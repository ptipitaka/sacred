<Division number="238" book="pari">

<Paragraph number="954" book="pari">
ឩនវីសតិវស្សំ កុមារិភូតំ វុដ្ឋាបេន្តី ទ្វេ អាបត្តិយោ អាបជ្ជតិ. វុដ្ឋាបេតិ, បយោគេ ទុក្កដំ; វុដ្ឋាបិតេ, អាបត្តិ បាចិត្តិយស្ស. (1)
</Paragraph>

<Paragraph number="955" book="pari">
បរិបុណ្ណវីសតិវស្សំ កុមារិភូតំ ទ្វេ វស្សានិ ឆសុ ធម្មេសុ អសិក្ខិតសិក្ខំ វុដ្ឋាបេន្តី ទ្វេ អាបត្តិយោ អាបជ្ជតិ. វុដ្ឋាបេតិ, បយោគេ ទុក្កដំ; វុដ្ឋាបិតេ, អាបត្តិ បាចិត្តិយស្ស. (2)
</Paragraph>

<Paragraph number="956" book="pari">
បរិបុណ្ណវីសតិវស្សំ កុមារិភូតំ ទ្វេ វស្សានិ ឆសុ ធម្មេសុ សិក្ខិតសិក្ខំ សំឃេន អសម្មតំ វុដ្ឋាបេន្តី ទ្វេ អាបត្តិយោ អាបជ្ជតិ. វុដ្ឋាបេតិ, បយោគេ ទុក្កដំ; វុដ្ឋាបិតេ, អាបត្តិ បាចិត្តិយស្ស. (3)
</Paragraph>

<Paragraph number="957" book="pari">
ឩនទ្វាទសវស្សា វុដ្ឋាបេន្តី ទ្វេ អាបត្តិយោ អាបជ្ជតិ. វុដ្ឋាបេតិ, បយោគេ ទុក្កដំ; វុដ្ឋាបិតេ, អាបត្តិ បាចិត្តិយស្ស. (4)
</Paragraph>

<Paragraph number="958" book="pari">
បរិបុណ្ណទ្វាទសវស្សា សំឃេន អសម្មតា វុដ្ឋាបេន្តី ទ្វេ អាបត្តិយោ អាបជ្ជតិ. វុដ្ឋាបេតិ, បយោគេ ទុក្កដំ; វុដ្ឋាបិតេ, អាបត្តិ បាចិត្តិយស្ស. (5)
</Paragraph>

<Paragraph number="959" book="pari">
“អលំ តាវ តេ, អយ្យេ, វុដ្ឋាបិតេនា”តិ វុច្ចមានា “សាធូ”តិ បដិស្សុណិត្វា បច្ឆា ខីយនធម្មំ អាបជ្ជន្តី ទ្វេ អាបត្តិយោ អាបជ្ជតិ. ខិយ្យតិ, បយោគេ ទុក្កដំ; ខិយ្យិតេ, អាបត្តិ បាចិត្តិយស្ស. (6)
</Paragraph>

<Paragraph number="960" book="pari">
សិក្ខមានំ—  “សចេ មេ ត្វំ, អយ្យេ, ចីវរំ ទស្សសិ, ឯវាហំ តំ វុដ្ឋាបេស្សាមី”តិ វត្វា នេវ វុដ្ឋាបេន្តី ន វុដ្ឋាបនាយ ឧស្សុក្កំ ករោន្តី ឯកំ អាបត្តឹ អាបជ្ជតិ. បាចិត្តិយំ. (7)
</Paragraph>

<Paragraph number="961" book="pari">
សិក្ខមានំ—  “សចេ មំ ត្វំ, អយ្យេ, ទ្វេ វស្សានិ អនុពន្ធិស្សសិ, ឯវាហំ តំ វុដ្ឋាបេស្សាមី”តិ វត្វា នេវ វុដ្ឋាបេន្តី ន វុដ្ឋាបនាយ ឧស្សុក្កំ ករោន្តី ឯកំ អាបត្តឹ អាបជ្ជតិ. បាចិត្តិយំ. (8)
</Paragraph>

<Paragraph number="962" book="pari">
បុរិសសំសដ្ឋំ កុមារកសំសដ្ឋំ ចណ្ឌឹ សោកាវាសំ សិក្ខមានំ វុដ្ឋាបេន្តី ទ្វេ អាបត្តិយោ អាបជ្ជតិ. វុដ្ឋាបេតិ, បយោគេ ទុក្កដំ; វុដ្ឋាបិតេ, អាបត្តិ បាចិត្តិយស្ស. (9)
</Paragraph>

<Paragraph number="963" book="pari">
មាតាបិតូហិ វា សាមិកេន វា អននុញ្ញាតំ សិក្ខមានំ វុដ្ឋាបេន្តី ទ្វេ អាបត្តិយោ អាបជ្ជតិ. វុដ្ឋាបេតិ, បយោគេ ទុក្កដំ; វុដ្ឋាបិតេ, អាបត្តិ បាចិត្តិយស្ស. (10)
</Paragraph>

<Paragraph number="964" book="pari">
បារិវាសិកឆន្ទទានេន សិក្ខមានំ វុដ្ឋាបេន្តី ទ្វេ អាបត្តិយោ អាបជ្ជតិ. វុដ្ឋាបេតិ, បយោគេ ទុក្កដំ; វុដ្ឋាបិតេ, អាបត្តិ បាចិត្តិយស្ស. (11)
</Paragraph>

<Paragraph number="965" book="pari">
អនុវស្សំ វុដ្ឋាបេន្តី ទ្វេ អាបត្តិយោ អាបជ្ជតិ. វុដ្ឋាបេតិ, បយោគេ ទុក្កដំ; វុដ្ឋាបិតេ, អាបត្តិ បាចិត្តិយស្ស. (12)
</Paragraph>

<Paragraph number="966" book="pari">
ឯកំ វស្សំ ទ្វេ វុដ្ឋាបេន្តី ទ្វេ អាបត្តិយោ អាបជ្ជតិ. វុដ្ឋាបេតិ, បយោគេ ទុក្កដំ; វុដ្ឋាបិតេ, អាបត្តិ បាចិត្តិយស្ស. (13)

</Paragraph>
</Division>
