<Division number="230" book="pari">

<Paragraph number="864" book="pari">
បត្តសន្និចយំ ករោន្តី ឯកំ អាបត្តឹ អាបជ្ជតិ. និស្សគ្គិយំ បាចិត្តិយំ. (1)
</Paragraph>

<Paragraph number="865" book="pari">
អកាលចីវរំ “កាលចីវរន៑”តិ អធិដ្ឋហិត្វា ភាជាបេន្តី ទ្វេ អាបត្តិយោ អាបជ្ជតិ. ភាជាបេតិ, បយោគេ ទុក្កដំ; ភាជាបិតេ និស្សគ្គិយំ បាចិត្តិយំ. (2)
</Paragraph>

<Paragraph number="866" book="pari">
ភិក្ខុនិយា សទ្ធឹ ចីវរំ បរិវត្តេត្វា អច្ឆិន្ទន្តី ទ្វេ អាបត្តិយោ អាបជ្ជតិ. អច្ឆិន្ទតិ, បយោគេ ទុក្កដំ; អច្ឆិន្នេ និស្សគ្គិយំ បាចិត្តិយំ. (3)
</Paragraph>

<Paragraph number="867" book="pari">
អញ្ញំ វិញ្ញាបេត្វា អញ្ញំ វិញ្ញាបេន្តី ទ្វេ អាបត្តិយោ អាបជ្ជតិ. វិញ្ញាបេតិ, បយោគេ ទុក្កដំ; វិញ្ញាបិតេ និស្សគ្គិយំ បាចិត្តិយំ. (4)
</Paragraph>

<Paragraph number="868" book="pari">
អញ្ញំ ចេតាបេត្វា អញ្ញំ ចេតាបេន្តី ទ្វេ អាបត្តិយោ អាបជ្ជតិ. ចេតាបេតិ, បយោគេ ទុក្កដំ; ចេតាបិតេ និស្សគ្គិយំ បាចិត្តិយំ. (5)
</Paragraph>

<Paragraph number="869" book="pari">
អញ្ញទត្ថិកេន បរិក្ខារេន អញ្ញុទ្ទិសិកេន សំឃិកេន អញ្ញំ ចេតាបេន្តី ទ្វេ អាបត្តិយោ អាបជ្ជតិ. ចេតាបេតិ, បយោគេ ទុក្កដំ; ចេតាបិតេ និស្សគ្គិយំ បាចិត្តិយំ. (6)
</Paragraph>

<Paragraph number="870" book="pari">
អញ្ញទត្ថិកេន បរិក្ខារេន អញ្ញុទ្ទិសិកេន សំឃិកេន សំយាចិកេន អញ្ញំ ចេតាបេន្តី ទ្វេ អាបត្តិយោ អាបជ្ជតិ. ចេតាបេតិ, បយោគេ ទុក្កដំ; ចេតាបិតេ និស្សគ្គិយំ បាចិត្តិយំ. (7)
</Paragraph>

<Paragraph number="871" book="pari">
អញ្ញទត្ថិកេន បរិក្ខារេន អញ្ញុទ្ទិសិកេន មហាជនិកេន អញ្ញំ ចេតាបេន្តី ទ្វេ អាបត្តិយោ អាបជ្ជតិ. ចេតាបេតិ, បយោគេ ទុក្កដំ; ចេតាបិតេ និស្សគ្គិយំ បាចិត្តិយំ. (8)
</Paragraph>

<Paragraph number="872" book="pari">
អញ្ញទត្ថិកេន បរិក្ខារេន អញ្ញុទ្ទិសិកេន មហាជនិកេន សំយាចិកេន អញ្ញំ ចេតាបេន្តី ទ្វេ អាបត្តិយោ អាបជ្ជតិ. ចេតាបេតិ, បយោគេ ទុក្កដំ; ចេតាបិតេ និស្សគ្គិយំ បាចិត្តិយំ. (9)
</Paragraph>

<Paragraph number="873" book="pari">
អញ្ញទត្ថិកេន បរិក្ខារេន អញ្ញុទ្ទិសិកេន បុគ្គលិកេន សំយាចិកេន អញ្ញំ ចេតាបេន្តី ទ្វេ អាបត្តិយោ អាបជ្ជតិ. ចេតាបេតិ, បយោគេ ទុក្កដំ; ចេតាបិតេ និស្សគ្គិយំ បាចិត្តិយំ. (10)
</Paragraph>

<Paragraph number="874" book="pari">
អតិរេកចតុក្កំសបរមំ គរុបាវុរណំ ចេតាបេន្តី ទ្វេ អាបត្តិយោ អាបជ្ជតិ. ចេតាបេតិ, បយោគេ ទុក្កដំ; ចេតាបិតេ និស្សគ្គិយំ បាចិត្តិយំ. (11)
</Paragraph>

<Paragraph number="875" book="pari">

</Paragraph>
