<Division number="231" book="pari">

<Paragraph number="877" book="pari">
លសុណំ ខាទន្តី ទ្វេ អាបត្តិយោ អាបជ្ជតិ. ខាទិស្សាមីតិ បដិគ្គណ្ហាតិ, អាបត្តិ ទុក្កដស្ស; អជ្ឈោហារេ អជ្ឈោហារេ អាបត្តិ បាចិត្តិយស្ស. (1)
</Paragraph>

<Paragraph number="878" book="pari">
សម្ពាធេ លោមំ សំហរាបេន្តី ទ្វេ អាបត្តិយោ អាបជ្ជតិ. សំហរាបេតិ, បយោគេ ទុក្កដំ; សំហរាបិតេ អាបត្តិ បាចិត្តិយស្ស. (2)
</Paragraph>

<Paragraph number="879" book="pari">
តលឃាតកំ ករោន្តី ទ្វេ អាបត្តិយោ អាបជ្ជតិ. ករោតិ, បយោគេ ទុក្កដំ; កតេ អាបត្តិ បាចិត្តិយស្ស. (3)
</Paragraph>

<Paragraph number="880" book="pari">
ជតុមដ្ឋកំ អាទិយន្តី ទ្វេ អាបត្តិយោ អាបជ្ជតិ. អាទិយតិ, បយោគេ ទុក្កដំ, អាទិន្នេ អាបត្តិ បាចិត្តិយស្ស. (4)
</Paragraph>

<Paragraph number="881" book="pari">
អតិរេកទ្វង្គុលបព្ពបរមំ ឧទកសុទ្ធិកំ អាទិយន្តី ទ្វេ អាបត្តិយោ អាបជ្ជតិ. អាទិយតិ, បយោគេ ទុក្កដំ; អាទិន្នេ អាបត្តិ បាចិត្តិយស្ស. (5)
</Paragraph>

<Paragraph number="882" book="pari">
ភិក្ខុស្ស ភុញ្ជន្តស្ស បានីយេន វា វិធូបនេន វា ឧបតិដ្ឋន្តី ទ្វេ អាបត្តិយោ អាបជ្ជតិ. ហត្ថបាសេ តិដ្ឋតិ, អាបត្តិ បាចិត្តិយស្ស; ហត្ថបាសំ វិជហិត្វា តិដ្ឋតិ, អាបត្តិ ទុក្កដស្ស. (6)
</Paragraph>

<Paragraph number="883" book="pari">
អាមកធញ្ញំ វិញ្ញាបេត្វា ភុញ្ជន្តី ទ្វេ អាបត្តិយោ អាបជ្ជតិ. “ភុញ្ជិស្សាមី”តិ បដិគ្គណ្ហាតិ, អាបត្តិ ទុក្កដស្ស; អជ្ឈោហារេ អជ្ឈោហារេ អាបត្តិ បាចិត្តិយស្ស. (7)
</Paragraph>

<Paragraph number="884" book="pari">
ឧច្ចារំ វា បស្សាវំ វា សង្ការំ វា វិឃាសំ វា តិរោកុដ្ដេ វា តិរោបាការេ វា ឆឌ្ឌេន្តី ទ្វេ អាបត្តិយោ អាបជ្ជតិ. ឆឌ្ឌេតិ, បយោគេ ទុក្កដំ; ឆឌ្ឌិតេ អាបត្តិ បាចិត្តិយស្ស. (8)
</Paragraph>

<Paragraph number="885" book="pari">
ឧច្ចារំ វា បស្សាវំ វា សង្ការំ វា វិឃាសំ វា ហរិតេ ឆឌ្ឌេន្តី ទ្វេ អាបត្តិយោ អាបជ្ជតិ. ឆឌ្ឌេតិ, បយោគេ ទុក្កដំ; ឆឌ្ឌិតេ អាបត្តិ បាចិត្តិយស្ស. (9)
</Paragraph>

<Paragraph number="886" book="pari">
នច្ចំ វា គីតំ វា វាទិតំ វា ទស្សនាយ គច្ឆន្តី ទ្វេ អាបត្តិយោ អាបជ្ជតិ. គច្ឆតិ, អាបត្តិ ទុក្កដស្ស; យត្ថ ឋិតា បស្សតិ វា សុណាតិ វា, អាបត្តិ បាចិត្តិយស្ស. (10)

</Paragraph>
</Division>

<Paragraph number="887" type="center" book="pari">
លសុណវគ្គោ បឋមោ.
</Paragraph>
