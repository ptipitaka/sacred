<Division number="235" book="pari">

<Paragraph number="921" book="pari">
រាជាគារំ វា ចិត្តាគារំ វា អារាមំ វា ឧយ្យានំ វា បោក្ខរណឹ វា ទស្សនាយ គច្ឆន្តី ទ្វេ អាបត្តិយោ អាបជ្ជតិ. គច្ឆតិ, អាបត្តិ ទុក្កដស្ស; យត្ថ ឋិតា បស្សតិ, អាបត្តិ បាចិត្តិយស្ស. (1)
</Paragraph>

<Paragraph number="922" book="pari">
អាសន្ទឹ វា បល្លង្កំ វា បរិភុញ្ជន្តី ទ្វេ អាបត្តិយោ អាបជ្ជតិ. បរិភុញ្ជតិ, បយោគេ ទុក្កដំ; បរិភុត្តេ, អាបត្តិ បាចិត្តិយស្ស. (2)
</Paragraph>

<Paragraph number="923" book="pari">
សុត្តំ កន្តន្តី ទ្វេ អាបត្តិយោ អាបជ្ជតិ. កន្តតិ, បយោគេ ទុក្កដំ; ឧជ្ជវុជ្ជវេ, អាបត្តិ បាចិត្តិយស្ស. (3)
</Paragraph>

<Paragraph number="924" book="pari">
គិហិវេយ្យាវច្ចំ ករោន្តី ទ្វេ អាបត្តិយោ អាបជ្ជតិ. ករោតិ, បយោគេ ទុក្កដំ; កតេ, អាបត្តិ បាចិត្តិយស្ស. (4)
</Paragraph>

<Paragraph number="925" book="pari">
ភិក្ខុនិយា—  “ឯហាយ្យេ, ឥមំ អធិករណំ វូបសមេហី”តិ វុច្ចមានា—  “សាធូ”តិ បដិស្សុណិត្វា នេវ វូបសមេន្តី ន វូបសមាយ ឧស្សុក្កំ ករោន្តី ឯកំ អាបត្តឹ អាបជ្ជតិ. បាចិត្តិយំ. (5)
</Paragraph>

<Paragraph number="926" book="pari">
អគារិកស្ស វា បរិព្ពាជកស្ស វា បរិព្ពាជិកាយ វា សហត្ថា ខាទនីយំ វា ភោជនីយំ វា ទេន្តី ទ្វេ អាបត្តិយោ អាបជ្ជតិ. ទេតិ, បយោគេ ទុក្កដំ; ទិន្នេ, អាបត្តិ បាចិត្តិយស្ស. (6)
</Paragraph>

<Paragraph number="927" book="pari">
អាវសថចីវរំ អនិស្សជ្ជិត្វា បរិភុញ្ជន្តី ទ្វេ អាបត្តិយោ អាបជ្ជតិ. បរិភុញ្ជតិ, បយោគេ ទុក្កដំ; បរិភុត្តេ, អាបត្តិ បាចិត្តិយស្ស. (7)
</Paragraph>

<Paragraph number="928" book="pari">
អាវសថំ អនិស្សជ្ជិត្វា ចារិកំ បក្កមន្តី ទ្វេ អាបត្តិយោ អាបជ្ជតិ. បឋមំ បាទំ បរិក្ខេបំ អតិក្កាមេតិ, អាបត្តិ ទុក្កដស្ស; ទុតិយំ បាទំ អតិក្កាមេតិ, អាបត្តិ បាចិត្តិយស្ស. (8)
</Paragraph>

<Paragraph number="929" book="pari">
តិរច្ឆានវិជ្ជំ បរិយាបុណន្តី ទ្វេ អាបត្តិយោ អាបជ្ជតិ. បរិយាបុណាតិ, បយោគេ ទុក្កដំ; បទេ បទេ អាបត្តិ បាចិត្តិយស្ស. (9)
</Paragraph>

<Paragraph number="930" book="pari">
តិរច្ឆានវិជ្ជំ វាចេន្តី ទ្វេ អាបត្តិយោ អាបជ្ជតិ. វាចេតិ, បយោគេ ទុក្កដំ; បទេ បទេ អាបត្តិ បាចិត្តិយស្ស. (10)

</Paragraph>
</Division>

<Paragraph number="931" type="center" book="pari">
ចិត្តាគារវគ្គោ បញ្ចមោ.
</Paragraph>
