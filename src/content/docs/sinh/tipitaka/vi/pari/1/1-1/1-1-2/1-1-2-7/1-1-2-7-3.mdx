<Paragraph number="515" book="pari">

</Paragraph>

<Paragraph number="516" book="pari">

</Paragraph>

<Paragraph number="517" book="pari">

</Paragraph>

<Paragraph number="518" book="pari">

</Paragraph>

<Paragraph number="519" book="pari">

</Paragraph>

</Paragraph>
