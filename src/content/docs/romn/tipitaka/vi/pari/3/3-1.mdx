<Paragraph number="1076" type="verses" book="pari">
_Aniccā sabbe saṅkhārā,_  
_dukkhānattā ca saṅkhatā;_  
_Nibbānañceva paññatti,_  
_anattā iti nicchayā._  

</Paragraph>

<Paragraph number="1077" type="verses" book="pari">
_Buddhacande anuppanne,_  
_buddhādicce anuggate;_  
_Tesaṃ sabhāgadhammānaṃ,_  
_nāmamattaṃ na nāyati._  

</Paragraph>

<Paragraph number="1078" type="verses" book="pari">
_Dukkaraṃ vividhaṃ katvā,_  
_pūrayitvāna pāramī;_  
_Uppajjanti mahāvīrā,_  
_cakkhubhūtā sabrahmake._  

</Paragraph>

<Paragraph number="1079" type="verses" book="pari">
_Te desayanti saddhammaṃ,_  
_dukkhahāniṃ sukhāvahaṃ;_  
_Aṅgīraso sakyamuni,_  
_sabbabhūtānukampako._  

</Paragraph>

<Paragraph number="1080" type="verses" book="pari">
_Sabbasattuttamo sīho,_  
_piṭake tīṇi desayi;_  
_Suttantamabhidhammañca,_  
_vinayañca mahāguṇaṃ._  

</Paragraph>

<Paragraph number="1081" type="verses" book="pari">
_Evaṃ nīyati saddhammo,_  
_vinayo yadi tiṭṭhati;_  
_Ubhato ca vibhaṅgāni,_  
_khandhakā yā ca mātikā._  

</Paragraph>

<Paragraph number="1082" type="verses" book="pari">
_Mālā suttaguṇeneva,_  
_parivārena ganthitā;_  
_Tasseva parivārassa,_  
_samuṭṭhānaṃ niyato kataṃ._  

</Paragraph>

<Paragraph number="1083" type="verses" book="pari">
_Sambhedaṃ nidānañcaññaṃ,_  
_sutte dissanti upari;_  
_Tasmā sikkhe parivāraṃ,_  
_dhammakāmo supesaloti._
</Paragraph>
</Division>
