<Paragraph number="1093" type="verses" book="pari">
_Adinnaṃ viggahuttari,_  
_duṭṭhullā attakāminaṃ;_  
_Amūlā aññabhāgiyā,_  
_aniyatā dutiyikā._  

</Paragraph>

<Paragraph number="1094" type="verses" book="pari">
_Acchinde pariṇāmane,_  
_musā omasapesuṇā;_  
_Duṭṭhullā pathavīkhaṇe,_  
_bhūtaṃ aññāya ujjhāpe._  

</Paragraph>

<Paragraph number="1095" type="verses" book="pari">
_Nikkaḍḍhanaṃ siñcanañca,_  
_āmisahetu bhuttāvī;_  
_Ehi anādari bhiṃsā,_  
_apanidhe ca jīvitaṃ._  

</Paragraph>

<Paragraph number="1096" type="verses" book="pari">
_Jānaṃ sappāṇakaṃ kammaṃ,_  
_ūnasaṃvāsanāsanā;_  
_Sahadhammikavilekhā,_  
_moho amūlakena ca._  

</Paragraph>

<Paragraph number="1097" type="verses" book="pari">
_Kukkuccaṃ dhammikaṃ cīvaraṃ datvā,_  
_Pariṇāmeyya puggale;_  
_Kiṃ te akālaṃ acchinde,_  
_Duggahī nirayena ca._  

</Paragraph>

<Paragraph number="1098" type="verses" book="pari">
_Gaṇaṃ vibhaṅgaṃ dubbalaṃ,_  
_kathināphāsupassayaṃ;_  
_Akkosacaṇḍī maccharī,_  
_gabbhinī ca pāyantiyā._  

</Paragraph>

<Paragraph number="1099" type="verses" book="pari">
_Dvevassaṃ sikkhā saṃghena,_  
_tayo ceva gihīgatā;_  
_Kumāribhūtā tisso ca,_  
_ūnadvādasasammatā._  

</Paragraph>

<Paragraph number="1100" type="verses" book="pari">
_Alaṃ tāva sokāvāsaṃ,_  
_chandā anuvassā ca dve;_  
_Sikkhāpadā sattatime,_  
_samuṭṭhānā tikā katā._  

</Paragraph>

<Paragraph number="1101" type="verses" book="pari">
_Kāyacittena na vācā,_  
_vācācittaṃ na kāyikaṃ;_  
_Tīhi dvārehi jāyanti,_  
_pārājikaṃ dutiyaṃ yathā._  


</Paragraph>
</Division>

<Paragraph number="1102" type="center" book="pari">
Dutiyapārājikasamuṭṭhānaṃ niṭṭhitaṃ.
</Paragraph>
