<Paragraph number="2242" type="verses" book="pari">
_Kati desanāgāminiyo,_  
_Kati sappaṭikammā katā;_  
_Katettha appaṭikammā vuttā,_  
_Buddhenādiccabandhunā._  

</Paragraph>

<Paragraph number="2243" type="verses" book="pari">
_Pañca desanāgāminiyo,_  
_Cha sappaṭikammā katā;_  
_Ekettha appaṭikammā vuttā,_  
_Buddhenādiccabandhunā._  

</Paragraph>

<Paragraph number="2244" type="verses" book="pari">
_Vinayagarukā kati vuttā,_  
_Kāyavācasikāni ca;_  
_Kati vikāle dhaññaraso,_  
_Kati ñatticatutthena sammuti._  

</Paragraph>

<Paragraph number="2245" type="verses" book="pari">
_Vinayagarukā dve vuttā,_  
_Kāyavācasikāni ca;_  
_Eko vikāle dhaññaraso,_  
_Ekā ñatticatutthena sammuti._  

</Paragraph>

<Paragraph number="2246" type="verses" book="pari">
_Pārājikā kāyikā kati,_  
_kati saṃvāsakabhūmiyo;_  
_Katinaṃ ratticchedo,_  
_paññattā dvaṅgulā kati._  

</Paragraph>

<Paragraph number="2247" type="verses" book="pari">
_Pārājikā kāyikā dve,_  
_dve saṃvāsakabhūmiyo;_  
_Dvinnaṃ ratticchedo,_  
_paññattā dvaṅgulā duve._  

</Paragraph>

<Paragraph number="2248" type="verses" book="pari">
_Katattānaṃ vadhitvāna,_  
_katihi saṃgho bhijjati;_  
_Katettha paṭhamāpattikā,_  
_ñattiyā karaṇā kati._  

</Paragraph>

<Paragraph number="2249" type="verses" book="pari">
_Dve attānaṃ vadhitvāna,_  
_dvīhi saṃgho bhijjati;_  
_Dvettha paṭhamāpattikā,_  
_ñattiyā karaṇā duve._  

</Paragraph>

<Paragraph number="2250" type="verses" book="pari">
_Pāṇātipāte kati āpattiyo,_  
_Vācā pārājikā kati;_  
_Obhāsanā kati vuttā,_  
_Sañcarittena vā kati._  

</Paragraph>

<Paragraph number="2251" type="verses" book="pari">
_Pāṇātipāte tisso āpattiyo,_  
_Vācā pārājikā tayo;_  
_Obhāsanā tayo vuttā,_  
_Sañcarittena vā tayo._  

</Paragraph>

<Paragraph number="2252" type="verses" book="pari">
_Kati puggalā na upasampādetabbā,_  
_Kati kammānaṃ saṅgahā;_  
_Nāsitakā kati vuttā,_  
_Katinaṃ ekavācikā._  

</Paragraph>

<Paragraph number="2253" type="verses" book="pari">
_Tayo puggalā na upasampādetabbā,_  
_Tayo kammānaṃ saṅgahā;_  
_Nāsitakā tayo vuttā,_  
_Tiṇṇannaṃ ekavācikā._  

</Paragraph>

<Paragraph number="2254" type="verses" book="pari">
_Adinnādāne kati āpattiyo,_  
_Kati methunapaccayā;_  
_Chindantassa kati āpattiyo,_  
_Kati chaḍḍitapaccayā._  

</Paragraph>

<Paragraph number="2255" type="verses" book="pari">
_Adinnādāne tisso āpattiyo,_  
_Catasso methunapaccayā;_  
_Chindantassa tisso āpattiyo,_  
_Pañca chaḍḍitapaccayā._  

</Paragraph>

<Paragraph number="2256" type="verses" book="pari">
_Bhikkhunovādakavaggasmiṃ,_  
_pācittiyena dukkaṭā;_  
_Katettha navakā vuttā,_  
_katinaṃ cīvarena ca._  

</Paragraph>

<Paragraph number="2257" type="verses" book="pari">
_Bhikkhunovādakavaggasmiṃ,_  
_Pācittiyena dukkaṭā katā;_  
_Caturettha navakā vuttā,_  
_Dvinnaṃ cīvarena ca._  

</Paragraph>

<Paragraph number="2258" type="verses" book="pari">
_Bhikkhunīnañca akkhātā,_  
_Pāṭidesaniyā kati;_  
_Bhuñjantāmakadhaññena,_  
_Pācittiyena dukkaṭā kati._  

</Paragraph>

<Paragraph number="2259" type="verses" book="pari">
_Bhikkhunīnañca akkhātā,_  
_Aṭṭha pāṭidesanīyā katā;_  
_Bhuñjantāmakadhaññena,_  
_Pācittiyena dukkaṭā katā._  

</Paragraph>

<Paragraph number="2260" type="verses" book="pari">
_Gacchantassa kati āpattiyo,_  
_Ṭhitassa cāpi kittakā;_  
_Nisinnassa kati āpattiyo,_  
_Nipannassāpi kittakā._  

</Paragraph>

<Paragraph number="2261" type="verses" book="pari">
_Gacchantassa catasso āpattiyo,_  
_Ṭhitassa cāpi tattakā;_  
_Nisinnassa catasso āpattiyo,_  
_Nipannassāpi tattakā._
</Paragraph>
