_Kati bhojanapaccayā._  

</Paragraph>

<Paragraph number="2275" type="verses" book="pari">

</Paragraph>

<Paragraph number="2276" type="verses" book="pari">

<Paragraph number="2280" type="verses" book="pari">

</Paragraph>

</Paragraph>

<Paragraph number="2283" type="verses" book="pari">

</Paragraph>

<Paragraph number="2284" type="verses" book="pari">

</Paragraph>
