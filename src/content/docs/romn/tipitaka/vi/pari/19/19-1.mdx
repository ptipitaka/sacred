<Division number="474" book="pari">

<Paragraph number="2232" type="verses" book="pari">
_Kati āpattiyo kāyikā,_  
_Kati vācasikā katā;_  
_Chādentassa kati āpattiyo,_  
_Kati saṃsaggapaccayā._  

</Paragraph>

<Paragraph number="2233" type="verses" book="pari">
_Chāpattiyo kāyikā,_  
_Cha vācasikā katā;_  
_Chādentassa tisso āpattiyo,_  
_Pañca saṃsaggapaccayā._  

</Paragraph>

<Paragraph number="2234" type="verses" book="pari">
_Aruṇugge kati āpattiyo,_  
_Kati yāvatatiyakā;_  
_Katettha aṭṭha vatthukā,_  
_Katihi sabbasaṅgaho._  

</Paragraph>

<Paragraph number="2235" type="verses" book="pari">
_Aruṇugge tisso āpattiyo,_  
_Dve yāvatatiyakā;_  
_Ekettha aṭṭha vatthukā,_  
_Ekena sabbasaṅgaho._  

</Paragraph>

<Paragraph number="2236" type="verses" book="pari">
_Vinayassa kati mūlāni,_  
_Yāni buddhena paññattā;_  
_Vinayagarukā kati vuttā,_  
_Duṭṭhullacchādanā kati._  

</Paragraph>

<Paragraph number="2237" type="verses" book="pari">
_Vinayassa dve mūlāni,_  
_yāni buddhena paññattā;_  
_Vinayagarukā dve vuttā,_  
_dve duṭṭhullacchādanā._  

</Paragraph>

<Paragraph number="2238" type="verses" book="pari">
_Gāmantare kati āpattiyo,_  
_Kati nadipārapaccayā;_  
_Katimaṃsesu thullaccayaṃ,_  
_Katimaṃsesu dukkaṭaṃ._  

</Paragraph>

<Paragraph number="2239" type="verses" book="pari">
_Gāmantare catasso āpattiyo,_  
_Catasso nadipārapaccayā;_  
_Ekamaṃse thullaccayaṃ,_  
_Navamaṃsesu dukkaṭaṃ._  

</Paragraph>

<Paragraph number="2240" type="verses" book="pari">
_Kati vācasikā rattiṃ,_  
_Kati vācasikā divā;_  
_Dadamānassa kati āpattiyo,_  
_Paṭiggaṇhantassa kittakā._  

</Paragraph>

<Paragraph number="2241" type="verses" book="pari">
_Dve vācasikā rattiṃ,_  
_Dve vācasikā divā;_  
_Dadamānassa tisso āpattiyo,_  
_Cattāro ca paṭiggahe._
</Paragraph>
</Division>
