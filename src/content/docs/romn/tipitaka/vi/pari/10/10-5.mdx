<Paragraph number="1677" type="verses" book="pari">
_“Pārājikan”ti yaṃ vuttaṃ,_  
_taṃ suṇohi yathātathaṃ;_  
_Cuto paraddho bhaṭṭho ca,_  
_saddhammā hi niraṅkato;_  
_Saṃvāsopi tahiṃ natthi,_  
_tenetaṃ iti vuccati._  

</Paragraph>

<Paragraph number="1678" type="verses" book="pari">
_“Saṃghādiseso”ti yaṃ vuttaṃ,_  
_taṃ suṇohi yathātathaṃ;_  
_Saṃghova deti parivāsaṃ,_  
_mūlāya paṭikassati;_  
_Mānattaṃ deti abbheti,_  
_tenetaṃ iti vuccati._  

</Paragraph>

<Paragraph number="1679" type="verses" book="pari">
_“Aniyato”ti yaṃ vuttaṃ,_  
_taṃ suṇohi yathātathaṃ;_  
_Aniyato na niyato,_  
_anekaṃsikataṃ padaṃ;_  
_Tiṇṇamaññataraṃ ṭhānaṃ,_  
_“aniyato”ti pavuccati._  

</Paragraph>

<Paragraph number="1680" type="verses" book="pari">
_“Thullaccayan”ti yaṃ vuttaṃ,_  
_taṃ suṇohi yathātathaṃ;_  
_Ekassa mūle yo deseti,_  
_yo ca taṃ paṭigaṇhati;_  
_Accayo tena samo natthi,_  
_tenetaṃ iti vuccati._  

</Paragraph>

<Paragraph number="1681" type="verses" book="pari">
_“Nissaggiyan”ti yaṃ vuttaṃ,_  
_taṃ suṇohi yathātathaṃ;_  
_Saṃghamajjhe gaṇamajjhe,_  
_ekasseva ca ekato;_  
_Nissajjitvāna deseti,_  
_tenetaṃ iti vuccati._  

</Paragraph>

<Paragraph number="1682" type="verses" book="pari">
_“Pācittiyan”ti yaṃ vuttaṃ,_  
_Taṃ suṇohi yathātathaṃ;_  
_Pāteti kusalaṃ dhammaṃ,_  
_Ariyamaggaṃ aparajjhati;_  
_Cittasammohanaṭṭhānaṃ,_  
_Tenetaṃ iti vuccati._  

</Paragraph>

<Paragraph number="1683" type="verses" book="pari">
_“Pāṭidesanīyan”ti yaṃ vuttaṃ,_  
_Taṃ suṇohi yathātathaṃ;_  
_Bhikkhu aññātako santo,_  
_Kicchā laddhāya bhojanaṃ;_  
_Sāmaṃ gahetvā bhuñjeyya,_  
_“Gārayhan”ti pavuccati._  

</Paragraph>

<Paragraph number="1684" type="verses" book="pari">
_Nimantanāsu bhuñjantā chandāya,_  
_Vosāsati tattha bhikkhuniṃ;_  
_Anivāretvā tahiṃ bhuñje,_  
_“Gārayhan”ti pavuccati._  

</Paragraph>

<Paragraph number="1685" type="verses" book="pari">
_Saddhācittaṃ kulaṃ gantvā,_  
_appabhogaṃ anāḷiyaṃ;_  
_Agilāno tahiṃ bhuñje,_  
_“gārayhan”ti pavuccati._  

</Paragraph>

<Paragraph number="1686" type="verses" book="pari">
_Yo ce araññe viharanto,_  
_sāsaṅke sabhayānake;_  
_Aviditaṃ tahiṃ bhuñje,_  
_“gārayhan”ti pavuccati._  

</Paragraph>

<Paragraph number="1687" type="verses" book="pari">
_Bhikkhunī aññātikā santā,_  
_Yaṃ paresaṃ mamāyitaṃ;_  
_Sappi telaṃ madhuṃ phāṇitaṃ,_  
_Macchamaṃsaṃ atho khīraṃ;_  
_Dadhiṃ sayaṃ viññāpeyya bhikkhunī,_  
_Gārayhapattā sugatassa sāsane._  

</Paragraph>

<Paragraph number="1688" type="verses" book="pari">
_“Dukkaṭan”ti yaṃ vuttaṃ,_  
_taṃ suṇohi yathātathaṃ;_  
_Aparaddhaṃ viraddhañca,_  
_khalitaṃ yañca dukkaṭaṃ._  

</Paragraph>

<Paragraph number="1689" type="verses" book="pari">
_Yaṃ manusso kare pāpaṃ,_  
_āvi vā yadi vā raho;_  
_“Dukkaṭan”ti pavedenti,_  
_tenetaṃ iti vuccati._  

</Paragraph>

<Paragraph number="1690" type="verses" book="pari">
_“Dubbhāsitan”ti yaṃ vuttaṃ,_  
_taṃ suṇohi yathātathaṃ;_  
_Dubbhāsitaṃ durābhaṭṭhaṃ,_  
_saṃkiliṭṭhañca yaṃ padaṃ;_  
_Yañca viññū garahanti,_  
_tenetaṃ iti vuccati._  

</Paragraph>

<Paragraph number="1691" type="verses" book="pari">
_“Sekhiyan”ti yaṃ vuttaṃ,_  
_taṃ suṇohi yathātathaṃ;_  
_Sekkhassa sikkhamānassa,_  
_ujumaggānusārino._  

</Paragraph>

<Paragraph number="1692" type="verses" book="pari">
_Ādi cetaṃ caraṇañca,_  
_mukhaṃ saññamasaṃvaro;_  
_Sikkhā etādisī natthi,_  
_tenetaṃ iti vuccati._  

</Paragraph>

<Paragraph number="1693" type="verses" book="pari">
_Channamativassati,_  
_vivaṭaṃ nātivassati;_  
_Tasmā channaṃ vivaretha,_  
_evaṃ taṃ nātivassati._  

</Paragraph>

<Paragraph number="1694" type="verses" book="pari">
_Gati migānaṃ pavanaṃ,_  
_Ākāso pakkhinaṃ gati;_  
_Vibhavo gati dhammānaṃ,_  
_Nibbānaṃ arahato gatīti._  


</Paragraph>
</Division>

</Paragraph>

_Sattanagaresu paññattā,_  
_vipatti caturopi ca;_  
_Bhikkhūnaṃ bhikkhunīnañca,_  
_sādhāraṇā asādhāraṇā;_  
_Sāsanaṃ anuggahāya,_  
_gāthāsaṅgaṇikaṃ idanti._  


</Paragraph>
