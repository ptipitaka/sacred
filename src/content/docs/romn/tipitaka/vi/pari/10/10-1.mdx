<Paragraph number="1615" type="verses" book="pari">
_“Ekaṃsaṃ cīvaraṃ katvā,_  
_paggaṇhitvāna añjaliṃ;_  
_Āsīsamānarūpova,_  
_kissa tvaṃ idha māgato”._  

</Paragraph>

<Paragraph number="1616" type="verses" book="pari">
_“Dvīsu vinayesu ye paññattā,_  
_Uddesaṃ āgacchanti uposathesu;_  
_Kati te sikkhāpadā honti,_  
_Katisu nagaresu paññattā”._  

</Paragraph>

<Paragraph number="1617" type="verses" book="pari">
_“Bhaddako te ummaṅgo,_  
_yoniso paripucchasi;_  
_Taggha te ahamakkhissaṃ,_  
_yathāsi kusalo tathā”._  

</Paragraph>

<Paragraph number="1618" type="verses" book="pari">
_“Dvīsu vinayesu ye paññattā,_  
_Uddesaṃ āgacchanti uposathesu;_  
_Aḍḍhuḍḍhasatāni te honti,_  
_Sattasu nagaresu paññattā”._  

</Paragraph>

<Paragraph number="1619" type="verses" book="pari">
_“Katamesu sattasu nagaresu paññattā,_  
_Iṅgha me tvaṃ byākara naṃ;_  
_Taṃ vacanapathaṃ nisāmayitvā,_  
_Paṭipajjema hitāya no siyā”._  

</Paragraph>

<Paragraph number="1620" type="verses" book="pari">
_“Vesāliyaṃ rājagahe,_  
_sāvatthiyañca āḷaviyaṃ;_  
_Kosambiyañca sakkesu,_  
_bhaggesu ceva paññattā”._  

</Paragraph>

<Paragraph number="1621" type="verses" book="pari">
_“Kati vesāliyaṃ paññattā,_  
_kati rājagahe katā;_  
_Sāvatthiyaṃ kati honti,_  
_kati āḷaviyaṃ katā”._  

</Paragraph>

<Paragraph number="1622" type="verses" book="pari">
_“Kati kosambiyaṃ paññattā,_  
_kati sakkesu vuccanti;_  
_Kati bhaggesu paññattā,_  
_taṃ me akkhāhi pucchito”._  

</Paragraph>

<Paragraph number="1623" type="verses" book="pari">
_“Dasa vesāliyaṃ paññattā,_  
_Ekavīsa rājagahe katā;_  
_Chaūna tīṇisatāni,_  
_Sabbe sāvatthiyaṃ katā._  

</Paragraph>

<Paragraph number="1624" type="verses" book="pari">
_Cha āḷaviyaṃ paññattā,_  
_aṭṭha kosambiyaṃ katā;_  
_Aṭṭha sakkesu vuccanti,_  
_tayo bhaggesu paññattā._  

</Paragraph>

<Paragraph number="1625" type="verses" book="pari">
_Ye vesāliyaṃ paññattā,_  
_te suṇohi yathātathaṃ;_  
_Methunaviggahuttari,_  
_atirekañca kāḷakaṃ._  

</Paragraph>

<Paragraph number="1626" type="verses" book="pari">
_Bhūtaṃ paramparabhattaṃ,_  
_dantaponena acelako;_  
_Bhikkhunīsu ca akkoso,_  
_dasete vesāliyaṃ katā._  

</Paragraph>

<Paragraph number="1627" type="verses" book="pari">
_Ye rājagahe paññattā,_  
_Te suṇohi yathātathaṃ;_  
_Adinnādānaṃ rājagahe,_  
_Dve anuddhaṃsanā dvepi ca bhedā._  

</Paragraph>

<Paragraph number="1628" type="verses" book="pari">
_Antaravāsakaṃ rūpiyaṃ suttaṃ,_  
_Ujjhāpanena ca pācitapiṇḍaṃ;_  
_Gaṇabhojanaṃ vikāle ca,_  
_Cārittaṃ nahānaṃ ūnavīsati._  

</Paragraph>

<Paragraph number="1629" type="verses" book="pari">
_Cīvaraṃ datvā vosāsanti,_  
_Ete rājagahe katā;_  
_Giraggacariyā tattheva,_  
_Chandadānena ekavīsati._  

</Paragraph>

<Paragraph number="1630" type="verses" book="pari">
_Ye sāvatthiyaṃ paññattā,_  
_Te suṇohi yathātathaṃ;_  
_Pārājikāni cattāri,_  
_Saṃghādisesā bhavanti soḷasa._  

</Paragraph>

<Paragraph number="1631" type="verses" book="pari">
_Aniyatā ca dve honti,_  
_nissaggiyā catuvīsati;_  
_Chapaññāsasatañceva,_  
_khuddakāni pavuccanti._  

</Paragraph>

<Paragraph number="1632" type="verses" book="pari">
_Dasayeva ca gārayhā,_  
_dvesattati ca sekhiyā;_  
_Chaūna tīṇisatāni,_  
_sabbe sāvatthiyaṃ katā._  

</Paragraph>

<Paragraph number="1633" type="verses" book="pari">
_Ye āḷaviyaṃ paññattā,_  
_te suṇohi yathātathaṃ;_  
_Kuṭikosiyaseyyā ca,_  
_khaṇane gaccha devate;_  
_Sappāṇakañca siñcanti,_  
_cha ete āḷaviyaṃ katā._  

</Paragraph>

<Paragraph number="1634" type="verses" book="pari">
_Ye kosambiyaṃ paññattā,_  
_te suṇohi yathātathaṃ;_  
_Mahāvihāro dovacassaṃ,_  
_aññaṃ dvāraṃ surāya ca;_  
_Anādariyaṃ sahadhammo,_  
_payopānena aṭṭhamaṃ._  

</Paragraph>

<Paragraph number="1635" type="verses" book="pari">
_Ye sakkesu paññattā,_  
_te suṇohi yathātathaṃ;_  
_Eḷakalomāni patto ca,_  
_ovādo ceva bhesajjaṃ._  

</Paragraph>

<Paragraph number="1636" type="verses" book="pari">
_Sūci āraññiko ceva,_  
_aṭṭhete kāpilavatthave;_  
_Udakasuddhiyā ovādo,_  
_bhikkhunīsu pavuccanti._  

</Paragraph>

<Paragraph number="1637" type="verses" book="pari">
_Ye bhaggesu paññattā,_  
_te suṇohi yathātathaṃ;_  
_Samādahitvā visibbenti,_  
_sāmisena sasitthakaṃ._  

</Paragraph>

<Paragraph number="1638" type="verses" book="pari">
_Pārājikāni cattāri,_  
_saṃghādisesāni bhavanti;_  
_Satta ca nissaggiyāni,_  
_aṭṭha dvattiṃsa khuddakā._  

</Paragraph>

<Paragraph number="1639" type="verses" book="pari">
_Dve gārayhā tayo sekkhā,_  
_chappaññāsa sikkhāpadā;_  
_Chasu nagaresu paññattā,_  
_buddhenādiccabandhunā._  

</Paragraph>

<Paragraph number="1640" type="verses" book="pari">
_Chaūna tīṇisatāni,_  
_sabbe sāvatthiyaṃ katā;_  
_Kāruṇikena buddhena,_  
_gotamena yasassinā”._
</Paragraph>
</Division>
