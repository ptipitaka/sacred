<Paragraph number="1652" type="verses" book="pari">
_“Vīsaṃ dve satāni,_  
_Bhikkhūnaṃ sikkhāpadāni;_  
_Uddesaṃ āgacchanti uposathesu,_  
_Tīṇi satāni cattāri;_  
_Bhikkhunīnaṃ sikkhāpadāni,_  
_Uddesaṃ āgacchanti uposathesu._  

</Paragraph>

<Paragraph number="1653" type="verses" book="pari">
_Chacattārīsā bhikkhūnaṃ,_  
_bhikkhunīhi asādhāraṇā;_  
_Sataṃ tiṃsā ca bhikkhunīnaṃ,_  
_bhikkhūhi asādhāraṇā._  

</Paragraph>

<Paragraph number="1654" type="verses" book="pari">
_Sataṃ sattati chacceva,_  
_ubhinnaṃ asādhāraṇā;_  
_Sataṃ sattati cattāri,_  
_ubhinnaṃ samasikkhatā._  

</Paragraph>

<Paragraph number="1655" type="verses" book="pari">
_Vīsaṃ dve satāni,_  
_Bhikkhūnaṃ sikkhāpadāni;_  
_Uddesaṃ āgacchanti uposathesu,_  
_Te suṇohi yathātathaṃ._  

</Paragraph>

<Paragraph number="1656" type="verses" book="pari">
_Pārājikāni cattāri,_  
_Saṃghādisesāni bhavanti terasa;_  
_Aniyatā dve honti._  

</Paragraph>

<Paragraph number="1657" type="verses" book="pari">
_Nissaggiyāni tiṃseva,_  
_dvenavuti ca khuddakā;_  
_Cattāro pāṭidesanīyā,_  
_pañcasattati sekhiyā._  

</Paragraph>

<Paragraph number="1658" type="verses" book="pari">
_Vīsaṃ dve satāni cime honti,_  
_Bhikkhūnaṃ sikkhāpadāni;_  
_Uddesaṃ āgacchanti uposathesu._  

</Paragraph>

<Paragraph number="1659" type="verses" book="pari">
_Tīṇi satāni cattāri,_  
_Bhikkhunīnaṃ sikkhāpadāni;_  
_Uddesaṃ āgacchanti uposathesu,_  
_Te suṇohi yathātathaṃ._  

</Paragraph>

<Paragraph number="1660" type="verses" book="pari">
_Pārājikāni aṭṭha,_  
_Saṃghādisesāni bhavanti sattarasa;_  
_Nissaggiyāni tiṃseva,_  
_Sataṃ saṭṭhi cha ceva;_  
_Khuddakāni pavuccanti._  

</Paragraph>

<Paragraph number="1661" type="verses" book="pari">
_Aṭṭha pāṭidesanīyā,_  
_Pañcasattati sekhiyā;_  
_Tīṇi satāni cattāri cime honti,_  
_Bhikkhunīnaṃ sikkhāpadāni;_  
_Uddesaṃ āgacchanti uposathesu._  

</Paragraph>

<Paragraph number="1662" type="verses" book="pari">
_Chacattārīsā bhikkhūnaṃ,_  
_bhikkhunīhi asādhāraṇā;_  
_Te suṇohi yathātathaṃ._  

</Paragraph>

<Paragraph number="1663" type="verses" book="pari">
_Cha saṃghādisesā,_  
_dve aniyatehi aṭṭha;_  
_Nissaggiyāni dvādasa,_  
_tehi te honti vīsati._  

</Paragraph>

<Paragraph number="1664" type="verses" book="pari">
_Dvevīsati khuddakā,_  
_Caturo pāṭidesanīyā;_  
_Chacattārīsā cime honti,_  
_Bhikkhūnaṃ bhikkhunīhi asādhāraṇā._  

</Paragraph>

<Paragraph number="1665" type="verses" book="pari">
_Sataṃ tiṃsā ca bhikkhunīnaṃ,_  
_bhikkhūhi asādhāraṇā;_  
_Te suṇohi yathātathaṃ._  

</Paragraph>

<Paragraph number="1666" type="verses" book="pari">
_Pārājikāni cattāri,_  
_saṃghamhā dasa nissare;_  
_Nissaggiyāni dvādasa,_  
_channavuti ca khuddakā;_  
_Aṭṭha pāṭidesanīyā._  

</Paragraph>

<Paragraph number="1667" type="verses" book="pari">
_Sataṃ tiṃsā cime honti,_  
_Bhikkhunīnaṃ bhikkhūhi asādhāraṇā;_  
_Sataṃ sattati chacceva,_  
_Ubhinnaṃ asādhāraṇā;_  
_Te suṇohi yathātathaṃ._  

</Paragraph>

<Paragraph number="1668" type="verses" book="pari">
_Pārājikāni cattāri,_  
_Saṃghādisesāni bhavanti soḷasa;_  
_Aniyatā dve honti,_  
_Nissaggiyāni catuvīsati;_  
_Sataṃ aṭṭhārasā ceva,_  
_Khuddakāni pavuccanti;_  
_Dvādasa pāṭidesanīyā._  

</Paragraph>

<Paragraph number="1669" type="verses" book="pari">
_Sataṃ sattati chaccevime honti,_  
_Ubhinnaṃ asādhāraṇā;_  
_Sataṃ sattati cattāri,_  
_Ubhinnaṃ samasikkhatā;_  
_Te suṇohi yathātathaṃ._  

</Paragraph>

<Paragraph number="1670" type="verses" book="pari">
_Pārājikāni cattāri,_  
_Saṃghādisesāni bhavanti satta;_  
_Nissaggiyāni aṭṭhārasa,_  
_Samasattati khuddakā;_  
_Pañcasattati sekhiyāni._  

</Paragraph>

<Paragraph number="1671" type="verses" book="pari">
_Sataṃ sattati cattāri cime honti,_  
_Ubhinnaṃ samasikkhatā;_  
_Aṭṭheva pārājikā ye durāsadā,_  
_Tālavatthusamūpamā._  

</Paragraph>

<Paragraph number="1672" type="verses" book="pari">
_Paṇḍupalāso puthusilā,_  
_sīsacchinnova so naro;_  
_Tālova matthakacchinno,_  
_aviruḷhī bhavanti te._  

</Paragraph>

<Paragraph number="1673" type="verses" book="pari">
_Tevīsati saṃghādisesā,_  
_Dve aniyatā;_  
_Dve cattārīsa nissaggiyā,_  
_Aṭṭhāsītisataṃ pācittiyā;_  
_Dvādasa pāṭidesanīyā._  

</Paragraph>

<Paragraph number="1674" type="verses" book="pari">
_Pañcasattati sekhiyā,_  
_tīhi samathehi sammanti;_  
_Sammukhā ca paṭiññāya,_  
_tiṇavatthārakena ca._  

</Paragraph>

<Paragraph number="1675" type="verses" book="pari">
_Dve uposathā dve pavāraṇā,_  
_Cattāri kammāni jinena desitā;_  
_Pañceva uddesā caturo bhavanti,_  
_Anaññathā āpattikkhandhā ca bhavanti satta._  

</Paragraph>

<Paragraph number="1676" type="verses" book="pari">
_Adhikaraṇāni cattāri,_  
_Sattahi samathehi sammanti;_  
_Dvīhi catūhi tīhi,_  
_Kiccaṃ ekena sammati”._
</Paragraph>
