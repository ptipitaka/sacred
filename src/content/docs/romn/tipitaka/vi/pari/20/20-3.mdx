<Paragraph number="2369" type="verses" book="pari">
_Adhiṭṭhitaṃ rajanāya rattaṃ,_  
_Kappakatampi santaṃ;_  
_Paribhuñjantassa āpatti,_  
_Pañhā mesā kusalehi cintitā._  

</Paragraph>

<Paragraph number="2370" type="verses" book="pari">
_Atthaṅgate sūriye bhikkhu maṃsāni khādati,_  
_Na ummattako na ca pana khittacitto;_  
_Na cāpi so vedanāṭṭo bhaveyya,_  
_Na cassa hoti āpatti;_  
_So ca dhammo sugatena desito,_  
_Pañhā mesā kusalehi cintitā._  

</Paragraph>

<Paragraph number="2371" type="verses" book="pari">
_Na rattacitto na ca pana theyyacitto,_  
_Na cāpi so paraṃ maraṇāya cetayi;_  
_Salākaṃ dentassa hoti chejjaṃ,_  
_Paṭiggaṇhantassa thullaccayaṃ;_  
_Pañhā mesā kusalehi cintitā._  

</Paragraph>

<Paragraph number="2372" type="verses" book="pari">
_Na cāpi āraññakaṃ sāsaṅkasammataṃ,_  
_Na cāpi saṃghena sammuti dinnā;_  
_Na cassa kathinaṃ atthataṃ tattheva,_  
_Cīvaraṃ nikkhipitvā gaccheyya aḍḍhayojanaṃ;_  
_Tattheva aruṇaṃ uggacchantassa anāpatti,_  
_Pañhā mesā kusalehi cintitā._  

</Paragraph>

<Paragraph number="2373" type="verses" book="pari">
_Kāyikāni na vācasikāni,_  
_Sabbāni nānāvatthukāni;_  
_Apubbaṃ acarimaṃ āpajjeyya ekato,_  
_Pañhā mesā kusalehi cintitā._  

</Paragraph>

<Paragraph number="2374" type="verses" book="pari">
_Vācasikāni na kāyikāni,_  
_Sabbāni nānāvatthukāni;_  
_Apubbaṃ acarimaṃ āpajjeyya ekato,_  
_Pañhā mesā kusalehi cintitā._  

</Paragraph>

<Paragraph number="2375" type="verses" book="pari">
_Tissitthiyo methunaṃ taṃ na seve,_  
_Tayo purise tayo anariyapaṇḍake;_  
_Na cācare methunaṃ byañjanasmiṃ,_  
_Chejjaṃ siyā methunadhammapaccayā;_  
_Pañhā mesā kusalehi cintitā._  

</Paragraph>

<Paragraph number="2376" type="verses" book="pari">
_Mātaraṃ cīvaraṃ yāce,_  
_No ca saṃghe pariṇataṃ;_  
_Kenassa hoti āpatti,_  
_Anāpatti ca ñātake;_  
_Pañhā mesā kusalehi cintitā._  

</Paragraph>

<Paragraph number="2377" type="verses" book="pari">
_Kuddho ārādhako hoti,_  
_Kuddho hoti garahiyo;_  
_Atha ko nāma so dhammo,_  
_Yena kuddho pasaṃsiyo;_  
_Pañhā mesā kusalehi cintitā._  

</Paragraph>

<Paragraph number="2378" type="verses" book="pari">
_Tuṭṭho ārādhako hoti,_  
_Tuṭṭho hoti garahiyo;_  
_Atha ko nāma so dhammo,_  
_Yena tuṭṭho garahiyo;_  
_Pañhā mesā kusalehi cintitā._  

</Paragraph>

<Paragraph number="2379" type="verses" book="pari">
_Saṃghādisesaṃ thullaccayaṃ,_  
_Pācittiyaṃ pāṭidesanīyaṃ;_  
_Dukkaṭaṃ āpajjeyya ekato,_  
_Pañhā mesā kusalehi cintitā._  

</Paragraph>

<Paragraph number="2380" type="verses" book="pari">
_Ubho paripuṇṇavīsativassā,_  
_Ubhinnaṃ ekupajjhāyo;_  
_Ekācariyo ekā kammavācā,_  
_Eko upasampanno eko anupasampanno;_  
_Pañhā mesā kusalehi cintitā._  

</Paragraph>

<Paragraph number="2381" type="verses" book="pari">
_Akappakataṃ nāpi rajanāya rattaṃ,_  
_Tena nivattho yena kāmaṃ vajeyya;_  
_Na cassa hoti āpatti,_  
_So ca dhammo sugatena desito;_  
_Pañhā mesā kusalehi cintitā._  

</Paragraph>

<Paragraph number="2382" type="verses" book="pari">
_Na deti na paṭiggaṇhāti,_  
_Paṭiggaho tena na vijjati;_  
_Āpajjati garukaṃ na lahukaṃ,_  
_Tañca paribhogapaccayā;_  
_Pañhā mesā kusalehi cintitā._  

</Paragraph>

<Paragraph number="2383" type="verses" book="pari">
_Na deti na paṭiggaṇhāti,_  
_Paṭiggaho tena na vijjati;_  
_Āpajjati lahukaṃ na garukaṃ,_  
_Tañca paribhogapaccayā;_  
_Pañhā mesā kusalehi cintitā._  

</Paragraph>

<Paragraph number="2384" type="verses" book="pari">
_Āpajjati garukaṃ sāvasesaṃ,_  
_Chādeti anādariyaṃ paṭicca;_  
_Na bhikkhunī no ca phuseyya vajjaṃ,_  
_Pañhā mesā kusalehi cintitā._  


</Paragraph>

<Paragraph number="2387" type="verses" book="pari">
_Asaṃvāso avissajji,_  
_Dasa ca anukkhittako;_  
_Upeti dhammaṃ ubbhakkhakaṃ,_  
_Tato saññācikā ca dve._  

</Paragraph>

<Paragraph number="2388" type="verses" book="pari">
_Na kāyikañca garukaṃ,_  
_na kāyikaṃ na vācasikaṃ;_  
_Anālapanto sikkhā ca,_  
_ubho ca caturo janā._  

</Paragraph>

<Paragraph number="2389" type="verses" book="pari">
_Itthī telañca nissaggi,_  
_bhikkhu ca padavītiyo;_  
_Nivattho ca na ca ñatti,_  
_na mātaraṃ pitaraṃ hane._  

</Paragraph>

<Paragraph number="2390" type="verses" book="pari">
_Acodayitvā codayitvā,_  
_chindantaṃ saccameva ca;_  
_Adhiṭṭhitañcatthaṅgate,_  
_na rattaṃ na cāraññakaṃ._  

</Paragraph>

<Paragraph number="2391" type="verses" book="pari">
_Kāyikā vācasikā ca,_  
_tissitthī cāpi mātaraṃ;_  
_Kuddho ārādhako tuṭṭho,_  
_saṃghādisesā ca ubho._  

</Paragraph>

<Paragraph number="2392" type="verses" book="pari">
_Akappakataṃ na deti,_  
_Na detāpajjatī garuṃ;_  
_Sedamocanikā gāthā,_  
_Pañhā viññūhi vibhāvitāti._
</Paragraph>
