<Division number="479" book="pari" e="ch" v="5" p="373">

<Paragraph number="2342" type="verses" book="pari">
_Asaṃvāso bhikkhūhi ca bhikkhunīhi ca,_  
_Sambhogo ekacco tahiṃ na labbhati;_  
_Avippavāsena anāpatti,_  
_Pañhā mesā kusalehi cintitā._  

</Paragraph>

<Paragraph number="2343" type="verses" book="pari">
_Avissajjiyaṃ avebhaṅgiyaṃ,_  
_Pañca vuttā mahesinā;_  
_Vissajjantassa paribhuñjantassa anāpatti,_  
_Pañhā mesā kusalehi cintitā._  

</Paragraph>

<Paragraph number="2344" type="verses" book="pari">
_Dasa puggale na vadāmi,_  
_Ekādasa vivajjiya;_  
_Vuḍḍhaṃ vandantassa āpatti,_  
_Pañhā mesā kusalehi cintitā._  

</Paragraph>

<Paragraph number="2345" type="verses" book="pari">
_Na ukkhittako na ca pana pārivāsiko,_  
_Na saṃghabhinno na ca pana pakkhasaṅkanto;_  
_Samānasaṃvāsakabhūmiyā ṭhito,_  
_Kathaṃ nu sikkhāya asādhāraṇo siyā;_  
_Pañhā mesā kusalehi cintitā._  

</Paragraph>

<Paragraph number="2346" type="verses" book="pari">
_Upeti dhammaṃ paripucchamāno,_  
_Kusalaṃ atthūpasañhitaṃ;_  
_Na jīvati na mato na nibbuto,_  
_Taṃ puggalaṃ katamaṃ vadanti buddhā;_  
_Pañhā mesā kusalehi cintitā._  

</Paragraph>

<Paragraph number="2347" type="verses" book="pari">
_Ubbhakkhake na vadāmi,_  
_Adho nābhiṃ vivajjiya;_  
_Methunadhammapaccayā,_  
_Kathaṃ pārājiko siyā;_  
_Pañhā mesā kusalehi cintitā._  

</Paragraph>

<Paragraph number="2348" type="verses" book="pari">
_Bhikkhu saññācikāya kuṭiṃ karoti,_  
_Adesitavatthukaṃ pamāṇātikkantaṃ;_  
_Sārambhaṃ aparikkamanaṃ anāpatti,_  
_Pañhā mesā kusalehi cintitā._  

</Paragraph>

<Paragraph number="2349" type="verses" book="pari">
_Bhikkhu saññācikāya kuṭiṃ karoti,_  
_Desitavatthukaṃ pamāṇikaṃ;_  
_Anārambhaṃ saparikkamanaṃ āpatti,_  
_Pañhā mesā kusalehi cintitā._  

</Paragraph>

<Paragraph number="2350" type="verses" book="pari">
_Na kāyikaṃ kiñci payogamācare,_  
_Na cāpi vācāya pare bhaṇeyya;_  
_Āpajjeyya garukaṃ chejjavatthuṃ,_  
_Pañhā mesā kusalehi cintitā._  

</Paragraph>

<Paragraph number="2351" type="verses" book="pari">
_Na kāyikaṃ vācasikañca kiñci,_  
_Manasāpi santo na kareyya pāpaṃ;_  
_So nāsito kinti sunāsito bhave,_  
_Pañhā mesā kusalehi cintitā._  

</Paragraph>

<Paragraph number="2352" type="verses" book="pari">
_Anālapanto manujena kenaci,_  
_Vācāgiraṃ no ca pare bhaṇeyya;_  
_Āpajjeyya vācasikaṃ na kāyikaṃ,_  
_Pañhā mesā kusalehi cintitā._  

</Paragraph>

<Paragraph number="2353" type="verses" book="pari">
_Sikkhāpadā buddhavarena vaṇṇitā,_  
_Saṃghādisesā caturo bhaveyyuṃ;_  
_Āpajjeyya ekapayogena sabbe,_  
_Pañhā mesā kusalehi cintitā._  

</Paragraph>

<Paragraph number="2354" type="verses" book="pari">
_Ubho ekato upasampannā,_  
_Ubhinnaṃ hatthato cīvaraṃ paṭiggaṇheyya;_  
_Siyā āpattiyo nānā,_  
_Pañhā mesā kusalehi cintitā._  

</Paragraph>

<Paragraph number="2355" type="verses" book="pari">
_Caturo janā saṃvidhāya,_  
_Garubhaṇḍaṃ avāharuṃ;_  
_Tayo pārājikā eko na pārājiko,_  
_Pañhā mesā kusalehi cintitā._
</Paragraph>
</Division>
