<Paragraph number="341" type="verses" book="saga">
_“Kiṃsu uppatataṃ seṭṭhaṃ,_  
_kiṃsu nipatataṃ varaṃ;_  
_Kiṃsu pavajamānānaṃ,_  
_kiṃsu pavadataṃ varan”ti._  

</Paragraph>

<Paragraph number="342" type="verses" book="saga">
_“Bījaṃ uppatataṃ seṭṭhaṃ,_  
_vuṭṭhi nipatataṃ varā;_  
_Gāvo pavajamānānaṃ,_  
_putto pavadataṃ varo”ti._  

</Paragraph>

<Paragraph number="343" type="verses" book="saga">
_“Vijjā uppatataṃ seṭṭhā,_  
_avijjā nipatataṃ varā;_  
_Saṃgho pavajamānānaṃ,_  
_buddho pavadataṃ varo”ti._
</Paragraph>
</Division>
