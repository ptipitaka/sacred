<Paragraph number="101" type="verses" book="saga">
_“Antojaṭā bahijaṭā,_  
_Jaṭāya jaṭitā pajā;_  
_Taṃ taṃ gotama pucchāmi,_  
_Ko imaṃ vijaṭaye jaṭan”ti._  

</Paragraph>

<Paragraph number="102" type="verses" book="saga">
_“Sīle patiṭṭhāya naro sapañño,_  
_Cittaṃ paññañca bhāvayaṃ;_  
_Ātāpī nipako bhikkhu,_  
_So imaṃ vijaṭaye jaṭaṃ._  

</Paragraph>

<Paragraph number="103" type="verses" book="saga">
_Yesaṃ rāgo ca doso ca,_  
_avijjā ca virājitā;_  
_Khīṇāsavā arahanto,_  
_tesaṃ vijaṭitā jaṭā._  

</Paragraph>

<Paragraph number="104" type="verses" book="saga">
_Yattha nāmañca rūpañca,_  
_asesaṃ uparujjhati;_  
_Paṭighaṃ rūpasaññā ca,_  
_etthesā chijjate jaṭā”ti._
</Paragraph>
</Division>
