<Paragraph number="290" type="verses" book="saga">
_“Kiṃsu ajarasā sādhu,_  
_kiṃsu sādhu adhiṭṭhitaṃ;_  
_Kiṃsu narānaṃ ratanaṃ,_  
_kiṃsu corehyahāriyan”ti._  

</Paragraph>

<Paragraph number="291" type="verses" book="saga">
_“Sīlaṃ ajarasā sādhu,_  
_saddhā sādhu adhiṭṭhitā;_  
_Paññā narānaṃ ratanaṃ,_  
_puññaṃ corehyahāriyan”ti._
</Paragraph>
