<Paragraph number="255" type="verses" book="saga">
_“Kesaṃ divā ca ratto ca,_  
_sadā puññaṃ pavaḍḍhati;_  
_Dhammaṭṭhā sīlasampannā,_  
_ke janā saggagāmino”ti._  

</Paragraph>

<Paragraph number="256" type="verses" book="saga">
_“Ārāmaropā vanaropā,_  
_ye janā setukārakā;_  
_Papañca udapānañca,_  
_ye dadanti upassayaṃ._  

</Paragraph>

<Paragraph number="257" type="verses" book="saga">
_Tesaṃ divā ca ratto ca,_  
_sadā puññaṃ pavaḍḍhati;_  
_Dhammaṭṭhā sīlasampannā,_  
_te janā saggagāmino”ti._
</Paragraph>
</Division>
