<Paragraph number="359" type="verses" book="saga">
_“Kiṃsu lokasmi pajjoto,_  
_kiṃsu lokasmi jāgaro;_  
_Kiṃsu kamme sajīvānaṃ,_  
_kimassa iriyāpatho._  

</Paragraph>

<Paragraph number="360" type="verses" book="saga">
_Kiṃsu alasaṃ analasañca,_  
_Mātā puttaṃva posati;_  
_Kiṃ bhūtā upajīvanti,_  
_Ye pāṇā pathavissitā”ti._  

</Paragraph>

<Paragraph number="361" type="verses" book="saga">
_“Paññā lokasmi pajjoto,_  
_sati lokasmi jāgaro;_  
_Gāvo kamme sajīvānaṃ,_  
_sītassa iriyāpatho._  

</Paragraph>

<Paragraph number="362" type="verses" book="saga">
_Vuṭṭhi alasaṃ analasañca,_  
_Mātā puttaṃva posati;_  
_Vuṭṭhiṃ bhūtā upajīvanti,_  
_Ye pāṇā pathavissitā”ti._
</Paragraph>
</Division>
