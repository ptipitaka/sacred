<Paragraph number="251" type="verses" book="saga">
_“Accharāgaṇasaṅghuṭṭhaṃ,_  
_pisācagaṇasevitaṃ;_  
_Vanantaṃ mohanaṃ nāma,_  
_kathaṃ yātrā bhavissatī”ti._  

</Paragraph>

<Paragraph number="252" type="verses" book="saga">
_“Ujuko nāma so maggo,_  
_abhayā nāma sā disā;_  
_Ratho akūjano nāma,_  
_dhammacakkehi saṃyuto._  

</Paragraph>

<Paragraph number="253" type="verses" book="saga">
_Hirī tassa apālambo,_  
_satyassa parivāraṇaṃ;_  
_Dhammāhaṃ sārathiṃ brūmi,_  
_sammādiṭṭhipurejavaṃ._  

</Paragraph>

<Paragraph number="254" type="verses" book="saga">
_Yassa etādisaṃ yānaṃ,_  
_itthiyā purisassa vā;_  
_Sa ve etena yānena,_  
_nibbānasseva santike”ti._
</Paragraph>
</Division>
