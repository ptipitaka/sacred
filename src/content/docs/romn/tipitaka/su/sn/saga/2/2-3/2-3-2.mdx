<Paragraph number="487" book="saga">
Ekamantaṃ ṭhito kho khemo devaputto bhagavato santike imā gāthāyo abhāsi—
</Paragraph>

<Paragraph number="488" type="verses" book="saga">
_“Caranti bālā dummedhā,_  
_amitteneva attanā;_  
_Karontā pāpakaṃ kammaṃ,_  
_yaṃ hoti kaṭukapphalaṃ._  

</Paragraph>

<Paragraph number="489" type="verses" book="saga">
_Na taṃ kammaṃ kataṃ sādhu,_  
_yaṃ katvā anutappati;_  
_Yassa assumukho rodaṃ,_  
_vipākaṃ paṭisevati._  

</Paragraph>

<Paragraph number="490" type="verses" book="saga">
_Tañca kammaṃ kataṃ sādhu,_  
_yaṃ katvā nānutappati;_  
_Yassa patīto sumano,_  
_vipākaṃ paṭisevati._  

</Paragraph>

<Paragraph number="491" type="verses" book="saga">
_Paṭikacceva taṃ kayirā,_  
_yaṃ jaññā hitamattano;_  
_Na sākaṭikacintāya,_  
_mantā dhīro parakkame._  

</Paragraph>

<Paragraph number="492" type="verses" book="saga">
_Yathā sākaṭiko maṭṭhaṃ,_  
_samaṃ hitvā mahāpathaṃ;_  
_Visamaṃ maggamāruyha,_  
_akkhacchinnova jhāyati._  

</Paragraph>

<Paragraph number="493" type="verses" book="saga">
_Evaṃ dhammā apakkamma,_  
_adhammamanuvattiya;_  
_Mando maccumukhaṃ patto,_  
_akkhacchinnova jhāyatī”ti._
</Paragraph>
</Division>
