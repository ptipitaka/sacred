<Paragraph number="122" type="verses" book="saga">
_“Eṇijaṅghaṃ kisaṃ vīraṃ,_  
_appāhāraṃ alolupaṃ;_  
_Sīhaṃvekacaraṃ nāgaṃ,_  
_kāmesu anapekkhinaṃ;_  
_Upasaṅkamma pucchāma,_  
_kathaṃ dukkhā pamuccatī”ti._  

</Paragraph>

<Paragraph number="123" type="verses" book="saga">
_“Pañca kāmaguṇā loke,_  
_manochaṭṭhā paveditā;_  
_Ettha chandaṃ virājetvā,_  
_evaṃ dukkhā pamuccatī”ti._  


</Paragraph>
</Division>

<Paragraph number="124" type="center" book="saga">
Sattivaggo tatiyo.



</Paragraph>

<Paragraph number="125" type="center" book="saga">
Tassuddānaṃ


</Paragraph>

<Paragraph number="126" type="verses" book="saga">
_Sattiyā phusati ceva,_  
_jaṭā manonivāraṇā;_  
_Arahantena pajjoto,_  
_sarā mahaddhanena ca;_  
_Catucakkena navamaṃ,_  
_eṇijaṅghena te dasāti._
</Paragraph>
