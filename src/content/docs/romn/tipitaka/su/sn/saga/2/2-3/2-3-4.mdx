<Division number="105" book="saga">

<Paragraph number="508" book="saga">
Ekamantaṃ ṭhito kho ghaṭīkāro devaputto bhagavato santike imaṃ gāthaṃ abhāsi—
</Paragraph>

<Paragraph number="509" type="verses" book="saga">
_“Avihaṃ upapannāse,_  
_vimuttā satta bhikkhavo;_  
_Rāgadosaparikkhīṇā,_  
_tiṇṇā loke visattikan”ti._  

</Paragraph>

<Paragraph number="510" type="verses" book="saga">
_“Ke ca te ataruṃ paṅkaṃ,_  
_maccudheyyaṃ suduttaraṃ;_  
_Ke hitvā mānusaṃ dehaṃ,_  
_dibbayogaṃ upaccagun”ti._  

</Paragraph>

<Paragraph number="511" type="verses" book="saga">
_“Upako palagaṇḍo ca,_  
_pukkusāti ca te tayo;_  
_Bhaddiyo khaṇḍadevo ca,_  
_bāhuraggi ca saṅgiyo;_  
_Te hitvā mānusaṃ dehaṃ,_  
_dibbayogaṃ upaccagun”ti._  

</Paragraph>

<Paragraph number="512" type="verses" book="saga">
_“Kusalī bhāsasī tesaṃ,_  
_mārapāsappahāyinaṃ;_  
_Kassa te dhammamaññāya,_  
_acchiduṃ bhavabandhanan”ti._  

</Paragraph>

<Paragraph number="513" type="verses" book="saga">
_“Na aññatra bhagavatā,_  
_nāññatra tava sāsanā;_  
_Yassa te dhammamaññāya,_  
_acchiduṃ bhavabandhanaṃ._  

</Paragraph>

<Paragraph number="514" type="verses" book="saga">
_Yattha nāmañca rūpañca,_  
_asesaṃ uparujjhati;_  
_Taṃ te dhammaṃ idhaññāya,_  
_acchiduṃ bhavabandhanan”ti._  

</Paragraph>

<Paragraph number="515" type="verses" book="saga">
_“Gambhīraṃ bhāsasī vācaṃ,_  
_dubbijānaṃ sudubbudhaṃ;_  
_Kassa tvaṃ dhammamaññāya,_  
_vācaṃ bhāsasi īdisan”ti._  

</Paragraph>

<Paragraph number="516" type="verses" book="saga">
_“Kumbhakāro pure āsiṃ,_  
_vekaḷiṅge ghaṭīkaro;_  
_Mātāpettibharo āsiṃ,_  
_kassapassa upāsako._  

</Paragraph>

<Paragraph number="517" type="verses" book="saga">
_Virato methunā dhammā,_  
_brahmacārī nirāmiso;_  
_Ahuvā te sagāmeyyo,_  
_ahuvā te pure sakhā._  

</Paragraph>

<Paragraph number="518" type="verses" book="saga">
_Sohamete pajānāmi,_  
_vimutte satta bhikkhavo;_  
_Rāgadosaparikkhīṇe,_  
_tiṇṇe loke visattikan”ti._  

</Paragraph>

<Paragraph number="519" type="verses" book="saga">
_“Evametaṃ tadā āsi,_  
_yathā bhāsasi bhaggava;_  
_Kumbhakāro pure āsi,_  
_vekaḷiṅge ghaṭīkaro._  

</Paragraph>

<Paragraph number="520" type="verses" book="saga">
_Mātāpettibharo āsi,_  
_kassapassa upāsako;_  
_Virato methunā dhammā,_  
_brahmacārī nirāmiso;_  
_Ahuvā me sagāmeyyo,_  
_ahuvā me pure sakhā”ti._  

</Paragraph>

<Paragraph number="521" type="verses" book="saga">
_“Evametaṃ purāṇānaṃ,_  
_sahāyānaṃ ahu saṅgamo;_  
_Ubhinnaṃ bhāvitattānaṃ,_  
_sarīrantimadhārinan”ti._
</Paragraph>
</Division>
