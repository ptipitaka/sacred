<Paragraph number="306" type="verses" book="saga">
_“Kiṃsu nidānaṃ gāthānaṃ,_  
_kiṃsu tāsaṃ viyañjanaṃ;_  
_Kiṃsu sannissitā gāthā,_  
_kiṃsu gāthānamāsayo”ti._  

</Paragraph>

<Paragraph number="307" type="verses" book="saga">
_“Chando nidānaṃ gāthānaṃ,_  
_akkharā tāsaṃ viyañjanaṃ;_  
_Nāmasannissitā gāthā,_  
_kavi gāthānamāsayo”ti._  


</Paragraph>
