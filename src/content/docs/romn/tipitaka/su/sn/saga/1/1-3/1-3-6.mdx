<Paragraph number="112" type="verses" book="saga">
_“Kati lokasmiṃ pajjotā,_  
_Tehi loko pakāsati;_  
_Bhagavantaṃ puṭṭhumāgamma,_  
_Kathaṃ jānemu taṃ mayan”ti._  

</Paragraph>

<Paragraph number="113" type="verses" book="saga">
_“Cattāro loke pajjotā,_  
_pañcamettha na vijjati;_  
_Divā tapati ādicco,_  
_rattimābhāti candimā._  

</Paragraph>

<Paragraph number="114" type="verses" book="saga">
_Atha aggi divārattiṃ,_  
_tattha tattha pakāsati;_  
_Sambuddho tapataṃ seṭṭho,_  
_esā ābhā anuttarā”ti._
</Paragraph>
</Division>
