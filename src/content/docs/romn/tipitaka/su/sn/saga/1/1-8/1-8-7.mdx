<Paragraph number="351" type="verses" book="saga">
_“Kiṃsu issariyaṃ loke,_  
_kiṃsu bhaṇḍānamuttamaṃ;_  
_Kiṃsu satthamalaṃ loke,_  
_kiṃsu lokasmimabbudaṃ._  

</Paragraph>

<Paragraph number="352" type="verses" book="saga">
_Kiṃsu harantaṃ vārenti,_  
_haranto pana ko piyo;_  
_Kiṃsu punappunāyantaṃ,_  
_abhinandanti paṇḍitā”ti._  

</Paragraph>

<Paragraph number="353" type="verses" book="saga">
_“Vaso issariyaṃ loke,_  
_itthī bhaṇḍānamuttamaṃ;_  
_Kodho satthamalaṃ loke,_  
_corā lokasmimabbudā._  

</Paragraph>

<Paragraph number="354" type="verses" book="saga">
_Coraṃ harantaṃ vārenti,_  
_haranto samaṇo piyo;_  
_Samaṇaṃ punappunāyantaṃ,_  
_abhinandanti paṇḍitā”ti._
</Paragraph>
</Division>
