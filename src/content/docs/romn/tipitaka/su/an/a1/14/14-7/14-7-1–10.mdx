<Division number="259" book="a1">

<Paragraph number="348" book="a1">

</Paragraph>

</Division>
<Division number="260" book="a1">

<Paragraph number="349" book="a1">

</Paragraph>

</Division>
<Division number="261" book="a1">

<Paragraph number="350" book="a1">
…  Mettāvihārīnaṃ yadidaṃ sāmāvatī. (4)
</Paragraph>

</Division>
<Division number="262" book="a1">

<Paragraph number="351" book="a1">
…  Jhāyīnaṃ yadidaṃ uttarānandamātā. (5)
</Paragraph>

</Division>
<Division number="263" book="a1">

<Paragraph number="352" book="a1">
…  Paṇītadāyikānaṃ yadidaṃ suppavāsā koliyadhītā. (6)
</Paragraph>

</Division>
<Division number="264" book="a1">

<Paragraph number="353" book="a1">

</Paragraph>

</Division>
<Division number="265" book="a1">

<Paragraph number="354" book="a1">

</Paragraph>

</Division>
<Division number="266" book="a1">

<Paragraph number="355" book="a1">

</Paragraph>

<Division number="267" book="a1">

<Paragraph number="356" book="a1">
