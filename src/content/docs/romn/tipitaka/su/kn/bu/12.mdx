<Division number="1" book="bu">

<Paragraph number="566" type="verses" book="bu">
_“Nāradassa aparena,_  
_sambuddho dvipaduttamo;_  
_Padumuttaro nāma jino,_  
_akkhobbho sāgarūpamo._  

</Paragraph>

</Division>
<Division number="2" book="bu">

<Paragraph number="567" type="verses" book="bu">
_Maṇḍakappova so āsi,_  
_yamhi buddho ajāyatha;_  
_Ussannakusalā janatā,_  
_tamhi kappe ajāyatha._  

</Paragraph>

</Division>
<Division number="3" book="bu">

<Paragraph number="568" type="verses" book="bu">
_Padumuttarassa bhagavato,_  
_Paṭhame dhammadesane;_  
_Koṭisatasahassānaṃ,_  
_Dhammābhisamayo ahu._  

</Paragraph>

</Division>
<Division number="4" book="bu">

<Paragraph number="569" type="verses" book="bu">
_Tato parampi vassante,_  
_Tappayante ca pāṇine;_  
_Sattatiṃsasatasahassānaṃ,_  
_Dutiyābhisamayo ahu._  

</Paragraph>

</Division>
<Division number="5" book="bu">

<Paragraph number="570" type="verses" book="bu">
_Yamhi kāle mahāvīro,_  
_ānandaṃ upasaṅkami;_  
_Pitusantikaṃ upagantvā,_  
_āhanī amatadundubhiṃ._  

</Paragraph>

</Division>
<Division number="6" book="bu">

<Paragraph number="571" type="verses" book="bu">
_Āhate amatabherimhi,_  
_vassante dhammavuṭṭhiyā;_  
_Paññāsasatasahassānaṃ,_  
_tatiyābhisamayo ahu._  

</Paragraph>

</Division>
<Division number="7" book="bu">

<Paragraph number="572" type="verses" book="bu">
_Ovādako viññāpako,_  
_tārako sabbapāṇinaṃ;_  
_Desanākusalo buddho,_  
_tāresi janataṃ bahuṃ._  

</Paragraph>

</Division>
<Division number="8" book="bu">

<Paragraph number="573" type="verses" book="bu">
_Sannipātā tayo āsuṃ,_  
_padumuttarassa satthuno;_  
_Koṭisatasahassānaṃ,_  
_paṭhamo āsi samāgamo._  

</Paragraph>

</Division>
<Division number="9" book="bu">

<Paragraph number="574" type="verses" book="bu">
_Yadā buddho asamasamo,_  
_vasi vebhārapabbate;_  
_Navutikoṭisahassānaṃ,_  
_dutiyo āsi samāgamo._  

</Paragraph>

</Division>
<Division number="10" book="bu">

<Paragraph number="575" type="verses" book="bu">
_Puna cārikaṃ pakkante,_  
_gāmanigamaraṭṭhato;_  
_Asītikoṭisahassānaṃ,_  
_tatiyo āsi samāgamo._  

</Paragraph>

</Division>
<Division number="11" book="bu">

<Paragraph number="576" type="verses" book="bu">
_Ahaṃ tena samayena,_  
_Jaṭilo nāma raṭṭhiko;_  
_Sambuddhappamukhaṃ saṃghaṃ,_  
_Sabhattaṃ dussamadāsahaṃ._  

</Paragraph>

</Division>
<Division number="12" book="bu">

<Paragraph number="577" type="verses" book="bu">
_Sopi maṃ buddho byākāsi,_  
_saṃghamajjhe nisīdiya;_  
_‘Satasahassito kappe,_  
_ayaṃ buddho bhavissati._  

</Paragraph>

</Division>
<Division number="13" book="bu">

<Paragraph number="578" type="verses" book="bu">
_Padhānaṃ padahitvāna,_  
_ …pe… _  
_hessāma sammukhā imaṃ’._  

</Paragraph>

</Division>
<Division number="14" book="bu">

<Paragraph number="579" type="verses" book="bu">
_Tassāpi vacanaṃ sutvā,_  
_uttariṃ vatamadhiṭṭhahiṃ;_  
_Akāsiṃ uggadaḷhaṃ dhitiṃ,_  
_dasapāramipūriyā._  

</Paragraph>

</Division>
<Division number="15" book="bu">

<Paragraph number="580" type="verses" book="bu">
_Byāhatā titthiyā sabbe,_  
_Vimanā dummanā tadā;_  
_Na tesaṃ keci paricaranti,_  
_Raṭṭhato nicchubhanti te._  

</Paragraph>

</Division>
<Division number="16" book="bu">

<Paragraph number="581" type="verses" book="bu">
_Sabbe tattha samāgantvā,_  
_upagacchuṃ buddhasantike;_  
_Tuvaṃ nātho mahāvīra,_  
_saraṇaṃ hohi cakkhuma._  

</Paragraph>

</Division>
<Division number="17" book="bu">

<Paragraph number="582" type="verses" book="bu">
_Anukampako kāruṇiko,_  
_hitesī sabbapāṇinaṃ;_  
_Sampatte titthiye sabbe,_  
_pañcasīle patiṭṭhapi._  

</Paragraph>

</Division>
<Division number="18" book="bu">

<Paragraph number="583" type="verses" book="bu">
_Evaṃ nirākulaṃ āsi,_  
_suññataṃ titthiyehi taṃ;_  
_Vicittaṃ arahantehi,_  
_vasībhūtehi tādihi._  

</Paragraph>

</Division>
<Division number="19" book="bu">

<Paragraph number="584" type="verses" book="bu">
_Nagaraṃ haṃsavatī nāma,_  
_ānando nāma khattiyo;_  
_Sujātā nāma janikā,_  
_padumuttarassa satthuno._  

</Paragraph>

</Division>
<Division number="20" book="bu">

<Paragraph number="585" type="verses" book="bu">
_Dasavassasahassāni,_  
_Agāraṃ ajjha so vasi;_  
_Naravāhano yaso vasavattī,_  
_Tayo pāsādamuttamā._  

</Paragraph>

</Division>
<Division number="21" book="bu">

<Paragraph number="586" type="verses" book="bu">
_Ticattārīsasahassāni,_  
_nāriyo samalaṅkatā;_  
_Vasudattā nāma nārī,_  
_uttamo nāma atrajo._  

</Paragraph>

</Division>
<Division number="22" book="bu">

<Paragraph number="587" type="verses" book="bu">
_Nimitte caturo disvā,_  
_pāsādenābhinikkhami;_  
_Sattāhaṃ padhānacāraṃ,_  
_acarī purisuttamo._  

</Paragraph>

</Division>
<Division number="23" book="bu">

<Paragraph number="588" type="verses" book="bu">
_Brahmunā yācito santo,_  
_padumuttaro vināyako;_  
_Vatti cakkaṃ mahāvīro,_  
_mithiluyyānamuttame._  

</Paragraph>

</Division>
<Division number="24" book="bu">

<Paragraph number="589" type="verses" book="bu">
_Devalo ca sujāto ca,_  
_Ahesuṃ aggasāvakā;_  
_Sumano nāmupaṭṭhāko,_  
_Padumuttarassa mahesino._  

</Paragraph>

</Division>
<Division number="25" book="bu">

<Paragraph number="590" type="verses" book="bu">
_Amitā ca asamā ca,_  
_ahesuṃ aggasāvikā;_  
_Bodhi tassa bhagavato,_  
_salaloti pavuccati._  

</Paragraph>

</Division>
<Division number="26" book="bu">

<Paragraph number="591" type="verses" book="bu">
_Vitiṇṇo ceva tisso ca,_  
_ahesuṃ aggupaṭṭhakā;_  
_Haṭṭhā ceva vicittā ca,_  
_ahesuṃ aggupaṭṭhikā._  

</Paragraph>

</Division>
<Division number="27" book="bu">

<Paragraph number="592" type="verses" book="bu">
_Aṭṭhapaṇṇāsaratanaṃ,_  
_accuggato mahāmuni;_  
_Kañcanagghiyasaṅkāso,_  
_dvattiṃsavaralakkhaṇo._  

</Paragraph>

</Division>
<Division number="28" book="bu">

<Paragraph number="593" type="verses" book="bu">
_Kuṭṭā kavāṭā bhittī ca,_  
_rukkhā nagasiluccayā;_  
_Na tassāvaraṇaṃ atthi,_  
_samantā dvādasayojane._  

</Paragraph>

</Division>
<Division number="29" book="bu">

<Paragraph number="594" type="verses" book="bu">
_Vassasatasahassāni,_  
_āyu vijjati tāvade;_  
_Tāvatā tiṭṭhamāno so,_  
_tāresi janataṃ bahuṃ._  

</Paragraph>

</Division>
<Division number="30" book="bu">

<Paragraph number="595" type="verses" book="bu">
_Santāretvā bahujanaṃ,_  
_chinditvā sabbasaṃsayaṃ;_  
_Jalitvā aggikkhandhova,_  
_nibbuto so sasāvako._  

</Paragraph>

</Division>
<Division number="31" book="bu">

<Paragraph number="596" type="verses" book="bu">
_Padumuttaro jino buddho,_  
_Nandārāmamhi nibbuto;_  
_Tatthevassa thūpavaro,_  
_Dvādasubbedhayojano”ti. (584)_  


</Paragraph>
</Division>

<Paragraph number="597" type="center" book="bu">
Padumuttarassa bhagavato vaṃso dasamo.
