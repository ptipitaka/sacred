_“Siddhatthassa aparena,_  
_Asamo appaṭipuggalo;_  
_Anantatejo amitayaso,_  
_Tisso lokagganāyako._  

</Paragraph>

</Division>
<Division number="2" book="bu" bookNo="21" page="341">

<Paragraph number="774" type="verses" book="bu">
_Tamandhakāraṃ vidhamitvā,_  
_obhāsetvā sadevakaṃ;_  
_Anukampako mahāvīro,_  
_loke uppajji cakkhumā._  

</Paragraph>

</Division>
<Division number="3" book="bu" bookNo="21" page="341">

<Paragraph number="775" type="verses" book="bu">
_Tassāpi atulā iddhi,_  
_atulaṃ sīlaṃ samādhi ca;_  
_Sabbattha pāramiṃ gantvā,_  
_dhammacakkaṃ pavattayi._  

</Paragraph>

</Division>
<Division number="4" book="bu" bookNo="21" page="341">

<Paragraph number="776" type="verses" book="bu">
_So buddho dasasahassimhi,_  
_Viññāpesi giraṃ suciṃ;_  
_Koṭisatāni abhisamiṃsu,_  
_Paṭhame dhammadesane._  

</Paragraph>

</Division>
<Division number="5" book="bu" bookNo="21" page="341">

<Paragraph number="777" type="verses" book="bu">
_Dutiyo navutikoṭīnaṃ,_  
_tatiyo saṭṭhikoṭiyo;_  
_Bandhanāto pamocesi,_  
_satte naramarū tadā._  

</Paragraph>

</Division>
<Division number="6" book="bu" bookNo="21" page="341">

<Paragraph number="778" type="verses" book="bu">
_Sannipātā tayo āsuṃ,_  
_tisse lokagganāyake;_  
_Khīṇāsavānaṃ vimalānaṃ,_  
_santacittāna tādinaṃ._  

</Paragraph>

</Division>
<Division number="7" book="bu" bookNo="21" page="341">

<Paragraph number="779" type="verses" book="bu">
_Khīṇāsavasatasahassānaṃ,_  
_Paṭhamo āsi samāgamo;_  
_Navutisatasahassānaṃ,_  
_Dutiyo āsi samāgamo._  

</Paragraph>

</Division>
<Division number="8" book="bu" bookNo="21" page="341">

<Paragraph number="780" type="verses" book="bu">
_Asītisatasahassānaṃ,_  
_tatiyo āsi samāgamo;_  
_Khīṇāsavānaṃ vimalānaṃ,_  
_pupphitānaṃ vimuttiyā._  

</Paragraph>

</Division>
<Division number="9" book="bu" bookNo="21" page="341">

<Paragraph number="781" type="verses" book="bu">
_Ahaṃ tena samayena,_  
_sujāto nāma khattiyo;_  
_Mahābhogaṃ chaḍḍayitvā,_  
_pabbajiṃ isipabbajaṃ._  

</Paragraph>

</Division>
<Division number="10" book="bu" bookNo="21" page="341">

<Paragraph number="782" type="verses" book="bu">
_Mayi pabbajite sante,_  
_uppajji lokanāyako;_  
_Buddhoti saddaṃ sutvāna,_  
_pīti me upapajjatha._  

</Paragraph>

</Division>
<Division number="11" book="bu" bookNo="21" page="341">

<Paragraph number="783" type="verses" book="bu">
_Dibbaṃ mandāravaṃ pupphaṃ,_  
_padumaṃ pārichattakaṃ;_  
_Ubho hatthehi paggayha,_  
_dhunamāno upāgamiṃ._  

</Paragraph>

</Division>
<Division number="12" book="bu" bookNo="21" page="341">

<Paragraph number="784" type="verses" book="bu">
_Catuvaṇṇaparivutaṃ,_  
_tissaṃ lokagganāyakaṃ;_  
_Tamahaṃ pupphaṃ gahetvā,_  
_matthake dhārayiṃ jinaṃ._  

</Paragraph>

</Division>
<Division number="13" book="bu" bookNo="21" page="342">

<Paragraph number="785" type="verses" book="bu">
_Sopi maṃ buddho byākāsi,_  
_janamajjhe nisīdiya;_  
_‘Dvenavute ito kappe,_  
_ayaṃ buddho bhavissati._  

</Paragraph>

</Division>
<Division number="14" book="bu" bookNo="21" page="342">

<Paragraph number="786" type="verses" book="bu">

_ …pe… _  
_hessāma sammukhā imaṃ’._  

</Paragraph>

</Division>
<Division number="15" book="bu" bookNo="21" page="342">

<Paragraph number="787" type="verses" book="bu">
_Tassāpi vacanaṃ sutvā,_  
_bhiyyo cittaṃ pasādayiṃ;_  
_Uttariṃ vatamadhiṭṭhāsiṃ,_  
_dasapāramipūriyā._  

</Paragraph>

</Division>
<Division number="16" book="bu" bookNo="21" page="342">

<Paragraph number="788" type="verses" book="bu">
_Khemakaṃ nāma nagaraṃ,_  
_janasandho nāma khattiyo;_  
_Padumā nāma janikā,_  
_tissassa ca mahesino._  

</Paragraph>

</Division>
<Division number="17" book="bu" bookNo="21" page="342">

<Paragraph number="789" type="verses" book="bu">
_Sattavassasahassāni,_  
_Agāraṃ ajjha so vasi;_  
_Guhāsela nārisaya nisabhā,_  
_Tayo pāsādamuttamā._  

</Paragraph>

</Division>
<Division number="18" book="bu" bookNo="21" page="342">

<Paragraph number="790" type="verses" book="bu">
_Samatiṃsasahassāni,_  
_nāriyo samalaṅkatā;_  
_Subhaddānāmikā nārī,_  
_ānando nāma atrajo._  

</Paragraph>

</Division>
<Division number="19" book="bu" bookNo="21" page="342">

<Paragraph number="791" type="verses" book="bu">
_Nimitte caturo disvā,_  
_assayānena nikkhami;_  
_Anūnaaṭṭhamāsāni,_  
_padhānaṃ padahī jino._  

</Paragraph>

</Division>
<Division number="20" book="bu" bookNo="21" page="342">

<Paragraph number="792" type="verses" book="bu">
_Brahmunā yācito santo,_  
_tisso lokagganāyako;_  
_Vatti cakkaṃ mahāvīro,_  
_yasavatiyamuttame._  

</Paragraph>

</Division>
<Division number="21" book="bu" bookNo="21" page="342">

<Paragraph number="793" type="verses" book="bu">
_Brahmadevo udayo ca,_  
_ahesuṃ aggasāvakā;_  
_Samaṅgo nāmupaṭṭhāko,_  
_tissassa ca mahesino._  

</Paragraph>

</Division>
<Division number="22" book="bu" bookNo="21" page="342">

<Paragraph number="794" type="verses" book="bu">
_Phussā ceva sudattā ca,_  
_ahesuṃ aggasāvikā;_  
_Bodhi tassa bhagavato,_  
_asanoti pavuccati._  

</Paragraph>

</Division>
<Division number="23" book="bu" bookNo="21" page="342">

<Paragraph number="795" type="verses" book="bu">
_Sambalo ca sirimā ceva,_  
_ahesuṃ aggupaṭṭhakā;_  
_Kisāgotamī upasenā,_  
_ahesuṃ aggupaṭṭhikā._  

</Paragraph>

</Division>

<Paragraph number="796" type="verses" book="bu">

_ahu uccattane jino;_  
_Anūpamo asadiso,_  
_himavā viya dissati._  

</Paragraph>

</Division>

_āyu āsi anuttaro;_  
_Vassasatasahassāni,_  
_loke aṭṭhāsi cakkhumā._  

</Paragraph>

_anubhotvā mahāyasaṃ;_  
_Jalitvā aggikkhandhova,_  
_nibbuto so sasāvako._  

</Paragraph>
