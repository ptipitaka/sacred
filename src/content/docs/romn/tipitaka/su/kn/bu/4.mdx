<Paragraph number="304" type="verses" book="bu">
_“Dīpaṅkarassa aparena,_  
_Koṇḍañño nāma nāyako;_  
_Anantatejo amitayaso,_  
_Appameyyo durāsado._  

</Paragraph>

</Division>
<Division number="2" book="bu" bookNo="21" page="371">

<Paragraph number="305" type="verses" book="bu">
_Dharaṇūpamo khamanena,_  
_sīlena sāgarūpamo;_  
_Samādhinā merūpamo,_  
_ñāṇena gaganūpamo._  

</Paragraph>

</Division>
<Division number="3" book="bu" bookNo="21" page="371">

<Paragraph number="306" type="verses" book="bu">
_Indriyabalabojjhaṅga-_  
_maggasaccappakāsanaṃ;_  
_Pakāsesi sadā buddho,_  
_hitāya sabbapāṇinaṃ._  

</Paragraph>

</Division>
<Division number="4" book="bu" bookNo="21" page="371">

<Paragraph number="307" type="verses" book="bu">
_Dhammacakkaṃ pavattente,_  
_koṇḍaññe lokanāyake;_  
_Koṭisatasahassānaṃ,_  
_paṭhamābhisamayo ahu._  

</Paragraph>

</Division>
<Division number="5" book="bu" bookNo="21" page="371">

<Paragraph number="308" type="verses" book="bu">
_Tato parampi desente,_  
_naramarūnaṃ samāgame;_  
_Navutikoṭisahassānaṃ,_  
_dutiyābhisamayo ahu._  

</Paragraph>

</Division>
<Division number="6" book="bu" bookNo="21" page="371">

<Paragraph number="309" type="verses" book="bu">
_Titthiye abhimaddanto,_  
_yadā dhammamadesayi;_  
_Asītikoṭisahassānaṃ,_  
_tatiyābhisamayo ahu._  

</Paragraph>

</Division>
<Division number="7" book="bu" bookNo="21" page="371">

<Paragraph number="310" type="verses" book="bu">
_Sannipātā tayo āsuṃ,_  
_koṇḍaññassa mahesino;_  
_Khīṇāsavānaṃ vimalānaṃ,_  
_santacittāna tādinaṃ._  

</Paragraph>

</Division>
<Division number="8" book="bu" bookNo="21" page="371">

<Paragraph number="311" type="verses" book="bu">
_Koṭisatasahassānaṃ,_  
_paṭhamo āsi samāgamo;_  
_Dutiyo koṭisahassānaṃ,_  
_tatiyo navutikoṭinaṃ._  

</Paragraph>

</Division>
<Division number="9" book="bu" bookNo="21" page="371">

<Paragraph number="312" type="verses" book="bu">
_Ahaṃ tena samayena,_  
_vijitāvī nāma khattiyo;_  
_Samuddaṃ antamantena,_  
_issariyaṃ vattayāmahaṃ._  

</Paragraph>

</Division>
<Division number="10" book="bu" bookNo="21" page="372">

<Paragraph number="313" type="verses" book="bu">
_Koṭisatasahassānaṃ,_  
_vimalānaṃ mahesinaṃ;_  
_Saha lokagganāthena,_  
_paramannena tappayiṃ._  

</Paragraph>

</Division>
<Division number="11" book="bu" bookNo="21" page="372">

<Paragraph number="314" type="verses" book="bu">
_Sopi maṃ buddho byākāsi,_  
_koṇḍañño lokanāyako;_  
_‘Aparimeyyito kappe,_  
_buddho loke bhavissati._  

</Paragraph>

</Division>
<Division number="12" book="bu" bookNo="21" page="372">

<Paragraph number="315" type="verses" book="bu">
_Padhānaṃ padahitvāna,_  
_katvā dukkarakārikaṃ;_  
_Assatthamūle sambuddho,_  
_bujjhissati mahāyaso._  

</Paragraph>

</Division>
<Division number="13" book="bu" bookNo="21" page="372">

<Paragraph number="316" type="verses" book="bu">
_Imassa janikā mātā,_  
_māyā nāma bhavissati;_  
_Pitā suddhodano nāma,_  
_ayaṃ hessati gotamo._  

</Paragraph>

</Division>
<Division number="14" book="bu" bookNo="21" page="372">

<Paragraph number="317" type="verses" book="bu">
_Kolito upatisso ca,_  
_aggā hessanti sāvakā;_  
_Ānando nāmupaṭṭhāko,_  
_upaṭṭhissatimaṃ jinaṃ._  

</Paragraph>

</Division>
<Division number="15" book="bu" bookNo="21" page="372">

<Paragraph number="318" type="verses" book="bu">
_Khemā uppalavaṇṇā ca,_  
_aggā hessanti sāvikā;_  
_Bodhi tassa bhagavato,_  
_assatthoti pavuccati._  

</Paragraph>

</Division>
<Division number="16" book="bu" bookNo="21" page="372">

<Paragraph number="319" type="verses" book="bu">
_Citto ca hatthāḷavako,_  
_aggā hessantupaṭṭhakā;_  
_Nandamātā ca uttarā,_  
_aggā hessantupaṭṭhikā;_  
_Āyu vassasataṃ tassa,_  
_gotamassa yasassino’._  

</Paragraph>

</Division>
<Division number="17" book="bu" bookNo="21" page="372">

<Paragraph number="320" type="verses" book="bu">
_Idaṃ sutvāna vacanaṃ,_  
_asamassa mahesino;_  
_Āmoditā naramarū,_  
_‘buddhabījaṃ kira ayaṃ’._  

</Paragraph>

</Division>
<Division number="18" book="bu" bookNo="21" page="372">

<Paragraph number="321" type="verses" book="bu">
_Ukkuṭṭhisaddā vattanti,_  
_apphoṭenti hasanti ca;_  
_Katañjalī namassanti,_  
_dasasahassidevatā._  

</Paragraph>

</Division>
<Division number="19" book="bu" bookNo="21" page="370">

<Paragraph number="322" type="verses" book="bu">
_‘Yadimassa lokanāthassa,_  
_virajjhissāma sāsanaṃ;_  
_Anāgatamhi addhāne,_  
_hessāma sammukhā imaṃ._  

</Paragraph>

</Division>
<Division number="20" book="bu" bookNo="21" page="370">

<Paragraph number="323" type="verses" book="bu">
_Yathā manussā nadiṃ tarantā,_  
_Paṭititthaṃ virajjhiya;_  
_Heṭṭhātitthe gahetvāna,_  
_Uttaranti mahānadiṃ._  

</Paragraph>

</Division>
<Division number="21" book="bu" bookNo="21" page="368">

<Paragraph number="324" type="verses" book="bu">
_Evamevaṃ mayaṃ sabbe,_  
_yadi muñcāmimaṃ jinaṃ;_  
_Anāgatamhi addhāne,_  
_hessāma sammukhā imaṃ’._  

</Paragraph>

</Division>
<Division number="22" book="bu" bookNo="21" page="368">

<Paragraph number="325" type="verses" book="bu">
_Tassāhaṃ vacanaṃ sutvā,_  
_bhiyyo cittaṃ pasādayiṃ;_  
_Tameva atthaṃ sādhento,_  
_mahārajjaṃ jine adaṃ;_  
_Mahārajjaṃ daditvāna,_  
_pabbajiṃ tassa santike._  

</Paragraph>

</Division>
<Division number="23" book="bu" bookNo="21" page="368">

<Paragraph number="326" type="verses" book="bu">
_Suttantaṃ vinayañcāpi,_  
_navaṅgaṃ satthusāsanaṃ;_  
_Sabbaṃ pariyāpuṇitvāna,_  
_sobhayiṃ jinasāsanaṃ._  

</Paragraph>

</Division>
<Division number="24" book="bu" bookNo="21" page="368">

<Paragraph number="327" type="verses" book="bu">
_Tatthappamatto viharanto,_  
_nisajjaṭṭhānacaṅkame;_  
_Abhiññāpāramiṃ gantvā,_  
_brahmalokamagañchahaṃ._  

</Paragraph>

</Division>
<Division number="25" book="bu" bookNo="21" page="365">

<Paragraph number="328" type="verses" book="bu">
_Nagaraṃ rammavatī nāma,_  
_sunando nāma khattiyo;_  
_Sujātā nāma janikā,_  
_koṇḍaññassa mahesino._  

</Paragraph>

</Division>
<Division number="26" book="bu" bookNo="21" page="365">

<Paragraph number="329" type="verses" book="bu">
_Dasavassasahassāni,_  
_agāraṃ ajjha so vasi;_  
_Suci suruci subho ca,_  
_tayo pāsādamuttamā._  

</Paragraph>

</Division>
<Division number="27" book="bu" bookNo="21" page="365">

<Paragraph number="330" type="verses" book="bu">
_Tīṇi satasahassāni,_  
_nāriyo samalaṅkatā;_  
_Rucidevī nāma nārī,_  
_vijitaseno atrajo._  

</Paragraph>

</Division>
<Division number="28" book="bu" bookNo="21" page="361">

<Paragraph number="331" type="verses" book="bu">
_Nimitte caturo disvā,_  
_rathayānena nikkhami;_  
_Anūnadasamāsāni,_  
_padhānaṃ padahī jino._  

</Paragraph>

</Division>
<Division number="29" book="bu" bookNo="21" page="348">

<Paragraph number="332" type="verses" book="bu">
_Brahmunā yācito santo,_  
_koṇḍañño dvipaduttamo;_  
_Vatti cakkaṃ mahāvīro,_  
_devānaṃ nagaruttame._  

</Paragraph>

</Division>
<Division number="30" book="bu" bookNo="21" page="350">

<Paragraph number="333" type="verses" book="bu">
_Bhaddo ceva subhaddo ca,_  
_ahesuṃ aggasāvakā;_  
_Anuruddho nāmupaṭṭhāko,_  
_koṇḍaññassa mahesino._  

</Paragraph>

</Division>
<Division number="31" book="bu" bookNo="21" page="350">

<Paragraph number="334" type="verses" book="bu">
_Tissā ca upatissā ca,_  
_ahesuṃ aggasāvikā;_  
_Sālakalyāṇiko bodhi,_  
_koṇḍaññassa mahesino._  

</Paragraph>

</Division>
<Division number="32" book="bu" bookNo="21" page="350">

<Paragraph number="335" type="verses" book="bu">
_Soṇo ca upasoṇo ca,_  
_ahesuṃ aggupaṭṭhakā;_  
_Nandā ceva sirīmā ca,_  
_ahesuṃ aggupaṭṭhikā._  

</Paragraph>

</Division>
<Division number="33" book="bu" bookNo="21" page="366">

<Paragraph number="336" type="verses" book="bu">
_So aṭṭhāsīti hatthāni,_  
_accuggato mahāmuni;_  
_Sobhate uḷurājāva,_  
_sūriyo majjhanhike yathā._  

</Paragraph>

</Division>
<Division number="34" book="bu" bookNo="21" page="366">

<Paragraph number="337" type="verses" book="bu">
_Vassasatasahassāni,_  
_āyu vijjati tāvade;_  
_Tāvatā tiṭṭhamāno so,_  
_tāresi janataṃ bahuṃ._  

</Paragraph>

</Division>
<Division number="35" book="bu" bookNo="21" page="378">

<Paragraph number="338" type="verses" book="bu">
_Khīṇāsavehi vimalehi,_  
_vicittā āsi medanī;_  
_Yathā gaganamuḷūhi,_  
_evaṃ so upasobhatha._  

</Paragraph>

</Division>
<Division number="36" book="bu" bookNo="21" page="378">

<Paragraph number="339" type="verses" book="bu">
_Tepi nāgā appameyyā,_  
_asaṅkhobbhā durāsadā;_  
_Vijjupātaṃva dassetvā,_  
_nibbutā te mahāyasā._  

</Paragraph>

</Division>
<Division number="37" book="bu" bookNo="21" page="378">

<Paragraph number="340" type="verses" book="bu">
_Sā ca atuliyā jinassa iddhi,_  
_Ñāṇaparibhāvito ca samādhi;_  
_Sabbaṃ tamantarahitaṃ,_  
_Nanu rittā sabbasaṅkhārā._  

</Paragraph>

</Division>
