<Paragraph number="954" type="verses" book="bu">
_“Kakusandhassa aparena,_  
_sambuddho dvipaduttamo;_  
_Koṇāgamano nāma jino,_  
_lokajeṭṭho narāsabho._  

</Paragraph>

</Division>
<Division number="2" book="bu" bookNo="21" page="357">

<Paragraph number="955" type="verses" book="bu">
_Dasa dhamme pūrayitvāna,_  
_kantāraṃ samatikkami;_  
_Pavāhiya malaṃ sabbaṃ,_  
_patto sambodhimuttamaṃ._  

</Paragraph>

</Division>
<Division number="3" book="bu" bookNo="21" page="357">

<Paragraph number="956" type="verses" book="bu">
_Dhammacakkaṃ pavattente,_  
_koṇāgamananāyake;_  
_Tiṃsakoṭisahassānaṃ,_  
_paṭhamābhisamayo ahu._  

</Paragraph>

</Division>
<Division number="4" book="bu" bookNo="21" page="357">

<Paragraph number="957" type="verses" book="bu">
_Pāṭihīraṃ karonte ca,_  
_paravādappamaddane;_  
_Vīsatikoṭisahassānaṃ,_  
_dutiyābhisamayo ahu._  

</Paragraph>

</Division>
<Division number="5" book="bu" bookNo="21" page="357">

<Paragraph number="958" type="verses" book="bu">
_Tato vikubbanaṃ katvā,_  
_jino devapuraṃ gato;_  
_Vasate tattha sambuddho,_  
_silāya paṇḍukambale._  

</Paragraph>

</Division>
<Division number="6" book="bu" bookNo="21" page="357">

<Paragraph number="959" type="verses" book="bu">
_Pakaraṇe satta desento,_  
_vassaṃ vasati so muni;_  
_Dasakoṭisahassānaṃ,_  
_tatiyābhisamayo ahu._  

</Paragraph>

</Division>
<Division number="7" book="bu" bookNo="21" page="358">

<Paragraph number="960" type="verses" book="bu">
_Tassāpi devadevassa,_  
_eko āsi samāgamo;_  
_Khīṇāsavānaṃ vimalānaṃ,_  
_santacittāna tādinaṃ._  

</Paragraph>

</Division>
<Division number="8" book="bu" bookNo="21" page="358">

<Paragraph number="961" type="verses" book="bu">
_Tiṃsabhikkhusahassānaṃ,_  
_tadā āsi samāgamo;_  
_Oghānamatikkantānaṃ,_  
_bhijjitānañca maccuyā._  

</Paragraph>

</Division>
<Division number="9" book="bu" bookNo="21" page="358">

<Paragraph number="962" type="verses" book="bu">
_Ahaṃ tena samayena,_  
_pabbato nāma khattiyo;_  
_Mittāmaccehi sampanno,_  
_anantabalavāhano._  

</Paragraph>

</Division>
<Division number="10" book="bu" bookNo="21" page="358">

<Paragraph number="963" type="verses" book="bu">
_Sambuddhadassanaṃ gantvā,_  
_sutvā dhammamanuttaraṃ;_  
_Nimantetvā sajinasaṃghaṃ,_  
_dānaṃ datvā yadicchakaṃ._  

</Paragraph>

</Division>
<Division number="11" book="bu" bookNo="21" page="358">

<Paragraph number="964" type="verses" book="bu">
_Pattuṇṇaṃ cīnapaṭṭañca,_  
_koseyyaṃ kambalampi ca;_  
_Sovaṇṇapādukañceva,_  
_adāsiṃ satthusāvake._  

</Paragraph>

</Division>
<Division number="12" book="bu" bookNo="21" page="358">

<Paragraph number="965" type="verses" book="bu">
_Sopi maṃ buddho byākāsi,_  
_saṃghamajjhe nisīdiya;_  
_‘Imamhi bhaddake kappe,_  
_ayaṃ buddho bhavissati._  

</Paragraph>

</Division>
<Division number="13" book="bu" bookNo="21" page="358">

<Paragraph number="966" type="verses" book="bu">
_Ahu kapilavhayā rammā,_  
_ …pe… _  
_hessāma sammukhā imaṃ’._  

</Paragraph>

</Division>
<Division number="14" book="bu" bookNo="21" page="358">

<Paragraph number="967" type="verses" book="bu">
_Tassāpi vacanaṃ sutvā,_  
_bhiyyo cittaṃ pasādayiṃ;_  
_Uttariṃ vatamadhiṭṭhāsiṃ,_  
_dasapāramipūriyā._  

</Paragraph>

</Division>
<Division number="15" book="bu" bookNo="21" page="358">

<Paragraph number="968" type="verses" book="bu">
_Sabbaññutaṃ gavesanto,_  
_dānaṃ datvā naruttame;_  
_Ohāyāhaṃ mahārajjaṃ,_  
_pabbajiṃ jinasantike._  

</Paragraph>

</Division>
<Division number="16" book="bu" bookNo="21" page="358">

<Paragraph number="969" type="verses" book="bu">
_Nagaraṃ sobhavatī nāma,_  
_sobho nāmāsi khattiyo;_  
_Vasate tattha nagare,_  
_sambuddhassa mahākulaṃ._  

</Paragraph>

</Division>
<Division number="17" book="bu" bookNo="21" page="358">

<Paragraph number="970" type="verses" book="bu">
_Brāhmaṇo yaññadatto ca,_  
_āsi buddhassa so pitā;_  
_Uttarā nāma janikā,_  
_koṇāgamanassa satthuno._  

</Paragraph>

</Division>
<Division number="18" book="bu" bookNo="21" page="358">

<Paragraph number="971" type="verses" book="bu">
_Tīṇi vassasahassāni,_  
_Agāraṃ ajjha so vasi;_  
_Tusitasantusitasantuṭṭhā,_  
_Tayo pāsādamuttamā._  

</Paragraph>

</Division>
<Division number="19" book="bu" bookNo="21" page="358">

<Paragraph number="972" type="verses" book="bu">
_Anūnasoḷasasahassāni,_  
_Nāriyo samalaṅkatā;_  
_Rucigattā nāma nārī,_  
_Satthavāho nāma atrajo._  

</Paragraph>

</Division>
<Division number="20" book="bu" bookNo="21" page="358">

<Paragraph number="973" type="verses" book="bu">
_Nimitte caturo disvā,_  
_hatthiyānena nikkhami;_  
_Chamāsaṃ padhānacāraṃ,_  
_acarī purisuttamo._  

</Paragraph>

</Division>
<Division number="21" book="bu" bookNo="21" page="359">

<Paragraph number="974" type="verses" book="bu">
_Brahmunā yācito santo,_  
_koṇāgamananāyako;_  
_Vatti cakkaṃ mahāvīro,_  
_migadāye naruttamo._  

</Paragraph>

</Division>
<Division number="22" book="bu" bookNo="21" page="359">

<Paragraph number="975" type="verses" book="bu">
_Bhiyyaso uttaro nāma,_  
_ahesuṃ aggasāvakā;_  
_Sotthijo nāmupaṭṭhāko,_  
_koṇāgamanassa satthuno._  

</Paragraph>

</Division>
<Division number="23" book="bu" bookNo="21" page="359">

<Paragraph number="976" type="verses" book="bu">
_Samuddā uttarā ceva,_  
_ahesuṃ aggasāvikā;_  
_Bodhi tassa bhagavato,_  
_udumbaroti pavuccati._  

</Paragraph>

</Division>
<Division number="24" book="bu" bookNo="21" page="359">

<Paragraph number="977" type="verses" book="bu">
_Uggo ca somadevo ca,_  
_ahesuṃ aggupaṭṭhakā;_  
_Sīvalā ceva sāmā ca,_  
_ahesuṃ aggupaṭṭhikā._  

</Paragraph>

</Division>
<Division number="25" book="bu" bookNo="21" page="357">

<Paragraph number="978" type="verses" book="bu">
_Uccattanena so buddho,_  
_tiṃsahatthasamuggato;_  
_Ukkāmukhe yathā kambu,_  
_evaṃ raṃsīhi maṇḍito._  

</Paragraph>

</Division>
<Division number="26" book="bu" bookNo="21" page="353">

<Paragraph number="979" type="verses" book="bu">
_Tiṃsavassasahassāni,_  
_āyu buddhassa tāvade;_  
_Tāvatā tiṭṭhamāno so,_  
_tāresi janataṃ bahuṃ._  

</Paragraph>

</Division>
<Division number="27" book="bu" bookNo="21" page="350">

<Paragraph number="980" type="verses" book="bu">
_Dhammacetiṃ samussetvā,_  
_dhammadussavibhūsitaṃ;_  
_Dhammapupphaguḷaṃ katvā,_  
_nibbuto so sasāvako._  

</Paragraph>

</Division>
<Division number="28" book="bu" bookNo="21" page="345">

<Paragraph number="981" type="verses" book="bu">
_Mahāvilāso tassa jano,_  
_siridhammappakāsano;_  
_Sabbaṃ tamantarahitaṃ,_  
_nanu rittā sabbasaṅkhārā._  

</Paragraph>

</Division>

_Pabbatārāmamhi nibbuto;_  
_Dhātuvitthārikaṃ āsi,_  
_Tesu tesu padesato”ti. (957)_  


</Paragraph>
