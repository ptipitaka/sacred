<Paragraph number="376" type="verses" book="bu">
_“Maṅgalassa aparena,_  
_sumano nāma nāyako;_  
_Sabbadhammehi asamo,_  
_sabbasattānamuttamo._  

</Paragraph>

</Division>
<Division number="2" book="bu">

<Paragraph number="377" type="verses" book="bu">
_Tadā amatabheriṃ so,_  
_āhanī mekhale pure;_  
_Dhammasaṅkhasamāyuttaṃ,_  
_navaṅgaṃ jinasāsanaṃ._  

</Paragraph>

</Division>
<Division number="3" book="bu">

<Paragraph number="378" type="verses" book="bu">
_Nijjinitvā kilese so,_  
_patvā sambodhimuttamaṃ;_  
_Māpesi nagaraṃ satthā,_  
_saddhammapuravaruttamaṃ._  

</Paragraph>

</Division>
<Division number="4" book="bu">

<Paragraph number="379" type="verses" book="bu">
_Nirantaraṃ akuṭilaṃ,_  
_ujuṃ vipulavitthataṃ;_  
_Māpesi so mahāvīthiṃ,_  
_satipaṭṭhānavaruttamaṃ._  

</Paragraph>

</Division>
<Division number="5" book="bu">

<Paragraph number="380" type="verses" book="bu">
_Phale cattāri sāmaññe,_  
_catasso paṭisambhidā;_  
_Chaḷabhiññāṭṭhasamāpattī,_  
_pasāresi tattha vīthiyaṃ._  

</Paragraph>

</Division>
<Division number="6" book="bu">

<Paragraph number="381" type="verses" book="bu">
_Ye appamattā akhilā,_  
_hirivīriyehupāgatā;_  
_Te te ime guṇavare,_  
_ādiyanti yathā sukhaṃ._  

</Paragraph>

</Division>
<Division number="7" book="bu">

<Paragraph number="382" type="verses" book="bu">
_Evametena yogena,_  
_uddharanto mahājanaṃ;_  
_Bodhesi paṭhamaṃ satthā,_  
_koṭisatasahassiyo._  

</Paragraph>

</Division>
<Division number="8" book="bu">

<Paragraph number="383" type="verses" book="bu">
_Yamhi kāle mahāvīro,_  
_ovadī titthiye gaṇe;_  
_Koṭisahassābhisamiṃsu,_  
_dutiye dhammadesane._  

</Paragraph>

</Division>
<Division number="9" book="bu">

<Paragraph number="384" type="verses" book="bu">
_Yadā devā manussā ca,_  
_samaggā ekamānasā;_  
_Nirodhapañhaṃ pucchiṃsu,_  
_saṃsayañcāpi mānasaṃ._  

</Paragraph>

</Division>
<Division number="10" book="bu">

<Paragraph number="385" type="verses" book="bu">
_Tadāpi dhammadesane,_  
_nirodhaparidīpane;_  
_Navutikoṭisahassānaṃ,_  
_tatiyābhisamayo ahu._  

</Paragraph>

</Division>
<Division number="11" book="bu">

<Paragraph number="386" type="verses" book="bu">
_Sannipātā tayo āsuṃ,_  
_sumanassa mahesino;_  
_Khīṇāsavānaṃ vimalānaṃ,_  
_santacittāna tādinaṃ._  

</Paragraph>

</Division>
<Division number="12" book="bu">

<Paragraph number="387" type="verses" book="bu">
_Vassaṃvutthassa bhagavato,_  
_abhighuṭṭhe pavāraṇe;_  
_Koṭisatasahassehi,_  
_pavāresi tathāgato._  

</Paragraph>

</Division>
<Division number="13" book="bu">

<Paragraph number="388" type="verses" book="bu">
_Tatoparaṃ sannipāte,_  
_vimale kañcanapabbate;_  
_Navutikoṭisahassānaṃ,_  
_dutiyo āsi samāgamo._  

</Paragraph>

</Division>
<Division number="14" book="bu">

<Paragraph number="389" type="verses" book="bu">
_Yadā sakko devarājā,_  
_buddhadassanupāgami;_  
_Asītikoṭisahassānaṃ,_  
_tatiyo āsi samāgamo._  

</Paragraph>

</Division>
<Division number="15" book="bu">

<Paragraph number="390" type="verses" book="bu">
_Ahaṃ tena samayena,_  
_nāgarājā mahiddhiko;_  
_Atulo nāma nāmena,_  
_ussannakusalasañcayo._  

</Paragraph>

</Division>
<Division number="16" book="bu">

<Paragraph number="391" type="verses" book="bu">
_Tadāhaṃ nāgabhavanā,_  
_nikkhamitvā sañātibhi;_  
_Nāgānaṃ dibbaturiyehi,_  
_sasaṃghaṃ jinamupaṭṭhahiṃ._  

</Paragraph>

</Division>
<Division number="17" book="bu">

<Paragraph number="392" type="verses" book="bu">
_Koṭisatasahassānaṃ,_  
_annapānena tappayiṃ;_  
_Paccekadussayugaṃ datvā,_  
_saraṇaṃ tamupāgamiṃ._  

</Paragraph>

</Division>
<Division number="18" book="bu">

<Paragraph number="393" type="verses" book="bu">
_Sopi maṃ buddho byākāsi,_  
_sumano lokanāyako;_  
_‘Aparimeyyito kappe,_  
_ayaṃ buddho bhavissati._  

</Paragraph>

</Division>
<Division number="19" book="bu">

<Paragraph number="394" type="verses" book="bu">
_Padhānaṃ padahitvāna,_  
_ …pe… _  
_hessāma sammukhā imaṃ’._  

</Paragraph>

</Division>
<Division number="20" book="bu">

<Paragraph number="395" type="verses" book="bu">
_Tassāpi vacanaṃ sutvā,_  
_bhiyyo cittaṃ pasādayiṃ;_  
_Uttariṃ vatamadhiṭṭhāsiṃ,_  
_dasapāramipūriyā._  

</Paragraph>

</Division>
<Division number="21" book="bu">

<Paragraph number="396" type="verses" book="bu">
_Nagaraṃ mekhalaṃ nāma,_  
_sudatto nāma khattiyo;_  
_Sirimā nāma janikā,_  
_sumanassa mahesino._  

</Paragraph>

</Division>
<Division number="22" book="bu">

<Paragraph number="397" type="verses" book="bu">
_Navavassasahassāni,_  
_agāraṃ ajjha so vasi;_  
_Cando sucando vaṭaṃso ca,_  
_tayo pāsādamuttamā._  

</Paragraph>

</Division>
<Division number="23" book="bu">

<Paragraph number="398" type="verses" book="bu">
_Tesaṭṭhisatasahassāni,_  
_nāriyo samalaṅkatā;_  
_Vaṭaṃsikā nāma nārī,_  
_anūpamo nāma atrajo._  

</Paragraph>

</Division>
<Division number="24" book="bu">

<Paragraph number="399" type="verses" book="bu">
_Nimitte caturo disvā,_  
_hatthiyānena nikkhami;_  
_Anūnadasamāsāni,_  
_padhānaṃ padahī jino._  

</Paragraph>

</Division>
<Division number="25" book="bu">

<Paragraph number="400" type="verses" book="bu">
_Brahmunā yācito santo,_  
_sumano lokanāyako;_  
_Vatti cakkaṃ mahāvīro,_  
_mekhale puramuttame._  

</Paragraph>

</Division>
<Division number="26" book="bu">

<Paragraph number="401" type="verses" book="bu">
_Saraṇo bhāvitatto ca,_  
_ahesuṃ aggasāvakā;_  
_Udeno nāmupaṭṭhāko,_  
_sumanassa mahesino._  

</Paragraph>

</Division>
<Division number="27" book="bu">

<Paragraph number="402" type="verses" book="bu">
_Soṇā ca upasoṇā ca,_  
_ahesuṃ aggasāvikā;_  
_Sopi buddho amitayaso,_  
_nāgamūle abujjhatha._  

</Paragraph>

</Division>
<Division number="28" book="bu">

<Paragraph number="403" type="verses" book="bu">
_Varuṇo ceva saraṇo ca,_  
_ahesuṃ aggupaṭṭhakā;_  
_Cālā ca upacālā ca,_  
_ahesuṃ aggupaṭṭhikā._  

</Paragraph>

</Division>
<Division number="29" book="bu">

<Paragraph number="404" type="verses" book="bu">
_Uccattanena so buddho,_  
_navutihatthamuggato;_  
_Kañcanagghiyasaṅkāso,_  
_dasasahassī virocati._  

</Paragraph>

</Division>
<Division number="30" book="bu">

<Paragraph number="405" type="verses" book="bu">
_Navutivassasahassāni,_  
_āyu vijjati tāvade;_  
_Tāvatā tiṭṭhamāno so,_  
_tāresi janataṃ bahuṃ._  

</Paragraph>

</Division>
<Division number="31" book="bu">

<Paragraph number="406" type="verses" book="bu">
_Tāraṇīye tārayitvā,_  
_bodhanīye ca bodhayi;_  
_Parinibbāyi sambuddho,_  
_uḷurājāva atthami._  

</Paragraph>

</Division>
<Division number="32" book="bu">

<Paragraph number="407" type="verses" book="bu">
_Te ca khīṇāsavā bhikkhū,_  
_so ca buddho asādiso;_  
_Atulappabhaṃ dassayitvā,_  
_nibbutā ye mahāyasā._  

</Paragraph>

</Division>
<Division number="33" book="bu">

<Paragraph number="408" type="verses" book="bu">
_Tañca ñāṇaṃ atuliyaṃ,_  
_Tāni ca atulāni ratanāni;_  
_Sabbaṃ tamantarahitaṃ,_  
_Nanu rittā sabbasaṅkhārā._  

</Paragraph>

</Division>
