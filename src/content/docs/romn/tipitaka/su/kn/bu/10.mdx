_Sambuddho dvipaduttamo;_  
_Padumo nāma nāmena,_  
_Asamo appaṭipuggalo._

<Division number="2" book="bu" bookNo="21" page="306">

<Paragraph number="503" type="verses" book="bu">
_Tassāpi asamaṃ sīlaṃ,_  
_samādhipi anantako;_  
_Asaṅkheyyaṃ ñāṇavaraṃ,_  
_vimuttipi anūpamā._  

</Paragraph>

</Division>
<Division number="3" book="bu" bookNo="21" page="306">

<Paragraph number="504" type="verses" book="bu">
_Tassāpi atulatejassa,_  
_dhammacakkappavattane;_  
_Abhisamayā tayo āsuṃ,_  
_mahātamapavāhanā._  

</Paragraph>

</Division>
<Division number="4" book="bu" bookNo="21" page="306">

<Paragraph number="505" type="verses" book="bu">
_Paṭhamābhisamaye buddho,_  
_koṭisatamabodhayi;_  
_Dutiyābhisamaye dhīro,_  
_navutikoṭimabodhayi._  

</Paragraph>

</Division>
<Division number="5" book="bu" bookNo="21" page="307">

<Paragraph number="506" type="verses" book="bu">
_Yadā ca padumo buddho,_  
_ovadī sakamatrajaṃ;_  
_Tadā asītikoṭīnaṃ,_  
_tatiyābhisamayo ahu._  

</Paragraph>

</Division>
<Division number="6" book="bu" bookNo="21" page="307">

<Paragraph number="507" type="verses" book="bu">
_Sannipātā tayo āsuṃ,_  
_padumassa mahesino;_  
_Koṭisatasahassānaṃ,_  
_paṭhamo āsi samāgamo._  

</Paragraph>

</Division>
<Division number="7" book="bu" bookNo="21" page="307">

<Paragraph number="508" type="verses" book="bu">
_Kathinatthārasamaye,_  
_uppanne kathinacīvare;_  
_Dhammasenāpatitthāya,_  
_bhikkhū sibbiṃsu cīvaraṃ._  

</Paragraph>

</Division>
<Division number="8" book="bu" bookNo="21" page="307">

<Paragraph number="509" type="verses" book="bu">
_Tadā te vimalā bhikkhū,_  
_chaḷabhiññā mahiddhikā;_  
_Tīṇi satasahassāni,_  
_samiṃsu aparājitā._  

</Paragraph>

</Division>
<Division number="9" book="bu" bookNo="21" page="307">

<Paragraph number="510" type="verses" book="bu">
_Punāparaṃ so narāsabho,_  
_pavane vāsaṃ upāgami;_  
_Tadā samāgamo āsi,_  
_dvinnaṃ satasahassinaṃ._  

</Paragraph>

</Division>
<Division number="10" book="bu" bookNo="21" page="307">

<Paragraph number="511" type="verses" book="bu">
_Ahaṃ tena samayena,_  
_sīho āsiṃ migādhibhū;_  
_Vivekamanubrūhantaṃ,_  
_pavane addasaṃ jinaṃ._  

</Paragraph>

</Division>
<Division number="11" book="bu" bookNo="21" page="307">

<Paragraph number="512" type="verses" book="bu">
_Vanditvā sirasā pāde,_  
_katvāna taṃ padakkhiṇaṃ;_  
_Tikkhattuṃ abhināditvā,_  
_sattāhaṃ jinamupaṭṭhahaṃ._  

</Paragraph>

</Division>
<Division number="12" book="bu" bookNo="21" page="307">

<Paragraph number="513" type="verses" book="bu">
_Sattāhaṃ varasamāpattiyā,_  
_Vuṭṭhahitvā tathāgato;_  
_Manasā cintayitvāna,_  
_Koṭibhikkhū samānayi._  

</Paragraph>

</Division>
<Division number="13" book="bu" bookNo="21" page="307">

<Paragraph number="514" type="verses" book="bu">
_Tadāpi so mahāvīro,_  
_tesaṃ majjhe viyākari;_  
_‘Aparimeyyito kappe,_  
_ayaṃ buddho bhavissati._  

</Paragraph>

</Division>
<Division number="14" book="bu" bookNo="21" page="307">

<Paragraph number="515" type="verses" book="bu">
_Padhānaṃ padahitvāna,_  
_ …pe… _  
_hessāma sammukhā imaṃ’._  

</Paragraph>

</Division>
<Division number="15" book="bu" bookNo="21" page="307">

<Paragraph number="516" type="verses" book="bu">
_Tassāpi vacanaṃ sutvā,_  
_bhiyyo cittaṃ pasādayiṃ;_  
_Uttariṃ vatamadhiṭṭhāsiṃ,_  
_dasapāramipūriyā._  

</Paragraph>

</Division>
<Division number="16" book="bu" bookNo="21" page="307">

<Paragraph number="517" type="verses" book="bu">
_Campakaṃ nāma nagaraṃ,_  
_asamo nāma khattiyo;_  
_Asamā nāma janikā,_  
_padumassa mahesino._  

</Paragraph>

</Division>
<Division number="17" book="bu" bookNo="21" page="307">

<Paragraph number="518" type="verses" book="bu">
_Dasavassasahassāni,_  
_agāraṃ ajjha so vasi;_  
_Nandā-vasu-yasuttarā,_  
_tayo pāsādamuttamā._  

</Paragraph>

</Division>
<Division number="18" book="bu" bookNo="21" page="308">

<Paragraph number="519" type="verses" book="bu">
_Tettiṃsa ca sahassāni,_  
_nāriyo samalaṅkatā;_  
_Uttarā nāma sā nārī,_  
_rammo nāmāsi atrajo._  

</Paragraph>

</Division>
<Division number="19" book="bu" bookNo="21" page="308">

<Paragraph number="520" type="verses" book="bu">
_Nimitte caturo disvā,_  
_rathayānena nikkhami;_  
_Anūnaaṭṭhamāsāni,_  
_padhānaṃ padahī jino._  

</Paragraph>

</Division>
<Division number="20" book="bu" bookNo="21" page="308">

<Paragraph number="521" type="verses" book="bu">
_Brahmunā yācito santo,_  
_padumo lokanāyako;_  
_Vatti cakkaṃ mahāvīro,_  
_dhanañcuyyānamuttame._  

</Paragraph>

</Division>
<Division number="21" book="bu" bookNo="21" page="308">

<Paragraph number="522" type="verses" book="bu">
_Sālo ca upasālo ca,_  
_ahesuṃ aggasāvakā;_  
_Varuṇo nāmupaṭṭhāko,_  
_padumassa mahesino._  

</Paragraph>

</Division>
<Division number="22" book="bu" bookNo="21" page="308">

<Paragraph number="523" type="verses" book="bu">
_Rādhā ceva surādhā ca,_  
_ahesuṃ aggasāvikā;_  
_Bodhi tassa bhagavato,_  
_mahāsoṇoti vuccati._  

</Paragraph>

</Division>
<Division number="23" book="bu" bookNo="21" page="308">

<Paragraph number="524" type="verses" book="bu">
_Bhiyyo ceva asamo ca,_  
_ahesuṃ aggupaṭṭhakā;_  
_Rucī ca nandarāmā ca,_  
_ahesuṃ aggupaṭṭhikā._  

</Paragraph>

</Division>
<Division number="24" book="bu" bookNo="21" page="308">

<Paragraph number="525" type="verses" book="bu">
_Aṭṭhapaṇṇāsaratanaṃ,_  
_accuggato mahāmuni;_  
_Pabhā niddhāvatī tassa,_  
_asamā sabbaso disā._  

</Paragraph>

</Division>
<Division number="25" book="bu" bookNo="21" page="308">

<Paragraph number="526" type="verses" book="bu">
_Candappabhā sūriyappabhā,_  
_ratanaggimaṇippabhā;_  
_Sabbāpi tā hatā honti,_  
_patvā jinapabhuttamaṃ._  

</Paragraph>

</Division>
<Division number="26" book="bu" bookNo="21" page="308">

<Paragraph number="527" type="verses" book="bu">
_Vassasatasahassāni,_  
_āyu vijjati tāvade;_  
_Tāvatā tiṭṭhamāno so,_  
_tāresi janataṃ bahuṃ._  

</Paragraph>

</Division>
<Division number="27" book="bu" bookNo="21" page="308">

<Paragraph number="528" type="verses" book="bu">
_Paripakkamānase satte,_  
_bodhayitvā asesato;_  
_Sesake anusāsitvā,_  
_nibbuto so sasāvako._  

</Paragraph>

</Division>
<Division number="28" book="bu" bookNo="21" page="308">

<Paragraph number="529" type="verses" book="bu">
_Uragova tacaṃ jiṇṇaṃ,_  
_vaddhapattaṃva pādapo;_  
_Jahitvā sabbasaṅkhāre,_  
_nibbuto so yathā sikhī._  

</Paragraph>

</Division>
