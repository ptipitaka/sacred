<Division number="1" book="bu">

<Paragraph number="1061" type="verses" book="bu">
_Aparimeyyito kappe,_  
_caturo āsuṃ vināyakā;_  
_Taṇhaṅkaro medhaṅkaro,_  
_athopi saraṇaṅkaro;_  
_Dīpaṅkaro ca sambuddho,_  
_ekakappamhi te jinā._  

</Paragraph>

</Division>
<Division number="2" book="bu">

<Paragraph number="1062" type="verses" book="bu">
_Dīpaṅkarassa aparena,_  
_koṇḍañño nāma nāyako;_  
_Ekova ekakappamhi,_  
_tāresi janataṃ bahuṃ._  

</Paragraph>

</Division>
<Division number="3" book="bu">

<Paragraph number="1063" type="verses" book="bu">
_Dīpaṅkarassa bhagavato,_  
_koṇḍaññassa ca satthuno;_  
_Etesaṃ antarā kappā,_  
_gaṇanāto asaṅkhiyā._  

</Paragraph>

</Division>
<Division number="4" book="bu">

<Paragraph number="1064" type="verses" book="bu">
_Koṇḍaññassa aparena,_  
_maṅgalo nāma nāyako;_  
_Tesampi antarā kappā,_  
_gaṇanāto asaṅkhiyā._  

</Paragraph>

</Division>
<Division number="5" book="bu">

<Paragraph number="1065" type="verses" book="bu">
_Maṅgalo ca sumano ca,_  
_revato sobhito muni;_  
_Tepi buddhā ekakappe,_  
_cakkhumanto pabhaṅkarā._  

</Paragraph>

</Division>
<Division number="6" book="bu">

<Paragraph number="1066" type="verses" book="bu">
_Sobhitassa aparena,_  
_anomadassī mahāyaso;_  
_Tesampi antarā kappā,_  
_gaṇanāto asaṅkhiyā._  

</Paragraph>

</Division>
<Division number="7" book="bu">

<Paragraph number="1067" type="verses" book="bu">
_Anomadassī padumo,_  
_nārado cāpi nāyako;_  
_Tepi buddhā ekakappe,_  
_tamantakārakā munī._  

</Paragraph>

</Division>
<Division number="8" book="bu">

<Paragraph number="1068" type="verses" book="bu">
_Nāradassa aparena,_  
_Padumuttaro nāma nāyako;_  
_Ekakappamhi uppanno,_  
_Tāresi janataṃ bahuṃ._  

</Paragraph>

</Division>
<Division number="9" book="bu">

<Paragraph number="1069" type="verses" book="bu">
_Nāradassa bhagavato,_  
_padumuttarassa satthuno;_  
_Tesampi antarā kappā,_  
_gaṇanāto asaṅkhiyā._  

</Paragraph>

</Division>
<Division number="10" book="bu">

<Paragraph number="1070" type="verses" book="bu">
_Kappasatasahassamhi,_  
_eko āsi mahāmuni;_  
_Padumuttaro lokavidū,_  
_āhutīnaṃ paṭiggaho._  

</Paragraph>

</Division>
<Division number="11" book="bu">

<Paragraph number="1071" type="verses" book="bu">
_Tiṃsakappasahassamhi,_  
_duve āsuṃ vināyakā;_  
_Sumedho ca sujāto ca,_  
_orato padumuttarā._  

</Paragraph>

</Division>
<Division number="12" book="bu">

<Paragraph number="1072" type="verses" book="bu">
_Aṭṭhārase kappasate,_  
_tayo āsuṃ vināyakā;_  
_Piyadassī atthadassī,_  
_dhammadassī ca nāyakā._  

</Paragraph>

</Division>
<Division number="13" book="bu">

<Paragraph number="1073" type="verses" book="bu">
_Orato ca sujātassa,_  
_sambuddhā dvipaduttamā;_  
_Ekakappamhi te buddhā,_  
_loke appaṭipuggalā._  

</Paragraph>

</Division>
<Division number="14" book="bu">

<Paragraph number="1074" type="verses" book="bu">
_Catunnavutito kappe,_  
_eko āsi mahāmuni;_  
_Siddhattho so lokavidū,_  
_sallakatto anuttaro._  

</Paragraph>

</Division>
<Division number="15" book="bu">

<Paragraph number="1075" type="verses" book="bu">
_Dvenavute ito kappe,_  
_duve āsuṃ vināyakā;_  
_Tisso phusso ca sambuddhā,_  
_asamā appaṭipuggalā._  

</Paragraph>

</Division>
<Division number="16" book="bu">

<Paragraph number="1076" type="verses" book="bu">
_Ekanavutito kappe,_  
_vipassī nāma nāyako;_  
_Sopi buddho kāruṇiko,_  
_satte mocesi bandhanā._  

</Paragraph>

</Division>
<Division number="17" book="bu">

<Paragraph number="1077" type="verses" book="bu">
_Ekatiṃse ito kappe,_  
_duve āsuṃ vināyakā;_  
_Sikhī ca vessabhū ceva,_  
_asamā appaṭipuggalā._  

</Paragraph>

</Division>
<Division number="18" book="bu">

<Paragraph number="1078" type="verses" book="bu">
_Imamhi bhaddake kappe,_  
_tayo āsuṃ vināyakā;_  
_Kakusandho koṇāgamano,_  
_kassapo cāpi nāyako._  

</Paragraph>

</Division>
<Division number="19" book="bu">

<Paragraph number="1079" type="verses" book="bu">
_Ahametarahi sambuddho,_  
_metteyyo cāpi hessati;_  
_Etepime pañca buddhā,_  
_dhīrā lokānukampakā._  

</Paragraph>

</Division>
<Division number="20" book="bu">

<Paragraph number="1080" type="verses" book="bu">
_Etesaṃ dhammarājūnaṃ,_  
_Aññesaṃnekakoṭinaṃ;_  
_Ācikkhitvāna taṃ maggaṃ,_  
_Nibbutā te sasāvakāti. (1052)_  


</Paragraph>
</Division>

<Paragraph number="1081" type="center" book="bu">
Buddhapakiṇṇakakaṇḍaṃ niṭṭhitaṃ.
</Paragraph>
