<Paragraph number="1036" type="verses" book="bu">
_“Ahametarahi sambuddho,_  
_gotamo sakyavaḍḍhano;_  
_Padhānaṃ padahitvāna,_  
_patto sambodhimuttamaṃ._  

</Paragraph>

</Division>
<Division number="2" book="bu" bookNo="21" page="361">

<Paragraph number="1037" type="verses" book="bu">
_Brahmunā yācito santo,_  
_dhammacakkaṃ pavattayiṃ;_  
_Aṭṭhārasannaṃ koṭīnaṃ,_  
_paṭhamābhisamayo ahu._  

</Paragraph>

</Division>
<Division number="3" book="bu" bookNo="21" page="361">

<Paragraph number="1038" type="verses" book="bu">
_Tato parañca desente,_  
_naradevasamāgame;_  
_Gaṇanāya na vattabbo,_  
_dutiyābhisamayo ahu._  

</Paragraph>

</Division>
<Division number="4" book="bu" bookNo="21" page="361">

<Paragraph number="1039" type="verses" book="bu">
_Idhevāhaṃ etarahi,_  
_ovadiṃ mama atrajaṃ;_  
_Gaṇanāya na vattabbo,_  
_tatiyābhisamayo ahu._  

</Paragraph>

</Division>
<Division number="5" book="bu" bookNo="21" page="361">

<Paragraph number="1040" type="verses" book="bu">
_Ekosi sannipāto me,_  
_sāvakānaṃ mahesinaṃ;_  
_Aḍḍhateḷasasatānaṃ,_  
_bhikkhūnāsi samāgamo._  

</Paragraph>

</Division>
<Division number="6" book="bu" bookNo="21" page="362">

<Paragraph number="1041" type="verses" book="bu">
_Virocamāno vimalo,_  
_bhikkhusaṃghassa majjhago;_  
_Dadāmi patthitaṃ sabbaṃ,_  
_maṇīva sabbakāmado._  

</Paragraph>

</Division>
<Division number="7" book="bu" bookNo="21" page="362">

<Paragraph number="1042" type="verses" book="bu">
_Phalamākaṅkhamānānaṃ,_  
_bhavacchandajahesinaṃ;_  
_Catusaccaṃ pakāsemi,_  
_anukampāya pāṇinaṃ._  

</Paragraph>

</Division>
<Division number="8" book="bu" bookNo="21" page="362">

<Paragraph number="1043" type="verses" book="bu">
_Dasavīsasahassānaṃ,_  
_dhammābhisamayo ahu;_  
_Ekadvinnaṃ abhisamayo,_  
_gaṇanāto asaṅkhiyo._  

</Paragraph>

</Division>
<Division number="9" book="bu" bookNo="21" page="362">

<Paragraph number="1044" type="verses" book="bu">
_Vitthārikaṃ bāhujaññaṃ,_  
_iddhaṃ phītaṃ suphullitaṃ;_  
_Idha mayhaṃ sakyamunino,_  
_sāsanaṃ suvisodhitaṃ._  

</Paragraph>

</Division>
<Division number="10" book="bu" bookNo="21" page="362">

<Paragraph number="1045" type="verses" book="bu">
_Anāsavā vītarāgā,_  
_santacittā samāhitā;_  
_Bhikkhūnekasatā sabbe,_  
_parivārenti maṃ sadā._  

</Paragraph>

</Division>
<Division number="11" book="bu" bookNo="21" page="362">

<Paragraph number="1046" type="verses" book="bu">
_Idāni ye etarahi,_  
_jahanti mānusaṃ bhavaṃ;_  
_Appattamānasā sekhā,_  
_te bhikkhū viññugarahitā._  

</Paragraph>

</Division>
<Division number="12" book="bu" bookNo="21" page="362">

<Paragraph number="1047" type="verses" book="bu">
_Ariyañca santhomayantā,_  
_sadā dhammaratā janā;_  
_Bujjhissanti satimanto,_  
_saṃsārasaritaṃ gatā._  

</Paragraph>

</Division>
<Division number="13" book="bu" bookNo="21" page="362">

<Paragraph number="1048" type="verses" book="bu">
_Nagaraṃ kapilavatthu me,_  
_rājā suddhodano pitā;_  
_Mayhaṃ janettikā mātā,_  
_māyādevīti vuccati._  

</Paragraph>

</Division>
<Division number="14" book="bu" bookNo="21" page="362">

<Paragraph number="1049" type="verses" book="bu">
_Ekūnatiṃsavassāni,_  
_agāraṃ ajjhahaṃ vasiṃ;_  
_Rammo surammo subhako,_  
_tayo pāsādamuttamā._  

</Paragraph>

</Division>
<Division number="15" book="bu" bookNo="21" page="362">

<Paragraph number="1050" type="verses" book="bu">
_Cattārīsasahassāni,_  
_nāriyo samalaṅkatā;_  
_Bhaddakañcanā nāma nārī,_  
_rāhulo nāma atrajo._  

</Paragraph>

</Division>
<Division number="16" book="bu" bookNo="21" page="362">

<Paragraph number="1051" type="verses" book="bu">
_Nimitte caturo disvā,_  
_assayānena nikkhamiṃ;_  
_Chabbassaṃ padhānacāraṃ,_  
_acariṃ dukkaraṃ ahaṃ._  

</Paragraph>

</Division>
<Division number="17" book="bu" bookNo="21" page="362">

<Paragraph number="1052" type="verses" book="bu">
_Bārāṇasiyaṃ isipatane,_  
_Cakkaṃ pavattitaṃ mayā;_  
_Ahaṃ gotamasambuddho,_  
_Saraṇaṃ sabbapāṇinaṃ._  

</Paragraph>

</Division>
<Division number="18" book="bu" bookNo="21" page="362">

<Paragraph number="1053" type="verses" book="bu">
_Kolito upatisso ca,_  
_dve bhikkhū aggasāvakā;_  
_Ānando nāmupaṭṭhāko,_  
_santikāvacaro mama;_  
_Khemā uppalavaṇṇā ca,_  
_bhikkhunī aggasāvikā._  

</Paragraph>

</Division>
<Division number="19" book="bu" bookNo="21" page="362">

<Paragraph number="1054" type="verses" book="bu">
_Citto hatthāḷavako ca,_  
_aggupaṭṭhākupāsakā;_  
_Nandamātā ca uttarā,_  
_aggupaṭṭhākupāsikā._  

</Paragraph>

</Division>
<Division number="20" book="bu" bookNo="21" page="363">

<Paragraph number="1055" type="verses" book="bu">
_Ahaṃ assatthamūlamhi,_  
_patto sambodhimuttamaṃ;_  
_Byāmappabhā sadā mayhaṃ,_  
_soḷasahatthamuggatā._  

</Paragraph>

</Division>
<Division number="21" book="bu" bookNo="21" page="363">

<Paragraph number="1056" type="verses" book="bu">
_Appaṃ vassasataṃ āyu,_  
_idānetarahi vijjati;_  
_Tāvatā tiṭṭhamānohaṃ,_  
_tāremi janataṃ bahuṃ._  

</Paragraph>

</Division>
<Division number="22" book="bu" bookNo="21" page="363">

<Paragraph number="1057" type="verses" book="bu">
_Ṭhapayitvāna dhammukkaṃ,_  
_pacchimaṃ janabodhanaṃ;_  
_Ahampi nacirasseva,_  
_saddhiṃ sāvakasaṃghato;_  
_Idheva parinibbissaṃ,_  
_aggīvāhārasaṅkhayā._  

</Paragraph>

</Division>
<Division number="23" book="bu" bookNo="21" page="363">

<Paragraph number="1058" type="verses" book="bu">
_Tāni ca atulatejāni,_  
_Imāni ca dasabalāni;_  
_Ayañca guṇadhāraṇo deho,_  
_Dvattiṃsavaralakkhaṇavicitto._  

</Paragraph>

</Division>

_Sataraṃsīva chappabhā;_  
_Sabbaṃ tamantarahissanti,_  
_Nanu rittā sabbasaṅkhārā”ti. (1032)_  


</Paragraph>
