<Paragraph number="667" type="verses" book="bu">
_“Sujātassa aparena,_  
_sayambhū lokanāyako;_  
_Durāsado asamasamo,_  
_piyadassī mahāyaso._  

</Paragraph>

</Division>
<Division number="2" book="bu" bookNo="21" page="332">

<Paragraph number="668" type="verses" book="bu">
_Sopi buddho amitayaso,_  
_ādiccova virocati;_  
_Sabbaṃ tamaṃ nihantvāna,_  
_dhammacakkaṃ pavattayi._  

</Paragraph>

</Division>
<Division number="3" book="bu" bookNo="21" page="332">

<Paragraph number="669" type="verses" book="bu">
_Tassāpi atulatejassa,_  
_Ahesuṃ abhisamayā tayo;_  
_Koṭisatasahassānaṃ,_  
_Paṭhamābhisamayo ahu._  

</Paragraph>

</Division>
<Division number="4" book="bu" bookNo="21" page="332">

<Paragraph number="670" type="verses" book="bu">
_Sudassano devarājā,_  
_micchādiṭṭhimarocayi;_  
_Tassa diṭṭhiṃ vinodento,_  
_satthā dhammamadesayi._  

</Paragraph>

</Division>
<Division number="5" book="bu" bookNo="21" page="332">

<Paragraph number="671" type="verses" book="bu">
_Janasannipāto atulo,_  
_mahāsannipatī tadā;_  
_Navutikoṭisahassānaṃ,_  
_dutiyābhisamayo ahu._  

</Paragraph>

</Division>
<Division number="6" book="bu" bookNo="21" page="332">

<Paragraph number="672" type="verses" book="bu">
_Yadā doṇamukhaṃ hatthiṃ,_  
_vinesi narasārathi;_  
_Asītikoṭisahassānaṃ,_  
_tatiyābhisamayo ahu._  

</Paragraph>

</Division>
<Division number="7" book="bu" bookNo="21" page="332">

<Paragraph number="673" type="verses" book="bu">
_Sannipātā tayo āsuṃ,_  
_tassāpi piyadassino;_  
_Koṭisatasahassānaṃ,_  
_paṭhamo āsi samāgamo._  

</Paragraph>

</Division>
<Division number="8" book="bu" bookNo="21" page="332">

<Paragraph number="674" type="verses" book="bu">
_Tato paraṃ navutikoṭī,_  
_samiṃsu ekato munī;_  
_Tatiye sannipātamhi,_  
_asītikoṭiyo ahū._  

</Paragraph>

</Division>
<Division number="9" book="bu" bookNo="21" page="332">

<Paragraph number="675" type="verses" book="bu">
_Ahaṃ tena samayena,_  
_kassapo nāma brāhmaṇo;_  
_Ajjhāyako mantadharo,_  
_tiṇṇaṃ vedāna pāragū._  

</Paragraph>

</Division>
<Division number="10" book="bu" bookNo="21" page="332">

<Paragraph number="676" type="verses" book="bu">
_Tassa dhammaṃ suṇitvāna,_  
_pasādaṃ janayiṃ ahaṃ;_  
_Koṭisatasahassehi,_  
_saṃghārāmaṃ amāpayiṃ._  

</Paragraph>

</Division>
<Division number="11" book="bu" bookNo="21" page="332">

<Paragraph number="677" type="verses" book="bu">
_Tassa datvāna ārāmaṃ,_  
_haṭṭho saṃviggamānaso;_  
_Saraṇe pañca sīle ca,_  
_daḷhaṃ katvā samādiyiṃ._  

</Paragraph>

</Division>
<Division number="12" book="bu" bookNo="21" page="332">

<Paragraph number="678" type="verses" book="bu">
_Sopi maṃ buddho byākāsi,_  
_saṃghamajjhe nisīdiya;_  
_‘Aṭṭhārase kappasate,_  
_ayaṃ buddho bhavissati._  

</Paragraph>

</Division>
<Division number="13" book="bu" bookNo="21" page="332">

<Paragraph number="679" type="verses" book="bu">
_Padhānaṃ padahitvāna,_  
_ …pe… _  
_hessāma sammukhā imaṃ’._  

</Paragraph>

</Division>
<Division number="14" book="bu" bookNo="21" page="332">

<Paragraph number="680" type="verses" book="bu">
_Tassāpi vacanaṃ sutvā,_  
_bhiyyo cittaṃ pasādayiṃ;_  
_Uttariṃ vatamadhiṭṭhāsiṃ,_  
_dasapāramipūriyā._  

</Paragraph>

</Division>
<Division number="15" book="bu" bookNo="21" page="332">

<Paragraph number="681" type="verses" book="bu">
_Sudhaññaṃ nāma nagaraṃ,_  
_sudatto nāma khattiyo;_  
_Candā nāmāsi janikā,_  
_piyadassissa satthuno._  

</Paragraph>

</Division>
<Division number="16" book="bu" bookNo="21" page="333">

<Paragraph number="682" type="verses" book="bu">
_Navavassasahassāni,_  
_Agāraṃ ajjha so vasi;_  
_Sunimmalavimalagiriguhā,_  
_Tayo pāsādamuttamā._  

</Paragraph>

</Division>
<Division number="17" book="bu" bookNo="21" page="333">

<Paragraph number="683" type="verses" book="bu">
_Tettiṃsasahassāni ca,_  
_Nāriyo samalaṅkatā;_  
_Vimalā nāma nārī ca,_  
_Kañcanāveḷo nāma atrajo._  

</Paragraph>

</Division>
<Division number="18" book="bu" bookNo="21" page="333">

<Paragraph number="684" type="verses" book="bu">
_Nimitte caturo disvā,_  
_rathayānena nikkhami;_  
_Chamāsaṃ padhānacāraṃ,_  
_acarī purisuttamo._  

</Paragraph>

</Division>
<Division number="19" book="bu" bookNo="21" page="333">

<Paragraph number="685" type="verses" book="bu">
_Brahmunā yācito santo,_  
_piyadassī mahāmuni;_  
_Vatti cakkaṃ mahāvīro,_  
_usabhuyyāne manorame._  

</Paragraph>

</Division>
<Division number="20" book="bu" bookNo="21" page="333">

<Paragraph number="686" type="verses" book="bu">
_Pālito sabbadassī ca,_  
_ahesuṃ aggasāvakā;_  
_Sobhito nāmupaṭṭhāko,_  
_piyadassissa satthuno._  

</Paragraph>

</Division>
<Division number="21" book="bu" bookNo="21" page="333">

<Paragraph number="687" type="verses" book="bu">
_Sujātā dhammadinnā ca,_  
_ahesuṃ aggasāvikā;_  
_Bodhi tassa bhagavato,_  
_kakudhoti pavuccati._  

</Paragraph>

</Division>
<Division number="22" book="bu" bookNo="21" page="333">

<Paragraph number="688" type="verses" book="bu">
_Sandhako dhammako ceva,_  
_ahesuṃ aggupaṭṭhakā;_  
_Visākhā dhammadinnā ca,_  
_ahesuṃ aggupaṭṭhikā._  

</Paragraph>

</Division>
<Division number="23" book="bu" bookNo="21" page="333">

<Paragraph number="689" type="verses" book="bu">
_Sopi buddho amitayaso,_  
_dvattiṃsavaralakkhaṇo;_  
_Asītihatthamubbedho,_  
_sālarājāva dissati._  

</Paragraph>

</Division>
<Division number="24" book="bu" bookNo="21" page="333">

<Paragraph number="690" type="verses" book="bu">
_Aggicandasūriyānaṃ,_  
_natthi tādisikā pabhā;_  
_Yathā ahu pabhā tassa,_  
_asamassa mahesino._  

</Paragraph>

</Division>
<Division number="25" book="bu" bookNo="21" page="333">

<Paragraph number="691" type="verses" book="bu">
_Tassāpi devadevassa,_  
_āyu tāvatakaṃ ahu;_  
_Navutivassasahassāni,_  
_loke aṭṭhāsi cakkhumā._  

</Paragraph>

</Division>
<Division number="26" book="bu" bookNo="21" page="333">

<Paragraph number="692" type="verses" book="bu">
_Sopi buddho asamasamo,_  
_Yugānipi tāni atuliyāni;_  
_Sabbaṃ tamantarahitaṃ,_  
_Nanu rittā sabbasaṅkhārā._  

</Paragraph>

</Division>
<Division number="27" book="bu" bookNo="21" page="333">

<Paragraph number="693" type="verses" book="bu">
_Piyadassī munivaro,_  
_Assatthārāmamhi nibbuto;_  
_Tatthevassa jinathūpo,_  
_Tīṇiyojanamuggato”ti. (678)_  


</Paragraph>
</Division>

<Paragraph number="694" type="center" book="bu">
