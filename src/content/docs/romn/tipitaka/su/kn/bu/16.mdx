<Paragraph number="695" type="verses" book="bu">
_“Tattheva maṇḍakappamhi,_  
_atthadassī mahāyaso;_  
_Mahātamaṃ nihantvāna,_  
_patto sambodhimuttamaṃ._  

</Paragraph>

</Division>
<Division number="2" book="bu" bookNo="21" page="334">

<Paragraph number="696" type="verses" book="bu">
_Brahmunā yācito santo,_  
_dhammacakkaṃ pavattayi;_  
_Amatena tappayī lokaṃ,_  
_dasasahassisadevakaṃ._  

</Paragraph>

</Division>
<Division number="3" book="bu" bookNo="21" page="334">

<Paragraph number="697" type="verses" book="bu">
_Tassāpi lokanāthassa,_  
_Ahesuṃ abhisamayā tayo;_  
_Koṭisatasahassānaṃ,_  
_Paṭhamābhisamayo ahu._  

</Paragraph>

</Division>
<Division number="4" book="bu" bookNo="21" page="334">

<Paragraph number="698" type="verses" book="bu">
_Yadā buddho atthadassī,_  
_carate devacārikaṃ;_  
_Koṭisatasahassānaṃ,_  
_dutiyābhisamayo ahu._  

</Paragraph>

</Division>
<Division number="5" book="bu" bookNo="21" page="334">

<Paragraph number="699" type="verses" book="bu">
_Punāparaṃ yadā buddho,_  
_desesi pitusantike;_  
_Koṭisatasahassānaṃ,_  
_tatiyābhisamayo ahu._  

</Paragraph>

</Division>
<Division number="6" book="bu" bookNo="21" page="334">

<Paragraph number="700" type="verses" book="bu">
_Sannipātā tayo āsuṃ,_  
_tassāpi ca mahesino;_  
_Khīṇāsavānaṃ vimalānaṃ,_  
_santacittāna tādinaṃ._  

</Paragraph>

</Division>
<Division number="7" book="bu" bookNo="21" page="334">

<Paragraph number="701" type="verses" book="bu">
_Aṭṭhanavutisahassānaṃ,_  
_paṭhamo āsi samāgamo;_  
_Aṭṭhāsītisahassānaṃ,_  
_dutiyo āsi samāgamo._  

</Paragraph>

</Division>
<Division number="8" book="bu" bookNo="21" page="334">

<Paragraph number="702" type="verses" book="bu">
_Aṭṭhasattatisatasahassānaṃ,_  
_Tatiyo āsi samāgamo;_  
_Anupādā vimuttānaṃ,_  
_Vimalānaṃ mahesinaṃ._  

</Paragraph>

</Division>
<Division number="9" book="bu" bookNo="21" page="334">

<Paragraph number="703" type="verses" book="bu">
_Ahaṃ tena samayena,_  
_jaṭilo uggatāpano;_  
_Susīmo nāma nāmena,_  
_mahiyā seṭṭhasammato._  

</Paragraph>

</Division>
<Division number="10" book="bu" bookNo="21" page="334">

<Paragraph number="704" type="verses" book="bu">
_Dibbaṃ mandāravaṃ pupphaṃ,_  
_padumaṃ pārichattakaṃ;_  
_Devalokāharitvāna,_  
_sambuddhamabhipūjayiṃ._  

</Paragraph>

</Division>
<Division number="11" book="bu" bookNo="21" page="335">

<Paragraph number="705" type="verses" book="bu">
_Sopi maṃ buddho byākāsi,_  
_atthadassī mahāmuni;_  
_‘Aṭṭhārase kappasate,_  
_ayaṃ buddho bhavissati._  

</Paragraph>

</Division>
<Division number="12" book="bu" bookNo="21" page="335">

<Paragraph number="706" type="verses" book="bu">
_Padhānaṃ padahitvāna,_  
_ …pe… _  
_hessāma sammukhā imaṃ’._  

</Paragraph>

</Division>
<Division number="13" book="bu" bookNo="21" page="335">

<Paragraph number="707" type="verses" book="bu">
_Tassāpi vacanaṃ sutvā,_  
_haṭṭho saṃviggamānaso;_  
_Uttariṃ vatamadhiṭṭhāsiṃ,_  
_dasapāramipūriyā._  

</Paragraph>

</Division>
<Division number="14" book="bu" bookNo="21" page="335">

<Paragraph number="708" type="verses" book="bu">
_Sobhanaṃ nāma nagaraṃ,_  
_sāgaro nāma khattiyo;_  
_Sudassanā nāma janikā,_  
_atthadassissa satthuno._  

</Paragraph>

</Division>
<Division number="15" book="bu" bookNo="21" page="335">

<Paragraph number="709" type="verses" book="bu">
_Dasavassasahassāni,_  
_Agāraṃ ajjha so vasi;_  
_Amaragiri sugiri vāhanā,_  
_Tayo pāsādamuttamā._  

</Paragraph>

</Division>
<Division number="16" book="bu" bookNo="21" page="335">

<Paragraph number="710" type="verses" book="bu">
_Tettiṃsañca sahassāni,_  
_nāriyo samalaṅkatā;_  
_Visākhā nāma nārī ca,_  
_selo nāmāsi atrajo._  

</Paragraph>

</Division>
<Division number="17" book="bu" bookNo="21" page="335">

<Paragraph number="711" type="verses" book="bu">
_Nimitte caturo disvā,_  
_assayānena nikkhami;_  
_Anūnaaṭṭhamāsāni,_  
_padhānaṃ padahī jino._  

</Paragraph>

</Division>
<Division number="18" book="bu" bookNo="21" page="335">

<Paragraph number="712" type="verses" book="bu">
_Brahmunā yācito santo,_  
_atthadassī mahāyaso;_  
_Vatti cakkaṃ mahāvīro,_  
_anomuyyāne narāsabho._  

</Paragraph>

</Division>
<Division number="19" book="bu" bookNo="21" page="335">

<Paragraph number="713" type="verses" book="bu">
_Santo ca upasanto ca,_  
_ahesuṃ aggasāvakā;_  
_Abhayo nāmupaṭṭhāko,_  
_atthadassissa satthuno._  

</Paragraph>

</Division>
<Division number="20" book="bu" bookNo="21" page="335">

<Paragraph number="714" type="verses" book="bu">
_Dhammā ceva sudhammā ca,_  
_ahesuṃ aggasāvikā;_  
_Bodhi tassa bhagavato,_  
_campakoti pavuccati._  

</Paragraph>

</Division>
<Division number="21" book="bu" bookNo="21" page="335">

<Paragraph number="715" type="verses" book="bu">
_Nakulo ca nisabho ca,_  
_ahesuṃ aggupaṭṭhakā;_  
_Makilā ca sunandā ca,_  
_ahesuṃ aggupaṭṭhikā._  

</Paragraph>

</Division>
<Division number="22" book="bu" bookNo="21" page="335">

<Paragraph number="716" type="verses" book="bu">
_Sopi buddho asamasamo,_  
_asītihatthamuggato;_  
_Sobhate sālarājāva,_  
_uḷurājāva pūrito._  

</Paragraph>

</Division>
<Division number="23" book="bu" bookNo="21" page="335">

<Paragraph number="717" type="verses" book="bu">
_Tassa pākatikā raṃsī,_  
_anekasatakoṭiyo;_  
_Uddhaṃ adho dasa disā,_  
_pharanti yojanaṃ sadā._  

</Paragraph>

</Division>
<Division number="24" book="bu" bookNo="21" page="335">

<Paragraph number="718" type="verses" book="bu">
_Sopi buddho narāsabho,_  
_sabbasattuttamo muni;_  
_Vassasatasahassāni,_  
_loke aṭṭhāsi cakkhumā._  

</Paragraph>

</Division>
<Division number="25" book="bu" bookNo="21" page="336">

<Paragraph number="719" type="verses" book="bu">
_Atulaṃ dassetvā obhāsaṃ,_  
_virocetvā sadevake;_  
_Sopi aniccataṃ patto,_  
_yathaggupādānasaṅkhayā._  

</Paragraph>

</Division>
<Division number="26" book="bu" bookNo="21" page="336">

_Anomārāmamhi nibbuto;_  
_Dhātuvitthārikaṃ āsi,_  
_Tesu tesu padesato”ti. (704)_  


</Paragraph>
</Division>
