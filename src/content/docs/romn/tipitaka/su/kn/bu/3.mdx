<Division number="1" book="bu">

<Paragraph number="272" type="verses" book="bu">
_“Tadā te bhojayitvāna,_  
_sasaṃghaṃ lokanāyakaṃ;_  
_Upagacchuṃ saraṇaṃ tassa,_  
_dīpaṅkarassa satthuno._  

</Paragraph>

</Division>
<Division number="2" book="bu">

<Paragraph number="273" type="verses" book="bu">
_Saraṇāgamane kañci,_  
_nivesesi tathāgato;_  
_Kañci pañcasu sīlesu,_  
_sīle dasavidhe paraṃ._  

</Paragraph>

</Division>
<Division number="3" book="bu">

<Paragraph number="274" type="verses" book="bu">
_Kassaci deti sāmaññaṃ,_  
_caturo phalamuttame;_  
_Kassaci asame dhamme,_  
_deti so paṭisambhidā._  

</Paragraph>

</Division>
<Division number="4" book="bu">

<Paragraph number="275" type="verses" book="bu">
_Kassaci varasamāpattiyo,_  
_Aṭṭha deti narāsabho;_  
_Tisso kassaci vijjāyo,_  
_Chaḷabhiññā pavecchati._  

</Paragraph>

</Division>
<Division number="5" book="bu">

<Paragraph number="276" type="verses" book="bu">
_Tena yogena janakāyaṃ,_  
_ovadati mahāmuni;_  
_Tena vitthārikaṃ āsi,_  
_lokanāthassa sāsanaṃ._  

</Paragraph>

</Division>
<Division number="6" book="bu">

<Paragraph number="277" type="verses" book="bu">
_Mahāhanusabhakkhandho,_  
_dīpaṅkarassa nāmako;_  
_Bahū jane tārayati,_  
_parimoceti duggatiṃ._  

</Paragraph>

</Division>
<Division number="7" book="bu">

<Paragraph number="278" type="verses" book="bu">
_Bodhaneyyaṃ janaṃ disvā,_  
_satasahassepi yojane;_  
_Khaṇena upagantvāna,_  
_bodheti taṃ mahāmuni._  

</Paragraph>

</Division>
<Division number="8" book="bu">

<Paragraph number="279" type="verses" book="bu">
_Paṭhamābhisamaye buddho,_  
_koṭisatamabodhayi;_  
_Dutiyābhisamaye nātho,_  
_navutikoṭimabodhayi._  

</Paragraph>

</Division>
<Division number="9" book="bu">

<Paragraph number="280" type="verses" book="bu">
_Yadā ca devabhavanamhi,_  
_buddho dhammamadesayi;_  
_Navutikoṭisahassānaṃ,_  
_tatiyābhisamayo ahu._  

</Paragraph>

</Division>
<Division number="10" book="bu">

<Paragraph number="281" type="verses" book="bu">
_Sannipātā tayo āsuṃ,_  
_dīpaṅkarassa satthuno;_  
_Koṭisatasahassānaṃ,_  
_paṭhamo āsi samāgamo._  

</Paragraph>

</Division>
<Division number="11" book="bu">

<Paragraph number="282" type="verses" book="bu">
_Puna nāradakūṭamhi,_  
_pavivekagate jine;_  
_Khīṇāsavā vītamalā,_  
_samiṃsu satakoṭiyo._  

</Paragraph>

</Division>
<Division number="12" book="bu">

<Paragraph number="283" type="verses" book="bu">
_Yamhi kāle mahāvīro,_  
_sudassanasiluccaye;_  
_Navakoṭisahassehi,_  
_pavāresi mahāmuni._  

</Paragraph>

</Division>
<Division number="13" book="bu">

<Paragraph number="284" type="verses" book="bu">
_Dasavīsasahassānaṃ,_  
_dhammābhisamayo ahu;_  
_Ekadvinnaṃ abhisamayā,_  
_gaṇanāto asaṅkhiyā._  

</Paragraph>

</Division>
<Division number="14" book="bu">

<Paragraph number="285" type="verses" book="bu">
_Vitthārikaṃ bāhujaññaṃ,_  
_iddhaṃ phītaṃ ahū tadā;_  
_Dīpaṅkarassa bhagavato,_  
_sāsanaṃ suvisodhitaṃ._  

</Paragraph>

</Division>
<Division number="15" book="bu">

<Paragraph number="286" type="verses" book="bu">
_Cattāri satasahassāni,_  
_chaḷabhiññā mahiddhikā;_  
_Dīpaṅkaraṃ lokaviduṃ,_  
_parivārenti sabbadā._  

</Paragraph>

</Division>
<Division number="16" book="bu">

<Paragraph number="287" type="verses" book="bu">
_Ye keci tena samayena,_  
_jahanti mānusaṃ bhavaṃ;_  
_Apattamānasā sekhā,_  
_garahitā bhavanti te._  

</Paragraph>

</Division>
<Division number="17" book="bu">

<Paragraph number="288" type="verses" book="bu">
_Supupphitaṃ pāvacanaṃ,_  
_arahantehi tādihi;_  
_Khīṇāsavehi vimalehi,_  
_upasobhati sabbadā._  

</Paragraph>

</Division>
<Division number="18" book="bu">

<Paragraph number="289" type="verses" book="bu">
_Nagaraṃ rammavatī nāma,_  
_sudevo nāma khattiyo;_  
_Sumedhā nāma janikā,_  
_dīpaṅkarassa satthuno._  

</Paragraph>

</Division>
<Division number="19" book="bu">

<Paragraph number="290" type="verses" book="bu">
_Dasavassasahassāni,_  
_agāraṃ ajjha so vasi;_  
_Haṃsā koñcā mayūrā ca,_  
_tayo pāsādamuttamā._  

</Paragraph>

</Division>
<Division number="20" book="bu">

<Paragraph number="291" type="verses" book="bu">
_Tīṇi satasahassāni,_  
_nāriyo samalaṅkatā;_  
_Padumā nāma sā nārī,_  
_usabhakkhandho atrajo._  

</Paragraph>

</Division>
<Division number="21" book="bu">

<Paragraph number="292" type="verses" book="bu">
_Nimitte caturo disvā,_  
_hatthiyānena nikkhami;_  
_Anūnadasamāsāni,_  
_padhāne padahī jino._  

</Paragraph>

</Division>
<Division number="22" book="bu">

<Paragraph number="293" type="verses" book="bu">
_Padhānacāraṃ caritvāna,_  
_abujjhi mānasaṃ muni;_  
_Brahmunā yācito santo,_  
_dīpaṅkaro mahāmuni._  

</Paragraph>

</Division>
<Division number="23" book="bu">

<Paragraph number="294" type="verses" book="bu">
_Vatti cakkaṃ mahāvīro,_  
_nandārāme sirīghare;_  
_Nisinno sirīsamūlamhi,_  
_akā titthiyamaddanaṃ._  

</Paragraph>

</Division>
<Division number="24" book="bu">

<Paragraph number="295" type="verses" book="bu">
_Sumaṅgalo ca tisso ca,_  
_ahesuṃ aggasāvakā;_  
_Sāgato nāmupaṭṭhāko,_  
_dīpaṅkarassa satthuno._  

</Paragraph>

</Division>
<Division number="25" book="bu">

<Paragraph number="296" type="verses" book="bu">
_Nandā ceva sunandā ca,_  
_ahesuṃ aggasāvikā;_  
_Bodhi tassa bhagavato,_  
_pipphalīti pavuccati._  

</Paragraph>

</Division>
<Division number="26" book="bu">

<Paragraph number="297" type="verses" book="bu">
_Tapussabhallikā nāma,_  
_ahesuṃ aggupaṭṭhakā;_  
_Sirimā koṇā upaṭṭhikā,_  
_dīpaṅkarassa satthuno._  

</Paragraph>

</Division>
<Division number="27" book="bu">

<Paragraph number="298" type="verses" book="bu">
_Asītihatthamubbedho,_  
_dīpaṅkaro mahāmuni;_  
_Sobhati dīparukkhova,_  
_sālarājāva phullito._  

</Paragraph>

</Division>
<Division number="28" book="bu">

<Paragraph number="299" type="verses" book="bu">
_Satasahassavassāni,_  
_āyu tassa mahesino;_  
_Tāvatā tiṭṭhamāno so,_  
_tāresi janataṃ bahuṃ._  

</Paragraph>

</Division>
<Division number="29" book="bu">

<Paragraph number="300" type="verses" book="bu">
_Jotayitvāna saddhammaṃ,_  
_santāretvā mahājanaṃ;_  
_Jalitvā aggikkhandhova,_  
_nibbuto so sasāvako._  

</Paragraph>

</Division>
<Division number="30" book="bu">

<Paragraph number="301" type="verses" book="bu">
_Sā ca iddhi so ca yaso,_  
_Tāni ca pādesu cakkaratanāni;_  
_Sabbaṃ tamantarahitaṃ,_  
_Nanu rittā sabbasaṅkhārā._  

</Paragraph>

</Division>
<Division number="31" book="bu">

<Paragraph number="302" type="verses" book="bu">
_Dīpaṅkaro jino satthā,_  
_Nandārāmamhi nibbuto;_  
_Tatthevassa jinathūpo,_  
_Chattiṃsubbedhayojano”ti. (299)_  


</Paragraph>
</Division>

<Paragraph number="303" type="center" book="bu">
Dīpaṅkarassa bhagavato vaṃso paṭhamo.
</Paragraph>
