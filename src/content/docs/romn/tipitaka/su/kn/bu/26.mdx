<Division number="1" book="bu">

<Paragraph number="984" type="verses" book="bu">
_“Koṇāgamanassa aparena,_  
_Sambuddho dvipaduttamo;_  
_Kassapo nāma gottena,_  
_Dhammarājā pabhaṅkaro._  

</Paragraph>

</Division>
<Division number="2" book="bu">

<Paragraph number="985" type="verses" book="bu">
_Sañchaḍḍitaṃ kulamūlaṃ,_  
_bahvannapānabhojanaṃ;_  
_Datvāna yācake dānaṃ,_  
_pūrayitvāna mānasaṃ;_  
_Usabhova āḷakaṃ bhetvā,_  
_patto sambodhimuttamaṃ._  

</Paragraph>

</Division>
<Division number="3" book="bu">

<Paragraph number="986" type="verses" book="bu">
_Dhammacakkaṃ pavattente,_  
_kassape lokanāyake;_  
_Vīsakoṭisahassānaṃ,_  
_paṭhamābhisamayo ahu._  

</Paragraph>

</Division>
<Division number="4" book="bu">

<Paragraph number="987" type="verses" book="bu">
_Catumāsaṃ yadā buddho,_  
_loke carati cārikaṃ;_  
_Dasakoṭisahassānaṃ,_  
_dutiyābhisamayo ahu._  

</Paragraph>

</Division>
<Division number="5" book="bu">

<Paragraph number="988" type="verses" book="bu">
_Yamakaṃ vikubbanaṃ katvā,_  
_ñāṇadhātuṃ pakittayi;_  
_Pañcakoṭisahassānaṃ,_  
_tatiyābhisamayo ahu._  

</Paragraph>

</Division>
<Division number="6" book="bu">

<Paragraph number="989" type="verses" book="bu">
_Sudhammā devapure ramme,_  
_tattha dhammaṃ pakittayi;_  
_Tīṇikoṭisahassānaṃ,_  
_devānaṃ bodhayī jino._  

</Paragraph>

</Division>
<Division number="7" book="bu">

<Paragraph number="990" type="verses" book="bu">
_Naradevassa yakkhassa,_  
_apare dhammadesane;_  
_Etesānaṃ abhisamayā,_  
_gaṇanāto asaṅkhiyā._  

</Paragraph>

</Division>
<Division number="8" book="bu">

<Paragraph number="991" type="verses" book="bu">
_Tassāpi devadevassa,_  
_eko āsi samāgamo;_  
_Khīṇāsavānaṃ vimalānaṃ,_  
_santacittāna tādinaṃ._  

</Paragraph>

</Division>
<Division number="9" book="bu">

<Paragraph number="992" type="verses" book="bu">
_Vīsabhikkhusahassānaṃ,_  
_tadā āsi samāgamo;_  
_Atikkantabhavantānaṃ,_  
_hirisīlena tādinaṃ._  

</Paragraph>

</Division>
<Division number="10" book="bu">

<Paragraph number="993" type="verses" book="bu">
_Ahaṃ tadā māṇavako,_  
_jotipāloti vissuto;_  
_Ajjhāyako mantadharo,_  
_tiṇṇaṃ vedāna pāragū._  

</Paragraph>

</Division>
<Division number="11" book="bu">

<Paragraph number="994" type="verses" book="bu">
_Lakkhaṇe itihāse ca,_  
_sadhamme pāramiṃ gato;_  
_Bhūmantalikkhakusalo,_  
_katavijjo anāvayo._  

</Paragraph>

</Division>
<Division number="12" book="bu">

<Paragraph number="995" type="verses" book="bu">
_Kassapassa bhagavato,_  
_ghaṭīkāro nāmupaṭṭhāko;_  
_Sagāravo sappatisso,_  
_nibbuto tatiye phale._  

</Paragraph>

</Division>
<Division number="13" book="bu">

<Paragraph number="996" type="verses" book="bu">
_Ādāya maṃ ghaṭīkāro,_  
_upagañchi kassapaṃ jinaṃ;_  
_Tassa dhammaṃ suṇitvāna,_  
_pabbajiṃ tassa santike._  

</Paragraph>

</Division>
<Division number="14" book="bu">

<Paragraph number="997" type="verses" book="bu">
_Āraddhavīriyo hutvā,_  
_vattāvattesu kovido;_  
_Na kvaci parihāyāmi,_  
_pūresiṃ jinasāsanaṃ._  

</Paragraph>

</Division>
<Division number="15" book="bu">

<Paragraph number="998" type="verses" book="bu">
_Yāvatā buddhabhaṇitaṃ,_  
_navaṅgaṃ jinasāsanaṃ;_  
_Sabbaṃ pariyāpuṇitvāna,_  
_sobhayiṃ jinasāsanaṃ._  

</Paragraph>

</Division>
<Division number="16" book="bu">

<Paragraph number="999" type="verses" book="bu">
_Mama acchariyaṃ disvā,_  
_sopi buddho viyākari;_  
_‘Imamhi bhaddake kappe,_  
_ayaṃ buddho bhavissati._  

</Paragraph>

</Division>
<Division number="17" book="bu">

<Paragraph number="1000" type="verses" book="bu">
_Ahu kapilavhayā rammā,_  
_nikkhamitvā tathāgato;_  
_Padhānaṃ padahitvāna,_  
_katvā dukkarakārikaṃ._  

</Paragraph>

</Division>
<Division number="18" book="bu">

<Paragraph number="1001" type="verses" book="bu">
_Ajapālarukkhamūle,_  
_nisīditvā tathāgato;_  
_Tattha pāyāsaṃ paggayha,_  
_nerañjaramupehiti._  

</Paragraph>

</Division>
<Division number="19" book="bu">

<Paragraph number="1002" type="verses" book="bu">
_Nerañjarāya tīramhi,_  
_pāyāsaṃ paribhuñjiya;_  
_Paṭiyattavaramaggena,_  
_bodhimūlamupehiti._  

</Paragraph>

</Division>
<Division number="20" book="bu">

<Paragraph number="1003" type="verses" book="bu">
_Tato padakkhiṇaṃ katvā,_  
_bodhimaṇḍaṃ anuttaro;_  
_Aparājitaṭṭhānamhi,_  
_bodhipallaṅkamuttame;_  
_Pallaṅkena nisīditvā,_  
_bujjhissati mahāyaso._  

</Paragraph>

</Division>
<Division number="21" book="bu">

<Paragraph number="1004" type="verses" book="bu">
_Imassa janikā mātā,_  
_māyā nāma bhavissati;_  
_Pitā suddhodano nāma,_  
_ayaṃ hessati gotamo._  

</Paragraph>

</Division>
<Division number="22" book="bu">

<Paragraph number="1005" type="verses" book="bu">
_Anāsavā vītarāgā,_  
_santacittā samāhitā;_  
_Kolito upatisso ca,_  
_aggā hessanti sāvakā;_  
_Ānando nāmupaṭṭhāko,_  
_upaṭṭhissatimaṃ jinaṃ._  

</Paragraph>

</Division>
<Division number="23" book="bu">

<Paragraph number="1006" type="verses" book="bu">
_Khemā uppalavaṇṇā ca,_  
_aggā hessanti sāvikā;_  
_Anāsavā santacittā,_  
_vītarāgā samāhitā;_  
_Bodhi tassa bhagavato,_  
_assatthoti pavuccati._  

</Paragraph>

</Division>
<Division number="24" book="bu">

<Paragraph number="1007" type="verses" book="bu">
_Citto hatthāḷavako ca,_  
_aggā hessantupaṭṭhakā;_  
_Nandamātā ca uttarā,_  
_aggā hessantupaṭṭhikā’._  

</Paragraph>

</Division>
<Division number="25" book="bu">

<Paragraph number="1008" type="verses" book="bu">
_Idaṃ sutvāna vacanaṃ,_  
_assamassa mahesino;_  
_Āmoditā naramarū,_  
_buddhabījaṃ kira ayaṃ._  

</Paragraph>

</Division>
<Division number="26" book="bu">

<Paragraph number="1009" type="verses" book="bu">
_Ukkuṭṭhisaddā pavattanti,_  
_apphoṭenti hasanti ca;_  
_Katañjalī namassanti,_  
_dasasahassī sadevakā._  

</Paragraph>

</Division>
<Division number="27" book="bu">

<Paragraph number="1010" type="verses" book="bu">
_‘Yadimassa lokanāthassa,_  
_virajjhissāma sāsanaṃ;_  
_Anāgatamhi addhāne,_  
_hessāma sammukhā imaṃ._  

</Paragraph>

</Division>
<Division number="28" book="bu">

<Paragraph number="1011" type="verses" book="bu">
_Yathā manussā nadiṃ tarantā,_  
_Paṭititthaṃ virajjhiya;_  
_Heṭṭhā titthe gahetvāna,_  
_Uttaranti mahānadiṃ._  

</Paragraph>

</Division>
<Division number="29" book="bu">

<Paragraph number="1012" type="verses" book="bu">
_Evamevaṃ mayaṃ sabbe,_  
_yadi muñcāmimaṃ jinaṃ;_  
_Anāgatamhi addhāne,_  
_hessāma sammukhā imaṃ’._  

</Paragraph>

</Division>
<Division number="30" book="bu">

<Paragraph number="1013" type="verses" book="bu">
_Tassāpi vacanaṃ sutvā,_  
_bhiyyo cittaṃ pasādayiṃ;_  
_Uttariṃ vatamadhiṭṭhāsiṃ,_  
_dasapāramipūriyā._  

</Paragraph>

</Division>
<Division number="31" book="bu">

<Paragraph number="1014" type="verses" book="bu">
_Evamahaṃ saṃsaritvā,_  
_parivajjento anācaraṃ;_  
_Dukkarañca kataṃ mayhaṃ,_  
_bodhiyāyeva kāraṇā._  

</Paragraph>

</Division>
<Division number="32" book="bu">

<Paragraph number="1015" type="verses" book="bu">
_Nagaraṃ bārāṇasī nāma,_  
_kikī nāmāsi khattiyo;_  
_Vasate tattha nagare,_  
_sambuddhassa mahākulaṃ._  

</Paragraph>

</Division>
<Division number="33" book="bu">

<Paragraph number="1016" type="verses" book="bu">
_Brāhmaṇo brahmadattova,_  
_āsi buddhassa so pitā;_  
_Dhanavatī nāma janikā,_  
_kassapassa mahesino._  

</Paragraph>

</Division>
<Division number="34" book="bu">

<Paragraph number="1017" type="verses" book="bu">
_Duve vassasahassāni,_  
_agāraṃ ajjha so vasi;_  
_Haṃso yaso sirinando,_  
_tayo pāsādamuttamā._  

</Paragraph>

</Division>
<Division number="35" book="bu">

<Paragraph number="1018" type="verses" book="bu">
_Tisoḷasasahassāni,_  
_Nāriyo samalaṅkatā;_  
_Sunandā nāma sā nārī,_  
_Vijitaseno nāma atrajo._  

</Paragraph>

</Division>
<Division number="36" book="bu">

<Paragraph number="1019" type="verses" book="bu">
_Nimitte caturo disvā,_  
_pāsādenābhinikkhami;_  
_Sattāhaṃ padhānacāraṃ,_  
_acarī purisuttamo._  

</Paragraph>

</Division>
<Division number="37" book="bu">

<Paragraph number="1020" type="verses" book="bu">
_Brahmunā yācito santo,_  
_kassapo lokanāyako;_  
_Vatti cakkaṃ mahāvīro,_  
_migadāye naruttamo._  

</Paragraph>

</Division>
<Division number="38" book="bu">

<Paragraph number="1021" type="verses" book="bu">
_Tisso ca bhāradvājo ca,_  
_ahesuṃ aggasāvakā;_  
_Sabbamitto nāmupaṭṭhāko,_  
_kassapassa mahesino._  

</Paragraph>

</Division>
<Division number="39" book="bu">

<Paragraph number="1022" type="verses" book="bu">
_Anuḷā uruvelā ca,_  
_ahesuṃ aggasāvikā;_  
_Bodhi tassa bhagavato,_  
_nigrodhoti pavuccati._  

</Paragraph>

</Division>
<Division number="40" book="bu">

<Paragraph number="1023" type="verses" book="bu">
_Sumaṅgalo ghaṭikāro ca,_  
_ahesuṃ aggupaṭṭhakā;_  
_Vicitasenā bhaddā ca,_  
_ahesuṃ aggupaṭṭhikā._  

</Paragraph>

</Division>
<Division number="41" book="bu">

<Paragraph number="1024" type="verses" book="bu">
_Uccattanena so buddho,_  
_vīsatiratanuggato;_  
_Vijjulaṭṭhīva ākāse,_  
_candova gahapūrito._  

</Paragraph>

</Division>
<Division number="42" book="bu">

<Paragraph number="1025" type="verses" book="bu">
_Vīsativassasahassāni,_  
_āyu tassa mahesino;_  
_Tāvatā tiṭṭhamāno so,_  
_tāresi janataṃ bahuṃ._  

</Paragraph>

</Division>
<Division number="43" book="bu">

<Paragraph number="1026" type="verses" book="bu">
_Dhammataḷākaṃ māpayitvā,_  
_sīlaṃ datvā vilepanaṃ;_  
_Dhammadussaṃ nivāsetvā,_  
_dhammamālaṃ vibhajjiya._  

</Paragraph>

</Division>
<Division number="44" book="bu">

<Paragraph number="1027" type="verses" book="bu">
_Dhammavimalamādāsaṃ,_  
_ṭhapayitvā mahājane;_  
_Keci nibbānaṃ patthentā,_  
_passantu me alaṅkaraṃ._  

</Paragraph>

</Division>
<Division number="45" book="bu">

<Paragraph number="1028" type="verses" book="bu">
_Sīlakañcukaṃ datvāna,_  
_jhānakavacavammitaṃ;_  
_Dhammacammaṃ pārupitvā,_  
_datvā sannāhamuttamaṃ._  

</Paragraph>

</Division>
<Division number="46" book="bu">

<Paragraph number="1029" type="verses" book="bu">
_Satiphalakaṃ datvāna,_  
_tikhiṇañāṇakuntimaṃ;_  
_Dhammakhaggavaraṃ datvā,_  
_sīlasaṃsaggamaddanaṃ._  

</Paragraph>

</Division>
<Division number="47" book="bu">

<Paragraph number="1030" type="verses" book="bu">
_Tevijjābhūsanaṃ datvāna,_  
_āveḷaṃ caturo phale;_  
_Chaḷabhiññābharaṇaṃ datvā,_  
_dhammapupphapiḷandhanaṃ._  

</Paragraph>

</Division>
<Division number="48" book="bu">

<Paragraph number="1031" type="verses" book="bu">
_Saddhammapaṇḍaracchattaṃ,_  
_datvā pāpanivāraṇaṃ;_  
_Māpayitvābhayaṃ pupphaṃ,_  
_nibbuto so sasāvako._  

</Paragraph>

</Division>
<Division number="49" book="bu">

<Paragraph number="1032" type="verses" book="bu">
_Eso hi sammāsambuddho,_  
_appameyyo durāsado;_  
_Eso hi dhammaratano,_  
_svākkhāto ehipassiko._  

</Paragraph>

</Division>
<Division number="50" book="bu">

<Paragraph number="1033" type="verses" book="bu">
_Eso hi saṃgharatano,_  
_suppaṭipanno anuttaro;_  
_Sabbaṃ tamantarahitaṃ,_  
_nanu rittā sabbasaṅkhārā._  

</Paragraph>

</Division>
<Division number="51" book="bu">

_Setabyārāmamhi nibbuto;_  
_Tatthevassa jinathūpo,_  
_Yojanubbedhamuggato”ti. (1008)_  


</Paragraph>
</Division>
