<Division number="1" book="bu">

<Paragraph number="343" type="verses" book="bu">
_“Koṇḍaññassa aparena,_  
_maṅgalo nāma nāyako;_  
_Tamaṃ loke nihantvāna,_  
_dhammokkamabhidhārayi._  

</Paragraph>

</Division>
<Division number="2" book="bu">

<Paragraph number="344" type="verses" book="bu">
_Atulāsi pabhā tassa,_  
_jinehaññehi uttariṃ;_  
_Candasūriyapabhaṃ hantvā,_  
_dasasahassī virocati._  

</Paragraph>

</Division>
<Division number="3" book="bu">

<Paragraph number="345" type="verses" book="bu">
_Sopi buddho pakāsesi,_  
_caturo saccavaruttame;_  
_Te te saccarasaṃ pītvā,_  
_vinodenti mahātamaṃ._  

</Paragraph>

</Division>
<Division number="4" book="bu">

<Paragraph number="346" type="verses" book="bu">
_Patvāna bodhimatulaṃ,_  
_paṭhame dhammadesane;_  
_Koṭisatasahassānaṃ,_  
_dhammābhisamayo ahu._  

</Paragraph>

</Division>
<Division number="5" book="bu">

<Paragraph number="347" type="verses" book="bu">
_Surindadevabhavane,_  
_buddho dhammamadesayi;_  
_Tadā koṭisahassānaṃ,_  
_dutiyo samayo ahu._  

</Paragraph>

</Division>
<Division number="6" book="bu">

<Paragraph number="348" type="verses" book="bu">
_Yadā sunando cakkavattī,_  
_sambuddhaṃ upasaṅkami;_  
_Tadā āhani sambuddho,_  
_dhammabheriṃ varuttamaṃ._  

</Paragraph>

</Division>
<Division number="7" book="bu">

<Paragraph number="349" type="verses" book="bu">
_Sunandassānucarā janatā,_  
_Tadāsuṃ navutikoṭiyo;_  
_Sabbepi te niravasesā,_  
_Ahesuṃ ehi bhikkhukā._  

</Paragraph>

</Division>
<Division number="8" book="bu">

<Paragraph number="350" type="verses" book="bu">
_Sannipātā tayo āsuṃ,_  
_maṅgalassa mahesino;_  
_Koṭisatasahassānaṃ,_  
_paṭhamo āsi samāgamo._  

</Paragraph>

</Division>
<Division number="9" book="bu">

<Paragraph number="351" type="verses" book="bu">
_Dutiyo koṭisatasahassānaṃ,_  
_Tatiyo navutikoṭinaṃ;_  
_Khīṇāsavānaṃ vimalānaṃ,_  
_Tadā āsi samāgamo._  

</Paragraph>

</Division>
<Division number="10" book="bu">

<Paragraph number="352" type="verses" book="bu">
_Ahaṃ tena samayena,_  
_surucī nāma brāhmaṇo;_  
_Ajjhāyako mantadharo,_  
_tiṇṇaṃ vedāna pāragū._  

</Paragraph>

</Division>
<Division number="11" book="bu">

<Paragraph number="353" type="verses" book="bu">
_Tamahaṃ upasaṅkamma,_  
_Saraṇaṃ gantvāna satthuno;_  
_Sambuddhappamukhaṃ saṃghaṃ,_  
_Gandhamālena pūjayiṃ;_  
_Pūjetvā gandhamālena,_  
_Gavapānena tappayiṃ._  

</Paragraph>

</Division>
<Division number="12" book="bu">

<Paragraph number="354" type="verses" book="bu">
_Sopi maṃ buddho byākāsi,_  
_maṅgalo dvipaduttamo;_  
_‘Aparimeyyito kappe,_  
_ayaṃ buddho bhavissati._  

</Paragraph>

</Division>
<Division number="13" book="bu">

<Paragraph number="355" type="verses" book="bu">
_Padhānaṃ padahitvāna,_  
_ …pe… _  
_hessāma sammukhā imaṃ’._  

</Paragraph>

</Division>
<Division number="14" book="bu">

<Paragraph number="356" type="verses" book="bu">
_Tassāpi vacanaṃ sutvā,_  
_bhiyyo cittaṃ pasādayiṃ;_  
_Uttariṃ vatamadhiṭṭhāsiṃ,_  
_dasapāramipūriyā._  

</Paragraph>

</Division>
<Division number="15" book="bu">

<Paragraph number="357" type="verses" book="bu">
_Tadā pītimanubrūhanto,_  
_sambodhivarapattiyā;_  
_Buddhe datvāna maṃ gehaṃ,_  
_pabbajiṃ tassa santike._  

</Paragraph>

</Division>
<Division number="16" book="bu">

<Paragraph number="358" type="verses" book="bu">
_Suttantaṃ vinayañcāpi,_  
_navaṅgaṃ satthusāsanaṃ;_  
_Sabbaṃ pariyāpuṇitvā,_  
_sobhayiṃ jinasāsanaṃ._  

</Paragraph>

</Division>
<Division number="17" book="bu">

<Paragraph number="359" type="verses" book="bu">
_Tatthappamatto viharanto,_  
_brahmaṃ bhāvetva bhāvanaṃ;_  
_Abhiññāpāramiṃ gantvā,_  
_brahmalokamagacchahaṃ._  

</Paragraph>

</Division>
<Division number="18" book="bu">

<Paragraph number="360" type="verses" book="bu">
_Uttaraṃ nāma nagaraṃ,_  
_uttaro nāma khattiyo;_  
_Uttarā nāma janikā,_  
_maṅgalassa mahesino._  

</Paragraph>

</Division>
<Division number="19" book="bu">

<Paragraph number="361" type="verses" book="bu">
_Navavassasahassāni,_  
_agāraṃ ajjha so vasi;_  
_Yasavā sucimā sirīmā,_  
_tayo pāsādamuttamā._  

</Paragraph>

</Division>
<Division number="20" book="bu">

<Paragraph number="362" type="verses" book="bu">
_Samatiṃsasahassāni,_  
_nāriyo samalaṅkatā;_  
_Yasavatī nāma nārī,_  
_sīvalo nāma atrajo._  

</Paragraph>

</Division>
<Division number="21" book="bu">

<Paragraph number="363" type="verses" book="bu">
_Nimitte caturo disvā,_  
_assayānena nikkhami;_  
_Anūnaaṭṭhamāsāni,_  
_padhānaṃ padahī jino._  

</Paragraph>

</Division>
<Division number="22" book="bu">

<Paragraph number="364" type="verses" book="bu">
_Brahmunā yācito santo,_  
_maṅgalo nāma nāyako;_  
_Vatti cakkaṃ mahāvīro,_  
_vane sirīvaruttame._  

</Paragraph>

</Division>
<Division number="23" book="bu">

<Paragraph number="365" type="verses" book="bu">
_Sudevo dhammaseno ca,_  
_ahesuṃ aggasāvakā;_  
_Pālito nāmupaṭṭhāko,_  
_maṅgalassa mahesino._  

</Paragraph>

</Division>
<Division number="24" book="bu">

<Paragraph number="366" type="verses" book="bu">
_Sīvalā ca asokā ca,_  
_ahesuṃ aggasāvikā;_  
_Bodhi tassa bhagavato,_  
_nāgarukkhoti vuccati._  

</Paragraph>

</Division>
<Division number="25" book="bu">

<Paragraph number="367" type="verses" book="bu">
_Nando ceva visākho ca,_  
_ahesuṃ aggupaṭṭhakā;_  
_Anulā ceva sutanā ca,_  
_ahesuṃ aggupaṭṭhikā._  

</Paragraph>

</Division>
<Division number="26" book="bu">

<Paragraph number="368" type="verses" book="bu">
_Aṭṭhāsīti ratanāni,_  
_accuggato mahāmuni;_  
_Tato niddhāvatī raṃsī,_  
_anekasatasahassiyo._  

</Paragraph>

</Division>
<Division number="27" book="bu">

<Paragraph number="369" type="verses" book="bu">
_Navutivassasahassāni,_  
_āyu vijjati tāvade;_  
_Tāvatā tiṭṭhamāno so,_  
_tāresi janataṃ bahuṃ._  

</Paragraph>

</Division>
<Division number="28" book="bu">

<Paragraph number="370" type="verses" book="bu">
_Yathāpi sāgare ūmī,_  
_na sakkā tā gaṇetuye;_  
_Tatheva sāvakā tassa,_  
_na sakkā te gaṇetuye._  

</Paragraph>

</Division>
<Division number="29" book="bu">

<Paragraph number="371" type="verses" book="bu">
_Yāva aṭṭhāsi sambuddho,_  
_maṅgalo lokanāyako;_  
_Na tassa sāsane atthi,_  
_sakilesamaraṇaṃ tadā._  

</Paragraph>

</Division>
<Division number="30" book="bu">

<Paragraph number="372" type="verses" book="bu">
_Dhammokkaṃ dhārayitvāna,_  
_santāretvā mahājanaṃ;_  
_Jalitvā dhūmaketūva,_  
_nibbuto so mahāyaso._  

</Paragraph>

</Division>
<Division number="31" book="bu">

<Paragraph number="373" type="verses" book="bu">
_Saṅkhārānaṃ sabhāvatthaṃ,_  
_dassayitvā sadevake;_  
_Jalitvā aggikkhandhova,_  
_sūriyo atthaṅgato yathā._  

</Paragraph>

</Division>
<Division number="32" book="bu">

<Paragraph number="374" type="verses" book="bu">
_Uyyāne vassare nāma,_  
_Buddho nibbāyi maṅgalo;_  
_Tatthevassa jinathūpo,_  
_Tiṃsayojanamuggato”ti. (369)_  


</Paragraph>
</Division>

<Paragraph number="375" type="center" book="bu">
Maṅgalassa bhagavato vaṃso tatiyo.
