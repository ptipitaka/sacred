<Paragraph number="828" type="verses" book="bu">
_“Phussassa ca aparena,_  
_sambuddho dvipaduttamo;_  
_Vipassī nāma nāmena,_  
_loke uppajji cakkhumā._  

</Paragraph>

</Division>
<Division number="2" book="bu" bookNo="21" page="348">

<Paragraph number="829" type="verses" book="bu">
_Avijjaṃ sabbaṃ padāletvā,_  
_patto sambodhimuttamaṃ;_  
_Dhammacakkaṃ pavattetuṃ,_  
_pakkāmi bandhumatīpuraṃ._  

</Paragraph>

</Division>
<Division number="3" book="bu" bookNo="21" page="348">

<Paragraph number="830" type="verses" book="bu">
_Dhammacakkaṃ pavattetvā,_  
_ubho bodhesi nāyako;_  
_Gaṇanāya na vattabbo,_  
_paṭhamābhisamayo ahu._  

</Paragraph>

</Division>
<Division number="4" book="bu" bookNo="21" page="348">

<Paragraph number="831" type="verses" book="bu">
_Punāparaṃ amitayaso,_  
_tattha saccaṃ pakāsayi;_  
_Caturāsītisahassānaṃ,_  
_dutiyābhisamayo ahu._  

</Paragraph>

</Division>
<Division number="5" book="bu" bookNo="21" page="348">

<Paragraph number="832" type="verses" book="bu">
_Caturāsītisahassāni,_  
_sambuddhaṃ anupabbajuṃ;_  
_Tesamārāmapattānaṃ,_  
_dhammaṃ desesi cakkhumā._  

</Paragraph>

</Division>
<Division number="6" book="bu" bookNo="21" page="348">

<Paragraph number="833" type="verses" book="bu">
_Sabbākārena bhāsato,_  
_sutvā upanisādino;_  
_Tepi dhammavaraṃ gantvā,_  
_tatiyābhisamayo ahu._  

</Paragraph>

</Division>
<Division number="7" book="bu" bookNo="21" page="348">

<Paragraph number="834" type="verses" book="bu">
_Sannipātā tayo āsuṃ,_  
_vipassissa mahesino;_  
_Khīṇāsavānaṃ vimalānaṃ,_  
_santacittāna tādinaṃ._  

</Paragraph>

</Division>
<Division number="8" book="bu" bookNo="21" page="349">

<Paragraph number="835" type="verses" book="bu">
_Aṭṭhasaṭṭhisatasahassānaṃ,_  
_Paṭhamo āsi samāgamo;_  
_Bhikkhusatasahassānaṃ,_  
_Dutiyo āsi samāgamo._  

</Paragraph>

</Division>
<Division number="9" book="bu" bookNo="21" page="349">

<Paragraph number="836" type="verses" book="bu">
_Asītibhikkhusahassānaṃ,_  
_tatiyo āsi samāgamo;_  
_Tattha bhikkhugaṇamajjhe,_  
_sambuddho atirocati._  

</Paragraph>

</Division>
<Division number="10" book="bu" bookNo="21" page="349">

<Paragraph number="837" type="verses" book="bu">
_Ahaṃ tena samayena,_  
_nāgarājā mahiddhiko;_  
_Atulo nāma nāmena,_  
_puññavanto jutindharo._  

</Paragraph>

</Division>
<Division number="11" book="bu" bookNo="21" page="349">

<Paragraph number="838" type="verses" book="bu">
_Nekānaṃ nāgakoṭīnaṃ,_  
_parivāretvānahaṃ tadā;_  
_Vajjanto dibbaturiyehi,_  
_lokajeṭṭhaṃ upāgamiṃ._  

</Paragraph>

</Division>
<Division number="12" book="bu" bookNo="21" page="349">

<Paragraph number="839" type="verses" book="bu">
_Upasaṅkamitvā sambuddhaṃ,_  
_Vipassiṃ lokanāyakaṃ;_  
_Maṇimuttaratanakhacitaṃ,_  
_Sabbābharaṇavibhūsitaṃ;_  
_Nimantetvā dhammarājassa,_  
_Suvaṇṇapīṭhamadāsahaṃ._  

</Paragraph>

</Division>
<Division number="13" book="bu" bookNo="21" page="349">

<Paragraph number="840" type="verses" book="bu">
_Sopi maṃ buddho byākāsi,_  
_saṃghamajjhe nisīdiya;_  
_‘Ekanavutito kappe,_  
_ayaṃ buddho bhavissati._  

</Paragraph>

</Division>
<Division number="14" book="bu" bookNo="21" page="349">

<Paragraph number="841" type="verses" book="bu">
_Ahu kapilavhayā rammā,_  
_nikkhamitvā tathāgato;_  
_Padhānaṃ padahitvāna,_  
_katvā dukkarakārikaṃ._  

</Paragraph>

</Division>
<Division number="15" book="bu" bookNo="21" page="349">

<Paragraph number="842" type="verses" book="bu">
_Ajapālarukkhamūlasmiṃ,_  
_nisīditvā tathāgato;_  
_Tattha pāyāsaṃ paggayha,_  
_nerañjaramupehiti._  

</Paragraph>

</Division>
<Division number="16" book="bu" bookNo="21" page="349">

<Paragraph number="843" type="verses" book="bu">
_Nerañjarāya tīramhi,_  
_pāyāsaṃ ada so jino;_  
_Paṭiyattavaramaggena,_  
_bodhimūlamupehiti._  

</Paragraph>

</Division>
<Division number="17" book="bu" bookNo="21" page="349">

<Paragraph number="844" type="verses" book="bu">
_Tato padakkhiṇaṃ katvā,_  
_bodhimaṇḍaṃ anuttaro;_  
_Assatthamūle sambodhiṃ,_  
_bujjhissati mahāyaso._  

</Paragraph>

</Division>
<Division number="18" book="bu" bookNo="21" page="349">

<Paragraph number="845" type="verses" book="bu">
_Imassa janikā mātā,_  
_māyā nāma bhavissati;_  
_Pitā suddhodano nāma,_  
_ayaṃ hessati gotamo._  

</Paragraph>

</Division>
<Division number="19" book="bu" bookNo="21" page="349">

<Paragraph number="846" type="verses" book="bu">

_santacittā samāhitā;_  
_Kolito upatisso ca,_  
_aggā hessanti sāvakā;_  
_Ānando nāmupaṭṭhāko,_  
_upaṭṭhissatimaṃ jinaṃ._  

</Paragraph>

</Division>
<Division number="20" book="bu" bookNo="21" page="349">

<Paragraph number="847" type="verses" book="bu">

_aggā hessanti sāvikā;_  
_Anāsavā vītarāgā,_  
_santacittā samāhitā;_  
_Bodhi tassa bhagavato,_  
_assatthoti pavuccati._  

</Paragraph>

</Division>

_aggā hessantupaṭṭhakā;_  
_Nandamātā ca uttarā,_  
_aggā hessantupaṭṭhikā;_  
_Āyu vassasataṃ tassa,_  
_gotamassa yasassino._  

</Paragraph>

</Division>

_bhiyyo cittaṃ pasādayiṃ;_  
_Uttariṃ vatamadhiṭṭhāsiṃ,_  
_dasapāramipūriyā._  

</Paragraph>

_bandhumā nāma khattiyo;_  
_Mātā bandhumatī nāma,_  
_vipassissa mahesino._  

</Paragraph>

_agāraṃ ajjha so vasi;_  
_Nando sunando sirimā,_  
_tayo pāsādamuttamā._  

</Paragraph>

_Nāriyo samalaṅkatā;_  
_Sudassanā nāma sā nārī,_  
_Samavattakkhandho nāma atrajo._  

</Paragraph>

</Division>

_rathayānena nikkhami;_  
_Anūnaaṭṭhamāsāni,_  
_padhānaṃ padahī jino._  

</Paragraph>

</Division>

_vipassī lokanāyako;_  
_Vatti cakkaṃ mahāvīro,_  
_migadāye naruttamo._  

</Paragraph>

_vipassissa mahesino._  

</Paragraph>

_ahesuṃ aggupaṭṭhakā;_

_ahesuṃ aggupaṭṭhikā._  

</Paragraph>

_vipassī lokanāyako;_  
_Pabhā niddhāvati tassa,_  
_samantā sattayojane._  

</Paragraph>

</Division>

_āyu buddhassa tāvade;_  
_Tāvatā tiṭṭhamāno so,_  
_tāresi janataṃ bahuṃ._  

</Paragraph>

</Division>

_avasesaputhujjane._  

</Paragraph>

_desetvā amataṃ padaṃ;_  
_Jalitvā aggikkhandhova,_  
_nibbuto so sasāvako._  

</Paragraph>

_Sabbaṃ tamantarahitaṃ,_  
_nanu rittā sabbasaṅkhārā._  

</Paragraph>
