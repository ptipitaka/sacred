_“Tattheva maṇḍakappamhi,_  
_asamo appaṭipuggalo;_  
_Vessabhū nāma nāmena,_  
_loke uppajji nāyako._  

</Paragraph>

<Division number="2" book="bu" bookNo="21" page="353">

<Paragraph number="896" type="verses" book="bu">
_Ādittaṃ vata rāgaggi,_  
_taṇhānaṃ vijitaṃ tadā;_  
_Nāgova bandhanaṃ chetvā,_  
_patto sambodhimuttamaṃ._  

</Paragraph>

</Division>
<Division number="3" book="bu" bookNo="21" page="353">

<Paragraph number="897" type="verses" book="bu">
_Dhammacakkaṃ pavattente,_  
_vessabhūlokanāyake;_  
_Asītikoṭisahassānaṃ,_  
_paṭhamābhisamayo ahu._  

</Paragraph>

</Division>
<Division number="4" book="bu" bookNo="21" page="353">

<Paragraph number="898" type="verses" book="bu">
_Pakkante cārikaṃ raṭṭhe,_  
_lokajeṭṭhe narāsabhe;_  
_Sattatikoṭisahassānaṃ,_  
_dutiyābhisamayo ahu._  

</Paragraph>

</Division>
<Division number="5" book="bu" bookNo="21" page="353">

<Paragraph number="899" type="verses" book="bu">
_Mahādiṭṭhiṃ vinodento,_  
_pāṭiheraṃ karoti so;_  
_Samāgatā naramarū,_  
_dasasahassī sadevake._  

</Paragraph>

</Division>
<Division number="6" book="bu" bookNo="21" page="353">

<Paragraph number="900" type="verses" book="bu">
_Mahāacchariyaṃ disvā,_  
_abbhutaṃ lomahaṃsanaṃ;_  
_Devā ceva manussā ca,_  
_bujjhare saṭṭhikoṭiyo._  

</Paragraph>

</Division>
<Division number="7" book="bu" bookNo="21" page="353">

<Paragraph number="901" type="verses" book="bu">
_Sannipātā tayo āsuṃ,_  
_vessabhussa mahesino;_  
_Khīṇāsavānaṃ vimalānaṃ,_  
_santacittāna tādinaṃ._  

</Paragraph>

</Division>
<Division number="8" book="bu" bookNo="21" page="354">

<Paragraph number="902" type="verses" book="bu">
_Asītibhikkhusahassānaṃ,_  
_paṭhamo āsi samāgamo;_  
_Sattatibhikkhusahassānaṃ,_  
_dutiyo āsi samāgamo._  

</Paragraph>

</Division>
<Division number="9" book="bu" bookNo="21" page="354">

<Paragraph number="903" type="verses" book="bu">
_Saṭṭhibhikkhusahassānaṃ,_  
_tatiyo āsi samāgamo;_  
_Jarādibhayabhītānaṃ,_  
_orasānaṃ mahesino._  

</Paragraph>

</Division>
<Division number="10" book="bu" bookNo="21" page="354">

<Paragraph number="904" type="verses" book="bu">
_Ahaṃ tena samayena,_  
_sudassano nāma khattiyo;_  
_Nimantetvā mahāvīraṃ,_  
_dānaṃ datvā mahārahaṃ;_  
_Annapānena vatthena,_  
_sasaṃghaṃ jinapūjayiṃ._  

</Paragraph>

</Division>
<Division number="11" book="bu" bookNo="21" page="354">

<Paragraph number="905" type="verses" book="bu">
_Tassa buddhassa asamassa,_  
_cakkaṃ vattitamuttamaṃ;_  
_Sutvāna paṇitaṃ dhammaṃ,_  
_pabbajjamabhirocayiṃ._  

</Paragraph>

</Division>
<Division number="12" book="bu" bookNo="21" page="354">

<Paragraph number="906" type="verses" book="bu">
_Mahādānaṃ pavattetvā,_  
_rattindivamatandito;_  
_Pabbajjaṃ guṇasampannaṃ,_  
_pabbajiṃ jinasantike._  

</Paragraph>

</Division>
<Division number="13" book="bu" bookNo="21" page="354">

<Paragraph number="907" type="verses" book="bu">
_Ācāraguṇasampanno,_  
_vattasīlasamāhito;_  
_Sabbaññutaṃ gavesanto,_  
_ramāmi jinasāsane._  

</Paragraph>

</Division>
<Division number="14" book="bu" bookNo="21" page="354">

<Paragraph number="908" type="verses" book="bu">
_Saddhāpītiṃ upagantvā,_  
_buddhaṃ vandāmi sattharaṃ;_  
_Pīti uppajjati mayhaṃ,_  
_bodhiyāyeva kāraṇā._  

</Paragraph>

</Division>
<Division number="15" book="bu" bookNo="21" page="354">

<Paragraph number="909" type="verses" book="bu">
_Anivattamānasaṃ ñatvā,_  
_sambuddho etadabravi;_  
_‘Ekatiṃse ito kappe,_  
_ayaṃ buddho bhavissati._  

</Paragraph>

</Division>

_ …pe… _

</Paragraph>

<Paragraph number="911" type="verses" book="bu">
_Tassāhaṃ vacanaṃ sutvā,_  
_bhiyyo cittaṃ pasādayiṃ;_  
_Uttariṃ vatamadhiṭṭhāsiṃ,_  
_dasapāramipūriyā._  

</Paragraph>

</Division>
<Division number="18" book="bu" bookNo="21" page="354">

<Paragraph number="912" type="verses" book="bu">
_Anomaṃ nāma nagaraṃ,_  
_suppatīto nāma khattiyo;_  
_Mātā yasavatī nāma,_  
_vessabhussa mahesino._  

</Paragraph>

</Division>
<Division number="19" book="bu" bookNo="21" page="354">

<Paragraph number="913" type="verses" book="bu">
_Cha ca vassasahassāni,_  
_Agāraṃ ajjha so vasi;_  
_Ruci suruci rativaḍḍhano,_  
_Tayo pāsādamuttamā._  

</Paragraph>

</Division>
<Division number="20" book="bu" bookNo="21" page="354">

<Paragraph number="914" type="verses" book="bu">
_Anūnatiṃsasahassāni,_  
_nāriyo samalaṅkatā;_  
_Sucittā nāma sā nārī,_  
_suppabuddho nāma atrajo._  

</Paragraph>

</Division>
<Division number="21" book="bu" bookNo="21" page="354">

<Paragraph number="915" type="verses" book="bu">
_Nimitte caturo disvā,_  
_sivikāyābhinikkhami;_  
_Chamāsaṃ padhānacāraṃ,_  
_acarī purisuttamo._  

</Paragraph>

</Division>
<Division number="22" book="bu" bookNo="21" page="355">

<Paragraph number="916" type="verses" book="bu">
_Brahmunā yācito santo,_  
_vessabhūlokanāyako;_  
_Vatti cakkaṃ mahāvīro,_  
_aruṇārāme naruttamo._  

</Paragraph>

</Division>
<Division number="23" book="bu" bookNo="21" page="355">

<Paragraph number="917" type="verses" book="bu">
_Soṇo ca uttaro ceva,_  
_ahesuṃ aggasāvakā;_  
_Upasanto nāmupaṭṭhāko,_  
_vessabhussa mahesino._  

</Paragraph>

</Division>
<Division number="24" book="bu" bookNo="21" page="355">

<Paragraph number="918" type="verses" book="bu">
_Rāmā ceva samālā ca,_  
_ahesuṃ aggasāvikā;_  
_Bodhi tassa bhagavato,_  
_mahāsāloti vuccati._  

</Paragraph>

</Division>
<Division number="25" book="bu" bookNo="21" page="353">

<Paragraph number="919" type="verses" book="bu">
_Sotthiko ceva rambho ca,_  
_ahesuṃ aggupaṭṭhakā;_  
_Gotamī sirimā ceva,_  
_ahesuṃ aggupaṭṭhikā._  

</Paragraph>

</Division>

_hemayūpasamūpamo;_  
_Kāyā niccharati rasmi,_  
_rattiṃva pabbate sikhī._  

</Paragraph>

</Division>
<Division number="27" book="bu" bookNo="21" page="345">

<Paragraph number="921" type="verses" book="bu">
_Saṭṭhivassasahassāni,_  
_āyu tassa mahesino;_  
_Tāvatā tiṭṭhamāno so,_  
_tāresi janataṃ bahuṃ._  

</Paragraph>

</Division>

_vibhajitvā mahājanaṃ;_  
_Dhammanāvaṃ ṭhapetvāna,_  
_nibbuto so sasāvako._  

</Paragraph>

</Division>

_vihāraṃ iriyāpathaṃ;_  
_Sabbaṃ tamantarahitaṃ,_  
_nanu rittā sabbasaṅkhārā._  

</Paragraph>

</Division>

_Dhātuvitthārikaṃ āsi,_  
_Tesu tesu padesato”ti. (901)_
