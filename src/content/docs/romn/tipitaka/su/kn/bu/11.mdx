<Paragraph number="532" type="verses" book="bu">
_“Padumassa aparena,_  
_sambuddho dvipaduttamo;_  
_Nārado nāma nāmena,_  
_asamo appaṭipuggalo._  

</Paragraph>

</Division>
<Division number="2" book="bu" bookNo="21" page="321">

<Paragraph number="533" type="verses" book="bu">
_So buddho cakkavattissa,_  
_jeṭṭho dayitaoraso;_  
_Āmukkamālābharaṇo,_  
_uyyānaṃ upasaṅkami._  

</Paragraph>

</Division>
<Division number="3" book="bu" bookNo="21" page="321">

<Paragraph number="534" type="verses" book="bu">
_Tatthāsi rukkho yasavipulo,_  
_Abhirūpo brahā suci;_  
_Tamajjhapatvā upanisīdi,_  
_Mahāsoṇassa heṭṭhato._  

</Paragraph>

</Division>
<Division number="4" book="bu" bookNo="21" page="321">

<Paragraph number="535" type="verses" book="bu">
_Tattha ñāṇavaruppajji,_  
_anantaṃ vajirūpamaṃ;_  
_Tena vicini saṅkhāre,_  
_ukkujjamavakujjakaṃ._  

</Paragraph>

</Division>
<Division number="5" book="bu" bookNo="21" page="321">

<Paragraph number="536" type="verses" book="bu">
_Tattha sabbakilesāni,_  
_asesamabhivāhayi;_  
_Pāpuṇī kevalaṃ bodhiṃ,_  
_buddhañāṇe ca cuddasa._  

</Paragraph>

</Division>
<Division number="6" book="bu" bookNo="21" page="321">

<Paragraph number="537" type="verses" book="bu">
_Pāpuṇitvāna sambodhiṃ,_  
_dhammacakkaṃ pavattayi;_  
_Koṭisatasahassānaṃ,_  
_paṭhamābhisamayo ahu._  

</Paragraph>

</Division>
<Division number="7" book="bu" bookNo="21" page="321">

<Paragraph number="538" type="verses" book="bu">
_Mahādoṇaṃ nāgarājaṃ,_  
_vinayanto mahāmuni;_  
_Pāṭiheraṃ tadākāsi,_  
_dassayanto sadevake._  

</Paragraph>

</Division>
<Division number="8" book="bu" bookNo="21" page="321">

<Paragraph number="539" type="verses" book="bu">
_Tadā devamanussānaṃ,_  
_tamhi dhammappakāsane;_  
_Navutikoṭisahassāni,_  
_tariṃsu sabbasaṃsayaṃ._  

</Paragraph>

</Division>
<Division number="9" book="bu" bookNo="21" page="321">

<Paragraph number="540" type="verses" book="bu">
_Yamhi kāle mahāvīro,_  
_ovadī sakamatrajaṃ;_  
_Asītikoṭisahassānaṃ,_  
_tatiyābhisamayo ahu._  

</Paragraph>

</Division>
<Division number="10" book="bu" bookNo="21" page="321">

<Paragraph number="541" type="verses" book="bu">
_Sannipātā tayo āsuṃ,_  
_nāradassa mahesino;_  
_Koṭisatasahassānaṃ,_  
_paṭhamo āsi samāgamo._  

</Paragraph>

</Division>
<Division number="11" book="bu" bookNo="21" page="322">

<Paragraph number="542" type="verses" book="bu">
_Yadā buddho buddhaguṇaṃ,_  
_sanidānaṃ pakāsayi;_  
_Navutikoṭisahassāni,_  
_samiṃsu vimalā tadā._  

</Paragraph>

</Division>
<Division number="12" book="bu" bookNo="21" page="322">

<Paragraph number="543" type="verses" book="bu">
_Yadā verocano nāgo,_  
_dānaṃ dadāti satthuno;_  
_Tadā samiṃsu jinaputtā,_  
_asītisatasahassiyo._  

</Paragraph>

</Division>
<Division number="13" book="bu" bookNo="21" page="322">

<Paragraph number="544" type="verses" book="bu">
_Ahaṃ tena samayena,_  
_jaṭilo uggatāpano;_  
_Antalikkhacaro āsiṃ,_  
_pañcābhiññāsu pāragū._  

</Paragraph>

</Division>
<Division number="14" book="bu" bookNo="21" page="322">

<Paragraph number="545" type="verses" book="bu">
_Tadāpāhaṃ asamasamaṃ,_  
_sasaṃghaṃ saparijjanaṃ;_  
_Annapānena tappetvā,_  
_candanenābhipūjayiṃ._  

</Paragraph>

</Division>
<Division number="15" book="bu" bookNo="21" page="322">

<Paragraph number="546" type="verses" book="bu">
_Sopi maṃ tadā byākāsi,_  
_nārado lokanāyako;_  
_‘Aparimeyyito kappe,_  
_ayaṃ buddho bhavissati._  

</Paragraph>

</Division>
<Division number="16" book="bu" bookNo="21" page="322">

<Paragraph number="547" type="verses" book="bu">
_Padhānaṃ padahitvāna,_  
_ …pe… _  
_hessāma sammukhā imaṃ’._  

</Paragraph>

</Division>
<Division number="17" book="bu" bookNo="21" page="322">

<Paragraph number="548" type="verses" book="bu">
_Tassāpi vacanaṃ sutvā,_  
_bhiyyo hāsetva mānasaṃ;_  
_Adhiṭṭhahiṃ vataṃ uggaṃ,_  
_dasapāramipūriyā._  

</Paragraph>

</Division>
<Division number="18" book="bu" bookNo="21" page="322">

<Paragraph number="549" type="verses" book="bu">
_Nagaraṃ dhaññavatī nāma,_  
_sudevo nāma khattiyo;_  
_Anomā nāma janikā,_  
_nāradassa mahesino._  

</Paragraph>

</Division>
<Division number="19" book="bu" bookNo="21" page="322">

<Paragraph number="550" type="verses" book="bu">
_Navavassasahassāni,_  
_agāraṃ ajjha so vasi;_  
_Jito vijitābhirāmo,_  
_tayo pāsādamuttamā._  

</Paragraph>

</Division>
<Division number="20" book="bu" bookNo="21" page="322">

<Paragraph number="551" type="verses" book="bu">
_Ticattārīsasahassāni,_  
_nāriyo samalaṅkatā;_  
_Vijitasenā nāma nārī,_  
_nanduttaro nāma atrajo._  

</Paragraph>

</Division>
<Division number="21" book="bu" bookNo="21" page="322">

<Paragraph number="552" type="verses" book="bu">
_Nimitte caturo disvā,_  
_Padasā gamanena nikkhami;_  
_Sattāhaṃ padhānacāraṃ,_  
_Acarī purisuttamo._  

</Paragraph>

</Division>
<Division number="22" book="bu" bookNo="21" page="322">

<Paragraph number="553" type="verses" book="bu">
_Brahmunā yācito santo,_  
_nārado lokanāyako;_  
_Vatti cakkaṃ mahāvīro,_  
_dhanañcuyyānamuttame._  

</Paragraph>

</Division>
<Division number="23" book="bu" bookNo="21" page="322">

<Paragraph number="554" type="verses" book="bu">
_Bhaddasālo jitamitto,_  
_ahesuṃ aggasāvakā;_  
_Vāseṭṭho nāmupaṭṭhāko,_  
_nāradassa mahesino._  

</Paragraph>

</Division>
<Division number="24" book="bu" bookNo="21" page="323">

<Paragraph number="555" type="verses" book="bu">
_Uttarā phaggunī ceva,_  
_ahesuṃ aggasāvikā;_  
_Bodhi tassa bhagavato,_  
_mahāsoṇoti vuccati._  

</Paragraph>

</Division>
<Division number="25" book="bu" bookNo="21" page="323">

<Paragraph number="556" type="verses" book="bu">
_Uggarindo vasabho ca,_  
_ahesuṃ aggupaṭṭhakā;_  
_Indāvarī ca vaṇḍī ca,_  
_ahesuṃ aggupaṭṭhikā._  

</Paragraph>

</Division>
<Division number="26" book="bu" bookNo="21" page="323">

<Paragraph number="557" type="verses" book="bu">
_Aṭṭhāsītiratanāni,_  
_accuggato mahāmuni;_  
_Kañcanagghiyasaṅkāso,_  
_dasasahassī virocati._  

</Paragraph>

</Division>
<Division number="27" book="bu" bookNo="21" page="323">

<Paragraph number="558" type="verses" book="bu">
_Tassa byāmappabhā kāyā,_  
_niddhāvati disodisaṃ;_  
_Nirantaraṃ divārattiṃ,_  
_yojanaṃ pharate sadā._  

</Paragraph>

</Division>
<Division number="28" book="bu" bookNo="21" page="323">

<Paragraph number="559" type="verses" book="bu">
_Na keci tena samayena,_  
_samantā yojane janā;_  
_Ukkāpadīpe ujjālenti,_  
_buddharaṃsīhi otthaṭā._  

</Paragraph>

</Division>
<Division number="29" book="bu" bookNo="21" page="323">

<Paragraph number="560" type="verses" book="bu">
_Navutivassasahassāni,_  
_āyu vijjati tāvade;_  
_Tāvatā tiṭṭhamāno so,_  
_tāresi janataṃ bahuṃ._  

</Paragraph>

</Division>
<Division number="30" book="bu" bookNo="21" page="308">

<Paragraph number="561" type="verses" book="bu">
_Yathā uḷūhi gaganaṃ,_  
_vicittaṃ upasobhati;_  
_Tatheva sāsanaṃ tassa,_  
_arahantehi sobhati._  

</Paragraph>

</Division>
<Division number="31" book="bu" bookNo="21" page="309">

<Paragraph number="562" type="verses" book="bu">
_Saṃsārasotaṃ taraṇāya,_  
_sesake paṭipannake;_  
_Dhammasetuṃ daḷhaṃ katvā,_  
_nibbuto so narāsabho._  

</Paragraph>

</Division>
<Division number="32" book="bu" bookNo="21" page="309">

<Paragraph number="563" type="verses" book="bu">
_Sopi buddho asamasamo,_  
_Tepi khīṇāsavā atulatejā;_  
_Sabbaṃ tamantarahitaṃ,_  
_Nanu rittā sabbasaṅkhārā._  

</Paragraph>

</Division>
