_Sumedho nāma nāyako;_

_brahā uju patāpavā;_  
_Hitesī sabbasattānaṃ,_  
_bahū mocesi bandhanā._  

</Paragraph>

<Division number="3" book="bu" bookNo="21" page="326">

<Paragraph number="600" type="verses" book="bu">
_Yadā buddho pāpuṇitvā,_  
_kevalaṃ bodhimuttamaṃ;_  
_Sudassanamhi nagare,_  
_dhammacakkaṃ pavattayi._  

</Paragraph>

</Division>
<Division number="4" book="bu" bookNo="21" page="326">

<Paragraph number="601" type="verses" book="bu">
_Tassāpi abhisamayā tīṇi,_  
_Ahesuṃ dhammadesane;_  
_Koṭisatasahassānaṃ,_  
_Paṭhamābhisamayo ahu._  

</Paragraph>

</Division>
<Division number="5" book="bu" bookNo="21" page="327">

<Paragraph number="602" type="verses" book="bu">
_Punāparaṃ kumbhakaṇṇaṃ,_  
_yakkhaṃ so damayī jino;_  
_Navutikoṭisahassānaṃ,_  
_dutiyābhisamayo ahu._  

</Paragraph>

</Division>
<Division number="6" book="bu" bookNo="21" page="327">

<Paragraph number="603" type="verses" book="bu">
_Punāparaṃ amitayaso,_  
_catusaccaṃ pakāsayi;_  
_Asītikoṭisahassānaṃ,_  
_tatiyābhisamayo ahu._  

</Paragraph>

</Division>
<Division number="7" book="bu" bookNo="21" page="327">

<Paragraph number="604" type="verses" book="bu">
_Sannipātā tayo āsuṃ,_  
_sumedhassa mahesino;_  
_Khīṇāsavānaṃ vimalānaṃ,_  
_santacittāna tādinaṃ._  

</Paragraph>

</Division>
<Division number="8" book="bu" bookNo="21" page="327">

<Paragraph number="605" type="verses" book="bu">
_Sudassanaṃ nāma nagaraṃ,_  
_upagañchi jino yadā;_  
_Tadā khīṇāsavā bhikkhū,_  
_samiṃsu satakoṭiyo._  

</Paragraph>

</Division>
<Division number="9" book="bu" bookNo="21" page="327">

<Paragraph number="606" type="verses" book="bu">
_Punāparaṃ devakūṭe,_  
_bhikkhūnaṃ kathinatthate;_  
_Tadā navutikoṭīnaṃ,_  
_dutiyo āsi samāgamo._  

</Paragraph>

</Division>
<Division number="10" book="bu" bookNo="21" page="327">

<Paragraph number="607" type="verses" book="bu">
_Punāparaṃ dasabalo,_  
_yadā carati cārikaṃ;_  
_Tadā asītikoṭīnaṃ,_  
_tatiyo āsi samāgamo._  

</Paragraph>

</Division>
<Division number="11" book="bu" bookNo="21" page="327">

<Paragraph number="608" type="verses" book="bu">
_Ahaṃ tena samayena,_  
_uttaro nāma māṇavo;_  
_Asītikoṭiyo mayhaṃ,_  
_ghare sannicitaṃ dhanaṃ._  

</Paragraph>

</Division>
<Division number="12" book="bu" bookNo="21" page="327">

<Paragraph number="609" type="verses" book="bu">
_Kevalaṃ sabbaṃ datvāna,_  
_sasaṃghe lokanāyake;_  
_Saraṇaṃ tassupagañchiṃ,_  
_pabbajjañcābhirocayiṃ._  

</Paragraph>

</Division>
<Division number="13" book="bu" bookNo="21" page="327">

<Paragraph number="610" type="verses" book="bu">
_Sopi maṃ buddho byākāsi,_  
_karonto anumodanaṃ;_  
_‘Tiṃsakappasahassamhi,_  
_ayaṃ buddho bhavissati._  

</Paragraph>

</Division>
<Division number="14" book="bu" bookNo="21" page="327">

<Paragraph number="611" type="verses" book="bu">
_Padhānaṃ padahitvāna,_  
_ …pe… _  
_hessāma sammukhā imaṃ’._  

</Paragraph>

</Division>
<Division number="15" book="bu" bookNo="21" page="327">

<Paragraph number="612" type="verses" book="bu">
_Tassāpi vacanaṃ sutvā,_  
_bhiyyo cittaṃ pasādayiṃ;_  
_Uttariṃ vatamadhiṭṭhāsiṃ,_  
_dasapāramipūriyā._  

</Paragraph>

</Division>
<Division number="16" book="bu" bookNo="21" page="327">

<Paragraph number="613" type="verses" book="bu">
_Suttantaṃ vinayañcāpi,_  
_navaṅgaṃ satthusāsanaṃ;_  
_Sabbaṃ pariyāpuṇitvāna,_  
_sobhayiṃ jinasāsanaṃ._  

</Paragraph>

</Division>
<Division number="17" book="bu" bookNo="21" page="327">

<Paragraph number="614" type="verses" book="bu">
_Tatthappamatto viharanto,_  
_Nisajjaṭṭhānacaṅkame;_  
_Abhiññāsu pāramiṃ gantvā,_  
_Brahmalokamagañchahaṃ._  

</Paragraph>

</Division>
<Division number="18" book="bu" bookNo="21" page="328">

<Paragraph number="615" type="verses" book="bu">
_Sudassanaṃ nāma nagaraṃ,_  
_sudatto nāma khattiyo;_  
_Sudattā nāma janikā,_  
_sumedhassa mahesino._  

</Paragraph>

</Division>
<Division number="19" book="bu" bookNo="21" page="328">

<Paragraph number="616" type="verses" book="bu">
_Navavassasahassāni,_  
_Agāraṃ ajjha so vasi;_  
_Sucandakañcanasirivaḍḍhā,_  
_Tayo pāsādamuttamā._  

</Paragraph>

</Division>
<Division number="20" book="bu" bookNo="21" page="328">

<Paragraph number="617" type="verses" book="bu">
_Tisoḷasasahassāni,_  
_nāriyo samalaṅkatā;_  
_Sumanā nāma sā nārī,_  
_punabbasu nāma atrajo._  

</Paragraph>

</Division>
<Division number="21" book="bu" bookNo="21" page="328">

<Paragraph number="618" type="verses" book="bu">
_Nimitte caturo disvā,_  
_hatthiyānena nikkhami;_  
_Anūnakaṃ aḍḍhamāsaṃ,_  
_padhānaṃ padahī jino._  

</Paragraph>

</Division>
<Division number="22" book="bu" bookNo="21" page="328">

<Paragraph number="619" type="verses" book="bu">
_Brahmunā yācito santo,_  
_sumedho lokanāyako;_  
_Vatti cakkaṃ mahāvīro,_  
_sudassanuyyānamuttame._  

</Paragraph>

</Division>
<Division number="23" book="bu" bookNo="21" page="328">

<Paragraph number="620" type="verses" book="bu">
_Saraṇo sabbakāmo ca,_  
_ahesuṃ aggasāvakā;_  
_Sāgaro nāmupaṭṭhāko,_  
_sumedhassa mahesino._  

</Paragraph>

</Division>
<Division number="24" book="bu" bookNo="21" page="328">

<Paragraph number="621" type="verses" book="bu">
_Rāmā ceva surāmā ca,_  
_ahesuṃ aggasāvikā;_  
_Bodhi tassa bhagavato,_  
_mahānīpoti vuccati._  

</Paragraph>

</Division>
<Division number="25" book="bu" bookNo="21" page="328">

<Paragraph number="622" type="verses" book="bu">
_Uruvelā yasavā ca,_  
_ahesuṃ aggupaṭṭhakā;_  
_Yasodharā sirimā ca,_  
_ahesuṃ aggupaṭṭhikā._  

</Paragraph>

</Division>
<Division number="26" book="bu" bookNo="21" page="328">

<Paragraph number="623" type="verses" book="bu">
_Aṭṭhāsītiratanāni,_  
_accuggato mahāmuni;_  
_Obhāseti disā sabbā,_  
_cando tāragaṇe yathā._  

</Paragraph>

</Division>
<Division number="27" book="bu" bookNo="21" page="328">

<Paragraph number="624" type="verses" book="bu">
_Cakkavattimaṇī nāma,_  
_yathā tapati yojanaṃ;_  
_Tatheva tassa ratanaṃ,_  
_samantā pharati yojanaṃ._  

</Paragraph>

</Division>
<Division number="28" book="bu" bookNo="21" page="328">

<Paragraph number="625" type="verses" book="bu">
_Navutivassasahassāni,_  
_āyu vijjati tāvade;_  
_Tāvatā tiṭṭhamāno so,_  
_tāresi janataṃ bahuṃ._  

</Paragraph>

</Division>
<Division number="29" book="bu" bookNo="21" page="328">

<Paragraph number="626" type="verses" book="bu">
_Tevijjachaḷabhiññehi,_  
_balappattehi tādihi;_  
_Samākulamidaṃ āsi,_  
_arahantehi sādhuhi._  

</Paragraph>

</Division>

_vippamuttā nirūpadhī;_  
_Ñāṇālokaṃ dassayitvā,_  
_nibbutā te mahāyasā._  

</Paragraph>
