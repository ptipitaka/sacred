<Paragraph number="802" type="verses" book="bu">
_“Tattheva maṇḍakappamhi,_  
_ahu satthā anuttaro;_  
_Anūpamo asamasamo,_  
_phusso lokagganāyako._  

</Paragraph>

</Division>
<Division number="2" book="bu">

<Paragraph number="803" type="verses" book="bu">
_Sopi sabbaṃ tamaṃ hantvā,_  
_vijaṭetvā mahājaṭaṃ;_  
_Sadevakaṃ tappayanto,_  
_abhivassi amatambunā._  

</Paragraph>

</Division>
<Division number="3" book="bu">

<Paragraph number="804" type="verses" book="bu">
_Dhammacakkaṃ pavattente,_  
_phusse nakkhattamaṅgale;_  
_Koṭisatasahassānaṃ,_  
_paṭhamābhisamayo ahu._  

</Paragraph>

</Division>
<Division number="4" book="bu">

<Paragraph number="805" type="verses" book="bu">
_Navutisatasahassānaṃ,_  
_dutiyābhisamayo ahu;_  
_Asītisatasahassānaṃ,_  
_tatiyābhisamayo ahu._  

</Paragraph>

</Division>
<Division number="5" book="bu">

<Paragraph number="806" type="verses" book="bu">
_Sannipātā tayo āsuṃ,_  
_phussassāpi mahesino;_  
_Khīṇāsavānaṃ vimalānaṃ,_  
_santacittāna tādinaṃ._  

</Paragraph>

</Division>
<Division number="6" book="bu">

<Paragraph number="807" type="verses" book="bu">
_Saṭṭhisatasahassānaṃ,_  
_paṭhamo āsi samāgamo;_  
_Paññāsasatasahassānaṃ,_  
_dutiyo āsi samāgamo._  

</Paragraph>

</Division>
<Division number="7" book="bu">

<Paragraph number="808" type="verses" book="bu">
_Cattārīsasatasahassānaṃ,_  
_Tatiyo āsi samāgamo;_  
_Anupādā vimuttānaṃ,_  
_Vocchinnapaṭisandhinaṃ._  

</Paragraph>

</Division>
<Division number="8" book="bu">

<Paragraph number="809" type="verses" book="bu">
_Ahaṃ tena samayena,_  
_vijitāvī nāma khattiyo;_  
_Chaḍḍayitvā mahārajjaṃ,_  
_pabbajiṃ tassa santike._  

</Paragraph>

</Division>
<Division number="9" book="bu">

<Paragraph number="810" type="verses" book="bu">
_Sopi maṃ buddho byākāsi,_  
_phusso lokagganāyako;_  
_‘Dvenavute ito kappe,_  
_ayaṃ buddho bhavissati._  

</Paragraph>

</Division>
<Division number="10" book="bu">

<Paragraph number="811" type="verses" book="bu">
_Padhānaṃ padahitvāna,_  
_ …pe… _  
_hessāma sammukhā imaṃ’._  

</Paragraph>

</Division>
<Division number="11" book="bu">

<Paragraph number="812" type="verses" book="bu">
_Tassāpi vacanaṃ sutvā,_  
_bhiyyo cittaṃ pasādayiṃ;_  
_Uttariṃ vatamadhiṭṭhāsiṃ,_  
_dasapāramipūriyā._  

</Paragraph>

</Division>
<Division number="12" book="bu">

<Paragraph number="813" type="verses" book="bu">
_Suttantaṃ vinayañcāpi,_  
_navaṅgaṃ satthusāsanaṃ;_  
_Sabbaṃ pariyāpuṇitvā,_  
_sobhayiṃ jinasāsanaṃ._  

</Paragraph>

</Division>
<Division number="13" book="bu">

<Paragraph number="814" type="verses" book="bu">
_Tatthappamatto viharanto,_  
_brahmaṃ bhāvetva bhāvanaṃ;_  
_Abhiññāsu pāramiṃ gantvā,_  
_brahmalokamagañchahaṃ._  

</Paragraph>

</Division>
<Division number="14" book="bu">

<Paragraph number="815" type="verses" book="bu">
_Kāsikaṃ nāma nagaraṃ,_  
_jayaseno nāma khattiyo;_  
_Sirimā nāma janikā,_  
_phussassāpi mahesino._  

</Paragraph>

</Division>
<Division number="15" book="bu">

<Paragraph number="816" type="verses" book="bu">
_Navavassasahassāni,_  
_Agāraṃ ajjha so vasi;_  
_Garuḷapakkha haṃsa suvaṇṇabhārā,_  
_Tayo pāsādamuttamā._  

</Paragraph>

</Division>
<Division number="16" book="bu">

<Paragraph number="817" type="verses" book="bu">
_Tiṃsaitthisahassāni,_  
_nāriyo samalaṅkatā;_  
_Kisāgotamī nāma nārī,_  
_anūpamo nāma atrajo._  

</Paragraph>

</Division>
<Division number="17" book="bu">

<Paragraph number="818" type="verses" book="bu">
_Nimitte caturo disvā,_  
_hatthiyānena nikkhami;_  
_Chamāsaṃ padhānacāraṃ,_  
_acarī purisuttamo._  

</Paragraph>

</Division>
<Division number="18" book="bu">

<Paragraph number="819" type="verses" book="bu">
_Brahmunā yācito santo,_  
_phusso lokagganāyako;_  
_Vatti cakkaṃ mahāvīro,_  
_migadāye naruttamo._  

</Paragraph>

</Division>
<Division number="19" book="bu">

<Paragraph number="820" type="verses" book="bu">
_Surakkhito dhammaseno,_  
_ahesuṃ aggasāvakā;_  
_Sabhiyo nāmupaṭṭhāko,_  
_phussassāpi mahesino._  

</Paragraph>

</Division>
<Division number="20" book="bu">

<Paragraph number="821" type="verses" book="bu">
_Cālā ca upacālā ca,_  
_ahesuṃ aggasāvikā;_  
_Bodhi tassa bhagavato,_  
_āmaṇḍoti pavuccati._  

</Paragraph>

</Division>
<Division number="21" book="bu">

<Paragraph number="822" type="verses" book="bu">
_Dhanañcayo visākho ca,_  
_ahesuṃ aggupaṭṭhakā;_  
_Padumā ceva nāgā ca,_  
_ahesuṃ aggupaṭṭhikā._  

</Paragraph>

</Division>
<Division number="22" book="bu">

<Paragraph number="823" type="verses" book="bu">
_Aṭṭhapaṇṇāsaratanaṃ,_  
_sopi accuggato muni;_  
_Sobhate sataraṃsīva,_  
_uḷurājāva pūrito._  

</Paragraph>

</Division>
<Division number="23" book="bu">

<Paragraph number="824" type="verses" book="bu">
_Navutivassasahassāni,_  
_āyu vijjati tāvade;_  
_Tāvatā tiṭṭhamāno so,_  
_tāresi janataṃ bahuṃ._  

</Paragraph>

</Division>
<Division number="24" book="bu">

<Paragraph number="825" type="verses" book="bu">
_Ovaditvā bahū satte,_  
_santāretvā bahū jane;_  
_Sopi satthā atulayaso,_  
_nibbuto so sasāvako._  

</Paragraph>

</Division>
<Division number="25" book="bu">

<Paragraph number="826" type="verses" book="bu">
_Phusso jinavaro satthā,_  
_Senārāmamhi nibbuto;_  
_Dhātuvitthārikaṃ āsi,_  
_Tesu tesu padesato”ti. (806)_  


</Paragraph>
</Division>
