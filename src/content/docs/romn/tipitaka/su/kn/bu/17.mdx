<Division number="1" book="bu">

<Paragraph number="722" type="verses" book="bu">
_“Tattheva maṇḍakappamhi,_  
_dhammadassī mahāyaso;_  
_Tamandhakāraṃ vidhamitvā,_  
_atirocati sadevake._  

</Paragraph>

</Division>
<Division number="2" book="bu">

<Paragraph number="723" type="verses" book="bu">
_Tassāpi atulatejassa,_  
_dhammacakkappavattane;_  
_Koṭisatasahassānaṃ,_  
_paṭhamābhisamayo ahu._  

</Paragraph>

</Division>
<Division number="3" book="bu">

<Paragraph number="724" type="verses" book="bu">
_Yadā buddho dhammadassī,_  
_vinesi sañjayaṃ isiṃ;_  
_Tadā navutikoṭīnaṃ,_  
_dutiyābhisamayo ahu._  

</Paragraph>

</Division>
<Division number="4" book="bu">

<Paragraph number="725" type="verses" book="bu">
_Yadā sakko upāgañchi,_  
_sapariso vināyakaṃ;_  
_Tadā asītikoṭīnaṃ,_  
_tatiyābhisamayo ahu._  

</Paragraph>

</Division>
<Division number="5" book="bu">

<Paragraph number="726" type="verses" book="bu">
_Tassāpi devadevassa,_  
_sannipātā tayo ahuṃ;_  
_Khīṇāsavānaṃ vimalānaṃ,_  
_santacittāna tādinaṃ._  

</Paragraph>

</Division>
<Division number="6" book="bu">

<Paragraph number="727" type="verses" book="bu">
_Yadā buddho dhammadassī,_  
_Saraṇe vassaṃ upāgami;_  
_Tadā koṭisatasahassānaṃ,_  
_Paṭhamo āsi samāgamo._  

</Paragraph>

</Division>
<Division number="7" book="bu">

<Paragraph number="728" type="verses" book="bu">
_Punāparaṃ yadā buddho,_  
_devato eti mānusaṃ;_  
_Tadāpi satakoṭīnaṃ,_  
_dutiyo āsi samāgamo._  

</Paragraph>

</Division>
<Division number="8" book="bu">

<Paragraph number="729" type="verses" book="bu">
_Punāparaṃ yadā buddho,_  
_pakāsesi dhute guṇe;_  
_Tadā asītikoṭīnaṃ,_  
_tatiyo āsi samāgamo._  

</Paragraph>

</Division>
<Division number="9" book="bu">

<Paragraph number="730" type="verses" book="bu">
_Ahaṃ tena samayena,_  
_sakko āsiṃ purindado;_  
_Dibbena gandhamālena,_  
_tūriyenābhipūjayiṃ._  

</Paragraph>

</Division>
<Division number="10" book="bu">

<Paragraph number="731" type="verses" book="bu">
_Sopi maṃ buddho byākāsi,_  
_devamajjhe nisīdiya;_  
_‘Aṭṭhārase kappasate,_  
_ayaṃ buddho bhavissati._  

</Paragraph>

</Division>
<Division number="11" book="bu">

<Paragraph number="732" type="verses" book="bu">
_Padhānaṃ padahitvāna,_  
_ …pe… _  
_hessāma sammukhā imaṃ’._  

</Paragraph>

</Division>
<Division number="12" book="bu">

<Paragraph number="733" type="verses" book="bu">
_Tassāpi vacanaṃ sutvā,_  
_bhiyyo cittaṃ pasādayiṃ;_  
_Uttariṃ vatamadhiṭṭhāsiṃ,_  
_dasapāramipūriyā._  

</Paragraph>

</Division>
<Division number="13" book="bu">

<Paragraph number="734" type="verses" book="bu">
_Saraṇaṃ nāma nagaraṃ,_  
_saraṇo nāma khattiyo;_  
_Sunandā nāma janikā,_  
_dhammadassissa satthuno._  

</Paragraph>

</Division>
<Division number="14" book="bu">

<Paragraph number="735" type="verses" book="bu">
_Aṭṭhavassasahassāni,_  
_Agāraṃ ajjha so vasi;_  
_Arajo virajo sudassano,_  
_Tayo pāsādamuttamā._  

</Paragraph>

</Division>
<Division number="15" book="bu">

<Paragraph number="736" type="verses" book="bu">
_Ticattārīsasahassāni,_  
_nāriyo samalaṅkatā;_  
_Vicikoḷi nāma nārī,_  
_atrajo puññavaḍḍhano._  

</Paragraph>

</Division>
<Division number="16" book="bu">

<Paragraph number="737" type="verses" book="bu">
_Nimitte caturo disvā,_  
_pāsādenābhinikkhami;_  
_Sattāhaṃ padhānacāraṃ,_  
_acarī purisuttamo._  

</Paragraph>

</Division>
<Division number="17" book="bu">

<Paragraph number="738" type="verses" book="bu">
_Brahmunā yācito santo,_  
_dhammadassī narāsabho;_  
_Vatti cakkaṃ mahāvīro,_  
_migadāye naruttamo._  

</Paragraph>

</Division>
<Division number="18" book="bu">

<Paragraph number="739" type="verses" book="bu">
_Padumo phussadevo ca,_  
_ahesuṃ aggasāvakā;_  
_Sunetto nāmupaṭṭhāko,_  
_dhammadassissa satthuno._  

</Paragraph>

</Division>
<Division number="19" book="bu">

<Paragraph number="740" type="verses" book="bu">
_Khemā ca saccanāmā ca,_  
_ahesuṃ aggasāvikā;_  
_Bodhi tassa bhagavato,_  
_bimbijāloti vuccati._  

</Paragraph>

</Division>
<Division number="20" book="bu">

<Paragraph number="741" type="verses" book="bu">
_Subhaddo kaṭissaho ceva,_  
_ahesuṃ aggupaṭṭhakā;_  
_Sāḷiyā ca kaḷiyā ca,_  
_ahesuṃ aggupaṭṭhikā._  

</Paragraph>

</Division>
<Division number="21" book="bu">

<Paragraph number="742" type="verses" book="bu">
_Sopi buddho asamasamo,_  
_asītihatthamuggato;_  
_Atirocati tejena,_  
_dasasahassimhi dhātuyā._  

</Paragraph>

</Division>
<Division number="22" book="bu">

<Paragraph number="743" type="verses" book="bu">
_Suphullo sālarājāva,_  
_vijjūva gagane yathā;_  
_Majjhanhikeva sūriyo,_  
_evaṃ so upasobhatha._  

</Paragraph>

</Division>
<Division number="23" book="bu">

<Paragraph number="744" type="verses" book="bu">
_Tassāpi atulatejassa,_  
_samakaṃ āsi jīvitaṃ;_  
_Vassasatasahassāni,_  
_loke aṭṭhāsi cakkhumā._  

</Paragraph>

</Division>
<Division number="24" book="bu">

<Paragraph number="745" type="verses" book="bu">
_Obhāsaṃ dassayitvāna,_  
_vimalaṃ katvāna sāsanaṃ;_  
_Cavi candova gagane,_  
_nibbuto so sasāvako._  

</Paragraph>

</Division>
<Division number="25" book="bu">

<Paragraph number="746" type="verses" book="bu">
_Dhammadassī mahāvīro,_  
_Sālārāmamhi nibbuto;_  
_Tatthevassa thūpavaro,_  
_Tīṇiyojanamuggato”ti. (729)_  


</Paragraph>
</Division>

<Paragraph number="747" type="center" book="bu">
Dhammadassissa bhagavato vaṃso pannarasamo.
</Paragraph>
