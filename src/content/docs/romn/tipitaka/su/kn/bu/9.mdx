<Division number="1" book="bu">

<Paragraph number="472" type="verses" book="bu">
_“Sobhitassa aparena,_  
_Sambuddho dvipaduttamo;_  
_Anomadassī amitayaso,_  
_Tejassī duratikkamo._  

</Paragraph>

</Division>
<Division number="2" book="bu">

<Paragraph number="473" type="verses" book="bu">
_So chetvā bandhanaṃ sabbaṃ,_  
_viddhaṃsetvā tayo bhave;_  
_Anivattigamanaṃ maggaṃ,_  
_desesi devamānuse._  

</Paragraph>

</Division>
<Division number="3" book="bu">

<Paragraph number="474" type="verses" book="bu">
_Sāgarova asaṅkhobbho,_  
_pabbatova durāsado;_  
_Ākāsova ananto so,_  
_sālarājāva phullito._  

</Paragraph>

</Division>
<Division number="4" book="bu">

<Paragraph number="475" type="verses" book="bu">
_Dassanenapi taṃ buddhaṃ,_  
_tositā honti pāṇino;_  
_Byāharantaṃ giraṃ sutvā,_  
_amataṃ pāpuṇanti te._  

</Paragraph>

</Division>
<Division number="5" book="bu">

<Paragraph number="476" type="verses" book="bu">
_Dhammābhisamayo tassa,_  
_Iddho phīto tadā ahu;_  
_Koṭisatāni abhisamiṃsu,_  
_Paṭhame dhammadesane._  

</Paragraph>

</Division>
<Division number="6" book="bu">

<Paragraph number="477" type="verses" book="bu">
_Tato paraṃ abhisamaye,_  
_Vassante dhammavuṭṭhiyo;_  
_Asītikoṭiyobhisamiṃsu,_  
_Dutiye dhammadesane._  

</Paragraph>

</Division>
<Division number="7" book="bu">

<Paragraph number="478" type="verses" book="bu">
_Tatoparañhi vassante,_  
_tappayante ca pāṇinaṃ;_  
_Aṭṭhasattatikoṭīnaṃ,_  
_tatiyābhisamayo ahu._  

</Paragraph>

</Division>
<Division number="8" book="bu">

<Paragraph number="479" type="verses" book="bu">
_Sannipātā tayo āsuṃ,_  
_tassāpi ca mahesino;_  
_Abhiññābalappattānaṃ,_  
_pupphitānaṃ vimuttiyā._  

</Paragraph>

</Division>
<Division number="9" book="bu">

<Paragraph number="480" type="verses" book="bu">
_Aṭṭhasatasahassānaṃ,_  
_sannipāto tadā ahu;_  
_Pahīnamadamohānaṃ,_  
_santacittāna tādinaṃ._  

</Paragraph>

</Division>
<Division number="10" book="bu">

<Paragraph number="481" type="verses" book="bu">
_Sattasatasahassānaṃ,_  
_dutiyo āsi samāgamo;_  
_Anaṅgaṇānaṃ virajānaṃ,_  
_upasantāna tādinaṃ._  

</Paragraph>

</Division>
<Division number="11" book="bu">

<Paragraph number="482" type="verses" book="bu">
_Channaṃ satasahassānaṃ,_  
_tatiyo āsi samāgamo;_  
_Abhiññābalappattānaṃ,_  
_nibbutānaṃ tapassinaṃ._  

</Paragraph>

</Division>
<Division number="12" book="bu">

<Paragraph number="483" type="verses" book="bu">
_Ahaṃ tena samayena,_  
_yakkho āsiṃ mahiddhiko;_  
_Nekānaṃ yakkhakoṭīnaṃ,_  
_vasavattimhi issaro._  

</Paragraph>

</Division>
<Division number="13" book="bu">

<Paragraph number="484" type="verses" book="bu">
_Tadāpi taṃ buddhavaraṃ,_  
_upagantvā mahesinaṃ;_  
_Annapānena tappesiṃ,_  
_sasaṃghaṃ lokanāyakaṃ._  

</Paragraph>

</Division>
<Division number="14" book="bu">

<Paragraph number="485" type="verses" book="bu">
_Sopi maṃ tadā byākāsi,_  
_visuddhanayano muni;_  
_‘Aparimeyyito kappe,_  
_ayaṃ buddho bhavissati._  

</Paragraph>

</Division>
<Division number="15" book="bu">

<Paragraph number="486" type="verses" book="bu">
_Padhānaṃ padahitvāna,_  
_ …pe… _  
_hessāma sammukhā imaṃ’._  

</Paragraph>

</Division>
<Division number="16" book="bu">

<Paragraph number="487" type="verses" book="bu">
_Tassāpi vacanaṃ sutvā,_  
_haṭṭho saṃviggamānaso;_  
_Uttariṃ vatamadhiṭṭhāsiṃ,_  
_dasapāramipūriyā._  

</Paragraph>

</Division>
<Division number="17" book="bu">

<Paragraph number="488" type="verses" book="bu">
_Nagaraṃ candavatī nāma,_  
_yasavā nāma khattiyo;_  
_Mātā yasodharā nāma,_  
_anomadassissa satthuno._  

</Paragraph>

</Division>
<Division number="18" book="bu">

<Paragraph number="489" type="verses" book="bu">
_Dasavassasahassāni,_  
_agāraṃ ajjha so vasi;_  
_Sirī upasirī vaḍḍho,_  
_tayo pāsādamuttamā._  

</Paragraph>

</Division>
<Division number="19" book="bu">

<Paragraph number="490" type="verses" book="bu">
_Tevīsatisahassāni,_  
_nāriyo samalaṅkatā;_  
_Sirimā nāma sā nārī,_  
_upavāṇo nāma atrajo._  

</Paragraph>

</Division>
<Division number="20" book="bu">

<Paragraph number="491" type="verses" book="bu">
_Nimitte caturo disvā,_  
_sivikāyābhinikkhami;_  
_Anūnadasamāsāni,_  
_padhānaṃ padahī jino._  

</Paragraph>

</Division>
<Division number="21" book="bu">

<Paragraph number="492" type="verses" book="bu">
_Brahmunā yācito santo,_  
_anomadassī mahāmuni;_  
_Vatti cakkaṃ mahāvīro,_  
_uyyāne so sudassane._  

</Paragraph>

</Division>
<Division number="22" book="bu">

<Paragraph number="493" type="verses" book="bu">
_Nisabho ca anomo ca,_  
_ahesuṃ aggasāvakā;_  
_Varuṇo nāmupaṭṭhāko,_  
_anomadassissa satthuno._  

</Paragraph>

</Division>
<Division number="23" book="bu">

<Paragraph number="494" type="verses" book="bu">
_Sundarī ca sumanā ca,_  
_ahesuṃ aggasāvikā;_  
_Bodhi tassa bhagavato,_  
_ajjunoti pavuccati._  

</Paragraph>

</Division>
<Division number="24" book="bu">

<Paragraph number="495" type="verses" book="bu">
_Nandivaḍḍho sirivaḍḍho,_  
_ahesuṃ aggupaṭṭhakā;_  
_Uppalā ceva padumā ca,_  
_ahesuṃ aggupaṭṭhikā._  

</Paragraph>

</Division>
<Division number="25" book="bu">

<Paragraph number="496" type="verses" book="bu">
_Aṭṭhapaṇṇāsaratanaṃ,_  
_accuggato mahāmuni;_  
_Pabhā niddhāvatī tassa,_  
_sataraṃsīva uggato._  

</Paragraph>

</Division>
<Division number="26" book="bu">

<Paragraph number="497" type="verses" book="bu">
_Vassasatasahassāni,_  
_āyu vijjati tāvade;_  
_Tāvatā tiṭṭhamāno so,_  
_tāresi janataṃ bahuṃ._  

</Paragraph>

</Division>
<Division number="27" book="bu">

<Paragraph number="498" type="verses" book="bu">
_Supupphitaṃ pāvacanaṃ,_  
_arahantehi tādihi;_  
_Vītarāgehi vimalehi,_  
_sobhittha jinasāsanaṃ._  

</Paragraph>

</Division>
<Division number="28" book="bu">

<Paragraph number="499" type="verses" book="bu">
_So ca satthā amitayaso,_  
_Yugāni tāni atuliyāni;_  
_Sabbaṃ tamantarahitaṃ,_  
_Nanu rittā sabbasaṅkhārā._  

</Paragraph>

</Division>
<Division number="29" book="bu">

<Paragraph number="500" type="verses" book="bu">
_Anomadassī jino satthā,_  
_Dhammārāmamhi nibbuto;_  
_Tatthevassa jinathūpo,_  
_Ubbedho pañcavīsatī”ti. (491)_  


</Paragraph>
</Division>

<Paragraph number="501" type="center" book="bu">
Anomadassissa bhagavato vaṃso sattamo.
</Paragraph>
