<Paragraph number="630" type="verses" book="bu">
_“Tattheva maṇḍakappamhi,_  
_sujāto nāma nāyako;_  
_Sīhahanusabhakkhandho,_  
_appameyyo durāsado._  

</Paragraph>

</Division>
<Division number="2" book="bu">

<Paragraph number="631" type="verses" book="bu">
_Candova vimalo suddho,_  
_sataraṃsīva patāpavā;_  
_Evaṃ sobhati sambuddho,_  
_jalanto siriyā sadā._  

</Paragraph>

</Division>
<Division number="3" book="bu">

<Paragraph number="632" type="verses" book="bu">
_Pāpuṇitvāna sambuddho,_  
_kevalaṃ bodhimuttamaṃ;_  
_Sumaṅgalamhi nagare,_  
_dhammacakkaṃ pavattayi._  

</Paragraph>

</Division>
<Division number="4" book="bu">

<Paragraph number="633" type="verses" book="bu">
_Desente pavaraṃ dhammaṃ,_  
_Sujāte lokanāyake;_  
_Asītikoṭī abhisamiṃsu,_  
_Paṭhame dhammadesane._  

</Paragraph>

</Division>
<Division number="5" book="bu">

<Paragraph number="634" type="verses" book="bu">
_Yadā sujāto amitayaso,_  
_Deve vassaṃ upāgami;_  
_Sattatiṃsasatasahassānaṃ,_  
_Dutiyābhisamayo ahu._  

</Paragraph>

</Division>
<Division number="6" book="bu">

<Paragraph number="635" type="verses" book="bu">
_Yadā sujāto asamasamo,_  
_Upagacchi pitusantikaṃ;_  
_Saṭṭhisatasahassānaṃ,_  
_Tatiyābhisamayo ahu._  

</Paragraph>

</Division>
<Division number="7" book="bu">

<Paragraph number="636" type="verses" book="bu">
_Sannipātā tayo āsuṃ,_  
_sujātassa mahesino;_  
_Khīṇāsavānaṃ vimalānaṃ,_  
_santacittāna tādinaṃ._  

</Paragraph>

</Division>
<Division number="8" book="bu">

<Paragraph number="637" type="verses" book="bu">
_Abhiññābalappattānaṃ,_  
_appattānaṃ bhavābhave;_  
_Saṭṭhisatasahassāni,_  
_paṭhamaṃ sannipatiṃsu te._  

</Paragraph>

</Division>
<Division number="9" book="bu">

<Paragraph number="638" type="verses" book="bu">
_Punāparaṃ sannipāte,_  
_tidivorohaṇe jine;_  
_Paññāsasatasahassānaṃ,_  
_dutiyo āsi samāgamo._  

</Paragraph>

</Division>
<Division number="10" book="bu">

<Paragraph number="639" type="verses" book="bu">
_Upasaṅkamanto narāsabhaṃ,_  
_Tassa yo aggasāvako;_  
_Catūhi satasahassehi,_  
_Sambuddhaṃ upasaṅkami._  

</Paragraph>

</Division>
<Division number="11" book="bu">

<Paragraph number="640" type="verses" book="bu">
_Ahaṃ tena samayena,_  
_catudīpamhi issaro;_  
_Antalikkhacaro āsiṃ,_  
_cakkavattī mahabbalo._  

</Paragraph>

</Division>
<Division number="12" book="bu">

<Paragraph number="641" type="verses" book="bu">
_Loke acchariyaṃ disvā,_  
_abbhutaṃ lomahaṃsanaṃ;_  
_Upagantvāna vandiṃ so,_  
_sujātaṃ lokanāyakaṃ._  

</Paragraph>

</Division>
<Division number="13" book="bu">

<Paragraph number="642" type="verses" book="bu">
_Catudīpe mahārajjaṃ,_  
_ratane satta uttame;_  
_Buddhe niyyādayitvāna,_  
_pabbajiṃ tassa santike._  

</Paragraph>

</Division>
<Division number="14" book="bu">

<Paragraph number="643" type="verses" book="bu">
_Ārāmikā janapade,_  
_uṭṭhānaṃ paṭipiṇḍiya;_  
_Upanenti bhikkhusaṃghassa,_  
_paccayaṃ sayanāsanaṃ._  

</Paragraph>

</Division>
<Division number="15" book="bu">

<Paragraph number="644" type="verses" book="bu">
_Sopi maṃ buddho byākāsi,_  
_dasasahassimhi issaro;_  
_‘Tiṃsakappasahassamhi,_  
_ayaṃ buddho bhavissati._  

</Paragraph>

</Division>
<Division number="16" book="bu">

<Paragraph number="645" type="verses" book="bu">
_Padhānaṃ padahitvāna,_  
_ …pe… _  
_hessāma sammukhā imaṃ’._  

</Paragraph>

</Division>
<Division number="17" book="bu">

<Paragraph number="646" type="verses" book="bu">
_Tassāpi vacanaṃ sutvā,_  
_bhiyyo hāsaṃ janesahaṃ;_  
_Adhiṭṭhahiṃ vataṃ uggaṃ,_  
_dasapāramipūriyā._  

</Paragraph>

</Division>
<Division number="18" book="bu">

<Paragraph number="647" type="verses" book="bu">
_Suttantaṃ vinayañcāpi,_  
_navaṅgaṃ satthusāsanaṃ;_  
_Sabbaṃ pariyāpuṇitvāna,_  
_sobhayiṃ jinasāsanaṃ._  

</Paragraph>

</Division>
<Division number="19" book="bu">

<Paragraph number="648" type="verses" book="bu">
_Tatthappamatto viharanto,_  
_brahmaṃ bhāvetva bhāvanaṃ;_  
_Abhiññāpāramiṃ gantvā,_  
_brahmalokamagañchahaṃ._  

</Paragraph>

</Division>
<Division number="20" book="bu">

<Paragraph number="649" type="verses" book="bu">
_Sumaṅgalaṃ nāma nagaraṃ,_  
_uggato nāma khattiyo;_  
_Mātā pabhāvatī nāma,_  
_sujātassa mahesino._  

</Paragraph>

</Division>
<Division number="21" book="bu">

<Paragraph number="650" type="verses" book="bu">
_Navavassasahassāni,_  
_agāraṃ ajjha so vasi;_  
_Sirī upasirī nando,_  
_tayo pāsādamuttamā._  

</Paragraph>

</Division>
<Division number="22" book="bu">

<Paragraph number="651" type="verses" book="bu">
_Tevīsatisahassāni,_  
_nāriyo samalaṅkatā;_  
_Sirinandā nāma nārī,_  
_upaseno nāma atrajo._  

</Paragraph>

</Division>
<Division number="23" book="bu">

<Paragraph number="652" type="verses" book="bu">
_Nimitte caturo disvā,_  
_assayānena nikkhami;_  
_Anūnanavamāsāni,_  
_padhānaṃ padahī jino._  

</Paragraph>

</Division>
<Division number="24" book="bu">

<Paragraph number="653" type="verses" book="bu">
_Brahmunā yācito santo,_  
_sujāto lokanāyako;_  
_Vatti cakkaṃ mahāvīro,_  
_sumaṅgaluyyānamuttame._  

</Paragraph>

</Division>
<Division number="25" book="bu">

<Paragraph number="654" type="verses" book="bu">
_Sudassano sudevo ca,_  
_ahesuṃ aggasāvakā;_  
_Nārado nāmupaṭṭhāko,_  
_sujātassa mahesino._  

</Paragraph>

</Division>
<Division number="26" book="bu">

<Paragraph number="655" type="verses" book="bu">
_Nāgā ca nāgasamālā ca,_  
_ahesuṃ aggasāvikā;_  
_Bodhi tassa bhagavato,_  
_mahāveḷūti vuccati._  

</Paragraph>

</Division>
<Division number="27" book="bu">

<Paragraph number="656" type="verses" book="bu">
_So ca rukkho ghanakkhandho,_  
_Acchiddo hoti pattiko;_  
_Uju vaṃso brahā hoti,_  
_Dassanīyo manoramo._  

</Paragraph>

</Division>
<Division number="28" book="bu">

<Paragraph number="657" type="verses" book="bu">
_Ekakkhandho pavaḍḍhitvā,_  
_tato sākhā pabhijjati;_  
_Yathā subaddho morahattho,_  
_evaṃ sobhati so dumo._  

</Paragraph>

</Division>
<Division number="29" book="bu">

<Paragraph number="658" type="verses" book="bu">
_Na tassa kaṇṭakā honti,_  
_nāpi chiddaṃ mahā ahu;_  
_Vitthiṇṇasākho aviralo,_  
_sandacchāyo manoramo._  

</Paragraph>

</Division>
<Division number="30" book="bu">

<Paragraph number="659" type="verses" book="bu">
_Sudatto ceva citto ca,_  
_ahesuṃ aggupaṭṭhakā;_  
_Subhaddā ca padumā ca,_  
_ahesuṃ aggupaṭṭhikā._  

</Paragraph>

</Division>
<Division number="31" book="bu">

<Paragraph number="660" type="verses" book="bu">
_Paññāsaratano āsi,_  
_uccattanena so jino;_  
_Sabbākāravarūpeto,_  
_sabbaguṇamupāgato._  

</Paragraph>

</Division>
<Division number="32" book="bu">

<Paragraph number="661" type="verses" book="bu">
_Tassa pabhā asamasamā,_  
_niddhāvati samantato;_  
_Appamāṇo atuliyo,_  
_opammehi anūpamo._  

</Paragraph>

</Division>
<Division number="33" book="bu">

<Paragraph number="662" type="verses" book="bu">
_Navutivassasahassāni,_  
_āyu vijjati tāvade;_  
_Tāvatā tiṭṭhamāno so,_  
_tāresi janataṃ bahuṃ._  

</Paragraph>

</Division>
<Division number="34" book="bu">

<Paragraph number="663" type="verses" book="bu">
_Yathāpi sāgare ūmī,_  
_gagane tārakā yathā;_  
_Evaṃ tadā pāvacanaṃ,_  
_arahantehi cittitaṃ._  

</Paragraph>

</Division>
<Division number="35" book="bu">

<Paragraph number="664" type="verses" book="bu">
_So ca buddho asamasamo,_  
_Guṇāni ca tāni atuliyāni;_  
_Sabbaṃ tamantarahitaṃ,_  
_Nanu rittā sabbasaṅkhārā._  

</Paragraph>

</Division>
<Division number="36" book="bu">

_Silārāmamhi nibbuto;_  
_Tattheva tassa cetiyo,_  
_Tīṇigāvutamuggato”ti. (651)_  


</Paragraph>
</Division>
