<Paragraph number="1082" type="verses" book="bu">
_Mahāgotamo jinavaro,_  
_kusināramhi nibbuto;_  
_Dhātuvitthārikaṃ āsi,_  
_tesu tesu padesato._  

</Paragraph>

</Division>
<Division number="2" book="bu" bookNo="21" page="366">

<Paragraph number="1083" type="verses" book="bu">
_Eko ajātasattussa,_  
_eko vesāliyā pure;_  
_Eko kapilavatthusmiṃ,_  
_eko ca allakappake._  

</Paragraph>

</Division>
<Division number="3" book="bu" bookNo="21" page="366">

<Paragraph number="1084" type="verses" book="bu">
_Eko ca rāmagāmamhi,_  
_eko ca veṭhadīpake;_  
_Eko pāveyyake malle,_  
_eko ca kosinārake._  

</Paragraph>

</Division>
<Division number="4" book="bu" bookNo="21" page="366">

<Paragraph number="1085" type="verses" book="bu">
_Kumbhassa thūpaṃ kāresi,_  
_brāhmaṇo doṇasavhayo;_  
_Aṅgārathūpaṃ kāresuṃ,_  
_moriyā tuṭṭhamānasā._  

</Paragraph>

</Division>
<Division number="5" book="bu" bookNo="21" page="366">

<Paragraph number="1086" type="verses" book="bu">
_Aṭṭha sārīrikā thūpā,_  
_navamo kumbhacetiyo;_  
_Aṅgārathūpo dasamo,_  
_tadāyeva patiṭṭhito._  

</Paragraph>

</Division>
<Division number="6" book="bu" bookNo="21" page="367">

<Paragraph number="1087" type="verses" book="bu">
_Uṇhīsaṃ catasso dāṭhā,_  
_akkhakā dve ca dhātuyo;_  
_Asambhinnā imā satta,_  
_sesā bhinnāva dhātuyo._  

</Paragraph>

</Division>
<Division number="7" book="bu" bookNo="21" page="367">

<Paragraph number="1088" type="verses" book="bu">
_Mahantā muggamattā ca,_  
_majjhimā bhinnataṇḍulā;_  
_Khuddakā sāsapamattā ca,_  
_nānāvaṇṇā ca dhātuyo._  

</Paragraph>

</Division>
<Division number="8" book="bu" bookNo="21" page="367">

<Paragraph number="1089" type="verses" book="bu">
_Mahantā suvaṇṇavaṇṇā ca,_  
_muttavaṇṇā ca majjhimā;_  
_Khuddakā makulavaṇṇā ca,_  
_soḷasadoṇamattikā._  

</Paragraph>

</Division>
<Division number="9" book="bu" bookNo="21" page="367">

<Paragraph number="1090" type="verses" book="bu">
_Mahantā pañca nāḷiyo,_  
_nāḷiyo pañca majjhimā;_  
_Khuddakā cha nāḷī ceva,_  
_etā sabbāpi dhātuyo._  

</Paragraph>

</Division>
<Division number="10" book="bu" bookNo="21" page="367">

<Paragraph number="1091" type="verses" book="bu">
_Uṇhīsaṃ sīhaḷe dīpe,_  
_brahmaloke ca vāmakaṃ;_  
_Sīhaḷe dakkhiṇakkhañca,_  
_sabbāpetā patiṭṭhitā._  

</Paragraph>

</Division>
<Division number="11" book="bu" bookNo="21" page="367">

<Paragraph number="1092" type="verses" book="bu">
_Ekā dāṭhā tidasapure,_  
_ekā nāgapure ahu;_  
_Ekā gandhāravisaye,_  
_ekā kaliṅgarājino._  

</Paragraph>

</Division>
<Division number="12" book="bu" bookNo="21" page="367">

<Paragraph number="1093" type="verses" book="bu">
_Cattālīsasamā dantā,_  
_kesā lomā ca sabbaso;_  
_Devā hariṃsu ekekaṃ,_  
_cakkavāḷaparamparā._  

</Paragraph>

</Division>
<Division number="13" book="bu" bookNo="21" page="367">

<Paragraph number="1094" type="verses" book="bu">
_Vajirāyaṃ bhagavato,_  
_Patto daṇḍañca cīvaraṃ;_  
_Nivāsanaṃ kulaghare,_  
_Paccattharaṇaṃ kapilavhaye._  

</Paragraph>

</Division>
<Division number="14" book="bu" bookNo="21" page="367">

<Paragraph number="1095" type="verses" book="bu">
_Pāṭaliputtapuramhi,_  
_karaṇaṃ kāyabandhanaṃ;_  
_Campāyudakasāṭiyaṃ,_  
_uṇṇalomañca kosale._  

</Paragraph>

</Division>
<Division number="15" book="bu" bookNo="21" page="367">

<Paragraph number="1096" type="verses" book="bu">
_Kāsāvaṃ brahmaloke ca,_  
_veṭhanaṃ tidase pure;_  
_Nisīdanaṃ avantīsu,_  
_raṭṭhe attharaṇaṃ tadā._  

</Paragraph>

</Division>
<Division number="16" book="bu" bookNo="21" page="367">

<Paragraph number="1097" type="verses" book="bu">
_Araṇī ca mithilāyaṃ,_  
_videhe parisāvanaṃ;_  
_Vāsi sūcigharañcāpi,_  
_indapatthapure tadā._  

</Paragraph>

</Division>
<Division number="17" book="bu" bookNo="21" page="367">

<Paragraph number="1098" type="verses" book="bu">
_Parikkhārā avasesā,_  
_janapade aparantake;_  
_Paribhuttāni muninā,_  
_akaṃsu manujā tadā._  

</Paragraph>

</Division>
<Division number="18" book="bu" bookNo="21" page="367">

<Paragraph number="1099" type="verses" book="bu">
_Dhātuvitthārikaṃ āsi,_  
_Gotamassa mahesino;_  
_Pāṇīnaṃ anukampāya,_  
_Ahu porāṇikaṃ tadāti. (1070)_
</Paragraph>
</Division>
