<Division number="1" book="bu">

<Paragraph number="926" type="verses" book="bu">
_“Vessabhussa aparena,_  
_sambuddho dvipaduttamo;_  
_Kakusandho nāma nāmena,_  
_appameyyo durāsado._  

</Paragraph>

</Division>
<Division number="2" book="bu">

<Paragraph number="927" type="verses" book="bu">
_Ugghāṭetvā sabbabhavaṃ,_  
_cariyāya pāramiṃ gato;_  
_Sīhova pañjaraṃ bhetvā,_  
_patto sambodhimuttamaṃ._  

</Paragraph>

</Division>
<Division number="3" book="bu">

<Paragraph number="928" type="verses" book="bu">
_Dhammacakkaṃ pavattente,_  
_Kakusandhe lokanāyake;_  
_Cattārīsakoṭisahassānaṃ,_  
_Dhammābhisamayo ahu._  

</Paragraph>

</Division>
<Division number="4" book="bu">

<Paragraph number="929" type="verses" book="bu">
_Antalikkhamhi ākāse,_  
_yamakaṃ katvā vikubbanaṃ;_  
_Tiṃsakoṭisahassānaṃ,_  
_bodhesi devamānuse._  

</Paragraph>

</Division>
<Division number="5" book="bu">

<Paragraph number="930" type="verses" book="bu">
_Naradevassa yakkhassa,_  
_catusaccappakāsane;_  
_Dhammābhisamayo tassa,_  
_gaṇanāto asaṅkhiyo._  

</Paragraph>

</Division>
<Division number="6" book="bu">

<Paragraph number="931" type="verses" book="bu">
_Kakusandhassa bhagavato,_  
_eko āsi samāgamo;_  
_Khīṇāsavānaṃ vimalānaṃ,_  
_santacittāna tādinaṃ._  

</Paragraph>

</Division>
<Division number="7" book="bu">

<Paragraph number="932" type="verses" book="bu">
_Cattālīsasahassānaṃ,_  
_tadā āsi samāgamo;_  
_Dantabhūmimanuppattānaṃ,_  
_āsavārigaṇakkhayā._  

</Paragraph>

</Division>
<Division number="8" book="bu">

<Paragraph number="933" type="verses" book="bu">
_Ahaṃ tena samayena,_  
_khemo nāmāsi khattiyo;_  
_Tathāgate jinaputte,_  
_dānaṃ datvā anappakaṃ._  

</Paragraph>

</Division>
<Division number="9" book="bu">

<Paragraph number="934" type="verses" book="bu">
_Pattañca cīvaraṃ datvā,_  
_añjanaṃ madhulaṭṭhikaṃ;_  
_Imetaṃ patthitaṃ sabbaṃ,_  
_paṭiyādemi varaṃ varaṃ._  

</Paragraph>

</Division>
<Division number="10" book="bu">

<Paragraph number="935" type="verses" book="bu">
_Sopi maṃ buddho byākāsi,_  
_kakusandho vināyako;_  
_‘Imamhi bhaddake kappe,_  
_ayaṃ buddho bhavissati._  

</Paragraph>

</Division>
<Division number="11" book="bu">

<Paragraph number="936" type="verses" book="bu">
_Ahu kapilavhayā rammā,_  
_ …pe… _  
_hessāma sammukhā imaṃ’._  

</Paragraph>

</Division>
<Division number="12" book="bu">

<Paragraph number="937" type="verses" book="bu">
_Tassāpi vacanaṃ sutvā,_  
_bhiyyo cittaṃ pasādayiṃ;_  
_Uttariṃ vatamadhiṭṭhāsiṃ,_  
_dasapāramipūriyā._  

</Paragraph>

</Division>
<Division number="13" book="bu">

<Paragraph number="938" type="verses" book="bu">
_Nagaraṃ khemāvatī nāma,_  
_khemo nāmāsahaṃ tadā;_  
_Sabbaññutaṃ gavesanto,_  
_pabbajiṃ tassa santike._  

</Paragraph>

</Division>
<Division number="14" book="bu">

<Paragraph number="939" type="verses" book="bu">
_Brāhmaṇo aggidatto ca,_  
_āsi buddhassa so pitā;_  
_Visākhā nāma janikā,_  
_kakusandhassa satthuno._  

</Paragraph>

</Division>
<Division number="15" book="bu">

<Paragraph number="940" type="verses" book="bu">
_Vasate tattha kheme pure,_  
_sambuddhassa mahākulaṃ;_  
_Narānaṃ pavaraṃ seṭṭhaṃ,_  
_jātimantaṃ mahāyasaṃ._  

</Paragraph>

</Division>
<Division number="16" book="bu">

<Paragraph number="941" type="verses" book="bu">
_Catuvassasahassāni,_  
_Agāraṃ ajjha so vasi;_  
_Kāmakāmavaṇṇakāmasuddhināmā,_  
_Tayo pāsādamuttamā._  

</Paragraph>

</Division>
<Division number="17" book="bu">

<Paragraph number="942" type="verses" book="bu">
_Samattiṃsasahassāni,_  
_nāriyo samalaṅkatā;_  
_Rocinī nāma sā nārī,_  
_uttaro nāma atrajo._  

</Paragraph>

</Division>
<Division number="18" book="bu">

<Paragraph number="943" type="verses" book="bu">
_Nimitte caturo disvā,_  
_rathayānena nikkhami;_  
_Anūnaaṭṭhamāsāni,_  
_padhānaṃ padahī jino._  

</Paragraph>

</Division>
<Division number="19" book="bu">

<Paragraph number="944" type="verses" book="bu">
_Brahmunā yācito santo,_  
_kakusandho vināyako;_  
_Vatti cakkaṃ mahāvīro,_  
_migadāye naruttamo._  

</Paragraph>

</Division>
<Division number="20" book="bu">

<Paragraph number="945" type="verses" book="bu">
_Vidhuro ca sañjīvo ca,_  
_ahesuṃ aggasāvakā;_  
_Buddhijo nāmupaṭṭhāko,_  
_kakusandhassa satthuno._  

</Paragraph>

</Division>
<Division number="21" book="bu">

<Paragraph number="946" type="verses" book="bu">
_Sāmā ca campānāmā ca,_  
_ahesuṃ aggasāvikā;_  
_Bodhi tassa bhagavato,_  
_sirīsoti pavuccati._  

</Paragraph>

</Division>
<Division number="22" book="bu">

<Paragraph number="947" type="verses" book="bu">
_Accuto ca sumano ca,_  
_ahesuṃ aggupaṭṭhakā;_  
_Nandā ceva sunandā ca,_  
_ahesuṃ aggupaṭṭhikā._  

</Paragraph>

</Division>
<Division number="23" book="bu">

<Paragraph number="948" type="verses" book="bu">
_Cattālīsaratanāni,_  
_accuggato mahāmuni;_  
_Kanakappabhā niccharati,_  
_samantā dasayojanaṃ._  

</Paragraph>

</Division>
<Division number="24" book="bu">

<Paragraph number="949" type="verses" book="bu">
_Cattālīsavassasahassāni,_  
_Āyu tassa mahesino;_  
_Tāvatā tiṭṭhamāno so,_  
_Tāresi janataṃ bahuṃ._  

</Paragraph>

</Division>
<Division number="25" book="bu">

<Paragraph number="950" type="verses" book="bu">
_Dhammāpaṇaṃ pasāretvā,_  
_naranārīnaṃ sadevake;_  
_Naditvā sīhanādaṃva,_  
_nibbuto so sasāvako._  

</Paragraph>

</Division>
<Division number="26" book="bu">

<Paragraph number="951" type="verses" book="bu">
_Aṭṭhaṅgavacanasampanno,_  
_acchiddāni nirantaraṃ;_  
_Sabbaṃ tamantarahitaṃ,_  
_nanu rittā sabbasaṅkhārā._  

</Paragraph>

</Division>
<Division number="27" book="bu">

<Paragraph number="952" type="verses" book="bu">
_Kakusandho jinavaro,_  
_Khemārāmamhi nibbuto;_  
_Tatthevassa thūpavaro,_  
_Gāvutaṃ nabhamuggato”ti. (928)_  


</Paragraph>
</Division>

<Paragraph number="953" type="center" book="bu">
Kakusandhassa bhagavato vaṃso dvāvīsatimo.
</Paragraph>
