<Division number="1" book="bu">

<Paragraph number="748" type="verses" book="bu">
_“Dhammadassissa aparena,_  
_siddhattho nāma nāyako;_  
_Nihanitvā tamaṃ sabbaṃ,_  
_sūriyo abbhuggato yathā._  

</Paragraph>

</Division>
<Division number="2" book="bu">

<Paragraph number="749" type="verses" book="bu">
_Sopi patvāna sambodhiṃ,_  
_santārento sadevakaṃ;_  
_Abhivassi dhammameghena,_  
_nibbāpento sadevakaṃ._  

</Paragraph>

</Division>
<Division number="3" book="bu">

<Paragraph number="750" type="verses" book="bu">
_Tassāpi atulatejassa,_  
_Ahesuṃ abhisamayā tayo;_  
_Koṭisatasahassānaṃ,_  
_Paṭhamābhisamayo ahu._  

</Paragraph>

</Division>
<Division number="4" book="bu">

<Paragraph number="751" type="verses" book="bu">
_Punāparaṃ bhīmarathe,_  
_yadā āhani dundubhiṃ;_  
_Tadā navutikoṭīnaṃ,_  
_dutiyābhisamayo ahu._  

</Paragraph>

</Division>
<Division number="5" book="bu">

<Paragraph number="752" type="verses" book="bu">
_Yadā buddho dhammaṃ desesi,_  
_vebhāre so puruttame;_  
_Tadā navutikoṭīnaṃ,_  
_tatiyābhisamayo ahu._  

</Paragraph>

</Division>
<Division number="6" book="bu">

<Paragraph number="753" type="verses" book="bu">
_Sannipātā tayo āsuṃ,_  
_tasmimpi dvipaduttame;_  
_Khīṇāsavānaṃ vimalānaṃ,_  
_santacittāna tādinaṃ._  

</Paragraph>

</Division>
<Division number="7" book="bu">

<Paragraph number="754" type="verses" book="bu">
_Koṭisatānaṃ navutīnaṃ,_  
_asītiyāpi ca koṭinaṃ;_  
_Ete āsuṃ tayo ṭhānā,_  
_vimalānaṃ samāgame._  

</Paragraph>

</Division>
<Division number="8" book="bu">

<Paragraph number="755" type="verses" book="bu">
_Ahaṃ tena samayena,_  
_maṅgalo nāma tāpaso;_  
_Uggatejo duppasaho,_  
_abhiññābalasamāhito._  

</Paragraph>

</Division>
<Division number="9" book="bu">

<Paragraph number="756" type="verses" book="bu">
_Jambuto phalamānetvā,_  
_siddhatthassa adāsahaṃ;_  
_Paṭiggahetvā sambuddho,_  
_idaṃ vacanamabravi._  

</Paragraph>

</Division>
<Division number="10" book="bu">

<Paragraph number="757" type="verses" book="bu">
_‘Passatha imaṃ tāpasaṃ,_  
_jaṭilaṃ uggatāpanaṃ;_  
_Catunnavutito kappe,_  
_ayaṃ buddho bhavissati._  

</Paragraph>

</Division>
<Division number="11" book="bu">

<Paragraph number="758" type="verses" book="bu">
_Padhānaṃ padahitvāna,_  
_ …pe… _  
_hessāma sammukhā imaṃ’._  

</Paragraph>

</Division>
<Division number="12" book="bu">

<Paragraph number="759" type="verses" book="bu">
_Tassāpi vacanaṃ sutvā,_  
_bhiyyo cittaṃ pasādayiṃ;_  
_Uttariṃ vatamadhiṭṭhāsiṃ,_  
_dasapāramipūriyā._  

</Paragraph>

</Division>
<Division number="13" book="bu">

<Paragraph number="760" type="verses" book="bu">
_Vebhāraṃ nāma nagaraṃ,_  
_udeno nāma khattiyo;_  
_Suphassā nāma janikā,_  
_siddhatthassa mahesino._  

</Paragraph>

</Division>
<Division number="14" book="bu">

<Paragraph number="761" type="verses" book="bu">
_Dasavassasahassāni,_  
_agāraṃ ajjha so vasi;_  
_Kokāsuppalakokanadā,_  
_tayo pāsādamuttamā._  

</Paragraph>

</Division>
<Division number="15" book="bu">

<Paragraph number="762" type="verses" book="bu">
_Tisoḷasasahassāni,_  
_nāriyo samalaṅkatā;_  
_Somanassā nāma sā nārī,_  
_anūpamo nāma atrajo._  

</Paragraph>

</Division>
<Division number="16" book="bu">

<Paragraph number="763" type="verses" book="bu">
_Nimitte caturo disvā,_  
_sivikāyābhinikkhami;_  
_Anūnadasamāsāni,_  
_padhānaṃ padahī jino._  

</Paragraph>

</Division>
<Division number="17" book="bu">

<Paragraph number="764" type="verses" book="bu">
_Brahmunā yācito santo,_  
_siddhattho lokanāyako;_  
_Vatti cakkaṃ mahāvīro,_  
_migadāye naruttamo._  

</Paragraph>

</Division>
<Division number="18" book="bu">

<Paragraph number="765" type="verses" book="bu">
_Sambalo ca sumitto ca,_  
_ahesuṃ aggasāvakā;_  
_Revato nāmupaṭṭhāko,_  
_siddhatthassa mahesino._  

</Paragraph>

</Division>
<Division number="19" book="bu">

<Paragraph number="766" type="verses" book="bu">
_Sīvalā ca surāmā ca,_  
_ahesuṃ aggasāvikā;_  
_Bodhi tassa bhagavato,_  
_kaṇikāroti vuccati._  

</Paragraph>

</Division>
<Division number="20" book="bu">

<Paragraph number="767" type="verses" book="bu">
_Suppiyo ca samuddo ca,_  
_ahesuṃ aggupaṭṭhakā;_  
_Rammā ceva surammā ca,_  
_ahesuṃ aggupaṭṭhikā._  

</Paragraph>

</Division>
<Division number="21" book="bu">

<Paragraph number="768" type="verses" book="bu">
_So buddho saṭṭhiratanaṃ,_  
_ahosi nabhamuggato;_  
_Kañcanagghiyasaṅkāso,_  
_dasasahassī virocati._  

</Paragraph>

</Division>
<Division number="22" book="bu">

<Paragraph number="769" type="verses" book="bu">
_Sopi buddho asamasamo,_  
_atulo appaṭipuggalo;_  
_Vassasatasahassāni,_  
_loke aṭṭhāsi cakkhumā._  

</Paragraph>

</Division>
<Division number="23" book="bu">

<Paragraph number="770" type="verses" book="bu">
_Vipulaṃ pabhaṃ dassayitvā,_  
_pupphāpetvāna sāvake;_  
_Vilāsetvā samāpatyā,_  
_nibbuto so sasāvako._  

</Paragraph>

</Division>
<Division number="24" book="bu">

<Paragraph number="771" type="verses" book="bu">
_Siddhattho munivaro buddho,_  
_Anomārāmamhi nibbuto;_  
_Tatthevassa thūpavaro,_  
_Catuyojanamuggato”ti. (753)_  


</Paragraph>
</Division>

<Paragraph number="772" type="center" book="bu">
Siddhatthassa bhagavato vaṃso soḷasamo.
