_Nissaraṇampi ca phalaṃ upāyo ca;_

<Paragraph number="26" type="verses" book="ne">
_Yaṃ pucchitañca vissajjitañca,_  
_Suttassa yā ca anugīti;_  
_Suttassa yo pavicayo,_  
_Hāro <Emphasis>vicayo</Emphasis>ti niddiṭṭho._  

</Paragraph>

<Paragraph number="27" type="verses" book="ne">
_Sabbesaṃ hārānaṃ,_  
_Yā bhūmī yo ca gocaro tesaṃ;_  
_Yuttāyuttaparikkhā,_  
_Hāro <Emphasis>yuttī</Emphasis>ti niddiṭṭho._  

</Paragraph>

<Paragraph number="28" type="verses" book="ne">
_Dhammaṃ deseti jino,_  
_Tassa ca dhammassa yaṃ padaṭṭhānaṃ;_  
_Iti yāva sabbadhammā,_  
_Eso hāro <Emphasis>padaṭṭhāno</Emphasis>._  

</Paragraph>

<Paragraph number="29" type="verses" book="ne">
_Vuttamhi ekadhamme,_  
_Ye dhammā ekalakkhaṇā keci;_  
_Vuttā bhavanti sabbe,_  
_So hāro <Emphasis>lakkhaṇo</Emphasis> nāma._  

</Paragraph>

<Paragraph number="30" type="verses" book="ne">
_Neruttamadhippāyo,_  
_Byañjanamatha desanānidānañca;_  
_Pubbāparānusandhī,_  
_Eso hāro <Emphasis>catubyūho</Emphasis>._  

</Paragraph>

<Paragraph number="31" type="verses" book="ne">
_Ekamhi padaṭṭhāne,_  
_Pariyesati sesakaṃ padaṭṭhānaṃ;_  
_Āvaṭṭati paṭipakkhe,_  
_<Emphasis>Āvaṭṭo</Emphasis> nāma so hāro._  

</Paragraph>

<Paragraph number="32" type="verses" book="ne">
_Dhammañca padaṭṭhānaṃ,_  
_Bhūmiñca vibhajjate ayaṃ hāro;_  
_Sādhāraṇe asādhā-_  
_Raṇe ca neyyo <Emphasis>vibhattī</Emphasis>ti._  

</Paragraph>

<Paragraph number="33" type="verses" book="ne">
_Kusalākusale dhamme,_  
_Niddiṭṭhe bhāvite pahīne ca;_  
_Parivattati paṭipakkhe,_  
_Hāro <Emphasis>parivattano</Emphasis> nāma._  

</Paragraph>

<Paragraph number="34" type="verses" book="ne">
_Vevacanāni bahūni tu,_  
_Sutte vuttāni ekadhammassa;_  
_Yo jānāti suttavidū,_  
_<Emphasis>Vevacano</Emphasis> nāma so hāro._  

</Paragraph>

<Paragraph number="35" type="verses" book="ne">
_Ekaṃ bhagavā dhammaṃ,_  
_Paññattīhi vividhāhi deseti;_  
_So ākāro ñeyyo,_  
_<Emphasis>Paññattī</Emphasis> nāma hāroti._  

</Paragraph>

<Paragraph number="36" type="verses" book="ne">
_Yo ca paṭiccuppādo,_  
_Indriyakhandhā ca dhātu āyatanā;_  
_Etehi otarati yo,_  
_<Emphasis>Otaraṇo</Emphasis> nāma so hāro._  

</Paragraph>

<Paragraph number="37" type="verses" book="ne">
_Vissajjitamhi pañhe,_  
_Gāthāyaṃ pucchitā yamārabbha;_  
_Suddhāsuddhaparikkhā,_  
_Hāro so <Emphasis>sodhano</Emphasis> nāma._  

</Paragraph>

<Paragraph number="38" type="verses" book="ne">
_Ekattatāya dhammā,_  
_Yepi ca vemattatāya niddiṭṭhā;_  
_Te na vikappayitabbā,_  
_Eso hāro <Emphasis>adhiṭṭhāno</Emphasis>._  

</Paragraph>

<Paragraph number="39" type="verses" book="ne">
_Ye dhammā yaṃ dhammaṃ,_  
_Janayantippaccayā paramparato;_  
_Hetumavakaḍḍhayitvā,_  
_Eso hāro <Emphasis>parikkhāro</Emphasis>._  

</Paragraph>

<Paragraph number="40" type="verses" book="ne">
_Ye dhammā yaṃmūlā,_  
_Ye cekatthā pakāsitā muninā;_  
_Te samaropayitabbā,_  
_Esa <Emphasis>samāropano</Emphasis> hāro._
</Paragraph>
