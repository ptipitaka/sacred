<Division number="316" book="th1">

<Paragraph number="568" type="verses" book="th1">

_mataṃ disvāna pāpakaṃ;_  
_Kāmarāgo pāturahu,_  
_andhova savatī ahuṃ._  

</Paragraph>

</Division>
<Division number="317" book="th1">

<Paragraph number="569" type="verses" book="th1">

_tamhā ṭhānā apakkamiṃ;_  
_Satimā sampajānohaṃ,_  
_ekamantaṃ upāvisiṃ._  

</Paragraph>

</Division>
<Division number="318" book="th1">

<Paragraph number="570" type="verses" book="th1">

_yoniso udapajjatha;_  
_Ādīnavo pāturahu,_  
_nibbidā samatiṭṭhatha._  

</Paragraph>

</Division>

_kataṃ buddhassa sāsanan”ti._  


</Paragraph>
