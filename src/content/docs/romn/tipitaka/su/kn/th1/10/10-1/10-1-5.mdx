<Paragraph number="874" type="verses" book="th1">
_“Nānākulamalasampuṇṇo,_  
_mahāukkārasambhavo;_  
_Candanikaṃva paripakkaṃ,_  
_mahāgaṇḍo mahāvaṇo._  

</Paragraph>

</Division>
<Division number="568" book="th1" bookNo="19" page="304">

<Paragraph number="875" type="verses" book="th1">
_Pubbaruhirasampuṇṇo,_  
_gūthakūpena gāḷhito;_  
_Āpopaggharaṇo kāyo,_  
_sadā sandati pūtikaṃ._  

</Paragraph>

</Division>
<Division number="569" book="th1" bookNo="19" page="304">

<Paragraph number="876" type="verses" book="th1">
_Saṭṭhikaṇḍarasambandho,_  
_maṃsalepanalepito;_  
_Cammakañcukasannaddho,_  
_pūtikāyo niratthako._  

</Paragraph>

</Division>
<Division number="570" book="th1" bookNo="19" page="304">

<Paragraph number="877" type="verses" book="th1">
_Aṭṭhisaṅghātaghaṭito,_  
_nhārusuttanibandhano;_  
_Nekesaṃ saṅgatībhāvā,_  
_kappeti iriyāpathaṃ._  

</Paragraph>

</Division>
<Division number="571" book="th1" bookNo="19" page="304">

<Paragraph number="878" type="verses" book="th1">
_Dhuvappayāto maraṇāya,_  
_maccurājassa santike;_  
_Idheva chaḍḍayitvāna,_  
_yenakāmaṅgamo naro._  

</Paragraph>

</Division>
<Division number="572" book="th1" bookNo="19" page="304">

<Paragraph number="879" type="verses" book="th1">
_Avijjāya nivuto kāyo,_  
_catuganthena ganthito;_  
_Oghasaṃsīdano kāyo,_  
_anusayajālamotthato._  

</Paragraph>

</Division>
<Division number="573" book="th1" bookNo="19" page="304">

<Paragraph number="880" type="verses" book="th1">
_Pañcanīvaraṇe yutto,_  
_vitakkena samappito;_  
_Taṇhāmūlenānugato,_  
_mohacchādanachādito._  

</Paragraph>

</Division>
<Division number="574" book="th1" bookNo="19" page="304">

<Paragraph number="881" type="verses" book="th1">
_Evāyaṃ vattate kāyo,_  
_kammayantena yantito;_  
_Sampatti ca vipatyantā,_  
_nānābhāvo vipajjati._  

</Paragraph>

</Division>
<Division number="575" book="th1" bookNo="19" page="305">

<Paragraph number="882" type="verses" book="th1">
_Yemaṃ kāyaṃ mamāyanti,_  
_andhabālā puthujjanā;_  
_Vaḍḍhenti kaṭasiṃ ghoraṃ,_  
_ādiyanti punabbhavaṃ._  

</Paragraph>

</Division>
<Division number="576" book="th1" bookNo="19" page="305">

<Paragraph number="883" type="verses" book="th1">
_Yemaṃ kāyaṃ vivajjenti,_  
_Gūthalittaṃva pannagaṃ;_  
_Bhavamūlaṃ vamitvāna,_  
_Parinibbissantināsavā”ti._  


</Paragraph>
</Division>
