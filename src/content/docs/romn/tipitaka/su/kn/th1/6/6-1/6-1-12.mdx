<Paragraph number="720" type="verses" book="th1">
_“Akkodhassa kuto kodho,_  
_dantassa samajīvino;_  
_Sammadaññāvimuttassa,_  
_upasantassa tādino._  

</Paragraph>

</Division>
<Division number="442" book="th1" bookNo="19" page="288">

<Paragraph number="721" type="verses" book="th1">
_Tasseva tena pāpiyo,_  
_yo kuddhaṃ paṭikujjhati;_  
_Kuddhaṃ appaṭikujjhanto,_  
_saṅgāmaṃ jeti dujjayaṃ._  

</Paragraph>

</Division>
<Division number="443" book="th1" bookNo="19" page="288">

<Paragraph number="722" type="verses" book="th1">
_Ubhinnamatthaṃ carati,_  
_attano ca parassa ca;_  
_Paraṃ saṅkupitaṃ ñatvā,_  
_yo sato upasammati._  

</Paragraph>

</Division>
<Division number="444" book="th1" bookNo="19" page="288">

<Paragraph number="723" type="verses" book="th1">
_Ubhinnaṃ tikicchantaṃ taṃ,_  
_attano ca parassa ca;_  
_Janā maññanti bāloti,_  
_ye dhammassa akovidā._  

</Paragraph>

</Division>
<Division number="445" book="th1" bookNo="19" page="288">

<Paragraph number="724" type="verses" book="th1">
_Uppajje te sace kodho,_  
_āvajja kakacūpamaṃ;_  
_Uppajje ce rase taṇhā,_  
_puttamaṃsūpamaṃ sara._  

</Paragraph>

</Division>
<Division number="446" book="th1" bookNo="19" page="288">

<Paragraph number="725" type="verses" book="th1">
_Sace dhāvati cittaṃ te,_  
_kāmesu ca bhavesu ca;_  
_Khippaṃ niggaṇha satiyā,_  
_kiṭṭhādaṃ viya duppasun”ti._  


</Paragraph>
</Division>
