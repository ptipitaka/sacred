_“Yathācārī yathāsato satīmā,_  
_Yatasaṅkappajjhāyi appamatto;_  
_Ajjhattarato samāhitatto,_  
_Eko santusito tamāhu bhikkhuṃ._  

</Paragraph>

</Division>
<Division number="982" book="th1" bookNo="19" page="343">

<Paragraph number="1332" type="verses" book="th1">
_Allaṃ sukkhaṃ vā bhuñjanto,_  
_Na bāḷhaṃ suhito siyā;_  
_Ūnūdaro mitāhāro,_  
_Sato bhikkhu paribbaje._  

</Paragraph>

</Division>
<Division number="983" book="th1" bookNo="19" page="343">

<Paragraph number="1333" type="verses" book="th1">
_Cattāro pañca ālope,_  
_abhutvā udakaṃ pive;_  
_Alaṃ phāsuvihārāya,_  
_pahitattassa bhikkhuno._  

</Paragraph>

</Division>
<Division number="984" book="th1" bookNo="19" page="343">

<Paragraph number="1334" type="verses" book="th1">
_Kappiyaṃ tañce chādeti,_  
_cīvaraṃ idamatthikaṃ;_  
_Alaṃ phāsuvihārāya,_  
_pahitattassa bhikkhuno._  

</Paragraph>

</Division>
<Division number="985" book="th1" bookNo="19" page="344">

<Paragraph number="1335" type="verses" book="th1">
_Pallaṅkena nisinnassa,_  
_jaṇṇuke nābhivassati;_  
_Alaṃ phāsuvihārāya,_  
_pahitattassa bhikkhuno._  

</Paragraph>

</Division>
<Division number="986" book="th1" bookNo="19" page="344">

<Paragraph number="1336" type="verses" book="th1">
_Yo sukhaṃ dukkhato adda,_  
_dukkhamaddakkhi sallato;_  
_Ubhayantarena nāhosi,_  
_kena lokasmi kiṃ siyā._  

</Paragraph>

</Division>
<Division number="987" book="th1" bookNo="19" page="344">

<Paragraph number="1337" type="verses" book="th1">
_Mā me kadāci pāpiccho,_  
_kusīto hīnavīriyo;_  
_Appassuto anādaro,_  
_kena lokasmi kiṃ siyā._  

</Paragraph>

</Division>
<Division number="988" book="th1" bookNo="19" page="344">

<Paragraph number="1338" type="verses" book="th1">
_Bahussuto ca medhāvī,_  
_sīlesu susamāhito;_  
_Cetosamathamanuyutto,_  
_api muddhani tiṭṭhatu._  

</Paragraph>

</Division>
<Division number="989" book="th1" bookNo="19" page="344">

<Paragraph number="1339" type="verses" book="th1">
_Yo papañcamanuyutto,_  
_papañcābhirato mago;_  
_Virādhayī so nibbānaṃ,_  
_yogakkhemaṃ anuttaraṃ._  

</Paragraph>

</Division>
<Division number="990" book="th1" bookNo="19" page="344">

<Paragraph number="1340" type="verses" book="th1">
_Yo ca papañcaṃ hitvāna,_  
_nippapañcapathe rato;_  
_Ārādhayī so nibbānaṃ,_  
_yogakkhemaṃ anuttaraṃ._  

</Paragraph>

</Division>
<Division number="991" book="th1" bookNo="19" page="344">

<Paragraph number="1341" type="verses" book="th1">
_Gāme vā yadi vāraññe,_  
_Ninne vā yadi vā thale;_  
_Yattha arahanto viharanti,_  
_Taṃ bhūmirāmaṇeyyakaṃ._  

</Paragraph>

</Division>
<Division number="992" book="th1" bookNo="19" page="344">

<Paragraph number="1342" type="verses" book="th1">
_Ramaṇīyāni araññāni,_  
_yattha na ramatī jano;_  
_Vītarāgā ramissanti,_  
_na te kāmagavesino._  

</Paragraph>

</Division>
<Division number="993" book="th1" bookNo="19" page="344">

<Paragraph number="1343" type="verses" book="th1">
_Nidhīnaṃva pavattāraṃ,_  
_yaṃ passe vajjadassinaṃ;_  
_Niggayhavādiṃ medhāviṃ,_  
_tādisaṃ paṇḍitaṃ bhaje;_  
_Tādisaṃ bhajamānassa,_  
_seyyo hoti na pāpiyo._  

</Paragraph>

</Division>
<Division number="994" book="th1" bookNo="19" page="344">

<Paragraph number="1344" type="verses" book="th1">
_Ovadeyyānusāseyya,_  
_asabbhā ca nivāraye;_  
_Satañhi so piyo hoti,_  
_asataṃ hoti appiyo._  

</Paragraph>

</Division>
<Division number="995" book="th1" bookNo="19" page="344">

<Paragraph number="1345" type="verses" book="th1">
_Aññassa bhagavā buddho,_  
_dhammaṃ desesi cakkhumā;_  
_Dhamme desiyamānamhi,_  
_sotamodhesimatthiko;_  
_Taṃ me amoghaṃ savanaṃ,_  
_vimuttomhi anāsavo._  

</Paragraph>

</Division>
<Division number="996" book="th1" bookNo="19" page="344">

<Paragraph number="1346" type="verses" book="th1">
_Neva pubbenivāsāya,_  
_napi dibbassa cakkhuno;_  
_Cetopariyāya iddhiyā,_  
_cutiyā upapattiyā;_  
_Sotadhātuvisuddhiyā,_  
_paṇidhī me na vijjati”._  

</Paragraph>

</Division>
<Division number="997" book="th1" bookNo="19" page="345">

<Paragraph number="1347" type="verses" book="th1">
_“Rukkhamūlaṃva nissāya,_  
_muṇḍo saṅghāṭipāruto;_  
_Paññāya uttamo thero,_  
_upatissova jhāyati._  

</Paragraph>

</Division>
<Division number="998" book="th1" bookNo="19" page="345">

<Paragraph number="1348" type="verses" book="th1">
_Avitakkaṃ samāpanno,_  
_sammāsambuddhasāvako;_  
_Ariyena tuṇhībhāvena,_  
_upeto hoti tāvade._  

</Paragraph>

</Division>
<Division number="999" book="th1" bookNo="19" page="345">

<Paragraph number="1349" type="verses" book="th1">
_Yathāpi pabbato selo,_  
_acalo suppatiṭṭhito;_  
_Evaṃ mohakkhayā bhikkhu,_  
_pabbatova na vedhati”._  

</Paragraph>

</Division>
<Division number="1000" book="th1" bookNo="19" page="345">

<Paragraph number="1350" type="verses" book="th1">
_“Anaṅgaṇassa posassa,_  
_niccaṃ sucigavesino;_  
_Vālaggamattaṃ pāpassa,_  
_abbhamattaṃva khāyati._  

</Paragraph>

</Division>
<Division number="1001" book="th1" bookNo="19" page="345">

<Paragraph number="1351" type="verses" book="th1">
_Nābhinandāmi maraṇaṃ,_  
_nābhinandāmi jīvitaṃ;_  
_Nikkhipissaṃ imaṃ kāyaṃ,_  
_sampajāno patissato._  

</Paragraph>

</Division>
<Division number="1002" book="th1" bookNo="19" page="345">

<Paragraph number="1352" type="verses" book="th1">
_Nābhinandāmi maraṇaṃ,_  
_nābhinandāmi jīvitaṃ;_  
_Kālañca paṭikaṅkhāmi,_  
_nibbisaṃ bhatako yathā”._  

</Paragraph>

</Division>
<Division number="1003" book="th1" bookNo="19" page="345">

<Paragraph number="1353" type="verses" book="th1">
_“Ubhayena midaṃ maraṇameva,_  
_Nāmaraṇaṃ pacchā vā pure vā;_  
_Paṭipajjatha mā vinassatha,_  
_Khaṇo vo mā upaccagā._  

</Paragraph>

</Division>
<Division number="1004" book="th1" bookNo="19" page="345">

<Paragraph number="1354" type="verses" book="th1">
_Nagaraṃ yathā paccantaṃ,_  
_guttaṃ santarabāhiraṃ;_  
_Evaṃ gopetha attānaṃ,_  
_khaṇo vo mā upaccagā;_  
_Khaṇātītā hi socanti,_  
_nirayamhi samappitā”._  

</Paragraph>

</Division>
<Division number="1005" book="th1" bookNo="19" page="345">

_mantabhāṇī anuddhato;_  
_Dhunāti pāpake dhamme,_  
_dumapattaṃva māluto._  

</Paragraph>

</Division>
<Division number="1006" book="th1" bookNo="19" page="345">

_mantabhāṇī anuddhato;_  
_Appāsi pāpake dhamme,_  
_dumapattaṃva māluto._  

</Paragraph>

</Division>
<Division number="1007" book="th1" bookNo="19" page="345">

<Paragraph number="1357" type="verses" book="th1">

_vippasanno anāvilo;_  
_Kalyāṇasīlo medhāvī,_  
_dukkhassantakaro siyā”._  

</Paragraph>

</Division>
<Division number="1008" book="th1" bookNo="19" page="345">

<Paragraph number="1358" type="verses" book="th1">
_“Na vissase ekatiyesu evaṃ,_  
_Agārisu pabbajitesu cāpi;_  
_Sādhūpi hutvā na asādhu honti,_  
_Asādhu hutvā puna sādhu honti._  

</Paragraph>

</Division>
<Division number="1009" book="th1" bookNo="19" page="346">

<Paragraph number="1359" type="verses" book="th1">
_Kāmacchando ca byāpādo,_  
_thinamiddhañca bhikkhuno;_  
_Uddhaccaṃ vicikicchā ca,_  
_pañcete cittakelisā._  

</Paragraph>

</Division>
<Division number="1010" book="th1" bookNo="19" page="346">

<Paragraph number="1360" type="verses" book="th1">
_Yassa sakkariyamānassa,_  
_asakkārena cūbhayaṃ;_  
_Samādhi na vikampati,_  
_appamādavihārino._  

</Paragraph>

</Division>
<Division number="1011" book="th1" bookNo="19" page="346">

<Paragraph number="1361" type="verses" book="th1">
_Taṃ jhāyinaṃ sātatikaṃ,_  
_sukhumadiṭṭhivipassakaṃ;_  
_Upādānakkhayārāmaṃ,_  
_āhu sappuriso iti._  

</Paragraph>

</Division>
<Division number="1012" book="th1" bookNo="19" page="346">

<Paragraph number="1362" type="verses" book="th1">
_Mahāsamuddo pathavī,_  
_pabbato anilopi ca;_  
_Upamāya na yujjanti,_  
_satthu varavimuttiyā._  

</Paragraph>

</Division>

_mahāñāṇī samāhito;_  
_Pathavāpaggisamāno,_  
_na rajjati na dussati._  

</Paragraph>

</Division>

_mahābuddhi mahāmati;_  
_Ajaḷo jaḷasamāno,_  
_sadā carati nibbuto._  

</Paragraph>
