<Paragraph number="538" type="verses" book="th1">
_“Ubhayeneva sampanno,_  
_rāhulabhaddoti maṃ vidū;_  
_Yañcamhi putto buddhassa,_  
_yañca dhammesu cakkhumā._  

</Paragraph>

</Division>
<Division number="296" book="th1" bookNo="19" page="271">

<Paragraph number="539" type="verses" book="th1">
_Yañca me āsavā khīṇā,_  
_yañca natthi punabbhavo;_  
_Arahā dakkhiṇeyyomhi,_  
_tevijjo amataddaso._  

</Paragraph>

</Division>
<Division number="297" book="th1" bookNo="19" page="272">

<Paragraph number="540" type="verses" book="th1">
_Kāmandhā jālapacchannā,_  
_taṇhāchādanachāditā;_  
_Pamattabandhunā baddhā,_  
_macchāva kumināmukhe._  

</Paragraph>

</Division>
<Division number="298" book="th1" bookNo="19" page="272">

<Paragraph number="541" type="verses" book="th1">
_Taṃ kāmaṃ ahamujjhitvā,_  
_chetvā mārassa bandhanaṃ;_  
_Samūlaṃ taṇhamabbuyha,_  
_sītibhūtosmi nibbuto”ti._  


</Paragraph>
</Division>
