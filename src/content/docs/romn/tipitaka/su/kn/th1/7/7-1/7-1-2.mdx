<Paragraph number="752" type="verses" book="th1">
_“Pare ambāṭakārāme,_  
_vanasaṇḍamhi bhaddiyo;_  
_Samūlaṃ taṇhamabbuyha,_  
_tattha bhaddova jhāyati._  

</Paragraph>

</Division>
<Division number="467" book="th1" bookNo="19" page="291">

<Paragraph number="753" type="verses" book="th1">
_Ramanteke mudiṅgehi,_  
_vīṇāhi paṇavehi ca;_  
_Ahañca rukkhamūlasmiṃ,_  
_rato buddhassa sāsane._  

</Paragraph>

</Division>
<Division number="468" book="th1" bookNo="19" page="291">

<Paragraph number="754" type="verses" book="th1">
_Buddho ce me varaṃ dajjā,_  
_so ca labbhetha me varo;_  
_Gaṇhehaṃ sabbalokassa,_  
_niccaṃ kāyagataṃ satiṃ._  

</Paragraph>

</Division>
<Division number="469" book="th1" bookNo="19" page="292">

<Paragraph number="755" type="verses" book="th1">
_Ye maṃ rūpena pāmiṃsu,_  
_ye ca ghosena anvagū;_  
_Chandarāgavasūpetā,_  
_na maṃ jānanti te janā._  

</Paragraph>

</Division>
<Division number="470" book="th1" bookNo="19" page="292">

<Paragraph number="756" type="verses" book="th1">
_Ajjhattañca na jānāti,_  
_bahiddhā ca na passati;_  
_Samantāvaraṇo bālo,_  
_sa ve ghosena vuyhati._  

</Paragraph>

</Division>
<Division number="471" book="th1" bookNo="19" page="292">

<Paragraph number="757" type="verses" book="th1">
_Ajjhattañca na jānāti,_  
_bahiddhā ca vipassati;_  
_Bahiddhā phaladassāvī,_  
_sopi ghosena vuyhati._  

</Paragraph>

</Division>
<Division number="472" book="th1" bookNo="19" page="292">

<Paragraph number="758" type="verses" book="th1">
_Ajjhattañca pajānāti,_  
_bahiddhā ca vipassati;_  
_Anāvaraṇadassāvī,_  
_na so ghosena vuyhatī”ti._  


</Paragraph>
</Division>
