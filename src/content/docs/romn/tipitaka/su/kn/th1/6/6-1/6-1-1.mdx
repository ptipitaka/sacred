<Paragraph number="643" type="verses" book="th1">
_“Disvāna pāṭihīrāni,_  
_gotamassa yasassino;_  
_Na tāvāhaṃ paṇipatiṃ,_  
_issāmānena vañcito._  

</Paragraph>

</Division>
<Division number="376" book="th1" bookNo="19" page="282">

<Paragraph number="644" type="verses" book="th1">
_Mama saṅkappamaññāya,_  
_codesi narasārathi;_  
_Tato me āsi saṃvego,_  
_abbhuto lomahaṃsano._  

</Paragraph>

</Division>
<Division number="377" book="th1" bookNo="19" page="282">

<Paragraph number="645" type="verses" book="th1">
_Pubbe jaṭilabhūtassa,_  
_yā me siddhi parittikā;_  
_Tāhaṃ tadā nirākatvā,_  
_pabbajiṃ jinasāsane._  

</Paragraph>

</Division>
<Division number="378" book="th1" bookNo="19" page="282">

<Paragraph number="646" type="verses" book="th1">
_Pubbe yaññena santuṭṭho,_  
_kāmadhātupurakkhato;_  
_Pacchā rāgañca dosañca,_  
_mohañcāpi samūhaniṃ._  

</Paragraph>

</Division>
<Division number="379" book="th1" bookNo="19" page="282">

<Paragraph number="647" type="verses" book="th1">
_Pubbenivāsaṃ jānāmi,_  
_dibbacakkhu visodhitaṃ;_  
_Iddhimā paracittaññū,_  
_dibbasotañca pāpuṇiṃ._  

</Paragraph>

</Division>
<Division number="380" book="th1" bookNo="19" page="282">

<Paragraph number="648" type="verses" book="th1">
_Yassa catthāya pabbajito,_  
_agārasmānagāriyaṃ;_  
_So me attho anuppatto,_  
_sabbasaṃyojanakkhayo”ti._  


</Paragraph>
</Division>
