</Paragraph>

<Paragraph number="1375" type="verses" book="th1">
_Bahussuto cittakathī,_  
_buddhassa paricārako;_  
_Pannabhāro visaññutto,_  
_seyyaṃ kappeti gotamo._  

</Paragraph>

</Division>
<Division number="1025" book="th1" bookNo="19" page="347">

<Paragraph number="1376" type="verses" book="th1">
_Khīṇāsavo visaññutto,_  
_saṅgātīto sunibbuto;_  
_Dhāreti antimaṃ dehaṃ,_  
_jātimaraṇapāragū._  

</Paragraph>

</Division>
<Division number="1026" book="th1" bookNo="19" page="347">

<Paragraph number="1377" type="verses" book="th1">
_Yasmiṃ patiṭṭhitā dhammā,_  
_buddhassādiccabandhuno;_  
_Nibbānagamane magge,_  
_soyaṃ tiṭṭhati gotamo._  

</Paragraph>

</Division>
<Division number="1027" book="th1" bookNo="19" page="347">

<Paragraph number="1378" type="verses" book="th1">
_Dvāsīti buddhato gaṇhiṃ,_  
_dve sahassāni bhikkhuto;_  
_Caturāsītisahassāni,_  
_ye me dhammā pavattino._  

</Paragraph>

</Division>
<Division number="1028" book="th1" bookNo="19" page="347">

<Paragraph number="1379" type="verses" book="th1">
_Appassutāyaṃ puriso,_  
_balibaddova jīrati;_  
_Maṃsāni tassa vaḍḍhanti,_  
_paññā tassa na vaḍḍhati._  

</Paragraph>

</Division>
<Division number="1029" book="th1" bookNo="19" page="347">

<Paragraph number="1380" type="verses" book="th1">
_Bahussuto appassutaṃ,_  
_yo sutenātimaññati;_  
_Andho padīpadhārova,_  
_tatheva paṭibhāti maṃ._  

</Paragraph>

</Division>
<Division number="1030" book="th1" bookNo="19" page="347">

<Paragraph number="1381" type="verses" book="th1">
_Bahussutaṃ upāseyya,_  
_sutañca na vināsaye;_  
_Taṃ mūlaṃ brahmacariyassa,_  
_tasmā dhammadharo siyā._  

</Paragraph>

</Division>
<Division number="1031" book="th1" bookNo="19" page="347">

<Paragraph number="1382" type="verses" book="th1">
_Pubbāparaññū atthaññū,_  
_niruttipadakovido;_  
_Suggahītañca gaṇhāti,_  
_atthañcopaparikkhati._  

</Paragraph>

</Division>
<Division number="1032" book="th1" bookNo="19" page="347">

<Paragraph number="1383" type="verses" book="th1">
_Khantyā chandikato hoti,_  
_ussahitvā tuleti taṃ;_  
_Samaye so padahati,_  
_ajjhattaṃ susamāhito._  

</Paragraph>

</Division>
<Division number="1033" book="th1" bookNo="19" page="347">

<Paragraph number="1384" type="verses" book="th1">

_sappaññaṃ buddhasāvakaṃ;_  
_Dhammaviññāṇamākaṅkhaṃ,_  
_taṃ bhajetha tathāvidhaṃ._  

</Paragraph>

</Division>
<Division number="1034" book="th1" bookNo="19" page="347">

_kosārakkho mahesino;_  
_Cakkhu sabbassa lokassa,_  
_pūjanīyo bahussuto._  

</Paragraph>

</Division>
<Division number="1035" book="th1" bookNo="19" page="347">

_dhammaṃ anuvicintayaṃ;_  
_Dhammaṃ anussaraṃ bhikkhu,_  
_saddhammā na parihāyati._  

</Paragraph>

</Division>
<Division number="1036" book="th1" bookNo="19" page="348">

<Paragraph number="1387" type="verses" book="th1">
_Kāyamaccheragaruno,_  
_hiyyamāne anuṭṭhahe;_  
_Sarīrasukhagiddhassa,_  
_kuto samaṇaphāsutā._  

</Paragraph>

</Division>
<Division number="1037" book="th1" bookNo="19" page="348">

<Paragraph number="1388" type="verses" book="th1">

_dhammā na paṭibhanti maṃ;_  
_Gate kalyāṇamittamhi,_  
_andhakāraṃva khāyati._  

</Paragraph>

</Division>
<Division number="1038" book="th1" bookNo="19" page="348">

_atītagatasatthuno;_  
_Natthi etādisaṃ mittaṃ,_  
_yathā kāyagatā sati._  

</Paragraph>

</Division>

_navehi na sameti me;_  
_Svajja ekova jhāyāmi,_  
_vassupetova pakkhimā”._  

</Paragraph>

</Division>
