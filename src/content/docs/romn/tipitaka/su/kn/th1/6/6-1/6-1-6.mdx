<Paragraph number="678" type="verses" book="th1">
_“Paṇṇavīsativassāni,_  
_yato pabbajito ahaṃ;_  
_Accharāsaṅghātamattampi,_  
_cetosantimanajjhagaṃ._  

</Paragraph>

</Division>
<Division number="406" book="th1" bookNo="19" page="285">

<Paragraph number="679" type="verses" book="th1">
_Aladdhā cittassekaggaṃ,_  
_kāmarāgena aṭṭito;_  
_Bāhā paggayha kandanto,_  
_vihārā upanikkhamiṃ._  

</Paragraph>

</Division>
<Division number="407" book="th1" bookNo="19" page="285">

<Paragraph number="680" type="verses" book="th1">
_Satthaṃ vā āharissāmi,_  
_Ko attho jīvitena me;_  
_Kathaṃ hi sikkhaṃ paccakkhaṃ,_  
_Kālaṃ kubbetha mādiso._  

</Paragraph>

</Division>
<Division number="408" book="th1" bookNo="19" page="285">

<Paragraph number="681" type="verses" book="th1">
_Tadāhaṃ khuramādāya,_  
_mañcakamhi upāvisiṃ;_  
_Parinīto khuro āsi,_  
_dhamaniṃ chettumattano._  

</Paragraph>

</Division>
<Division number="409" book="th1" bookNo="19" page="285">

<Paragraph number="682" type="verses" book="th1">
_Tato me manasīkāro,_  
_yoniso udapajjatha;_  
_Ādīnavo pāturahu,_  
_nibbidā samatiṭṭhatha._  

</Paragraph>

</Division>
<Division number="410" book="th1" bookNo="19" page="285">

<Paragraph number="683" type="verses" book="th1">
_Tato cittaṃ vimucci me,_  
_passa dhammasudhammataṃ;_  
_Tisso vijjā anuppattā,_  
_kataṃ buddhassa sāsanan”ti._  


</Paragraph>
</Division>
