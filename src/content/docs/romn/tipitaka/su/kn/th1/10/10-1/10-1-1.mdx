<Paragraph number="830" type="verses" book="th1">
_“Aṅgārino dāni dumā bhadante,_  
_Phalesino chadanaṃ vippahāya;_  
_Te accimantova pabhāsayanti,_  
_Samayo mahāvīra bhāgī rasānaṃ._  

</Paragraph>

</Division>
<Division number="528" book="th1">

<Paragraph number="831" type="verses" book="th1">
_Dumāni phullāni manoramāni,_  
_Samantato sabbadisā pavanti;_  
_Pattaṃ pahāya phalamāsasānā,_  
_Kālo ito pakkamanāya vīra._  

</Paragraph>

</Division>
<Division number="529" book="th1">

<Paragraph number="832" type="verses" book="th1">
_Nevātisītaṃ na panātiuṇhaṃ,_  
_Sukhā utu addhaniyā bhadante;_  
_Passantu taṃ sākiyā koḷiyā ca,_  
_Pacchāmukhaṃ rohiniyaṃ tarantaṃ._  

</Paragraph>

</Division>
<Division number="530" book="th1">

<Paragraph number="833" type="verses" book="th1">
_Āsāya kasate khettaṃ,_  
_bījaṃ āsāya vappati;_  
_Āsāya vāṇijā yanti,_  
_samuddaṃ dhanahārakā;_  
_Yāya āsāya tiṭṭhāmi,_  
_sā me āsā samijjhatu._  

</Paragraph>

</Division>
<Division number="531" book="th1">

<Paragraph number="834" type="verses" book="th1">
_Punappunaṃ ceva vapanti bījaṃ,_  
_Punappunaṃ vassati devarājā;_  
_Punappunaṃ khettaṃ kasanti kassakā,_  
_Punappunaṃ dhaññamupeti raṭṭhaṃ._  

</Paragraph>

</Division>
<Division number="532" book="th1">

<Paragraph number="835" type="verses" book="th1">
_Punappunaṃ yācanakā caranti,_  
_Punappunaṃ dānapatī dadanti;_  
_Punappunaṃ dānapatī daditvā,_  
_Punappunaṃ saggamupenti ṭhānaṃ._  

</Paragraph>

</Division>
<Division number="533" book="th1">

<Paragraph number="836" type="verses" book="th1">
_Vīro have sattayugaṃ puneti,_  
_Yasmiṃ kule jāyati bhūripañño;_  
_Maññāmahaṃ sakkati devadevo,_  
_Tayā hi jāto muni saccanāmo._  

</Paragraph>

</Division>
<Division number="534" book="th1">

<Paragraph number="837" type="verses" book="th1">
_Suddhodano nāma pitā mahesino,_  
_Buddhassa mātā pana māyanāmā;_  
_Yā bodhisattaṃ parihariya kucchinā,_  
_Kāyassa bhedā tidivamhi modati._  

</Paragraph>

</Division>
<Division number="535" book="th1">

<Paragraph number="838" type="verses" book="th1">
_Sā gotamī kālakatā ito cutā,_  
_Dibbehi kāmehi samaṅgibhūtā;_  
_Sā modati kāmaguṇehi pañcahi,_  
_Parivāritā devagaṇehi tehi”._  

</Paragraph>

</Division>
<Division number="536" book="th1">

<Paragraph number="839" type="verses" book="th1">
_“Buddhassa puttomhi asayhasāhino,_  
_Aṅgīrasassappaṭimassa tādino;_  
_Pitupitā mayhaṃ tuvaṃsi sakka,_  
_Dhammena me gotama ayyakosī”ti._  


</Paragraph>
</Division>
