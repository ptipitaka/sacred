<Paragraph number="841" type="verses" book="th1">
_“Purato pacchato vāpi,_  
_aparo ce na vijjati;_  
_Atīva phāsu bhavati,_  
_ekassa vasato vane._  

</Paragraph>

</Division>
<Division number="538" book="th1" bookNo="19" page="301">

<Paragraph number="842" type="verses" book="th1">
_Handa eko gamissāmi,_  
_araññaṃ buddhavaṇṇitaṃ;_  
_Phāsu ekavihārissa,_  
_pahitattassa bhikkhuno._  

</Paragraph>

</Division>
<Division number="539" book="th1" bookNo="19" page="301">

<Paragraph number="843" type="verses" book="th1">
_Yogi-pītikaraṃ rammaṃ,_  
_mattakuñjarasevitaṃ;_  
_Eko attavasī khippaṃ,_  
_pavisissāmi kānanaṃ._  

</Paragraph>

</Division>
<Division number="540" book="th1" bookNo="19" page="301">

<Paragraph number="844" type="verses" book="th1">
_Supupphite sītavane,_  
_sītale girikandare;_  
_Gattāni parisiñcitvā,_  
_caṅkamissāmi ekako._  

</Paragraph>

</Division>
<Division number="541" book="th1" bookNo="19" page="301">

<Paragraph number="845" type="verses" book="th1">
_Ekākiyo adutiyo,_  
_ramaṇīye mahāvane;_  
_Kadāhaṃ viharissāmi,_  
_katakicco anāsavo._  

</Paragraph>

</Division>
<Division number="542" book="th1" bookNo="19" page="301">

<Paragraph number="846" type="verses" book="th1">
_Evaṃ me kattukāmassa,_  
_adhippāyo samijjhatu;_  
_Sādhayissāmahaṃyeva,_  
_nāñño aññassa kārako._  

</Paragraph>

</Division>
<Division number="543" book="th1" bookNo="19" page="302">

<Paragraph number="847" type="verses" book="th1">
_Esa bandhāmi sannāhaṃ,_  
_pavisissāmi kānanaṃ;_  
_Na tato nikkhamissāmi,_  
_appatto āsavakkhayaṃ._  

</Paragraph>

</Division>
<Division number="544" book="th1" bookNo="19" page="302">

<Paragraph number="848" type="verses" book="th1">
_Mālute upavāyante,_  
_sīte surabhigandhike;_  
_Avijjaṃ dālayissāmi,_  
_nisinno nagamuddhani._  

</Paragraph>

</Division>
<Division number="545" book="th1" bookNo="19" page="302">

<Paragraph number="849" type="verses" book="th1">
_Vane kusumasañchanne,_  
_pabbhāre nūna sītale;_  
_Vimuttisukhena sukhito,_  
_ramissāmi giribbaje._  

</Paragraph>

</Division>
<Division number="546" book="th1" bookNo="19" page="302">

<Paragraph number="850" type="verses" book="th1">
_Sohaṃ paripuṇṇasaṅkappo,_  
_cando pannaraso yathā;_  
_Sabbāsavaparikkhīṇo,_  
_natthi dāni punabbhavo”ti._  


</Paragraph>
</Division>

…  Ekavihāriyo thero… .
