<Division number="920" book="th1">

<Paragraph number="1265" type="verses" book="th1">
_Samaṇassa ahu cintā,_  
_pupphitamhi mahāvane;_  
_Ekaggassa nisinnassa,_  
_pavivittassa jhāyino._  

</Paragraph>

</Division>
<Division number="921" book="th1">

<Paragraph number="1266" type="verses" book="th1">
_“Aññathā lokanāthamhi,_  
_tiṭṭhante purisuttame;_  
_Iriyaṃ āsi bhikkhūnaṃ,_  
_aññathā dāni dissati._  

</Paragraph>

</Division>
<Division number="922" book="th1">

<Paragraph number="1267" type="verses" book="th1">
_Sītavātaparittāṇaṃ,_  
_hirikopīnachādanaṃ;_  
_Mattaṭṭhiyaṃ abhuñjiṃsu,_  
_santuṭṭhā itarītare._  

</Paragraph>

</Division>
<Division number="923" book="th1">

<Paragraph number="1268" type="verses" book="th1">
_Paṇītaṃ yadi vā lūkhaṃ,_  
_appaṃ vā yadi vā bahuṃ;_  
_Yāpanatthaṃ abhuñjiṃsu,_  
_agiddhā nādhimucchitā._  

</Paragraph>

</Division>
<Division number="924" book="th1">

<Paragraph number="1269" type="verses" book="th1">
_Jīvitānaṃ parikkhāre,_  
_bhesajje atha paccaye;_  
_Na bāḷhaṃ ussukā āsuṃ,_  
_yathā te āsavakkhaye._  

</Paragraph>

</Division>
<Division number="925" book="th1">

<Paragraph number="1270" type="verses" book="th1">
_Araññe rukkhamūlesu,_  
_kandarāsu guhāsu ca;_  
_Vivekamanubrūhantā,_  
_vihaṃsu tapparāyaṇā._  

</Paragraph>

</Division>
<Division number="926" book="th1">

<Paragraph number="1271" type="verses" book="th1">
_Nīcā niviṭṭhā subharā,_  
_mudū atthaddhamānasā;_  
_Abyāsekā amukharā,_  
_atthacintā vasānugā._  

</Paragraph>

</Division>
<Division number="927" book="th1">

<Paragraph number="1272" type="verses" book="th1">
_Tato pāsādikaṃ āsi,_  
_gataṃ bhuttaṃ nisevitaṃ;_  
_Siniddhā teladhārāva,_  
_ahosi iriyāpatho._  

</Paragraph>

</Division>
<Division number="928" book="th1">

<Paragraph number="1273" type="verses" book="th1">
_Sabbāsavaparikkhīṇā,_  
_mahājhāyī mahāhitā;_  
_Nibbutā dāni te therā,_  
_parittā dāni tādisā._  

</Paragraph>

</Division>
<Division number="929" book="th1">

<Paragraph number="1274" type="verses" book="th1">
_Kusalānañca dhammānaṃ,_  
_paññāya ca parikkhayā;_  
_Sabbākāravarūpetaṃ,_  
_lujjate jinasāsanaṃ._  

</Paragraph>

</Division>
<Division number="930" book="th1">

<Paragraph number="1275" type="verses" book="th1">
_Pāpakānañca dhammānaṃ,_  
_kilesānañca yo utu;_  
_Upaṭṭhitā vivekāya,_  
_ye ca saddhammasesakā._  

</Paragraph>

</Division>
<Division number="931" book="th1">

<Paragraph number="1276" type="verses" book="th1">
_Te kilesā pavaḍḍhantā,_  
_āvisanti bahuṃ janaṃ;_  
_Kīḷanti maññe bālehi,_  
_ummattehiva rakkhasā._  

</Paragraph>

</Division>
<Division number="932" book="th1">

<Paragraph number="1277" type="verses" book="th1">
_Kilesehābhibhūtā te,_  
_tena tena vidhāvitā;_  
_Narā kilesavatthūsu,_  
_sasaṅgāmeva ghosite._  

</Paragraph>

</Division>
<Division number="933" book="th1">

<Paragraph number="1278" type="verses" book="th1">
_Pariccajitvā saddhammaṃ,_  
_aññamaññehi bhaṇḍare;_  
_Diṭṭhigatāni anventā,_  
_idaṃ seyyoti maññare._  

</Paragraph>

</Division>
<Division number="934" book="th1">

<Paragraph number="1279" type="verses" book="th1">
_Dhanañca puttaṃ bhariyañca,_  
_chaḍḍayitvāna niggatā;_  
_Kaṭacchubhikkhahetūpi,_  
_akicchāni nisevare._  

</Paragraph>

</Division>
<Division number="935" book="th1">

<Paragraph number="1280" type="verses" book="th1">
_Udarāvadehakaṃ bhutvā,_  
_sayantuttānaseyyakā;_  
_Kathaṃ vattenti paṭibuddhā,_  
_yā kathā satthugarahitā._  

</Paragraph>

</Division>
<Division number="936" book="th1">

<Paragraph number="1281" type="verses" book="th1">
_Sabbakārukasippāni,_  
_cittiṃ katvāna sikkhare;_  
_Avūpasantā ajjhattaṃ,_  
_sāmaññatthotiacchati._  

</Paragraph>

</Division>
<Division number="937" book="th1">

<Paragraph number="1282" type="verses" book="th1">
_Mattikaṃ telacuṇṇañca,_  
_udakāsanabhojanaṃ;_  
_Gihīnaṃ upanāmenti,_  
_ākaṅkhantā bahuttaraṃ._  

</Paragraph>

</Division>
<Division number="938" book="th1">

<Paragraph number="1283" type="verses" book="th1">
_Dantaponaṃ kapitthañca,_  
_pupphaṃ khādaniyāni ca;_  
_Piṇḍapāte ca sampanne,_  
_ambe āmalakāni ca._  

</Paragraph>

</Division>
<Division number="939" book="th1">

<Paragraph number="1284" type="verses" book="th1">
_Bhesajjesu yathā vejjā,_  
_kiccākicce yathā gihī;_  
_Gaṇikāva vibhūsāyaṃ,_  
_issare khattiyā yathā._  

</Paragraph>

</Division>
<Division number="940" book="th1">

<Paragraph number="1285" type="verses" book="th1">
_Nekatikā vañcanikā,_  
_kūṭasakkhī apāṭukā;_  
_Bahūhi parikappehi,_  
_āmisaṃ paribhuñjare._  

</Paragraph>

</Division>
<Division number="941" book="th1">

<Paragraph number="1286" type="verses" book="th1">
_Lesakappe pariyāye,_  
_parikappenudhāvitā;_  
_Jīvikatthā upāyena,_  
_saṅkaḍḍhanti bahuṃ dhanaṃ._  

</Paragraph>

</Division>
<Division number="942" book="th1">

<Paragraph number="1287" type="verses" book="th1">
_Upaṭṭhāpenti parisaṃ,_  
_kammato no ca dhammato;_  
_Dhammaṃ paresaṃ desenti,_  
_lābhato no ca atthato._  

</Paragraph>

</Division>
<Division number="943" book="th1">

<Paragraph number="1288" type="verses" book="th1">
_Saṃghalābhassa bhaṇḍanti,_  
_saṃghato paribāhirā;_  
_Paralābhopajīvantā,_  
_ahirīkā na lajjare._  

</Paragraph>

</Division>
<Division number="944" book="th1">

<Paragraph number="1289" type="verses" book="th1">
_Nānuyuttā tathā eke,_  
_muṇḍā saṅghāṭipārutā;_  
_Sambhāvanaṃyevicchanti,_  
_lābhasakkāramucchitā._  

</Paragraph>

</Division>
<Division number="945" book="th1">

<Paragraph number="1290" type="verses" book="th1">
_Evaṃ nānappayātamhi,_  
_na dāni sukaraṃ tathā;_  
_Aphusitaṃ vā phusituṃ,_  
_phusitaṃ vānurakkhituṃ._  

</Paragraph>

</Division>
<Division number="946" book="th1">

<Paragraph number="1291" type="verses" book="th1">
_Yathā kaṇṭakaṭṭhānamhi,_  
_careyya anupāhano;_  
_Satiṃ upaṭṭhapetvāna,_  
_evaṃ gāme munī care._  

</Paragraph>

</Division>
<Division number="947" book="th1">

<Paragraph number="1292" type="verses" book="th1">
_Saritvā pubbake yogī,_  
_tesaṃ vattamanussaraṃ;_  
_Kiñcāpi pacchimo kālo,_  
_phuseyya amataṃ padaṃ._  

</Paragraph>

</Division>
<Division number="948" book="th1">

<Paragraph number="1293" type="verses" book="th1">
_Idaṃ vatvā sālavane,_  
_samaṇo bhāvitindriyo;_  
_Brāhmaṇo parinibbāyī,_  
_isi khīṇapunabbhavo”ti._  


</Paragraph>
</Division>

…  Pārāpariyo thero… .



</Paragraph>

_telakāni raṭṭhapālo;_  
_Mālukyaselo bhaddiyo,_  
_aṅguli dibbacakkhuko;_  
_Pārāpariyo dasete,_  
_vīsamhi parikittitā;_  
_Gāthāyo dve satā honti,_  
_pañcatālīsa uttarinti._
