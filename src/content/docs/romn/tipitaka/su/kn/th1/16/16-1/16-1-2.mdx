<Paragraph number="1063" type="verses" book="th1">
_“Samaṇassa ahu cintā,_  
_pārāpariyassa bhikkhuno;_  
_Ekakassa nisinnassa,_  
_pavivittassa jhāyino._  

</Paragraph>

</Division>
<Division number="727" book="th1" bookNo="19" page="321">

<Paragraph number="1064" type="verses" book="th1">
_Kimānupubbaṃ puriso,_  
_kiṃ vataṃ kiṃ samācāraṃ;_  
_Attano kiccakārīssa,_  
_na ca kañci viheṭhaye._  

</Paragraph>

</Division>
<Division number="728" book="th1" bookNo="19" page="322">

<Paragraph number="1065" type="verses" book="th1">
_Indriyāni manussānaṃ,_  
_hitāya ahitāya ca;_  
_Arakkhitāni ahitāya,_  
_rakkhitāni hitāya ca._  

</Paragraph>

</Division>
<Division number="729" book="th1" bookNo="19" page="322">

<Paragraph number="1066" type="verses" book="th1">
_Indriyāneva sārakkhaṃ,_  
_indriyāni ca gopayaṃ;_  
_Attano kiccakārīssa,_  
_na ca kañci viheṭhaye._  

</Paragraph>

</Division>
<Division number="730" book="th1" bookNo="19" page="322">

<Paragraph number="1067" type="verses" book="th1">
_Cakkhundriyañce rūpesu,_  
_gacchantaṃ anivārayaṃ;_  
_Anādīnavadassāvī,_  
_so dukkhā na hi muccati._  

</Paragraph>

</Division>
<Division number="731" book="th1" bookNo="19" page="322">

<Paragraph number="1068" type="verses" book="th1">
_Sotindriyañce saddesu,_  
_gacchantaṃ anivārayaṃ;_  
_Anādīnavadassāvī,_  
_so dukkhā na hi muccati._  

</Paragraph>

</Division>
<Division number="732" book="th1" bookNo="19" page="322">

<Paragraph number="1069" type="verses" book="th1">
_Anissaraṇadassāvī,_  
_gandhe ce paṭisevati;_  
_Na so muccati dukkhamhā,_  
_gandhesu adhimucchito._  

</Paragraph>

</Division>
<Division number="733" book="th1" bookNo="19" page="322">

<Paragraph number="1070" type="verses" book="th1">
_Ambilaṃ madhuraggañca,_  
_tittakaggamanussaraṃ;_  
_Rasataṇhāya gadhito,_  
_hadayaṃ nāvabujjhati._  

</Paragraph>

</Division>
<Division number="734" book="th1" bookNo="19" page="322">

<Paragraph number="1071" type="verses" book="th1">
_Subhānyappaṭikūlāni,_  
_phoṭṭhabbāni anussaraṃ;_  
_Ratto rāgādhikaraṇaṃ,_  
_vividhaṃ vindate dukhaṃ._  

</Paragraph>

</Division>
<Division number="735" book="th1" bookNo="19" page="322">

<Paragraph number="1072" type="verses" book="th1">
_Manaṃ cetehi dhammehi,_  
_yo na sakkoti rakkhituṃ;_  
_Tato naṃ dukkhamanveti,_  
_sabbehetehi pañcahi._  

</Paragraph>

</Division>
<Division number="736" book="th1" bookNo="19" page="322">

<Paragraph number="1073" type="verses" book="th1">
_Pubbalohitasampuṇṇaṃ,_  
_bahussa kuṇapassa ca;_  
_Naravīrakataṃ vagguṃ,_  
_samuggamiva cittitaṃ._  

</Paragraph>

</Division>
<Division number="737" book="th1" bookNo="19" page="322">

<Paragraph number="1074" type="verses" book="th1">
_Kaṭukaṃ madhurassādaṃ,_  
_piyanibandhanaṃ dukhaṃ;_  
_Khuraṃva madhunā littaṃ,_  
_ullihaṃ nāvabujjhati._  

</Paragraph>

</Division>
<Division number="738" book="th1" bookNo="19" page="322">

<Paragraph number="1075" type="verses" book="th1">
_Itthirūpe itthisare,_  
_phoṭṭhabbepi ca itthiyā;_  
_Itthigandhesu sāratto,_  
_vividhaṃ vindate dukhaṃ._  

</Paragraph>

</Division>
<Division number="739" book="th1" bookNo="19" page="322">

<Paragraph number="1076" type="verses" book="th1">
_Itthisotāni sabbāni,_  
_sandanti pañca pañcasu;_  
_Tesamāvaraṇaṃ kātuṃ,_  
_yo sakkoti vīriyavā._  

</Paragraph>

</Division>
<Division number="740" book="th1" bookNo="19" page="322">

<Paragraph number="1077" type="verses" book="th1">
_So atthavā so dhammaṭṭho,_  
_so dakkho so vicakkhaṇo;_  
_Kareyya ramamānopi,_  
_kiccaṃ dhammatthasaṃhitaṃ._  

</Paragraph>

</Division>
<Division number="741" book="th1" bookNo="19" page="323">

<Paragraph number="1078" type="verses" book="th1">
_Atho sīdati saññuttaṃ,_  
_vajje kiccaṃ niratthakaṃ;_  
_‘Na taṃ kiccan’ti maññitvā,_  
_appamatto vicakkhaṇo._  

</Paragraph>

</Division>
<Division number="742" book="th1" bookNo="19" page="323">

<Paragraph number="1079" type="verses" book="th1">
_Yañca atthena saññuttaṃ,_  
_yā ca dhammagatā rati;_  
_Taṃ samādāya vattetha,_  
_sā hi ve uttamā rati._  

</Paragraph>

</Division>
<Division number="743" book="th1" bookNo="19" page="323">

<Paragraph number="1080" type="verses" book="th1">
_Uccāvacehupāyehi,_  
_Paresamabhijigīsati;_  
_Hantvā vadhitvā atha socayitvā,_  
_Ālopati sāhasā yo paresaṃ._  

</Paragraph>

</Division>
<Division number="744" book="th1" bookNo="19" page="323">

<Paragraph number="1081" type="verses" book="th1">
_Tacchanto āṇiyā āṇiṃ,_  
_nihanti balavā yathā;_  
_Indriyānindriyeheva,_  
_nihanti kusalo tathā._  

</Paragraph>

</Division>
<Division number="745" book="th1" bookNo="19" page="323">

<Paragraph number="1082" type="verses" book="th1">
_Saddhaṃ vīriyaṃ samādhiñca,_  
_satipaññañca bhāvayaṃ;_  
_Pañca pañcahi hantvāna,_  
_anīgho yāti brāhmaṇo._  

</Paragraph>

</Division>
<Division number="746" book="th1" bookNo="19" page="323">

<Paragraph number="1083" type="verses" book="th1">
_So atthavā so dhammaṭṭho,_  
_katvā vākyānusāsaniṃ;_  
_Sabbena sabbaṃ buddhassa,_  
_so naro sukhamedhatī”ti._  


</Paragraph>
</Division>

<Paragraph number="1084" type="center" book="th1">
… Pārāpariyo thero… .
</Paragraph>
