<Paragraph number="1159" type="verses" book="th1">
_“Paripuṇṇakāyo suruci,_  
_Sujāto cārudassano;_  
_Suvaṇṇavaṇṇosi bhagavā,_  
_Susukkadāṭhosi vīriyavā._  

</Paragraph>

</Division>
<Division number="819" book="th1">

<Paragraph number="1160" type="verses" book="th1">
_Narassa hi sujātassa,_  
_ye bhavanti viyañjanā;_  
_Sabbe te tava kāyasmiṃ,_  
_mahāpurisalakkhaṇā._  

</Paragraph>

</Division>
<Division number="820" book="th1">

<Paragraph number="1161" type="verses" book="th1">
_Pasannanetto sumukho,_  
_brahā uju patāpavā;_  
_Majjhe samaṇasaṃghassa,_  
_ādiccova virocasi._  

</Paragraph>

</Division>
<Division number="821" book="th1">

<Paragraph number="1162" type="verses" book="th1">
_Kalyāṇadassano bhikkhu,_  
_kañcanasannibhattaco;_  
_Kiṃ te samaṇabhāvena,_  
_evaṃ uttamavaṇṇino._  

</Paragraph>

</Division>
<Division number="822" book="th1">

<Paragraph number="1163" type="verses" book="th1">
_Rājā arahasi bhavituṃ,_  
_cakkavattī rathesabho;_  
_Cāturanto vijitāvī,_  
_jambusaṇḍassa issaro._  

</Paragraph>

</Division>
<Division number="823" book="th1">

<Paragraph number="1164" type="verses" book="th1">
_Khattiyā bhogī rājāno,_  
_anuyantā bhavanti te;_  
_Rājābhirājā manujindo,_  
_rajjaṃ kārehi gotama”._  

</Paragraph>

</Division>
<Division number="824" book="th1">

<Paragraph number="1165" type="verses" book="th1">
_“Rājāhamasmi sela, __(selāti bhagavā)___  
_Dhammarājā anuttaro;_  
_Dhammena cakkaṃ vattemi,_  
_Cakkaṃ appaṭivattiyaṃ”._  

</Paragraph>

</Division>
<Division number="825" book="th1">

<Paragraph number="1166" type="verses" book="th1">
_“Sambuddho paṭijānāsi, __(iti selo brāhmaṇo)___  
_Dhammarājā anuttaro;_  
_‘Dhammena cakkaṃ vattemi’,_  
_Iti bhāsatha gotama._  

</Paragraph>

</Division>
<Division number="826" book="th1">

<Paragraph number="1167" type="verses" book="th1">
_Ko nu senāpati bhoto,_  
_sāvako satthuranvayo;_  
_Ko tetamanuvatteti,_  
_dhammacakkaṃ pavattitaṃ”._  

</Paragraph>

</Division>
<Division number="827" book="th1">

<Paragraph number="1168" type="verses" book="th1">
_“Mayā pavattitaṃ cakkaṃ, __(selāti bhagavā)___  
_Dhammacakkaṃ anuttaraṃ;_  
_Sāriputto anuvatteti,_  
_Anujāto tathāgataṃ._  

</Paragraph>

</Division>
<Division number="828" book="th1">

<Paragraph number="1169" type="verses" book="th1">
_Abhiññeyyaṃ abhiññātaṃ,_  
_bhāvetabbañca bhāvitaṃ;_  
_Pahātabbaṃ pahīnaṃ me,_  
_tasmā buddhosmi brāhmaṇa._  

</Paragraph>

</Division>
<Division number="829" book="th1">

<Paragraph number="1170" type="verses" book="th1">
_Vinayassu mayi kaṅkhaṃ,_  
_adhimuccassu brāhmaṇa;_  
_Dullabhaṃ dassanaṃ hoti,_  
_sambuddhānaṃ abhiṇhaso._  

</Paragraph>

</Division>
<Division number="830" book="th1">

<Paragraph number="1171" type="verses" book="th1">
_Yesaṃ ve dullabho loke,_  
_pātubhāvo abhiṇhaso;_  
_Sohaṃ brāhmaṇa buddhosmi,_  
_sallakatto anuttaro._  

</Paragraph>

</Division>
<Division number="831" book="th1">

<Paragraph number="1172" type="verses" book="th1">
_Brahmabhūto atitulo,_  
_mārasenappamaddano;_  
_Sabbāmitte vase katvā,_  
_modāmi akutobhayo”._  

</Paragraph>

</Division>
<Division number="832" book="th1">

<Paragraph number="1173" type="verses" book="th1">
_“Idaṃ bhonto nisāmetha,_  
_yathā bhāsati cakkhumā;_  
_Sallakatto mahāvīro,_  
_sīhova nadatī vane._  

</Paragraph>

</Division>
<Division number="833" book="th1">

<Paragraph number="1174" type="verses" book="th1">
_Brahmabhūtaṃ atitulaṃ,_  
_mārasenappamaddanaṃ;_  
_Ko disvā nappasīdeyya,_  
_api kaṇhābhijātiko._  

</Paragraph>

</Division>
<Division number="834" book="th1">

<Paragraph number="1175" type="verses" book="th1">
_Yo maṃ icchati anvetu,_  
_yo vā nicchati gacchatu;_  
_Idhāhaṃ pabbajissāmi,_  
_varapaññassa santike”._  

</Paragraph>

</Division>
<Division number="835" book="th1">

<Paragraph number="1176" type="verses" book="th1">
_“Etañce ruccati bhoto,_  
_sammāsambuddhasāsanaṃ;_  
_Mayampi pabbajissāma,_  
_varapaññassa santike”._  

</Paragraph>

</Division>
<Division number="836" book="th1">

<Paragraph number="1177" type="verses" book="th1">
_“Brāhmaṇā tisatā ime,_  
_yācanti pañjalīkatā;_  
_‘Brahmacariyaṃ carissāma,_  
_bhagavā tava santike’”._  

</Paragraph>

</Division>
<Division number="837" book="th1">

<Paragraph number="1178" type="verses" book="th1">
_“Svākkhātaṃ brahmacariyaṃ, __(selāti bhagavā)___  
_Sandiṭṭhikamakālikaṃ;_  
_Yattha amoghā pabbajjā,_  
_Appamattassa sikkhato”._  

</Paragraph>

</Division>
<Division number="838" book="th1">

<Paragraph number="1179" type="verses" book="th1">
_“Yaṃ taṃ saraṇamāgamha,_  
_ito aṭṭhame cakkhuma;_  
_Sattarattena bhagavā,_  
_dantāmha tava sāsane._  

</Paragraph>

</Division>
<Division number="839" book="th1">

<Paragraph number="1180" type="verses" book="th1">
_Tuvaṃ buddho tuvaṃ satthā,_  
_tuvaṃ mārābhibhū muni;_  
_Tuvaṃ anusaye chetvā,_  
_tiṇṇo tāresimaṃ pajaṃ._  

</Paragraph>

</Division>
<Division number="840" book="th1">

<Paragraph number="1181" type="verses" book="th1">
_Upadhī te samatikkantā,_  
_āsavā te padālitā;_  
_Sīhova anupādāno,_  
_pahīnabhayabheravo._  

</Paragraph>

</Division>
<Division number="841" book="th1">

<Paragraph number="1182" type="verses" book="th1">
_Bhikkhavo tisatā ime,_  
_tiṭṭhanti pañjalīkatā;_  
_Pāde vīra pasārehi,_  
_nāgā vandantu satthuno”ti._  


</Paragraph>
</Division>
