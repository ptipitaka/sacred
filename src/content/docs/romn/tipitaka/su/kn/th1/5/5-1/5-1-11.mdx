<Paragraph number="627" type="verses" book="th1">
_“Upasampadā ca me laddhā,_  
_vimutto camhi anāsavo;_  
_So ca me bhagavā diṭṭho,_  
_vihāre ca sahāvasiṃ._  

</Paragraph>

</Division>
<Division number="366" book="th1" bookNo="19" page="280">

<Paragraph number="628" type="verses" book="th1">
_Bahudeva rattiṃ bhagavā,_  
_abbhokāsetināmayi;_  
_Vihārakusalo satthā,_  
_vihāraṃ pāvisī tadā._  

</Paragraph>

</Division>
<Division number="367" book="th1" bookNo="19" page="280">

<Paragraph number="629" type="verses" book="th1">
_Santharitvāna saṅghāṭiṃ,_  
_seyyaṃ kappesi gotamo;_  
_Sīho selaguhāyaṃva,_  
_pahīnabhayabheravo._  

</Paragraph>

</Division>
<Division number="368" book="th1" bookNo="19" page="280">

<Paragraph number="630" type="verses" book="th1">
_Tato kalyāṇavākkaraṇo,_  
_sammāsambuddhasāvako;_  
_Soṇo abhāsi saddhammaṃ,_  
_buddhaseṭṭhassa sammukhā._  

</Paragraph>

</Division>
<Division number="369" book="th1" bookNo="19" page="280">

<Paragraph number="631" type="verses" book="th1">
_Pañcakkhandhe pariññāya,_  
_Bhāvayitvāna añjasaṃ;_  
_Pappuyya paramaṃ santiṃ,_  
_Parinibbissatyanāsavo”ti._  


</Paragraph>
</Division>
