<Division number="659" book="th1">

<Paragraph number="986" type="verses" book="th1">
_“Yathāpi bhaddo ājañño,_  
_dhure yutto dhurassaho;_  
_Mathito atibhārena,_  
_saṃyugaṃ nātivattati._  

</Paragraph>

</Division>
<Division number="660" book="th1">

<Paragraph number="987" type="verses" book="th1">
_Evaṃ paññāya ye tittā,_  
_samuddo vārinā yathā;_  
_Na pare atimaññanti,_  
_ariyadhammova pāṇinaṃ._  

</Paragraph>

</Division>
<Division number="661" book="th1">

<Paragraph number="988" type="verses" book="th1">
_Kāle kālavasaṃ pattā,_  
_bhavābhavavasaṃ gatā;_  
_Narā dukkhaṃ nigacchanti,_  
_tedha socanti māṇavā._  

</Paragraph>

</Division>
<Division number="662" book="th1">

<Paragraph number="989" type="verses" book="th1">
_Unnatā sukhadhammena,_  
_dukkhadhammena conatā;_  
_Dvayena bālā haññanti,_  
_yathābhūtaṃ adassino._  

</Paragraph>

</Division>
<Division number="663" book="th1">

<Paragraph number="990" type="verses" book="th1">
_Ye ca dukkhe sukhasmiñca,_  
_majjhe sibbinimaccagū;_  
_Ṭhitā te indakhīlova,_  
_na te unnataonatā._  

</Paragraph>

</Division>
<Division number="664" book="th1">

<Paragraph number="991" type="verses" book="th1">
_Na heva lābhe nālābhe,_  
_na yase na ca kittiyā;_  
_Na nindāyaṃ pasaṃsāya,_  
_na te dukkhe sukhamhi ca._  

</Paragraph>

</Division>
<Division number="665" book="th1">

<Paragraph number="992" type="verses" book="th1">
_Sabbattha te na limpanti,_  
_udabinduva pokkhare;_  
_Sabbattha sukhitā dhīrā,_  
_sabbattha aparājitā._  

</Paragraph>

</Division>
<Division number="666" book="th1">

<Paragraph number="993" type="verses" book="th1">
_Dhammena ca alābho yo,_  
_yo ca lābho adhammiko;_  
_Alābho dhammiko seyyo,_  
_yañce lābho adhammiko._  

</Paragraph>

</Division>
<Division number="667" book="th1">

<Paragraph number="994" type="verses" book="th1">
_Yaso ca appabuddhīnaṃ,_  
_viññūnaṃ ayaso ca yo;_  
_Ayasova seyyo viññūnaṃ,_  
_na yaso appabuddhinaṃ._  

</Paragraph>

</Division>
<Division number="668" book="th1">

<Paragraph number="995" type="verses" book="th1">
_Dummedhehi pasaṃsā ca,_  
_viññūhi garahā ca yā;_  
_Garahāva seyyo viññūhi,_  
_yañce bālappasaṃsanā._  

</Paragraph>

</Division>
<Division number="669" book="th1">

<Paragraph number="996" type="verses" book="th1">
_Sukhañca kāmamayikaṃ,_  
_dukkhañca pavivekiyaṃ;_  
_Pavivekadukkhaṃ seyyo,_  
_yañce kāmamayaṃ sukhaṃ._  

</Paragraph>

</Division>
<Division number="670" book="th1">

<Paragraph number="997" type="verses" book="th1">
_Jīvitañca adhammena,_  
_dhammena maraṇañca yaṃ;_  
_Maraṇaṃ dhammikaṃ seyyo,_  
_yañce jīve adhammikaṃ._  

</Paragraph>

</Division>
<Division number="671" book="th1">

<Paragraph number="998" type="verses" book="th1">
_Kāmakopappahīnā ye,_  
_santacittā bhavābhave;_  
_Caranti loke asitā,_  
_natthi tesaṃ piyāpiyaṃ._  

</Paragraph>

</Division>
<Division number="672" book="th1">

<Paragraph number="999" type="verses" book="th1">
_Bhāvayitvāna bojjhaṅge,_  
_indriyāni balāni ca;_  
_Pappuyya paramaṃ santiṃ,_  
_parinibbantināsavā”ti._  


</Paragraph>
</Division>
