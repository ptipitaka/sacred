<Paragraph number="633" type="verses" book="th1">
_“Yo ve garūnaṃ vacanaññu dhīro,_  
_Vase ca tamhi janayetha pemaṃ;_  
_So bhattimā nāma ca hoti paṇḍito,_  
_Ñatvā ca dhammesu visesi assa._  

</Paragraph>

</Division>
<Division number="371" book="th1">

<Paragraph number="634" type="verses" book="th1">
_Yaṃ āpadā uppatitā uḷārā,_  
_Nakkhambhayante paṭisaṅkhayantaṃ;_  
_So thāmavā nāma ca hoti paṇḍito,_  
_Ñatvā ca dhammesu visesi assa._  

</Paragraph>

</Division>
<Division number="372" book="th1">

<Paragraph number="635" type="verses" book="th1">
_Yo ve samuddova ṭhito anejo,_  
_Gambhīrapañño nipuṇatthadassī;_  
_Asaṃhāriyo nāma ca hoti paṇḍito,_  
_Ñatvā ca dhammesu visesi assa._  

</Paragraph>

</Division>
<Division number="373" book="th1">

<Paragraph number="636" type="verses" book="th1">
_Bahussuto dhammadharo ca hoti,_  
_Dhammassa hoti anudhammacārī;_  
_So tādiso nāma ca hoti paṇḍito,_  
_Ñatvā ca dhammesu visesi assa._  

</Paragraph>

</Division>
<Division number="374" book="th1">

<Paragraph number="637" type="verses" book="th1">
_Atthañca yo jānāti bhāsitassa,_  
_Atthañca ñatvāna tathā karoti;_  
_Atthantaro nāma sa hoti paṇḍito,_  
_Ñatvā ca dhammesu visesi assā”ti._  


</Paragraph>
</Division>

_girimānandasumanā;_

_gayākassapavakkalī._

_therā ca ettha dvādasāti._
</Paragraph>
