<Paragraph number="1453" type="verses" book="th1">
_“Kadā nuhaṃ pabbatakandarāsu,_  
_Ekākiyo addutiyo vihassaṃ;_  
_Aniccato sabbabhavaṃ vipassaṃ,_  
_Taṃ me idaṃ taṃ nu kadā bhavissati._  

</Paragraph>

</Division>
<Division number="1095" book="th1">

<Paragraph number="1454" type="verses" book="th1">
_Kadā nuhaṃ bhinnapaṭandharo muni,_  
_Kāsāvavattho amamo nirāso;_  
_Rāgañca dosañca tatheva mohaṃ,_  
_Hantvā sukhī pavanagato vihassaṃ._  

</Paragraph>

</Division>
<Division number="1096" book="th1">

<Paragraph number="1455" type="verses" book="th1">
_Kadā aniccaṃ vadharoganīḷaṃ,_  
_Kāyaṃ imaṃ maccujarāyupaddutaṃ;_  
_Vipassamāno vītabhayo vihassaṃ,_  
_Eko vane taṃ nu kadā bhavissati._  

</Paragraph>

</Division>
<Division number="1097" book="th1">

<Paragraph number="1456" type="verses" book="th1">
_Kadā nuhaṃ bhayajananiṃ dukhāvahaṃ,_  
_Taṇhālataṃ bahuvidhānuvattaniṃ;_  
_Paññāmayaṃ tikhiṇamasiṃ gahetvā,_  
_Chetvā vase tampi kadā bhavissati._  

</Paragraph>

</Division>
<Division number="1098" book="th1">

<Paragraph number="1457" type="verses" book="th1">
_Kadā nu paññāmayamuggatejaṃ,_  
_Satthaṃ isīnaṃ sahasādiyitvā;_  
_Māraṃ sasenaṃ sahasā bhañjissaṃ,_  
_Sīhāsane taṃ nu kadā bhavissati._  

</Paragraph>

</Division>
<Division number="1099" book="th1">

<Paragraph number="1458" type="verses" book="th1">
_Kadā nuhaṃ sabbhi samāgamesu,_  
_Diṭṭho bhave dhammagarūhi tādibhi;_  
_Yāthāvadassīhi jitindriyehi,_  
_Padhāniyo taṃ nu kadā bhavissati._  

</Paragraph>

</Division>
<Division number="1100" book="th1">

<Paragraph number="1459" type="verses" book="th1">
_Kadā nu maṃ tandi khudā pipāsā,_  
_Vātātapā kīṭasarīsapā vā;_  
_Na bādhayissanti na taṃ giribbaje,_  
_Atthatthiyaṃ taṃ nu kadā bhavissati._  

</Paragraph>

</Division>
<Division number="1101" book="th1">

<Paragraph number="1460" type="verses" book="th1">
_Kadā nu kho yaṃ viditaṃ mahesinā,_  
_Cattāri saccāni sududdasāni;_  
_Samāhitatto satimā agacchaṃ,_  
_Paññāya taṃ taṃ nu kadā bhavissati._  

</Paragraph>

</Division>
<Division number="1102" book="th1">

<Paragraph number="1461" type="verses" book="th1">
_Kadā nu rūpe amite ca sadde,_  
_Gandhe rase phusitabbe ca dhamme;_  
_Ādittatohaṃ samathehi yutto,_  
_Paññāya dacchaṃ tadidaṃ kadā me._  

</Paragraph>

</Division>
<Division number="1103" book="th1">

<Paragraph number="1462" type="verses" book="th1">
_Kadā nuhaṃ dubbacanena vutto,_  
_Tato nimittaṃ vimano na hessaṃ;_  
_Atho pasatthopi tato nimittaṃ,_  
_Tuṭṭho na hessaṃ tadidaṃ kadā me._  

</Paragraph>

</Division>
<Division number="1104" book="th1">

<Paragraph number="1463" type="verses" book="th1">
_Kadā nu kaṭṭhe ca tiṇe latā ca,_  
_Khandhe imehaṃ amite ca dhamme;_  
_Ajjhattikāneva ca bāhirāni ca,_  
_Samaṃ tuleyyaṃ tadidaṃ kadā me._  

</Paragraph>

</Division>
<Division number="1105" book="th1">

<Paragraph number="1464" type="verses" book="th1">
_Kadā nu maṃ pāvusakālamegho,_  
_Navena toyena sacīvaraṃ vane;_  
_Isippayātamhi pathe vajantaṃ,_  
_Ovassate taṃ nu kadā bhavissati._  

</Paragraph>

</Division>
<Division number="1106" book="th1">

<Paragraph number="1465" type="verses" book="th1">
_Kadā mayūrassa sikhaṇḍino vane,_  
_Dijassa sutvā girigabbhare rutaṃ;_  
_Paccuṭṭhahitvā amatassa pattiyā,_  
_Sañcintaye taṃ nu kadā bhavissati._  

</Paragraph>

</Division>
<Division number="1107" book="th1">

<Paragraph number="1466" type="verses" book="th1">
_Kadā nu gaṅgaṃ yamunaṃ sarassatiṃ,_  
_Pātālakhittaṃ vaḷavāmukhañca;_  
_Asajjamāno patareyyamiddhiyā,_  
_Vibhiṃsanaṃ taṃ nu kadā bhavissati._  

</Paragraph>

</Division>
<Division number="1108" book="th1">

<Paragraph number="1467" type="verses" book="th1">
_Kadā nu nāgova asaṅgacārī,_  
_Padālaye kāmaguṇesu chandaṃ;_  
_Nibbajjayaṃ sabbasubhaṃ nimittaṃ,_  
_Jhāne yuto taṃ nu kadā bhavissati._  

</Paragraph>

</Division>
<Division number="1109" book="th1">

<Paragraph number="1468" type="verses" book="th1">
_Kadā iṇaṭṭova daliddako nidhiṃ,_  
_Ārādhayitvā dhanikehi pīḷito;_  
_Tuṭṭho bhavissaṃ adhigamma sāsanaṃ,_  
_Mahesino taṃ nu kadā bhavissati._  

</Paragraph>

</Division>
<Division number="1110" book="th1">

<Paragraph number="1469" type="verses" book="th1">
_Bahūni vassāni tayāmhi yācito,_  
_‘Agāravāsena alaṃ nu te idaṃ’;_  
_Taṃ dāni maṃ pabbajitaṃ samānaṃ,_  
_Kiṃkāraṇā citta tuvaṃ na yuñjasi._  

</Paragraph>

</Division>
<Division number="1111" book="th1">

<Paragraph number="1470" type="verses" book="th1">
_Nanu ahaṃ citta tayāmhi yācito,_  
_‘Giribbaje citrachadā vihaṅgamā’;_  
_Mahindaghosatthanitābhigajjino,_  
_Te taṃ ramessanti vanamhi jhāyinaṃ._  

</Paragraph>

</Division>
<Division number="1112" book="th1">

<Paragraph number="1471" type="verses" book="th1">
_Kulamhi mitte ca piye ca ñātake,_  
_Khiḍḍāratiṃ kāmaguṇañca loke;_  
_Sabbaṃ pahāya imamajjhupāgato,_  
_Athopi tvaṃ citta na mayha tussasi._  

</Paragraph>

</Division>
<Division number="1113" book="th1">

<Paragraph number="1472" type="verses" book="th1">
_Mameva etaṃ na hi tvaṃ paresaṃ,_  
_Sannāhakāle paridevitena kiṃ;_  
_Sabbaṃ idaṃ calamiti pekkhamāno,_  
_Abhinikkhamiṃ amatapadaṃ jigīsaṃ._  

</Paragraph>

</Division>
<Division number="1114" book="th1">

<Paragraph number="1473" type="verses" book="th1">
_Suyuttavādī dvipadānamuttamo,_  
_Mahābhisakko naradammasārathi;_  
_‘Cittaṃ calaṃ makkaṭasannibhaṃ iti,_  
_Avītarāgena sudunnivārayaṃ’._  

</Paragraph>

</Division>
<Division number="1115" book="th1">

<Paragraph number="1474" type="verses" book="th1">
_Kāmā hi citrā madhurā manoramā,_  
_Aviddasū yattha sitā puthujjanā;_  
_Te dukkhamicchanti punabbhavesino,_  
_Cittena nītā niraye nirākatā._  

</Paragraph>

</Division>
<Division number="1116" book="th1">

_Dīpīhi byagghehi purakkhato vasaṃ;_  
_Kāye apekkhaṃ jaha mā virādhaya’,_  
_Itissu maṃ citta pure niyuñjasi._  

</Paragraph>

</Division>
<Division number="1117" book="th1">

_Tisso ca vijjā phusa buddhasāsane’,_  
_Itissu maṃ citta pure niyuñjasi._  

</Paragraph>

</Division>
<Division number="1118" book="th1">

_Aṭṭhaṅgikaṃ sabbakilesasodhanaṃ’,_  
_Itissu maṃ citta pure niyuñjasi._  

</Paragraph>

</Division>
<Division number="1119" book="th1">

_Itissu maṃ citta pure niyuñjasi._  

</Paragraph>

</Division>
<Division number="1120" book="th1">

_Itissu maṃ citta pure niyuñjasi._  

</Paragraph>

</Division>

_Itissu maṃ citta pure niyuñjasi._  

</Paragraph>

</Division>

_Itissu maṃ citta pure niyuñjasi._

_Itissu maṃ citta pure niyuñjasi._  

</Paragraph>

</Division>
<Division number="1124" book="th1">

_Mūle taruṃ chettu tameva icchasi;_  
_Tathūpamaṃ cittamidaṃ karosi,_  
_Yaṃ maṃ aniccamhi cale niyuñjasi._  

</Paragraph>

</Division>
<Division number="1125" book="th1">

_Na te karissaṃ vacanaṃ idānihaṃ;_  
_Dukkhā hi kāmā kaṭukā mahabbhayā,_  
_Nibbānamevābhimano carissaṃ._  

</Paragraph>

</Division>
<Division number="1126" book="th1">

_Na cittahetū na ca dūrakantanā;_  
_Ājīvahetū ca ahaṃ na nikkhamiṃ,_  
_Kato ca te citta paṭissavo mayā._  

</Paragraph>

</Division>
<Division number="1127" book="th1">

_Itissu maṃ citta tadā niyuñjasi,_  
_Idāni tvaṃ gacchasi pubbaciṇṇaṃ._  

</Paragraph>

</Division>
<Division number="1128" book="th1">

<Paragraph number="1487" type="verses" book="th1">

_Subhāni rūpāni sukhā ca vedanā;_  
_Manāpiyā kāmaguṇā ca vantā,_  
_Vante ahaṃ āvamituṃ na ussahe._  

</Paragraph>

</Division>
<Division number="1129" book="th1">

<Paragraph number="1488" type="verses" book="th1">

_Bahūsu jātīsu na mesi kopito;_  
_Ajjhattasambhavo kataññutāya te,_  
_Dukkhe ciraṃ saṃsaritaṃ tayā kate._  

</Paragraph>

</Division>
<Division number="1130" book="th1">

<Paragraph number="1489" type="verses" book="th1">

_Tvaṃ khattiyo rājadasī karosi;_  
_Vessā ca suddā ca bhavāma ekadā,_  
_Devattanaṃ vāpi taveva vāhasā._  

</Paragraph>

</Division>
<Division number="1131" book="th1">

_Petattanaṃ vāpi taveva vāhasā._  

</Paragraph>

</Division>
<Division number="1132" book="th1">

<Paragraph number="1491" type="verses" book="th1">

_Muhuṃ muhuṃ cāraṇikaṃva dassayaṃ;_  
_Ummattakeneva mayā palobhasi,_  
_Kiñcāpi te citta virādhitaṃ mayā._  

</Paragraph>

</Division>
<Division number="1133" book="th1">

_Yenicchakaṃ yatthakāmaṃ yathāsukhaṃ;_  
_Tadajjahaṃ niggahessāmi yoniso,_  
_Hatthippabhinnaṃ viya aṅkusaggaho._  

</Paragraph>

</Division>

</Paragraph>

</Paragraph>
