<Paragraph number="518" type="verses" book="th1">
_“Dhiratthu pūre duggandhe,_  
_mārapakkhe avassute;_  
_Navasotāni te kāye,_  
_yāni sandanti sabbadā._

<Division number="280" book="th1">

<Paragraph number="519" type="verses" book="th1">
_Mā purāṇaṃ amaññittho,_  
_māsādesi tathāgate;_  
_Saggepi te na rajjanti,_  
_kimaṅgaṃ pana mānuse._  

</Paragraph>

</Division>
<Division number="281" book="th1">

<Paragraph number="520" type="verses" book="th1">
_Ye ca kho bālā dummedhā,_  
_dummantī mohapārutā;_  
_Tādisā tattha rajjanti,_  
_mārakhittamhi bandhane._  

</Paragraph>

</Division>
<Division number="282" book="th1">

<Paragraph number="521" type="verses" book="th1">
_Yesaṃ rāgo ca doso ca,_  
_avijjā ca virājitā;_  
_Tādī tattha na rajjanti,_  
_chinnasuttā abandhanā”ti._  


</Paragraph>
</Division>
