<Paragraph number="533" type="verses" book="th1">
_“Yo dandhakāle tarati,_  
_taraṇīye ca dandhaye;_  
_Ayoni saṃvidhānena,_  
_bālo dukkhaṃ nigacchati._  

</Paragraph>

</Division>
<Division number="292" book="th1" bookNo="19" page="271">

<Paragraph number="534" type="verses" book="th1">
_Tassatthā parihāyanti,_  
_kāḷapakkheva candimā;_  
_Āyasakyañca pappoti,_  
_mittehi ca virujjhati._  

</Paragraph>

</Division>
<Division number="293" book="th1" bookNo="19" page="271">

<Paragraph number="535" type="verses" book="th1">
_Yo dandhakāle dandheti,_  
_taraṇīye ca tāraye;_  
_Yoniso saṃvidhānena,_  
_sukhaṃ pappoti paṇḍito._  

</Paragraph>

</Division>
<Division number="294" book="th1" bookNo="19" page="271">

<Paragraph number="536" type="verses" book="th1">
_Tassatthā paripūrenti,_  
_sukkapakkheva candimā;_  
_Yaso kittiñca pappoti,_  
_mittehi na virujjhatī”ti._  


</Paragraph>
</Division>
