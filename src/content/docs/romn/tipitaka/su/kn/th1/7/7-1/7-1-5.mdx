<Paragraph number="776" type="verses" book="th1">
_“Sare hatthehi bhañjitvā,_  
_katvāna kuṭimacchisaṃ;_  
_Tena me sarabhaṅgoti,_  
_nāmaṃ sammutiyā ahu._  

</Paragraph>

</Division>
<Division number="488" book="th1">

<Paragraph number="777" type="verses" book="th1">
_Na mayhaṃ kappate ajja,_  
_sare hatthehi bhañjituṃ;_  
_Sikkhāpadā no paññattā,_  
_gotamena yasassinā._  

</Paragraph>

</Division>
<Division number="489" book="th1">

<Paragraph number="778" type="verses" book="th1">
_Sakalaṃ samattaṃ rogaṃ,_  
_Sarabhaṅgo nāddasaṃ pubbe;_  
_Soyaṃ rogo diṭṭho,_  
_Vacanakarenātidevassa._  

</Paragraph>

</Division>
<Division number="490" book="th1">

<Paragraph number="779" type="verses" book="th1">
_Yeneva maggena gato vipassī,_  
_Yeneva maggena sikhī ca vessabhū;_  
_Kakusandhakoṇāgamano ca kassapo,_  
_Tenañjasena agamāsi gotamo._  

</Paragraph>

</Division>
<Division number="491" book="th1">

<Paragraph number="780" type="verses" book="th1">
_Vītataṇhā anādānā,_  
_satta buddhā khayogadhā;_  
_Yehāyaṃ desito dhammo,_  
_dhammabhūtehi tādibhi._  

</Paragraph>

</Division>
<Division number="492" book="th1">

<Paragraph number="781" type="verses" book="th1">
_Cattāri ariyasaccāni,_  
_anukampāya pāṇinaṃ;_  
_Dukkhaṃ samudayo maggo,_  
_nirodho dukkhasaṅkhayo._  

</Paragraph>

</Division>
<Division number="493" book="th1">

<Paragraph number="782" type="verses" book="th1">
_Yasmiṃ nivattate dukkhaṃ,_  
_saṃsārasmiṃ anantakaṃ;_  
_Bhedā imassa kāyassa,_  
_jīvitassa ca saṅkhayā;_  
_Añño punabbhavo natthi,_  
_suvimuttomhi sabbadhī”ti._  


</Paragraph>
</Division>

<Paragraph number="783" type="center" book="th1">
…  Sarabhaṅgo thero… .



</Paragraph>

<Paragraph number="784" type="center" book="th1">
Sattakanipāto niṭṭhito.



</Paragraph>

<Paragraph number="785" type="center" book="th1">
Tatruddānaṃ


</Paragraph>

<Paragraph number="786" type="verses" book="th1">
_Sundarasamuddo thero,_  
_thero lakuṇḍabhaddiyo;_  
_Bhaddo thero ca sopāko,_  
_sarabhaṅgo mahāisi;_  
_Sattake pañcakā therā,_  
_gāthāyo pañcatiṃsatīti._
</Paragraph>
