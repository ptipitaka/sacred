<Paragraph number="1041" type="verses" book="th1">
_“Yaññatthaṃ vā dhanatthaṃ vā,_  
_ye hanāma mayaṃ pure;_  
_Avasesaṃ bhayaṃ hoti,_  
_vedhanti vilapanti ca._  

</Paragraph>

</Division>
<Division number="706" book="th1" bookNo="19" page="320">

<Paragraph number="1042" type="verses" book="th1">
_Tassa te natthi bhītattaṃ,_  
_bhiyyo vaṇṇo pasīdati;_  
_Kasmā na paridevesi,_  
_evarūpe mahabbhaye”._  

</Paragraph>

</Division>
<Division number="707" book="th1" bookNo="19" page="320">

<Paragraph number="1043" type="verses" book="th1">
_“Natthi cetasikaṃ dukkhaṃ,_  
_anapekkhassa gāmaṇi;_  
_Atikkantā bhayā sabbe,_  
_khīṇasaṃyojanassa ve._  

</Paragraph>

</Division>
<Division number="708" book="th1" bookNo="19" page="320">

<Paragraph number="1044" type="verses" book="th1">
_Khīṇāya bhavanettiyā,_  
_diṭṭhe dhamme yathātathe;_  
_Na bhayaṃ maraṇe hoti,_  
_bhāranikkhepane yathā._  

</Paragraph>

</Division>
<Division number="709" book="th1" bookNo="19" page="320">

<Paragraph number="1045" type="verses" book="th1">
_Suciṇṇaṃ brahmacariyaṃ me,_  
_maggo cāpi subhāvito;_  
_Maraṇe me bhayaṃ natthi,_  
_rogānamiva saṅkhaye._  

</Paragraph>

</Division>
<Division number="710" book="th1" bookNo="19" page="320">

<Paragraph number="1046" type="verses" book="th1">
_Suciṇṇaṃ brahmacariyaṃ me,_  
_maggo cāpi subhāvito;_  
_Nirassādā bhavā diṭṭhā,_  
_visaṃ pitvāva chaḍḍitaṃ._  

</Paragraph>

</Division>
<Division number="711" book="th1" bookNo="19" page="320">

<Paragraph number="1047" type="verses" book="th1">
_Pāragū anupādāno,_  
_katakicco anāsavo;_  
_Tuṭṭho āyukkhayā hoti,_  
_mutto āghātanā yathā._  

</Paragraph>

</Division>
<Division number="712" book="th1" bookNo="19" page="320">

<Paragraph number="1048" type="verses" book="th1">
_Uttamaṃ dhammataṃ patto,_  
_sabbaloke anatthiko;_  
_Ādittāva gharā mutto,_  
_maraṇasmiṃ na socati._  

</Paragraph>

</Division>
<Division number="713" book="th1" bookNo="19" page="320">

<Paragraph number="1049" type="verses" book="th1">
_Yadatthi saṅgataṃ kiñci,_  
_bhavo vā yattha labbhati;_  
_Sabbaṃ anissaraṃ etaṃ,_  
_iti vuttaṃ mahesinā._  

</Paragraph>

</Division>
<Division number="714" book="th1" bookNo="19" page="320">

<Paragraph number="1050" type="verses" book="th1">
_Yo taṃ tathā pajānāti,_  
_yathā buddhena desitaṃ;_  
_Na gaṇhāti bhavaṃ kiñci,_  
_sutattaṃva ayoguḷaṃ._  

</Paragraph>

</Division>
<Division number="715" book="th1" bookNo="19" page="320">

<Paragraph number="1051" type="verses" book="th1">
_Na me hoti ‘ahosin’ti,_  
_‘bhavissan’ti na hoti me;_  
_Saṅkhārā vigamissanti,_  
_tattha kā paridevanā._  

</Paragraph>

</Division>
<Division number="716" book="th1" bookNo="19" page="320">

<Paragraph number="1052" type="verses" book="th1">
_Suddhaṃ dhammasamuppādaṃ,_  
_Suddhaṃ saṅkhārasantatiṃ;_  
_Passantassa yathābhūtaṃ,_  
_Na bhayaṃ hoti gāmaṇi._  

</Paragraph>

</Division>
<Division number="717" book="th1" bookNo="19" page="321">

<Paragraph number="1053" type="verses" book="th1">
_Tiṇakaṭṭhasamaṃ lokaṃ,_  
_yadā paññāya passati;_  
_Mamattaṃ so asaṃvindaṃ,_  
_‘natthi me’ti na socati._  

</Paragraph>

</Division>
<Division number="718" book="th1" bookNo="19" page="321">

<Paragraph number="1054" type="verses" book="th1">
_Ukkaṇṭhāmi sarīrena,_  
_bhavenamhi anatthiko;_  
_Soyaṃ bhijjissati kāyo,_  
_añño ca na bhavissati._  

</Paragraph>

</Division>
<Division number="719" book="th1" bookNo="19" page="321">

<Paragraph number="1055" type="verses" book="th1">
_Yaṃ vo kiccaṃ sarīrena,_  
_taṃ karotha yadicchatha;_  
_Na me tappaccayā tattha,_  
_doso pemañca hehiti”._  

</Paragraph>

</Division>
<Division number="720" book="th1" bookNo="19" page="321">

<Paragraph number="1056" type="verses" book="th1">
_Tassa taṃ vacanaṃ sutvā,_  
_abbhutaṃ lomahaṃsanaṃ;_  
_Satthāni nikkhipitvāna,_  
_māṇavā etadabravuṃ._  

</Paragraph>

</Division>
<Division number="721" book="th1" bookNo="19" page="321">

<Paragraph number="1057" type="verses" book="th1">
_“Kiṃ bhadante karitvāna,_  
_ko vā ācariyo tava;_  
_Kassa sāsanamāgamma,_  
_labbhate taṃ asokatā”._  

</Paragraph>

</Division>
<Division number="722" book="th1" bookNo="19" page="321">

<Paragraph number="1058" type="verses" book="th1">
_“Sabbaññū sabbadassāvī,_  
_jino ācariyo mama;_  
_Mahākāruṇiko satthā,_  
_sabbalokatikicchako._  

</Paragraph>

</Division>
<Division number="723" book="th1" bookNo="19" page="321">

<Paragraph number="1059" type="verses" book="th1">
_Tenāyaṃ desito dhammo,_  
_khayagāmī anuttaro;_  
_Tassa sāsanamāgamma,_  
_labbhate taṃ asokatā”._  

</Paragraph>

</Division>
<Division number="724" book="th1" bookNo="19" page="321">

<Paragraph number="1060" type="verses" book="th1">
_Sutvāna corā isino subhāsitaṃ,_  
_Nikkhippa satthāni ca āvudhāni ca;_  
_Tamhā ca kammā viramiṃsu eke,_  
_Eke ca pabbajjamarocayiṃsu._  

</Paragraph>

</Division>
<Division number="725" book="th1" bookNo="19" page="321">

<Paragraph number="1061" type="verses" book="th1">
_Te pabbajitvā sugatassa sāsane,_  
_Bhāvetva bojjhaṅgabalāni paṇḍitā;_  
_Udaggacittā sumanā katindriyā,_  
_Phusiṃsu nibbānapadaṃ asaṅkhatanti._  


</Paragraph>
</Division>

… Adhimutto thero… .
