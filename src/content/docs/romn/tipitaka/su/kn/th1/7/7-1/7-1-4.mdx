<Division number="480" book="th1">

<Paragraph number="768" type="verses" book="th1">
_“Disvā pāsādachāyāyaṃ,_  
_caṅkamantaṃ naruttamaṃ;_  
_Tattha naṃ upasaṅkamma,_  
_vandissaṃ purisuttamaṃ._  

</Paragraph>

</Division>
<Division number="481" book="th1">

<Paragraph number="769" type="verses" book="th1">
_Ekaṃsaṃ cīvaraṃ katvā,_  
_saṃharitvāna pāṇayo;_  
_Anucaṅkamissaṃ virajaṃ,_  
_sabbasattānamuttamaṃ._  

</Paragraph>

</Division>
<Division number="482" book="th1">

<Paragraph number="770" type="verses" book="th1">
_Tato pañhe apucchi maṃ,_  
_pañhānaṃ kovido vidū;_  
_Acchambhī ca abhīto ca,_  
_byākāsiṃ satthuno ahaṃ._  

</Paragraph>

</Division>
<Division number="483" book="th1">

<Paragraph number="771" type="verses" book="th1">
_Vissajjitesu pañhesu,_  
_anumodi tathāgato;_  
_Bhikkhusaṃghaṃ viloketvā,_  
_imamatthaṃ abhāsatha._  

</Paragraph>

</Division>
<Division number="484" book="th1">

<Paragraph number="772" type="verses" book="th1">
_‘Lābhā aṅgānaṃ magadhānaṃ,_  
_Yesāyaṃ paribhuñjati;_  
_Cīvaraṃ piṇḍapātañca,_  
_Paccayaṃ sayanāsanaṃ;_  
_Paccuṭṭhānañca sāmīciṃ,_  
_Tesaṃ lābhāti cābravi._  

</Paragraph>

</Division>
<Division number="485" book="th1">

<Paragraph number="773" type="verses" book="th1">
_Ajjatagge maṃ sopāka,_  
_dassanāyopasaṅkama;_  
_Esā ceva te sopāka,_  
_bhavatu upasampadā’._  

</Paragraph>

</Division>
<Division number="486" book="th1">

<Paragraph number="774" type="verses" book="th1">
_Jātiyā sattavassohaṃ,_  
_laddhāna upasampadaṃ;_  
_Dhāremi antimaṃ dehaṃ,_  
_aho dhammasudhammatā”ti._  


</Paragraph>
</Division>
