<Paragraph number="760" type="verses" book="th1">
_“Ekaputto ahaṃ āsiṃ,_  
_piyo mātu piyo pitu;_  
_Bahūhi vatacariyāhi,_  
_laddho āyācanāhi ca._  

</Paragraph>

</Division>
<Division number="474" book="th1" bookNo="19" page="292">

<Paragraph number="761" type="verses" book="th1">
_Te ca maṃ anukampāya,_  
_atthakāmā hitesino;_  
_Ubho pitā ca mātā ca,_  
_buddhassa upanāmayuṃ._  

</Paragraph>

</Division>
<Division number="475" book="th1" bookNo="19" page="292">

<Paragraph number="762" type="verses" book="th1">
_‘Kicchā laddho ayaṃ putto,_  
_sukhumālo sukhedhito;_  
_Imaṃ dadāma te nātha,_  
_jinassa paricārakaṃ’._  

</Paragraph>

</Division>
<Division number="476" book="th1" bookNo="19" page="292">

<Paragraph number="763" type="verses" book="th1">
_Satthā ca maṃ paṭiggayha,_  
_ānandaṃ etadabravi;_  
_‘Pabbājehi imaṃ khippaṃ,_  
_hessatyājāniyo ayaṃ’._  

</Paragraph>

</Division>
<Division number="477" book="th1" bookNo="19" page="292">

<Paragraph number="764" type="verses" book="th1">
_Pabbājetvāna maṃ satthā,_  
_vihāraṃ pāvisī jino;_  
_Anoggatasmiṃ sūriyasmiṃ,_  
_tato cittaṃ vimucci me._  

</Paragraph>

</Division>
<Division number="478" book="th1" bookNo="19" page="292">

<Paragraph number="765" type="verses" book="th1">
_Tato satthā nirākatvā,_  
_paṭisallānavuṭṭhito;_  
_‘Ehi bhaddā’ti maṃ āha,_  
_sā me āsūpasampadā._  

</Paragraph>

</Division>
<Division number="479" book="th1" bookNo="19" page="292">

<Paragraph number="766" type="verses" book="th1">
_Jātiyā sattavassena,_  
_laddhā me upasampadā;_  
_Tisso vijjā anuppattā,_  
_aho dhammasudhammatā”ti._  


</Paragraph>
</Division>
