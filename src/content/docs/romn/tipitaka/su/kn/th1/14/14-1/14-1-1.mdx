<Paragraph number="971" type="verses" book="th1">
_“Yadā ahaṃ pabbajito,_  
_agārasmānagāriyaṃ;_  
_Nābhijānāmi saṅkappaṃ,_  
_anariyaṃ dosasaṃhitaṃ._  

</Paragraph>

</Division>
<Division number="646" book="th1" bookNo="19" page="314">

<Paragraph number="972" type="verses" book="th1">
_‘Ime haññantu vajjhantu,_  
_dukkhaṃ pappontu pāṇino’;_  
_Saṅkappaṃ nābhijānāmi,_  
_imasmiṃ dīghamantare._  

</Paragraph>

</Division>
<Division number="647" book="th1" bookNo="19" page="314">

<Paragraph number="973" type="verses" book="th1">
_Mettañca abhijānāmi,_  
_appamāṇaṃ subhāvitaṃ;_  
_Anupubbaṃ paricitaṃ,_  
_yathā buddhena desitaṃ._  

</Paragraph>

</Division>
<Division number="648" book="th1" bookNo="19" page="314">

<Paragraph number="974" type="verses" book="th1">
_Sabbamitto sabbasakho,_  
_sabbabhūtānukampako;_  
_Mettacittañca bhāvemi,_  
_abyāpajjarato sadā._  

</Paragraph>

</Division>
<Division number="649" book="th1" bookNo="19" page="314">

<Paragraph number="975" type="verses" book="th1">
_Asaṃhīraṃ asaṃkuppaṃ,_  
_cittaṃ āmodayāmahaṃ;_  
_Brahmavihāraṃ bhāvemi,_  
_akāpurisasevitaṃ._  

</Paragraph>

</Division>
<Division number="650" book="th1" bookNo="19" page="314">

<Paragraph number="976" type="verses" book="th1">
_Avitakkaṃ samāpanno,_  
_sammāsambuddhasāvako;_  
_Ariyena tuṇhībhāvena,_  
_upeto hoti tāvade._  

</Paragraph>

</Division>
<Division number="651" book="th1" bookNo="19" page="314">

<Paragraph number="977" type="verses" book="th1">
_Yathāpi pabbato selo,_  
_acalo suppatiṭṭhito;_  
_Evaṃ mohakkhayā bhikkhu,_  
_pabbatova na vedhati._  

</Paragraph>

</Division>
<Division number="652" book="th1" bookNo="19" page="314">

<Paragraph number="978" type="verses" book="th1">
_Anaṅgaṇassa posassa,_  
_niccaṃ sucigavesino;_  
_Vālaggamattaṃ pāpassa,_  
_abbhamattaṃva khāyati._  

</Paragraph>

</Division>
<Division number="653" book="th1" bookNo="19" page="314">

<Paragraph number="979" type="verses" book="th1">
_Nagaraṃ yathā paccantaṃ,_  
_guttaṃ santarabāhiraṃ;_  
_Evaṃ gopetha attānaṃ,_  
_khaṇo vo mā upaccagā._  

</Paragraph>

</Division>
<Division number="654" book="th1" bookNo="19" page="314">

<Paragraph number="980" type="verses" book="th1">
_Nābhinandāmi maraṇaṃ,_  
_nābhinandāmi jīvitaṃ;_  
_Kālañca paṭikaṅkhāmi,_  
_nibbisaṃ bhatako yathā._  

</Paragraph>

</Division>
<Division number="655" book="th1" bookNo="19" page="314">

<Paragraph number="981" type="verses" book="th1">
_Nābhinandāmi maraṇaṃ,_  
_ …pe… _  
_sampajāno patissato._  

</Paragraph>

</Division>
<Division number="656" book="th1" bookNo="19" page="314">

<Paragraph number="982" type="verses" book="th1">
_Pariciṇṇo mayā satthā,_  
_kataṃ buddhassa sāsanaṃ;_  
_Ohito garuko bhāro,_  
_bhavanetti samūhatā._  

</Paragraph>

</Division>
<Division number="657" book="th1" bookNo="19" page="315">

<Paragraph number="983" type="verses" book="th1">
_Yassa catthāya pabbajito,_  
_agārasmānagāriyaṃ;_  
_So me attho anuppatto,_  
_sabbasaṃyojanakkhayo._  

</Paragraph>

</Division>
<Division number="658" book="th1" bookNo="19" page="315">

<Paragraph number="984" type="verses" book="th1">
_Sampādethappamādena,_  
_esā me anusāsanī;_  
_Handāhaṃ parinibbissaṃ,_  
_vippamuttomhi sabbadhī”ti._  


</Paragraph>
</Division>
