_“Nīce kulamhi jātohaṃ,_  
_daliddo appabhojano;_  
_Hīnakammaṃ mamaṃ āsi,_  
_ahosiṃ pupphachaḍḍako._  

</Paragraph>

</Division>
<Division number="621" book="th1">

<Paragraph number="939" type="verses" book="th1">
_Jigucchito manussānaṃ,_  
_paribhūto ca vambhito;_  
_Nīcaṃ manaṃ karitvāna,_  
_vandissaṃ bahukaṃ janaṃ._  

</Paragraph>

</Division>
<Division number="622" book="th1">

<Paragraph number="940" type="verses" book="th1">
_Athaddasāsiṃ sambuddhaṃ,_  
_bhikkhusaṃghapurakkhataṃ;_  
_Pavisantaṃ mahāvīraṃ,_  
_magadhānaṃ puruttamaṃ._  

</Paragraph>

</Division>
<Division number="623" book="th1">

<Paragraph number="941" type="verses" book="th1">
_Nikkhipitvāna byābhaṅgiṃ,_  
_vandituṃ upasaṅkamiṃ;_  
_Mameva anukampāya,_  
_aṭṭhāsi purisuttamo._  

</Paragraph>

</Division>
<Division number="624" book="th1">

<Paragraph number="942" type="verses" book="th1">
_Vanditvā satthuno pāde,_  
_ekamantaṃ ṭhito tadā;_  
_Pabbajjaṃ ahamāyāciṃ,_  
_sabbasattānamuttamaṃ._  

</Paragraph>

</Division>
<Division number="625" book="th1">

<Paragraph number="943" type="verses" book="th1">
_Tato kāruṇiko satthā,_  
_sabbalokānukampako;_  
_‘Ehi bhikkhū’ti maṃ āha,_  
_sā me āsūpasampadā._  

</Paragraph>

</Division>
<Division number="626" book="th1">

<Paragraph number="944" type="verses" book="th1">
_Sohaṃ eko araññasmiṃ,_  
_viharanto atandito;_  
_Akāsiṃ satthu vacanaṃ,_  
_yathā maṃ ovadī jino._  

</Paragraph>

</Division>
<Division number="627" book="th1">

<Paragraph number="945" type="verses" book="th1">
_Rattiyā paṭhamaṃ yāmaṃ,_  
_Pubbajātimanussariṃ;_  
_Rattiyā majjhimaṃ yāmaṃ,_  
_Dibbacakkhuṃ visodhayiṃ;_  
_Rattiyā pacchime yāme,_  
_Tamokhandhaṃ padālayiṃ._  

</Paragraph>

</Division>
<Division number="628" book="th1">

<Paragraph number="946" type="verses" book="th1">
_Tato ratyā vivasāne,_  
_sūriyassuggamanaṃ pati;_  
_Indo brahmā ca āgantvā,_  
_maṃ namassiṃsu pañjalī._  

</Paragraph>

</Division>
<Division number="629" book="th1">

<Paragraph number="947" type="verses" book="th1">
_‘Namo te purisājañña,_  
_namo te purisuttama;_  
_Yassa te āsavā khīṇā,_  
_dakkhiṇeyyosi mārisa’._  

</Paragraph>

</Division>
<Division number="630" book="th1">

<Paragraph number="948" type="verses" book="th1">
_Tato disvāna maṃ satthā,_  
_devasaṅghapurakkhataṃ;_  
_Sitaṃ pātukaritvāna,_  
_imamatthaṃ abhāsatha._  

</Paragraph>

</Division>
<Division number="631" book="th1">

<Paragraph number="949" type="verses" book="th1">
_‘Tapena brahmacariyena,_  
_saṃyamena damena ca;_  
_Etena brāhmaṇo hoti,_  
_etaṃ brāhmaṇamuttaman’”ti._  


</Paragraph>
</Division>
