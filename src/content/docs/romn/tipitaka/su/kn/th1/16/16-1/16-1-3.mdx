<Paragraph number="1085" type="verses" book="th1">
_“Cirarattaṃ vatātāpī,_  
_dhammaṃ anuvicintayaṃ;_  
_Samaṃ cittassa nālatthaṃ,_  
_pucchaṃ samaṇabrāhmaṇe._  

</Paragraph>

</Division>
<Division number="748" book="th1" bookNo="19" page="323">

<Paragraph number="1086" type="verses" book="th1">
_‘Ko so pāraṅgato loke,_  
_ko patto amatogadhaṃ;_  
_Kassa dhammaṃ paṭicchāmi,_  
_paramatthavijānanaṃ’._  

</Paragraph>

</Division>
<Division number="749" book="th1" bookNo="19" page="323">

<Paragraph number="1087" type="verses" book="th1">
_Antovaṅkagato āsi,_  
_macchova ghasamāmisaṃ;_  
_Baddho mahindapāsena,_  
_vepacityasuro yathā._  

</Paragraph>

</Division>
<Division number="750" book="th1" bookNo="19" page="323">

<Paragraph number="1088" type="verses" book="th1">
_Añchāmi naṃ na muñcāmi,_  
_asmā sokapariddavā;_  
_Ko me bandhaṃ muñcaṃ loke,_  
_sambodhiṃ vedayissati._  

</Paragraph>

</Division>
<Division number="751" book="th1" bookNo="19" page="323">

<Paragraph number="1089" type="verses" book="th1">
_Samaṇaṃ brāhmaṇaṃ vā kaṃ,_  
_ādisantaṃ pabhaṅgunaṃ;_  
_Kassa dhammaṃ paṭicchāmi,_  
_jarāmaccupavāhanaṃ._  

</Paragraph>

</Division>
<Division number="752" book="th1" bookNo="19" page="323">

<Paragraph number="1090" type="verses" book="th1">
_Vicikicchākaṅkhāganthitaṃ,_  
_sārambhabalasaññutaṃ;_  
_Kodhappattamanatthaddhaṃ,_  
_abhijappappadāraṇaṃ._  

</Paragraph>

</Division>
<Division number="753" book="th1" bookNo="19" page="324">

<Paragraph number="1091" type="verses" book="th1">
_Taṇhādhanusamuṭṭhānaṃ,_  
_dve ca pannarasāyutaṃ;_  
_Passa orasikaṃ bāḷhaṃ,_  
_bhetvāna yadi tiṭṭhati._  

</Paragraph>

</Division>
<Division number="754" book="th1" bookNo="19" page="324">

<Paragraph number="1092" type="verses" book="th1">
_Anudiṭṭhīnaṃ appahānaṃ,_  
_saṅkappaparatejitaṃ;_  
_Tena viddho pavedhāmi,_  
_pattaṃva māluteritaṃ._  

</Paragraph>

</Division>
<Division number="755" book="th1" bookNo="19" page="324">

<Paragraph number="1093" type="verses" book="th1">
_Ajjhattaṃ me samuṭṭhāya,_  
_khippaṃ paccati māmakaṃ;_  
_Chaphassāyatanī kāyo,_  
_yattha sarati sabbadā._  

</Paragraph>

</Division>
<Division number="756" book="th1" bookNo="19" page="324">

<Paragraph number="1094" type="verses" book="th1">
_Taṃ na passāmi tekicchaṃ,_  
_yo metaṃ sallamuddhare;_  
_Nānārajjena satthena,_  
_nāññena vicikicchitaṃ._  

</Paragraph>

</Division>
<Division number="757" book="th1" bookNo="19" page="324">

<Paragraph number="1095" type="verses" book="th1">
_Ko me asattho avaṇo,_  
_Sallamabbhantarapassayaṃ;_  
_Ahiṃsaṃ sabbagattāni,_  
_Sallaṃ me uddharissati._  

</Paragraph>

</Division>
<Division number="758" book="th1" bookNo="19" page="324">

<Paragraph number="1096" type="verses" book="th1">
_Dhammappati hi so seṭṭho,_  
_visadosappavāhako;_  
_Gambhīre patitassa me,_  
_thalaṃ pāṇiñca dassaye._  

</Paragraph>

</Division>
<Division number="759" book="th1" bookNo="19" page="324">

<Paragraph number="1097" type="verses" book="th1">
_Rahadehamasmi ogāḷho,_  
_ahāriyarajamattike;_  
_Māyāusūyasārambha,_  
_thinamiddhamapatthaṭe._  

</Paragraph>

</Division>
<Division number="760" book="th1" bookNo="19" page="324">

<Paragraph number="1098" type="verses" book="th1">
_Uddhaccameghathanitaṃ,_  
_saṃyojanavalāhakaṃ;_  
_Vāhā vahanti kuddiṭṭhiṃ,_  
_saṅkappā rāganissitā._  

</Paragraph>

</Division>
<Division number="761" book="th1" bookNo="19" page="324">

<Paragraph number="1099" type="verses" book="th1">
_Savanti sabbadhi sotā,_  
_latā ubbhijja tiṭṭhati;_  
_Te sote ko nivāreyya,_  
_taṃ lataṃ ko hi checchati._  

</Paragraph>

</Division>
<Division number="762" book="th1" bookNo="19" page="324">

<Paragraph number="1100" type="verses" book="th1">
_Velaṃ karotha bhaddante,_  
_sotānaṃ sannivāraṇaṃ;_  
_Mā te manomayo soto,_  
_rukkhaṃva sahasā luve._  

</Paragraph>

</Division>
<Division number="763" book="th1" bookNo="19" page="324">

<Paragraph number="1101" type="verses" book="th1">
_Evaṃ me bhayajātassa,_  
_apārā pāramesato;_  
_Tāṇo paññāvudho satthā,_  
_isisaṃghanisevito._  

</Paragraph>

</Division>
<Division number="764" book="th1" bookNo="19" page="324">

<Paragraph number="1102" type="verses" book="th1">
_Sopāṇaṃ sugataṃ suddhaṃ,_  
_dhammasāramayaṃ daḷhaṃ;_  
_Pādāsi vuyhamānassa,_  
_‘mā bhāyī’ti ca mabravi._  

</Paragraph>

</Division>
<Division number="765" book="th1" bookNo="19" page="324">

<Paragraph number="1103" type="verses" book="th1">
_Satipaṭṭhānapāsādaṃ,_  
_āruyha paccavekkhisaṃ;_  
_Yaṃ taṃ pubbe amaññissaṃ,_  
_sakkāyābhirataṃ pajaṃ._  

</Paragraph>

</Division>
<Division number="766" book="th1" bookNo="19" page="325">

<Paragraph number="1104" type="verses" book="th1">
_Yadā ca maggamaddakkhiṃ,_  
_nāvāya abhirūhanaṃ;_  
_Anadhiṭṭhāya attānaṃ,_  
_titthamaddakkhimuttamaṃ._  

</Paragraph>

</Division>
<Division number="767" book="th1" bookNo="19" page="325">

<Paragraph number="1105" type="verses" book="th1">
_Sallaṃ attasamuṭṭhānaṃ,_  
_bhavanettippabhāvitaṃ;_  
_Etesaṃ appavattāya,_  
_desesi maggamuttamaṃ._  

</Paragraph>

</Division>
<Division number="768" book="th1" bookNo="19" page="325">

<Paragraph number="1106" type="verses" book="th1">
_Dīgharattānusayitaṃ,_  
_cirarattamadhiṭṭhitaṃ;_  
_Buddho mepānudī ganthaṃ,_  
_visadosappavāhano”ti._  


</Paragraph>
</Division>

… Telakāni thero… .
