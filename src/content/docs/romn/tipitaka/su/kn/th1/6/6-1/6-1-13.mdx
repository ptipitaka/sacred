<Paragraph number="727" type="verses" book="th1">
_“Channamativassati,_  
_vivaṭaṃ nātivassati;_  
_Tasmā channaṃ vivaretha,_  
_evaṃ taṃ nātivassati._

</Division>
<Division number="448" book="th1">

<Paragraph number="728" type="verses" book="th1">
_Maccunābbhahato loko,_  
_jarāya parivārito;_  
_Taṇhāsallena otiṇṇo,_  
_icchādhūpāyito sadā._  

</Paragraph>

</Division>
<Division number="449" book="th1">

<Paragraph number="729" type="verses" book="th1">
_Maccunābbhahato loko,_  
_parikkhitto jarāya ca;_  
_Haññati niccamattāṇo,_  
_pattadaṇḍova takkaro._  

</Paragraph>

</Division>
<Division number="450" book="th1">

<Paragraph number="730" type="verses" book="th1">
_Āgacchantaggikhandhāva,_  
_maccu byādhi jarā tayo;_  
_Paccuggantuṃ balaṃ natthi,_  
_javo natthi palāyituṃ._  

</Paragraph>

</Division>
<Division number="451" book="th1">

<Paragraph number="731" type="verses" book="th1">
_Amoghaṃ divasaṃ kayirā,_  
_appena bahukena vā;_  
_Yaṃ yaṃ vijahate rattiṃ,_  
_tadūnaṃ tassa jīvitaṃ._  

</Paragraph>

</Division>
<Division number="452" book="th1">

<Paragraph number="732" type="verses" book="th1">
_Carato tiṭṭhato vāpi,_  
_āsīnasayanassa vā;_  
_Upeti carimā ratti,_  
_na te kālo pamajjitun”ti._  


</Paragraph>
</Division>
