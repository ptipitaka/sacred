<Division number="866" book="th1">

<Paragraph number="1209" type="verses" book="th1">
_“Gacchaṃ vadesi samaṇaṭṭhitomhi,_  
_Mamañca brūsi ṭhitamaṭṭhitoti;_  
_Pucchāmi taṃ samaṇa etamatthaṃ,_  
_‘Kathaṃ ṭhito tvaṃ ahamaṭṭhitomhi’”._  

</Paragraph>

</Division>
<Division number="867" book="th1">

<Paragraph number="1210" type="verses" book="th1">
_“Ṭhito ahaṃ aṅgulimāla sabbadā,_  
_Sabbesu bhūtesu nidhāya daṇḍaṃ;_  
_Tuvañca pāṇesu asaññatosi,_  
_Tasmā ṭhitohaṃ tuvamaṭṭhitosi”._  

</Paragraph>

</Division>
<Division number="868" book="th1">

<Paragraph number="1211" type="verses" book="th1">
_“Cirassaṃ vata me mahito mahesī,_  
_Mahāvanaṃ samaṇo paccapādi;_  
_Sohaṃ cajissāmi sahassapāpaṃ,_  
_Sutvāna gāthaṃ tava dhammayuttaṃ”._  

</Paragraph>

</Division>
<Division number="869" book="th1">

<Paragraph number="1212" type="verses" book="th1">
_Icceva coro asimāvudhañca,_  
_Sobbhe papāte narake anvakāsi;_  
_Avandi coro sugatassa pāde,_  
_Tattheva pabbajjamayāci buddhaṃ._  

</Paragraph>

</Division>
<Division number="870" book="th1">

<Paragraph number="1213" type="verses" book="th1">
_Buddho ca kho kāruṇiko mahesi,_  
_Yo satthā lokassa sadevakassa;_  
_“Tamehi bhikkhū”ti tadā avoca,_  
_Eseva tassa ahu bhikkhubhāvo._  

</Paragraph>

</Division>
<Division number="871" book="th1">

<Paragraph number="1214" type="verses" book="th1">
_“Yo ca pubbe pamajjitvā,_  
_pacchā so nappamajjati;_  
_Somaṃ lokaṃ pabhāseti,_  
_abbhā muttova candimā._  

</Paragraph>

</Division>
<Division number="872" book="th1">

<Paragraph number="1215" type="verses" book="th1">
_Yassa pāpaṃ kataṃ kammaṃ,_  
_kusalena pidhīyati;_  
_Somaṃ lokaṃ pabhāseti,_  
_abbhā muttova candimā._  

</Paragraph>

</Division>
<Division number="873" book="th1">

<Paragraph number="1216" type="verses" book="th1">
_Yo have daharo bhikkhu,_  
_yuñjati buddhasāsane;_  
_Somaṃ lokaṃ pabhāseti,_  
_abbhā muttova candimā._  

</Paragraph>

</Division>
<Division number="874" book="th1">

<Paragraph number="1217" type="verses" book="th1">
_Disāpi me dhammakathaṃ suṇantu,_  
_Disāpi me yuñjantu buddhasāsane;_  
_Disāpi me te manuje bhajantu,_  
_Ye dhammamevādapayanti santo._  

</Paragraph>

</Division>
<Division number="875" book="th1">

<Paragraph number="1218" type="verses" book="th1">
_Disā hi me khantivādānaṃ,_  
_avirodhappasaṃsinaṃ;_  
_Suṇantu dhammaṃ kālena,_  
_tañca anuvidhīyantu._  

</Paragraph>

</Division>
<Division number="876" book="th1">

<Paragraph number="1219" type="verses" book="th1">
_Na hi jātu so mamaṃ hiṃse,_  
_aññaṃ vā pana kiñcanaṃ;_  
_Pappuyya paramaṃ santiṃ,_  
_rakkheyya tasathāvare._  

</Paragraph>

</Division>
<Division number="877" book="th1">

<Paragraph number="1220" type="verses" book="th1">
_Udakañhi nayanti nettikā,_  
_Usukārā namayanti tejanaṃ;_  
_Dāruṃ namayanti tacchakā,_  
_Attānaṃ damayanti paṇḍitā._  

</Paragraph>

</Division>
<Division number="878" book="th1">

<Paragraph number="1221" type="verses" book="th1">
_Daṇḍeneke damayanti,_  
_aṅkusebhi kasāhi ca;_  
_Adaṇḍena asatthena,_  
_ahaṃ dantomhi tādinā._  

</Paragraph>

</Division>
<Division number="879" book="th1">

<Paragraph number="1222" type="verses" book="th1">
_‘Ahiṃsako’ti me nāmaṃ,_  
_hiṃsakassa pure sato;_  
_Ajjāhaṃ saccanāmomhi,_  
_na naṃ hiṃsāmi kiñcanaṃ._  

</Paragraph>

</Division>
<Division number="880" book="th1">

<Paragraph number="1223" type="verses" book="th1">
_Coro ahaṃ pure āsiṃ,_  
_aṅgulimāloti vissuto;_  
_Vuyhamāno mahoghena,_  
_buddhaṃ saraṇamāgamaṃ._  

</Paragraph>

</Division>
<Division number="881" book="th1">

<Paragraph number="1224" type="verses" book="th1">
_Lohitapāṇi pure āsiṃ,_  
_aṅgulimāloti vissuto;_  
_Saraṇagamanaṃ passa,_  
_bhavanetti samūhatā._  

</Paragraph>

</Division>
<Division number="882" book="th1">

<Paragraph number="1225" type="verses" book="th1">
_Tādisaṃ kammaṃ katvāna,_  
_bahuṃ duggatigāminaṃ;_  
_Phuṭṭho kammavipākena,_  
_anaṇo bhuñjāmi bhojanaṃ._  

</Paragraph>

</Division>
<Division number="883" book="th1">

<Paragraph number="1226" type="verses" book="th1">
_Pamādamanuyuñjanti,_  
_bālā dummedhino janā;_  
_Appamādañca medhāvī,_  
_dhanaṃ seṭṭhaṃva rakkhati._  

</Paragraph>

</Division>
<Division number="884" book="th1">

<Paragraph number="1227" type="verses" book="th1">
_Mā pamādamanuyuñjetha,_  
_mā kāmaratisanthavaṃ;_  
_Appamatto hi jhāyanto,_  
_pappoti paramaṃ sukhaṃ._  

</Paragraph>

</Division>
<Division number="885" book="th1">

<Paragraph number="1228" type="verses" book="th1">
_Svāgataṃ nāpagataṃ,_  
_netaṃ dummantitaṃ mama;_  
_Savibhattesu dhammesu,_  
_yaṃ seṭṭhaṃ tadupāgamaṃ._  

</Paragraph>

</Division>
<Division number="886" book="th1">

<Paragraph number="1229" type="verses" book="th1">
_Svāgataṃ nāpagataṃ,_  
_netaṃ dummantitaṃ mama;_  
_Tisso vijjā anuppattā,_  
_kataṃ buddhassa sāsanaṃ._  

</Paragraph>

</Division>
<Division number="887" book="th1">

<Paragraph number="1230" type="verses" book="th1">
_Araññe rukkhamūle vā,_  
_pabbatesu guhāsu vā;_  
_Tattha tattheva aṭṭhāsiṃ,_  
_ubbiggamanaso tadā._  

</Paragraph>

</Division>
<Division number="888" book="th1">

<Paragraph number="1231" type="verses" book="th1">
_Sukhaṃ sayāmi ṭhāyāmi,_  
_sukhaṃ kappemi jīvitaṃ;_  
_Ahatthapāso mārassa,_  
_aho satthānukampito._  

</Paragraph>

</Division>
<Division number="889" book="th1">

<Paragraph number="1232" type="verses" book="th1">
_Brahmajacco pure āsiṃ,_  
_udicco ubhato ahu;_  
_Sojja putto sugatassa,_  
_dhammarājassa satthuno._  

</Paragraph>

</Division>
<Division number="890" book="th1">

<Paragraph number="1233" type="verses" book="th1">
_Vītataṇho anādāno,_  
_guttadvāro susaṃvuto;_  
_Aghamūlaṃ vadhitvāna,_  
_patto me āsavakkhayo._  

</Paragraph>

</Division>
<Division number="891" book="th1">

<Paragraph number="1234" type="verses" book="th1">
_Pariciṇṇo mayā satthā,_  
_kataṃ buddhassa sāsanaṃ;_  
_Ohito garuko bhāro,_  
_bhavanetti samūhatā”ti._  


</Paragraph>
</Division>
