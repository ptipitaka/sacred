</Paragraph>

<Division number="506" book="th1">

<Paragraph number="800" type="verses" book="th1">

_amāyo rittapesuṇo;_  
_Kalyāṇapañño so bhikkhu,_  
_evaṃ pecca na socati._  

</Paragraph>

</Division>
<Division number="507" book="th1">

<Paragraph number="801" type="verses" book="th1">
_Yassa saddhā tathāgate,_  
_acalā suppatiṭṭhitā;_  
_Sīlañca yassa kalyāṇaṃ,_  
_ariyakantaṃ pasaṃsitaṃ._  

</Paragraph>

</Division>
<Division number="508" book="th1">

<Paragraph number="802" type="verses" book="th1">
_Saṃghe pasādo yassatthi,_  
_ujubhūtañca dassanaṃ;_  
_‘Adaliddo’ti taṃ āhu,_  
_amoghaṃ tassa jīvitaṃ._  

</Paragraph>

</Division>
<Division number="509" book="th1">

_Anuyuñjetha medhāvī,_  
_saraṃ buddhāna sāsanan”ti._  


</Paragraph>
