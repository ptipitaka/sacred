_Pāsādike bahū disvā,_  
_bhāvitatte susaṃvute;_  
_Isi paṇḍarasagotto,_  
_apucchi phussasavhayaṃ._  

</Paragraph>

</Division>
<Division number="950" book="th1" bookNo="19" page="341">

<Paragraph number="1299" type="verses" book="th1">
_“Kiṃchandā kimadhippāyā,_  
_kimākappā bhavissare;_  
_Anāgatamhi kālamhi,_  
_taṃ me akkhāhi pucchito”._  

</Paragraph>

</Division>
<Division number="951" book="th1" bookNo="19" page="341">

<Paragraph number="1300" type="verses" book="th1">
_“Suṇohi vacanaṃ mayhaṃ,_  
_isipaṇḍarasavhaya;_  
_Sakkaccaṃ upadhārehi,_  
_ācikkhissāmyanāgataṃ._  

</Paragraph>

</Division>
<Division number="952" book="th1" bookNo="19" page="341">

<Paragraph number="1301" type="verses" book="th1">
_Kodhanā upanāhī ca,_  
_makkhī thambhī saṭhā bahū;_  
_Ussukī nānāvādā ca,_  
_bhavissanti anāgate._  

</Paragraph>

</Division>
<Division number="953" book="th1" bookNo="19" page="341">

<Paragraph number="1302" type="verses" book="th1">
_Aññātamānino dhamme,_  
_gambhīre tīragocarā;_  
_Lahukā agarū dhamme,_  
_aññamaññamagāravā._  

</Paragraph>

</Division>
<Division number="954" book="th1" bookNo="19" page="341">

<Paragraph number="1303" type="verses" book="th1">
_Bahū ādīnavā loke,_  
_uppajjissantyanāgate;_  
_Sudesitaṃ imaṃ dhammaṃ,_  
_kilesessanti dummatī._  

</Paragraph>

</Division>
<Division number="955" book="th1" bookNo="19" page="341">

<Paragraph number="1304" type="verses" book="th1">
_Guṇahīnāpi saṃghamhi,_  
_voharantā visāradā;_  
_Balavanto bhavissanti,_  
_mukharā assutāvino._  

</Paragraph>

</Division>
<Division number="956" book="th1" bookNo="19" page="341">

<Paragraph number="1305" type="verses" book="th1">
_Guṇavantopi saṃghamhi,_  
_voharantā yathātthato;_  
_Dubbalā te bhavissanti,_  
_hirīmanā anatthikā._  

</Paragraph>

</Division>
<Division number="957" book="th1" bookNo="19" page="341">

<Paragraph number="1306" type="verses" book="th1">
_Rajataṃ jātarūpañca,_  
_khettaṃ vatthumajeḷakaṃ;_  
_Dāsidāsañca dummedhā,_  
_sādiyissantyanāgate._  

</Paragraph>

</Division>
<Division number="958" book="th1" bookNo="19" page="341">

<Paragraph number="1307" type="verses" book="th1">
_Ujjhānasaññino bālā,_  
_sīlesu asamāhitā;_  
_Unnaḷā vicarissanti,_  
_kalahābhiratā magā._  

</Paragraph>

</Division>
<Division number="959" book="th1" bookNo="19" page="341">

<Paragraph number="1308" type="verses" book="th1">
_Uddhatā ca bhavissanti,_  
_nīlacīvarapārutā;_  
_Kuhā thaddhā lapā siṅgī,_  
_carissantyariyā viya._  

</Paragraph>

</Division>
<Division number="960" book="th1" bookNo="19" page="341">

<Paragraph number="1309" type="verses" book="th1">
_Telasaṇṭhehi kesehi,_  
_capalā añjanakkhikā;_  
_Rathiyāya gamissanti,_  
_dantavaṇṇikapārutā._  

</Paragraph>

</Division>
<Division number="961" book="th1" bookNo="19" page="342">

<Paragraph number="1310" type="verses" book="th1">
_Ajegucchaṃ vimuttehi,_  
_surattaṃ arahaddhajaṃ;_  
_Jigucchissanti kāsāvaṃ,_  
_odātesu samucchitā._  

</Paragraph>

</Division>
<Division number="962" book="th1" bookNo="19" page="342">

<Paragraph number="1311" type="verses" book="th1">
_Lābhakāmā bhavissanti,_  
_kusītā hīnavīriyā;_  
_Kicchantā vanapatthāni,_  
_gāmantesu vasissare._  

</Paragraph>

</Division>
<Division number="963" book="th1" bookNo="19" page="342">

<Paragraph number="1312" type="verses" book="th1">
_Ye ye lābhaṃ labhissanti,_  
_micchājīvaratā sadā;_  
_Te teva anusikkhantā,_  
_bhajissanti asaṃyatā._  

</Paragraph>

</Division>
<Division number="964" book="th1" bookNo="19" page="342">

<Paragraph number="1313" type="verses" book="th1">
_Ye ye alābhino lābhaṃ,_  
_na te pujjā bhavissare;_  
_Supesalepi te dhīre,_  
_sevissanti na te tadā._  

</Paragraph>

</Division>
<Division number="965" book="th1" bookNo="19" page="342">

<Paragraph number="1314" type="verses" book="th1">
_Milakkhurajanaṃ rattaṃ,_  
_garahantā sakaṃ dhajaṃ;_  
_Titthiyānaṃ dhajaṃ keci,_  
_dhārissantyavadātakaṃ._  

</Paragraph>

</Division>
<Division number="966" book="th1" bookNo="19" page="342">

<Paragraph number="1315" type="verses" book="th1">
_Agāravo ca kāsāve,_  
_tadā tesaṃ bhavissati;_  
_Paṭisaṅkhā ca kāsāve,_  
_bhikkhūnaṃ na bhavissati._  

</Paragraph>

</Division>
<Division number="967" book="th1" bookNo="19" page="342">

<Paragraph number="1316" type="verses" book="th1">
_Abhibhūtassa dukkhena,_  
_sallaviddhassa ruppato;_  
_Paṭisaṅkhā mahāghorā,_  
_nāgassāsi acintiyā._  

</Paragraph>

</Division>
<Division number="968" book="th1" bookNo="19" page="342">

<Paragraph number="1317" type="verses" book="th1">
_Chaddanto hi tadā disvā,_  
_surattaṃ arahaddhajaṃ;_  
_Tāvadeva bhaṇī gāthā,_  
_gajo atthopasaṃhitā._  

</Paragraph>

</Division>
<Division number="969" book="th1" bookNo="19" page="342">

<Paragraph number="1318" type="verses" book="th1">
_‘Anikkasāvo kāsāvaṃ,_  
_yo vatthaṃ paridhassati;_  
_Apeto damasaccena,_  
_na so kāsāvamarahati._  

</Paragraph>

</Division>
<Division number="970" book="th1" bookNo="19" page="342">

<Paragraph number="1319" type="verses" book="th1">
_Yo ca vantakasāvassa,_  
_sīlesu susamāhito;_  
_Upeto damasaccena,_  
_sa ve kāsāvamarahati._  

</Paragraph>

</Division>
<Division number="971" book="th1" bookNo="19" page="342">

<Paragraph number="1320" type="verses" book="th1">
_Vipannasīlo dummedho,_  
_pākaṭo kāmakāriyo;_  
_Vibbhantacitto nissukko,_  
_na so kāsāvamarahati._  

</Paragraph>

</Division>
<Division number="972" book="th1" bookNo="19" page="342">

<Paragraph number="1321" type="verses" book="th1">
_Yo ca sīlena sampanno,_  
_vītarāgo samāhito;_  
_Odātamanasaṅkappo,_  
_sa ve kāsāvamarahati._  

</Paragraph>

</Division>
<Division number="973" book="th1" bookNo="19" page="342">

<Paragraph number="1322" type="verses" book="th1">
_Uddhato unnaḷo bālo,_  
_sīlaṃ yassa na vijjati;_  
_Odātakaṃ arahati,_  
_kāsāvaṃ kiṃ karissati._  

</Paragraph>

</Division>
<Division number="974" book="th1" bookNo="19" page="343">

<Paragraph number="1323" type="verses" book="th1">
_Bhikkhū ca bhikkhuniyo ca,_  
_duṭṭhacittā anādarā;_  
_Tādīnaṃ mettacittānaṃ,_  
_niggaṇhissantyanāgate._  

</Paragraph>

</Division>
<Division number="975" book="th1" bookNo="19" page="343">

<Paragraph number="1324" type="verses" book="th1">
_Sikkhāpentāpi therehi,_  
_bālā cīvaradhāraṇaṃ;_  
_Na suṇissanti dummedhā,_  
_pākaṭā kāmakāriyā._  

</Paragraph>

</Division>
<Division number="976" book="th1" bookNo="19" page="343">

<Paragraph number="1325" type="verses" book="th1">
_Te tathā sikkhitā bālā,_  
_aññamaññaṃ agāravā;_  
_Nādiyissantupajjhāye,_  
_khaḷuṅko viya sārathiṃ._  

</Paragraph>

</Division>
<Division number="977" book="th1" bookNo="19" page="343">

<Paragraph number="1326" type="verses" book="th1">
_Evaṃ anāgataddhānaṃ,_  
_paṭipatti bhavissati;_  
_Bhikkhūnaṃ bhikkhunīnañca,_  
_patte kālamhi pacchime’._  

</Paragraph>

</Division>
<Division number="978" book="th1" bookNo="19" page="343">

<Paragraph number="1327" type="verses" book="th1">
_Purā āgacchate etaṃ,_  
_anāgataṃ mahabbhayaṃ;_  
_Subbacā hotha sakhilā,_  
_aññamaññaṃ sagāravā._  

</Paragraph>

</Division>
<Division number="979" book="th1" bookNo="19" page="343">

<Paragraph number="1328" type="verses" book="th1">
_Mettacittā kāruṇikā,_  
_Hotha sīlesu saṃvutā;_  
_Āraddhavīriyā pahitattā,_  
_Niccaṃ daḷhaparakkamā._  

</Paragraph>

</Division>
<Division number="980" book="th1" bookNo="19" page="343">

<Paragraph number="1329" type="verses" book="th1">
_Pamādaṃ bhayato disvā,_  
_Appamādañca khemato;_  
_Bhāvethaṭṭhaṅgikaṃ maggaṃ,_  
_Phusantā amataṃ padan”ti._  


</Paragraph>
</Division>
