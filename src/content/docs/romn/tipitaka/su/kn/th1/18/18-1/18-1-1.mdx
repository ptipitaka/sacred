<Paragraph number="1409" type="verses" book="th1">
_“Na gaṇena purakkhato care,_  
_Vimano hoti samādhi dullabho;_  
_Nānājanasaṅgaho dukho,_  
_Iti disvāna gaṇaṃ na rocaye._  

</Paragraph>

</Division>
<Division number="1055" book="th1">

<Paragraph number="1410" type="verses" book="th1">
_Na kulāni upabbaje muni,_  
_Vimano hoti samādhi dullabho;_  
_So ussukko rasānugiddho,_  
_Atthaṃ riñcati yo sukhāvaho._  

</Paragraph>

</Division>
<Division number="1056" book="th1">

<Paragraph number="1411" type="verses" book="th1">
_Paṅkoti hi naṃ avedayuṃ,_  
_Yāyaṃ vandanapūjanā kulesu;_  
_Sukhumaṃ sallaṃ durubbahaṃ,_  
_Sakkāro kāpurisena dujjaho._  

</Paragraph>

</Division>
<Division number="1057" book="th1">

<Paragraph number="1412" type="verses" book="th1">
_Senāsanamhā oruyha,_  
_nagaraṃ piṇḍāya pāvisiṃ;_  
_Bhuñjantaṃ purisaṃ kuṭṭhiṃ,_  
_sakkaccaṃ taṃ upaṭṭhahiṃ._  

</Paragraph>

</Division>
<Division number="1058" book="th1">

<Paragraph number="1413" type="verses" book="th1">
_So me pakkena hatthena,_  
_ālopaṃ upanāmayi;_  
_Ālopaṃ pakkhipantassa,_  
_aṅguli cettha chijjatha._  

</Paragraph>

</Division>
<Division number="1059" book="th1">

<Paragraph number="1414" type="verses" book="th1">
_Kuṭṭamūlañca nissāya,_  
_ālopaṃ taṃ abhuñjisaṃ;_  
_Bhuñjamāne vā bhutte vā,_  
_jegucchaṃ me na vijjati._  

</Paragraph>

</Division>
<Division number="1060" book="th1">

<Paragraph number="1415" type="verses" book="th1">
_Uttiṭṭhapiṇḍo āhāro,_  
_pūtimuttañca osadhaṃ;_  
_Senāsanaṃ rukkhamūlaṃ,_  
_paṃsukūlañca cīvaraṃ;_  
_Yassete abhisambhutvā,_  
_sa ve cātuddiso naro._  

</Paragraph>

</Division>
<Division number="1061" book="th1">

<Paragraph number="1416" type="verses" book="th1">
_Yattha eke vihaññanti,_  
_āruhantā siluccayaṃ;_  
_Tassa buddhassa dāyādo,_  
_sampajāno patissato;_  
_Iddhibalenupatthaddho,_  
_kassapo abhirūhati._  

</Paragraph>

</Division>
<Division number="1062" book="th1">

<Paragraph number="1417" type="verses" book="th1">
_Piṇḍapātapaṭikkanto,_  
_selamāruyha kassapo;_  
_Jhāyati anupādāno,_  
_pahīnabhayabheravo._  

</Paragraph>

</Division>
<Division number="1063" book="th1">

<Paragraph number="1418" type="verses" book="th1">
_Piṇḍapātapaṭikkanto,_  
_selamāruyha kassapo;_  
_Jhāyati anupādāno,_  
_ḍayhamānesu nibbuto._  

</Paragraph>

</Division>
<Division number="1064" book="th1">

<Paragraph number="1419" type="verses" book="th1">
_Piṇḍapātapaṭikkanto,_  
_selamāruyha kassapo;_  
_Jhāyati anupādāno,_  
_katakicco anāsavo._  

</Paragraph>

</Division>
<Division number="1065" book="th1">

<Paragraph number="1420" type="verses" book="th1">
_Karerimālāvitatā,_  
_bhūmibhāgā manoramā;_  
_Kuñjarābhirudā rammā,_  
_te selā ramayanti maṃ._  

</Paragraph>

</Division>
<Division number="1066" book="th1">

<Paragraph number="1421" type="verses" book="th1">
_Nīlabbhavaṇṇā rucirā,_  
_vārisītā sucindharā;_  
_Indagopakasañchannā,_  
_te selā ramayanti maṃ._  

</Paragraph>

</Division>
<Division number="1067" book="th1">

<Paragraph number="1422" type="verses" book="th1">
_Nīlabbhakūṭasadisā,_  
_kūṭāgāravarūpamā;_  
_Vāraṇābhirudā rammā,_  
_te selā ramayanti maṃ._  

</Paragraph>

</Division>
<Division number="1068" book="th1">

<Paragraph number="1423" type="verses" book="th1">
_Abhivuṭṭhā rammatalā,_  
_nagā isibhi sevitā;_  
_Abbhunnaditā sikhīhi,_  
_te selā ramayanti maṃ._  

</Paragraph>

</Division>
<Division number="1069" book="th1">

<Paragraph number="1424" type="verses" book="th1">
_Alaṃ jhāyitukāmassa,_  
_pahitattassa me sato;_  
_Alaṃ me atthakāmassa,_  
_pahitattassa bhikkhuno._  

</Paragraph>

</Division>
<Division number="1070" book="th1">

<Paragraph number="1425" type="verses" book="th1">
_Alaṃ me phāsukāmassa,_  
_pahitattassa bhikkhuno;_  
_Alaṃ me yogakāmassa,_  
_pahitattassa tādino._  

</Paragraph>

</Division>
<Division number="1071" book="th1">

<Paragraph number="1426" type="verses" book="th1">
_Umāpupphena samānā,_  
_gaganāvabbhachāditā;_  
_Nānādijagaṇākiṇṇā,_  
_te selā ramayanti maṃ._  

</Paragraph>

</Division>
<Division number="1072" book="th1">

<Paragraph number="1427" type="verses" book="th1">
_Anākiṇṇā gahaṭṭhehi,_  
_migasaṅghanisevitā;_  
_Nānādijagaṇākiṇṇā,_  
_te selā ramayanti maṃ._  

</Paragraph>

</Division>
<Division number="1073" book="th1">

<Paragraph number="1428" type="verses" book="th1">
_Acchodikā puthusilā,_  
_gonaṅgulamigāyutā;_  
_Ambusevālasañchannā,_  
_te selā ramayanti maṃ._  

</Paragraph>

</Division>
<Division number="1074" book="th1">

<Paragraph number="1429" type="verses" book="th1">
_Na pañcaṅgikena turiyena,_  
_Rati me hoti tādisī;_  
_Yathā ekaggacittassa,_  
_Sammā dhammaṃ vipassato._  

</Paragraph>

</Division>
<Division number="1075" book="th1">

<Paragraph number="1430" type="verses" book="th1">
_Kammaṃ bahukaṃ na kāraye,_  
_Parivajjeyya janaṃ na uyyame;_  
_Ussukko so rasānugiddho,_  
_Atthaṃ riñcati yo sukhāvaho._  

</Paragraph>

</Division>
<Division number="1076" book="th1">

<Paragraph number="1431" type="verses" book="th1">
_Kammaṃ bahukaṃ na kāraye,_  
_Parivajjeyya anattaneyyametaṃ;_  
_Kicchati kāyo kilamati,_  
_Dukkhito so samathaṃ na vindati._  

</Paragraph>

</Division>
<Division number="1077" book="th1">

<Paragraph number="1432" type="verses" book="th1">
_Oṭṭhappahatamattena,_  
_attānampi na passati;_  
_Patthaddhagīvo carati,_  
_ahaṃ seyyoti maññati._  

</Paragraph>

</Division>
<Division number="1078" book="th1">

<Paragraph number="1433" type="verses" book="th1">
_Aseyyo seyyasamānaṃ,_  
_bālo maññati attānaṃ;_  
_Na taṃ viññū pasaṃsanti,_  
_patthaddhamānasaṃ naraṃ._  

</Paragraph>

</Division>
<Division number="1079" book="th1">

<Paragraph number="1434" type="verses" book="th1">
_Yo ca seyyohamasmīti,_  
_nāhaṃ seyyoti vā pana;_  
_Hīno taṃsadiso vāti,_  
_vidhāsu na vikampati._  

</Paragraph>

</Division>
<Division number="1080" book="th1">

<Paragraph number="1435" type="verses" book="th1">
_Paññavantaṃ tathā tādiṃ,_  
_sīlesu susamāhitaṃ;_  
_Cetosamathamanuttaṃ,_  
_tañce viññū pasaṃsare._  

</Paragraph>

</Division>
<Division number="1081" book="th1">

<Paragraph number="1436" type="verses" book="th1">
_Yassa sabrahmacārīsu,_  
_gāravo nūpalabbhati;_  
_Ārakā hoti saddhammā,_  
_nabhato puthavī yathā._  

</Paragraph>

</Division>
<Division number="1082" book="th1">

<Paragraph number="1437" type="verses" book="th1">
_Yesañca hiriottappaṃ,_  
_sadā sammā upaṭṭhitaṃ;_  
_Virūḷhabrahmacariyā te,_  
_tesaṃ khīṇā punabbhavā._  

</Paragraph>

</Division>
<Division number="1083" book="th1">

<Paragraph number="1438" type="verses" book="th1">
_Uddhato capalo bhikkhu,_  
_paṃsukūlena pāruto;_  
_Kapīva sīhacammena,_  
_na so tenupasobhati._  

</Paragraph>

</Division>
<Division number="1084" book="th1">

<Paragraph number="1439" type="verses" book="th1">
_Anuddhato acapalo,_  
_nipako saṃvutindriyo;_  
_Sobhati paṃsukūlena,_  
_sīhova girigabbhare._  

</Paragraph>

</Division>
<Division number="1085" book="th1">

<Paragraph number="1440" type="verses" book="th1">
_Ete sambahulā devā,_  
_iddhimanto yasassino;_  
_Dasadevasahassāni,_  
_sabbe te brahmakāyikā._  

</Paragraph>

</Division>
<Division number="1086" book="th1">

<Paragraph number="1441" type="verses" book="th1">
_Dhammasenāpatiṃ vīraṃ,_  
_mahājhāyiṃ samāhitaṃ;_  
_Sāriputtaṃ namassantā,_  
_tiṭṭhanti pañjalīkatā._  

</Paragraph>

</Division>
<Division number="1087" book="th1">

<Paragraph number="1442" type="verses" book="th1">
_‘Namo te purisājañña,_  
_namo te purisuttama;_  
_Yassa te nābhijānāma,_  
_yampi nissāya jhāyati._  

</Paragraph>

</Division>
<Division number="1088" book="th1">

<Paragraph number="1443" type="verses" book="th1">
_Accheraṃ vata buddhānaṃ,_  
_gambhīro gocaro sako;_  
_Ye mayaṃ nābhijānāma,_  
_vālavedhisamāgatā’._  

</Paragraph>

</Division>
<Division number="1089" book="th1">

<Paragraph number="1444" type="verses" book="th1">
_Taṃ tathā devakāyehi,_  
_pūjitaṃ pūjanārahaṃ;_  
_Sāriputtaṃ tadā disvā,_  
_kappinassa sitaṃ ahu._  

</Paragraph>

</Division>
<Division number="1090" book="th1">

<Paragraph number="1445" type="verses" book="th1">
_Yāvatā buddhakhettamhi,_  
_ṭhapayitvā mahāmuniṃ;_  
_Dhutaguṇe visiṭṭhohaṃ,_  
_sadiso me na vijjati._  

</Paragraph>

</Division>
<Division number="1091" book="th1">

<Paragraph number="1446" type="verses" book="th1">
_Pariciṇṇo mayā satthā,_  
_kataṃ buddhassa sāsanaṃ;_  
_Ohito garuko bhāro,_  
_natthi dāni punabbhavo._  

</Paragraph>

</Division>
<Division number="1092" book="th1">

<Paragraph number="1447" type="verses" book="th1">
_Na cīvare na sayane,_  
_bhojane nupalimpati;_  
_Gotamo anappameyyo,_  
_muḷālapupphaṃ vimalaṃva;_  
_Ambunā nekkhammaninno,_  
_tibhavābhinissaṭo._  

</Paragraph>

</Division>
<Division number="1093" book="th1">

<Paragraph number="1448" type="verses" book="th1">
_Satipaṭṭhānagīvo so,_  
_saddhāhattho mahāmuni;_  
_Paññāsīso mahāñāṇī,_  
_sadā carati nibbuto”ti._  


</Paragraph>
