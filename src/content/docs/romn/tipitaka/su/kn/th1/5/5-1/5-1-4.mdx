<Paragraph number="585" type="verses" book="th1">
_“Yaṃ patthayāno dhammesu,_  
_upajjhāyo anuggahi;_  
_Amataṃ abhikaṅkhantaṃ,_  
_kataṃ kattabbakaṃ mayā._  

</Paragraph>

</Division>
<Division number="331" book="th1">

<Paragraph number="586" type="verses" book="th1">
_Anuppatto sacchikato,_  
_sayaṃ dhammo anītiho;_  
_Visuddhañāṇo nikkaṅkho,_  
_byākaromi tavantike._  

</Paragraph>

</Division>
<Division number="332" book="th1">

<Paragraph number="587" type="verses" book="th1">
_Pubbenivāsaṃ jānāmi,_  
_dibbacakkhu visodhitaṃ;_  
_Sadattho me anuppatto,_  
_kataṃ buddhassa sāsanaṃ._  

</Paragraph>

</Division>
<Division number="333" book="th1">

<Paragraph number="588" type="verses" book="th1">
_Appamattassa me sikkhā,_  
_sussutā tava sāsane;_  
_Sabbe me āsavā khīṇā,_  
_natthi dāni punabbhavo._  

</Paragraph>

</Division>
<Division number="334" book="th1">

<Paragraph number="589" type="verses" book="th1">
_Anusāsi maṃ ariyavatā,_  
_Anukampi anuggahi;_  
_Amogho tuyhamovādo,_  
_Antevāsimhi sikkhito”ti._  


</Paragraph>
</Division>
