<Paragraph number="805" type="verses" book="th1">
_“Yadā paṭhamamaddakkhiṃ,_  
_satthāramakutobhayaṃ;_  
_Tato me ahu saṃvego,_  
_passitvā purisuttamaṃ._  

</Paragraph>

</Division>
<Division number="511" book="th1" bookNo="19" page="296">

<Paragraph number="806" type="verses" book="th1">
_Siriṃ hatthehi pādehi,_  
_yo paṇāmeyya āgataṃ;_  
_Etādisaṃ so satthāraṃ,_  
_ārādhetvā virādhaye._  

</Paragraph>

</Division>
<Division number="512" book="th1" bookNo="19" page="296">

<Paragraph number="807" type="verses" book="th1">
_Tadāhaṃ puttadārañca,_  
_dhanadhaññañca chaḍḍayiṃ;_  
_Kesamassūni chedetvā,_  
_pabbajiṃ anagāriyaṃ._  

</Paragraph>

</Division>
<Division number="513" book="th1" bookNo="19" page="296">

<Paragraph number="808" type="verses" book="th1">
_Sikkhāsājīvasampanno,_  
_indriyesu susaṃvuto;_  
_Namassamāno sambuddhaṃ,_  
_vihāsiṃ aparājito._  

</Paragraph>

</Division>
<Division number="514" book="th1" bookNo="19" page="296">

<Paragraph number="809" type="verses" book="th1">
_Tato me paṇidhī āsi,_  
_cetaso abhipatthito;_  
_Na nisīde muhuttampi,_  
_taṇhāsalle anūhate._  

</Paragraph>

</Division>
<Division number="515" book="th1" bookNo="19" page="296">

<Paragraph number="810" type="verses" book="th1">
_Tassa mevaṃ viharato,_  
_passa vīriyaparakkamaṃ;_  
_Tisso vijjā anuppattā,_  
_kataṃ buddhassa sāsanaṃ._  

</Paragraph>

</Division>
<Division number="516" book="th1" bookNo="19" page="297">

<Paragraph number="811" type="verses" book="th1">
_Pubbenivāsaṃ jānāmi,_  
_dibbacakkhu visodhitaṃ;_  
_Arahā dakkhiṇeyyomhi,_  
_vippamutto nirūpadhi._  

</Paragraph>

</Division>
<Division number="517" book="th1" bookNo="19" page="297">

<Paragraph number="812" type="verses" book="th1">
_Tato ratyā vivasāne,_  
_sūriyuggamanaṃ pati;_  
_Sabbaṃ taṇhaṃ visosetvā,_  
_pallaṅkena upāvisin”ti._  


</Paragraph>
</Division>

…  Mahāpanthako thero… .



</Paragraph>

<Paragraph number="814" type="center" book="th1">

</Paragraph>

<Paragraph number="815" type="center" book="th1">
Tatruddānaṃ


</Paragraph>

<Paragraph number="816" type="verses" book="th1">
_Mahākaccāyano thero,_  
_sirimitto mahāpanthako;_  
_Ete aṭṭhanipātamhi,_  
_gāthāyo catuvīsatīti._
</Paragraph>
