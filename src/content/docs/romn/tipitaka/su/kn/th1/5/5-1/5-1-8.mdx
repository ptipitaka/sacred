<Paragraph number="609" type="verses" book="th1">
_“Vātarogābhinīto tvaṃ,_  
_viharaṃ kānane vane;_  
_Paviṭṭhagocare lūkhe,_  
_kathaṃ bhikkhu karissasi”._  

</Paragraph>

</Division>
<Division number="351" book="th1" bookNo="19" page="278">

<Paragraph number="610" type="verses" book="th1">
_“Pītisukhena vipulena,_  
_pharamāno samussayaṃ;_  
_Lūkhampi abhisambhonto,_  
_viharissāmi kānane._  

</Paragraph>

</Division>
<Division number="352" book="th1" bookNo="19" page="278">

<Paragraph number="611" type="verses" book="th1">
_Bhāvento satipaṭṭhāne,_  
_indriyāni balāni ca;_  
_Bojjhaṅgāni ca bhāvento,_  
_viharissāmi kānane._  

</Paragraph>

</Division>
<Division number="353" book="th1" bookNo="19" page="278">

<Paragraph number="612" type="verses" book="th1">
_Āraddhavīriye pahitatte,_  
_Niccaṃ daḷhaparakkame;_  
_Samagge sahite disvā,_  
_Viharissāmi kānane._  

</Paragraph>

</Division>
<Division number="354" book="th1" bookNo="19" page="278">

<Paragraph number="613" type="verses" book="th1">
_Anussaranto sambuddhaṃ,_  
_aggaṃ dantaṃ samāhitaṃ;_  
_Atandito rattindivaṃ,_  
_viharissāmi kānane”ti._  


</Paragraph>
</Division>
