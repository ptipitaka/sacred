<Division number="518" book="th1">

<Paragraph number="817" type="verses" book="th1">
_“Yadā dukkhaṃ jarāmaraṇanti paṇḍito,_  
_Aviddasū yattha sitā puthujjanā;_  
_Dukkhaṃ pariññāya satova jhāyati,_  
_Tato ratiṃ paramataraṃ na vindati._  

</Paragraph>

</Division>
<Division number="519" book="th1">

<Paragraph number="818" type="verses" book="th1">
_Yadā dukkhassāvahaniṃ visattikaṃ,_  
_Papañcasaṅghātadukhādhivāhiniṃ;_  
_Taṇhaṃ pahantvāna satova jhāyati,_  
_Tato ratiṃ paramataraṃ na vindati._  

</Paragraph>

</Division>
<Division number="520" book="th1">

<Paragraph number="819" type="verses" book="th1">
_Yadā sivaṃ dvecaturaṅgagāminaṃ,_  
_Magguttamaṃ sabbakilesasodhanaṃ;_  
_Paññāya passitva satova jhāyati,_  
_Tato ratiṃ paramataraṃ na vindati._  

</Paragraph>

</Division>
<Division number="521" book="th1">

<Paragraph number="820" type="verses" book="th1">
_Yadā asokaṃ virajaṃ asaṅkhataṃ,_  
_Santaṃ padaṃ sabbakilesasodhanaṃ;_  
_Bhāveti saṃyojanabandhanacchidaṃ,_  
_Tato ratiṃ paramataraṃ na vindati._  

</Paragraph>

</Division>
<Division number="522" book="th1">

<Paragraph number="821" type="verses" book="th1">
_Yadā nabhe gajjati meghadundubhi,_  
_Dhārākulā vihagapathe samantato;_  
_Bhikkhū ca pabbhāragatova jhāyati,_  
_Tato ratiṃ paramataraṃ na vindati._  

</Paragraph>

</Division>
<Division number="523" book="th1">

<Paragraph number="822" type="verses" book="th1">
_Yadā nadīnaṃ kusumākulānaṃ,_  
_Vicitta-vāneyya-vaṭaṃsakānaṃ;_  
_Tīre nisinno sumanova jhāyati,_  
_Tato ratiṃ paramataraṃ na vindati._  

</Paragraph>

</Division>
<Division number="524" book="th1">

<Paragraph number="823" type="verses" book="th1">
_Yadā nisīthe rahitamhi kānane,_  
_Deve gaḷantamhi nadanti dāṭhino;_  
_Bhikkhū ca pabbhāragatova jhāyati,_  
_Tato ratiṃ paramataraṃ na vindati._  

</Paragraph>

</Division>
<Division number="525" book="th1">

<Paragraph number="824" type="verses" book="th1">
_Yadā vitakke uparundhiyattano,_  
_Nagantare nagavivaraṃ samassito;_  
_Vītaddaro vītakhilova jhāyati,_  
_Tato ratiṃ paramataraṃ na vindati._  

</Paragraph>

</Division>
<Division number="526" book="th1">

<Paragraph number="825" type="verses" book="th1">
_Yadā sukhī malakhilasokanāsano,_  
_Niraggaḷo nibbanatho visallo;_  
_Sabbāsave byantikatova jhāyati,_  
_Tato ratiṃ paramataraṃ na vindatī”ti._  


</Paragraph>
</Division>
