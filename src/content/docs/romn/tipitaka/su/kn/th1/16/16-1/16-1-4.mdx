<Paragraph number="1108" type="verses" book="th1">
_“Passa cittakataṃ bimbaṃ,_  
_arukāyaṃ samussitaṃ;_  
_Āturaṃ bahusaṅkappaṃ,_  
_yassa natthi dhuvaṃ ṭhiti._  

</Paragraph>

</Division>
<Division number="770" book="th1" bookNo="19" page="325">

<Paragraph number="1109" type="verses" book="th1">
_Passa cittakataṃ rūpaṃ,_  
_maṇinā kuṇḍalena ca;_  
_Aṭṭhiṃ tacena onaddhaṃ,_  
_saha vatthehi sobhati._  

</Paragraph>

</Division>
<Division number="771" book="th1" bookNo="19" page="325">

<Paragraph number="1110" type="verses" book="th1">
_Alattakakatā pādā,_  
_mukhaṃ cuṇṇakamakkhitaṃ;_  
_Alaṃ bālassa mohāya,_  
_no ca pāragavesino._  

</Paragraph>

</Division>
<Division number="772" book="th1" bookNo="19" page="325">

<Paragraph number="1111" type="verses" book="th1">
_Aṭṭhapadakatā kesā,_  
_nettā añjanamakkhitā;_  
_Alaṃ bālassa mohāya,_  
_no ca pāragavesino._  

</Paragraph>

</Division>
<Division number="773" book="th1" bookNo="19" page="325">

<Paragraph number="1112" type="verses" book="th1">
_Añjanīva navā cittā,_  
_pūtikāyo alaṅkato;_  
_Alaṃ bālassa mohāya,_  
_no ca pāragavesino._  

</Paragraph>

</Division>
<Division number="774" book="th1" bookNo="19" page="325">

<Paragraph number="1113" type="verses" book="th1">
_Odahi migavo pāsaṃ,_  
_nāsadā vāguraṃ migo;_  
_Bhutvā nivāpaṃ gacchāma,_  
_kandante migabandhake._  

</Paragraph>

</Division>
<Division number="775" book="th1" bookNo="19" page="325">

<Paragraph number="1114" type="verses" book="th1">
_Chinno pāso migavassa,_  
_nāsadā vāguraṃ migo;_  
_Bhutvā nivāpaṃ gacchāma,_  
_socante migaluddake._  

</Paragraph>

</Division>
<Division number="776" book="th1" bookNo="19" page="325">

<Paragraph number="1115" type="verses" book="th1">
_Passāmi loke sadhane manusse,_  
_Laddhāna vittaṃ na dadanti mohā;_  
_Luddhā dhanaṃ sannicayaṃ karonti,_  
_Bhiyyova kāme abhipatthayanti._  

</Paragraph>

</Division>
<Division number="777" book="th1" bookNo="19" page="326">

<Paragraph number="1116" type="verses" book="th1">
_Rājā pasayhappathaviṃ vijetvā,_  
_Sasāgarantaṃ mahimāvasanto;_  
_Oraṃ samuddassa atittarūpo,_  
_Pāraṃ samuddassapi patthayetha._  

</Paragraph>

</Division>
<Division number="778" book="th1" bookNo="19" page="326">

<Paragraph number="1117" type="verses" book="th1">
_Rājā ca aññe ca bahū manussā,_  
_Avītataṇhā maraṇaṃ upenti;_  
_Ūnāva hutvāna jahanti dehaṃ,_  
_Kāmehi lokamhi na hatthi titti._  

</Paragraph>

</Division>
<Division number="779" book="th1" bookNo="19" page="326">

<Paragraph number="1118" type="verses" book="th1">
_Kandanti naṃ ñātī pakiriya kese,_  
_Aho vatā no amarāti cāhu;_  
_Vatthena naṃ pārutaṃ nīharitvā,_  
_Citaṃ samodhāya tato ḍahanti._  

</Paragraph>

</Division>
<Division number="780" book="th1" bookNo="19" page="326">

<Paragraph number="1119" type="verses" book="th1">
_So ḍayhati sūlehi tujjamāno,_  
_Ekena vatthena pahāya bhoge;_  
_Na mīyamānassa bhavanti tāṇā,_  
_Ñātī ca mittā atha vā sahāyā._  

</Paragraph>

</Division>
<Division number="781" book="th1" bookNo="19" page="326">

<Paragraph number="1120" type="verses" book="th1">
_Dāyādakā tassa dhanaṃ haranti,_  
_Satto pana gacchati yena kammaṃ;_  
_Na mīyamānaṃ dhanamanveti kiñci,_  
_Puttā ca dārā ca dhanañca raṭṭhaṃ._  

</Paragraph>

</Division>
<Division number="782" book="th1" bookNo="19" page="326">

<Paragraph number="1121" type="verses" book="th1">
_Na dīghamāyuṃ labhate dhanena,_  
_Na cāpi vittena jaraṃ vihanti;_  
_Appaṃ hidaṃ jīvitamāhu dhīrā,_  
_Asassataṃ vippariṇāmadhammaṃ._  

</Paragraph>

</Division>
<Division number="783" book="th1" bookNo="19" page="326">

<Paragraph number="1122" type="verses" book="th1">
_Aḍḍhā daliddā ca phusanti phassaṃ,_  
_Bālo ca dhīro ca tatheva phuṭṭho;_  
_Bālo hi bālyā vadhitova seti,_  
_Dhīro ca no vedhati phassaphuṭṭho._  

</Paragraph>

</Division>
<Division number="784" book="th1" bookNo="19" page="327">

<Paragraph number="1123" type="verses" book="th1">
_Tasmā hi paññāva dhanena seyyā,_  
_Yāya vosānamidhādhigacchati;_  
_Abyositattā hi bhavābhavesu,_  
_Pāpāni kammāni karoti mohā._  

</Paragraph>

</Division>
<Division number="785" book="th1" bookNo="19" page="327">

<Paragraph number="1124" type="verses" book="th1">
_Upeti gabbhañca parañca lokaṃ,_  
_Saṃsāramāpajja paramparāya;_  
_Tassappapañño abhisaddahanto,_  
_Upeti gabbhañca parañca lokaṃ._  

</Paragraph>

</Division>
<Division number="786" book="th1" bookNo="19" page="327">

<Paragraph number="1125" type="verses" book="th1">
_Coro yathā sandhimukhe gahīto,_  
_Sakammunā haññati pāpadhammo;_  
_Evaṃ pajā pecca paramhi loke,_  
_Sakammunā haññati pāpadhammo._  

</Paragraph>

</Division>
<Division number="787" book="th1" bookNo="19" page="327">

<Paragraph number="1126" type="verses" book="th1">
_Kāmā hi citrā madhurā manoramā,_  
_Virūparūpena mathenti cittaṃ;_  
_Ādīnavaṃ kāmaguṇesu disvā,_  
_Tasmā ahaṃ pabbajitomhi rāja._  

</Paragraph>

</Division>
<Division number="788" book="th1" bookNo="19" page="327">

<Paragraph number="1127" type="verses" book="th1">
_Dumapphalānīva patanti māṇavā,_  
_Daharā ca vuḍḍhā ca sarīrabhedā;_  
_Etampi disvā pabbajitomhi rāja,_  
_Apaṇṇakaṃ sāmaññameva seyyo._  

</Paragraph>

</Division>
<Division number="789" book="th1" bookNo="19" page="327">

<Paragraph number="1128" type="verses" book="th1">
_Saddhāyāhaṃ pabbajito,_  
_upeto jinasāsane;_  
_Avañjhā mayhaṃ pabbajjā,_  
_anaṇo bhuñjāmi bhojanaṃ._  

</Paragraph>

</Division>
<Division number="790" book="th1" bookNo="19" page="327">

<Paragraph number="1129" type="verses" book="th1">
_Kāme ādittato disvā,_  
_Jātarūpāni satthato;_  
_Gabbhavokkantito dukkhaṃ,_  
_Nirayesu mahabbhayaṃ._  

</Paragraph>

</Division>
<Division number="791" book="th1" bookNo="19" page="327">

<Paragraph number="1130" type="verses" book="th1">
_Etamādīnavaṃ ñatvā,_  
_saṃvegaṃ alabhiṃ tadā;_  
_Sohaṃ viddho tadā santo,_  
_sampatto āsavakkhayaṃ._  

</Paragraph>

</Division>
<Division number="792" book="th1" bookNo="19" page="327">

<Paragraph number="1131" type="verses" book="th1">
_Pariciṇṇo mayā satthā,_  
_kataṃ buddhassa sāsanaṃ;_  
_Ohito garuko bhāro,_  
_bhavanetti samūhatā._  

</Paragraph>

</Division>
<Division number="793" book="th1" bookNo="19" page="327">

<Paragraph number="1132" type="verses" book="th1">
_Yassatthāya pabbajito,_  
_agārasmānagāriyaṃ;_  
_So me attho anuppatto,_  
_sabbasaṃyojanakkhayo”ti._  


</Paragraph>
</Division>

…  Raṭṭhapālo thero… .
