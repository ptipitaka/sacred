<Division number="608" book="th1">

<Paragraph number="925" type="verses" book="th1">
_“Sīlamevidha sikkhetha,_  
_asmiṃ loke susikkhitaṃ;_  
_Sīlañhi sabbasampattiṃ,_  
_upanāmeti sevitaṃ._  

</Paragraph>

</Division>
<Division number="609" book="th1">

<Paragraph number="926" type="verses" book="th1">
_Sīlaṃ rakkheyya medhāvī,_  
_patthayāno tayo sukhe;_  
_Pasaṃsaṃ vittilābhañca,_  
_pecca sagge pamodanaṃ._  

</Paragraph>

</Division>
<Division number="610" book="th1">

<Paragraph number="927" type="verses" book="th1">
_Sīlavā hi bahū mitte,_  
_saññamenādhigacchati;_  
_Dussīlo pana mittehi,_  
_dhaṃsate pāpamācaraṃ._  

</Paragraph>

</Division>
<Division number="611" book="th1">

<Paragraph number="928" type="verses" book="th1">
_Avaṇṇañca akittiñca,_  
_dussīlo labhate naro;_  
_Vaṇṇaṃ kittiṃ pasaṃsañca,_  
_sadā labhati sīlavā._  

</Paragraph>

</Division>
<Division number="612" book="th1">

<Paragraph number="929" type="verses" book="th1">
_Ādi sīlaṃ patiṭṭhā ca,_  
_kalyāṇānañca mātukaṃ;_  
_Pamukhaṃ sabbadhammānaṃ,_  
_tasmā sīlaṃ visodhaye._  

</Paragraph>

</Division>
<Division number="613" book="th1">

<Paragraph number="930" type="verses" book="th1">
_Velā ca saṃvaraṃ sīlaṃ,_  
_cittassa abhihāsanaṃ;_  
_Titthañca sabbabuddhānaṃ,_  
_tasmā sīlaṃ visodhaye._  

</Paragraph>

</Division>
<Division number="614" book="th1">

<Paragraph number="931" type="verses" book="th1">
_Sīlaṃ balaṃ appaṭimaṃ,_  
_sīlaṃ āvudhamuttamaṃ;_  
_Sīlamābharaṇaṃ seṭṭhaṃ,_  
_sīlaṃ kavacamabbhutaṃ._  

</Paragraph>

</Division>
<Division number="615" book="th1">

<Paragraph number="932" type="verses" book="th1">
_Sīlaṃ setu mahesakkho,_  
_sīlaṃ gandho anuttaro;_  
_Sīlaṃ vilepanaṃ seṭṭhaṃ,_  
_yena vāti disodisaṃ._  

</Paragraph>

</Division>
<Division number="616" book="th1">

<Paragraph number="933" type="verses" book="th1">
_Sīlaṃ sambalamevaggaṃ,_  
_sīlaṃ pātheyyamuttamaṃ;_  
_Sīlaṃ seṭṭho ativāho,_  
_yena yāti disodisaṃ._  

</Paragraph>

</Division>
<Division number="617" book="th1">

<Paragraph number="934" type="verses" book="th1">
_Idheva nindaṃ labhati,_  
_peccāpāye ca dummano;_  
_Sabbattha dummano bālo,_  
_sīlesu asamāhito._  

</Paragraph>

</Division>
<Division number="618" book="th1">

<Paragraph number="935" type="verses" book="th1">
_Idheva kittiṃ labhati,_  
_pecca sagge ca summano;_  
_Sabbattha sumano dhīro,_  
_sīlesu susamāhito._  

</Paragraph>

</Division>
<Division number="619" book="th1">

<Paragraph number="936" type="verses" book="th1">
_Sīlameva idha aggaṃ,_  
_paññavā pana uttamo;_  
_Manussesu ca devesu,_  
_sīlapaññāṇato jayan”ti._  


</Paragraph>
</Division>
