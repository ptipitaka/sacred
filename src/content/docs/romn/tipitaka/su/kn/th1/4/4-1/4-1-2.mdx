_vihārā upanikkhamiṃ;_

_tattheva papatiṃ chamā._

<Division number="272" book="th1">

<Paragraph number="509" type="verses" book="th1">
_Gattāni parimajjitvā,_  
_punapāruyha caṅkamaṃ;_  
_Caṅkame caṅkamiṃ sohaṃ,_  
_ajjhattaṃ susamāhito._  

</Paragraph>

</Division>
<Division number="273" book="th1">

<Paragraph number="510" type="verses" book="th1">
_Tato me manasīkāro,_  
_yoniso udapajjatha;_  
_Ādīnavo pāturahu,_  
_nibbidā samatiṭṭhatha._  

</Paragraph>

</Division>

<Paragraph number="511" type="verses" book="th1">

_passa dhammasudhammataṃ;_  
_Tisso vijjā anuppattā,_  
_kataṃ buddhassa sāsanan”ti._  


</Paragraph>
