<Paragraph number="954" type="verses" book="th1">
_“Yāhu raṭṭhe samukkaṭṭho,_  
_rañño aṅgassa paddhagū;_  
_Svajja dhammesu ukkaṭṭho,_  
_soṇo dukkhassa pāragū._  

</Paragraph>

</Division>
<Division number="633" book="th1" bookNo="19" page="312">

<Paragraph number="955" type="verses" book="th1">
_Pañca chinde pañca jahe,_  
_pañca cuttari bhāvaye;_  
_Pañcasaṅgātigo bhikkhu,_  
_oghatiṇṇoti vuccati._  

</Paragraph>

</Division>
<Division number="634" book="th1" bookNo="19" page="312">

<Paragraph number="956" type="verses" book="th1">
_Unnaḷassa pamattassa,_  
_bāhirāsassa bhikkhuno;_  
_Sīlaṃ samādhi paññā ca,_  
_pāripūriṃ na gacchati._  

</Paragraph>

</Division>
<Division number="635" book="th1" bookNo="19" page="312">

<Paragraph number="957" type="verses" book="th1">
_Yañhi kiccaṃ apaviddhaṃ,_  
_akiccaṃ pana karīyati;_  
_Unnaḷānaṃ pamattānaṃ,_  
_tesaṃ vaḍḍhanti āsavā._  

</Paragraph>

</Division>
<Division number="636" book="th1" bookNo="19" page="312">

<Paragraph number="958" type="verses" book="th1">
_Yesañca susamāraddhā,_  
_niccaṃ kāyagatā sati;_  
_Akiccaṃ te na sevanti,_  
_kicce sātaccakārino;_  
_Satānaṃ sampajānānaṃ,_  
_atthaṃ gacchanti āsavā._  

</Paragraph>

</Division>
<Division number="637" book="th1" bookNo="19" page="312">

<Paragraph number="959" type="verses" book="th1">
_Ujumaggamhi akkhāte,_  
_gacchatha mā nivattatha;_  
_Attanā codayattānaṃ,_  
_nibbānamabhihāraye._  

</Paragraph>

</Division>
<Division number="638" book="th1" bookNo="19" page="312">

<Paragraph number="960" type="verses" book="th1">
_Accāraddhamhi vīriyamhi,_  
_satthā loke anuttaro;_  
_Vīṇopamaṃ karitvā me,_  
_dhammaṃ desesi cakkhumā;_  
_Tassāhaṃ vacanaṃ sutvā,_  
_vihāsiṃ sāsane rato._  

</Paragraph>

</Division>
<Division number="639" book="th1" bookNo="19" page="312">

<Paragraph number="961" type="verses" book="th1">
_Samathaṃ paṭipādesiṃ,_  
_uttamatthassa pattiyā;_  
_Tisso vijjā anuppattā,_  
_kataṃ buddhassa sāsanaṃ._  

</Paragraph>

</Division>
<Division number="640" book="th1" bookNo="19" page="312">

<Paragraph number="962" type="verses" book="th1">
_Nekkhamme adhimuttassa,_  
_pavivekañca cetaso;_  
_Abyāpajjādhimuttassa,_  
_upādānakkhayassa ca._  

</Paragraph>

</Division>
<Division number="641" book="th1" bookNo="19" page="312">

<Paragraph number="963" type="verses" book="th1">
_Taṇhakkhayādhimuttassa,_  
_asammohañca cetaso;_  
_Disvā āyatanuppādaṃ,_  
_sammā cittaṃ vimuccati._  

</Paragraph>

</Division>
<Division number="642" book="th1" bookNo="19" page="312">

<Paragraph number="964" type="verses" book="th1">
_Tassa sammā vimuttassa,_  
_santacittassa bhikkhuno;_  
_Katassa paṭicayo natthi,_  
_karaṇīyaṃ na vijjati._  

</Paragraph>

</Division>
<Division number="643" book="th1" bookNo="19" page="313">

<Paragraph number="965" type="verses" book="th1">
_Selo yathā ekagghano,_  
_vātena na samīrati;_  
_Evaṃ rūpā rasā saddā,_  
_gandhā phassā ca kevalā._  

</Paragraph>

</Division>
<Division number="644" book="th1" bookNo="19" page="313">

<Paragraph number="966" type="verses" book="th1">
_Iṭṭhā dhammā aniṭṭhā ca,_  
_nappavedhenti tādino;_  
_Ṭhitaṃ cittaṃ visaññuttaṃ,_  
_vayañcassānupassatī”ti._  


</Paragraph>
</Division>

<Paragraph number="967" type="center" book="th1">
…  Soṇo koḷiviso thero… .



</Paragraph>

<Paragraph number="968" type="center" book="th1">
Terasanipāto niṭṭhito.



</Paragraph>

Tatruddānaṃ


</Paragraph>

<Paragraph number="970" type="verses" book="th1">
_Soṇo koḷiviso thero,_  
_ekoyeva mahiddhiko;_  
_Terasamhi nipātamhi,_  
_gāthāyo cettha terasāti._
</Paragraph>
