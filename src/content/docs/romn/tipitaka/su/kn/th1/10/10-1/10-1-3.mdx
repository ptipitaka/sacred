<Division number="547" book="th1">

<Paragraph number="852" type="verses" book="th1">
_“Anāgataṃ yo paṭikacca passati,_  
_Hitañca atthaṃ ahitañca taṃ dvayaṃ;_  
_Viddesino tassa hitesino vā,_  
_Randhaṃ na passanti samekkhamānā._  

</Paragraph>

</Division>
<Division number="548" book="th1">

<Paragraph number="853" type="verses" book="th1">
_Ānāpānasatī yassa,_  
_paripuṇṇā subhāvitā;_  
_Anupubbaṃ paricitā,_  
_yathā buddhena desitā;_  
_Somaṃ lokaṃ pabhāseti,_  
_abbhā muttova candimā._  

</Paragraph>

</Division>
<Division number="549" book="th1">

<Paragraph number="854" type="verses" book="th1">
_Odātaṃ vata me cittaṃ,_  
_appamāṇaṃ subhāvitaṃ;_  
_Nibbiddhaṃ paggahītañca,_  
_sabbā obhāsate disā._  

</Paragraph>

</Division>
<Division number="550" book="th1">

<Paragraph number="855" type="verses" book="th1">
_Jīvate vāpi sappañño,_  
_api vittaparikkhayo;_  
_Paññāya ca alābhena,_  
_vittavāpi na jīvati._  

</Paragraph>

</Division>
<Division number="551" book="th1">

<Paragraph number="856" type="verses" book="th1">
_Paññā sutavinicchinī,_  
_Paññā kittisilokavaddhanī;_  
_Paññāsahito naro idha,_  
_Api dukkhesu sukhāni vindati._  

</Paragraph>

</Division>
<Division number="552" book="th1">

<Paragraph number="857" type="verses" book="th1">
_Nāyaṃ ajjatano dhammo,_  
_nacchero napi abbhuto;_  
_Yattha jāyetha mīyetha,_  
_tattha kiṃ viya abbhutaṃ._  

</Paragraph>

</Division>
<Division number="553" book="th1">

<Paragraph number="858" type="verses" book="th1">
_Anantaraṃ hi jātassa,_  
_jīvitā maraṇaṃ dhuvaṃ;_  
_Jātā jātā marantīdha,_  
_evaṃdhammā hi pāṇino._  

</Paragraph>

</Division>
<Division number="554" book="th1">

<Paragraph number="859" type="verses" book="th1">
_Na hetadatthāya matassa hoti,_  
_Yaṃ jīvitatthaṃ paraporisānaṃ;_  
_Matamhi ruṇṇaṃ na yaso na lokyaṃ,_  
_Na vaṇṇitaṃ samaṇabrāhmaṇehi._  

</Paragraph>

</Division>
<Division number="555" book="th1">

<Paragraph number="860" type="verses" book="th1">
_Cakkhuṃ sarīraṃ upahanti tena,_  
_Nihīyati vaṇṇabalaṃ matī ca;_  
_Ānandino tassa disā bhavanti,_  
_Hitesino nāssa sukhī bhavanti._  

</Paragraph>

</Division>
<Division number="556" book="th1">

<Paragraph number="861" type="verses" book="th1">
_Tasmā hi iccheyya kule vasante,_  
_Medhāvino ceva bahussute ca;_  
_Yesaṃ hi paññāvibhavena kiccaṃ,_  
_Taranti nāvāya nadiṃva puṇṇan”ti._  


</Paragraph>
</Division>
