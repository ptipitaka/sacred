<Paragraph number="591" type="verses" book="th1">
_“Sādhū hi kira me mātā,_  
_Patodaṃ upadaṃsayi;_  
_Yassāhaṃ vacanaṃ sutvā,_  
_Anusiṭṭho janettiyā;_  
_Āraddhavīriyo pahitatto,_  
_Patto sambodhimuttamaṃ._  

</Paragraph>

</Division>
<Division number="336" book="th1" bookNo="19" page="277">

<Paragraph number="592" type="verses" book="th1">
_Arahā dakkhiṇeyyomhi,_  
_tevijjo amataddaso;_  
_Jetvā namucino senaṃ,_  
_viharāmi anāsavo._  

</Paragraph>

</Division>
<Division number="337" book="th1" bookNo="19" page="277">

<Paragraph number="593" type="verses" book="th1">
_Ajjhattañca bahiddhā ca,_  
_ye me vijjiṃsu āsavā;_  
_Sabbe asesā ucchinnā,_  
_na ca uppajjare puna._  

</Paragraph>

</Division>
<Division number="338" book="th1" bookNo="19" page="277">

<Paragraph number="594" type="verses" book="th1">
_Visāradā kho bhaginī,_  
_etamatthaṃ abhāsayi;_  
_‘Apihā nūna mayipi,_  
_vanatho te na vijjati’._  

</Paragraph>

</Division>
<Division number="339" book="th1" bookNo="19" page="277">

<Paragraph number="595" type="verses" book="th1">
_Pariyantakataṃ dukkhaṃ,_  
_antimoyaṃ samussayo;_  
_Jātimaraṇasaṃsāro,_  
_natthi dāni punabbhavo”ti._  


</Paragraph>
