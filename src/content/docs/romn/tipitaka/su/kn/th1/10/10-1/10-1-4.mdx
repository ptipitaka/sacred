<Division number="557" book="th1">

<Paragraph number="863" type="verses" book="th1">
_“Dandhā mayhaṃ gatī āsi,_  
_paribhūto pure ahaṃ;_  
_Bhātā ca maṃ paṇāmesi,_  
_‘gaccha dāni tuvaṃ gharaṃ’._  

</Paragraph>

</Division>
<Division number="558" book="th1">

<Paragraph number="864" type="verses" book="th1">
_Sohaṃ paṇāmito santo,_  
_saṃghārāmassa koṭṭhake;_  
_Dummano tattha aṭṭhāsiṃ,_  
_sāsanasmiṃ apekkhavā._  

</Paragraph>

</Division>
<Division number="559" book="th1">

<Paragraph number="865" type="verses" book="th1">
_Bhagavā tattha āgacchi,_  
_sīsaṃ mayhaṃ parāmasi;_  
_Bāhāya maṃ gahetvāna,_  
_saṃghārāmaṃ pavesayi._  

</Paragraph>

</Division>
<Division number="560" book="th1">

<Paragraph number="866" type="verses" book="th1">
_Anukampāya me satthā,_  
_pādāsi pādapuñchaniṃ;_  
_‘Etaṃ suddhaṃ adhiṭṭhehi,_  
_ekamantaṃ svadhiṭṭhitaṃ’._  

</Paragraph>

</Division>
<Division number="561" book="th1">

<Paragraph number="867" type="verses" book="th1">
_Tassāhaṃ vacanaṃ sutvā,_  
_vihāsiṃ sāsane rato;_  
_Samādhiṃ paṭipādesiṃ,_  
_uttamatthassa pattiyā._  

</Paragraph>

</Division>
<Division number="562" book="th1">

<Paragraph number="868" type="verses" book="th1">
_Pubbenivāsaṃ jānāmi,_  
_dibbacakkhu visodhitaṃ;_  
_Tisso vijjā anuppattā,_  
_kataṃ buddhassa sāsanaṃ._  

</Paragraph>

</Division>
<Division number="563" book="th1">

<Paragraph number="869" type="verses" book="th1">
_Sahassakkhattumattānaṃ,_  
_nimminitvāna panthako;_  
_Nisīdambavane ramme,_  
_yāva kālappavedanā._  

</Paragraph>

</Division>
<Division number="564" book="th1">

<Paragraph number="870" type="verses" book="th1">
_Tato me satthā pāhesi,_  
_dūtaṃ kālappavedakaṃ;_  
_Paveditamhi kālamhi,_  
_vehāsādupasaṅkamiṃ._  

</Paragraph>

</Division>
<Division number="565" book="th1">

<Paragraph number="871" type="verses" book="th1">
_Vanditvā satthuno pāde,_  
_ekamantaṃ nisīdahaṃ;_  
_Nisinnaṃ maṃ viditvāna,_  
_atha satthā paṭiggahi._  

</Paragraph>

</Division>
<Division number="566" book="th1">

<Paragraph number="872" type="verses" book="th1">
_Āyāgo sabbalokassa,_  
_āhutīnaṃ paṭiggaho;_  
_Puññakkhettaṃ manussānaṃ,_  
_paṭiggaṇhittha dakkhiṇan”ti._  


</Paragraph>
</Division>

<Paragraph number="873" type="center" book="th1">
…  Cūḷapanthako thero… .
