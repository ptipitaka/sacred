<Paragraph number="885" type="verses" book="th1">
_“Vivittaṃ appanigghosaṃ,_  
_vāḷamiganisevitaṃ;_  
_Seve senāsanaṃ bhikkhu,_  
_paṭisallānakāraṇā._  

</Paragraph>

</Division>
<Division number="578" book="th1" bookNo="19" page="305">

<Paragraph number="886" type="verses" book="th1">
_Saṅkārapuñjā āhatvā,_  
_susānā rathiyāhi ca;_  
_Tato saṅghāṭikaṃ katvā,_  
_lūkhaṃ dhāreyya cīvaraṃ._  

</Paragraph>

</Division>
<Division number="579" book="th1" bookNo="19" page="305">

<Paragraph number="887" type="verses" book="th1">
_Nīcaṃ manaṃ karitvāna,_  
_sapadānaṃ kulā kulaṃ;_  
_Piṇḍikāya care bhikkhu,_  
_guttadvāro susaṃvuto._  

</Paragraph>

</Division>
<Division number="580" book="th1" bookNo="19" page="305">

<Paragraph number="888" type="verses" book="th1">
_Lūkhenapi vā santusse,_  
_nāññaṃ patthe rasaṃ bahuṃ;_  
_Rasesu anugiddhassa,_  
_jhāne na ramatī mano._  

</Paragraph>

</Division>
<Division number="581" book="th1" bookNo="19" page="305">

<Paragraph number="889" type="verses" book="th1">
_Appiccho ceva santuṭṭho,_  
_pavivitto vase muni;_  
_Asaṃsaṭṭho gahaṭṭhehi,_  
_anāgārehi cūbhayaṃ._  

</Paragraph>

</Division>
<Division number="582" book="th1" bookNo="19" page="305">

<Paragraph number="890" type="verses" book="th1">
_Yathā jaḷo va mūgo va,_  
_attānaṃ dassaye tathā;_  
_Nātivelaṃ sambhāseyya,_  
_saṃghamajjhamhi paṇḍito._  

</Paragraph>

</Division>
<Division number="583" book="th1" bookNo="19" page="305">

<Paragraph number="891" type="verses" book="th1">
_Na so upavade kañci,_  
_upaghātaṃ vivajjaye;_  
_Saṃvuto pātimokkhasmiṃ,_  
_mattaññū cassa bhojane._  

</Paragraph>

</Division>
<Division number="584" book="th1" bookNo="19" page="305">

<Paragraph number="892" type="verses" book="th1">
_Suggahītanimittassa,_  
_cittassuppādakovido;_  
_Samathaṃ anuyuñjeyya,_  
_kālena ca vipassanaṃ._  

</Paragraph>

</Division>
<Division number="585" book="th1" bookNo="19" page="305">

<Paragraph number="893" type="verses" book="th1">
_Vīriyasātaccasampanno,_  
_yuttayogo sadā siyā;_  
_Na ca appatvā dukkhantaṃ,_  
_vissāsaṃ eyya paṇḍito._  

</Paragraph>

</Division>
<Division number="586" book="th1" bookNo="19" page="305">

<Paragraph number="894" type="verses" book="th1">
_Evaṃ viharamānassa,_  
_suddhikāmassa bhikkhuno;_  
_Khīyanti āsavā sabbe,_  
_nibbutiñcādhigacchatī”ti._  


</Paragraph>
</Division>
