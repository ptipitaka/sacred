<Division number="393" book="th1">

<Paragraph number="664" type="verses" book="th1">
_“Kullo sivathikaṃ gantvā,_  
_addasa itthimujjhitaṃ;_  
_Apaviddhaṃ susānasmiṃ,_  
_khajjantiṃ kimihī phuṭaṃ._  

</Paragraph>

</Division>
<Division number="394" book="th1">

<Paragraph number="665" type="verses" book="th1">
_Āturaṃ asuciṃ pūtiṃ,_  
_passa kulla samussayaṃ;_  
_Uggharantaṃ paggharantaṃ,_  
_bālānaṃ abhinanditaṃ._  

</Paragraph>

</Division>
<Division number="395" book="th1">

<Paragraph number="666" type="verses" book="th1">
_Dhammādāsaṃ gahetvāna,_  
_ñāṇadassanapattiyā;_  
_Paccavekkhiṃ imaṃ kāyaṃ,_  
_tucchaṃ santarabāhiraṃ._  

</Paragraph>

</Division>
<Division number="396" book="th1">

<Paragraph number="667" type="verses" book="th1">
_Yathā idaṃ tathā etaṃ,_  
_yathā etaṃ tathā idaṃ;_  
_Yathā adho tathā uddhaṃ,_  
_yathā uddhaṃ tathā adho._  

</Paragraph>

</Division>
<Division number="397" book="th1">

<Paragraph number="668" type="verses" book="th1">
_Yathā divā tathā rattiṃ,_  
_yathā rattiṃ tathā divā;_  
_Yathā pure tathā pacchā,_  
_yathā pacchā tathā pure._  

</Paragraph>

</Division>
<Division number="398" book="th1">

<Paragraph number="669" type="verses" book="th1">
_Pañcaṅgikena turiyena,_  
_Na ratī hoti tādisī;_  
_Yathā ekaggacittassa,_  
_Sammā dhammaṃ vipassato”ti._  


</Paragraph>
</Division>
