_“Pahāya mātāpitaro,_  
_bhaginī ñātibhātaro;_  
_Pañca kāmaguṇe hitvā,_  
_anuruddhova jhāyati._  

</Paragraph>

</Division>
<Division number="893" book="th1">

<Paragraph number="1237" type="verses" book="th1">
_Sameto naccagītehi,_  
_sammatāḷappabodhano;_  
_Na tena suddhimajjhagaṃ,_  
_mārassa visaye rato._  

</Paragraph>

</Division>
<Division number="894" book="th1">

<Paragraph number="1238" type="verses" book="th1">
_Etañca samatikkamma,_  
_rato buddhassa sāsane;_  
_Sabboghaṃ samatikkamma,_  
_anuruddhova jhāyati._  

</Paragraph>

</Division>
<Division number="895" book="th1">

<Paragraph number="1239" type="verses" book="th1">
_Rūpā saddā rasā gandhā,_  
_phoṭṭhabbā ca manoramā;_  
_Ete ca samatikkamma,_  
_anuruddhova jhāyati._  

</Paragraph>

</Division>
<Division number="896" book="th1">

<Paragraph number="1240" type="verses" book="th1">
_Piṇḍapātamatikkanto,_  
_eko adutiyo muni;_  
_Esati paṃsukūlāni,_  
_anuruddho anāsavo._  

</Paragraph>

</Division>
<Division number="897" book="th1">

<Paragraph number="1241" type="verses" book="th1">
_Vicinī aggahī dhovi,_  
_rajayī dhārayī muni;_  
_Paṃsukūlāni matimā,_  
_anuruddho anāsavo._  

</Paragraph>

</Division>
<Division number="898" book="th1">

<Paragraph number="1242" type="verses" book="th1">
_Mahiccho ca asantuṭṭho,_  
_saṃsaṭṭho yo ca uddhato;_  
_Tassa dhammā ime honti,_  
_pāpakā saṃkilesikā._  

</Paragraph>

</Division>
<Division number="899" book="th1">

<Paragraph number="1243" type="verses" book="th1">
_Sato ca hoti appiccho,_  
_santuṭṭho avighātavā;_  
_Pavivekarato vitto,_  
_niccamāraddhavīriyo._  

</Paragraph>

</Division>
<Division number="900" book="th1">

<Paragraph number="1244" type="verses" book="th1">
_Tassa dhammā ime honti,_  
_kusalā bodhipakkhikā;_  
_Anāsavo ca so hoti,_  
_iti vuttaṃ mahesinā._  

</Paragraph>

</Division>
<Division number="901" book="th1">

<Paragraph number="1245" type="verses" book="th1">
_Mama saṅkappamaññāya,_  
_satthā loke anuttaro;_  
_Manomayena kāyena,_  
_iddhiyā upasaṅkami._  

</Paragraph>

</Division>
<Division number="902" book="th1">

<Paragraph number="1246" type="verses" book="th1">
_Yadā me ahu saṅkappo,_  
_tato uttari desayi;_  
_Nippapañcarato buddho,_  
_nippapañcamadesayi._  

</Paragraph>

</Division>
<Division number="903" book="th1">

<Paragraph number="1247" type="verses" book="th1">
_Tassāhaṃ dhammamaññāya,_  
_vihāsiṃ sāsane rato;_  
_Tisso vijjā anuppattā,_  
_kataṃ buddhassa sāsanaṃ._  

</Paragraph>

</Division>
<Division number="904" book="th1">

<Paragraph number="1248" type="verses" book="th1">
_Pañcapaññāsavassāni,_  
_yato nesajjiko ahaṃ;_  
_Pañcavīsativassāni,_  
_yato middhaṃ samūhataṃ._  

</Paragraph>

</Division>
<Division number="905" book="th1">

<Paragraph number="1249" type="verses" book="th1">
_Nāhu assāsapassāsā,_  
_ṭhitacittassa tādino;_  
_Anejo santimārabbha,_  
_cakkhumā parinibbuto._  

</Paragraph>

</Division>
<Division number="906" book="th1">

<Paragraph number="1250" type="verses" book="th1">
_Asallīnena cittena,_  
_vedanaṃ ajjhavāsayi;_  
_Pajjotasseva nibbānaṃ,_  
_vimokkho cetaso ahu._  

</Paragraph>

</Division>
<Division number="907" book="th1">

<Paragraph number="1251" type="verses" book="th1">
_Ete pacchimakā dāni,_  
_munino phassapañcamā;_  
_Nāññe dhammā bhavissanti,_  
_sambuddhe parinibbute._  

</Paragraph>

</Division>
<Division number="908" book="th1">

<Paragraph number="1252" type="verses" book="th1">
_Natthi dāni punāvāso,_  
_devakāyasmi jālini;_  
_Vikkhīṇo jātisaṃsāro,_  
_natthi dāni punabbhavo._  

</Paragraph>

</Division>
<Division number="909" book="th1">

<Paragraph number="1253" type="verses" book="th1">
_Yassa muhuttena sahassadhā,_  
_Loko saṃvidito sabrahmakappo;_  
_Vasī iddhiguṇe cutūpapāte,_  
_Kāle passati devatā sa bhikkhu._  

</Paragraph>

</Division>
<Division number="910" book="th1">

<Paragraph number="1254" type="verses" book="th1">
_Annabhāro pure āsiṃ,_  
_daliddo ghāsahārako;_  
_Samaṇaṃ paṭipādesiṃ,_  
_upariṭṭhaṃ yasassinaṃ._  

</Paragraph>

</Division>
<Division number="911" book="th1">

<Paragraph number="1255" type="verses" book="th1">
_Somhi sakyakule jāto,_  
_anuruddhoti maṃ vidū;_  
_Upeto naccagītehi,_  
_sammatāḷappabodhano._  

</Paragraph>

</Division>
<Division number="912" book="th1">

<Paragraph number="1256" type="verses" book="th1">
_Athaddasāsiṃ sambuddhaṃ,_  
_satthāraṃ akutobhayaṃ;_  
_Tasmiṃ cittaṃ pasādetvā,_  
_pabbajiṃ anagāriyaṃ._  

</Paragraph>

</Division>
<Division number="913" book="th1">

<Paragraph number="1257" type="verses" book="th1">
_Pubbenivāsaṃ jānāmi,_  
_yattha me vusitaṃ pure;_  
_Tāvatiṃsesu devesu,_  
_aṭṭhāsiṃ sakkajātiyā._  

</Paragraph>

</Division>
<Division number="914" book="th1">

<Paragraph number="1258" type="verses" book="th1">
_Sattakkhattuṃ manussindo,_  
_ahaṃ rajjamakārayiṃ;_  
_Cāturanto vijitāvī,_  
_jambusaṇḍassa issaro;_  
_Adaṇḍena asatthena,_  
_dhammena anusāsayiṃ._  

</Paragraph>

</Division>
<Division number="915" book="th1">

<Paragraph number="1259" type="verses" book="th1">
_Ito satta tato satta,_  
_saṃsārāni catuddasa;_  
_Nivāsamabhijānissaṃ,_  
_devaloke ṭhito tadā._  

</Paragraph>

</Division>
<Division number="916" book="th1">

<Paragraph number="1260" type="verses" book="th1">
_Pañcaṅgike samādhimhi,_  
_sante ekodibhāvite;_  
_Paṭippassaddhiladdhamhi,_  
_dibbacakkhu visujjhi me._  

</Paragraph>

</Division>

_sattānaṃ āgatiṃ gatiṃ;_  
_Itthabhāvaññathābhāvaṃ,_  
_jhāne pañcaṅgike ṭhito._  

</Paragraph>

</Division>
