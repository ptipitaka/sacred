_jātiyā sattavassiko;_  
_Iddhiyā abhibhotvāna,_  
_pannagindaṃ mahiddhikaṃ._

<Division number="430" book="th1" bookNo="19" page="287">

<Paragraph number="707" type="verses" book="th1">
_Upajjhāyassa udakaṃ,_  
_anotattā mahāsarā;_  
_Āharāmi tato disvā,_  
_maṃ satthā etadabravi”._  

</Paragraph>

</Division>
<Division number="431" book="th1" bookNo="19" page="287">

<Paragraph number="708" type="verses" book="th1">
_“Sāriputta imaṃ passa,_  
_āgacchantaṃ kumārakaṃ;_  
_Udakakumbhamādāya,_  
_ajjhattaṃ susamāhitaṃ._  

</Paragraph>

</Division>
<Division number="432" book="th1" bookNo="19" page="287">

<Paragraph number="709" type="verses" book="th1">
_Pāsādikena vattena,_  
_kalyāṇairiyāpatho;_  
_Sāmaṇeronuruddhassa,_  
_iddhiyā ca visārado._  

</Paragraph>

</Division>
<Division number="433" book="th1" bookNo="19" page="287">

<Paragraph number="710" type="verses" book="th1">
_Ājānīyena ājañño,_  
_sādhunā sādhukārito;_  
_Vinīto anuruddhena,_  
_katakiccena sikkhito._  

</Paragraph>

</Division>

_sacchikatvā akuppataṃ;_  
_Sāmaṇero sa sumano,_  
_mā maṃ jaññāti icchatī”ti._  


</Paragraph>
