<Division number="460" book="th1">

<Paragraph number="745" type="verses" book="th1">

_purato pañjalīkatā;_  
_Sā maṃ saṇhena mudunā,_  
_mhitapubbaṃ abhāsatha._  

</Paragraph>

</Division>
<Division number="461" book="th1">

<Paragraph number="746" type="verses" book="th1">
_Yuvāsi tvaṃ pabbajito,_  
_tiṭṭhāhi mama sāsane;_  
_Bhuñja mānusake kāme,_  
_ahaṃ vittaṃ dadāmi te;_  
_Saccaṃ te paṭijānāmi,_  
_aggiṃ vā te harāmahaṃ._  

</Paragraph>

</Division>
<Division number="462" book="th1">

<Paragraph number="747" type="verses" book="th1">

_ubho daṇḍaparāyanā;_  
_Ubhopi pabbajissāma,_  
_ubhayattha kaṭaggaho._  

</Paragraph>

</Division>

_Alaṅkataṃ suvasanaṃ,_  
_maccupāsaṃva oḍḍitaṃ._  

</Paragraph>

</Division>
