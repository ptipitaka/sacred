<Paragraph number="513" type="verses" book="th1">
_“Pare ca na vijānanti,_  
_mayamettha yamāmase;_  
_Ye ca tattha vijānanti,_  
_tato sammanti medhagā._  

</Paragraph>

</Division>
<Division number="276" book="th1" bookNo="19" page="269">

<Paragraph number="514" type="verses" book="th1">
_Yadā ca avijānantā,_  
_iriyantyamarā viya;_  
_Vijānanti ca ye dhammaṃ,_  
_āturesu anāturā._  

</Paragraph>

</Division>
<Division number="277" book="th1" bookNo="19" page="270">

<Paragraph number="515" type="verses" book="th1">
_Yaṃ kiñci sithilaṃ kammaṃ,_  
_saṃkiliṭṭhañca yaṃ vataṃ;_  
_Saṅkassaraṃ brahmacariyaṃ,_  
_na taṃ hoti mahapphalaṃ._  

</Paragraph>

</Division>
<Division number="278" book="th1" bookNo="19" page="270">

<Paragraph number="516" type="verses" book="th1">
_Yassa sabrahmacārīsu,_  
_gāravo nūpalabbhati;_  
_Ārakā hoti saddhammā,_  
_nabhaṃ puthaviyā yathā”ti._  


</Paragraph>
</Division>
