_“Manussabhūtaṃ sambuddhaṃ,_  
_Attadantaṃ samāhitaṃ;_  
_Iriyamānaṃ brahmapathe,_  
_Cittassūpasame rataṃ._  

</Paragraph>

</Division>
<Division number="690" book="th1">

<Paragraph number="1022" type="verses" book="th1">
_Yaṃ manussā namassanti,_  
_sabbadhammāna pāraguṃ;_  
_Devāpi taṃ namassanti,_  
_iti me arahato sutaṃ._  

</Paragraph>

</Division>
<Division number="691" book="th1">

<Paragraph number="1023" type="verses" book="th1">
_Sabbasaṃyojanātītaṃ,_  
_vanā nibbanamāgataṃ;_  
_Kāmehi nekkhammarataṃ,_  
_muttaṃ selāva kañcanaṃ._  

</Paragraph>

</Division>
<Division number="692" book="th1">

<Paragraph number="1024" type="verses" book="th1">
_Sa ve accaruci nāgo,_  
_himavāvaññe siluccaye;_  
_Sabbesaṃ nāganāmānaṃ,_  
_saccanāmo anuttaro._  

</Paragraph>

</Division>
<Division number="693" book="th1">

<Paragraph number="1025" type="verses" book="th1">
_Nāgaṃ vo kittayissāmi,_  
_na hi āguṃ karoti so;_  
_Soraccaṃ avihiṃsā ca,_  
_pādā nāgassa te duve._  

</Paragraph>

</Division>
<Division number="694" book="th1">

<Paragraph number="1026" type="verses" book="th1">
_Sati ca sampajaññañca,_  
_caraṇā nāgassa tepare;_  
_Saddhāhattho mahānāgo,_  
_upekkhāsetadantavā._  

</Paragraph>

</Division>
<Division number="695" book="th1">

<Paragraph number="1027" type="verses" book="th1">
_Sati gīvā siro paññā,_  
_vīmaṃsā dhammacintanā;_  
_Dhammakucchisamāvāso,_  
_viveko tassa vāladhi._  

</Paragraph>

</Division>
<Division number="696" book="th1">

<Paragraph number="1028" type="verses" book="th1">
_So jhāyī assāsarato,_  
_ajjhattaṃ susamāhito;_  
_Gacchaṃ samāhito nāgo,_  
_ṭhito nāgo samāhito._  

</Paragraph>

</Division>
<Division number="697" book="th1">

<Paragraph number="1029" type="verses" book="th1">
_Sayaṃ samāhito nāgo,_  
_nisinnopi samāhito;_  
_Sabbattha saṃvuto nāgo,_  
_esā nāgassa sampadā._  

</Paragraph>

</Division>
<Division number="698" book="th1">

<Paragraph number="1030" type="verses" book="th1">
_Bhuñjati anavajjāni,_  
_sāvajjāni na bhuñjati;_  
_Ghāsamacchādanaṃ laddhā,_  
_sannidhiṃ parivajjayaṃ._  

</Paragraph>

</Division>
<Division number="699" book="th1">

<Paragraph number="1031" type="verses" book="th1">
_Saṃyojanaṃ aṇuṃ thūlaṃ,_  
_sabbaṃ chetvāna bandhanaṃ;_  
_Yena yeneva gacchati,_  
_anapekkhova gacchati._  

</Paragraph>

</Division>
<Division number="700" book="th1">

<Paragraph number="1032" type="verses" book="th1">
_Yathāpi udake jātaṃ,_  
_puṇḍarīkaṃ pavaḍḍhati;_  
_Nopalippati toyena,_  
_sucigandhaṃ manoramaṃ._  

</Paragraph>

</Division>
<Division number="701" book="th1">

<Paragraph number="1033" type="verses" book="th1">
_Tatheva ca loke jāto,_  
_buddho loke viharati;_  
_Nopalippati lokena,_  
_toyena padumaṃ yathā._  

</Paragraph>

</Division>
<Division number="702" book="th1">

<Paragraph number="1034" type="verses" book="th1">
_Mahāgini pajjalito,_  
_anāhāropasammati;_  
_Aṅgāresu ca santesu,_  
_nibbutoti pavuccati._  

</Paragraph>

</Division>
<Division number="703" book="th1">

<Paragraph number="1035" type="verses" book="th1">
_Atthassāyaṃ viññāpanī,_  
_upamā viññūhi desitā;_  
_Viññissanti mahānāgā,_  
_nāgaṃ nāgena desitaṃ._  

</Paragraph>

</Division>
<Division number="704" book="th1">

<Paragraph number="1036" type="verses" book="th1">
_Vītarāgo vītadoso,_  
_Vītamoho anāsavo;_  
_Sarīraṃ vijahaṃ nāgo,_  
_Parinibbissatyanāsavo”ti._  


</Paragraph>
</Division>
