<Paragraph number="685" type="verses" book="th1">
_“Uṭṭhehi nisīda kātiyāna,_  
_Mā niddābahulo ahu jāgarassu;_  
_Mā taṃ alasaṃ pamattabandhu,_  
_Kūṭeneva jinātu maccurājā._  

</Paragraph>

</Division>
<Division number="412" book="th1" bookNo="19" page="285">

<Paragraph number="686" type="verses" book="th1">
_Seyyathāpi mahāsamuddavego,_  
_Evaṃ jātijarātivattate taṃ;_  
_So karohi sudīpamattano tvaṃ,_  
_Na hi tāṇaṃ tava vijjateva aññaṃ._  

</Paragraph>

</Division>
<Division number="413" book="th1" bookNo="19" page="285">

<Paragraph number="687" type="verses" book="th1">
_Satthā hi vijesi maggametaṃ,_  
_Saṅgā jātijarābhayā atītaṃ;_  
_Pubbāpararattamappamatto,_  
_Anuyuñjassu daḷhaṃ karohi yogaṃ._  

</Paragraph>

</Division>
<Division number="414" book="th1" bookNo="19" page="286">

<Paragraph number="688" type="verses" book="th1">
_Purimāni pamuñca bandhanāni,_  
_Saṅghāṭikhuramuṇḍabhikkhabhojī;_  
_Mā khiḍḍāratiñca mā niddaṃ,_  
_Anuyuñjittha jhāya kātiyāna._  

</Paragraph>

</Division>
<Division number="415" book="th1" bookNo="19" page="286">

<Paragraph number="689" type="verses" book="th1">
_Jhāyāhi jināhi kātiyāna,_  
_Yogakkhemapathesu kovidosi;_  
_Pappuyya anuttaraṃ visuddhiṃ,_  
_Parinibbāhisi vārināva joti._  

</Paragraph>

</Division>
<Division number="416" book="th1" bookNo="19" page="286">

<Paragraph number="690" type="verses" book="th1">
_Pajjotakaro parittaraṃso,_  
_Vātena vinamyate latāva;_  
_Evampi tuvaṃ anādiyāno,_  
_Māraṃ indasagotta niddhunāhi;_  
_So vedayitāsu vītarāgo,_  
_Kālaṃ kaṅkha idheva sītibhūto”ti._  


</Paragraph>
</Division>
