<Division number="362" book="sn">

<Paragraph number="419" type="verses" book="sn">
_“Pucchāmi muniṃ pahūtapaññaṃ,_  
_Tiṇṇaṃ pāraṅgataṃ parinibbutaṃ ṭhitattaṃ;_  
_Nikkhamma gharā panujja kāme,_  
_Kathaṃ bhikkhu sammā so loke paribbajeyya”. (1)_  

</Paragraph>

</Division>
<Division number="363" book="sn">

<Paragraph number="420" type="verses" book="sn">
_“Yassa maṅgalā samūhatā, __(iti bhagavā)___  
_Uppātā supinā ca lakkhaṇā ca;_  
_So maṅgaladosavippahīno,_  
_Sammā so loke paribbajeyya. (2)_  

</Paragraph>

</Division>
<Division number="364" book="sn">

<Paragraph number="421" type="verses" book="sn">
_Rāgaṃ vinayetha mānusesu,_  
_Dibbesu kāmesu cāpi bhikkhu;_  
_Atikkamma bhavaṃ samecca dhammaṃ,_  
_Sammā so loke paribbajeyya. (3)_  

</Paragraph>

</Division>
<Division number="365" book="sn">

<Paragraph number="422" type="verses" book="sn">
_Vipiṭṭhikatvāna pesuṇāni,_  
_Kodhaṃ kadarīyaṃ jaheyya bhikkhu;_  
_Anurodhavirodhavippahīno,_  
_Sammā so loke paribbajeyya. (4)_  

</Paragraph>

</Division>
<Division number="366" book="sn">

<Paragraph number="423" type="verses" book="sn">
_Hitvāna piyañca appiyañca,_  
_Anupādāya anissito kuhiñci;_  
_Saṃyojaniyehi vippamutto,_  
_Sammā so loke paribbajeyya. (5)_  

</Paragraph>

</Division>
<Division number="367" book="sn">

<Paragraph number="424" type="verses" book="sn">
_Na so upadhīsu sārameti,_  
_Ādānesu vineyya chandarāgaṃ;_  
_So anissito anaññaneyyo,_  
_Sammā so loke paribbajeyya. (6)_  

</Paragraph>

</Division>
<Division number="368" book="sn">

<Paragraph number="425" type="verses" book="sn">
_Vacasā manasā ca kammunā ca,_  
_Aviruddho sammā viditvā dhammaṃ;_  
_Nibbānapadābhipatthayāno,_  
_Sammā so loke paribbajeyya. (7)_  

</Paragraph>

</Division>
<Division number="369" book="sn">

<Paragraph number="426" type="verses" book="sn">
_Yo vandati manti nunnameyya,_  
_Akkuṭṭhopi na sandhiyetha bhikkhu;_  
_Laddhā parabhojanaṃ na majje,_  
_Sammā so loke paribbajeyya. (8)_  

</Paragraph>

</Division>
<Division number="370" book="sn">

<Paragraph number="427" type="verses" book="sn">
_Lobhañca bhavañca vippahāya,_  
_Virato chedanabandhanā ca bhikkhu;_  
_So tiṇṇakathaṃkatho visallo,_  
_Sammā so loke paribbajeyya. (9)_  

</Paragraph>

</Division>
<Division number="371" book="sn">

<Paragraph number="428" type="verses" book="sn">
_Sāruppaṃ attano viditvā,_  
_No ca bhikkhu hiṃseyya kañci loke;_  
_Yathātathiyaṃ viditvā dhammaṃ,_  
_Sammā so loke paribbajeyya. (10)_  

</Paragraph>

</Division>
<Division number="372" book="sn">

<Paragraph number="429" type="verses" book="sn">
_Yassānusayā na santi keci,_  
_Mūlā ca akusalā samūhatāse;_  
_So nirāso anāsisāno,_  
_Sammā so loke paribbajeyya. (11)_  

</Paragraph>

</Division>
<Division number="373" book="sn">

<Paragraph number="430" type="verses" book="sn">
_Āsavakhīṇo pahīnamāno,_  
_Sabbaṃ rāgapathaṃ upātivatto;_  
_Danto parinibbuto ṭhitatto,_  
_Sammā so loke paribbajeyya. (12)_  

</Paragraph>

</Division>
<Division number="374" book="sn">

<Paragraph number="431" type="verses" book="sn">
_Saddho sutavā niyāmadassī,_  
_Vaggagatesu na vaggasāri dhīro;_  
_Lobhaṃ dosaṃ vineyya paṭighaṃ,_  
_Sammā so loke paribbajeyya. (13)_  

</Paragraph>

</Division>
<Division number="375" book="sn">

<Paragraph number="432" type="verses" book="sn">
_Saṃsuddhajino vivaṭṭacchado,_  
_Dhammesu vasī pāragū anejo;_  
_Saṅkhāranirodhañāṇakusalo,_  
_Sammā so loke paribbajeyya. (14)_  

</Paragraph>

</Division>
<Division number="376" book="sn">

<Paragraph number="433" type="verses" book="sn">
_Atītesu anāgatesu cāpi,_  
_Kappātīto aticcasuddhipañño;_  
_Sabbāyatanehi vippamutto,_  
_Sammā so loke paribbajeyya. (15)_  

</Paragraph>

</Division>
<Division number="377" book="sn">

<Paragraph number="434" type="verses" book="sn">
_Aññāya padaṃ samecca dhammaṃ,_  
_Vivaṭaṃ disvāna pahānamāsavānaṃ;_  
_Sabbupadhīnaṃ parikkhayāno,_  
_Sammā so loke paribbajeyya”. (16)_  

</Paragraph>

</Division>
<Division number="378" book="sn">

<Paragraph number="435" type="verses" book="sn">
_“Addhā hi bhagavā tatheva etaṃ,_  
_Yo so evaṃvihārī danto bhikkhu;_  
_Sabbasaṃyojanayogavītivatto,_  
_Sammā so loke paribbajeyyā”ti. (17)_  


</Paragraph>
</Division>

<Paragraph number="436" type="center" book="sn">
Sammāparibbājanīyasuttaṃ terasamaṃ.
</Paragraph>
