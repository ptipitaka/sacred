<Paragraph number="968" type="verses" book="sn">
_Paramanti diṭṭhīsu paribbasāno,_  
_Yaduttari kurute jantu loke;_  
_Hīnāti aññe tato sabbamāha,_  
_Tasmā vivādāni avītivatto. (1)_  

</Paragraph>

</Division>
<Division number="804" book="sn" bookNo="18" page="403">

<Paragraph number="969" type="verses" book="sn">
_Yadattanī passati ānisaṃsaṃ,_  
_Diṭṭhe sute sīlavate mute vā;_  
_Tadeva so tattha samuggahāya,_  
_Nihīnato passati sabbamaññaṃ. (2)_  

</Paragraph>

</Division>
<Division number="805" book="sn" bookNo="18" page="404">

<Paragraph number="970" type="verses" book="sn">
_Taṃ vāpi ganthaṃ kusalā vadanti,_  
_Yaṃ nissito passati hīnamaññaṃ;_  
_Tasmā hi diṭṭhaṃ va sutaṃ mutaṃ vā,_  
_Sīlabbataṃ bhikkhu na nissayeyya. (3)_  

</Paragraph>

</Division>
<Division number="806" book="sn" bookNo="18" page="404">

<Paragraph number="971" type="verses" book="sn">
_Diṭṭhimpi lokasmiṃ na kappayeyya,_  
_Ñāṇena vā sīlavatena vāpi;_  
_Samoti attānamanūpaneyya,_  
_Hīno na maññetha visesi vāpi. (4)_  

</Paragraph>

</Division>
<Division number="807" book="sn" bookNo="18" page="404">

<Paragraph number="972" type="verses" book="sn">
_Attaṃ pahāya anupādiyāno,_  
_Ñāṇepi so nissayaṃ no karoti;_  
_Sa ve viyattesu na vaggasārī,_  
_Diṭṭhimpi so na pacceti kiñci. (5)_  

</Paragraph>

</Division>
<Division number="808" book="sn" bookNo="18" page="404">

<Paragraph number="973" type="verses" book="sn">
_Yassūbhayante paṇidhīdha natthi,_  
_Bhavābhavāya idha vā huraṃ vā;_  
_Nivesanā tassa na santi keci,_  
_Dhammesu niccheyya samuggahītaṃ. (6)_  

</Paragraph>

</Division>
<Division number="809" book="sn" bookNo="18" page="404">

<Paragraph number="974" type="verses" book="sn">
_Tassīdha diṭṭhe va sute mute vā,_  
_Pakappitā natthi aṇūpi saññā;_  
_Taṃ brāhmaṇaṃ diṭṭhimanādiyānaṃ,_  
_Kenīdha lokasmiṃ vikappayeyya. (7)_  

</Paragraph>

</Division>
<Division number="810" book="sn" bookNo="18" page="404">

<Paragraph number="975" type="verses" book="sn">
_Na kappayanti na purekkharonti,_  
_Dhammāpi tesaṃ na paṭicchitāse;_  
_Na brāhmaṇo sīlavatena neyyo,_  
_Pāraṅgato na pacceti tādīti. (8)_  


</Paragraph>
</Division>

<Paragraph number="976" type="center" book="sn">
Paramaṭṭhakasuttaṃ pañcamaṃ.
</Paragraph>
