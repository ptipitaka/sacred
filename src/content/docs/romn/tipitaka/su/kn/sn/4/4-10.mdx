<Division number="855" book="sn">

<Paragraph number="1025" type="verses" book="sn">
_“Kathaṃdassī kathaṃsīlo,_  
_upasantoti vuccati;_  
_Taṃ me gotama pabrūhi,_  
_pucchito uttamaṃ naraṃ”. (1)_  

</Paragraph>

</Division>
<Division number="856" book="sn">

<Paragraph number="1026" type="verses" book="sn">
_“Vītataṇho purā bhedā, __(iti bhagavā)___  
_Pubbamantamanissito;_  
_Vemajjhe nupasaṅkheyyo,_  
_Tassa natthi purakkhataṃ. (2)_  

</Paragraph>

</Division>
<Division number="857" book="sn">

<Paragraph number="1027" type="verses" book="sn">
_Akkodhano asantāsī,_  
_avikatthī akukkuco;_  
_Mantabhāṇī anuddhato,_  
_sa ve vācāyato muni. (3)_  

</Paragraph>

</Division>
<Division number="858" book="sn">

<Paragraph number="1028" type="verses" book="sn">
_Nirāsatti anāgate,_  
_atītaṃ nānusocati;_  
_Vivekadassī phassesu,_  
_diṭṭhīsu ca na nīyati. (4)_  

</Paragraph>

</Division>
<Division number="859" book="sn">

<Paragraph number="1029" type="verses" book="sn">
_Patilīno akuhako,_  
_apihālu amaccharī;_  
_Appagabbho ajeguccho,_  
_pesuṇeyye ca no yuto. (5)_  

</Paragraph>

</Division>
<Division number="860" book="sn">

<Paragraph number="1030" type="verses" book="sn">
_Sātiyesu anassāvī,_  
_atimāne ca no yuto;_  
_Saṇho ca paṭibhānavā,_  
_na saddho na virajjati. (6)_  

</Paragraph>

</Division>
<Division number="861" book="sn">

<Paragraph number="1031" type="verses" book="sn">
_Lābhakamyā na sikkhati,_  
_alābhe ca na kuppati;_  
_Aviruddho ca taṇhāya,_  
_rasesu nānugijjhati. (7)_  

</Paragraph>

</Division>
<Division number="862" book="sn">

<Paragraph number="1032" type="verses" book="sn">
_Upekkhako sadā sato,_  
_na loke maññate samaṃ;_  
_Na visesī na nīceyyo,_  
_tassa no santi ussadā. (8)_  

</Paragraph>

</Division>
<Division number="863" book="sn">

<Paragraph number="1033" type="verses" book="sn">
_Yassa nissayanā natthi,_  
_ñatvā dhammaṃ anissito;_  
_Bhavāya vibhavāya vā,_  
_taṇhā yassa na vijjati. (9)_  

</Paragraph>

</Division>
<Division number="864" book="sn">

<Paragraph number="1034" type="verses" book="sn">
_Taṃ brūmi upasantoti,_  
_Kāmesu anapekkhinaṃ;_  
_Ganthā tassa na vijjanti,_  
_Atarī so visattikaṃ. (10)_  

</Paragraph>

</Division>
<Division number="865" book="sn">

<Paragraph number="1035" type="verses" book="sn">
_Na tassa puttā pasavo,_  
_Khettaṃ vatthuñca vijjati;_  
_Attā vāpi nirattā vā,_  
_Na tasmiṃ upalabbhati. (11)_  

</Paragraph>

</Division>
<Division number="866" book="sn">

<Paragraph number="1036" type="verses" book="sn">
_Yena naṃ vajjuṃ puthujjanā,_  
_Atho samaṇabrāhmaṇā;_  
_Taṃ tassa apurakkhataṃ,_  
_Tasmā vādesu nejati. (12)_  

</Paragraph>

</Division>
<Division number="867" book="sn">

<Paragraph number="1037" type="verses" book="sn">
_Vītagedho amaccharī,_  
_Na ussesu vadate muni;_  
_Na samesu na omesu,_  
_Kappaṃ neti akappiyo. (13)_  

</Paragraph>

</Division>
<Division number="868" book="sn">

<Paragraph number="1038" type="verses" book="sn">
_Yassa loke sakaṃ natthi,_  
_Asatā ca na socati;_  
_Dhammesu ca na gacchati,_  
_Sa ve santoti vuccatī”ti. (14)_  


</Paragraph>
</Division>

<Paragraph number="1039" type="center" book="sn">
Purābhedasuttaṃ dasamaṃ.
</Paragraph>
