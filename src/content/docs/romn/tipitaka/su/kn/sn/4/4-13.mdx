<Paragraph number="1075" type="verses" book="sn">
_“Ye kecime diṭṭhiparibbasānā,_  
_Idameva saccanti vivādayanti;_  
_Sabbeva te nindamanvānayanti,_  
_Atho pasaṃsampi labhanti tattha”. (1)_  

</Paragraph>

</Division>
<Division number="903" book="sn" bookNo="18" page="418">

<Paragraph number="1076" type="verses" book="sn">
_“Appañhi etaṃ na alaṃ samāya,_  
_Duve vivādassa phalāni brūmi;_  
_Etampi disvā na vivādayetha,_  
_Khemābhipassaṃ avivādabhūmiṃ”. (2)_  

</Paragraph>

</Division>
<Division number="904" book="sn" bookNo="18" page="418">

<Paragraph number="1077" type="verses" book="sn">
_“Yā kācimā sammutiyo puthujjā,_  
_Sabbāva etā na upeti vidvā;_  
_Anūpayo so upayaṃ kimeyya,_  
_Diṭṭhe sute khantimakubbamāno”. (3)_  

</Paragraph>

</Division>
<Division number="905" book="sn" bookNo="18" page="418">

<Paragraph number="1078" type="verses" book="sn">
_“Sīluttamā saññamenāhu suddhiṃ,_  
_Vataṃ samādāya upaṭṭhitāse;_  
_Idheva sikkhema athassa suddhiṃ,_  
_Bhavūpanītā kusalāvadānā. (4)_  

</Paragraph>

</Division>
<Division number="906" book="sn" bookNo="18" page="418">

<Paragraph number="1079" type="verses" book="sn">
_Sace cuto sīlavatato hoti,_  
_Pavedhatī kamma virādhayitvā;_  
_Pajappatī patthayatī ca suddhiṃ,_  
_Satthāva hīno pavasaṃ gharamhā. (5)_  

</Paragraph>

</Division>
<Division number="907" book="sn" bookNo="18" page="419">

<Paragraph number="1080" type="verses" book="sn">
_Sīlabbataṃ vāpi pahāya sabbaṃ,_  
_Kammañca sāvajjanavajjametaṃ;_  
_Suddhiṃ asuddhinti apatthayāno,_  
_Virato care santimanuggahāya. (6)_  

</Paragraph>

</Division>
<Division number="908" book="sn" bookNo="18" page="419">

<Paragraph number="1081" type="verses" book="sn">
_Tamūpanissāya jigucchitaṃ vā,_  
_Atha vāpi diṭṭhaṃ va sutaṃ mutaṃ vā;_  
_Uddhaṃsarā suddhimanutthunanti,_  
_Avītataṇhāse bhavābhavesu. (7)_  

</Paragraph>

</Division>
<Division number="909" book="sn" bookNo="18" page="419">

<Paragraph number="1082" type="verses" book="sn">
_Patthayamānassa hi jappitāni,_  
_Pavedhitaṃ vāpi pakappitesu;_  
_Cutūpapāto idha yassa natthi,_  
_Sa kena vedheyya kuhiṃ va jappe”. (8)_  

</Paragraph>

</Division>
<Division number="910" book="sn" bookNo="18" page="419">

<Paragraph number="1083" type="verses" book="sn">
_“Yamāhu dhammaṃ paramanti eke,_  
_Tameva hīnanti panāhu aññe;_  
_Sacco nu vādo katamo imesaṃ,_  
_Sabbeva hīme kusalāvadānā”. (9)_  

</Paragraph>

</Division>
<Division number="911" book="sn" bookNo="18" page="419">

<Paragraph number="1084" type="verses" book="sn">
_“Sakañhi dhammaṃ paripuṇṇamāhu,_  
_Aññassa dhammaṃ pana hīnamāhu;_  
_Evampi viggayha vivādayanti,_  
_Sakaṃ sakaṃ sammutimāhu saccaṃ. (10)_  

</Paragraph>

</Division>
<Division number="912" book="sn" bookNo="18" page="419">

<Paragraph number="1085" type="verses" book="sn">
_Parassa ce vambhayitena hīno,_  
_Na koci dhammesu visesi assa;_  
_Puthū hi aññassa vadanti dhammaṃ,_  
_Nihīnato samhi daḷhaṃ vadānā. (11)_  

</Paragraph>

</Division>
<Division number="913" book="sn" bookNo="18" page="419">

<Paragraph number="1086" type="verses" book="sn">
_Saddhammapūjāpi nesaṃ tatheva,_  
_Yathā pasaṃsanti sakāyanāni;_  
_Sabbeva vādā tathiyā bhaveyyuṃ,_  
_Suddhī hi nesaṃ paccattameva. (12)_  

</Paragraph>

</Division>
<Division number="914" book="sn" bookNo="18" page="420">

<Paragraph number="1087" type="verses" book="sn">
_Na brāhmaṇassa paraneyyamatthi,_  
_Dhammesu niccheyya samuggahītaṃ;_  
_Tasmā vivādāni upātivatto,_  
_Na hi seṭṭhato passati dhammamaññaṃ. (13)_  

</Paragraph>

</Division>
<Division number="915" book="sn" bookNo="18" page="420">

<Paragraph number="1088" type="verses" book="sn">
_Jānāmi passāmi tatheva etaṃ,_  
_Diṭṭhiyā eke paccenti suddhiṃ;_  
_Addakkhi ce kiñhi tumassa tena,_  
_Atisitvā aññena vadanti suddhiṃ. (14)_  

</Paragraph>

</Division>
<Division number="916" book="sn" bookNo="18" page="420">

<Paragraph number="1089" type="verses" book="sn">
_Passaṃ naro dakkhati nāmarūpaṃ,_  
_Disvāna vā ñassati tānimeva;_  
_Kāmaṃ bahuṃ passatu appakaṃ vā,_  
_Na hi tena suddhiṃ kusalā vadanti. (15)_  

</Paragraph>

</Division>
<Division number="917" book="sn" bookNo="18" page="420">

<Paragraph number="1090" type="verses" book="sn">
_Nivissavādī na hi subbināyo,_  
_Pakappitaṃ diṭṭhi purakkharāno;_  
_Yaṃ nissito tattha subhaṃ vadāno,_  
_Suddhiṃ vado tattha tathaddasā so. (16)_  

</Paragraph>

</Division>
<Division number="918" book="sn" bookNo="18" page="420">

<Paragraph number="1091" type="verses" book="sn">
_Na brāhmaṇo kappamupeti saṅkhā,_  
_Na diṭṭhisārī napi ñāṇabandhu;_  
_Ñatvā ca so sammutiyo puthujjā,_  
_Upekkhatī uggahaṇanti maññe. (17)_  

</Paragraph>

</Division>
<Division number="919" book="sn" bookNo="18" page="420">

<Paragraph number="1092" type="verses" book="sn">
_Vissajja ganthāni munīdha loke,_  
_Vivādajātesu na vaggasārī;_  
_Santo asantesu upekkhako so,_  
_Anuggaho uggahaṇanti maññe. (18)_  

</Paragraph>

</Division>
<Division number="920" book="sn" bookNo="18" page="420">

<Paragraph number="1093" type="verses" book="sn">
_Pubbāsave hitvā nave akubbaṃ,_  
_Na chandagū nopi nivissavādī;_  
_Sa vippamutto diṭṭhigatehi dhīro,_  
_Na lippati loke anattagarahī. (19)_  

</Paragraph>

</Division>
<Division number="921" book="sn" bookNo="18" page="421">

<Paragraph number="1094" type="verses" book="sn">
_Sa sabbadhammesu visenibhūto,_  
_Yaṃ kiñci diṭṭhaṃ va sutaṃ mutaṃ vā;_  
_Sa pannabhāro muni vippamutto,_  
_Na kappiyo nūparato na patthiyo”ti. (20)_  


</Paragraph>
</Division>

<Paragraph number="1095" type="center" book="sn">
Mahābyūhasuttaṃ terasamaṃ.
</Paragraph>
