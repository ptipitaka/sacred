<Paragraph number="1272" type="verses" book="sn">
_“Santi loke munayo, __(iccāyasmā nando)___  
_Janā vadanti tayidaṃ kathaṃsu;_  
_Ñāṇūpapannaṃ no muniṃ vadanti,_  
_Udāhu ve jīvitenūpapannaṃ”. (1)_  

</Paragraph>

</Division>
<Division number="1085" book="sn" bookNo="18" page="441">

<Paragraph number="1273" type="verses" book="sn">
_“Na diṭṭhiyā na sutiyā na ñāṇena, ( )_  
_Munīdha nanda kusalā vadanti;_  
_Visenikatvā anīghā nirāsā,_  
_Caranti ye te munayoti brūmi”. (2)_  

</Paragraph>

</Division>
<Division number="1086" book="sn" bookNo="18" page="441">

<Paragraph number="1274" type="verses" book="sn">
_“Ye kecime samaṇabrāhmaṇāse, __(iccāyasmā nando)___  
_Diṭṭhassutenāpi vadanti suddhiṃ;_  
_Sīlabbatenāpi vadanti suddhiṃ,_  
_Anekarūpena vadanti suddhiṃ._  

</Paragraph>

<Paragraph number="1275" type="verses" book="sn">
_Kaccissu te bhagavā tattha yatā carantā,_  
_Atāru jātiñca jarañca mārisa;_  
_Pucchāmi taṃ bhagavā brūhi me taṃ”. (3)_  

</Paragraph>

</Division>
<Division number="1087" book="sn" bookNo="18" page="442">

<Paragraph number="1276" type="verses" book="sn">
_“Ye kecime samaṇabrāhmaṇāse, __(nandāti bhagavā)___  
_Diṭṭhassutenāpi vadanti suddhiṃ;_  
_Sīlabbatenāpi vadanti suddhiṃ,_  
_Anekarūpena vadanti suddhiṃ;_  
_Kiñcāpi te tattha yatā caranti,_  
_Nātariṃsu jātijaranti brūmi”. (4)_  

</Paragraph>

</Division>
<Division number="1088" book="sn" bookNo="18" page="442">

<Paragraph number="1277" type="verses" book="sn">
_“Ye kecime samaṇabrāhmaṇāse, __(iccāyasmā nando)___  
_Diṭṭhassutenāpi vadanti suddhiṃ;_  
_Sīlabbatenāpi vadanti suddhiṃ,_  
_Anekarūpena vadanti suddhiṃ._  

</Paragraph>

<Paragraph number="1278" type="verses" book="sn">
_Te ce muni brūsi anoghatiṇṇe,_  
_Atha ko carahi devamanussaloke;_  
_Atāri jātiñca jarañca mārisa,_  
_Pucchāmi taṃ bhagavā brūhi metaṃ”. (5)_  

</Paragraph>

</Division>
<Division number="1089" book="sn" bookNo="18" page="442">

<Paragraph number="1279" type="verses" book="sn">
_“Nāhaṃ sabbe samaṇabrāhmaṇāse, __(nandāti bhagavā)___  
_Jātijarāya nivutāti brūmi;_  
_Ye sīdha diṭṭhaṃ va sutaṃ mutaṃ vā,_  
_Sīlabbataṃ vāpi pahāya sabbaṃ._  

</Paragraph>

<Paragraph number="1280" type="verses" book="sn">
_Anekarūpampi pahāya sabbaṃ,_  
_Taṇhaṃ pariññāya anāsavāse;_  
_Te ve narā oghatiṇṇāti brūmi”. (6)_  

</Paragraph>

</Division>
<Division number="1090" book="sn" bookNo="18" page="442">

<Paragraph number="1281" type="verses" book="sn">
_“Etābhinandāmi vaco mahesino,_  
_Sukittitaṃ gotamanūpadhīkaṃ;_  
_Ye sīdha diṭṭhaṃ va sutaṃ mutaṃ vā,_  
_Sīlabbataṃ vāpi pahāya sabbaṃ._  

</Paragraph>

<Paragraph number="1282" type="verses" book="sn">
_Anekarūpampi pahāya sabbaṃ,_  
_Taṇhaṃ pariññāya anāsavāse;_  
_Ahampi te oghatiṇṇāti brūmī”ti. (7)_  


</Paragraph>
</Division>

<Paragraph number="1283" type="center" book="sn">
Nandamāṇavapucchā sattamā.
</Paragraph>
