<Paragraph number="378" type="verses" book="sn">
_“Kiṃsīlo kiṃsamācāro,_  
_kāni kammāni brūhayaṃ;_  
_Naro sammā niviṭṭhassa,_  
_uttamatthañca pāpuṇe”. (1)_  

</Paragraph>

</Division>
<Division number="328" book="sn" bookNo="18" page="327">

<Paragraph number="379" type="verses" book="sn">
_“Vuḍḍhāpacāyī anusūyako siyā,_  
_Kālaññū cassa garūnaṃ dassanāya;_  
_Dhammiṃ kathaṃ erayitaṃ khaṇaññū,_  
_Suṇeyya sakkacca subhāsitāni. (2)_  

</Paragraph>

</Division>
<Division number="329" book="sn" bookNo="18" page="327">

<Paragraph number="380" type="verses" book="sn">
_Kālena gacche garūnaṃ sakāsaṃ,_  
_Thambhaṃ niraṃkatvā nivātavutti;_  
_Atthaṃ dhammaṃ saṃyamaṃ brahmacariyaṃ,_  
_Anussare ceva samācare ca. (3)_  

</Paragraph>

</Division>
<Division number="330" book="sn" bookNo="18" page="327">

<Paragraph number="381" type="verses" book="sn">
_Dhammārāmo dhammarato,_  
_Dhamme ṭhito dhammavinicchayaññū;_  
_Nevācare dhammasandosavādaṃ,_  
_Tacchehi nīyetha subhāsitehi. (4)_  

</Paragraph>

</Division>
<Division number="331" book="sn" bookNo="18" page="327">

<Paragraph number="382" type="verses" book="sn">
_Hassaṃ jappaṃ paridevaṃ padosaṃ,_  
_Māyākataṃ kuhanaṃ giddhi mānaṃ;_  
_Sārambhaṃ kakkasaṃ kasāvañca mucchaṃ,_  
_Hitvā care vītamado ṭhitatto. (5)_  

</Paragraph>

</Division>
<Division number="332" book="sn" bookNo="18" page="327">

<Paragraph number="383" type="verses" book="sn">
_Viññātasārāni subhāsitāni,_  
_Sutañca viññātasamādhisāraṃ;_  
_Na tassa paññā ca sutañca vaḍḍhati,_  
_Yo sāhaso hoti naro pamatto. (6)_  

</Paragraph>

</Division>
<Division number="333" book="sn" bookNo="18" page="328">

<Paragraph number="384" type="verses" book="sn">
_Dhamme ca ye ariyapavedite ratā,_  
_Anuttarā te vacasā manasā kammunā ca;_  
_Te santisoraccasamādhisaṇṭhitā,_  
_Sutassa paññāya ca sāramajjhagū”ti. (7)_  


</Paragraph>
</Division>

<Paragraph number="385" type="center" book="sn">
Kiṃsīlasuttaṃ navamaṃ.
</Paragraph>
