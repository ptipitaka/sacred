<Paragraph number="1096" type="verses" book="sn">
_“Pucchāmi taṃ ādiccabandhu,_  
_Vivekaṃ santipadañca mahesi;_  
_Kathaṃ disvā nibbāti bhikkhu,_  
_Anupādiyāno lokasmiṃ kiñci”. (1)_  

</Paragraph>

</Division>
<Division number="923" book="sn" bookNo="18" page="421">

<Paragraph number="1097" type="verses" book="sn">
_“Mūlaṃ papañcasaṅkhāya, __(iti bhagavā)___  
_Mantā asmīti sabbamuparundhe;_  
_Yā kāci taṇhā ajjhattaṃ,_  
_Tāsaṃ vinayā sadā sato sikkhe. (2)_  

</Paragraph>

</Division>
<Division number="924" book="sn" bookNo="18" page="421">

<Paragraph number="1098" type="verses" book="sn">
_Yaṃ kiñci dhammamabhijaññā,_  
_Ajjhattaṃ atha vāpi bahiddhā;_  
_Na tena thāmaṃ kubbetha,_  
_Na hi sā nibbuti sataṃ vuttā. (3)_  

</Paragraph>

</Division>
<Division number="925" book="sn" bookNo="18" page="421">

<Paragraph number="1099" type="verses" book="sn">
_Seyyo na tena maññeyya,_  
_Nīceyyo atha vāpi sarikkho;_  
_Phuṭṭho anekarūpehi,_  
_Nātumānaṃ vikappayaṃ tiṭṭhe. (4)_  

</Paragraph>

</Division>
<Division number="926" book="sn" bookNo="18" page="421">

<Paragraph number="1100" type="verses" book="sn">
_Ajjhattamevupasame,_  
_Na aññato bhikkhu santimeseyya;_  
_Ajjhattaṃ upasantassa,_  
_Natthi attā kuto nirattā vā. (5)_  

</Paragraph>

</Division>
<Division number="927" book="sn" bookNo="18" page="422">

<Paragraph number="1101" type="verses" book="sn">
_Majjhe yathā samuddassa,_  
_Ūmi no jāyatī ṭhito hoti;_  
_Evaṃ ṭhito anejassa,_  
_Ussadaṃ bhikkhu na kareyya kuhiñci”. (6)_  

</Paragraph>

</Division>
<Division number="928" book="sn" bookNo="18" page="422">

<Paragraph number="1102" type="verses" book="sn">
_“Akittayī vivaṭacakkhu,_  
_Sakkhidhammaṃ parissayavinayaṃ;_  
_Paṭipadaṃ vadehi bhaddante,_  
_Pātimokkhaṃ atha vāpi samādhiṃ”. (7)_  

</Paragraph>

</Division>
<Division number="929" book="sn" bookNo="18" page="422">

<Paragraph number="1103" type="verses" book="sn">
_“Cakkhūhi neva lolassa,_  
_Gāmakathāya āvaraye sotaṃ;_  
_Rase ca nānugijjheyya,_  
_Na ca mamāyetha kiñci lokasmiṃ. (8)_  

</Paragraph>

</Division>
<Division number="930" book="sn" bookNo="18" page="422">

<Paragraph number="1104" type="verses" book="sn">
_Phassena yadā phuṭṭhassa,_  
_Paridevaṃ bhikkhu na kareyya kuhiñci;_  
_Bhavañca nābhijappeyya,_  
_Bheravesu ca na sampavedheyya. (9)_  

</Paragraph>

</Division>
<Division number="931" book="sn" bookNo="18" page="422">

<Paragraph number="1105" type="verses" book="sn">
_Annānamatho pānānaṃ,_  
_Khādanīyānaṃ athopi vatthānaṃ;_  
_Laddhā na sannidhiṃ kayirā,_  
_Na ca parittase tāni alabhamāno. (10)_  

</Paragraph>

</Division>
<Division number="932" book="sn" bookNo="18" page="422">

<Paragraph number="1106" type="verses" book="sn">
_Jhāyī na pādalolassa,_  
_Virame kukkuccā nappamajjeyya;_  
_Athāsanesu sayanesu,_  
_Appasaddesu bhikkhu vihareyya. (11)_  

</Paragraph>

</Division>
<Division number="933" book="sn" bookNo="18" page="422">

<Paragraph number="1107" type="verses" book="sn">
_Niddaṃ na bahulīkareyya,_  
_Jāgariyaṃ bhajeyya ātāpī;_  
_Tandiṃ māyaṃ hassaṃ khiḍḍaṃ,_  
_Methunaṃ vippajahe savibhūsaṃ. (12)_  

</Paragraph>

</Division>
<Division number="934" book="sn" bookNo="18" page="423">

<Paragraph number="1108" type="verses" book="sn">
_Āthabbaṇaṃ supinaṃ lakkhaṇaṃ,_  
_No vidahe athopi nakkhattaṃ;_  
_Virutañca gabbhakaraṇaṃ,_  
_Tikicchaṃ māmako na seveyya. (13)_  

</Paragraph>

</Division>
<Division number="935" book="sn" bookNo="18" page="423">

<Paragraph number="1109" type="verses" book="sn">
_Nindāya nappavedheyya,_  
_Na uṇṇameyya pasaṃsito bhikkhu;_  
_Lobhaṃ saha macchariyena,_  
_Kodhaṃ pesuṇiyañca panudeyya. (14)_  

</Paragraph>

</Division>
<Division number="936" book="sn" bookNo="18" page="423">

<Paragraph number="1110" type="verses" book="sn">
_Kayavikkaye na tiṭṭheyya,_  
_Upavādaṃ bhikkhu na kareyya kuhiñci;_  
_Gāme ca nābhisajjeyya,_  
_Lābhakamyā janaṃ na lapayeyya. (15)_  

</Paragraph>

</Division>
<Division number="937" book="sn" bookNo="18" page="423">

<Paragraph number="1111" type="verses" book="sn">
_Na ca katthitā siyā bhikkhu,_  
_Na ca vācaṃ payuttaṃ bhāseyya;_  
_Pāgabbhiyaṃ na sikkheyya,_  
_Kathaṃ viggāhikaṃ na kathayeyya. (16)_  

</Paragraph>

</Division>
<Division number="938" book="sn" bookNo="18" page="423">

<Paragraph number="1112" type="verses" book="sn">
_Mosavajje na nīyetha,_  
_Sampajāno saṭhāni na kayirā;_  
_Atha jīvitena paññāya,_  
_Sīlabbatena nāññamatimaññe. (17)_  

</Paragraph>

</Division>
<Division number="939" book="sn" bookNo="18" page="423">

<Paragraph number="1113" type="verses" book="sn">
_Sutvā rusito bahuṃ vācaṃ,_  
_Samaṇānaṃ vā puthujanānaṃ;_  
_Pharusena ne na paṭivajjā,_  
_Na hi santo paṭisenikaronti. (18)_  

</Paragraph>

</Division>
<Division number="940" book="sn" bookNo="18" page="423">

<Paragraph number="1114" type="verses" book="sn">
_Etañca dhammamaññāya,_  
_Vicinaṃ bhikkhu sadā sato sikkhe;_  
_Santīti nibbutiṃ ñatvā,_  
_Sāsane gotamassa na pamajjeyya. (19)_  

</Paragraph>

</Division>
<Division number="941" book="sn" bookNo="18" page="424">

<Paragraph number="1115" type="verses" book="sn">
_Abhibhū hi so anabhibhūto,_  
_Sakkhidhammamanītihamadassī;_  
_Tasmā hi tassa bhagavato sāsane,_  
_Appamatto sadā namassamanusikkhe”ti. (20)_  


</Paragraph>
</Division>

<Paragraph number="1116" type="center" book="sn">
Tuvaṭakasuttaṃ cuddasamaṃ.
</Paragraph>
