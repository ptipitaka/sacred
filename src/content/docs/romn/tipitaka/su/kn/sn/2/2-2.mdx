<Division number="242" book="sn">

<Paragraph number="279" type="verses" book="sn">
_“Sāmākaciṅgūlakacīnakāni ca,_  
_Pattapphalaṃ mūlaphalaṃ gavipphalaṃ;_  
_Dhammena laddhaṃ satamasnamānā,_  
_Na kāmakāmā alikaṃ bhaṇanti. (1)_  

</Paragraph>

</Division>
<Division number="243" book="sn">

<Paragraph number="280" type="verses" book="sn">
_Yadasnamāno sukataṃ suniṭṭhitaṃ,_  
_Parehi dinnaṃ payataṃ paṇītaṃ;_  
_Sālīnamannaṃ paribhuñjamāno,_  
_So bhuñjasī kassapa āmagandhaṃ. (2)_  

</Paragraph>

</Division>
<Division number="244" book="sn">

<Paragraph number="281" type="verses" book="sn">
_Na āmagandho mama kappatīti,_  
_Icceva tvaṃ bhāsasi brahmabandhu;_  
_Sālīnamannaṃ paribhuñjamāno,_  
_Sakuntamaṃsehi susaṅkhatehi;_  
_Pucchāmi taṃ kassapa etamatthaṃ,_  
_Kathaṃpakāro tava āmagandho”. (3)_  

</Paragraph>

</Division>
<Division number="245" book="sn">

<Paragraph number="282" type="verses" book="sn">
_“Pāṇātipāto vadhachedabandhanaṃ,_  
_Theyyaṃ musāvādo nikativañcanāni ca;_  
_Ajjhenakuttaṃ paradārasevanā,_  
_Esāmagandho na hi maṃsabhojanaṃ. (4)_  

</Paragraph>

</Division>
<Division number="246" book="sn">

<Paragraph number="283" type="verses" book="sn">
_Ye idha kāmesu asaññatā janā,_  
_Rasesu giddhā asucibhāvamassitā;_  
_Natthikadiṭṭhī visamā durannayā,_  
_Esāmagandho na hi maṃsabhojanaṃ. (5)_  

</Paragraph>

</Division>
<Division number="247" book="sn">

<Paragraph number="284" type="verses" book="sn">
_Ye lūkhasā dāruṇā piṭṭhimaṃsikā,_  
_Mittadduno nikkaruṇātimānino;_  
_Adānasīlā na ca denti kassaci,_  
_Esāmagandho na hi maṃsabhojanaṃ. (6)_  

</Paragraph>

</Division>
<Division number="248" book="sn">

<Paragraph number="285" type="verses" book="sn">
_Kodho mado thambho paccupaṭṭhāpanā,_  
_Māyā usūyā bhassasamussayo ca;_  
_Mānātimāno ca asabbhi santhavo,_  
_Esāmagandho na hi maṃsabhojanaṃ. (7)_  

</Paragraph>

</Division>
<Division number="249" book="sn">

<Paragraph number="286" type="verses" book="sn">
_Ye pāpasīlā iṇaghātasūcakā,_  
_Vohārakūṭā idha pāṭirūpikā;_  
_Narādhamā yedha karonti kibbisaṃ,_  
_Esāmagandho na hi maṃsabhojanaṃ. (8)_  

</Paragraph>

</Division>
<Division number="250" book="sn">

<Paragraph number="287" type="verses" book="sn">
_Ye idha pāṇesu asaññatā janā,_  
_Paresamādāya vihesamuyyutā;_  
_Dussīlaluddā pharusā anādarā,_  
_Esāmagandho na hi maṃsabhojanaṃ. (9)_  

</Paragraph>

</Division>
<Division number="251" book="sn">

<Paragraph number="288" type="verses" book="sn">
_Etesu giddhā viruddhātipātino,_  
_Niccuyyutā pecca tamaṃ vajanti ye;_  
_Patanti sattā nirayaṃ avaṃsirā,_  
_Esāmagandho na hi maṃsabhojanaṃ. (10)_  

</Paragraph>

</Division>
<Division number="252" book="sn">

<Paragraph number="289" type="verses" book="sn">
_Na macchamaṃsānamanāsakattaṃ,_  
_Na naggiyaṃ na muṇḍiyaṃ jaṭājallaṃ;_  
_Kharājināni nāggihuttassupasevanā,_  
_Ye vāpi loke amarā bahū tapā;_  
_Mantāhutī yaññamutūpasevanā,_  
_Sodhenti maccaṃ avitiṇṇakaṅkhaṃ. (11)_  

</Paragraph>

</Division>
<Division number="253" book="sn">

<Paragraph number="290" type="verses" book="sn">
_Sotesu gutto viditindriyo care,_  
_Dhamme ṭhito ajjavamaddave rato;_  
_Saṅgātigo sabbadukkhappahīno,_  
_Na lippati diṭṭhasutesu dhīro”. (12)_  

</Paragraph>

</Division>
<Division number="254" book="sn">

<Paragraph number="291" type="verses" book="sn">
_Iccetamatthaṃ bhagavā punappunaṃ,_  
_Akkhāsi naṃ vedayi mantapāragū;_  
_Citrāhi gāthāhi munī pakāsayi,_  
_Nirāmagandho asito durannayo. (13)_  

</Paragraph>

</Division>
<Division number="255" book="sn">

<Paragraph number="292" type="verses" book="sn">
_Sutvāna buddhassa subhāsitaṃ padaṃ,_  
_Nirāmagandhaṃ sabbadukkhappanūdanaṃ;_  
_Nīcamano vandi tathāgatassa,_  
_Tattheva pabbajjamarocayitthāti. (14)_  


</Paragraph>
</Division>

<Paragraph number="293" type="center" book="sn">
Āmagandhasuttaṃ dutiyaṃ.
</Paragraph>
