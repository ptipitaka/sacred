* Kapilasutta (dhammacariyasutta)
</Paragraph>

<Division number="277" book="sn" bookNo="18" page="321">

<Paragraph number="323" type="verses" book="sn">
_Dhammacariyaṃ brahmacariyaṃ,_  
_Etadāhu vasuttamaṃ;_  
_Pabbajitopi ce hoti,_  
_Agārā anagāriyaṃ. (1)_  

</Paragraph>

</Division>
<Division number="278" book="sn" bookNo="18" page="321">

<Paragraph number="324" type="verses" book="sn">
_So ce mukharajātiko,_  
_vihesābhirato mago;_  
_Jīvitaṃ tassa pāpiyo,_  
_rajaṃ vaḍḍheti attano. (2)_  

</Paragraph>

</Division>
<Division number="279" book="sn" bookNo="18" page="321">

<Paragraph number="325" type="verses" book="sn">
_Kalahābhirato bhikkhu,_  
_Mohadhammena āvuto;_  
_Akkhātampi na jānāti,_  
_Dhammaṃ buddhena desitaṃ. (3)_  

</Paragraph>

</Division>
<Division number="280" book="sn" bookNo="18" page="321">

<Paragraph number="326" type="verses" book="sn">
_Vihesaṃ bhāvitattānaṃ,_  
_avijjāya purakkhato;_  
_Saṃkilesaṃ na jānāti,_  
_maggaṃ nirayagāminaṃ. (4)_  

</Paragraph>

</Division>
<Division number="281" book="sn" bookNo="18" page="321">

<Paragraph number="327" type="verses" book="sn">
_Vinipātaṃ samāpanno,_  
_gabbhā gabbhaṃ tamā tamaṃ;_  
_Sa ve tādisako bhikkhu,_  
_pecca dukkhaṃ nigacchati. (5)_  

</Paragraph>

</Division>
<Division number="282" book="sn" bookNo="18" page="321">

<Paragraph number="328" type="verses" book="sn">
_Gūthakūpo yathā assa,_  
_sampuṇṇo gaṇavassiko;_  
_Yo ca evarūpo assa,_  
_dubbisodho hi sāṅgaṇo. (6)_  

</Paragraph>

</Division>
<Division number="283" book="sn" bookNo="18" page="321">

<Paragraph number="329" type="verses" book="sn">
_Yaṃ evarūpaṃ jānātha,_  
_bhikkhavo gehanissitaṃ;_  
_Pāpicchaṃ pāpasaṅkappaṃ,_  
_pāpaācāragocaraṃ. (7)_  

</Paragraph>

</Division>
<Division number="284" book="sn" bookNo="18" page="322">

<Paragraph number="330" type="verses" book="sn">
_Sabbe samaggā hutvāna,_  
_abhinibbajjiyātha naṃ;_  
_Kāraṇḍavaṃ niddhamatha,_  
_kasambuṃ apakassatha. (8)_  

</Paragraph>

</Division>
<Division number="285" book="sn" bookNo="18" page="322">

<Paragraph number="331" type="verses" book="sn">
_Tato palāpe vāhetha,_  
_Assamaṇe samaṇamānine;_  
_Niddhamitvāna pāpicche,_  
_Pāpaācāragocare. (9)_  

</Paragraph>

</Division>
<Division number="286" book="sn" bookNo="18" page="322">

<Paragraph number="332" type="verses" book="sn">
_Suddhā suddhehi saṃvāsaṃ,_  
_Kappayavho patissatā;_  
_Tato samaggā nipakā,_  
_Dukkhassantaṃ karissathāti. (10)_  


</Paragraph>
</Division>

<Paragraph number="333" type="center" book="sn">
Dhammacariyasuttaṃ chaṭṭhaṃ.
</Paragraph>
