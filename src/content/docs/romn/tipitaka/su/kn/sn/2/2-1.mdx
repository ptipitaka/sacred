<Division number="224" book="sn">

<Paragraph number="260" type="verses" book="sn">
_Yānīdha bhūtāni samāgatāni,_  
_Bhummāni vā yāni va antalikkhe;_  
_Sabbeva bhūtā sumanā bhavantu,_  
_Athopi sakkacca suṇantu bhāsitaṃ. (1)_  

</Paragraph>

</Division>
<Division number="225" book="sn">

<Paragraph number="261" type="verses" book="sn">
_Tasmā hi bhūtā nisāmetha sabbe,_  
_Mettaṃ karotha mānusiyā pajāya;_  
_Divā ca ratto ca haranti ye baliṃ,_  
_Tasmā hi ne rakkhatha appamattā. (2)_  

</Paragraph>

</Division>
<Division number="226" book="sn">

<Paragraph number="262" type="verses" book="sn">
_Yaṃ kiñci vittaṃ idha vā huraṃ vā,_  
_Saggesu vā yaṃ ratanaṃ paṇītaṃ;_  
_Na no samaṃ atthi tathāgatena,_  
_Idampi buddhe ratanaṃ paṇītaṃ;_  
_Etena saccena suvatthi hotu. (3)_  

</Paragraph>

</Division>
<Division number="227" book="sn">

<Paragraph number="263" type="verses" book="sn">
_Khayaṃ virāgaṃ amataṃ paṇītaṃ,_  
_Yadajjhagā sakyamunī samāhito;_  
_Na tena dhammena samatthi kiñci,_  
_Idampi dhamme ratanaṃ paṇītaṃ;_  
_Etena saccena suvatthi hotu. (4)_  

</Paragraph>

</Division>
<Division number="228" book="sn">

<Paragraph number="264" type="verses" book="sn">
_Yaṃ buddhaseṭṭho parivaṇṇayī suciṃ,_  
_Samādhimānantarikaññamāhu;_  
_Samādhinā tena samo na vijjati,_  
_Idampi dhamme ratanaṃ paṇītaṃ;_  
_Etena saccena suvatthi hotu. (5)_  

</Paragraph>

</Division>
<Division number="229" book="sn">

<Paragraph number="265" type="verses" book="sn">
_Ye puggalā aṭṭha sataṃ pasatthā,_  
_Cattāri etāni yugāni honti;_  
_Te dakkhiṇeyyā sugatassa sāvakā,_  
_Etesu dinnāni mahapphalāni;_  
_Idampi saṃghe ratanaṃ paṇītaṃ,_  
_Etena saccena suvatthi hotu. (6)_  

</Paragraph>

</Division>
<Division number="230" book="sn">

<Paragraph number="266" type="verses" book="sn">
_Ye suppayuttā manasā daḷhena,_  
_Nikkāmino gotamasāsanamhi;_  
_Te pattipattā amataṃ vigayha,_  
_Laddhā mudhā nibbutiṃ bhuñjamānā;_  
_Idampi saṃghe ratanaṃ paṇītaṃ,_  
_Etena saccena suvatthi hotu. (7)_  

</Paragraph>

</Division>
<Division number="231" book="sn">

<Paragraph number="267" type="verses" book="sn">
_Yathindakhīlo pathavissito siyā,_  
_Catubbhi vātehi asampakampiyo;_  
_Tathūpamaṃ sappurisaṃ vadāmi,_  
_Yo ariyasaccāni avecca passati;_  
_Idampi saṃghe ratanaṃ paṇītaṃ,_  
_Etena saccena suvatthi hotu. (8)_  

</Paragraph>

</Division>
<Division number="232" book="sn">

<Paragraph number="268" type="verses" book="sn">
_Ye ariyasaccāni vibhāvayanti,_  
_Gambhīrapaññena sudesitāni;_  
_Kiñcāpi te honti bhusaṃ pamattā,_  
_Na te bhavaṃ aṭṭhamamādiyanti;_  
_Idampi saṃghe ratanaṃ paṇītaṃ,_  
_Etena saccena suvatthi hotu. (9)_  

</Paragraph>

</Division>
<Division number="233" book="sn">

<Paragraph number="269" type="verses" book="sn">
_Sahāvassa dassanasampadāya,_  
_Tayassu dhammā jahitā bhavanti;_  
_Sakkāyadiṭṭhī vicikicchitañca,_  
_Sīlabbataṃ vāpi yadatthi kiñci. (10)_  

</Paragraph>

</Division>
<Division number="234" book="sn">

<Paragraph number="270" type="verses" book="sn">
_Catūhapāyehi ca vippamutto,_  
_Chaccābhiṭhānāni abhabba kātuṃ;_  
_Idampi saṃghe ratanaṃ paṇītaṃ,_  
_Etena saccena suvatthi hotu. (11)_  

</Paragraph>

</Division>
<Division number="235" book="sn">

<Paragraph number="271" type="verses" book="sn">
_Kiñcāpi so kamma karoti pāpakaṃ,_  
_Kāyena vācā uda cetasā vā;_  
_Abhabba so tassa paṭicchadāya,_  
_Abhabbatā diṭṭhapadassa vuttā;_  
_Idampi saṃghe ratanaṃ paṇītaṃ,_  
_Etena saccena suvatthi hotu. (12)_  

</Paragraph>

</Division>
<Division number="236" book="sn">

<Paragraph number="272" type="verses" book="sn">
_Vanappagumbe yatha phussitagge,_  
_Gimhānamāse paṭhamasmiṃ gimhe;_  
_Tathūpamaṃ dhammavaraṃ adesayi,_  
_Nibbānagāmiṃ paramaṃ hitāya;_  
_Idampi buddhe ratanaṃ paṇītaṃ,_  
_Etena saccena suvatthi hotu. (13)_  

</Paragraph>

</Division>
<Division number="237" book="sn">

<Paragraph number="273" type="verses" book="sn">
_Varo varaññū varado varāharo,_  
_Anuttaro dhammavaraṃ adesayi;_  
_Idampi buddhe ratanaṃ paṇītaṃ,_  
_Etena saccena suvatthi hotu. (14)_  

</Paragraph>

</Division>
<Division number="238" book="sn">

<Paragraph number="274" type="verses" book="sn">
_Khīṇaṃ purāṇaṃ nava natthi sambhavaṃ,_  
_Virattacittāyatike bhavasmiṃ;_  
_Te khīṇabījā avirūḷhichandā,_  
_Nibbanti dhīrā yathāyaṃ padīpo;_  
_Idampi saṃghe ratanaṃ paṇītaṃ,_  
_Etena saccena suvatthi hotu. (15)_  

</Paragraph>

</Division>
<Division number="239" book="sn">

<Paragraph number="275" type="verses" book="sn">
_Yānīdha bhūtāni samāgatāni,_  
_Bhummāni vā yāni va antalikkhe;_  
_Tathāgataṃ devamanussapūjitaṃ,_  
_Buddhaṃ namassāma suvatthi hotu. (16)_  

</Paragraph>

</Division>
<Division number="240" book="sn">

<Paragraph number="276" type="verses" book="sn">
_Yānīdha bhūtāni samāgatāni,_  
_Bhummāni vā yāni va antalikkhe;_  
_Tathāgataṃ devamanussapūjitaṃ,_  
_Dhammaṃ namassāma suvatthi hotu. (17)_  

</Paragraph>

</Division>
<Division number="241" book="sn">

<Paragraph number="277" type="verses" book="sn">
_Yānīdha bhūtāni samāgatāni,_  
_Bhummāni vā yāni va antalikkhe;_  
_Tathāgataṃ devamanussapūjitaṃ,_  
_Saṃghaṃ namassāma suvatthi hotūti. (18)_  


</Paragraph>
