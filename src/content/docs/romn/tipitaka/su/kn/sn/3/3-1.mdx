<Paragraph number="472" type="verses" book="sn">
_“Pabbajjaṃ kittayissāmi,_  
_yathā pabbaji cakkhumā;_  
_Yathā vīmaṃsamāno so,_  
_pabbajjaṃ samarocayi. (1)_  

</Paragraph>

</Division>
<Division number="409" book="sn" bookNo="18" page="340">

<Paragraph number="473" type="verses" book="sn">
_Sambādhoyaṃ gharāvāso,_  
_rajassāyatanaṃ iti;_  
_Abbhokāsova pabbajjā,_  
_iti disvāna pabbaji. (2)_  

</Paragraph>

</Division>
<Division number="410" book="sn" bookNo="18" page="340">

<Paragraph number="474" type="verses" book="sn">
_Pabbajitvāna kāyena,_  
_pāpakammaṃ vivajjayi;_  
_Vacīduccaritaṃ hitvā,_  
_ājīvaṃ parisodhayi. (3)_  

</Paragraph>

</Division>
<Division number="411" book="sn" bookNo="18" page="340">

<Paragraph number="475" type="verses" book="sn">
_Agamā rājagahaṃ buddho,_  
_magadhānaṃ giribbajaṃ;_  
_Piṇḍāya abhihāresi,_  
_ākiṇṇavaralakkhaṇo”. (4)_  

</Paragraph>

</Division>
<Division number="412" book="sn" bookNo="18" page="340">

<Paragraph number="476" type="verses" book="sn">
_Tamaddasā bimbisāro,_  
_pāsādasmiṃ patiṭṭhito;_  
_Disvā lakkhaṇasampannaṃ,_  
_imamatthaṃ abhāsatha. (5)_  

</Paragraph>

</Division>
<Division number="413" book="sn" bookNo="18" page="340">

<Paragraph number="477" type="verses" book="sn">
_“Imaṃ bhonto nisāmetha,_  
_abhirūpo brahā suci;_  
_Caraṇena ca sampanno,_  
_yugamattañca pekkhati. (6)_  

</Paragraph>

</Division>
<Division number="414" book="sn" bookNo="18" page="340">

<Paragraph number="478" type="verses" book="sn">
_Okkhittacakkhu satimā,_  
_nāyaṃ nīcakulāmiva;_  
_Rājadūtābhidhāvantu,_  
_kuhiṃ bhikkhu gamissati”. (7)_  

</Paragraph>

</Division>
<Division number="415" book="sn" bookNo="18" page="340">

<Paragraph number="479" type="verses" book="sn">
_Te pesitā rājadūtā,_  
_piṭṭhito anubandhisuṃ;_  
_Kuhiṃ gamissati bhikkhu,_  
_kattha vāso bhavissati. (8)_  

</Paragraph>

</Division>
<Division number="416" book="sn" bookNo="18" page="340">

<Paragraph number="480" type="verses" book="sn">
_Sapadānaṃ caramāno,_  
_guttadvāro susaṃvuto;_  
_Khippaṃ pattaṃ apūresi,_  
_sampajāno paṭissato. (9)_  

</Paragraph>

</Division>
<Division number="417" book="sn" bookNo="18" page="340">

<Paragraph number="481" type="verses" book="sn">
_Piṇḍacāraṃ caritvāna,_  
_Nikkhamma nagarā muni;_  
_Paṇḍavaṃ abhihāresi,_  
_Ettha vāso bhavissati. (10)_  

</Paragraph>

</Division>
<Division number="418" book="sn" bookNo="18" page="340">

<Paragraph number="482" type="verses" book="sn">
_Disvāna vāsūpagataṃ,_  
_Tayo dūtā upāvisuṃ;_  
_Tesu ekova āgantvā,_  
_Rājino paṭivedayi. (11)_  

</Paragraph>

</Division>
<Division number="419" book="sn" bookNo="18" page="341">

<Paragraph number="483" type="verses" book="sn">
_“Esa bhikkhu mahārāja,_  
_Paṇḍavassa puratthato;_  
_Nisinno byagghusabhova,_  
_Sīhova girigabbhare”. (12)_  

</Paragraph>

</Division>
<Division number="420" book="sn" bookNo="18" page="341">

<Paragraph number="484" type="verses" book="sn">
_Sutvāna dūtavacanaṃ,_  
_Bhaddayānena khattiyo;_  
_Taramānarūpo niyyāsi,_  
_Yena paṇḍavapabbato. (13)_  

</Paragraph>

</Division>
<Division number="421" book="sn" bookNo="18" page="341">

<Paragraph number="485" type="verses" book="sn">
_Sa yānabhūmiṃ yāyitvā,_  
_Yānā oruyha khattiyo;_  
_Pattiko upasaṅkamma,_  
_Āsajja naṃ upāvisi. (14)_  

</Paragraph>

</Division>
<Division number="422" book="sn" bookNo="18" page="341">

<Paragraph number="486" type="verses" book="sn">
_Nisajja rājā sammodi,_  
_Kathaṃ sāraṇīyaṃ tato;_  
_Kathaṃ so vītisāretvā,_  
_Imamatthaṃ abhāsatha. (15)_  

</Paragraph>

</Division>
<Division number="423" book="sn" bookNo="18" page="341">

<Paragraph number="487" type="verses" book="sn">
_“Yuvā ca daharo cāsi,_  
_Paṭhamuppattiko susu;_  
_Vaṇṇārohena sampanno,_  
_Jātimā viya khattiyo. (16)_  

</Paragraph>

</Division>
<Division number="424" book="sn" bookNo="18" page="341">

<Paragraph number="488" type="verses" book="sn">
_Sobhayanto anīkaggaṃ,_  
_Nāgasaṅghapurakkhato;_  
_Dadāmi bhoge bhuñjassu,_  
_Jātiṃ akkhāhi pucchito”. (17)_  

</Paragraph>

</Division>
<Division number="425" book="sn" bookNo="18" page="341">

<Paragraph number="489" type="verses" book="sn">
_“Ujuṃ janapado rāja,_  
_Himavantassa passato;_  
_Dhanaviriyena sampanno,_  
_Kosalesu niketino. (18)_  

</Paragraph>

</Division>
<Division number="426" book="sn" bookNo="18" page="341">

<Paragraph number="490" type="verses" book="sn">
_Ādiccā nāma gottena,_  
_Sākiyā nāma jātiyā;_  
_Tamhā kulā pabbajitomhi,_  
_Na kāme abhipatthayaṃ. (19)_  

</Paragraph>

</Division>
<Division number="427" book="sn" bookNo="18" page="341">

<Paragraph number="491" type="verses" book="sn">
_Kāmesvādīnavaṃ disvā,_  
_Nekkhammaṃ daṭṭhu khemato;_  
_Padhānāya gamissāmi,_  
_Ettha me rañjatī mano”ti. (20)_  


</Paragraph>
</Division>

<Paragraph number="492" type="center" book="sn">
Pabbajjāsuttaṃ paṭhamaṃ.
</Paragraph>
