<Paragraph number="2" type="verses" book="sn">
_Yo uppatitaṃ vineti kodhaṃ,_  
_Visaṭaṃ sappavisaṃva osadhehi;_  
_So bhikkhu jahāti orapāraṃ,_  
_Urago jiṇṇamivattacaṃ purāṇaṃ._  

</Paragraph>

</Division>
<Division number="2" book="sn">

<Paragraph number="3" type="verses" book="sn">
_Yo rāgamudacchidā asesaṃ,_  
_Bhisapupphaṃva saroruhaṃ vigayha;_  
_So bhikkhu jahāti orapāraṃ,_  
_Urago jiṇṇamivattacaṃ purāṇaṃ._  

</Paragraph>

</Division>
<Division number="3" book="sn">

<Paragraph number="4" type="verses" book="sn">
_Yo taṇhamudacchidā asesaṃ,_  
_Saritaṃ sīghasaraṃ visosayitvā;_  
_So bhikkhu jahāti orapāraṃ,_  
_Urago jiṇṇamivattacaṃ purāṇaṃ._  

</Paragraph>

</Division>
<Division number="4" book="sn">

<Paragraph number="5" type="verses" book="sn">
_Yo mānamudabbadhī asesaṃ,_  
_Naḷasetuṃva sudubbalaṃ mahogho;_  
_So bhikkhu jahāti orapāraṃ,_  
_Urago jiṇṇamivattacaṃ purāṇaṃ._  

</Paragraph>

</Division>
<Division number="5" book="sn">

<Paragraph number="6" type="verses" book="sn">
_Yo nājjhagamā bhavesu sāraṃ,_  
_Vicinaṃ pupphamivā udumbaresu;_  
_So bhikkhu jahāti orapāraṃ,_  
_Urago jiṇṇamivattacaṃ purāṇaṃ._  

</Paragraph>

</Division>
<Division number="6" book="sn">

<Paragraph number="7" type="verses" book="sn">
_Yassantarato na santi kopā,_  
_Itibhavābhavatañca vītivatto;_  
_So bhikkhu jahāti orapāraṃ,_  
_Urago jiṇṇamivattacaṃ purāṇaṃ._  

</Paragraph>

</Division>
<Division number="7" book="sn">

<Paragraph number="8" type="verses" book="sn">
_Yassa vitakkā vidhūpitā,_  
_Ajjhattaṃ suvikappitā asesā;_  
_So bhikkhu jahāti orapāraṃ,_  
_Urago jiṇṇamivattacaṃ purāṇaṃ._  

</Paragraph>

</Division>
<Division number="8" book="sn">

<Paragraph number="9" type="verses" book="sn">
_Yo nāccasārī na paccasārī,_  
_Sabbaṃ accagamā imaṃ papañcaṃ;_  
_So bhikkhu jahāti orapāraṃ,_  
_Urago jiṇṇamivattacaṃ purāṇaṃ._  

</Paragraph>

</Division>
<Division number="9" book="sn">

<Paragraph number="10" type="verses" book="sn">
_Yo nāccasārī na paccasārī,_  
_Sabbaṃ vitathamidanti ñatva loke;_  
_So bhikkhu jahāti orapāraṃ,_  
_Urago jiṇṇamivattacaṃ purāṇaṃ._  

</Paragraph>

</Division>
<Division number="10" book="sn">

<Paragraph number="11" type="verses" book="sn">
_Yo nāccasārī na paccasārī,_  
_Sabbaṃ vitathamidanti vītalobho;_  
_So bhikkhu jahāti orapāraṃ,_  
_Urago jiṇṇamivattacaṃ purāṇaṃ._  

</Paragraph>

</Division>
<Division number="11" book="sn">

<Paragraph number="12" type="verses" book="sn">
_Yo nāccasārī na paccasārī,_  
_Sabbaṃ vitathamidanti vītarāgo;_  
_So bhikkhu jahāti orapāraṃ,_  
_Urago jiṇṇamivattacaṃ purāṇaṃ._  

</Paragraph>

</Division>
<Division number="12" book="sn">

<Paragraph number="13" type="verses" book="sn">
_Yo nāccasārī na paccasārī,_  
_Sabbaṃ vitathamidanti vītadoso;_  
_So bhikkhu jahāti orapāraṃ,_  
_Urago jiṇṇamivattacaṃ purāṇaṃ._  

</Paragraph>

</Division>
<Division number="13" book="sn">

<Paragraph number="14" type="verses" book="sn">
_Yo nāccasārī na paccasārī,_  
_Sabbaṃ vitathamidanti vītamoho;_  
_So bhikkhu jahāti orapāraṃ,_  
_Urago jiṇṇamivattacaṃ purāṇaṃ._  

</Paragraph>

</Division>
<Division number="14" book="sn">

<Paragraph number="15" type="verses" book="sn">
_Yassānusayā na santi keci,_  
_Mūlā ca akusalā samūhatāse;_  
_So bhikkhu jahāti orapāraṃ,_  
_Urago jiṇṇamivattacaṃ purāṇaṃ._  

</Paragraph>

</Division>
<Division number="15" book="sn">

<Paragraph number="16" type="verses" book="sn">
_Yassa darathajā na santi keci,_  
_Oraṃ āgamanāya paccayāse;_  
_So bhikkhu jahāti orapāraṃ,_  
_Urago jiṇṇamivattacaṃ purāṇaṃ._  

</Paragraph>

</Division>
<Division number="16" book="sn">

<Paragraph number="17" type="verses" book="sn">
_Yassa vanathajā na santi keci,_  
_Vinibandhāya bhavāya hetukappā;_  
_So bhikkhu jahāti orapāraṃ,_  
_Urago jiṇṇamivattacaṃ purāṇaṃ._  

</Paragraph>

</Division>
<Division number="17" book="sn">

<Paragraph number="18" type="verses" book="sn">
_Yo nīvaraṇe pahāya pañca,_  
_Anigho tiṇṇakathaṃkatho visallo;_  
_So bhikkhu jahāti orapāraṃ,_  
_Urago jiṇṇamivattacaṃ purāṇanti._  


</Paragraph>
</Division>

<Paragraph number="19" type="center" book="sn">
Uragasuttaṃ paṭhamaṃ.
