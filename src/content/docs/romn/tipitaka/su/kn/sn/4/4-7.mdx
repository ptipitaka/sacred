<Division number="821" book="sn">

<Paragraph number="988" type="verses" book="sn">
_“Methunamanuyuttassa, __(iccāyasmā tisso metteyyo)___  
_Vighātaṃ brūhi mārisa;_  
_Sutvāna tava sāsanaṃ,_  
_Viveke sikkhissāmase”. (1)_  

</Paragraph>

</Division>
<Division number="822" book="sn">

<Paragraph number="989" type="verses" book="sn">
_“Methunamanuyuttassa, __(metteyyāti bhagavā)___  
_Mussate vāpi sāsanaṃ;_  
_Micchā ca paṭipajjati,_  
_Etaṃ tasmiṃ anāriyaṃ. (2)_  

</Paragraph>

</Division>
<Division number="823" book="sn">

<Paragraph number="990" type="verses" book="sn">
_Eko pubbe caritvāna,_  
_methunaṃ yo nisevati;_  
_Yānaṃ bhantaṃva taṃ loke,_  
_hīnamāhu puthujjanaṃ. (3)_  

</Paragraph>

</Division>
<Division number="824" book="sn">

<Paragraph number="991" type="verses" book="sn">
_Yaso kitti ca yā pubbe,_  
_hāyate vāpi tassa sā;_  
_Etampi disvā sikkhetha,_  
_methunaṃ vippahātave. (4)_  

</Paragraph>

</Division>
<Division number="825" book="sn">

<Paragraph number="992" type="verses" book="sn">
_Saṅkappehi pareto so,_  
_kapaṇo viya jhāyati;_  
_Sutvā paresaṃ nigghosaṃ,_  
_maṅku hoti tathāvidho. (5)_  

</Paragraph>

</Division>
<Division number="826" book="sn">

<Paragraph number="993" type="verses" book="sn">
_Atha satthāni kurute,_  
_paravādehi codito;_  
_Esa khvassa mahāgedho,_  
_mosavajjaṃ pagāhati. (6)_  

</Paragraph>

</Division>
<Division number="827" book="sn">

<Paragraph number="994" type="verses" book="sn">
_Paṇḍitoti samaññāto,_  
_ekacariyaṃ adhiṭṭhito;_  
_Athāpi methune yutto,_  
_mandova parikissati. (7)_  

</Paragraph>

</Division>
<Division number="828" book="sn">

<Paragraph number="995" type="verses" book="sn">
_Etamādīnavaṃ ñatvā,_  
_Muni pubbāpare idha;_  
_Ekacariyaṃ daḷhaṃ kayirā,_  
_Na nisevetha methunaṃ. (8)_  

</Paragraph>

</Division>
<Division number="829" book="sn">

<Paragraph number="996" type="verses" book="sn">
_Vivekaññeva sikkhetha,_  
_etaṃ ariyānamuttamaṃ;_  
_Na tena seṭṭho maññetha,_  
_sa ve nibbānasantike. (9)_  

</Paragraph>

</Division>
<Division number="830" book="sn">

<Paragraph number="997" type="verses" book="sn">
_Rittassa munino carato,_  
_Kāmesu anapekkhino;_  
_Oghatiṇṇassa pihayanti,_  
_Kāmesu gadhitā pajā”ti. (10)_  


</Paragraph>
</Division>

<Paragraph number="998" type="center" book="sn">
Tissametteyyasuttaṃ sattamaṃ.
</Paragraph>
