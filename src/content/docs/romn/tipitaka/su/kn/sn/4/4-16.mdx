<Division number="962" book="sn">

<Paragraph number="1138" type="verses" book="sn">
_“Na me diṭṭho ito pubbe, __(iccāyasmā sāriputto)___  
_Na suto uda kassaci;_  
_Evaṃ vagguvado satthā,_  
_Tusitā gaṇimāgato. (1)_  

</Paragraph>

</Division>
<Division number="963" book="sn">

<Paragraph number="1139" type="verses" book="sn">
_Sadevakassa lokassa,_  
_yathā dissati cakkhumā;_  
_Sabbaṃ tamaṃ vinodetvā,_  
_ekova ratimajjhagā. (2)_  

</Paragraph>

</Division>
<Division number="964" book="sn">

<Paragraph number="1140" type="verses" book="sn">
_Taṃ buddhaṃ asitaṃ tādiṃ,_  
_akuhaṃ gaṇimāgataṃ;_  
_Bahūnamidha baddhānaṃ,_  
_atthi pañhena āgamaṃ. (3)_  

</Paragraph>

</Division>
<Division number="965" book="sn">

<Paragraph number="1141" type="verses" book="sn">
_Bhikkhuno vijigucchato,_  
_bhajato rittamāsanaṃ;_  
_Rukkhamūlaṃ susānaṃ vā,_  
_pabbatānaṃ guhāsu vā. (4)_  

</Paragraph>

</Division>
<Division number="966" book="sn">

<Paragraph number="1142" type="verses" book="sn">
_Uccāvacesu sayanesu,_  
_kīvanto tattha bheravā;_  
_Yehi bhikkhu na vedheyya,_  
_nigghose sayanāsane. (5)_  

</Paragraph>

</Division>
<Division number="967" book="sn">

<Paragraph number="1143" type="verses" book="sn">
_Katī parissayā loke,_  
_gacchato agataṃ disaṃ;_  
_Ye bhikkhu abhisambhave,_  
_pantamhi sayanāsane. (6)_  

</Paragraph>

</Division>
<Division number="968" book="sn">

<Paragraph number="1144" type="verses" book="sn">
_Kyāssa byappathayo assu,_  
_kyāssassu idha gocarā;_  
_Kāni sīlabbatānāssu,_  
_pahitattassa bhikkhuno. (7)_  

</Paragraph>

</Division>
<Division number="969" book="sn">

<Paragraph number="1145" type="verses" book="sn">
_Kaṃ so sikkhaṃ samādāya,_  
_ekodi nipako sato;_  
_Kammāro rajatasseva,_  
_niddhame malamattano”. (8)_  

</Paragraph>

</Division>
<Division number="970" book="sn">

<Paragraph number="1146" type="verses" book="sn">
_“Vijigucchamānassa yadidaṃ phāsu, __(sāriputtāti bhagavā)___  
_Rittāsanaṃ sayanaṃ sevato ce;_  
_Sambodhikāmassa yathānudhammaṃ,_  
_Taṃ te pavakkhāmi yathā pajānaṃ. (9)_  

</Paragraph>

</Division>
<Division number="971" book="sn">

<Paragraph number="1147" type="verses" book="sn">
_Pañcannaṃ dhīro bhayānaṃ na bhāye,_  
_Bhikkhu sato sapariyantacārī;_  
_Ḍaṃsādhipātānaṃ sarīsapānaṃ,_  
_Manussaphassānaṃ catuppadānaṃ. (10)_  

</Paragraph>

</Division>
<Division number="972" book="sn">

<Paragraph number="1148" type="verses" book="sn">
_Paradhammikānampi na santaseyya,_  
_Disvāpi tesaṃ bahubheravāni;_  
_Athāparāni abhisambhaveyya,_  
_Parissayāni kusalānuesī. (11)_  

</Paragraph>

</Division>
<Division number="973" book="sn">

<Paragraph number="1149" type="verses" book="sn">
_Ātaṅkaphassena khudāya phuṭṭho,_  
_Sītaṃ athuṇhaṃ adhivāsayeyya;_  
_So tehi phuṭṭho bahudhā anoko,_  
_Viriyaṃ parakkammadaḷhaṃ kareyya. (12)_  

</Paragraph>

</Division>
<Division number="974" book="sn">

<Paragraph number="1150" type="verses" book="sn">
_Theyyaṃ na kāre na musā bhaṇeyya,_  
_Mettāya phasse tasathāvarāni;_  
_Yadāvilattaṃ manaso vijaññā,_  
_Kaṇhassa pakkhoti vinodayeyya. (13)_  

</Paragraph>

</Division>
<Division number="975" book="sn">

<Paragraph number="1151" type="verses" book="sn">
_Kodhātimānassa vasaṃ na gacche,_  
_Mūlampi tesaṃ palikhañña tiṭṭhe;_  
_Athappiyaṃ vā pana appiyaṃ vā,_  
_Addhā bhavanto abhisambhaveyya. (14)_  

</Paragraph>

</Division>
<Division number="976" book="sn">

<Paragraph number="1152" type="verses" book="sn">
_Paññaṃ purakkhatvā kalyāṇapīti,_  
_Vikkhambhaye tāni parissayāni;_  
_Aratiṃ sahetha sayanamhi pante,_  
_Caturo sahetha paridevadhamme. (15)_  

</Paragraph>

</Division>
<Division number="977" book="sn">

<Paragraph number="1153" type="verses" book="sn">
_Kiṃsū asissāmi kuva vā asissaṃ,_  
_Dukkhaṃ vata settha kvajja sessaṃ;_  
_Ete vitakke paridevaneyye,_  
_Vinayetha sekho aniketacārī. (16)_  

</Paragraph>

</Division>
<Division number="978" book="sn">

<Paragraph number="1154" type="verses" book="sn">
_Annañca laddhā vasanañca kāle,_  
_Mattaṃ so jaññā idha tosanatthaṃ;_  
_So tesu gutto yatacāri gāme,_  
_Rusitopi vācaṃ pharusaṃ na vajjā. (17)_  

</Paragraph>

</Division>
<Division number="979" book="sn">

<Paragraph number="1155" type="verses" book="sn">
_Okkhittacakkhu na ca pādalolo,_  
_Jhānānuyutto bahujāgarassa;_  
_Upekkhamārabbha samāhitatto,_  
_Takkāsayaṃ kukkucciyūpachinde. (18)_  

</Paragraph>

</Division>
<Division number="980" book="sn">

<Paragraph number="1156" type="verses" book="sn">
_Cudito vacībhi satimābhinande,_  
_Sabrahmacārīsu khilaṃ pabhinde;_  
_Vācaṃ pamuñce kusalaṃ nātivelaṃ,_  
_Janavādadhammāya na cetayeyya. (19)_  

</Paragraph>

</Division>
<Division number="981" book="sn">

<Paragraph number="1157" type="verses" book="sn">
_Athāparaṃ pañca rajāni loke,_  
_Yesaṃ satīmā vinayāya sikkhe;_  
_Rūpesu saddesu atho rasesu,_  
_Gandhesu phassesu sahetha rāgaṃ. (20)_  

</Paragraph>

</Division>
<Division number="982" book="sn">

<Paragraph number="1158" type="verses" book="sn">
_Etesu dhammesu vineyya chandaṃ,_  
_Bhikkhu satimā suvimuttacitto;_  
_Kālena so sammā dhammaṃ parivīmaṃsamāno,_  
_Ekodibhūto vihane tamaṃ so”ti. (21)_  


</Paragraph>
</Division>

<Paragraph number="1159" type="center" book="sn">
Sāriputtasuttaṃ soḷasamaṃ.



</Paragraph>

<Paragraph number="1160" type="center" book="sn">
Aṭṭhakavaggo catuttho.



</Paragraph>

<Paragraph number="1161" type="center" book="sn">
Tassuddānaṃ


</Paragraph>

<Paragraph number="1162" type="verses" book="sn">
_Kāmaṃ guhañca duṭṭhā ca,_  
_suddhañca paramā jarā;_  
_Metteyyo ca pasūro ca,_  
_māgaṇḍi purābhedanaṃ._  

</Paragraph>

<Paragraph number="1163" type="verses" book="sn">
_Kalahaṃ dve ca byūhāni,_  
_punadeva tuvaṭṭakaṃ;_  
_Attadaṇḍavaraṃ suttaṃ,_  
_therapuṭṭhena soḷasa;_  
_Iti etāni suttāni,_  
_sabbānaṭṭhakavaggikāti._
</Paragraph>
