<Division number="885" book="sn">

<Paragraph number="1057" type="verses" book="sn">
_“Sakaṃsakaṃdiṭṭhiparibbasānā,_  
_Viggayha nānā kusalā vadanti;_  
_Yo evaṃ jānāti sa vedi dhammaṃ,_  
_Idaṃ paṭikkosamakevalī so. (1)_  

</Paragraph>

</Division>
<Division number="886" book="sn">

<Paragraph number="1058" type="verses" book="sn">
_Evampi viggayha vivādayanti,_  
_Bālo paro akkusaloti cāhu;_  
_Sacco nu vādo katamo imesaṃ,_  
_Sabbeva hīme kusalāvadānā”. (2)_  

</Paragraph>

</Division>
<Division number="887" book="sn">

<Paragraph number="1059" type="verses" book="sn">
_“Parassa ce dhammamanānujānaṃ,_  
_Bālomako hoti nihīnapañño;_  
_Sabbeva bālā sunihīnapaññā,_  
_Sabbevime diṭṭhiparibbasānā. (3)_  

</Paragraph>

</Division>
<Division number="888" book="sn">

<Paragraph number="1060" type="verses" book="sn">
_Sandiṭṭhiyā ceva na vīvadātā,_  
_Saṃsuddhapaññā kusalā mutīmā;_  
_Na tesaṃ koci parihīnapañño,_  
_Diṭṭhī hi tesampi tathā samattā. (4)_  

</Paragraph>

</Division>
<Division number="889" book="sn">

<Paragraph number="1061" type="verses" book="sn">
_Na vāhametaṃ tathiyanti brūmi,_  
_Yamāhu bālā mithu aññamaññaṃ;_  
_Sakaṃ sakaṃ diṭṭhimakaṃsu saccaṃ,_  
_Tasmā hi bāloti paraṃ dahanti”. (5)_  

</Paragraph>

</Division>
<Division number="890" book="sn">

<Paragraph number="1062" type="verses" book="sn">
_“Yamāhu saccaṃ tathiyanti eke,_  
_Tamāhu aññe tucchaṃ musāti;_  
_Evampi vigayha vivādayanti,_  
_Kasmā na ekaṃ samaṇā vadanti”. (6)_  

</Paragraph>

</Division>
<Division number="891" book="sn">

<Paragraph number="1063" type="verses" book="sn">
_“Ekañhi saccaṃ na dutīyamatthi,_  
_Yasmiṃ pajā no vivade pajānaṃ;_  
_Nānā te saccāni sayaṃ thunanti,_  
_Tasmā na ekaṃ samaṇā vadanti”. (7)_  

</Paragraph>

</Division>
<Division number="892" book="sn">

<Paragraph number="1064" type="verses" book="sn">
_“Kasmā nu saccāni vadanti nānā,_  
_Pavādiyāse kusalāvadānā;_  
_Saccāni sutāni bahūni nānā,_  
_Udāhu te takkamanussaranti”. (8)_  

</Paragraph>

</Division>
<Division number="893" book="sn">

<Paragraph number="1065" type="verses" book="sn">
_“Na heva saccāni bahūni nānā,_  
_Aññatra saññāya niccāni loke;_  
_Takkañca diṭṭhīsu pakappayitvā,_  
_Saccaṃ musāti dvayadhammamāhu. (9)_  

</Paragraph>

</Division>
<Division number="894" book="sn">

<Paragraph number="1066" type="verses" book="sn">
_Diṭṭhe sute sīlavate mute vā,_  
_Ete ca nissāya vimānadassī;_  
_Vinicchaye ṭhatvā pahassamāno,_  
_Bālo paro akkusaloti cāha. (10)_  

</Paragraph>

</Division>
<Division number="895" book="sn">

<Paragraph number="1067" type="verses" book="sn">
_Yeneva bāloti paraṃ dahāti,_  
_Tenātumānaṃ kusaloti cāha;_  
_Sayamattanā so kusalāvadāno,_  
_Aññaṃ vimāneti tadeva pāva. (11)_  

</Paragraph>

</Division>
<Division number="896" book="sn">

<Paragraph number="1068" type="verses" book="sn">
_Atisāradiṭṭhiyā so samatto,_  
_Mānena matto paripuṇṇamānī;_  
_Sayameva sāmaṃ manasābhisitto,_  
_Diṭṭhī hi sā tassa tathā samattā. (12)_  

</Paragraph>

</Division>
<Division number="897" book="sn">

<Paragraph number="1069" type="verses" book="sn">
_Parassa ce hi vacasā nihīno,_  
_Tumo sahā hoti nihīnapañño;_  
_Atha ce sayaṃ vedagū hoti dhīro,_  
_Na koci bālo samaṇesu atthi. (13)_  

</Paragraph>

</Division>
<Division number="898" book="sn">

<Paragraph number="1070" type="verses" book="sn">
_Aññaṃ ito yābhivadanti dhammaṃ,_  
_Aparaddhā suddhimakevalī te;_  
_Evampi titthyā puthuso vadanti,_  
_Sandiṭṭhirāgena hi tebhirattā. (14)_  

</Paragraph>

</Division>
<Division number="899" book="sn">

<Paragraph number="1071" type="verses" book="sn">
_Idheva suddhiṃ iti vādayanti,_  
_Nāññesu dhammesu visuddhimāhu;_  
_Evampi titthyā puthuso niviṭṭhā,_  
_Sakāyane tattha daḷhaṃ vadānā. (15)_  

</Paragraph>

</Division>
<Division number="900" book="sn">

<Paragraph number="1072" type="verses" book="sn">
_Sakāyane vāpi daḷhaṃ vadāno,_  
_Kamettha bāloti paraṃ daheyya;_  
_Sayaṃva so medhagamāvaheyya,_  
_Paraṃ vadaṃ bālamasuddhidhammaṃ. (16)_  

</Paragraph>

</Division>
<Division number="901" book="sn">

<Paragraph number="1073" type="verses" book="sn">
_Vinicchaye ṭhatvā sayaṃ pamāya,_  
_Uddhaṃsa lokasmiṃ vivādameti;_  
_Hitvāna sabbāni vinicchayāni,_  
_Na medhagaṃ kubbati jantu loke”ti. (17)_  


</Paragraph>
</Division>

<Paragraph number="1074" type="center" book="sn">
Cūḷabyūhasuttaṃ dvādasamaṃ.
</Paragraph>
