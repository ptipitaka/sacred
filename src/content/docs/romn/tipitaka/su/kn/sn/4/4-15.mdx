<Paragraph number="1117" type="verses" book="sn">
_“Attadaṇḍā bhayaṃ jātaṃ,_  
_janaṃ passatha medhagaṃ;_  
_Saṃvegaṃ kittayissāmi,_  
_yathā saṃvijitaṃ mayā. (1)_  

</Paragraph>

</Division>
<Division number="943" book="sn" bookNo="18" page="424">

<Paragraph number="1118" type="verses" book="sn">
_Phandamānaṃ pajaṃ disvā,_  
_macche appodake yathā;_  
_Aññamaññehi byāruddhe,_  
_disvā maṃ bhayamāvisi. (2)_  

</Paragraph>

</Division>
<Division number="944" book="sn" bookNo="18" page="424">

<Paragraph number="1119" type="verses" book="sn">
_Samantamasāro loko,_  
_disā sabbā sameritā;_  
_Icchaṃ bhavanamattano,_  
_nāddasāsiṃ anositaṃ. (3)_  

</Paragraph>

</Division>
<Division number="945" book="sn" bookNo="18" page="424">

<Paragraph number="1120" type="verses" book="sn">
_Osāne tveva byāruddhe,_  
_Disvā me aratī ahu;_  
_Athettha sallamaddakkhiṃ,_  
_Duddasaṃ hadayanissitaṃ. (4)_  

</Paragraph>

</Division>
<Division number="946" book="sn" bookNo="18" page="424">

<Paragraph number="1121" type="verses" book="sn">
_Yena sallena otiṇṇo,_  
_disā sabbā vidhāvati;_  
_Tameva sallamabbuyha,_  
_na dhāvati na sīdati. (5)_  

</Paragraph>

</Division>
<Division number="947" book="sn" bookNo="18" page="424">

<Paragraph number="1122" type="verses" book="sn">
_Tattha sikkhānugīyanti,_  
_yāni loke gadhitāni;_  
_Na tesu pasuto siyā,_  
_nibbijjha sabbaso kāme;_  
_Sikkhe nibbānamattano. (6)_  

</Paragraph>

</Division>
<Division number="948" book="sn" bookNo="18" page="424">

<Paragraph number="1123" type="verses" book="sn">
_Sacco siyā appagabbho,_  
_amāyo rittapesuṇo;_  
_Akkodhano lobhapāpaṃ,_  
_vevicchaṃ vitare muni. (7)_  

</Paragraph>

</Division>
<Division number="949" book="sn" bookNo="18" page="424">

<Paragraph number="1124" type="verses" book="sn">
_Niddaṃ tandiṃ sahe thīnaṃ,_  
_pamādena na saṃvase;_  
_Atimāne na tiṭṭheyya,_  
_nibbānamanaso naro. (8)_  

</Paragraph>

</Division>
<Division number="950" book="sn" bookNo="18" page="425">

<Paragraph number="1125" type="verses" book="sn">
_Mosavajje na nīyetha,_  
_rūpe snehaṃ na kubbaye;_  
_Mānañca parijāneyya,_  
_sāhasā virato care. (9)_  

</Paragraph>

</Division>
<Division number="951" book="sn" bookNo="18" page="425">

<Paragraph number="1126" type="verses" book="sn">
_Purāṇaṃ nābhinandeyya,_  
_Nave khantiṃ na kubbaye;_  
_Hiyyamāne na soceyya,_  
_Ākāsaṃ na sito siyā. (10)_  

</Paragraph>

</Division>
<Division number="952" book="sn" bookNo="18" page="425">

<Paragraph number="1127" type="verses" book="sn">
_Gedhaṃ brūmi mahoghoti,_  
_Ājavaṃ brūmi jappanaṃ;_  
_Ārammaṇaṃ pakappanaṃ,_  
_Kāmapaṅko duraccayo. (11)_  

</Paragraph>

</Division>
<Division number="953" book="sn" bookNo="18" page="425">

<Paragraph number="1128" type="verses" book="sn">
_Saccā avokkamma muni,_  
_Thale tiṭṭhati brāhmaṇo;_  
_Sabbaṃ so paṭinissajja,_  
_Sa ve santoti vuccati. (12)_  

</Paragraph>

</Division>
<Division number="954" book="sn" bookNo="18" page="425">

<Paragraph number="1129" type="verses" book="sn">
_Sa ve vidvā sa vedagū,_  
_Ñatvā dhammaṃ anissito;_  
_Sammā so loke iriyāno,_  
_Na pihetīdha kassaci. (13)_  

</Paragraph>

</Division>
<Division number="955" book="sn" bookNo="18" page="425">

<Paragraph number="1130" type="verses" book="sn">
_Yodha kāme accatari,_  
_Saṅgaṃ loke duraccayaṃ;_  
_Na so socati nājjheti,_  
_Chinnasoto abandhano. (14)_  

</Paragraph>

</Division>
<Division number="956" book="sn" bookNo="18" page="425">

<Paragraph number="1131" type="verses" book="sn">
_Yaṃ pubbe taṃ visosehi,_  
_Pacchā te māhu kiñcanaṃ;_  
_Majjhe ce no gahessasi,_  
_Upasanto carissasi. (15)_  

</Paragraph>

</Division>
<Division number="957" book="sn" bookNo="18" page="425">

<Paragraph number="1132" type="verses" book="sn">
_Sabbaso nāmarūpasmiṃ,_  
_Yassa natthi mamāyitaṃ;_  
_Asatā ca na socati,_  
_Sa ve loke na jīyati. (16)_  

</Paragraph>

</Division>
<Division number="958" book="sn" bookNo="18" page="425">

<Paragraph number="1133" type="verses" book="sn">
_Yassa natthi idaṃ meti,_  
_Paresaṃ vāpi kiñcanaṃ;_  
_Mamattaṃ so asaṃvindaṃ,_  
_Natthi meti na socati. (17)_  

</Paragraph>

</Division>
<Division number="959" book="sn" bookNo="18" page="425">

<Paragraph number="1134" type="verses" book="sn">
_Aniṭṭhurī ananugiddho,_  
_Anejo sabbadhī samo;_  
_Tamānisaṃsaṃ pabrūmi,_  
_Pucchito avikampinaṃ. (18)_  

</Paragraph>

</Division>
<Division number="960" book="sn" bookNo="18" page="425">

<Paragraph number="1135" type="verses" book="sn">
_Anejassa vijānato,_  
_Natthi kāci nisaṅkhati;_  
_Virato so viyārabbhā,_  
_Khemaṃ passati sabbadhi. (19)_  

</Paragraph>

</Division>
<Division number="961" book="sn" bookNo="18" page="425">

<Paragraph number="1136" type="verses" book="sn">
_Na samesu na omesu,_  
_Na ussesu vadate muni;_  
_Santo so vītamaccharo,_  
_Nādeti na nirassatī”ti. (20)_  


</Paragraph>
</Division>

<Paragraph number="1137" type="center" book="sn">
Attadaṇḍasuttaṃ pannarasamaṃ.
</Paragraph>
