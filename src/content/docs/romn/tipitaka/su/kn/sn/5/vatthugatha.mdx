<Division number="983" book="sn">

<Paragraph number="1164" type="verses" book="sn">
_Kosalānaṃ purā rammā,_  
_agamā dakkhiṇāpathaṃ;_  
_Ākiñcaññaṃ patthayāno,_  
_brāhmaṇo mantapāragū. (1)_  

</Paragraph>

</Division>
<Division number="984" book="sn">

<Paragraph number="1165" type="verses" book="sn">
_So assakassa visaye,_  
_aḷakassa samāsane;_  
_Vasi godhāvarīkūle,_  
_uñchena ca phalena ca. (2)_  

</Paragraph>

</Division>
<Division number="985" book="sn">

<Paragraph number="1166" type="verses" book="sn">
_Tasseva upanissāya,_  
_gāmo ca vipulo ahu;_  
_Tato jātena āyena,_  
_mahāyaññamakappayi. (3)_  

</Paragraph>

</Division>
<Division number="986" book="sn">

<Paragraph number="1167" type="verses" book="sn">
_Mahāyaññaṃ yajitvāna,_  
_puna pāvisi assamaṃ;_  
_Tasmiṃ paṭipaviṭṭhamhi,_  
_añño āgañchi brāhmaṇo. (4)_  

</Paragraph>

</Division>
<Division number="987" book="sn">

<Paragraph number="1168" type="verses" book="sn">
_Ugghaṭṭapādo tasito,_  
_paṅkadanto rajassiro;_  
_So ca naṃ upasaṅkamma,_  
_satāni pañca yācati. (5)_  

</Paragraph>

</Division>
<Division number="988" book="sn">

<Paragraph number="1169" type="verses" book="sn">
_Tamenaṃ bāvarī disvā,_  
_āsanena nimantayi;_  
_Sukhañca kusalaṃ pucchi,_  
_idaṃ vacanamabravi. (6)_  

</Paragraph>

</Division>
<Division number="989" book="sn">

<Paragraph number="1170" type="verses" book="sn">
_“Yaṃ kho mama deyyadhammaṃ,_  
_Sabbaṃ visajjitaṃ mayā;_  
_Anujānāhi me brahme,_  
_Natthi pañcasatāni me”. (7)_  

</Paragraph>

</Division>
<Division number="990" book="sn">

<Paragraph number="1171" type="verses" book="sn">
_“Sace me yācamānassa,_  
_bhavaṃ nānupadassati;_  
_Sattame divase tuyhaṃ,_  
_muddhā phalatu sattadhā”. (8)_  

</Paragraph>

</Division>
<Division number="991" book="sn">

<Paragraph number="1172" type="verses" book="sn">
_Abhisaṅkharitvā kuhako,_  
_bheravaṃ so akittayi;_  
_Tassa taṃ vacanaṃ sutvā,_  
_bāvarī dukkhito ahu. (9)_  

</Paragraph>

</Division>
<Division number="992" book="sn">

<Paragraph number="1173" type="verses" book="sn">
_Ussussati anāhāro,_  
_Sokasallasamappito;_  
_Athopi evaṃ cittassa,_  
_Jhāne na ramatī mano. (10)_  

</Paragraph>

</Division>
<Division number="993" book="sn">

<Paragraph number="1174" type="verses" book="sn">
_Utrastaṃ dukkhitaṃ disvā,_  
_Devatā atthakāminī;_  
_Bāvariṃ upasaṅkamma,_  
_Idaṃ vacanamabravi. (11)_  

</Paragraph>

</Division>
<Division number="994" book="sn">

<Paragraph number="1175" type="verses" book="sn">
_“Na so muddhaṃ pajānāti,_  
_Kuhako so dhanatthiko;_  
_Muddhani muddhapāte vā,_  
_Ñāṇaṃ tassa na vijjati”. (12)_  

</Paragraph>

</Division>
<Division number="995" book="sn">

<Paragraph number="1176" type="verses" book="sn">
_“Bhotī carahi jānāsi,_  
_Taṃ me akkhāhi pucchitā;_  
_Muddhaṃ muddhādhipātañca,_  
_Taṃ suṇoma vaco tava”. (13)_  

</Paragraph>

</Division>
<Division number="996" book="sn">

<Paragraph number="1177" type="verses" book="sn">
_“Ahampetaṃ na jānāmi,_  
_Ñāṇamettha na vijjati;_  
_Muddhani muddhādhipāte ca,_  
_Jinānaṃ hettha dassanaṃ”. (14)_  

</Paragraph>

</Division>
<Division number="997" book="sn">

<Paragraph number="1178" type="verses" book="sn">
_“Atha ko carahi jānāti,_  
_Asmiṃ pathavimaṇḍale;_  
_Muddhaṃ muddhādhipātañca,_  
_Taṃ me akkhāhi devate”. (15)_  

</Paragraph>

</Division>
<Division number="998" book="sn">

<Paragraph number="1179" type="verses" book="sn">
_“Purā kapilavatthumhā,_  
_Nikkhanto lokanāyako;_  
_Apacco okkākarājassa,_  
_Sakyaputto pabhaṅkaro. (16)_  

</Paragraph>

</Division>
<Division number="999" book="sn">

<Paragraph number="1180" type="verses" book="sn">
_So hi brāhmaṇa sambuddho,_  
_Sabbadhammāna pāragū;_  
_Sabbābhiññābalappatto,_  
_Sabbadhammesu cakkhumā;_  
_Sabbakammakkhayaṃ patto,_  
_Vimutto upadhikkhaye. (17)_  

</Paragraph>

</Division>
<Division number="1000" book="sn">

<Paragraph number="1181" type="verses" book="sn">
_Buddho so bhagavā loke,_  
_Dhammaṃ deseti cakkhumā;_  
_Taṃ tvaṃ gantvāna pucchassu,_  
_So te taṃ byākarissati”. (18)_  

</Paragraph>

</Division>
<Division number="1001" book="sn">

<Paragraph number="1182" type="verses" book="sn">
_Sambuddhoti vaco sutvā,_  
_Udaggo bāvarī ahu;_  
_Sokassa tanuko āsi,_  
_Pītiñca vipulaṃ labhi. (19)_  

</Paragraph>

</Division>
<Division number="1002" book="sn">

<Paragraph number="1183" type="verses" book="sn">
_So bāvarī attamano udaggo,_  
_Taṃ devataṃ pucchati vedajāto;_  
_“Katamamhi gāme nigamamhi vā pana,_  
_Katamamhi vā janapade lokanātho;_  
_Yattha gantvāna passemu,_  
_Sambuddhaṃ dvipaduttamaṃ”. (20)_  

</Paragraph>

</Division>
<Division number="1003" book="sn">

<Paragraph number="1184" type="verses" book="sn">
_“Sāvatthiyaṃ kosalamandire jino,_  
_Pahūtapañño varabhūrimedhaso;_  
_So sakyaputto vidhuro anāsavo,_  
_Muddhādhipātassa vidū narāsabho”. (21)_  

</Paragraph>

</Division>
<Division number="1004" book="sn">

<Paragraph number="1185" type="verses" book="sn">
_Tato āmantayī sisse,_  
_Brāhmaṇe mantapārage;_  
_“Etha māṇavā akkhissaṃ,_  
_Suṇātha vacanaṃ mama. (22)_  

</Paragraph>

</Division>
<Division number="1005" book="sn">

<Paragraph number="1186" type="verses" book="sn">
_Yasseso dullabho loke,_  
_Pātubhāvo abhiṇhaso;_  
_Svājja lokamhi uppanno,_  
_Sambuddho iti vissuto;_  
_Khippaṃ gantvāna sāvatthiṃ,_  
_Passavho dvipaduttamaṃ”. (23)_  

</Paragraph>

</Division>
<Division number="1006" book="sn">

<Paragraph number="1187" type="verses" book="sn">
_“Kathaṃ carahi jānemu,_  
_Disvā buddhoti brāhmaṇa;_  
_Ajānataṃ no pabrūhi,_  
_Yathā jānemu taṃ mayaṃ”. (24)_  

</Paragraph>

</Division>
<Division number="1007" book="sn">

<Paragraph number="1188" type="verses" book="sn">
_“Āgatāni hi mantesu,_  
_Mahāpurisalakkhaṇā;_  
_Dvattiṃsāni ca byākkhātā,_  
_Samattā anupubbaso. (25)_  

</Paragraph>

</Division>
<Division number="1008" book="sn">

<Paragraph number="1189" type="verses" book="sn">
_Yassete honti gattesu,_  
_Mahāpurisalakkhaṇā;_  
_Dveyeva tassa gatiyo,_  
_Tatiyā hi na vijjati. (26)_  

</Paragraph>

</Division>
<Division number="1009" book="sn">

<Paragraph number="1190" type="verses" book="sn">
_Sace agāraṃ āvasati,_  
_Vijeyya pathaviṃ imaṃ;_  
_Adaṇḍena asatthena,_  
_Dhammena manusāsati. (27)_  

</Paragraph>

</Division>
<Division number="1010" book="sn">

<Paragraph number="1191" type="verses" book="sn">
_Sace ca so pabbajati,_  
_Agārā anagāriyaṃ;_  
_Vivaṭṭacchado sambuddho,_  
_Arahā bhavati anuttaro. (28)_  

</Paragraph>

</Division>
<Division number="1011" book="sn">

<Paragraph number="1192" type="verses" book="sn">
_Jātiṃ gottañca lakkhaṇaṃ,_  
_Mante sisse punāpare;_  
_Muddhaṃ muddhādhipātañca,_  
_Manasāyeva pucchatha. (29)_  

</Paragraph>

</Division>
<Division number="1012" book="sn">

<Paragraph number="1193" type="verses" book="sn">
_Anāvaraṇadassāvī,_  
_Yadi buddho bhavissati;_  
_Manasā pucchite pañhe,_  
_Vācāya vissajessati”. (30)_  

</Paragraph>

</Division>
<Division number="1013" book="sn">

<Paragraph number="1194" type="verses" book="sn">
_Bāvarissa vaco sutvā,_  
_Sissā soḷasa brāhmaṇā;_  
_Ajito tissametteyyo,_  
_Puṇṇako atha mettagū. (31)_  

</Paragraph>

</Division>
<Division number="1014" book="sn">

<Paragraph number="1195" type="verses" book="sn">
_Dhotako upasīvo ca,_  
_Nando ca atha hemako;_  
_Todeyya-kappā dubhayo,_  
_Jatukaṇṇī ca paṇḍito. (32)_  

</Paragraph>

</Division>
<Division number="1015" book="sn">

<Paragraph number="1196" type="verses" book="sn">
_Bhadrāvudho udayo ca,_  
_Posālo cāpi brāhmaṇo;_  
_Mogharājā ca medhāvī,_  
_Piṅgiyo ca mahāisi. (33)_  

</Paragraph>

</Division>
<Division number="1016" book="sn">

<Paragraph number="1197" type="verses" book="sn">
_Paccekagaṇino sabbe,_  
_Sabbalokassa vissutā;_  
_Jhāyī jhānaratā dhīrā,_  
_Pubbavāsanavāsitā. (34)_  

</Paragraph>

</Division>
<Division number="1017" book="sn">

<Paragraph number="1198" type="verses" book="sn">
_Bāvariṃ abhivādetvā,_  
_Katvā ca naṃ padakkhiṇaṃ;_  
_Jaṭājinadharā sabbe,_  
_Pakkāmuṃ uttarāmukhā. (35)_  

</Paragraph>

</Division>
<Division number="1018" book="sn">

<Paragraph number="1199" type="verses" book="sn">
_Aḷakassa patiṭṭhānaṃ,_  
_Purimāhissatiṃ tadā;_  
_Ujjeniñcāpi gonaddhaṃ,_  
_Vedisaṃ vanasavhayaṃ. (36)_  

</Paragraph>

</Division>
<Division number="1019" book="sn">

<Paragraph number="1200" type="verses" book="sn">
_Kosambiñcāpi sāketaṃ,_  
_Sāvatthiñca puruttamaṃ;_  
_Setabyaṃ kapilavatthuṃ,_  
_Kusinārañca mandiraṃ. (37)_  

</Paragraph>

</Division>
<Division number="1020" book="sn">

<Paragraph number="1201" type="verses" book="sn">
_Pāvañca bhoganagaraṃ,_  
_Vesāliṃ māgadhaṃ puraṃ;_  
_Pāsāṇakaṃ cetiyañca,_  
_Ramaṇīyaṃ manoramaṃ. (38)_  

</Paragraph>

</Division>
<Division number="1021" book="sn">

<Paragraph number="1202" type="verses" book="sn">
_Tasitovudakaṃ sītaṃ,_  
_Mahālābhaṃva vāṇijo;_  
_Chāyaṃ ghammābhitattova,_  
_Turitā pabbatamāruhuṃ. (39)_  

</Paragraph>

</Division>
<Division number="1022" book="sn">

<Paragraph number="1203" type="verses" book="sn">
_Bhagavā tamhi samaye,_  
_Bhikkhusaṃghapurakkhato;_  
_Bhikkhūnaṃ dhammaṃ deseti,_  
_Sīhova nadatī vane. (40)_  

</Paragraph>

</Division>
<Division number="1023" book="sn">

<Paragraph number="1204" type="verses" book="sn">
_Ajito addasa buddhaṃ,_  
_Sataraṃsiṃva bhāṇumaṃ;_  
_Candaṃ yathā pannarase,_  
_Pāripūriṃ upāgataṃ. (41)_  

</Paragraph>

</Division>
<Division number="1024" book="sn">

<Paragraph number="1205" type="verses" book="sn">
_Athassa gatte disvāna,_  
_Paripūrañca byañjanaṃ;_  
_Ekamantaṃ ṭhito haṭṭho,_  
_Manopañhe apucchatha. (42)_  

</Paragraph>

</Division>
<Division number="1025" book="sn">

<Paragraph number="1206" type="verses" book="sn">
_“Ādissa jammanaṃ brūhi,_  
_Gottaṃ brūhi salakkhaṇaṃ;_  
_Mantesu pāramiṃ brūhi,_  
_Kati vāceti brāhmaṇo”. (43)_  

</Paragraph>

</Division>
<Division number="1026" book="sn">

<Paragraph number="1207" type="verses" book="sn">
_“Vīsaṃ vassasataṃ āyu,_  
_So ca gottena bāvarī;_  
_Tīṇissa lakkhaṇā gatte,_  
_Tiṇṇaṃ vedāna pāragū. (44)_  

</Paragraph>

</Division>
<Division number="1027" book="sn">

<Paragraph number="1208" type="verses" book="sn">
_Lakkhaṇe itihāse ca,_  
_Sanighaṇḍusakeṭubhe;_  
_Pañcasatāni vāceti,_  
_Sadhamme pāramiṃ gato”. (45)_  

</Paragraph>

</Division>
<Division number="1028" book="sn">

<Paragraph number="1209" type="verses" book="sn">
_“Lakkhaṇānaṃ pavicayaṃ,_  
_Bāvarissa naruttama;_  
_Kaṅkhacchida pakāsehi,_  
_Mā no kaṅkhāyitaṃ ahu”. (46)_  

</Paragraph>

</Division>
<Division number="1029" book="sn">

<Paragraph number="1210" type="verses" book="sn">
_“Mukhaṃ jivhāya chādeti,_  
_Uṇṇassa bhamukantare;_  
_Kosohitaṃ vatthaguyhaṃ,_  
_Evaṃ jānāhi māṇava”. (47)_  

</Paragraph>

</Division>
<Division number="1030" book="sn">

<Paragraph number="1211" type="verses" book="sn">
_Pucchañhi kiñci asuṇanto,_  
_Sutvā pañhe viyākate;_  
_Vicinteti jano sabbo,_  
_Vedajāto katañjalī. (48)_  

</Paragraph>

</Division>
<Division number="1031" book="sn">

<Paragraph number="1212" type="verses" book="sn">
_“Ko nu devo vā brahmā vā,_  
_Indo vāpi sujampati;_  
_Manasā pucchite pañhe,_  
_Kametaṃ paṭibhāsati”. (49)_  

</Paragraph>

</Division>
<Division number="1032" book="sn">

<Paragraph number="1213" type="verses" book="sn">
_“Muddhaṃ muddhādhipātañca,_  
_Bāvarī paripucchati;_  
_Taṃ byākarohi bhagavā,_  
_Kaṅkhaṃ vinaya no ise”. (50)_  

</Paragraph>

</Division>
<Division number="1033" book="sn">

<Paragraph number="1214" type="verses" book="sn">
_“Avijjā muddhāti jānāhi,_  
_Vijjā muddhādhipātinī;_  
_Saddhāsatisamādhīhi,_  
_Chandaviriyena saṃyutā”. (51)_  

</Paragraph>

</Division>
<Division number="1034" book="sn">

<Paragraph number="1215" type="verses" book="sn">
_Tato vedena mahatā,_  
_Santhambhitvāna māṇavo;_  
_Ekaṃsaṃ ajinaṃ katvā,_  
_Pādesu sirasā pati. (52)_  

</Paragraph>

</Division>
<Division number="1035" book="sn">

<Paragraph number="1216" type="verses" book="sn">
_“Bāvarī brāhmaṇo bhoto,_  
_Saha sissehi mārisa;_  
_Udaggacitto sumano,_  
_Pāde vandati cakkhuma”. (53)_  

</Paragraph>

</Division>
<Division number="1036" book="sn">

<Paragraph number="1217" type="verses" book="sn">
_“Sukhito bāvarī hotu,_  
_Saha sissehi brāhmaṇo;_  
_Tvañcāpi sukhito hohi,_  
_Ciraṃ jīvāhi māṇava. (54)_  

</Paragraph>

</Division>
<Division number="1037" book="sn">

<Paragraph number="1218" type="verses" book="sn">
_Bāvarissa ca tuyhaṃ vā,_  
_Sabbesaṃ sabbasaṃsayaṃ;_  
_Katāvakāsā pucchavho,_  
_Yaṃ kiñci manasicchatha”. (55)_  

</Paragraph>

</Division>
<Division number="1038" book="sn">

<Paragraph number="1219" type="verses" book="sn">
_Sambuddhena katokāso,_  
_Nisīditvāna pañjalī;_  
_Ajito paṭhamaṃ pañhaṃ,_  
_Tattha pucchi tathāgataṃ. (56)_  


</Paragraph>
</Division>
