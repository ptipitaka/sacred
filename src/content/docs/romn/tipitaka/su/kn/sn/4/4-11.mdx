<Division number="869" book="sn">

<Paragraph number="1040" type="verses" book="sn">
_“Kutopahūtā kalahā vivādā,_  
_Paridevasokā sahamaccharā ca;_  
_Mānātimānā sahapesuṇā ca,_  
_Kutopahūtā te tadiṅgha brūhi”. (1)_  

</Paragraph>

</Division>
<Division number="870" book="sn">

<Paragraph number="1041" type="verses" book="sn">
_“Piyappahūtā kalahā vivādā,_  
_Paridevasokā sahamaccharā ca;_  
_Mānātimānā sahapesuṇā ca,_  
_Maccherayuttā kalahā vivādā;_  
_Vivādajātesu ca pesuṇāni”. (2)_  

</Paragraph>

</Division>
<Division number="871" book="sn">

<Paragraph number="1042" type="verses" book="sn">
_“Piyā su lokasmiṃ kutonidānā,_  
_Ye cāpi lobhā vicaranti loke;_  
_Āsā ca niṭṭhā ca kutonidānā,_  
_Ye samparāyāya narassa honti”. (3)_  

</Paragraph>

</Division>
<Division number="872" book="sn">

<Paragraph number="1043" type="verses" book="sn">
_“Chandānidānāni piyāni loke,_  
_Ye cāpi lobhā vicaranti loke;_  
_Āsā ca niṭṭhā ca itonidānā,_  
_Ye samparāyāya narassa honti”. (4)_  

</Paragraph>

</Division>
<Division number="873" book="sn">

<Paragraph number="1044" type="verses" book="sn">
_“Chando nu lokasmiṃ kutonidāno,_  
_Vinicchayā cāpi kutopahūtā;_  
_Kodho mosavajjañca kathaṃkathā ca,_  
_Ye vāpi dhammā samaṇena vuttā”. (5)_  

</Paragraph>

</Division>
<Division number="874" book="sn">

<Paragraph number="1045" type="verses" book="sn">
_“Sātaṃ asātanti yamāhu loke,_  
_Tamūpanissāya pahoti chando;_  
_Rūpesu disvā vibhavaṃ bhavañca,_  
_Vinicchayaṃ kubbati jantu loke. (6)_  

</Paragraph>

</Division>
<Division number="875" book="sn">

<Paragraph number="1046" type="verses" book="sn">
_Kodho mosavajjañca kathaṃkathā ca,_  
_Etepi dhammā dvayameva sante;_  
_Kathaṃkathī ñāṇapathāya sikkhe,_  
_Ñatvā pavuttā samaṇena dhammā”. (7)_  

</Paragraph>

</Division>
<Division number="876" book="sn">

<Paragraph number="1047" type="verses" book="sn">
_“Sātaṃ asātañca kutonidānā,_  
_Kismiṃ asante na bhavanti hete;_  
_Vibhavaṃ bhavañcāpi yametamatthaṃ,_  
_Etaṃ me pabrūhi yatonidānaṃ”. (8)_  

</Paragraph>

</Division>
<Division number="877" book="sn">

<Paragraph number="1048" type="verses" book="sn">
_“Phassanidānaṃ sātaṃ asātaṃ,_  
_Phasse asante na bhavanti hete;_  
_Vibhavaṃ bhavañcāpi yametamatthaṃ,_  
_Etaṃ te pabrūmi itonidānaṃ”. (9)_  

</Paragraph>

</Division>
<Division number="878" book="sn">

<Paragraph number="1049" type="verses" book="sn">
_“Phasso nu lokasmi kutonidāno,_  
_Pariggahā cāpi kutopahūtā;_  
_Kismiṃ asante na mamattamatthi,_  
_Kismiṃ vibhūte na phusanti phassā”. (10)_  

</Paragraph>

</Division>
<Division number="879" book="sn">

<Paragraph number="1050" type="verses" book="sn">
_“Nāmañca rūpañca paṭicca phasso,_  
_Icchānidānāni pariggahāni;_  
_Icchāyasantyā na mamattamatthi,_  
_Rūpe vibhūte na phusanti phassā”. (11)_  

</Paragraph>

</Division>
<Division number="880" book="sn">

<Paragraph number="1051" type="verses" book="sn">
_“Kathaṃ sametassa vibhoti rūpaṃ,_  
_Sukhaṃ dukhañcāpi kathaṃ vibhoti;_  
_Etaṃ me pabrūhi yathā vibhoti,_  
_Taṃ jāniyāmāti me mano ahu”. (12)_  

</Paragraph>

</Division>
<Division number="881" book="sn">

<Paragraph number="1052" type="verses" book="sn">
_“Na saññasaññī na visaññasaññī,_  
_Nopi asaññī na vibhūtasaññī;_  
_Evaṃ sametassa vibhoti rūpaṃ,_  
_Saññānidānā hi papañcasaṅkhā”. (13)_  

</Paragraph>

</Division>
<Division number="882" book="sn">

<Paragraph number="1053" type="verses" book="sn">
_“Yaṃ taṃ apucchimha akittayī no,_  
_Aññaṃ taṃ pucchāma tadiṅgha brūhi;_  
_Ettāvataggaṃ nu vadanti heke,_  
_Yakkhassa suddhiṃ idha paṇḍitāse;_  
_Udāhu aññampi vadanti etto”. (14)_  

</Paragraph>

</Division>
<Division number="883" book="sn">

<Paragraph number="1054" type="verses" book="sn">
_“Ettāvataggampi vadanti heke,_  
_Yakkhassa suddhiṃ idha paṇḍitāse;_  
_Tesaṃ paneke samayaṃ vadanti,_  
_Anupādisese kusalā vadānā. (15)_  

</Paragraph>

</Division>
<Division number="884" book="sn">

<Paragraph number="1055" type="verses" book="sn">
_Ete ca ñatvā upanissitāti,_  
_Ñatvā munī nissaye so vimaṃsī;_  
_Ñatvā vimutto na vivādameti,_  
_Bhavābhavāya na sameti dhīro”ti. (16)_  


</Paragraph>
</Division>

<Paragraph number="1056" type="center" book="sn">
Kalahavivādasuttaṃ ekādasamaṃ.
</Paragraph>
