<Paragraph number="391" type="verses" book="sn">
_“Kacci abhiṇhasaṃvāsā,_  
_nāvajānāsi paṇḍitaṃ;_  
_Ukkādhāro manussānaṃ,_  
_kacci apacito tayā”. (1)_  

</Paragraph>

</Division>
<Division number="339" book="sn" bookNo="18" page="328">

<Paragraph number="392" type="verses" book="sn">
_“Nāhaṃ abhiṇhasaṃvāsā,_  
_avajānāmi paṇḍitaṃ;_  
_Ukkādhāro manussānaṃ,_  
_niccaṃ apacito mayā”. (2)_  

</Paragraph>

</Division>
<Division number="340" book="sn" bookNo="18" page="329">

<Paragraph number="393" type="verses" book="sn">
_“Pañca kāmaguṇe hitvā,_  
_piyarūpe manorame;_  
_Saddhāya gharā nikkhamma,_  
_dukkhassantakaro bhava. (3)_  

</Paragraph>

</Division>
<Division number="341" book="sn" bookNo="18" page="329">

<Paragraph number="394" type="verses" book="sn">
_Mitte bhajassu kalyāṇe,_  
_pantañca sayanāsanaṃ;_  
_Vivittaṃ appanigghosaṃ,_  
_mattaññū hohi bhojane. (4)_  

</Paragraph>

</Division>
<Division number="342" book="sn" bookNo="18" page="329">

<Paragraph number="395" type="verses" book="sn">
_Cīvare piṇḍapāte ca,_  
_paccaye sayanāsane;_  
_Etesu taṇhaṃ mākāsi,_  
_mā lokaṃ punarāgami. (5)_  

</Paragraph>

</Division>
<Division number="343" book="sn" bookNo="18" page="329">

<Paragraph number="396" type="verses" book="sn">
_Saṃvuto pātimokkhasmiṃ,_  
_indriyesu ca pañcasu;_  
_Sati kāyagatā tyatthu,_  
_nibbidābahulo bhava. (6)_  

</Paragraph>

</Division>
<Division number="344" book="sn" bookNo="18" page="329">

<Paragraph number="397" type="verses" book="sn">
_Nimittaṃ parivajjehi,_  
_subhaṃ rāgūpasañhitaṃ;_  
_Asubhāya cittaṃ bhāvehi,_  
_ekaggaṃ susamāhitaṃ. (7)_  

</Paragraph>

</Division>
<Division number="345" book="sn" bookNo="18" page="329">

<Paragraph number="398" type="verses" book="sn">
_Animittañca bhāvehi,_  
_Mānānusayamujjaha;_  
_Tato mānābhisamayā,_  
_Upasanto carissatī”ti. (8)_  

</Paragraph>

<Paragraph number="399" book="sn">
Itthaṃ sudaṃ bhagavā āyasmantaṃ rāhulaṃ imāhi gāthāhi abhiṇhaṃ ovadatīti.

</Paragraph>
</Division>

<Paragraph number="400" type="center" book="sn">
Rāhulasuttaṃ ekādasamaṃ.
</Paragraph>
