<Paragraph number="176" type="verses" book="sn">
_“Ajja pannaraso uposatho, __(iti sātāgiro yakkho)___  
_Dibbā ratti upaṭṭhitā;_  
_Anomanāmaṃ satthāraṃ,_  
_Handa passāma gotamaṃ”. (1)_  

</Paragraph>

</Division>
<Division number="154" book="sn" bookNo="18" page="302">

<Paragraph number="177" type="verses" book="sn">
_“Kacci mano supaṇihito, __(iti hemavato yakkho)___  
_Sabbabhūtesu tādino;_  
_Kacci iṭṭhe aniṭṭhe ca,_  
_Saṅkappassa vasīkatā”. (2)_  

</Paragraph>

</Division>
<Division number="155" book="sn" bookNo="18" page="302">

<Paragraph number="178" type="verses" book="sn">
_“Mano cassa supaṇihito, __(iti sātāgiro yakkho)___  
_Sabbabhūtesu tādino;_  
_Atho iṭṭhe aniṭṭhe ca,_  
_Saṅkappassa vasīkatā”. (3)_  

</Paragraph>

</Division>
<Division number="156" book="sn" bookNo="18" page="302">

<Paragraph number="179" type="verses" book="sn">
_“Kacci adinnaṃ nādiyati, __(iti hemavato yakkho)___  
_Kacci pāṇesu saññato;_  
_Kacci ārā pamādamhā,_  
_Kacci jhānaṃ na riñcati”. (4)_  

</Paragraph>

</Division>
<Division number="157" book="sn" bookNo="18" page="303">

<Paragraph number="180" type="verses" book="sn">
_“Na so adinnaṃ ādiyati, __(iti sātāgiro yakkho)___  
_Atho pāṇesu saññato;_  
_Atho ārā pamādamhā,_  
_Buddho jhānaṃ na riñcati”. (5)_  

</Paragraph>

</Division>
<Division number="158" book="sn" bookNo="18" page="303">

<Paragraph number="181" type="verses" book="sn">
_“Kacci musā na bhaṇati, __(iti hemavato yakkho)___  
_Kacci na khīṇabyappatho;_  
_Kacci vebhūtiyaṃ nāha,_  
_Kacci samphaṃ na bhāsati”. (6)_  

</Paragraph>

</Division>
<Division number="159" book="sn" bookNo="18" page="303">

<Paragraph number="182" type="verses" book="sn">
_“Musā ca so na bhaṇati, __(iti sātāgiro yakkho)___  
_Atho na khīṇabyappatho;_  
_Atho vebhūtiyaṃ nāha,_  
_Mantā atthañca bhāsati”. (7)_  

</Paragraph>

</Division>
<Division number="160" book="sn" bookNo="18" page="303">

<Paragraph number="183" type="verses" book="sn">
_“Kacci na rajjati kāmesu, __(iti hemavato yakkho)___  
_Kacci cittaṃ anāvilaṃ;_  
_Kacci mohaṃ atikkanto,_  
_Kacci dhammesu cakkhumā”. (8)_  

</Paragraph>

</Division>
<Division number="161" book="sn" bookNo="18" page="303">

<Paragraph number="184" type="verses" book="sn">
_“Na so rajjati kāmesu, __(iti sātāgiro yakkho)___  
_Atho cittaṃ anāvilaṃ;_  
_Sabbamohaṃ atikkanto,_  
_Buddho dhammesu cakkhumā”. (9)_  

</Paragraph>

</Division>
<Division number="162" book="sn" bookNo="18" page="303">

<Paragraph number="185" type="verses" book="sn">
_“Kacci vijjāya sampanno, __(iti hemavato yakkho)___  
_Kacci saṃsuddhacāraṇo;_  
_Kaccissa āsavā khīṇā,_  
_Kacci natthi punabbhavo”. (10)_  

</Paragraph>

</Division>
<Division number="163" book="sn" bookNo="18" page="303">

<Paragraph number="186" type="verses" book="sn">
_“Vijjāya ceva sampanno, __(iti sātāgiro yakkho)___  
_Atho saṃsuddhacāraṇo;_  
_Sabbassa āsavā khīṇā,_  
_Natthi tassa punabbhavo”. (11)_  

</Paragraph>

</Division>
<Division number="164" book="sn" bookNo="18" page="304">

<Paragraph number="187" type="verses" book="sn">
_“Sampannaṃ munino cittaṃ,_  
_Kammunā byappathena ca;_  
_Vijjācaraṇasampannaṃ,_  
_Dhammato naṃ pasaṃsati”. (12)_  

</Paragraph>

</Division>
<Division number="165" book="sn" bookNo="18" page="304">

<Paragraph number="188" type="verses" book="sn">
_“Sampannaṃ munino cittaṃ,_  
_Kammunā byappathena ca;_  
_Vijjācaraṇasampannaṃ,_  
_Dhammato anumodasi. (13)_  

</Paragraph>

</Division>
<Division number="166" book="sn" bookNo="18" page="304">

<Paragraph number="189" type="verses" book="sn">
_Sampannaṃ munino cittaṃ,_  
_Kammunā byappathena ca;_  
_Vijjācaraṇasampannaṃ,_  
_Handa passāma gotamaṃ”. (14)_  

</Paragraph>

</Division>
<Division number="167" book="sn" bookNo="18" page="304">

<Paragraph number="190" type="verses" book="sn">
_“Eṇijaṅghaṃ kisaṃ vīraṃ,_  
_Appāhāraṃ alolupaṃ;_  
_Muniṃ vanasmiṃ jhāyantaṃ,_  
_Ehi passāma gotamaṃ. (15)_  

</Paragraph>

</Division>
<Division number="168" book="sn" bookNo="18" page="304">

<Paragraph number="191" type="verses" book="sn">
_Sīhaṃvekacaraṃ nāgaṃ,_  
_Kāmesu anapekkhinaṃ;_  
_Upasaṅkamma pucchāma,_  
_Maccupāsappamocanaṃ. (16)_  

</Paragraph>

</Division>
<Division number="169" book="sn" bookNo="18" page="304">

<Paragraph number="192" type="verses" book="sn">
_Akkhātāraṃ pavattāraṃ,_  
_Sabbadhammāna pāraguṃ;_  
_Buddhaṃ verabhayātītaṃ,_  
_Mayaṃ pucchāma gotamaṃ”. (17)_  

</Paragraph>

</Division>
<Division number="170" book="sn" bookNo="18" page="304">

<Paragraph number="193" type="verses" book="sn">
_“Kismiṃ loko samuppanno, __(iti hemavato yakkho)___  
_Kismiṃ kubbati santhavaṃ;_  
_Kissa loko upādāya,_  
_Kismiṃ loko vihaññati”. (18)_  

</Paragraph>

</Division>
<Division number="171" book="sn" bookNo="18" page="304">

<Paragraph number="194" type="verses" book="sn">
_“Chasu loko samuppanno, __(hemavatāti bhagavā)___  
_Chasu kubbati santhavaṃ;_  
_Channameva upādāya,_  
_Chasu loko vihaññati”. (19)_  

</Paragraph>

</Division>
<Division number="172" book="sn" bookNo="18" page="304">

<Paragraph number="195" type="verses" book="sn">
_“Katamaṃ taṃ upādānaṃ,_  
_Yattha loko vihaññati;_  
_Niyyānaṃ pucchito brūhi,_  
_Kathaṃ dukkhā pamuccati”. (20)_  

</Paragraph>

</Division>
<Division number="173" book="sn" bookNo="18" page="304">

<Paragraph number="196" type="verses" book="sn">
_“Pañca kāmaguṇā loke,_  
_Manochaṭṭhā paveditā;_  
_Ettha chandaṃ virājetvā,_  
_Evaṃ dukkhā pamuccati. (21)_  

</Paragraph>

</Division>
<Division number="174" book="sn" bookNo="18" page="304">

<Paragraph number="197" type="verses" book="sn">
_Etaṃ lokassa niyyānaṃ,_  
_Akkhātaṃ vo yathātathaṃ;_  
_Etaṃ vo ahamakkhāmi,_  
_Evaṃ dukkhā pamuccati”. (22)_  

</Paragraph>

</Division>
<Division number="175" book="sn" bookNo="18" page="305">

<Paragraph number="198" type="verses" book="sn">
_“Ko sūdha tarati oghaṃ,_  
_Kodha tarati aṇṇavaṃ;_  
_Appatiṭṭhe anālambe,_  
_Ko gambhīre na sīdati”. (23)_  

</Paragraph>

</Division>
<Division number="176" book="sn" bookNo="18" page="305">

<Paragraph number="199" type="verses" book="sn">
_“Sabbadā sīlasampanno,_  
_Paññavā susamāhito;_  
_Ajjhattacintī satimā,_  
_Oghaṃ tarati duttaraṃ. (24)_  

</Paragraph>

</Division>
<Division number="177" book="sn" bookNo="18" page="305">

<Paragraph number="200" type="verses" book="sn">
_Virato kāmasaññāya,_  
_Sabbasaṃyojanātigo;_  
_Nandībhavaparikkhīṇo,_  
_So gambhīre na sīdati”. (25)_  

</Paragraph>

</Division>
<Division number="178" book="sn" bookNo="18" page="305">

<Paragraph number="201" type="verses" book="sn">
_“Gambhīrapaññaṃ nipuṇatthadassiṃ,_  
_Akiñcanaṃ kāmabhave asattaṃ;_  
_Taṃ passatha sabbadhi vippamuttaṃ,_  
_Dibbe pathe kamamānaṃ mahesiṃ. (26)_  

</Paragraph>

</Division>
<Division number="179" book="sn" bookNo="18" page="305">

<Paragraph number="202" type="verses" book="sn">
_Anomanāmaṃ nipuṇatthadassiṃ,_  
_Paññādadaṃ kāmālaye asattaṃ;_  
_Taṃ passatha sabbaviduṃ sumedhaṃ,_  
_Ariye pathe kamamānaṃ mahesiṃ. (27)_  

</Paragraph>

</Division>
<Division number="180" book="sn" bookNo="18" page="305">

<Paragraph number="203" type="verses" book="sn">
_Sudiṭṭhaṃ vata no ajja,_  
_Suppabhātaṃ suhuṭṭhitaṃ;_  
_Yaṃ addasāma sambuddhaṃ,_  
_Oghatiṇṇamanāsavaṃ. (28)_  

</Paragraph>

</Division>
<Division number="181" book="sn" bookNo="18" page="305">

<Paragraph number="204" type="verses" book="sn">
_Ime dasasatā yakkhā,_  
_Iddhimanto yasassino;_  
_Sabbe taṃ saraṇaṃ yanti,_  
_Tvaṃ no satthā anuttaro. (29)_  

</Paragraph>

</Division>
<Division number="182" book="sn" bookNo="18" page="305">

<Paragraph number="205" type="verses" book="sn">
_Te mayaṃ vicarissāma,_  
_Gāmā gāmaṃ nagā nagaṃ;_  
_Namassamānā sambuddhaṃ,_  
_Dhammassa ca sudhammatan”ti. (30)_  


</Paragraph>
</Division>
