<Paragraph number="977" type="verses" book="sn">
_Appaṃ vata jīvitaṃ idaṃ,_  
_Oraṃ vassasatāpi miyyati;_  
_Yo cepi aticca jīvati,_  
_Atha kho so jarasāpi miyyati. (1)_  

</Paragraph>

</Division>
<Division number="812" book="sn" bookNo="18" page="405">

<Paragraph number="978" type="verses" book="sn">
_Socanti janā mamāyite,_  
_Na hi santi niccā pariggahā;_  
_Vinābhāvasantamevidaṃ,_  
_Iti disvā nāgāramāvase. (2)_  

</Paragraph>

</Division>
<Division number="813" book="sn" bookNo="18" page="405">

<Paragraph number="979" type="verses" book="sn">
_Maraṇenapi taṃ pahīyati,_  
_Yaṃ puriso mamidanti maññati;_  
_Etampi viditvā paṇḍito,_  
_Na mamattāya nametha māmako. (3)_  

</Paragraph>

</Division>
<Division number="814" book="sn" bookNo="18" page="405">

<Paragraph number="980" type="verses" book="sn">
_Supinena yathāpi saṅgataṃ,_  
_Paṭibuddho puriso na passati;_  
_Evampi piyāyitaṃ janaṃ,_  
_Petaṃ kālaṅkataṃ na passati. (4)_  

</Paragraph>

</Division>
<Division number="815" book="sn" bookNo="18" page="405">

<Paragraph number="981" type="verses" book="sn">
_Diṭṭhāpi sutāpi te janā,_  
_Yesaṃ nāmamidaṃ pavuccati;_  
_Nāmaṃyevāvasissati,_  
_Akkheyyaṃ petassa jantuno. (5)_  

</Paragraph>

</Division>
<Division number="816" book="sn" bookNo="18" page="405">

<Paragraph number="982" type="verses" book="sn">
_Sokapparidevamaccharaṃ,_  
_Na jahanti giddhā mamāyite;_  
_Tasmā munayo pariggahaṃ,_  
_Hitvā acariṃsu khemadassino. (6)_  

</Paragraph>

</Division>
<Division number="817" book="sn" bookNo="18" page="406">

<Paragraph number="983" type="verses" book="sn">
_Patilīnacarassa bhikkhuno,_  
_Bhajamānassa vivittamāsanaṃ;_  
_Sāmaggiyamāhu tassa taṃ,_  
_Yo attānaṃ bhavane na dassaye. (7)_  

</Paragraph>

</Division>
<Division number="818" book="sn" bookNo="18" page="406">

<Paragraph number="984" type="verses" book="sn">
_Sabbattha munī anissito,_  
_Na piyaṃ kubbati nopi appiyaṃ;_  
_Tasmiṃ paridevamaccharaṃ,_  
_Paṇṇe vāri yathā na limpati. (8)_  

</Paragraph>

</Division>
<Division number="819" book="sn" bookNo="18" page="406">

<Paragraph number="985" type="verses" book="sn">
_Udabindu yathāpi pokkhare,_  
_Padume vāri yathā na limpati;_  
_Evaṃ muni nopalimpati,_  
_Yadidaṃ diṭṭhasutaṃ mutesu vā. (9)_  

</Paragraph>

</Division>
<Division number="820" book="sn" bookNo="18" page="406">

<Paragraph number="986" type="verses" book="sn">
_Dhono na hi tena maññati,_  
_Yadidaṃ diṭṭhasutaṃ mutesu vā;_  
_Nāññena visuddhimicchati,_  
_Na hi so rajjati no virajjatīti. (10)_  


</Paragraph>
</Division>

<Paragraph number="987" type="center" book="sn">
Jarāsuttaṃ chaṭṭhaṃ.
</Paragraph>
