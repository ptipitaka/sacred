<Division number="842" book="sn">

<Paragraph number="1011" type="verses" book="sn">
_“Disvāna taṇhaṃ aratiṃ ragañca,_  
_Nāhosi chando api methunasmiṃ;_  
_Kimevidaṃ muttakarīsapuṇṇaṃ,_  
_Pādāpi naṃ samphusituṃ na icche”. (1)_  

</Paragraph>

</Division>
<Division number="843" book="sn">

<Paragraph number="1012" type="verses" book="sn">
_“Etādisañce ratanaṃ na icchasi,_  
_Nāriṃ narindehi bahūhi patthitaṃ;_  
_Diṭṭhigataṃ sīlavataṃ nu jīvitaṃ,_  
_Bhavūpapattiñca vadesi kīdisaṃ”. (2)_  

</Paragraph>

</Division>
<Division number="844" book="sn">

<Paragraph number="1013" type="verses" book="sn">
_“Idaṃ vadāmīti na tassa hoti, __(māgaṇḍiyāti bhagavā)___  
_Dhammesu niccheyya samuggahītaṃ;_  
_Passañca diṭṭhīsu anuggahāya,_  
_Ajjhattasantiṃ pacinaṃ adassaṃ”. (3)_  

</Paragraph>

</Division>
<Division number="845" book="sn">

<Paragraph number="1014" type="verses" book="sn">
_“Vinicchayā yāni pakappitāni, __(iti māgaṇḍiyo)___  
_Te ve munī brūsi anuggahāya;_  
_Ajjhattasantīti yametamatthaṃ,_  
_Kathaṃ nu dhīrehi paveditaṃ taṃ”. (4)_  

</Paragraph>

</Division>
<Division number="846" book="sn">

<Paragraph number="1015" type="verses" book="sn">
_“Na diṭṭhiyā na sutiyā na ñāṇena, __(māgaṇḍiyāti bhagavā)___  
_Sīlabbatenāpi na suddhimāha;_  
_Adiṭṭhiyā assutiyā añāṇā,_  
_Asīlatā abbatā nopi tena;_  
_Ete ca nissajja anuggahāya,_  
_Santo anissāya bhavaṃ na jappe”. (5)_  

</Paragraph>

</Division>
<Division number="847" book="sn">

<Paragraph number="1016" type="verses" book="sn">
_“No ce kira diṭṭhiyā na sutiyā na ñāṇena, __(iti māgaṇḍiyo)___  
_Sīlabbatenāpi na suddhimāha;_  
_Adiṭṭhiyā assutiyā añāṇā,_  
_Asīlatā abbatā nopi tena;_  
_Maññāmahaṃ momuhameva dhammaṃ,_  
_Diṭṭhiyā eke paccenti suddhiṃ”. (6)_  

</Paragraph>

</Division>
<Division number="848" book="sn">

<Paragraph number="1017" type="verses" book="sn">
_“Diṭṭhañca nissāya anupucchamāno, __(māgaṇḍiyāti bhagavā)___  
_Samuggahītesu pamohamāgā;_  
_Ito ca nāddakkhi aṇumpi saññaṃ,_  
_Tasmā tuvaṃ momuhato dahāsi. (7)_  

</Paragraph>

</Division>
<Division number="849" book="sn">

<Paragraph number="1018" type="verses" book="sn">
_Samo visesī uda vā nihīno,_  
_Yo maññati so vivadetha tena;_  
_Tīsu vidhāsu avikampamāno,_  
_Samo visesīti na tassa hoti. (8)_  

</Paragraph>

</Division>
<Division number="850" book="sn">

<Paragraph number="1019" type="verses" book="sn">
_Saccanti so brāhmaṇo kiṃ vadeyya,_  
_Musāti vā so vivadetha kena;_  
_Yasmiṃ samaṃ visamaṃ vāpi natthi,_  
_Sa kena vādaṃ paṭisaṃyujeyya. (9)_  

</Paragraph>

</Division>
<Division number="851" book="sn">

<Paragraph number="1020" type="verses" book="sn">
_Okaṃ pahāya aniketasārī,_  
_Gāme akubbaṃ muni santhavāni;_  
_Kāmehi ritto apurakkharāno,_  
_Kathaṃ na viggayha janena kayirā. (10)_  

</Paragraph>

</Division>
<Division number="852" book="sn">

<Paragraph number="1021" type="verses" book="sn">
_Yehi vivitto vicareyya loke,_  
_Na tāni uggayha vadeyya nāgo;_  
_Jalambujaṃ kaṇḍakavārijaṃ yathā,_  
_Jalena paṅkena canūpalittaṃ;_  
_Evaṃ munī santivādo agiddho,_  
_Kāme ca loke ca anūpalitto. (11)_  

</Paragraph>

</Division>
<Division number="853" book="sn">

<Paragraph number="1022" type="verses" book="sn">
_Na vedagū diṭṭhiyāyako na mutiyā,_  
_Sa mānameti na hi tammayo so;_  
_Na kammunā nopi sutena neyyo,_  
_Anūpanīto sa nivesanesu. (12)_  

</Paragraph>

</Division>
<Division number="854" book="sn">

<Paragraph number="1023" type="verses" book="sn">
_Saññāvirattassa na santi ganthā,_  
_Paññāvimuttassa na santi mohā;_  
_Saññañca diṭṭhiñca ye aggahesuṃ,_  
_Te ghaṭṭayantā vicaranti loke”ti. (13)_  


</Paragraph>
</Division>

<Paragraph number="1024" type="center" book="sn">
Māgaṇḍiyasuttaṃ navamaṃ.
</Paragraph>
