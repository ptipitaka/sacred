<Paragraph number="294" type="verses" book="sn">
_Hiriṃ tarantaṃ vijigucchamānaṃ,_  
_Tavāhamasmi iti bhāsamānaṃ;_  
_Sayhāni kammāni anādiyantaṃ,_  
_Neso mamanti iti naṃ vijaññā. (1)_  

</Paragraph>

</Division>
<Division number="257" book="sn" bookNo="18" page="318">

<Paragraph number="295" type="verses" book="sn">
_Ananvayaṃ piyaṃ vācaṃ,_  
_yo mittesu pakubbati;_  
_Akarontaṃ bhāsamānaṃ,_  
_parijānanti paṇḍitā. (2)_  

</Paragraph>

</Division>
<Division number="258" book="sn" bookNo="18" page="318">

<Paragraph number="296" type="verses" book="sn">
_Na so mitto yo sadā appamatto,_  
_Bhedāsaṅkī randhamevānupassī;_  
_Yasmiñca seti urasīva putto,_  
_Sa ve mitto yo parehi abhejjo. (3)_  

</Paragraph>

</Division>
<Division number="259" book="sn" bookNo="18" page="318">

<Paragraph number="297" type="verses" book="sn">
_Pāmujjakaraṇaṃ ṭhānaṃ,_  
_pasaṃsāvahanaṃ sukhaṃ;_  
_Phalānisaṃso bhāveti,_  
_vahanto porisaṃ dhuraṃ. (4)_  

</Paragraph>

</Division>
<Division number="260" book="sn" bookNo="18" page="318">

<Paragraph number="298" type="verses" book="sn">
_Pavivekarasaṃ pitvā,_  
_Rasaṃ upasamassa ca;_  
_Niddaro hoti nippāpo,_  
_Dhammapītirasaṃ pivanti. (5)_  


</Paragraph>
</Division>

<Paragraph number="299" type="center" book="sn">
Hirisuttaṃ tatiyaṃ.
</Paragraph>
