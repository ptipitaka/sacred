<Division number="580" book="sn">

<Paragraph number="695" type="verses" book="sn">
_Animittamanaññātaṃ,_  
_maccānaṃ idha jīvitaṃ;_  
_Kasirañca parittañca,_  
_tañca dukkhena saṃyutaṃ. (1)_  

</Paragraph>

</Division>
<Division number="581" book="sn">

<Paragraph number="696" type="verses" book="sn">
_Na hi so upakkamo atthi,_  
_yena jātā na miyyare;_  
_Jarampi patvā maraṇaṃ,_  
_evaṃdhammā hi pāṇino. (2)_  

</Paragraph>

</Division>
<Division number="582" book="sn">

<Paragraph number="697" type="verses" book="sn">
_Phalānamiva pakkānaṃ,_  
_pāto patanato bhayaṃ;_  
_Evaṃ jātāna maccānaṃ,_  
_niccaṃ maraṇato bhayaṃ. (3)_  

</Paragraph>

</Division>
<Division number="583" book="sn">

<Paragraph number="698" type="verses" book="sn">
_Yathāpi kumbhakārassa,_  
_katā mattikabhājanā;_  
_Sabbe bhedanapariyantā,_  
_evaṃ maccāna jīvitaṃ. (4)_  

</Paragraph>

</Division>
<Division number="584" book="sn">

<Paragraph number="699" type="verses" book="sn">
_Daharā ca mahantā ca,_  
_ye bālā ye ca paṇḍitā;_  
_Sabbe maccuvasaṃ yanti,_  
_sabbe maccuparāyaṇā. (5)_  

</Paragraph>

</Division>
<Division number="585" book="sn">

<Paragraph number="700" type="verses" book="sn">
_Tesaṃ maccuparetānaṃ,_  
_gacchataṃ paralokato;_  
_Na pitā tāyate puttaṃ,_  
_ñātī vā pana ñātake. (6)_  

</Paragraph>

</Division>
<Division number="586" book="sn">

<Paragraph number="701" type="verses" book="sn">
_Pekkhataṃyeva ñātīnaṃ,_  
_passa lālapataṃ puthu;_  
_Ekamekova maccānaṃ,_  
_govajjho viya nīyati. (7)_  

</Paragraph>

</Division>
<Division number="587" book="sn">

<Paragraph number="702" type="verses" book="sn">
_Evamabbhāhato loko,_  
_Maccunā ca jarāya ca;_  
_Tasmā dhīrā na socanti,_  
_Viditvā lokapariyāyaṃ. (8)_  

</Paragraph>

</Division>
<Division number="588" book="sn">

<Paragraph number="703" type="verses" book="sn">
_Yassa maggaṃ na jānāsi,_  
_āgatassa gatassa vā;_  
_Ubho ante asampassaṃ,_  
_niratthaṃ paridevasi. (9)_  

</Paragraph>

</Division>
<Division number="589" book="sn">

<Paragraph number="704" type="verses" book="sn">
_Paridevayamāno ce,_  
_Kiñcidatthaṃ udabbahe;_  
_Sammūḷho hiṃsamattānaṃ,_  
_Kayirā ce naṃ vicakkhaṇo. (10)_  

</Paragraph>

</Division>
<Division number="590" book="sn">

<Paragraph number="705" type="verses" book="sn">
_Na hi ruṇṇena sokena,_  
_Santiṃ pappoti cetaso;_  
_Bhiyyassuppajjate dukkhaṃ,_  
_Sarīraṃ cupahaññati. (11)_  

</Paragraph>

</Division>
<Division number="591" book="sn">

<Paragraph number="706" type="verses" book="sn">
_Kiso vivaṇṇo bhavati,_  
_Hiṃsamattānamattanā;_  
_Na tena petā pālenti,_  
_Niratthā paridevanā. (12)_  

</Paragraph>

</Division>
<Division number="592" book="sn">

<Paragraph number="707" type="verses" book="sn">
_Sokamappajahaṃ jantu,_  
_Bhiyyo dukkhaṃ nigacchati;_  
_Anutthunanto kālaṅkataṃ,_  
_Sokassa vasamanvagū. (13)_  

</Paragraph>

</Division>
<Division number="593" book="sn">

<Paragraph number="708" type="verses" book="sn">
_Aññepi passa gamine,_  
_Yathākammupage nare;_  
_Maccuno vasamāgamma,_  
_Phandantevidha pāṇino. (14)_  

</Paragraph>

</Division>
<Division number="594" book="sn">

<Paragraph number="709" type="verses" book="sn">
_Yena yena hi maññanti,_  
_Tato taṃ hoti aññathā;_  
_Etādiso vinābhāvo,_  
_Passa lokassa pariyāyaṃ. (15)_  

</Paragraph>

</Division>
<Division number="595" book="sn">

<Paragraph number="710" type="verses" book="sn">
_Api vassasataṃ jīve,_  
_Bhiyyo vā pana māṇavo;_  
_Ñātisaṅghā vinā hoti,_  
_Jahāti idha jīvitaṃ. (16)_  

</Paragraph>

</Division>
<Division number="596" book="sn">

<Paragraph number="711" type="verses" book="sn">
_Tasmā arahato sutvā,_  
_Vineyya paridevitaṃ;_  
_Petaṃ kālaṅkataṃ disvā,_  
_Neso labbhā mayā iti. (17)_  

</Paragraph>

</Division>
<Division number="597" book="sn">

<Paragraph number="712" type="verses" book="sn">
_Yathā saraṇamādittaṃ,_  
_Vārinā parinibbaye;_  
_Evampi dhīro sapañño,_  
_Paṇḍito kusalo naro;_  
_Khippamuppatitaṃ sokaṃ,_  
_Vāto tūlaṃva dhaṃsaye. (18)_  

</Paragraph>

</Division>
<Division number="598" book="sn">

<Paragraph number="713" type="verses" book="sn">
_Paridevaṃ pajappañca,_  
_Domanassañca attano;_  
_Attano sukhamesāno,_  
_Abbahe sallamattano. (19)_  

</Paragraph>

</Division>
<Division number="599" book="sn">

<Paragraph number="714" type="verses" book="sn">
_Abbūḷhasallo asito,_  
_santiṃ pappuyya cetaso;_  
_Sabbasokaṃ atikkanto,_  
_asoko hoti nibbutoti._  


</Paragraph>
</Division>

<Paragraph number="715" type="center" book="sn">
Sallasuttaṃ aṭṭhamaṃ.
</Paragraph>
