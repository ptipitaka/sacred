<Paragraph number="999" type="verses" book="sn">
_Idheva suddhī iti vādayanti,_  
_Nāññesu dhammesu visuddhimāhu;_  
_Yaṃ nissitā tattha subhaṃ vadānā,_  
_Paccekasaccesu puthū niviṭṭhā. (1)_  

</Paragraph>

</Division>
<Division number="832" book="sn" bookNo="18" page="408">

<Paragraph number="1000" type="verses" book="sn">
_Te vādakāmā parisaṃ vigayha,_  
_Bālaṃ dahantī mithu aññamaññaṃ;_  
_Vadanti te aññasitā kathojjaṃ,_  
_Pasaṃsakāmā kusalā vadānā. (2)_  

</Paragraph>

</Division>
<Division number="833" book="sn" bookNo="18" page="408">

<Paragraph number="1001" type="verses" book="sn">
_Yutto kathāyaṃ parisāya majjhe,_  
_Pasaṃsamicchaṃ vinighāti hoti;_  
_Apāhatasmiṃ pana maṅku hoti,_  
_Nindāya so kuppati randhamesī. (3)_  

</Paragraph>

</Division>
<Division number="834" book="sn" bookNo="18" page="408">

<Paragraph number="1002" type="verses" book="sn">
_Yamassa vādaṃ parihīnamāhu,_  
_Apāhataṃ pañhavimaṃsakāse;_  
_Paridevati socati hīnavādo,_  
_Upaccagā manti anutthunāti. (4)_  

</Paragraph>

</Division>
<Division number="835" book="sn" bookNo="18" page="408">

<Paragraph number="1003" type="verses" book="sn">
_Ete vivādā samaṇesu jātā,_  
_Etesu ugghātinighāti hoti;_  
_Etampi disvā virame kathojjaṃ,_  
_Na haññadatthatthi pasaṃsalābhā. (5)_  

</Paragraph>

</Division>
<Division number="836" book="sn" bookNo="18" page="408">

<Paragraph number="1004" type="verses" book="sn">
_Pasaṃsito vā pana tattha hoti,_  
_Akkhāya vādaṃ parisāya majjhe;_  
_So hassatī unnamatī ca tena,_  
_Pappuyya tamatthaṃ yathā mano ahu. (6)_  

</Paragraph>

</Division>
<Division number="837" book="sn" bookNo="18" page="408">

<Paragraph number="1005" type="verses" book="sn">
_Yā unnatī sāssa vighātabhūmi,_  
_Mānātimānaṃ vadate paneso;_  
_Etampi disvā na vivādayetha,_  
_Na hi tena suddhiṃ kusalā vadanti. (7)_  

</Paragraph>

</Division>
<Division number="838" book="sn" bookNo="18" page="408">

<Paragraph number="1006" type="verses" book="sn">
_Sūro yathā rājakhādāya puṭṭho,_  
_Abhigajjameti paṭisūramicchaṃ;_  
_Yeneva so tena palehi sūra,_  
_Pubbeva natthi yadidaṃ yudhāya. (8)_  

</Paragraph>

</Division>
<Division number="839" book="sn" bookNo="18" page="409">

<Paragraph number="1007" type="verses" book="sn">
_Ye diṭṭhimuggayha vivādayanti,_  
_Idameva saccanti ca vādayanti;_  
_Te tvaṃ vadassū na hi tedha atthi,_  
_Vādamhi jāte paṭisenikattā. (9)_  

</Paragraph>

</Division>
<Division number="840" book="sn" bookNo="18" page="409">

<Paragraph number="1008" type="verses" book="sn">
_Visenikatvā pana ye caranti,_  
_Diṭṭhīhi diṭṭhiṃ avirujjhamānā;_  
_Tesu tvaṃ kiṃ labhetho pasūra,_  
_Yesīdha natthī paramuggahītaṃ. (10)_  

</Paragraph>

</Division>
<Division number="841" book="sn" bookNo="18" page="409">

<Paragraph number="1009" type="verses" book="sn">
_Atha tvaṃ pavitakkamāgamā,_  
_Manasā diṭṭhigatāni cintayanto;_  
_Dhonena yugaṃ samāgamā,_  
_Na hi tvaṃ sakkhasi sampayātaveti. (11)_  


</Paragraph>
</Division>
