<Paragraph number="950" type="verses" book="sn">
_Vadanti ve duṭṭhamanāpi eke,_  
_Athopi ve saccamanā vadanti;_  
_Vādañca jātaṃ muni no upeti,_  
_Tasmā munī natthi khilo kuhiñci. (1)_  

</Paragraph>

</Division>
<Division number="788" book="sn" bookNo="18" page="401">

<Paragraph number="951" type="verses" book="sn">
_Sakañhi diṭṭhiṃ kathamaccayeyya,_  
_Chandānunīto ruciyā niviṭṭho;_  
_Sayaṃ samattāni pakubbamāno,_  
_Yathā hi jāneyya tathā vadeyya. (2)_  

</Paragraph>

</Division>
<Division number="789" book="sn" bookNo="18" page="401">

<Paragraph number="952" type="verses" book="sn">
_Yo attano sīlavatāni jantu,_  
_Anānupuṭṭhova paresa pāva;_  
_Anariyadhammaṃ kusalā tamāhu,_  
_Yo ātumānaṃ sayameva pāva. (3)_  

</Paragraph>

</Division>
<Division number="790" book="sn" bookNo="18" page="401">

<Paragraph number="953" type="verses" book="sn">
_Santo ca bhikkhu abhinibbutatto,_  
_Itihanti sīlesu akatthamāno;_  
_Tamariyadhammaṃ kusalā vadanti,_  
_Yassussadā natthi kuhiñci loke. (4)_  

</Paragraph>

</Division>
<Division number="791" book="sn" bookNo="18" page="401">

<Paragraph number="954" type="verses" book="sn">
_Pakappitā saṅkhatā yassa dhammā,_  
_Purakkhatā santi avīvadātā;_  
_Yadattani passati ānisaṃsaṃ,_  
_Taṃ nissito kuppapaṭiccasantiṃ. (5)_  

</Paragraph>

</Division>
<Division number="792" book="sn" bookNo="18" page="401">

<Paragraph number="955" type="verses" book="sn">
_Diṭṭhīnivesā na hi svātivattā,_  
_Dhammesu niccheyya samuggahītaṃ;_  
_Tasmā naro tesu nivesanesu,_  
_Nirassatī ādiyatī ca dhammaṃ. (6)_  

</Paragraph>

</Division>
<Division number="793" book="sn" bookNo="18" page="402">

<Paragraph number="956" type="verses" book="sn">
_Dhonassa hi natthi kuhiñci loke,_  
_Pakappitā diṭṭhi bhavābhavesu;_  
_Māyañca mānañca pahāya dhono,_  
_Sa kena gaccheyya anūpayo so. (7)_  

</Paragraph>

</Division>
<Division number="794" book="sn" bookNo="18" page="402">

<Paragraph number="957" type="verses" book="sn">
_Upayo hi dhammesu upeti vādaṃ,_  
_Anūpayaṃ kena kathaṃ vadeyya;_  
_Attā nirattā na hi tassa atthi,_  
_Adhosi so diṭṭhimidheva sabbanti. (8)_  


</Paragraph>
</Division>

<Paragraph number="958" type="center" book="sn">
Duṭṭhaṭṭhakasuttaṃ tatiyaṃ.
</Paragraph>
