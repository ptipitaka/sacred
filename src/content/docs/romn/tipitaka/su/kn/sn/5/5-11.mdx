<Division number="1103" book="sn">

<Paragraph number="1299" type="verses" book="sn">
_“Sutvānahaṃ vīramakāmakāmiṃ, __(iccāyasmā jatukaṇṇi)___  
_Oghātigaṃ puṭṭhumakāmamāgamaṃ;_  
_Santipadaṃ brūhi sahajanetta,_  
_Yathātacchaṃ bhagavā brūhi metaṃ. (1)_  

</Paragraph>

</Division>
<Division number="1104" book="sn">

<Paragraph number="1300" type="verses" book="sn">
_Bhagavā hi kāme abhibhuyya iriyati,_  
_Ādiccova pathaviṃ tejī tejasā;_  
_Parittapaññassa me bhūripañña,_  
_Ācikkha dhammaṃ yamahaṃ vijaññaṃ;_  
_Jātijarāya idha vippahānaṃ”. (2)_  

</Paragraph>

</Division>
<Division number="1105" book="sn">

<Paragraph number="1301" type="verses" book="sn">
_“Kāmesu vinaya gedhaṃ, __(jatukaṇṇīti bhagavā)___  
_Nekkhammaṃ daṭṭhu khemato;_  
_Uggahītaṃ nirattaṃ vā,_  
_Mā te vijjittha kiñcanaṃ. (3)_  

</Paragraph>

</Division>
<Division number="1106" book="sn">

<Paragraph number="1302" type="verses" book="sn">
_Yaṃ pubbe taṃ visosehi,_  
_pacchā te māhu kiñcanaṃ;_  
_Majjhe ce no gahessasi,_  
_upasanto carissasi. (4)_  

</Paragraph>

</Division>
<Division number="1107" book="sn">

<Paragraph number="1303" type="verses" book="sn">
_Sabbaso nāmarūpasmiṃ,_  
_Vītagedhassa brāhmaṇa;_  
_Āsavāssa na vijjanti,_  
_Yehi maccuvasaṃ vaje”ti. (5)_  


</Paragraph>
</Division>

<Paragraph number="1304" type="center" book="sn">
Jatukaṇṇimāṇavapucchā ekādasamā.
</Paragraph>
