<Paragraph number="817" type="verses" book="sn">
_Ānandajāte tidasagaṇe patīte,_  
_Sakkañca indaṃ sucivasane ca deve;_  
_Dussaṃ gahetvā atiriva thomayante,_  
_Asito isi addasa divāvihāre. (1)_  

</Paragraph>

</Division>
<Division number="686" book="sn" bookNo="18" page="385">

<Paragraph number="818" type="verses" book="sn">
_Disvāna deve muditamane udagge,_  
_Cittiṃ karitvāna idamavoca tattha;_  
_“Kiṃ devasaṅgho atiriva kalyarūpo,_  
_Dussaṃ gahetvā ramayatha kiṃ paṭicca. (2)_  

</Paragraph>

</Division>
<Division number="687" book="sn" bookNo="18" page="385">

<Paragraph number="819" type="verses" book="sn">
_Yadāpi āsī asurehi saṅgamo,_  
_Jayo surānaṃ asurā parājitā;_  
_Tadāpi netādiso lomahaṃsano,_  
_Kimabbhutaṃ daṭṭhu marū pamoditā. (3)_  

</Paragraph>

</Division>
<Division number="688" book="sn" bookNo="18" page="385">

<Paragraph number="820" type="verses" book="sn">
_Seḷenti gāyanti ca vādayanti ca,_  
_Bhujāni phoṭenti ca naccayanti ca;_  
_Pucchāmi vohaṃ merumuddhavāsine,_  
_Dhunātha me saṃsayaṃ khippa mārisā”. (4)_  

</Paragraph>

</Division>
<Division number="689" book="sn" bookNo="18" page="386">

<Paragraph number="821" type="verses" book="sn">
_“So bodhisatto ratanavaro atulyo,_  
_Manussaloke hitasukhatthāya jāto;_  
_Sakyāna gāme janapade lumbineyye,_  
_Tenamha tuṭṭhā atiriva kalyarūpā. (5)_  

</Paragraph>

</Division>
<Division number="690" book="sn" bookNo="18" page="386">

<Paragraph number="822" type="verses" book="sn">
_So sabbasattuttamo aggapuggalo,_  
_Narāsabho sabbapajānamuttamo;_  
_Vattessati cakkamisivhaye vane,_  
_Nadaṃva sīho balavā migābhibhū”. (6)_  

</Paragraph>

</Division>
<Division number="691" book="sn" bookNo="18" page="386">

<Paragraph number="823" type="verses" book="sn">
_Taṃ saddaṃ sutvā turitamavasarī so,_  
_Suddhodanassa tada bhavanaṃ upāvisi;_  
_Nisajja tattha idamavocāsi sakye,_  
_“Kuhiṃ kumāro ahamapi daṭṭhukāmo”. (7)_  

</Paragraph>

</Division>
<Division number="692" book="sn" bookNo="18" page="386">

<Paragraph number="824" type="verses" book="sn">
_Tato kumāraṃ jalitamiva suvaṇṇaṃ,_  
_Ukkāmukheva sukusalasampahaṭṭhaṃ;_  
_Daddallamānaṃ siriyā anomavaṇṇaṃ,_  
_Dassesu puttaṃ asitavhayassa sakyā. (8)_  

</Paragraph>

</Division>
<Division number="693" book="sn" bookNo="18" page="386">

<Paragraph number="825" type="verses" book="sn">
_Disvā kumāraṃ sikhimiva pajjalantaṃ,_  
_Tārāsabhaṃva nabhasigamaṃ visuddhaṃ;_  
_Suriyaṃ tapantaṃ saradarivabbhamuttaṃ,_  
_Ānandajāto vipulamalattha pītiṃ. (9)_  

</Paragraph>

</Division>
<Division number="694" book="sn" bookNo="18" page="386">

<Paragraph number="826" type="verses" book="sn">
_Anekasākhañca sahassamaṇḍalaṃ,_  
_Chattaṃ marū dhārayumantalikkhe;_  
_Suvaṇṇadaṇḍā vītipatanti cāmarā,_  
_Na dissare cāmarachattagāhakā. (10)_  

</Paragraph>

</Division>
<Division number="695" book="sn" bookNo="18" page="387">

<Paragraph number="827" type="verses" book="sn">
_Disvā jaṭī kaṇhasirivhayo isi,_  
_Suvaṇṇanikkhaṃ viya paṇḍukambale;_  
_Setañca chattaṃ dhariyanta muddhani,_  
_Udaggacitto sumano paṭiggahe. (11)_  

</Paragraph>

</Division>
<Division number="696" book="sn" bookNo="18" page="387">

<Paragraph number="828" type="verses" book="sn">
_Paṭiggahetvā pana sakyapuṅgavaṃ,_  
_Jigīsako lakkhaṇamantapāragū;_  
_Pasannacitto giramabbhudīrayi,_  
_“Anuttarāyaṃ dvipadānamuttamo”. (12)_  

</Paragraph>

</Division>
<Division number="697" book="sn" bookNo="18" page="387">

<Paragraph number="829" type="verses" book="sn">
_Athattano gamanamanussaranto,_  
_Akalyarūpo gaḷayati assukāni;_  
_Disvāna sakyā isimavocuṃ rudantaṃ,_  
_“No ce kumāre bhavissati antarāyo”. (13)_  

</Paragraph>

</Division>
<Division number="698" book="sn" bookNo="18" page="387">

<Paragraph number="830" type="verses" book="sn">
_Disvāna sakye isimavoca akalye,_  
_“Nāhaṃ kumāre ahitamanussarāmi;_  
_Na cāpimassa bhavissati antarāyo,_  
_Na orakāyaṃ adhimānasā bhavātha. (14)_  

</Paragraph>

</Division>
<Division number="699" book="sn" bookNo="18" page="387">

<Paragraph number="831" type="verses" book="sn">
_Sambodhiyaggaṃ phusissatāyaṃ kumāro,_  
_So dhammacakkaṃ paramavisuddhadassī;_  
_Vattessatāyaṃ bahujanahitānukampī,_  
_Vitthārikassa bhavissati brahmacariyaṃ. (15)_  

</Paragraph>

</Division>
<Division number="700" book="sn" bookNo="18" page="387">

<Paragraph number="832" type="verses" book="sn">
_Mamañca āyu na ciramidhāvaseso,_  
_Athantarā me bhavissati kālakiriyā;_  
_Sohaṃ na sossaṃ asamadhurassa dhammaṃ,_  
_Tenamhi aṭṭo byasanaṅgato aghāvī”. (16)_  

</Paragraph>

</Division>
<Division number="701" book="sn" bookNo="18" page="388">

<Paragraph number="833" type="verses" book="sn">
_So sākiyānaṃ vipulaṃ janetvā pītiṃ,_  
_Antepuramhā niggamā brahmacārī;_  
_So bhāgineyyaṃ sayaṃ anukampamāno,_  
_Samādapesi asamadhurassa dhamme. (17)_  

</Paragraph>

</Division>
<Division number="702" book="sn" bookNo="18" page="388">

<Paragraph number="834" type="verses" book="sn">
_“Buddhoti ghosaṃ yada parato suṇāsi,_  
_Sambodhipatto vivarati dhammamaggaṃ;_  
_Gantvāna tattha samayaṃ paripucchamāno,_  
_Carassu tasmiṃ bhagavati brahmacariyaṃ”. (18)_  

</Paragraph>

</Division>
<Division number="703" book="sn" bookNo="18" page="388">

<Paragraph number="835" type="verses" book="sn">
_Tenānusiṭṭho hitamanena tādinā,_  
_Anāgate paramavisuddhadassinā;_  
_So nālako upacitapuññasañcayo,_  
_Jinaṃ patikkhaṃ parivasi rakkhitindriyo. (19)_  

</Paragraph>

</Division>
<Division number="704" book="sn" bookNo="18" page="388">

<Paragraph number="836" type="verses" book="sn">
_Sutvāna ghosaṃ jinavaracakkavattane,_  
_Gantvāna disvā isinisabhaṃ pasanno;_  
_Moneyyaseṭṭhaṃ munipavaraṃ apucchi,_  
_Samāgate asitāvhayassa sāsaneti. (20)_  


</Paragraph>
</Division>

<Paragraph number="837" type="center" book="sn">
Vatthugāthā niṭṭhitā.


</Paragraph>

<Division number="705" book="sn" bookNo="18" page="388">

<Paragraph number="838" type="verses" book="sn">
_“Aññātametaṃ vacanaṃ,_  
_Asitassa yathātathaṃ;_  
_Taṃ taṃ gotama pucchāmi,_  
_Sabbadhammāna pāraguṃ. (21)_  

</Paragraph>

</Division>
<Division number="706" book="sn" bookNo="18" page="388">

<Paragraph number="839" type="verses" book="sn">
_Anagāriyupetassa,_  
_Bhikkhācariyaṃ jigīsato;_  
_Muni pabrūhi me puṭṭho,_  
_Moneyyaṃ uttamaṃ padaṃ”. (22)_  

</Paragraph>

</Division>
<Division number="707" book="sn" bookNo="18" page="388">

<Paragraph number="840" type="verses" book="sn">
_“Moneyyaṃ te upaññissaṃ, __(iti bhagavā)___  
_Dukkaraṃ durabhisambhavaṃ;_  
_Handa te naṃ pavakkhāmi,_  
_Santhambhassu daḷho bhava. (23)_  

</Paragraph>

</Division>
<Division number="708" book="sn" bookNo="18" page="389">

<Paragraph number="841" type="verses" book="sn">
_Samānabhāgaṃ kubbetha,_  
_Gāme akkuṭṭhavanditaṃ;_  
_Manopadosaṃ rakkheyya,_  
_Santo anuṇṇato care. (24)_  

</Paragraph>

</Division>
<Division number="709" book="sn" bookNo="18" page="389">

<Paragraph number="842" type="verses" book="sn">
_Uccāvacā niccharanti,_  
_Dāye aggisikhūpamā;_  
_Nāriyo muniṃ palobhenti,_  
_Tā su taṃ mā palobhayuṃ. (25)_  

</Paragraph>

</Division>
<Division number="710" book="sn" bookNo="18" page="389">

<Paragraph number="843" type="verses" book="sn">
_Virato methunā dhammā,_  
_Hitvā kāme paropare;_  
_Aviruddho asāratto,_  
_Pāṇesu tasathāvare. (26)_  

</Paragraph>

</Division>
<Division number="711" book="sn" bookNo="18" page="389">

<Paragraph number="844" type="verses" book="sn">
_Yathā ahaṃ tathā ete,_  
_Yathā ete tathā ahaṃ;_  
_Attānaṃ upamaṃ katvā,_  
_Na haneyya na ghātaye. (27)_  

</Paragraph>

</Division>
<Division number="712" book="sn" bookNo="18" page="389">

<Paragraph number="845" type="verses" book="sn">
_Hitvā icchañca lobhañca,_  
_Yattha satto puthujjano;_  
_Cakkhumā paṭipajjeyya,_  
_Tareyya narakaṃ imaṃ. (28)_  

</Paragraph>

</Division>
<Division number="713" book="sn" bookNo="18" page="389">

<Paragraph number="846" type="verses" book="sn">
_Ūnūdaro mitāhāro,_  
_Appicchassa alolupo;_  
_Sadā icchāya nicchāto,_  
_Aniccho hoti nibbuto. (29)_  

</Paragraph>

</Division>
<Division number="714" book="sn" bookNo="18" page="389">

<Paragraph number="847" type="verses" book="sn">
_Sa piṇḍacāraṃ caritvā,_  
_Vanantamabhihāraye;_  
_Upaṭṭhito rukkhamūlasmiṃ,_  
_Āsanūpagato muni. (30)_  

</Paragraph>

</Division>
<Division number="715" book="sn" bookNo="18" page="389">

<Paragraph number="848" type="verses" book="sn">
_Sa jhānapasuto dhīro,_  
_Vanante ramito siyā;_  
_Jhāyetha rukkhamūlasmiṃ,_  
_Attānamabhitosayaṃ. (31)_  

</Paragraph>

</Division>
<Division number="716" book="sn" bookNo="18" page="389">

<Paragraph number="849" type="verses" book="sn">
_Tato ratyā vivasāne,_  
_Gāmantamabhihāraye;_  
_Avhānaṃ nābhinandeyya,_  
_Abhihārañca gāmato. (32)_  

</Paragraph>

</Division>
<Division number="717" book="sn" bookNo="18" page="389">

<Paragraph number="850" type="verses" book="sn">
_Na munī gāmamāgamma,_  
_Kulesu sahasā care;_  
_Ghāsesanaṃ chinnakatho,_  
_Na vācaṃ payutaṃ bhaṇe. (33)_  

</Paragraph>

</Division>
<Division number="718" book="sn" bookNo="18" page="389">

<Paragraph number="851" type="verses" book="sn">
_Alatthaṃ yadidaṃ sādhu,_  
_Nālatthaṃ kusalaṃ iti;_  
_Ubhayeneva so tādī,_  
_Rukkhaṃvupanivattati. (34)_  

</Paragraph>

</Division>
<Division number="719" book="sn" bookNo="18" page="389">

<Paragraph number="852" type="verses" book="sn">
_Sa pattapāṇi vicaranto,_  
_Amūgo mūgasammato;_  
_Appaṃ dānaṃ na hīḷeyya,_  
_Dātāraṃ nāvajāniyā. (35)_  

</Paragraph>

</Division>
<Division number="720" book="sn" bookNo="18" page="390">

<Paragraph number="853" type="verses" book="sn">
_Uccāvacā hi paṭipadā,_  
_Samaṇena pakāsitā;_  
_Na pāraṃ diguṇaṃ yanti,_  
_Nayidaṃ ekaguṇaṃ mutaṃ. (36)_  

</Paragraph>

</Division>
<Division number="721" book="sn" bookNo="18" page="390">

<Paragraph number="854" type="verses" book="sn">
_Yassa ca visatā natthi,_  
_Chinnasotassa bhikkhuno;_  
_Kiccākiccappahīnassa,_  
_Pariḷāho na vijjati. (37)_  

</Paragraph>

</Division>
<Division number="722" book="sn" bookNo="18" page="390">

<Paragraph number="855" type="verses" book="sn">
_Moneyyaṃ te upaññissaṃ,_  
_Khuradhārūpamo bhave;_  
_Jivhāya tālumāhacca,_  
_Udare saññato siyā. (38)_  

</Paragraph>

</Division>
<Division number="723" book="sn" bookNo="18" page="390">

<Paragraph number="856" type="verses" book="sn">
_Alīnacitto ca siyā,_  
_Na cāpi bahu cintaye;_  
_Nirāmagandho asito,_  
_Brahmacariyaparāyaṇo. (39)_  

</Paragraph>

</Division>
<Division number="724" book="sn" bookNo="18" page="390">

<Paragraph number="857" type="verses" book="sn">
_Ekāsanassa sikkhetha,_  
_Samaṇūpāsanassa ca;_  
_Ekattaṃ monamakkhātaṃ,_  
_Eko ce abhiramissasi;_  
_Atha bhāhisi dasadisā. (40)_  

</Paragraph>

</Division>
<Division number="725" book="sn" bookNo="18" page="390">

<Paragraph number="858" type="verses" book="sn">
_Sutvā dhīrānaṃ nighosaṃ,_  
_Jhāyīnaṃ kāmacāginaṃ;_  
_Tato hiriñca saddhañca,_  
_Bhiyyo kubbetha māmako. (41)_  

</Paragraph>

</Division>
<Division number="726" book="sn" bookNo="18" page="390">

<Paragraph number="859" type="verses" book="sn">
_Taṃ nadīhi vijānātha,_  
_Sobbhesu padaresu ca;_  
_Saṇantā yanti kusobbhā,_  
_Tuṇhī yanti mahodadhī. (42)_  

</Paragraph>

</Division>
<Division number="727" book="sn" bookNo="18" page="390">

<Paragraph number="860" type="verses" book="sn">
_Yadūnakaṃ taṃ saṇati,_  
_Yaṃ pūraṃ santameva taṃ;_  
_Aḍḍhakumbhūpamo bālo,_  
_Rahado pūrova paṇḍito. (43)_  

</Paragraph>

</Division>
<Division number="728" book="sn" bookNo="18" page="390">

<Paragraph number="861" type="verses" book="sn">
_Yaṃ samaṇo bahuṃ bhāsati,_  
_Upetaṃ atthasañhitaṃ;_  
_Jānaṃ so dhammaṃ deseti,_  
_Jānaṃ so bahu bhāsati. (44)_  

</Paragraph>

</Division>
<Division number="729" book="sn" bookNo="18" page="390">

<Paragraph number="862" type="verses" book="sn">
_Yo ca jānaṃ saṃyatatto,_  
_Jānaṃ na bahu bhāsati;_  
_Sa munī monamarahati,_  
_Sa munī monamajjhagā”ti. (45)_  


</Paragraph>
</Division>

<Paragraph number="863" type="center" book="sn">
Nālakasuttaṃ ekādasamaṃ.
</Paragraph>
