<Paragraph number="225" type="verses" book="sn">
_Caraṃ vā yadi vā tiṭṭhaṃ,_  
_nisinno uda vā sayaṃ;_  
_Samiñjeti pasāreti,_  
_esā kāyassa iñjanā. (1)_  

</Paragraph>

</Division>
<Division number="196" book="sn" bookNo="18" page="307">

<Paragraph number="226" type="verses" book="sn">
_Aṭṭhinahārusaṃyutto,_  
_tacamaṃsāvalepano;_  
_Chaviyā kāyo paṭicchanno,_  
_yathābhūtaṃ na dissati. (2)_  

</Paragraph>

</Division>
<Division number="197" book="sn" bookNo="18" page="308">

<Paragraph number="227" type="verses" book="sn">
_Antapūro udarapūro,_  
_yakanapeḷassa vatthino;_  
_Hadayassa papphāsassa,_  
_vakkassa pihakassa ca. (3)_  

</Paragraph>

</Division>
<Division number="198" book="sn" bookNo="18" page="308">

<Paragraph number="228" type="verses" book="sn">
_Siṅghāṇikāya kheḷassa,_  
_sedassa ca medassa ca;_  
_Lohitassa lasikāya,_  
_pittassa ca vasāya ca. (4)_  

</Paragraph>

</Division>
<Division number="199" book="sn" bookNo="18" page="308">

<Paragraph number="229" type="verses" book="sn">
_Athassa navahi sotehi,_  
_asucī savati sabbadā;_  
_Akkhimhā akkhigūthako,_  
_kaṇṇamhā kaṇṇagūthako. (5)_  

</Paragraph>

</Division>
<Division number="200" book="sn" bookNo="18" page="308">

<Paragraph number="230" type="verses" book="sn">
_Siṅghāṇikā ca nāsato,_  
_mukhena vamatekadā;_  
_Pittaṃ semhañca vamati,_  
_kāyamhā sedajallikā. (6)_  

</Paragraph>

</Division>
<Division number="201" book="sn" bookNo="18" page="308">

<Paragraph number="231" type="verses" book="sn">
_Athassa susiraṃ sīsaṃ,_  
_matthaluṅgassa pūritaṃ;_  
_Subhato naṃ maññati bālo,_  
_avijjāya purakkhato. (7)_  

</Paragraph>

</Division>
<Division number="202" book="sn" bookNo="18" page="308">

<Paragraph number="232" type="verses" book="sn">
_Yadā ca so mato seti,_  
_Uddhumāto vinīlako;_  
_Apaviddho susānasmiṃ,_  
_Anapekkhā honti ñātayo. (8)_  

</Paragraph>

</Division>
<Division number="203" book="sn" bookNo="18" page="308">

<Paragraph number="233" type="verses" book="sn">
_Khādanti naṃ suvānā ca,_  
_siṅgālā ca vakā kimī;_  
_Kākā gijjhā ca khādanti,_  
_ye caññe santi pāṇino. (9)_  

</Paragraph>

</Division>
<Division number="204" book="sn" bookNo="18" page="308">

<Paragraph number="234" type="verses" book="sn">
_Sutvāna buddhavacanaṃ,_  
_Bhikkhu paññāṇavā idha;_  
_So kho naṃ parijānāti,_  
_Yathābhūtañhi passati. (10)_  

</Paragraph>

</Division>
<Division number="205" book="sn" bookNo="18" page="308">

<Paragraph number="235" type="verses" book="sn">
_Yathā idaṃ tathā etaṃ,_  
_Yathā etaṃ tathā idaṃ;_  
_Ajjhattañca bahiddhā ca,_  
_Kāye chandaṃ virājaye. (11)_  

</Paragraph>

</Division>
<Division number="206" book="sn" bookNo="18" page="308">

<Paragraph number="236" type="verses" book="sn">
_Chandarāgaviratto so,_  
_Bhikkhu paññāṇavā idha;_  
_Ajjhagā amataṃ santiṃ,_  
_Nibbānaṃ padamaccutaṃ. (12)_  

</Paragraph>

</Division>
<Division number="207" book="sn" bookNo="18" page="308">

<Paragraph number="237" type="verses" book="sn">
_Dvipādakoyaṃ asuci,_  
_Duggandho parihīrati;_  
_Nānākuṇapaparipūro,_  
_Vissavanto tato tato. (13)_  

</Paragraph>

</Division>
<Division number="208" book="sn" bookNo="18" page="308">

<Paragraph number="238" type="verses" book="sn">
_Etādisena kāyena,_  
_Yo maññe uṇṇametave;_  
_Paraṃ vā avajāneyya,_  
_Kimaññatra adassanāti. (14)_  


</Paragraph>
</Division>

<Paragraph number="239" type="center" book="sn">
Vijayasuttaṃ ekādasamaṃ.
</Paragraph>
