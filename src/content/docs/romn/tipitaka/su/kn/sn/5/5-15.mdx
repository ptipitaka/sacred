<Paragraph number="1323" type="verses" book="sn">
_“Dvāhaṃ sakkaṃ apucchissaṃ, __(iccāyasmā mogharājā)___  
_Na me byākāsi cakkhumā;_  
_Yāvatatiyañca devīsi,_  
_Byākarotīti me sutaṃ. (1)_  

</Paragraph>

</Division>
<Division number="1124" book="sn" bookNo="18" page="447">

<Paragraph number="1324" type="verses" book="sn">
_Ayaṃ loko paro loko,_  
_brahmaloko sadevako;_  
_Diṭṭhiṃ te nābhijānāti,_  
_gotamassa yasassino. (2)_  

</Paragraph>

</Division>
<Division number="1125" book="sn" bookNo="18" page="447">

<Paragraph number="1325" type="verses" book="sn">
_Evaṃ abhikkantadassāviṃ,_  
_atthi pañhena āgamaṃ;_  
_Kathaṃ lokaṃ avekkhantaṃ,_  
_maccurājā na passati”. (3)_  

</Paragraph>

</Division>
<Division number="1126" book="sn" bookNo="18" page="448">

<Paragraph number="1326" type="verses" book="sn">
_“Suññato lokaṃ avekkhassu,_  
_Mogharāja sadā sato;_  
_Attānudiṭṭhiṃ ūhacca,_  
_Evaṃ maccutaro siyā;_  
_Evaṃ lokaṃ avekkhantaṃ,_  
_Maccurājā na passatī”ti. (4)_  


</Paragraph>
</Division>

<Paragraph number="1327" type="center" book="sn">
Mogharājamāṇavapucchā pannarasamā.
</Paragraph>
