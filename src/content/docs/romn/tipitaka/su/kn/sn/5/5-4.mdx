<Paragraph number="1241" type="verses" book="sn">
_“Pucchāmi taṃ bhagavā brūhi me taṃ, __(iccāyasmā mettagū)___  
_Maññāmi taṃ vedaguṃ bhāvitattaṃ;_  
_Kuto nu dukkhā samudāgatā ime,_  
_Ye keci lokasmimanekarūpā”. (1)_  

</Paragraph>

</Division>
<Division number="1057" book="sn" bookNo="18" page="437">

<Paragraph number="1242" type="verses" book="sn">
_“Dukkhassa ve maṃ pabhavaṃ apucchasi, __(mettagūti bhagavā)___  
_Taṃ te pavakkhāmi yathā pajānaṃ;_  
_Upadhinidānā pabhavanti dukkhā,_  
_Ye keci lokasmimanekarūpā. (2)_  

</Paragraph>

</Division>
<Division number="1058" book="sn" bookNo="18" page="437">

<Paragraph number="1243" type="verses" book="sn">
_Yo ve avidvā upadhiṃ karoti,_  
_Punappunaṃ dukkhamupeti mando;_  
_Tasmā pajānaṃ upadhiṃ na kayirā,_  
_Dukkhassa jātippabhavānupassī”. (3)_  

</Paragraph>

</Division>
<Division number="1059" book="sn" bookNo="18" page="437">

<Paragraph number="1244" type="verses" book="sn">
_“Yaṃ taṃ apucchimha akittayī no,_  
_Aññaṃ taṃ pucchāma tadiṅgha brūhi;_  
_Kathaṃ nu dhīrā vitaranti oghaṃ,_  
_Jātiṃ jaraṃ sokapariddavañca;_  
_Taṃ me muni sādhu viyākarohi,_  
_Tathā hi te vidito esa dhammo”. (4)_  

</Paragraph>

</Division>
<Division number="1060" book="sn" bookNo="18" page="437">

<Paragraph number="1245" type="verses" book="sn">
_“Kittayissāmi te dhammaṃ, __(mettagūti bhagavā)___  
_Diṭṭhe dhamme anītihaṃ;_  
_Yaṃ viditvā sato caraṃ,_  
_Tare loke visattikaṃ”. (5)_  

</Paragraph>

</Division>
<Division number="1061" book="sn" bookNo="18" page="437">

<Paragraph number="1246" type="verses" book="sn">
_“Tañcāhaṃ abhinandāmi,_  
_mahesi dhammamuttamaṃ;_  
_Yaṃ viditvā sato caraṃ,_  
_tare loke visattikaṃ”. (6)_  

</Paragraph>

</Division>
<Division number="1062" book="sn" bookNo="18" page="437">

<Paragraph number="1247" type="verses" book="sn">
_“Yaṃ kiñci sampajānāsi, __(mettagūti bhagavā)___  
_Uddhaṃ adho tiriyañcāpi majjhe;_  
_Etesu nandiñca nivesanañca,_  
_Panujja viññāṇaṃ bhave na tiṭṭhe. (7)_  

</Paragraph>

</Division>
<Division number="1063" book="sn" bookNo="18" page="437">

<Paragraph number="1248" type="verses" book="sn">
_Evaṃvihārī sato appamatto,_  
_Bhikkhu caraṃ hitvā mamāyitāni;_  
_Jātiṃ jaraṃ sokapariddavañca,_  
_Idheva vidvā pajaheyya dukkhaṃ”. (8)_  

</Paragraph>

</Division>
<Division number="1064" book="sn" bookNo="18" page="438">

<Paragraph number="1249" type="verses" book="sn">
_“Etābhinandāmi vaco mahesino,_  
_Sukittitaṃ gotamanūpadhīkaṃ;_  
_Addhā hi bhagavā pahāsi dukkhaṃ,_  
_Tathā hi te vidito esa dhammo. (9)_  

</Paragraph>

</Division>
<Division number="1065" book="sn" bookNo="18" page="438">

<Paragraph number="1250" type="verses" book="sn">
_Te cāpi nūnappajaheyyu dukkhaṃ,_  
_Ye tvaṃ muni aṭṭhitaṃ ovadeyya;_  
_Taṃ taṃ namassāmi samecca nāga,_  
_Appeva maṃ bhagavā aṭṭhitaṃ ovadeyya”. (10)_  

</Paragraph>

</Division>
<Division number="1066" book="sn" bookNo="18" page="438">

<Paragraph number="1251" type="verses" book="sn">
_“Yaṃ brāhmaṇaṃ vedagumābhijaññā,_  
_Akiñcanaṃ kāmabhave asattaṃ;_  
_Addhā hi so oghamimaṃ atāri,_  
_Tiṇṇo ca pāraṃ akhilo akaṅkho. (11)_  

</Paragraph>

</Division>
<Division number="1067" book="sn" bookNo="18" page="438">

<Paragraph number="1252" type="verses" book="sn">
_Vidvā ca yo vedagū naro idha,_  
_Bhavābhave saṅgamimaṃ visajja;_  
_So vītataṇho anīgho nirāso,_  
_Atāri so jātijaranti brūmī”ti. (12)_  


</Paragraph>
</Division>

<Paragraph number="1253" type="center" book="sn">
Mettagūmāṇavapucchā catutthī.
</Paragraph>
