<Paragraph number="1341" type="verses" book="sn">
_“Pārāyanamanugāyissaṃ, __(iccāyasmā piṅgiyo)___  
_Yathāddakkhi tathākkhāsi;_  
_Vimalo bhūrimedhaso,_  
_Nikkāmo nibbano nāgo;_  
_Kissa hetu musā bhaṇe. (1)_  

</Paragraph>

</Division>
<Division number="1139" book="sn" bookNo="18" page="450">

<Paragraph number="1342" type="verses" book="sn">
_Pahīnamalamohassa,_  
_mānamakkhappahāyino;_  
_Handāhaṃ kittayissāmi,_  
_giraṃ vaṇṇūpasañhitaṃ. (2)_  

</Paragraph>

</Division>
<Division number="1140" book="sn" bookNo="18" page="450">

<Paragraph number="1343" type="verses" book="sn">
_Tamonudo buddho samantacakkhu,_  
_Lokantagū sabbabhavātivatto;_  
_Anāsavo sabbadukkhappahīno,_  
_Saccavhayo brahme upāsito me. (3)_  

</Paragraph>

</Division>
<Division number="1141" book="sn" bookNo="18" page="450">

<Paragraph number="1344" type="verses" book="sn">
_Dijo yathā kubbanakaṃ pahāya,_  
_Bahupphalaṃ kānanamāvaseyya;_  
_Evampahaṃ appadasse pahāya,_  
_Mahodadhiṃ haṃsoriva ajjhapatto. (4)_  

</Paragraph>

</Division>
<Division number="1142" book="sn" bookNo="18" page="450">

<Paragraph number="1345" type="verses" book="sn">
_Ye me pubbe viyākaṃsu,_  
_Huraṃ gotamasāsanā;_  
_Iccāsi iti bhavissati,_  
_Sabbaṃ taṃ itihītihaṃ;_  
_Sabbaṃ taṃ takkavaḍḍhanaṃ. (5)_  

</Paragraph>

</Division>
<Division number="1143" book="sn" bookNo="18" page="450">

<Paragraph number="1346" type="verses" book="sn">
_Eko tamanudāsino,_  
_jutimā so pabhaṅkaro;_  
_Gotamo bhūripaññāṇo,_  
_gotamo bhūrimedhaso. (6)_  

</Paragraph>

</Division>
<Division number="1144" book="sn" bookNo="18" page="450">

<Paragraph number="1347" type="verses" book="sn">

_Sandiṭṭhikamakālikaṃ;_  
_Taṇhakkhayamanītikaṃ,_  
_Yassa natthi upamā kvaci”. (7)_  

</Paragraph>

</Division>

</Paragraph>

</Division>

</Paragraph>

</Division>

<Paragraph number="1352" type="verses" book="sn">
_Passāmi naṃ manasā cakkhunāva,_  
_Rattindivaṃ brāhmaṇa appamatto;_  
_Namassamāno vivasemi rattiṃ,_  
_Teneva maññāmi avippavāsaṃ. (12)_  

</Paragraph>

</Division>
<Division number="1150" book="sn" bookNo="18" page="451">

<Paragraph number="1353" type="verses" book="sn">
_Saddhā ca pīti ca mano sati ca,_  
_Nāpentime gotamasāsanamhā;_  
_Yaṃ yaṃ disaṃ vajati bhūripañño,_  
_Sa tena teneva natohamasmi. (13)_  

</Paragraph>

</Division>
<Division number="1151" book="sn" bookNo="18" page="451">

<Paragraph number="1354" type="verses" book="sn">
_Jiṇṇassa me dubbalathāmakassa,_  
_Teneva kāyo na paleti tattha;_  
_Saṃkappayantāya vajāmi niccaṃ,_  
_Mano hi me brāhmaṇa tena yutto. (14)_  

</Paragraph>

</Division>
<Division number="1152" book="sn" bookNo="18" page="451">

<Paragraph number="1355" type="verses" book="sn">
_Paṅke sayāno pariphandamāno,_  
_Dīpā dīpaṃ upaplaviṃ;_  
_Athaddasāsiṃ sambuddhaṃ,_  
_Oghatiṇṇamanāsavaṃ”. (15)_  

</Paragraph>

</Division>
<Division number="1153" book="sn" bookNo="18" page="451">

<Paragraph number="1356" type="verses" book="sn">

_Bhadrāvudho āḷavigotamo ca;_  
_Evamevaṃ tvampi pamuñcassu saddhaṃ,_  
_Gamissasi tvaṃ piṅgiya maccudheyyassa pāraṃ”. ((16))_  

</Paragraph>

</Division>
<Division number="1154" book="sn" bookNo="18" page="451">

_Sutvāna munino vaco;_  
_Vivaṭṭacchado sambuddho,_  
_Akhilo paṭibhānavā. (17)_  

</Paragraph>

</Division>
<Division number="1155" book="sn" bookNo="18" page="451">

_Sabbaṃ vedi varovaraṃ;_  
_Pañhānantakaro satthā,_  
_Kaṅkhīnaṃ paṭijānataṃ. (18)_  

</Paragraph>

</Division>

_Yassa natthi upamā kvaci;_  
_Addhā gamissāmi na mettha kaṅkhā,_  
_Evaṃ maṃ dhārehi adhimuttacittan”ti. (19)_  


</Paragraph>
