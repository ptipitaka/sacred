<Paragraph number="240" type="verses" book="sn">
_Santhavāto bhayaṃ jātaṃ,_  
_niketā jāyate rajo;_  
_Aniketamasanthavaṃ,_  
_etaṃ ve munidassanaṃ. (1)_  

</Paragraph>

</Division>
<Division number="210" book="sn" bookNo="18" page="309">

<Paragraph number="241" type="verses" book="sn">
_Yo jātamucchijja na ropayeyya,_  
_Jāyantamassa nānuppavecche;_  
_Tamāhu ekaṃ muninaṃ carantaṃ,_  
_Addakkhi so santipadaṃ mahesi. (2)_  

</Paragraph>

</Division>
<Division number="211" book="sn" bookNo="18" page="309">

<Paragraph number="242" type="verses" book="sn">
_Saṅkhāya vatthūni pamāya bījaṃ,_  
_Sinehamassa nānuppavecche;_  
_Sa ve munī jātikhayantadassī,_  
_Takkaṃ pahāya na upeti saṅkhaṃ. (3)_  

</Paragraph>

</Division>
<Division number="212" book="sn" bookNo="18" page="309">

<Paragraph number="243" type="verses" book="sn">
_Aññāya sabbāni nivesanāni,_  
_Anikāmayaṃ aññatarampi tesaṃ;_  
_Sa ve munī vītagedho agiddho,_  
_Nāyūhatī pāragato hi hoti. (4)_  

</Paragraph>

</Division>
<Division number="213" book="sn" bookNo="18" page="309">

<Paragraph number="244" type="verses" book="sn">
_Sabbābhibhuṃ sabbaviduṃ sumedhaṃ,_  
_Sabbesu dhammesu anūpalittaṃ;_  
_Sabbañjahaṃ taṇhakkhaye vimuttaṃ,_  
_Taṃ vāpi dhīrā muni vedayanti. (5)_  

</Paragraph>

</Division>
<Division number="214" book="sn" bookNo="18" page="309">

<Paragraph number="245" type="verses" book="sn">
_Paññābalaṃ sīlavatūpapannaṃ,_  
_Samāhitaṃ jhānarataṃ satīmaṃ;_  
_Saṅgā pamuttaṃ akhilaṃ anāsavaṃ,_  
_Taṃ vāpi dhīrā muni vedayanti. (6)_  

</Paragraph>

</Division>
<Division number="215" book="sn" bookNo="18" page="310">

<Paragraph number="246" type="verses" book="sn">
_Ekaṃ carantaṃ munimappamattaṃ,_  
_Nindāpasaṃsāsu avedhamānaṃ;_  
_Sīhaṃva saddesu asantasantaṃ,_  
_Vātaṃva jālamhi asajjamānaṃ;_  
_Padmaṃva toyena alippamānaṃ,_  
_Netāramaññesamanaññaneyyaṃ;_  
_Taṃ vāpi dhīrā muni vedayanti. (7)_  

</Paragraph>

</Division>
<Division number="216" book="sn" bookNo="18" page="310">

<Paragraph number="247" type="verses" book="sn">
_Yo ogahaṇe thambhorivābhijāyati,_  
_Yasmiṃ pare vācāpariyantaṃ vadanti;_  
_Taṃ vītarāgaṃ susamāhitindriyaṃ,_  
_Taṃ vāpi dhīrā muni vedayanti. (8)_  

</Paragraph>

</Division>
<Division number="217" book="sn" bookNo="18" page="310">

<Paragraph number="248" type="verses" book="sn">
_Yo ve ṭhitatto tasaraṃva ujju,_  
_Jigucchati kammehi pāpakehi;_  
_Vīmaṃsamāno visamaṃ samañca,_  
_Taṃ vāpi dhīrā muni vedayanti. (9)_  

</Paragraph>

</Division>
<Division number="218" book="sn" bookNo="18" page="310">

<Paragraph number="249" type="verses" book="sn">
_Yo saññatatto na karoti pāpaṃ,_  
_Daharo majjhimo ca muni yatatto;_  
_Arosaneyyo na so roseti kañci,_  
_Taṃ vāpi dhīrā muni vedayanti. (10)_  

</Paragraph>

</Division>
<Division number="219" book="sn" bookNo="18" page="310">

<Paragraph number="250" type="verses" book="sn">
_Yadaggato majjhato sesato vā,_  
_Piṇḍaṃ labhetha paradattūpajīvī;_  
_Nālaṃ thutuṃ nopi nipaccavādī,_  
_Taṃ vāpi dhīrā muni vedayanti. (11)_  

</Paragraph>

</Division>
<Division number="220" book="sn" bookNo="18" page="310">

<Paragraph number="251" type="verses" book="sn">
_Muniṃ carantaṃ virataṃ methunasmā,_  
_Yo yobbane nopanibajjhate kvaci;_  
_Madappamādā virataṃ vippamuttaṃ,_  
_Taṃ vāpi dhīrā muni vedayanti. (12)_  

</Paragraph>

</Division>
<Division number="221" book="sn" bookNo="18" page="311">

<Paragraph number="252" type="verses" book="sn">
_Aññāya lokaṃ paramatthadassiṃ,_  
_Oghaṃ samuddaṃ atitariya tādiṃ;_  
_Taṃ chinnaganthaṃ asitaṃ anāsavaṃ,_  
_Taṃ vāpi dhīrā muni vedayanti. (13)_  

</Paragraph>

</Division>
<Division number="222" book="sn" bookNo="18" page="311">

<Paragraph number="253" type="verses" book="sn">
_Asamā ubho dūravihāravuttino,_  
_Gihī dāraposī amamo ca subbato;_  
_Parapāṇarodhāya gihī asaññato,_  
_Niccaṃ munī rakkhati pāṇine yato. (14)_  

</Paragraph>

</Division>
<Division number="223" book="sn" bookNo="18" page="311">

<Paragraph number="254" type="verses" book="sn">
_Sikhī yathā nīlagīvo vihaṅgamo,_  
_Haṃsassa nopeti javaṃ kudācanaṃ;_  
_Evaṃ gihī nānukaroti bhikkhuno,_  
_Munino vivittassa vanamhi jhāyatoti. (15)_  


</Paragraph>
</Division>
