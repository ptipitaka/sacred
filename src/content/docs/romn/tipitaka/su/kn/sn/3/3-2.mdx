<Paragraph number="493" type="verses" book="sn">
_“Taṃ maṃ padhānapahitattaṃ,_  
_nadiṃ nerañjaraṃ pati;_  
_Viparakkamma jhāyantaṃ,_  
_yogakkhemassa pattiyā. (1)_  

</Paragraph>

</Division>
<Division number="429" book="sn" bookNo="18" page="341">

<Paragraph number="494" type="verses" book="sn">
_Namucī karuṇaṃ vācaṃ,_  
_bhāsamāno upāgami;_  
_‘Kiso tvamasi dubbaṇṇo,_  
_santike maraṇaṃ tava. (2)_  

</Paragraph>

</Division>
<Division number="430" book="sn" bookNo="18" page="342">

<Paragraph number="495" type="verses" book="sn">
_Sahassabhāgo maraṇassa,_  
_ekaṃso tava jīvitaṃ;_  
_Jīva bho jīvitaṃ seyyo,_  
_jīvaṃ puññāni kāhasi. (3)_  

</Paragraph>

</Division>
<Division number="431" book="sn" bookNo="18" page="342">

<Paragraph number="496" type="verses" book="sn">
_Carato ca te brahmacariyaṃ,_  
_Aggihuttañca jūhato;_  
_Pahūtaṃ cīyate puññaṃ,_  
_Kiṃ padhānena kāhasi. (4)_  

</Paragraph>

</Division>
<Division number="432" book="sn" bookNo="18" page="342">

<Paragraph number="497" type="verses" book="sn">
_Duggo maggo padhānāya,_  
_dukkaro durabhisambhavo’”;_  
_Imā gāthā bhaṇaṃ māro,_  
_aṭṭhā buddhassa santike. (5)_  

</Paragraph>

</Division>
<Division number="433" book="sn" bookNo="18" page="342">

<Paragraph number="498" type="verses" book="sn">
_Taṃ tathāvādinaṃ māraṃ,_  
_bhagavā etadabravi;_  
_“Pamattabandhu pāpima,_  
_yenatthena idhāgato. (6)_  

</Paragraph>

</Division>
<Division number="434" book="sn" bookNo="18" page="342">

<Paragraph number="499" type="verses" book="sn">
_Aṇumattopi puññena,_  
_Attho mayhaṃ na vijjati;_  
_Yesañca attho puññena,_  
_Te māro vattumarahati. (7)_  

</Paragraph>

</Division>
<Division number="435" book="sn" bookNo="18" page="342">

<Paragraph number="500" type="verses" book="sn">
_Atthi saddhā tathā viriyaṃ,_  
_paññā ca mama vijjati;_  
_Evaṃ maṃ pahitattampi,_  
_kiṃ jīvamanupucchasi. (8)_  

</Paragraph>

</Division>
<Division number="436" book="sn" bookNo="18" page="342">

<Paragraph number="501" type="verses" book="sn">
_Nadīnamapi sotāni,_  
_ayaṃ vāto visosaye;_  
_Kiñca me pahitattassa,_  
_lohitaṃ nupasussaye. (9)_  

</Paragraph>

</Division>
<Division number="437" book="sn" bookNo="18" page="342">

<Paragraph number="502" type="verses" book="sn">
_Lohite sussamānamhi,_  
_Pittaṃ semhañca sussati;_  
_Maṃsesu khīyamānesu,_  
_Bhiyyo cittaṃ pasīdati;_  
_Bhiyyo sati ca paññā ca,_  
_Samādhi mama tiṭṭhati. (10)_  

</Paragraph>

</Division>
<Division number="438" book="sn" bookNo="18" page="342">

<Paragraph number="503" type="verses" book="sn">
_Tassa mevaṃ viharato,_  
_Pattassuttamavedanaṃ;_  
_Kāmesu nāpekkhate cittaṃ,_  
_Passa sattassa suddhataṃ. (11)_  

</Paragraph>

</Division>
<Division number="439" book="sn" bookNo="18" page="342">

<Paragraph number="504" type="verses" book="sn">
_Kāmā te paṭhamā senā,_  
_Dutiyā arati vuccati;_  
_Tatiyā khuppipāsā te,_  
_Catutthī taṇhā pavuccati. (12)_  

</Paragraph>

</Division>
<Division number="440" book="sn" bookNo="18" page="342">

<Paragraph number="505" type="verses" book="sn">
_Pañcamaṃ thinamiddhaṃ te,_  
_Chaṭṭhā bhīrū pavuccati;_  
_Sattamī vicikicchā te,_  
_Makkho thambho te aṭṭhamo. (13)_  

</Paragraph>

</Division>
<Division number="441" book="sn" bookNo="18" page="342">

<Paragraph number="506" type="verses" book="sn">
_Lābho siloko sakkāro,_  
_Micchāladdho ca yo yaso;_  
_Yo cattānaṃ samukkaṃse,_  
_Pare ca avajānati. (14)_  

</Paragraph>

</Division>
<Division number="442" book="sn" bookNo="18" page="343">

<Paragraph number="507" type="verses" book="sn">
_Esā namuci te senā,_  
_Kaṇhassābhippahārinī;_  
_Na naṃ asūro jināti,_  
_Jetvā ca labhate sukhaṃ. (15)_  

</Paragraph>

</Division>
<Division number="443" book="sn" bookNo="18" page="343">

<Paragraph number="508" type="verses" book="sn">
_Esa muñjaṃ parihare,_  
_Dhiratthu mama jīvitaṃ;_  
_Saṅgāme me mataṃ seyyo,_  
_Yañce jīve parājito. (16)_  

</Paragraph>

</Division>
<Division number="444" book="sn" bookNo="18" page="343">

<Paragraph number="509" type="verses" book="sn">
_Pagāḷhettha na dissanti,_  
_Eke samaṇabrāhmaṇā;_  
_Tañca maggaṃ na jānanti,_  
_Yena gacchanti subbatā. (17)_  

</Paragraph>

</Division>
<Division number="445" book="sn" bookNo="18" page="343">

<Paragraph number="510" type="verses" book="sn">
_Samantā dhajiniṃ disvā,_  
_Yuttaṃ māraṃ savāhanaṃ;_  
_Yuddhāya paccuggacchāmi,_  
_Mā maṃ ṭhānā acāvayi. (18)_  

</Paragraph>

</Division>
<Division number="446" book="sn" bookNo="18" page="343">

<Paragraph number="511" type="verses" book="sn">
_Yaṃ te taṃ nappasahati,_  
_Senaṃ loko sadevako;_  
_Taṃ te paññāya bhecchāmi,_  
_Āmaṃ pattaṃva asmanā. (19)_  

</Paragraph>

</Division>
<Division number="447" book="sn" bookNo="18" page="343">

<Paragraph number="512" type="verses" book="sn">
_Vasīkaritvā saṅkappaṃ,_  
_Satiñca sūpatiṭṭhitaṃ;_  
_Raṭṭhā raṭṭhaṃ vicarissaṃ,_  
_Sāvake vinayaṃ puthū. (20)_  

</Paragraph>

</Division>
<Division number="448" book="sn" bookNo="18" page="343">

<Paragraph number="513" type="verses" book="sn">
_Te appamattā pahitattā,_  
_Mama sāsanakārakā;_  
_Akāmassa te gamissanti,_  
_Yattha gantvā na socare”. (21)_  

</Paragraph>

</Division>
<Division number="449" book="sn" bookNo="18" page="343">

<Paragraph number="514" type="verses" book="sn">
_“Satta vassāni bhagavantaṃ,_  
_Anubandhiṃ padāpadaṃ;_  
_Otāraṃ nādhigacchissaṃ,_  
_Sambuddhassa satīmato. (22)_  

</Paragraph>

</Division>
<Division number="450" book="sn" bookNo="18" page="343">

<Paragraph number="515" type="verses" book="sn">
_Medavaṇṇaṃva pāsāṇaṃ,_  
_Vāyaso anupariyagā;_  
_Apettha muduṃ vindema,_  
_Api assādanā siyā. (23)_  

</Paragraph>

</Division>
<Division number="451" book="sn" bookNo="18" page="343">

<Paragraph number="516" type="verses" book="sn">
_Aladdhā tattha assādaṃ,_  
_Vāyasetto apakkami;_  
_Kākova selamāsajja,_  
_Nibbijjāpema gotamaṃ”. (24)_  

</Paragraph>

</Division>
<Division number="452" book="sn" bookNo="18" page="343">

<Paragraph number="517" type="verses" book="sn">
_Tassa sokaparetassa,_  
_Vīṇā kacchā abhassatha;_  
_Tato so dummano yakkho,_  
_Tatthevantaradhāyathāti. (25)_  


</Paragraph>
</Division>

<Paragraph number="518" type="center" book="sn">
Padhānasuttaṃ dutiyaṃ.
</Paragraph>
