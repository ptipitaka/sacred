<Paragraph number="38" type="verses" book="sn">
_“Sabbesu bhūtesu nidhāya daṇḍaṃ,_  
_Aviheṭhayaṃ aññatarampi tesaṃ;_  
_Na puttamiccheyya kuto sahāyaṃ,_  
_Eko care khaggavisāṇakappo”. (1)_  

</Paragraph>

</Division>
<Division number="36" book="sn" bookNo="18" page="285">

<Paragraph number="39" type="verses" book="sn">
_“Saṃsaggajātassa bhavanti snehā,_  
_Snehanvayaṃ dukkhamidaṃ pahoti;_  
_Ādīnavaṃ snehajaṃ pekkhamāno,_  
_Eko care khaggavisāṇakappo”. (2)_  

</Paragraph>

</Division>
<Division number="37" book="sn" bookNo="18" page="285">

<Paragraph number="40" type="verses" book="sn">
_“Mitte suhajje anukampamāno,_  
_Hāpeti atthaṃ paṭibaddhacitto;_  
_Etaṃ bhayaṃ santhave pekkhamāno,_  
_Eko care khaggavisāṇakappo”. (3)_  

</Paragraph>

</Division>
<Division number="38" book="sn" bookNo="18" page="285">

<Paragraph number="41" type="verses" book="sn">
_“Vaṃso visālova yathā visatto,_  
_Puttesu dāresu ca yā apekkhā;_  
_Vaṃsakkaḷīrova asajjamāno,_  
_Eko care khaggavisāṇakappo”. (4)_  

</Paragraph>

</Division>
<Division number="39" book="sn" bookNo="18" page="285">

<Paragraph number="42" type="verses" book="sn">
_“Migo araññamhi yathā abaddho,_  
_Yenicchakaṃ gacchati gocarāya;_  
_Viññū naro seritaṃ pekkhamāno,_  
_Eko care khaggavisāṇakappo”. (5)_  

</Paragraph>

</Division>
<Division number="40" book="sn" bookNo="18" page="285">

<Paragraph number="43" type="verses" book="sn">
_“Āmantanā hoti sahāyamajjhe,_  
_Vāse ṭhāne gamane cārikāya;_  
_Anabhijjhitaṃ seritaṃ pekkhamāno,_  
_Eko care khaggavisāṇakappo”. (6)_  

</Paragraph>

</Division>
<Division number="41" book="sn" bookNo="18" page="285">

<Paragraph number="44" type="verses" book="sn">
_“Khiḍḍā ratī hoti sahāyamajjhe,_  
_Puttesu ca vipulaṃ hoti pemaṃ;_  
_Piyavippayogaṃ vijigucchamāno,_  
_Eko care khaggavisāṇakappo”. (7)_  

</Paragraph>

</Division>
<Division number="42" book="sn" bookNo="18" page="286">

<Paragraph number="45" type="verses" book="sn">
_“Cātuddiso appaṭigho ca hoti,_  
_Santussamāno itarītarena;_  
_Parissayānaṃ sahitā achambhī,_  
_Eko care khaggavisāṇakappo”. (8)_  

</Paragraph>

</Division>
<Division number="43" book="sn" bookNo="18" page="286">

<Paragraph number="46" type="verses" book="sn">
_“Dussaṅgahā pabbajitāpi eke,_  
_Atho gahaṭṭhā gharamāvasantā;_  
_Appossukko paraputtesu hutvā,_  
_Eko care khaggavisāṇakappo”. (9)_  

</Paragraph>

</Division>
<Division number="44" book="sn" bookNo="18" page="286">

<Paragraph number="47" type="verses" book="sn">
_“Oropayitvā gihibyañjanāni,_  
_Sañchinnapatto yathā koviḷāro;_  
_Chetvāna vīro gihibandhanāni,_  
_Eko care khaggavisāṇakappo”. (10)_  

</Paragraph>

</Division>
<Division number="45" book="sn" bookNo="18" page="286">

<Paragraph number="48" type="verses" book="sn">
_“Sace labhetha nipakaṃ sahāyaṃ,_  
_Saddhiṃ caraṃ sādhuvihāridhīraṃ;_  
_Abhibhuyya sabbāni parissayāni,_  
_Careyya tenattamano satīmā”. (11)_  

</Paragraph>

</Division>
<Division number="46" book="sn" bookNo="18" page="286">

<Paragraph number="49" type="verses" book="sn">
_“No ce labhetha nipakaṃ sahāyaṃ,_  
_Saddhiṃ caraṃ sādhuvihāridhīraṃ;_  
_Rājāva raṭṭhaṃ vijitaṃ pahāya,_  
_Eko care mātaṅgaraññeva nāgo”. (12)_  

</Paragraph>

</Division>
<Division number="47" book="sn" bookNo="18" page="286">

<Paragraph number="50" type="verses" book="sn">
_“Addhā pasaṃsāma sahāyasampadaṃ,_  
_Seṭṭhā samā sevitabbā sahāyā;_  
_Ete aladdhā anavajjabhojī,_  
_Eko care khaggavisāṇakappo”. (13)_  

</Paragraph>

</Division>
<Division number="48" book="sn" bookNo="18" page="286">

<Paragraph number="51" type="verses" book="sn">
_“Disvā suvaṇṇassa pabhassarāni,_  
_Kammāraputtena suniṭṭhitāni;_  
_Saṅghaṭṭamānāni duve bhujasmiṃ,_  
_Eko care khaggavisāṇakappo”. (14)_  

</Paragraph>

</Division>
<Division number="49" book="sn" bookNo="18" page="287">

<Paragraph number="52" type="verses" book="sn">
_“Evaṃ dutīyena sahā mamassa,_  
_Vācābhilāpo abhisajjanā vā;_  
_Etaṃ bhayaṃ āyatiṃ pekkhamāno,_  
_Eko care khaggavisāṇakappo”. (15)_  

</Paragraph>

</Division>
<Division number="50" book="sn" bookNo="18" page="287">

<Paragraph number="53" type="verses" book="sn">
_“Kāmā hi citrā madhurā manoramā,_  
_Virūparūpena mathenti cittaṃ;_  
_Ādīnavaṃ kāmaguṇesu disvā,_  
_Eko care khaggavisāṇakappo”. (16)_  

</Paragraph>

</Division>
<Division number="51" book="sn" bookNo="18" page="287">

<Paragraph number="54" type="verses" book="sn">
_“Ītī ca gaṇḍo ca upaddavo ca,_  
_Rogo ca sallañca bhayañca metaṃ;_  
_Etaṃ bhayaṃ kāmaguṇesu disvā,_  
_Eko care khaggavisāṇakappo”. (17)_  

</Paragraph>

</Division>
<Division number="52" book="sn" bookNo="18" page="287">

<Paragraph number="55" type="verses" book="sn">
_“Sītañca uṇhañca khudaṃ pipāsaṃ,_  
_Vātātape ḍaṃsasarīsape ca;_  
_Sabbānipetāni abhisambhavitvā,_  
_Eko care khaggavisāṇakappo”. (18)_  

</Paragraph>

</Division>
<Division number="53" book="sn" bookNo="18" page="287">

<Paragraph number="56" type="verses" book="sn">
_“Nāgova yūthāni vivajjayitvā,_  
_Sañjātakhandho padumī uḷāro;_  
_Yathābhirantaṃ viharaṃ araññe,_  
_Eko care khaggavisāṇakappo”. (19)_  

</Paragraph>

</Division>
<Division number="54" book="sn" bookNo="18" page="287">

<Paragraph number="57" type="verses" book="sn">
_“Aṭṭhānataṃ saṅgaṇikāratassa,_  
_Yaṃ phassaye sāmayikaṃ vimuttiṃ;_  
_Ādiccabandhussa vaco nisamma,_  
_Eko care khaggavisāṇakappo”. (20)_  

</Paragraph>

</Division>
<Division number="55" book="sn" bookNo="18" page="287">

<Paragraph number="58" type="verses" book="sn">
_“Diṭṭhīvisūkāni upātivatto,_  
_Patto niyāmaṃ paṭiladdhamaggo;_  
_Uppannañāṇomhi anaññaneyyo,_  
_Eko care khaggavisāṇakappo”. (21)_  

</Paragraph>

</Division>
<Division number="56" book="sn" bookNo="18" page="288">

<Paragraph number="59" type="verses" book="sn">
_“Nillolupo nikkuho nippipāso,_  
_Nimmakkho niddhantakasāvamoho;_  
_Nirāsayo sabbaloke bhavitvā,_  
_Eko care khaggavisāṇakappo”. (22)_  

</Paragraph>

</Division>
<Division number="57" book="sn" bookNo="18" page="288">

<Paragraph number="60" type="verses" book="sn">
_“Pāpaṃ sahāyaṃ parivajjayetha,_  
_Anatthadassiṃ visame niviṭṭhaṃ;_  
_Sayaṃ na seve pasutaṃ pamattaṃ,_  
_Eko care khaggavisāṇakappo”. (23)_  

</Paragraph>

</Division>
<Division number="58" book="sn" bookNo="18" page="288">

<Paragraph number="61" type="verses" book="sn">
_“Bahussutaṃ dhammadharaṃ bhajetha,_  
_Mittaṃ uḷāraṃ paṭibhānavantaṃ;_  
_Aññāya atthāni vineyya kaṅkhaṃ,_  
_Eko care khaggavisāṇakappo”. (24)_  

</Paragraph>

</Division>
<Division number="59" book="sn" bookNo="18" page="288">

<Paragraph number="62" type="verses" book="sn">
_“Khiḍḍaṃ ratiṃ kāmasukhañca loke,_  
_Analaṅkaritvā anapekkhamāno;_  
_Vibhūsanaṭṭhānā virato saccavādī,_  
_Eko care khaggavisāṇakappo”. (25)_  

</Paragraph>

</Division>
<Division number="60" book="sn" bookNo="18" page="288">

<Paragraph number="63" type="verses" book="sn">
_“Puttañca dāraṃ pitarañca mātaraṃ,_  
_Dhanāni dhaññāni ca bandhavāni;_  
_Hitvāna kāmāni yathodhikāni,_  
_Eko care khaggavisāṇakappo”. (26)_  

</Paragraph>

</Division>
<Division number="61" book="sn" bookNo="18" page="288">

<Paragraph number="64" type="verses" book="sn">
_“Saṅgo eso parittamettha sokhyaṃ,_  
_Appassādo dukkhamettha bhiyyo;_  
_Gaḷo eso iti ñatvā matīmā,_  
_Eko care khaggavisāṇakappo”. (27)_  

</Paragraph>

</Division>
<Division number="62" book="sn" bookNo="18" page="288">

<Paragraph number="65" type="verses" book="sn">
_“Sandālayitvāna saṃyojanāni,_  
_Jālaṃva bhetvā salilambucārī;_  
_Aggīva daḍḍhaṃ anivattamāno,_  
_Eko care khaggavisāṇakappo”. (28)_  

</Paragraph>

</Division>
<Division number="63" book="sn" bookNo="18" page="289">

<Paragraph number="66" type="verses" book="sn">
_“Okkhittacakkhu na ca pādalolo,_  
_Guttindriyo rakkhitamānasāno;_  
_Anavassuto apariḍayhamāno,_  
_Eko care khaggavisāṇakappo”. (29)_  

</Paragraph>

</Division>
<Division number="64" book="sn" bookNo="18" page="289">

<Paragraph number="67" type="verses" book="sn">
_“Ohārayitvā gihibyañjanāni,_  
_Sañchannapatto yathā pārichatto;_  
_Kāsāyavattho abhinikkhamitvā,_  
_Eko care khaggavisāṇakappo”. (30)_  

</Paragraph>

</Division>
<Division number="65" book="sn" bookNo="18" page="289">

<Paragraph number="68" type="verses" book="sn">
_“Rasesu gedhaṃ akaraṃ alolo,_  
_Anaññaposī sapadānacārī;_  
_Kule kule appaṭibaddhacitto,_  
_Eko care khaggavisāṇakappo”. (31)_  

</Paragraph>

</Division>
<Division number="66" book="sn" bookNo="18" page="289">

<Paragraph number="69" type="verses" book="sn">
_“Pahāya pañcāvaraṇāni cetaso,_  
_Upakkilese byapanujja sabbe;_  
_Anissito chetva sinehadosaṃ,_  
_Eko care khaggavisāṇakappo”. (32)_  

</Paragraph>

</Division>
<Division number="67" book="sn" bookNo="18" page="289">

<Paragraph number="70" type="verses" book="sn">
_“Vipiṭṭhikatvāna sukhaṃ dukhañca,_  
_Pubbeva ca somanassadomanassaṃ;_  
_Laddhānupekkhaṃ samathaṃ visuddhaṃ,_  
_Eko care khaggavisāṇakappo”. (33)_  

</Paragraph>

</Division>
<Division number="68" book="sn" bookNo="18" page="289">

<Paragraph number="71" type="verses" book="sn">
_“Āraddhaviriyo paramatthapattiyā,_  
_Alīnacitto akusītavutti;_  
_Daḷhanikkamo thāmabalūpapanno,_  
_Eko care khaggavisāṇakappo”. (34)_  

</Paragraph>

</Division>
<Division number="69" book="sn" bookNo="18" page="289">

<Paragraph number="72" type="verses" book="sn">
_“Paṭisallānaṃ jhānamariñcamāno,_  
_Dhammesu niccaṃ anudhammacārī;_  
_Ādīnavaṃ sammasitā bhavesu,_  
_Eko care khaggavisāṇakappo”. (35)_  

</Paragraph>

</Division>
<Division number="70" book="sn" bookNo="18" page="290">

<Paragraph number="73" type="verses" book="sn">
_“Taṇhakkhayaṃ patthayamappamatto,_  
_Aneḷamūgo sutavā satīmā;_  
_Saṅkhātadhammo niyato padhānavā,_  
_Eko care khaggavisāṇakappo”. (36)_  

</Paragraph>

</Division>
<Division number="71" book="sn" bookNo="18" page="290">

<Paragraph number="74" type="verses" book="sn">
_“Sīhova saddesu asantasanto,_  
_Vātova jālamhi asajjamāno;_  
_Padumaṃva toyena alippamāno”,_  
_Eko care khaggavisāṇakappo. (37)_  

</Paragraph>

</Division>
<Division number="72" book="sn" bookNo="18" page="290">

<Paragraph number="75" type="verses" book="sn">
_“Sīho yathā dāṭhabalī pasayha,_  
_Rājā migānaṃ abhibhuyya cārī;_  
_Sevetha pantāni senāsanāni,_  
_Eko care khaggavisāṇakappo”. (38)_  

</Paragraph>

</Division>
<Division number="73" book="sn" bookNo="18" page="290">

<Paragraph number="76" type="verses" book="sn">
_“Mettaṃ upekkhaṃ karuṇaṃ vimuttiṃ,_  
_Āsevamāno muditañca kāle;_  
_Sabbena lokena avirujjhamāno,_  
_Eko care khaggavisāṇakappo”. (39)_  

</Paragraph>

</Division>
<Division number="74" book="sn" bookNo="18" page="290">

<Paragraph number="77" type="verses" book="sn">
_“Rāgañca dosañca pahāya mohaṃ,_  
_Sandālayitvāna saṃyojanāni;_  
_Asantasaṃ jīvitasaṅkhayamhi,_  
_Eko care khaggavisāṇakappo”. (40)_  

</Paragraph>

</Division>
<Division number="75" book="sn" bookNo="18" page="290">

<Paragraph number="78" type="verses" book="sn">
_“Bhajanti sevanti ca kāraṇatthā,_  
_Nikkāraṇā dullabhā ajja mittā;_  
_Attaṭṭhapaññā asucī manussā,_  
_Eko care khaggavisāṇakappo”. (41)_  


</Paragraph>
