<Paragraph number="941" type="verses" book="sn">
_Satto guhāyaṃ bahunābhichanno,_  
_Tiṭṭhaṃ naro mohanasmiṃ pagāḷho;_  
_Dūre vivekā hi tathāvidho so,_  
_Kāmā hi loke na hi suppahāyā. (1)_  

</Paragraph>

</Division>
<Division number="780" book="sn" bookNo="18" page="400">

<Paragraph number="942" type="verses" book="sn">
_Icchānidānā bhavasātabaddhā,_  
_Te duppamuñcā na hi aññamokkhā;_  
_Pacchā pure vāpi apekkhamānā,_  
_Ime va kāme purime va jappaṃ. (2)_  

</Paragraph>

</Division>
<Division number="781" book="sn" bookNo="18" page="400">

<Paragraph number="943" type="verses" book="sn">
_Kāmesu giddhā pasutā pamūḷhā,_  
_Avadāniyā te visame niviṭṭhā;_  
_Dukkhūpanītā paridevayanti,_  
_Kiṃsū bhavissāma ito cutāse. (3)_  

</Paragraph>

</Division>
<Division number="782" book="sn" bookNo="18" page="400">

<Paragraph number="944" type="verses" book="sn">
_Tasmā hi sikkhetha idheva jantu,_  
_Yaṃ kiñci jaññā visamanti loke;_  
_Na tassa hetū visamaṃ careyya,_  
_Appañhidaṃ jīvitamāhu dhīrā. (4)_  

</Paragraph>

</Division>
<Division number="783" book="sn" bookNo="18" page="400">

<Paragraph number="945" type="verses" book="sn">
_Passāmi loke pariphandamānaṃ,_  
_Pajaṃ imaṃ taṇhagataṃ bhavesu;_  
_Hīnā narā maccumukhe lapanti,_  
_Avītataṇhāse bhavābhavesu. (5)_  

</Paragraph>

</Division>
<Division number="784" book="sn" bookNo="18" page="400">

<Paragraph number="946" type="verses" book="sn">
_Mamāyite passatha phandamāne,_  
_Maccheva appodake khīṇasote;_  
_Etampi disvā amamo careyya,_  
_Bhavesu āsattimakubbamāno. (6)_  

</Paragraph>

</Division>
<Division number="785" book="sn" bookNo="18" page="400">

<Paragraph number="947" type="verses" book="sn">
_Ubhosu antesu vineyya chandaṃ,_  
_Phassaṃ pariññāya anānugiddho;_  
_Yadattagarahī tadakubbamāno,_  
_Na lippatī diṭṭhasutesu dhīro. (7)_  

</Paragraph>

</Division>
<Division number="786" book="sn" bookNo="18" page="400">

<Paragraph number="948" type="verses" book="sn">
_Saññaṃ pariññā vitareyya oghaṃ,_  
_Pariggahesu muni nopalitto;_  
_Abbūḷhasallo caramappamatto,_  
_Nāsīsatī lokamimaṃ parañcāti. (8)_  


</Paragraph>
</Division>

<Paragraph number="949" type="center" book="sn">
Guhaṭṭhakasuttaṃ dutiyaṃ.
</Paragraph>
