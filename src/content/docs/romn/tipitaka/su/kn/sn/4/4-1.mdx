<Paragraph number="934" type="verses" book="sn">
_Kāmaṃ kāmayamānassa,_  
_tassa ce taṃ samijjhati;_  
_Addhā pītimano hoti,_  
_laddhā macco yadicchati. (1)_  

</Paragraph>

</Division>
<Division number="774" book="sn" bookNo="18" page="399">

<Paragraph number="935" type="verses" book="sn">
_Tassa ce kāmayānassa,_  
_chandajātassa jantuno;_  
_Te kāmā parihāyanti,_  
_sallaviddhova ruppati. (2)_  

</Paragraph>

</Division>
<Division number="775" book="sn" bookNo="18" page="399">

<Paragraph number="936" type="verses" book="sn">
_Yo kāme parivajjeti,_  
_sappasseva padā siro;_  
_Somaṃ visattikaṃ loke,_  
_sato samativattati. (3)_  

</Paragraph>

</Division>
<Division number="776" book="sn" bookNo="18" page="399">

<Paragraph number="937" type="verses" book="sn">
_Khettaṃ vatthuṃ hiraññaṃ vā,_  
_gavāssaṃ dāsaporisaṃ;_  
_Thiyo bandhū puthu kāme,_  
_yo naro anugijjhati. (4)_  

</Paragraph>

</Division>
<Division number="777" book="sn" bookNo="18" page="399">

<Paragraph number="938" type="verses" book="sn">
_Abalā naṃ balīyanti,_  
_maddantenaṃ parissayā;_  
_Tato naṃ dukkhamanveti,_  
_nāvaṃ bhinnamivodakaṃ. (5)_  

</Paragraph>

</Division>
<Division number="778" book="sn" bookNo="18" page="399">

<Paragraph number="939" type="verses" book="sn">
_Tasmā jantu sadā sato,_  
_Kāmāni parivajjaye;_  
_Te pahāya tare oghaṃ,_  
_Nāvaṃ sitvāva pāragūti. (6)_  


</Paragraph>
</Division>

<Paragraph number="940" type="center" book="sn">
Kāmasuttaṃ paṭhamaṃ.
</Paragraph>
