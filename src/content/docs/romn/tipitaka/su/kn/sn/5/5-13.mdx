<Division number="1112" book="sn">

<Paragraph number="1310" type="verses" book="sn">
_“Jhāyiṃ virajamāsīnaṃ, __(iccāyasmā udayo)___  
_Katakiccaṃ anāsavaṃ;_  
_Pāraguṃ sabbadhammānaṃ,_  
_Atthi pañhena āgamaṃ;_  
_Aññāvimokkhaṃ pabrūhi,_  
_Avijjāya pabhedanaṃ”. (1)_  

</Paragraph>

</Division>
<Division number="1113" book="sn">

<Paragraph number="1311" type="verses" book="sn">
_“Pahānaṃ kāmacchandānaṃ, __(udayāti bhagavā)___  
_Domanassāna cūbhayaṃ;_  
_Thinassa ca panūdanaṃ,_  
_Kukkuccānaṃ nivāraṇaṃ. (2)_  

</Paragraph>

</Division>
<Division number="1114" book="sn">

<Paragraph number="1312" type="verses" book="sn">
_Upekkhāsatisaṃsuddhaṃ,_  
_dhammatakkapurejavaṃ;_  
_Aññāvimokkhaṃ pabrūmi,_  
_avijjāya pabhedanaṃ”. (3)_  

</Paragraph>

</Division>
<Division number="1115" book="sn">

<Paragraph number="1313" type="verses" book="sn">
_“Kiṃsu saṃyojano loko,_  
_kiṃsu tassa vicāraṇaṃ;_  
_Kissassa vippahānena,_  
_nibbānaṃ iti vuccati”. (4)_  

</Paragraph>

</Division>
<Division number="1116" book="sn">

<Paragraph number="1314" type="verses" book="sn">
_“Nandisaṃyojano loko,_  
_vitakkassa vicāraṇaṃ;_  
_Taṇhāya vippahānena,_  
_nibbānaṃ iti vuccati”. (5)_  

</Paragraph>

</Division>
<Division number="1117" book="sn">

<Paragraph number="1315" type="verses" book="sn">
_“Kathaṃ satassa carato,_  
_viññāṇaṃ uparujjhati;_  
_Bhagavantaṃ puṭṭhumāgamma,_  
_taṃ suṇoma vaco tava”. (6)_  

</Paragraph>

</Division>
<Division number="1118" book="sn">

<Paragraph number="1316" type="verses" book="sn">
_“Ajjhattañca bahiddhā ca,_  
_Vedanaṃ nābhinandato;_  
_Evaṃ satassa carato,_  
_Viññāṇaṃ uparujjhatī”ti. (7)_  


</Paragraph>
</Division>

<Paragraph number="1317" type="center" book="sn">
Udayamāṇavapucchā terasamā.
</Paragraph>
