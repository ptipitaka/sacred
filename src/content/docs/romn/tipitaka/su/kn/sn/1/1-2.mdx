<Division number="18" book="sn">

<Paragraph number="20" type="verses" book="sn">
_“Pakkodano duddhakhīrohamasmi, __(iti dhaniyo gopo)___  
_Anutīre mahiyā samānavāso;_  
_Channā kuṭi āhito gini,_  
_Atha ce patthayasī pavassa deva”. (1)_  

</Paragraph>

</Division>
<Division number="19" book="sn">

<Paragraph number="21" type="verses" book="sn">
_“Akkodhano vigatakhilohamasmi, __(iti bhagavā)___  
_Anutīre mahiyekarattivāso;_  
_Vivaṭā kuṭi nibbuto gini,_  
_Atha ce patthayasī pavassa deva”. (2)_  

</Paragraph>

</Division>
<Division number="20" book="sn">

<Paragraph number="22" type="verses" book="sn">
_“Andhakamakasā na vijjare, __(iti dhaniyo gopo)___  
_Kacche rūḷhatiṇe caranti gāvo;_  
_Vuṭṭhimpi saheyyumāgataṃ,_  
_Atha ce patthayasī pavassa deva”. (3)_  

</Paragraph>

</Division>
<Division number="21" book="sn">

<Paragraph number="23" type="verses" book="sn">
_“Baddhāsi bhisī susaṅkhatā, __(iti bhagavā)___  
_Tiṇṇo pāragato vineyya oghaṃ;_  
_Attho bhisiyā na vijjati,_  
_Atha ce patthayasī pavassa deva”. (4)_  

</Paragraph>

</Division>
<Division number="22" book="sn">

<Paragraph number="24" type="verses" book="sn">
_“Gopī mama assavā alolā, __(iti dhaniyo gopo)___  
_Dīgharattaṃ saṃvāsiyā manāpā;_  
_Tassā na suṇāmi kiñci pāpaṃ,_  
_Atha ce patthayasī pavassa deva”. (5)_  

</Paragraph>

</Division>
<Division number="23" book="sn">

<Paragraph number="25" type="verses" book="sn">
_“Cittaṃ mama assavaṃ vimuttaṃ, __(iti bhagavā)___  
_Dīgharattaṃ paribhāvitaṃ sudantaṃ;_  
_Pāpaṃ pana me na vijjati,_  
_Atha ce patthayasī pavassa deva”. (6)_  

</Paragraph>

</Division>
<Division number="24" book="sn">

<Paragraph number="26" type="verses" book="sn">
_“Attavetanabhatohamasmi, __(iti dhaniyo gopo)___  
_Puttā ca me samāniyā arogā;_  
_Tesaṃ na suṇāmi kiñci pāpaṃ,_  
_Atha ce patthayasī pavassa deva”. (7)_  

</Paragraph>

</Division>
<Division number="25" book="sn">

<Paragraph number="27" type="verses" book="sn">
_“Nāhaṃ bhatakosmi kassaci, __(iti bhagavā)___  
_Nibbiṭṭhena carāmi sabbaloke;_  
_Attho bhatiyā na vijjati,_  
_Atha ce patthayasī pavassa deva”. (8)_  

</Paragraph>

</Division>
<Division number="26" book="sn">

<Paragraph number="28" type="verses" book="sn">
_“Atthi vasā atthi dhenupā, __(iti dhaniyo gopo)___  
_Godharaṇiyo paveṇiyopi atthi;_  
_Usabhopi gavampatīdha atthi,_  
_Atha ce patthayasī pavassa deva”. (9)_  

</Paragraph>

</Division>
<Division number="27" book="sn">

<Paragraph number="29" type="verses" book="sn">
_“Natthi vasā natthi dhenupā, __(iti bhagavā)___  
_Godharaṇiyo paveṇiyopi natthi;_  
_Usabhopi gavampatīdha natthi,_  
_Atha ce patthayasī pavassa deva”. (10)_  

</Paragraph>

</Division>
<Division number="28" book="sn">

<Paragraph number="30" type="verses" book="sn">
_“Khilā nikhātā asampavedhī, __(iti dhaniyo gopo)___  
_Dāmā muñjamayā navā susaṇṭhānā;_  
_Na hi sakkhinti dhenupāpi chettuṃ,_  
_Atha ce patthayasī pavassa deva”. (11)_  

</Paragraph>

</Division>
<Division number="29" book="sn">

<Paragraph number="31" type="verses" book="sn">
_“Usabhoriva chetva bandhanāni, __(iti bhagavā)___  
_Nāgo pūtilataṃva dālayitvā;_  
_Nāhaṃ punupessaṃ gabbhaseyyaṃ,_  
_Atha ce patthayasī pavassa deva”. (12)_  

</Paragraph>

</Division>
<Division number="30" book="sn">

<Paragraph number="32" type="verses" book="sn">
_Ninnañca thalañca pūrayanto,_  
_Mahāmegho pavassi tāvadeva;_  
_Sutvā devassa vassato,_  
_Imamatthaṃ dhaniyo abhāsatha. (13)_  

</Paragraph>

</Division>
<Division number="31" book="sn">

<Paragraph number="33" type="verses" book="sn">
_“Lābhā vata no anappakā,_  
_Ye mayaṃ bhagavantaṃ addasāma;_  
_Saraṇaṃ taṃ upema cakkhuma,_  
_Satthā no hohi tuvaṃ mahāmuni. (14)_  

</Paragraph>

</Division>
<Division number="32" book="sn">

<Paragraph number="34" type="verses" book="sn">
_Gopī ca ahañca assavā,_  
_Brahmacariyaṃ sugate carāmase;_  
_Jātimaraṇassa pāragū,_  
_Dukkhassantakarā bhavāmase”. (15)_  

</Paragraph>

</Division>
<Division number="33" book="sn">

<Paragraph number="35" type="verses" book="sn">
_“Nandati puttehi puttimā, __(iti māro pāpimā)___  
_Gomā gohi tatheva nandati;_  
_Upadhī hi narassa nandanā,_  
_Na hi so nandati yo nirūpadhi”. (16)_  

</Paragraph>

</Division>
<Division number="34" book="sn">

<Paragraph number="36" type="verses" book="sn">
_“Socati puttehi puttimā, __(iti bhagavā)___  
_Gomā gohi tatheva socati;_  
_Upadhī hi narassa socanā,_  
_Na hi so socati yo nirūpadhī”ti. (17)_  


</Paragraph>
</Division>

<Paragraph number="37" type="center" book="sn">
Dhaniyasuttaṃ dutiyaṃ.
</Paragraph>
