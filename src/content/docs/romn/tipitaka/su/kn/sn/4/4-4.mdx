<Paragraph number="959" type="verses" book="sn">
_Passāmi suddhaṃ paramaṃ arogaṃ,_  
_Diṭṭhena saṃsuddhi narassa hoti;_  
_Evābhijānaṃ paramanti ñatvā,_  
_Suddhānupassīti pacceti ñāṇaṃ. (1)_  

</Paragraph>

</Division>
<Division number="796" book="sn" bookNo="18" page="402">

<Paragraph number="960" type="verses" book="sn">
_Diṭṭhena ce suddhi narassa hoti,_  
_Ñāṇena vā so pajahāti dukkhaṃ;_  
_Aññena so sujjhati sopadhīko,_  
_Diṭṭhī hi naṃ pāva tathā vadānaṃ. (2)_  

</Paragraph>

</Division>
<Division number="797" book="sn" bookNo="18" page="402">

<Paragraph number="961" type="verses" book="sn">
_Na brāhmaṇo aññato suddhimāha,_  
_Diṭṭhe sute sīlavate mute vā;_  
_Puññe ca pāpe ca anūpalitto,_  
_Attañjaho nayidha pakubbamāno. (3)_  

</Paragraph>

</Division>
<Division number="798" book="sn" bookNo="18" page="402">

<Paragraph number="962" type="verses" book="sn">
_Purimaṃ pahāya aparaṃ sitāse,_  
_Ejānugā te na taranti saṅgaṃ;_  
_Te uggahāyanti nirassajanti,_  
_Kapīva sākhaṃ pamuñcaṃ gahāyaṃ. (4)_  

</Paragraph>

</Division>
<Division number="799" book="sn" bookNo="18" page="403">

<Paragraph number="963" type="verses" book="sn">
_Sayaṃ samādāya vatāni jantu,_  
_Uccāvacaṃ gacchati saññasatto;_  
_Vidvā ca vedehi samecca dhammaṃ,_  
_Na uccāvacaṃ gacchati bhūripañño. (5)_  

</Paragraph>

</Division>
<Division number="800" book="sn" bookNo="18" page="403">

<Paragraph number="964" type="verses" book="sn">
_Sa sabbadhammesu visenibhūto,_  
_Yaṃ kiñci diṭṭhaṃ va sutaṃ mutaṃ vā;_  
_Tameva dassiṃ vivaṭaṃ carantaṃ,_  
_Kenīdha lokasmi vikappayeyya. (6)_  

</Paragraph>

</Division>
<Division number="801" book="sn" bookNo="18" page="403">

<Paragraph number="965" type="verses" book="sn">
_Na kappayanti na purekkharonti,_  
_Accantasuddhīti na te vadanti;_  
_Ādānaganthaṃ gathitaṃ visajja,_  
_Āsaṃ na kubbanti kuhiñci loke. (7)_  

</Paragraph>

</Division>
<Division number="802" book="sn" bookNo="18" page="403">

<Paragraph number="966" type="verses" book="sn">
_Sīmātigo brāhmaṇo tassa natthi,_  
_Ñatvā va disvā va samuggahītaṃ;_  
_Na rāgarāgī na virāgaratto,_  
_Tassīdha natthi paramuggahītanti. (8)_  


</Paragraph>
</Division>

<Paragraph number="967" type="center" book="sn">
Suddhaṭṭhakasuttaṃ catutthaṃ.
</Paragraph>
