<Paragraph number="356" type="verses" book="dh">
_Atthamhi jātamhi sukhā sahāyā,_  
_Tuṭṭhī sukhā yā itarītarena;_  
_Puññaṃ sukhaṃ jīvitasaṅkhayamhi,_  
_Sabbassa dukkhassa sukhaṃ pahānaṃ._  

</Paragraph>

</Division>
<Division number="332" book="dh" bookNo="18" page="61">

<Paragraph number="357" type="verses" book="dh">
_Sukhā matteyyatā loke,_  
_atho petteyyatā sukhā;_  
_Sukhā sāmaññatā loke,_  
_atho brahmaññatā sukhā._  

</Paragraph>

</Division>
<Division number="333" book="dh" bookNo="18" page="61">

<Paragraph number="358" type="verses" book="dh">
_Sukhaṃ yāva jarā sīlaṃ,_  
_sukhā saddhā patiṭṭhitā;_  
_Sukho paññāya paṭilābho,_  
_pāpānaṃ akaraṇaṃ sukhaṃ._  


</Paragraph>
</Division>
