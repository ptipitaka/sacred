<Division number="368" book="dh">

<Paragraph number="395" type="verses" book="dh">
_Mettāvihārī yo bhikkhu,_  
_pasanno buddhasāsane;_  
_Adhigacche padaṃ santaṃ,_  
_saṅkhārūpasamaṃ sukhaṃ._  

</Paragraph>

</Division>
<Division number="369" book="dh">

<Paragraph number="396" type="verses" book="dh">
_Siñca bhikkhu imaṃ nāvaṃ,_  
_sittā te lahumessati;_  
_Chetvā rāgañca dosañca,_  
_tato nibbānamehisi._  

</Paragraph>

</Division>
<Division number="370" book="dh">

<Paragraph number="397" type="verses" book="dh">
_Pañca chinde pañca jahe,_  
_pañca cuttari bhāvaye;_  
_Pañca saṅgātigo bhikkhu,_  
_“oghatiṇṇo”ti vuccati._  

</Paragraph>

</Division>
<Division number="371" book="dh">

<Paragraph number="398" type="verses" book="dh">
_Jhāya bhikkhu mā pamādo,_  
_Mā te kāmaguṇe ramessu cittaṃ;_  
_Mā lohaguḷaṃ gilī pamatto,_  
_Mā kandi “dukkhamidan”ti dayhamāno._  

</Paragraph>

</Division>
<Division number="372" book="dh">

<Paragraph number="399" type="verses" book="dh">
_Natthi jhānaṃ apaññassa,_  
_paññā natthi ajhāyato;_  
_Yamhi jhānañca paññā ca,_  
_sa ve nibbānasantike._  

</Paragraph>

</Division>
<Division number="373" book="dh">

<Paragraph number="400" type="verses" book="dh">
_Suññāgāraṃ paviṭṭhassa,_  
_santacittassa bhikkhuno;_  
_Amānusī rati hoti,_  
_sammā dhammaṃ vipassato._  

</Paragraph>

</Division>
<Division number="374" book="dh">

<Paragraph number="401" type="verses" book="dh">
_Yato yato sammasati,_  
_khandhānaṃ udayabbayaṃ;_  
_Labhatī pītipāmojjaṃ,_  
_amataṃ taṃ vijānataṃ._  

</Paragraph>

</Division>
<Division number="375" book="dh">

<Paragraph number="402" type="verses" book="dh">
_Tatrāyamādi bhavati,_  
_idha paññassa bhikkhuno;_  
_Indriyagutti santuṭṭhi,_  
_pātimokkhe ca saṃvaro._  

</Paragraph>

</Division>
<Division number="376" book="dh">

<Paragraph number="403" type="verses" book="dh">
_Mitte bhajassu kalyāṇe,_  
_suddhājīve atandite;_  
_Paṭisanthāravutyassa,_  
_ācārakusalo siyā;_  
_Tato pāmojjabahulo,_  
_dukkhassantaṃ karissati._
</Paragraph>
</Division>
