<Paragraph number="246" type="verses" book="dh">
_Porāṇametaṃ atula,_  
_netaṃ ajjatanāmiva;_  
_Nindanti tuṇhimāsīnaṃ,_  
_nindanti bahubhāṇinaṃ;_  
_Mitabhāṇimpi nindanti,_  
_natthi loke anindito._  

</Paragraph>

</Division>
<Division number="228" book="dh">

<Paragraph number="247" type="verses" book="dh">
_Na cāhu na ca bhavissati,_  
_na cetarahi vijjati;_  
_Ekantaṃ nindito poso,_  
_ekantaṃ vā pasaṃsito._  

</Paragraph>

</Division>
<Division number="229" book="dh">

<Paragraph number="248" type="verses" book="dh">
_Yañce viññū pasaṃsanti,_  
_anuvicca suve suve;_  
_Acchiddavuttiṃ medhāviṃ,_  
_paññāsīlasamāhitaṃ._  

</Paragraph>

</Division>
<Division number="230" book="dh">

<Paragraph number="249" type="verses" book="dh">
_Nikkhaṃ jambonadasseva,_  
_ko taṃ ninditumarahati;_  
_Devāpi naṃ pasaṃsanti,_  
_brahmunāpi pasaṃsito._
</Paragraph>
</Division>
