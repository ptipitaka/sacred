<Paragraph number="222" type="verses" book="dh">
_Sāhu dassanamariyānaṃ,_  
_sannivāso sadā sukho;_  
_Adassanena bālānaṃ,_  
_niccameva sukhī siyā._

<Division number="207" book="dh">

<Paragraph number="223" type="verses" book="dh">
_Bālasaṅgatacārī hi,_  
_dīghamaddhāna socati;_  
_Dukkho bālehi saṃvāso,_  
_amitteneva sabbadā;_  
_Dhīro ca sukhasaṃvāso,_  
_ñātīnaṃva samāgamo._  

</Paragraph>

</Division>
<Division number="208" book="dh">

<Paragraph number="224" book="dh">
Tasmā hi—
</Paragraph>

<Paragraph number="225" type="verses" book="dh">
_Dhīrañca paññañca bahussutañca,_  
_Dhorayhasīlaṃ vatavantamariyaṃ;_  
_Taṃ tādisaṃ sappurisaṃ sumedhaṃ,_  
_Bhajetha nakkhattapathaṃva candimā._  


</Paragraph>
</Division>
