<Paragraph number="353" type="verses" book="dh">
_Sace labhetha nipakaṃ sahāyaṃ,_  
_Saddhiṃ caraṃ sādhuvihāridhīraṃ;_  
_Abhibhuyya sabbāni parissayāni,_  
_Careyya tenattamano satīmā._  

</Paragraph>

<Division number="329" book="dh">

<Paragraph number="354" type="verses" book="dh">
_No ce labhetha nipakaṃ sahāyaṃ,_  
_Saddhiṃ caraṃ sādhuvihāridhīraṃ;_  
_Rājāva raṭṭhaṃ vijitaṃ pahāya,_  
_Eko care mātaṅgaraññeva nāgo._  

</Paragraph>

</Division>
<Division number="330" book="dh">

<Paragraph number="355" type="verses" book="dh">
_Ekassa caritaṃ seyyo,_  
_Natthi bāle sahāyatā;_  
_Eko care na ca pāpāni kayirā,_  
_Appossukko mātaṅgaraññeva nāgo._
</Paragraph>
</Division>
