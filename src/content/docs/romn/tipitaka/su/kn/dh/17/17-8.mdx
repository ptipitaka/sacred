<Paragraph number="250" type="verses" book="dh">
_Kāyappakopaṃ rakkheyya,_  
_kāyena saṃvuto siyā;_  
_Kāyaduccaritaṃ hitvā,_  
_kāyena sucaritaṃ care._  

</Paragraph>

</Division>
<Division number="232" book="dh">

<Paragraph number="251" type="verses" book="dh">
_Vacīpakopaṃ rakkheyya,_  
_vācāya saṃvuto siyā;_  
_Vacīduccaritaṃ hitvā,_  
_vācāya sucaritaṃ care._  

</Paragraph>

</Division>
<Division number="233" book="dh">

<Paragraph number="252" type="verses" book="dh">
_Manopakopaṃ rakkheyya,_  
_manasā saṃvuto siyā;_  
_Manoduccaritaṃ hitvā,_  
_manasā sucaritaṃ care._  

</Paragraph>

</Division>
<Division number="234" book="dh">

<Paragraph number="253" type="verses" book="dh">
_Kāyena saṃvutā dhīrā,_  
_atho vācāya saṃvutā;_  
_Manasā saṃvutā dhīrā,_  
_te ve suparisaṃvutā._  


</Paragraph>
</Division>

Kodhavaggo sattarasamo.
