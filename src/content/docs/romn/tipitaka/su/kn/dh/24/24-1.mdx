<Paragraph number="360" type="verses" book="dh">
_Manujassa pamattacārino,_  
_Taṇhā vaḍḍhati māluvā viya;_  
_So plavatī hurā huraṃ,_  
_Phalamicchaṃva vanasmi vānaro._  

</Paragraph>

</Division>
<Division number="335" book="dh" bookNo="18" page="61">

<Paragraph number="361" type="verses" book="dh">
_Yaṃ esā sahate jammī,_  
_taṇhā loke visattikā;_  
_Sokā tassa pavaḍḍhanti,_  
_abhivaṭṭhaṃva bīraṇaṃ._  

</Paragraph>

</Division>
<Division number="336" book="dh" bookNo="18" page="61">

<Paragraph number="362" type="verses" book="dh">
_Yo cetaṃ sahate jammiṃ,_  
_taṇhaṃ loke duraccayaṃ;_  
_Sokā tamhā papatanti,_  
_udabinduva pokkharā._  

</Paragraph>

</Division>
<Division number="337" book="dh" bookNo="18" page="61">

<Paragraph number="363" type="verses" book="dh">
_Taṃ vo vadāmi bhaddaṃ vo,_  
_yāvantettha samāgatā;_  
_Taṇhāya mūlaṃ khaṇatha,_  
_usīratthova bīraṇaṃ;_  
_Mā vo naḷaṃva sotova,_  
_māro bhañji punappunaṃ._
</Paragraph>
