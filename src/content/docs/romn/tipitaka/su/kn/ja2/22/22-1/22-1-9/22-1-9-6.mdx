<Division number="1472" book="ja2">

<Paragraph number="2811" type="verses" book="ja2">
_“Ethayyo rājavasatiṃ,_  
_nisīditvā suṇātha me;_  
_Yathā rājakulaṃ patto,_  
_yasaṃ poso nigacchati._  

</Paragraph>

</Division>
<Division number="1473" book="ja2">

<Paragraph number="2812" type="verses" book="ja2">

_aññāto labhate yasaṃ;_  
_Nāsūro nāpi dummedho,_  
_nappamatto kudācanaṃ._  

</Paragraph>

</Division>

_Atha vissasate tyamhi,_  
_guyhañcassa na rakkhati._  

</Paragraph>

_rañño suppaṭiyādito;_  
_Na tena vutto gaccheyya,_  
_sa rājavasatiṃ vase._  

</Paragraph>

</Division>
<Division number="1480" book="ja2">

<Paragraph number="2819" type="verses" book="ja2">

_kāmabhoge kudācanaṃ;_  
_Sabbattha pacchato gacche,_  
_sa rājavasatiṃ vase._  

</Paragraph>

</Division>
<Division number="1481" book="ja2">

<Paragraph number="2820" type="verses" book="ja2">

_na mālaṃ na vilepanaṃ;_  
_Ākappaṃ sarakuttiṃ vā,_  
_na rañño sadisamācare;_  
_Aññaṃ kareyya ākappaṃ,_  
_sa rājavasatiṃ vase._  

</Paragraph>

</Division>
<Division number="1482" book="ja2">

<Paragraph number="2821" type="verses" book="ja2">

_bhariyāhi parivārito;_  
_Nāmacco rājabhariyāsu,_  
_bhāvaṃ kubbetha paṇḍito._  

</Paragraph>

</Division>
<Division number="1483" book="ja2">

_nipako saṃvutindriyo;_  
_Manopaṇidhisampanno,_  
_sa rājavasatiṃ vase._  

</Paragraph>

</Division>
<Division number="1484" book="ja2">

_na manteyya rahogato;_  
_Nāssa kosā dhanaṃ gaṇhe,_  
_sa rājavasatiṃ vase._  

</Paragraph>

</Division>
<Division number="1485" book="ja2">

_na madāya suraṃ pive;_  
_Nāssa dāye mige haññe,_  
_sa rājavasatiṃ vase._  

</Paragraph>

</Division>
<Division number="1486" book="ja2">

_Sa rājavasatiṃ vase._  

</Paragraph>

</Division>

</Paragraph>

_rājānaṃ parisaṅgataṃ._  

</Paragraph>

_sa rājavasatiṃ vase._  

</Paragraph>

</Paragraph>

_sa rājavasatiṃ vase._  

</Paragraph>

</Paragraph>

</Paragraph>
