<Paragraph number="2770" type="verses" book="ja2">
_“Upāgataṃ rāja mupehi lakkhaṃ,_  
_Netādisaṃ maṇiratanaṃ tavatthi;_  
_Dhammena jissāma asāhasena,_  
_Jito ca no khippamavākarohi._  

</Paragraph>

</Division>
<Division number="1435" book="ja2" bookNo="23" page="285">

<Paragraph number="2771" type="verses" book="ja2">
_Pañcāla-paccuggata-sūrasena,_  
_Macchā ca maddā saha kekakebhi;_  
_Passantu note asaṭhena yuddhaṃ,_  
_Na no sabhāyaṃ na karonti kiñci”._  

</Paragraph>

</Division>
<Division number="1436" book="ja2" bookNo="23" page="285">

<Paragraph number="2772" type="verses" book="ja2">
_Te pāvisuṃ akkhamadena mattā,_  
_Rājā kurūnaṃ puṇṇako cāpi yakkho;_  
_Rājā kaliṃ viccinamaggahesi,_  
_Kaṭaṃ aggahī puṇṇako nāma yakkho._  

</Paragraph>

</Division>
<Division number="1437" book="ja2" bookNo="23" page="285">

<Paragraph number="2773" type="verses" book="ja2">
_Te tattha jūte ubhaye samāgate,_  
_Raññaṃ sakāse sakhīnañca majjhe;_  
_Ajesi yakkho naravīraseṭṭhaṃ,_  
_Tatthappanādo tumulo babhūva._  

</Paragraph>

</Division>
<Division number="1438" book="ja2" bookNo="23" page="285">

<Paragraph number="2774" type="verses" book="ja2">
_“Jayo mahārāja parājayo ca,_  
_Āyūhataṃ aññatarassa hoti;_  
_Janinda jīnosi varaddhanena,_  
_Jito ca me khippamavākarohi”._  

</Paragraph>

</Division>
<Division number="1439" book="ja2" bookNo="23" page="286">

<Paragraph number="2775" type="verses" book="ja2">
_“Hatthī gavāssā maṇikuṇḍalā ca,_  
_Yañcāpi mayhaṃ ratanaṃ pathabyā;_  
_Gaṇhāhi kaccāna varaṃ dhanānaṃ,_  
_Ādāya yenicchasi tena gaccha”._  

</Paragraph>

</Division>
<Division number="1440" book="ja2" bookNo="23" page="286">

<Paragraph number="2776" type="verses" book="ja2">
_“Hatthī gavāssā maṇikuṇḍalā ca,_  
_Yañcāpi tuyhaṃ ratanaṃ pathabyā;_  
_Tesaṃ varo vidhuro nāma kattā,_  
_So me jito taṃ me avākarohi”._  

</Paragraph>

</Division>
<Division number="1441" book="ja2" bookNo="23" page="286">

<Paragraph number="2777" type="verses" book="ja2">
_“Attā ca me so saraṇaṃ gatī ca,_  
_Dīpo ca leṇo ca parāyaṇo ca;_  
_Asantuleyyo mama so dhanena,_  
_Pāṇena me sādiso esa kattā”._  

</Paragraph>

</Division>
<Division number="1442" book="ja2" bookNo="23" page="286">

<Paragraph number="2778" type="verses" book="ja2">
_“Ciraṃ vivādo mama tuyhañcassa,_  
_Kāmañca pucchāma tameva gantvā;_  
_Esova no vivaratu etamatthaṃ,_  
_Yaṃ vakkhatī hotu kathā ubhinnaṃ”._  

</Paragraph>

</Division>
<Division number="1443" book="ja2" bookNo="23" page="286">

<Paragraph number="2779" type="verses" book="ja2">
_“Addhā hi saccaṃ bhaṇasi,_  
_na ca māṇava sāhasaṃ;_  
_Tameva gantvā pucchāma,_  
_tena tussāmubho janā”._  

</Paragraph>

</Division>
<Division number="1444" book="ja2" bookNo="23" page="286">

<Paragraph number="2780" type="verses" book="ja2">
_“Saccaṃ nu devā vidahū kurūnaṃ,_  
_Dhamme ṭhitaṃ vidhuraṃ nāmamaccaṃ;_  
_Dāsosi rañño uda vāsi ñāti,_  
_Vidhuroti saṅkhā katamāsi loke”._  

</Paragraph>

</Division>
<Division number="1445" book="ja2" bookNo="23" page="286">

<Paragraph number="2781" type="verses" book="ja2">
_“Āmāyadāsāpi bhavanti heke,_  
_Dhanena kītāpi bhavanti dāsā;_  
_Sayampi heke upayanti dāsā,_  
_Bhayā paṇunnāpi bhavanti dāsā._  

</Paragraph>

</Division>
<Division number="1446" book="ja2" bookNo="23" page="287">

<Paragraph number="2782" type="verses" book="ja2">
_Ete narānaṃ caturova dāsā,_  
_Addhā hi yonito ahampi jāto;_  
_Bhavo ca rañño abhavo ca rañño,_  
_Dāsāhaṃ devassa parampi gantvā;_  
_Dhammena maṃ māṇava tuyha dajjā”._  

</Paragraph>

</Division>
<Division number="1447" book="ja2" bookNo="23" page="287">

<Paragraph number="2783" type="verses" book="ja2">
_“Ayaṃ dutīyo vijayo mamajja,_  
_Puṭṭho hi kattā vivarettha pañhaṃ;_  
_Adhammarūpo vata rājaseṭṭho,_  
_Subhāsitaṃ nānujānāsi mayhaṃ”._  

</Paragraph>

</Division>
<Division number="1448" book="ja2" bookNo="23" page="287">

<Paragraph number="2784" type="verses" book="ja2">
_“Evañce no so vivarettha pañhaṃ,_  
_Dāsohamasmi na ca khosmi ñāti;_  
_Gaṇhāhi kaccāna varaṃ dhanānaṃ,_  
_Ādāya yenicchasi tena gaccha”._  


</Paragraph>
</Division>

<Paragraph number="2785" type="center" book="ja2">
Akkhakaṇḍaṃ nāma.
</Paragraph>
