<Paragraph number="2786" type="verses" book="ja2">
_“Vidhura vasamānassa,_  
_gahaṭṭhassa sakaṃ gharaṃ;_  
_Khemā vutti kathaṃ assa,_  
_kathaṃ nu assa saṅgaho._  

</Paragraph>

</Division>
<Division number="1450" book="ja2" bookNo="23" page="287">

<Paragraph number="2787" type="verses" book="ja2">
_Abyābajjhaṃ kathaṃ assa,_  
_saccavādī ca māṇavo;_  
_Asmā lokā paraṃ lokaṃ,_  
_kathaṃ pecca na socati”._  

</Paragraph>

</Division>
<Division number="1451" book="ja2" bookNo="23" page="287">

<Paragraph number="2788" type="verses" book="ja2">
_Taṃ tattha gatimā dhitimā,_  
_matimā atthadassimā;_  
_Saṅkhātā sabbadhammānaṃ,_  
_vidhuro etadabravi._  

</Paragraph>

</Division>
<Division number="1452" book="ja2" bookNo="23" page="287">

<Paragraph number="2789" type="verses" book="ja2">
_“Na sādhāraṇadārassa,_  
_na bhuñje sādumekako;_  
_Na seve lokāyatikaṃ,_  
_netaṃ paññāya vaḍḍhanaṃ._  

</Paragraph>

</Division>
<Division number="1453" book="ja2" bookNo="23" page="287">

<Paragraph number="2790" type="verses" book="ja2">
_Sīlavā vattasampanno,_  
_appamatto vicakkhaṇo;_  
_Nivātavutti atthaddho,_  
_surato sakhilo mudu._  

</Paragraph>

</Division>
<Division number="1454" book="ja2" bookNo="23" page="288">

<Paragraph number="2791" type="verses" book="ja2">
_Saṅgahetā ca mittānaṃ,_  
_saṃvibhāgī vidhānavā;_  
_Tappeyya annapānena,_  
_sadā samaṇabrāhmaṇe._  

</Paragraph>

</Division>
<Division number="1455" book="ja2" bookNo="23" page="288">

<Paragraph number="2792" type="verses" book="ja2">
_Dhammakāmo sutādhāro,_  
_bhaveyya paripucchako;_  
_Sakkaccaṃ payirupāseyya,_  
_sīlavante bahussute._  

</Paragraph>

</Division>
<Division number="1456" book="ja2" bookNo="23" page="288">

<Paragraph number="2793" type="verses" book="ja2">
_Gharamāvasamānassa,_  
_gahaṭṭhassa sakaṃ gharaṃ;_  
_Khemā vutti siyā evaṃ,_  
_evaṃ nu assa saṅgaho._  

</Paragraph>

</Division>
<Division number="1457" book="ja2" bookNo="23" page="288">

<Paragraph number="2794" type="verses" book="ja2">
_Abyābajjhaṃ siyā evaṃ,_  
_saccavādī ca māṇavo;_  
_Asmā lokā paraṃ lokaṃ,_  
_evaṃ pecca na socati”._  


</Paragraph>
</Division>

<Paragraph number="2795" type="center" book="ja2">
Gharāvāsapañhā nāma.
</Paragraph>
