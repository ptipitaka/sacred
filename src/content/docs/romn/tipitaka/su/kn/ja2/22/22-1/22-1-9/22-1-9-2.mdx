<Paragraph number="2728" type="verses" book="ja2">
_“Idañca me maṇiratanaṃ,_  
_passa tvaṃ dvipaduttama;_  
_Itthīnaṃ viggahā cettha,_  
_purisānañca viggahā._  

</Paragraph>

</Division>
<Division number="1394" book="ja2" bookNo="23" page="281">

<Paragraph number="2729" type="verses" book="ja2">
_Migānaṃ viggahā cettha,_  
_sakuṇānañca viggahā;_  
_Nāgarājā supaṇṇā ca,_  
_maṇimhi passa nimmitaṃ._  

</Paragraph>

</Division>
<Division number="1395" book="ja2" bookNo="23" page="282">

<Paragraph number="2730" type="verses" book="ja2">
_Hatthānīkaṃ rathānīkaṃ,_  
_asse pattī ca vammine;_  
_Caturaṅginimaṃ senaṃ,_  
_maṇimhi passa nimmitaṃ._  

</Paragraph>

</Division>
<Division number="1396" book="ja2" bookNo="23" page="282">

<Paragraph number="2731" type="verses" book="ja2">
_Hatthārohe anīkaṭṭhe,_  
_rathike pattikārake;_  
_Balaggāni viyūḷhāni,_  
_maṇimhi passa nimmitaṃ._  

</Paragraph>

</Division>
<Division number="1397" book="ja2" bookNo="23" page="282">

<Paragraph number="2732" type="verses" book="ja2">
_Puraṃ uddhāpasampannaṃ,_  
_bahupākāratoraṇaṃ;_  
_Siṅghāṭakesu bhūmiyo,_  
_maṇimhi passa nimmitaṃ._  

</Paragraph>

</Division>
<Division number="1398" book="ja2" bookNo="23" page="282">

<Paragraph number="2733" type="verses" book="ja2">
_Esikā parikhāyo ca,_  
_palikhaṃ aggaḷāni ca;_  
_Aṭṭālake ca dvāre ca,_  
_maṇimhi passa nimmitaṃ._  

</Paragraph>

</Division>
<Division number="1399" book="ja2" bookNo="23" page="282">

<Paragraph number="2734" type="verses" book="ja2">
_Passa toraṇamaggesu,_  
_nānādijagaṇā bahū;_  
_Haṃsā koñcā mayūrā ca,_  
_cakkavākā ca kukkuhā._  

</Paragraph>

</Division>
<Division number="1400" book="ja2" bookNo="23" page="282">

<Paragraph number="2735" type="verses" book="ja2">
_Kuṇālakā bahū citrā,_  
_sikhaṇḍī jīvajīvakā;_  
_Nānādijagaṇākiṇṇaṃ,_  
_maṇimhi passa nimmitaṃ._  

</Paragraph>

</Division>
<Division number="1401" book="ja2" bookNo="23" page="282">

<Paragraph number="2736" type="verses" book="ja2">
_Passa nagaraṃ supākāraṃ,_  
_abbhutaṃ lomahaṃsanaṃ;_  
_Samussitadhajaṃ rammaṃ,_  
_soṇṇavālukasanthataṃ._  

</Paragraph>

</Division>
<Division number="1402" book="ja2" bookNo="23" page="282">

<Paragraph number="2737" type="verses" book="ja2">
_Passettha paṇṇasālāyo,_  
_vibhattā bhāgaso mitā;_  
_Nivesane nivese ca,_  
_sandhibyūhe pathaddhiyo._  

</Paragraph>

</Division>
<Division number="1403" book="ja2" bookNo="23" page="282">

<Paragraph number="2738" type="verses" book="ja2">
_Pānāgāre ca soṇḍe ca,_  
_sūnā odaniyā gharā;_  
_Vesī ca gaṇikāyo ca,_  
_maṇimhi passa nimmitaṃ._  

</Paragraph>

</Division>
<Division number="1404" book="ja2" bookNo="23" page="282">

<Paragraph number="2739" type="verses" book="ja2">
_Mālākāre ca rajake,_  
_gandhike atha dussike;_  
_Suvaṇṇakāre maṇikāre,_  
_maṇimhi passa nimmitaṃ._  

</Paragraph>

</Division>
<Division number="1405" book="ja2" bookNo="23" page="282">

<Paragraph number="2740" type="verses" book="ja2">
_Āḷārike ca sūde ca,_  
_naṭanāṭakagāyino;_  
_Pāṇissare kumbhathūnike,_  
_maṇimhi passa nimmitaṃ._  

</Paragraph>

</Division>
<Division number="1406" book="ja2" bookNo="23" page="282">

<Paragraph number="2741" type="verses" book="ja2">
_Passa bherī mudiṅgā ca,_  
_saṅkhā paṇavadindimā;_  
_Sabbañca tāḷāvacaraṃ,_  
_maṇimhi passa nimmitaṃ._  

</Paragraph>

</Division>
<Division number="1407" book="ja2" bookNo="23" page="283">

<Paragraph number="2742" type="verses" book="ja2">
_Sammatālañca vīṇañca,_  
_naccagītaṃ suvāditaṃ;_  
_Tūriyatāḷitasaṅghuṭṭhaṃ,_  
_maṇimhi passa nimmitaṃ._  

</Paragraph>

</Division>
<Division number="1408" book="ja2" bookNo="23" page="283">

<Paragraph number="2743" type="verses" book="ja2">
_Laṅghikā muṭṭhikā cettha,_  
_māyākārā ca sobhiyā;_  
_Vetālike ca jalle ca,_  
_maṇimhi passa nimmitaṃ._  

</Paragraph>

</Division>
<Division number="1409" book="ja2" bookNo="23" page="283">

<Paragraph number="2744" type="verses" book="ja2">
_Samajjā cettha vattanti,_  
_ākiṇṇā naranāribhi;_  
_Mañcātimañce bhūmiyo,_  
_maṇimhi passa nimmitaṃ._  

</Paragraph>

</Division>
<Division number="1410" book="ja2" bookNo="23" page="283">

<Paragraph number="2745" type="verses" book="ja2">
_Passa malle samajjasmiṃ,_  
_phoṭente diguṇaṃ bhujaṃ;_  
_Nihate nihatamāne ca,_  
_maṇimhi passa nimmitaṃ._  

</Paragraph>

</Division>
<Division number="1411" book="ja2" bookNo="23" page="283">

<Paragraph number="2746" type="verses" book="ja2">
_Passa pabbatapādesu,_  
_nānāmigagaṇā bahū;_  
_Sīhā byagghā varāhā ca,_  
_acchakokataracchayo._  

</Paragraph>

</Division>
<Division number="1412" book="ja2" bookNo="23" page="283">

<Paragraph number="2747" type="verses" book="ja2">
_Palāsādā gavajā ca,_  
_mahiṃsā rohitā rurū;_  
_Eṇeyyā ca varāhā ca,_  
_gaṇino nīkasūkarā._  

</Paragraph>

</Division>
<Division number="1413" book="ja2" bookNo="23" page="283">

<Paragraph number="2748" type="verses" book="ja2">
_Kadalimigā bahū citrā,_  
_biḷārā sasakaṇṭakā;_  
_Nānāmigagaṇākiṇṇaṃ,_  
_maṇimhi passa nimmitaṃ._  

</Paragraph>

</Division>
<Division number="1414" book="ja2" bookNo="23" page="283">

<Paragraph number="2749" type="verses" book="ja2">
_Najjāyo supatitthāyo,_  
_soṇṇavālukasanthatā;_  
_Acchā savanti ambūni,_  
_macchagumbanisevitā._  

</Paragraph>

</Division>
<Division number="1415" book="ja2" bookNo="23" page="283">

<Paragraph number="2750" type="verses" book="ja2">
_Kumbhīlā makarā cettha,_  
_susumārā ca kacchapā;_  
_Pāṭhīnā pāvusā macchā,_  
_balajā muñjarohitā._  

</Paragraph>

</Division>
<Division number="1416" book="ja2" bookNo="23" page="283">

<Paragraph number="2751" type="verses" book="ja2">
_Nānādijagaṇākiṇṇā,_  
_nānādumagaṇāyutā;_  
_Veḷuriyakarodāyo,_  
_maṇimhi passa nimmitaṃ._  

</Paragraph>

</Division>
<Division number="1417" book="ja2" bookNo="23" page="283">

<Paragraph number="2752" type="verses" book="ja2">
_Passettha pokkharaṇiyo,_  
_suvibhattā catuddisā;_  
_Nānādijagaṇākiṇṇā,_  
_puthulomanisevitā._  

</Paragraph>

</Division>
<Division number="1418" book="ja2" bookNo="23" page="283">

<Paragraph number="2753" type="verses" book="ja2">
_Samantodakasampannaṃ,_  
_mahiṃ sāgarakuṇḍalaṃ;_  
_Upetaṃ vanarājehi,_  
_maṇimhi passa nimmitaṃ._  

</Paragraph>

</Division>
<Division number="1419" book="ja2" bookNo="23" page="283">

<Paragraph number="2754" type="verses" book="ja2">
_Purato videhe passa,_  
_goyāniye ca pacchato;_  
_Kuruyo jambudīpañca,_  
_maṇimhi passa nimmitaṃ._  

</Paragraph>

</Division>
<Division number="1420" book="ja2" bookNo="23" page="284">

<Paragraph number="2755" type="verses" book="ja2">
_Passa candaṃ sūriyañca,_  
_obhāsante catuddisā;_  
_Sineruṃ anupariyante,_  
_maṇimhi passa nimmitaṃ._  

</Paragraph>

</Division>
<Division number="1421" book="ja2" bookNo="23" page="284">

<Paragraph number="2756" type="verses" book="ja2">
_Sineruṃ himavantañca,_  
_sāgarañca mahītalaṃ;_  
_Cattāro ca mahārāje,_  
_maṇimhi passa nimmitaṃ._  

</Paragraph>

</Division>
<Division number="1422" book="ja2" bookNo="23" page="284">

<Paragraph number="2757" type="verses" book="ja2">
_Ārāme vanagumbe ca,_  
_pāṭiye ca siluccaye;_  
_Ramme kimpurisākiṇṇe,_  
_maṇimhi passa nimmitaṃ._  

</Paragraph>

</Division>
<Division number="1423" book="ja2" bookNo="23" page="284">

<Paragraph number="2758" type="verses" book="ja2">
_Phārusakaṃ cittalataṃ,_  
_missakaṃ nandanaṃ vanaṃ;_  
_Vejayantañca pāsādaṃ,_  
_maṇimhi passa nimmitaṃ._  

</Paragraph>

</Division>
<Division number="1424" book="ja2" bookNo="23" page="284">

<Paragraph number="2759" type="verses" book="ja2">
_Sudhammaṃ tāvatiṃsañca,_  
_pārichattañca pupphitaṃ;_  
_Erāvaṇaṃ nāgarājaṃ,_  
_maṇimhi passa nimmitaṃ._  

</Paragraph>

</Division>
<Division number="1425" book="ja2" bookNo="23" page="284">

<Paragraph number="2760" type="verses" book="ja2">
_Passettha devakaññāyo,_  
_nabhā vijjurivuggatā;_  
_Nandane vicarantiyo,_  
_maṇimhi passa nimmitaṃ._  

</Paragraph>

</Division>
<Division number="1426" book="ja2" bookNo="23" page="284">

<Paragraph number="2761" type="verses" book="ja2">
_Passettha devakaññāyo,_  
_devaputtapalobhinī;_  
_Devaputte ramamāne,_  
_maṇimhi passa nimmitaṃ._  

</Paragraph>

</Division>
<Division number="1427" book="ja2" bookNo="23" page="284">

<Paragraph number="2762" type="verses" book="ja2">
_Parosahassapāsāde,_  
_veḷuriyaphalasanthate;_  
_Pajjalante ca vaṇṇena,_  
_maṇimhi passa nimmitaṃ._  

</Paragraph>

</Division>
<Division number="1428" book="ja2" bookNo="23" page="284">

<Paragraph number="2763" type="verses" book="ja2">
_Tāvatiṃse ca yāme ca,_  
_Tusite cāpi nimmite;_  
_Paranimmitavasavattino,_  
_Maṇimhi passa nimmitaṃ._  

</Paragraph>

</Division>
<Division number="1429" book="ja2" bookNo="23" page="284">

<Paragraph number="2764" type="verses" book="ja2">
_Passettha pokkharaṇiyo,_  
_vippasannodikā sucī;_  
_Mandālakehi sañchannā,_  
_padumuppalakehi ca._  

</Paragraph>

</Division>
<Division number="1430" book="ja2" bookNo="23" page="284">

<Paragraph number="2765" type="verses" book="ja2">
_Dasettha rājiyo setā,_  
_dasanīlā manoramā;_  
_Cha piṅgalā pannarasa,_  
_haliddā ca catuddasa._  

</Paragraph>

</Division>
<Division number="1431" book="ja2" bookNo="23" page="284">

<Paragraph number="2766" type="verses" book="ja2">
_Vīsati tattha sovaṇṇā,_  
_vīsati rajatāmayā;_  
_Indagopakavaṇṇābhā,_  
_tāva dissanti tiṃsati._  

</Paragraph>

</Division>
<Division number="1432" book="ja2" bookNo="23" page="284">

<Paragraph number="2767" type="verses" book="ja2">
_Dasettha kāḷiyo chacca,_  
_mañjeṭṭhā pannavīsati;_  
_Missā bandhukapupphehi,_  
_nīluppalavicittikā._  

</Paragraph>

</Division>

_accimantaṃ pabhassaraṃ;_  
_Odhisuṅkaṃ mahārāja,_  
_passa tvaṃ dvipaduttama”._  


</Paragraph>
</Division>
