_kāmaṃ sumukha pakkama._

<Division number="90" book="ja2">

<Paragraph number="861" type="verses" book="ja2">

_ekaṃ pāsavasaṃ gataṃ;_  
_Anapekkhamānā gacchanti,_  
_kiṃ eko avahīyasi._  

</Paragraph>

</Division>
<Division number="91" book="ja2">

<Paragraph number="862" type="verses" book="ja2">
_Pateva patataṃ seṭṭha,_  
_natthi baddhe sahāyatā;_  
_Mā anīghāya hāpesi,_  
_kāmaṃ sumukha pakkama”._  

</Paragraph>

</Division>
<Division number="92" book="ja2">

<Paragraph number="863" type="verses" book="ja2">
_“Nāhaṃ dukkhaparetopi,_  
_dhataraṭṭha tuvaṃ jahe;_  
_Jīvitaṃ maraṇaṃ vā me,_  
_tayā saddhiṃ bhavissati._  

</Paragraph>

</Division>
<Division number="93" book="ja2">

<Paragraph number="864" type="verses" book="ja2">
_Nāhaṃ dukkhaparetopi,_  
_dhataraṭṭha tuvaṃ jahe;_  
_Na maṃ anariyasaṃyutte,_  
_kamme yojetumarahasi._  

</Paragraph>

</Division>
<Division number="94" book="ja2">

<Paragraph number="865" type="verses" book="ja2">
_Sakumāro sakhā tyasmi,_  
_sacitte casmi te ṭhito;_  
_Ñāto senāpati tyāhaṃ,_  
_haṃsānaṃ pavaruttama._  

</Paragraph>

</Division>
<Division number="95" book="ja2">

<Paragraph number="866" type="verses" book="ja2">
_Kathaṃ ahaṃ vikatthissaṃ,_  
_ñātimajjhe ito gato;_  
_Taṃ hitvā patataṃ seṭṭha,_  
_kiṃ te vakkhāmito gato;_  
_Idha pāṇaṃ cajissāmi,_  
_nānariyaṃ kattumussahe”._  

</Paragraph>

</Division>
<Division number="96" book="ja2">

<Paragraph number="867" type="verses" book="ja2">
_“Eso hi dhammo sumukha,_  
_yaṃ tvaṃ ariyapathe ṭhito;_  
_Yo bhattāraṃ sakhāraṃ maṃ,_  
_na pariccattumussahe._  

</Paragraph>

</Division>
<Division number="97" book="ja2">

<Paragraph number="868" type="verses" book="ja2">
_Tañhi me pekkhamānassa,_  
_bhayaṃ na tveva jāyati;_  
_Adhigacchasi tvaṃ mayhaṃ,_  
_evaṃ bhūtassa jīvitaṃ”._  

</Paragraph>

</Division>
<Division number="98" book="ja2">

<Paragraph number="869" type="verses" book="ja2">
_Iccevaṃ mantayantānaṃ,_  
_Ariyānaṃ ariyavuttinaṃ;_  
_Daṇḍamādāya nesādo,_  
_Āpatī turito bhusaṃ._  

</Paragraph>

</Division>
<Division number="99" book="ja2">

<Paragraph number="870" type="verses" book="ja2">
_Tamāpatantaṃ disvāna,_  
_Sumukho atibrūhayi;_  
_Aṭṭhāsi purato rañño,_  
_Haṃso vissāsayaṃ byathaṃ._  

</Paragraph>

</Division>
<Division number="100" book="ja2">

<Paragraph number="871" type="verses" book="ja2">
_“Mā bhāyi patataṃ seṭṭha,_  
_na hi bhāyanti tādisā;_  
_Ahaṃ yogaṃ payuñjissaṃ,_  
_yuttaṃ dhammūpasaṃhitaṃ;_  
_Tena pariyāpadānena,_  
_khippaṃ pāsā pamokkhasi”._  

</Paragraph>

</Division>
<Division number="101" book="ja2">

<Paragraph number="872" type="verses" book="ja2">
_Tassa taṃ vacanaṃ sutvā,_  
_sumukhassa subhāsitaṃ;_  
_Pahaṭṭhalomo nesādo,_  
_añjalissa paṇāmayi._  

</Paragraph>

</Division>
<Division number="102" book="ja2">

<Paragraph number="873" type="verses" book="ja2">
_“Na me sutaṃ vā diṭṭhaṃ vā,_  
_bhāsanto mānusiṃ dijo;_  
_Ariyaṃ bruvāno vakkaṅgo,_  
_cajanto mānusiṃ giraṃ._  

</Paragraph>

</Division>
<Division number="103" book="ja2">

<Paragraph number="874" type="verses" book="ja2">
_Kiṃ nu tāyaṃ dijo hoti,_  
_mutto baddhaṃ upāsasi;_  
_Ohāya sakuṇā yanti,_  
_kiṃ eko avahīyasi”._  

</Paragraph>

</Division>
<Division number="104" book="ja2">

<Paragraph number="875" type="verses" book="ja2">
_“Rājā me so dijāmitta,_  
_senāpaccassa kārayiṃ;_  
_Tamāpade pariccattuṃ,_  
_nussahe vihagādhipaṃ._  

</Paragraph>

</Division>
<Division number="105" book="ja2">

<Paragraph number="876" type="verses" book="ja2">
_Mahāgaṇāya bhattā me,_  
_mā eko byasanaṃ agā;_  
_Tathā taṃ samma nesāda,_  
_bhattāyaṃ abhito rame”._  

</Paragraph>

</Division>
<Division number="106" book="ja2">

<Paragraph number="877" type="verses" book="ja2">
_“Ariyavattasi vakkaṅga,_  
_yo piṇḍamapacāyasi;_  
_Cajāmi te taṃ bhattāraṃ,_  
_gacchathūbho yathāsukhaṃ”._  

</Paragraph>

</Division>
<Division number="107" book="ja2">

<Paragraph number="878" type="verses" book="ja2">
_“Sace attappayogena,_  
_ohito haṃsapakkhinaṃ;_  
_Paṭigaṇhāma te samma,_  
_etaṃ abhayadakkhiṇaṃ._  

</Paragraph>

</Division>
<Division number="108" book="ja2">

<Paragraph number="879" type="verses" book="ja2">
_No ce attappayogena,_  
_ohito haṃsapakkhinaṃ;_  
_Anissaro muñcamamhe,_  
_theyyaṃ kayirāsi luddaka”._  

</Paragraph>

</Division>
<Division number="109" book="ja2">

<Paragraph number="880" type="verses" book="ja2">
_“Yassa tvaṃ bhatako rañño,_  
_kāmaṃ tasseva pāpaya;_  
_Tattha saṃyamano rājā,_  
_yathābhiññaṃ karissati”._  

</Paragraph>

</Division>
<Division number="110" book="ja2">

<Paragraph number="881" type="verses" book="ja2">
_Iccevaṃ vutto nesādo,_  
_hemavaṇṇe harittace;_  
_Ubho hatthehi saṅgayha,_  
_pañjare ajjhavodahi._  

</Paragraph>

</Division>
<Division number="111" book="ja2">

<Paragraph number="882" type="verses" book="ja2">
_Te pañjaragate pakkhī,_  
_ubho bhassaravaṇṇine;_  
_Sumukhaṃ dhataraṭṭhañca,_  
_luddo ādāya pakkami._  

</Paragraph>

</Division>
<Division number="112" book="ja2">

<Paragraph number="883" type="verses" book="ja2">
_Harīyamāno dhataraṭṭho,_  
_sumukhaṃ etadabravi;_  
_“Bāḷhaṃ bhāyāmi sumukha,_  
_sāmāya lakkhaṇūruyā;_  
_Asmākaṃ vadhamaññāya,_  
_athattānaṃ vadhissati._  

</Paragraph>

</Division>
<Division number="113" book="ja2">

<Paragraph number="884" type="verses" book="ja2">
_Pākahaṃsā ca sumukha,_  
_suhemā hemasuttacā;_  
_Koñcī samuddatīreva,_  
_kapaṇā nūna rucchati”._  

</Paragraph>

</Division>
<Division number="114" book="ja2">

<Paragraph number="885" type="verses" book="ja2">
_“Evaṃ mahanto lokassa,_  
_appameyyo mahāgaṇī;_  
_Ekitthimanusoceyya,_  
_nayidaṃ paññavatāmiva._  

</Paragraph>

</Division>
<Division number="115" book="ja2">

<Paragraph number="886" type="verses" book="ja2">
_Vātova gandhamādeti,_  
_ubhayaṃ chekapāpakaṃ;_  
_Bālo āmakapakkaṃva,_  
_lolo andhova āmisaṃ._  

</Paragraph>

</Division>
<Division number="116" book="ja2">

<Paragraph number="887" type="verses" book="ja2">
_Avinicchayaññu atthesu,_  
_mandova paṭibhāsi maṃ;_  
_Kiccākiccaṃ na jānāsi,_  
_sampatto kālapariyāyaṃ._  

</Paragraph>

</Division>
<Division number="117" book="ja2">

<Paragraph number="888" type="verses" book="ja2">
_Aḍḍhummatto udīresi,_  
_yo seyyā maññasitthiyo;_  
_Bahusādhāraṇā hetā,_  
_soṇḍānaṃva surāgharaṃ._  

</Paragraph>

</Division>
<Division number="118" book="ja2">

<Paragraph number="889" type="verses" book="ja2">
_Māyā cesā marīcī ca,_  
_soko rogo cupaddavo;_  
_Kharā ca bandhanā cetā,_  
_maccupāsā guhāsayā;_  
_Tāsu yo vissase poso,_  
_so naresu narādhamo”._  

</Paragraph>

</Division>
<Division number="119" book="ja2">

<Paragraph number="890" type="verses" book="ja2">
_“Yaṃ vuddhehi upaññātaṃ,_  
_ko taṃ ninditumarahati;_  
_Mahābhūtitthiyo nāma,_  
_lokasmiṃ udapajjisuṃ._  

</Paragraph>

</Division>
<Division number="120" book="ja2">

<Paragraph number="891" type="verses" book="ja2">
_Khiḍḍā paṇihitā tyāsu,_  
_rati tyāsu patiṭṭhitā;_  
_Bījāni tyāsu rūhanti,_  
_yadidaṃ sattā pajāyare;_  
_Tāsu ko nibbide poso,_  
_pāṇamāsajja pāṇibhi._  

</Paragraph>

</Division>
<Division number="121" book="ja2">

<Paragraph number="892" type="verses" book="ja2">
_Tvameva nañño sumukha,_  
_thīnaṃ atthesu yuñjasi;_  
_Tassa tyajja bhaye jāte,_  
_bhītena jāyate mati._  

</Paragraph>

</Division>
<Division number="122" book="ja2">

<Paragraph number="893" type="verses" book="ja2">
_Sabbo hi saṃsayaṃ patto,_  
_bhayaṃ bhīru titikkhati;_  
_Paṇḍitā ca mahantāno,_  
_atthe yuñjanti duyyuje._  

</Paragraph>

</Division>
<Division number="123" book="ja2">

<Paragraph number="894" type="verses" book="ja2">
_Etadatthāya rājāno,_  
_sūramicchanti mantinaṃ;_  
_Paṭibāhati yaṃ sūro,_  
_āpadaṃ attapariyāyaṃ._  

</Paragraph>

</Division>
<Division number="124" book="ja2">

<Paragraph number="895" type="verses" book="ja2">
_Mā no ajja vikantiṃsu,_  
_rañño sūdā mahānase;_  
_Tathā hi vaṇṇo pattānaṃ,_  
_phalaṃ veḷuṃva taṃ vadhi._  

</Paragraph>

</Division>
<Division number="125" book="ja2">

<Paragraph number="896" type="verses" book="ja2">
_Muttopi na icchi uḍḍetuṃ,_  
_Sayaṃ bandhaṃ upāgami;_  
_Sopajja saṃsayaṃ patto,_  
_Atthaṃ gaṇhāhi mā mukhaṃ._  

</Paragraph>

</Division>
<Division number="126" book="ja2">

<Paragraph number="897" type="verses" book="ja2">
_So taṃ yogaṃ payuñjassu,_  
_yuttaṃ dhammūpasaṃhitaṃ;_  
_Tava pariyāpadānena,_  
_mama pāṇesanaṃ cara”._  

</Paragraph>

</Division>
<Division number="127" book="ja2">

<Paragraph number="898" type="verses" book="ja2">
_“Mā bhāyi patataṃ seṭṭha,_  
_na hi bhāyanti tādisā;_  
_Ahaṃ yogaṃ payuñjissaṃ,_  
_yuttaṃ dhammūpasaṃhitaṃ;_  
_Mama pariyāpadānena,_  
_khippaṃ pāsā pamokkhasi”._  

</Paragraph>

</Division>
<Division number="128" book="ja2">

<Paragraph number="899" type="verses" book="ja2">
_So luddo haṃsakājena,_  
_rājadvāraṃ upāgami;_  
_“Paṭivedetha maṃ rañño,_  
_dhataraṭṭhāyamāgato”._  

</Paragraph>

</Division>
<Division number="129" book="ja2">

<Paragraph number="900" type="verses" book="ja2">
_Te disvā puññasaṅkāse,_  
_ubho lakkhaṇasammate;_  
_Khalu saṃyamano rājā,_  
_amacce ajjhabhāsatha._  

</Paragraph>

</Division>
<Division number="130" book="ja2">

<Paragraph number="901" type="verses" book="ja2">
_“Detha luddassa vatthāni,_  
_annaṃ pānañca bhojanaṃ;_  
_Kāmaṅkaro hiraññassa,_  
_yāvanto esa icchati”._  

</Paragraph>

</Division>
<Division number="131" book="ja2">

<Paragraph number="902" type="verses" book="ja2">
_Disvā luddaṃ pasannattaṃ,_  
_Kāsirājā tadabravi;_  
_“Yadyāyaṃ samma khemaka,_  
_Puṇṇā haṃsehi tiṭṭhati._  

</Paragraph>

</Division>
<Division number="132" book="ja2">

<Paragraph number="903" type="verses" book="ja2">
_Kathaṃ rucimajjhagataṃ,_  
_pāsahattho upāgami;_  
_Okiṇṇaṃ ñātisaṅghehi,_  
_nimmajjhimaṃ kathaṃ gahi”._  

</Paragraph>

</Division>
<Division number="133" book="ja2">

<Paragraph number="904" type="verses" book="ja2">
_“Ajja me sattamā ratti,_  
_adanāni upāsato;_  
_Padametassa anvesaṃ,_  
_appamatto ghaṭassito._  

</Paragraph>

</Division>
<Division number="134" book="ja2">

<Paragraph number="905" type="verses" book="ja2">
_Athassa padamaddakkhiṃ,_  
_carato adanesanaṃ;_  
_Tatthāhaṃ odahiṃ pāsaṃ,_  
_evaṃ taṃ dijamaggahiṃ”._  

</Paragraph>

</Division>
<Division number="135" book="ja2">

<Paragraph number="906" type="verses" book="ja2">
_“Ludda dve ime sakuṇā,_  
_atha ekoti bhāsasi;_  
_Cittaṃ nu te vipariyattaṃ,_  
_adu kiṃ nu jigīsasi”._  

</Paragraph>

</Division>
<Division number="136" book="ja2">

<Paragraph number="907" type="verses" book="ja2">
_“Yassa lohitakā tālā,_  
_tapanīyanibhā subhā;_  
_Uraṃ saṃhacca tiṭṭhanti,_  
_so me bandhaṃ upāgami._  

</Paragraph>

</Division>
<Division number="137" book="ja2">

<Paragraph number="908" type="verses" book="ja2">
_Athāyaṃ bhassaro pakkhī,_  
_abaddho baddhamāturaṃ;_  
_Ariyaṃ bruvāno aṭṭhāsi,_  
_cajanto mānusiṃ giraṃ”._  

</Paragraph>

</Division>
<Division number="138" book="ja2">

<Paragraph number="909" type="verses" book="ja2">
_“Atha kiṃ dāni sumukha,_  
_Hanuṃ saṃhacca tiṭṭhasi;_  
_Adu me parisaṃ patto,_  
_Bhayā bhīto na bhāsasi”._  

</Paragraph>

</Division>
<Division number="139" book="ja2">

<Paragraph number="910" type="verses" book="ja2">
_“Nāhaṃ kāsipati bhīto,_  
_ogayha parisaṃ tava;_  
_Nāhaṃ bhayā na bhāsissaṃ,_  
_vākyaṃ atthamhi tādise”._  

</Paragraph>

</Division>
<Division number="140" book="ja2">

<Paragraph number="911" type="verses" book="ja2">
_“Na te abhisaraṃ passe,_  
_na rathe napi pattike;_  
_Nāssa cammaṃ va kīṭaṃ vā,_  
_vammite ca dhanuggahe._  

</Paragraph>

</Division>
<Division number="141" book="ja2">

<Paragraph number="912" type="verses" book="ja2">
_Na hiraññaṃ suvaṇṇaṃ vā,_  
_nagaraṃ vā sumāpitaṃ;_  
_Okiṇṇaparikhaṃ duggaṃ,_  
_daḷhamaṭṭālakoṭṭhakaṃ;_  
_Yattha paviṭṭho sumukha,_  
_bhāyitabbaṃ na bhāyasi”._  

</Paragraph>

</Division>
<Division number="142" book="ja2">

<Paragraph number="913" type="verses" book="ja2">
_“Na me abhisarenattho,_  
_nagarena dhanena vā;_  
_Apathena pathaṃ yāma,_  
_antalikkhecarā mayaṃ._  

</Paragraph>

</Division>
<Division number="143" book="ja2">

<Paragraph number="914" type="verses" book="ja2">
_Sutā ca paṇḍitā tyamhā,_  
_nipuṇā atthacintakā;_  
_Bhāsematthavatiṃ vācaṃ,_  
_sacce cassa patiṭṭhito._  

</Paragraph>

</Division>
<Division number="144" book="ja2">

<Paragraph number="915" type="verses" book="ja2">
_Kiñca tuyhaṃ asaccassa,_  
_anariyassa karissati;_  
_Musāvādissa luddassa,_  
_bhaṇitampi subhāsitaṃ._  

</Paragraph>

</Division>
<Division number="145" book="ja2">

<Paragraph number="916" type="verses" book="ja2">
_Taṃ brāhmaṇānaṃ vacanā,_  
_imaṃ khemamakārayi;_  
_Abhayañca tayā ghuṭṭhaṃ,_  
_imāyo dasadhā disā._  

</Paragraph>

</Division>
<Division number="146" book="ja2">

<Paragraph number="917" type="verses" book="ja2">
_Ogayha te pokkharaṇiṃ,_  
_vippasannodakaṃ suciṃ;_  
_Pahūtaṃ cādanaṃ tattha,_  
_ahiṃsā cettha pakkhinaṃ._  

</Paragraph>

</Division>
<Division number="147" book="ja2">

<Paragraph number="918" type="verses" book="ja2">
_Idaṃ sutvāna nigghosaṃ,_  
_āgatamha tavantike;_  
_Te te bandhasma pāsena,_  
_etaṃ te bhāsitaṃ musā._  

</Paragraph>

</Division>
<Division number="148" book="ja2">

<Paragraph number="919" type="verses" book="ja2">
_Musāvādaṃ purakkhatvā,_  
_icchālobhañca pāpakaṃ;_  
_Ubho sandhimatikkamma,_  
_asātaṃ upapajjati”._  

</Paragraph>

</Division>
<Division number="149" book="ja2">

<Paragraph number="920" type="verses" book="ja2">
_“Nāparajjhāma sumukha,_  
_napi lobhāva maggahiṃ;_  
_Sutā ca paṇḍitātyattha-_  
_nipuṇā atthacintakā._  

</Paragraph>

</Division>
<Division number="150" book="ja2">

<Paragraph number="921" type="verses" book="ja2">
_Appevatthavatiṃ vācaṃ,_  
_byāhareyyuṃ idhāgatā;_  
_Tathā taṃ samma nesādo,_  
_vutto sumukha maggahi”._  

</Paragraph>

</Division>
<Division number="151" book="ja2">

<Paragraph number="922" type="verses" book="ja2">
_“Neva bhītā kāsipati,_  
_upanītasmi jīvite;_  
_Bhāsematthavatiṃ vācaṃ,_  
_sampattā kālapariyāyaṃ._  

</Paragraph>

</Division>
<Division number="152" book="ja2">

<Paragraph number="923" type="verses" book="ja2">
_Yo migena migaṃ hanti,_  
_pakkhiṃ vā pana pakkhinā;_  
_Sutena vā sutaṃ kiṇyā,_  
_kiṃ anariyataraṃ tato._  

</Paragraph>

</Division>
<Division number="153" book="ja2">

<Paragraph number="924" type="verses" book="ja2">
_Yo cāriyarudaṃ bhāse,_  
_anariyadhammavassito;_  
_Ubho so dhaṃsate lokā,_  
_idha ceva parattha ca._  

</Paragraph>

</Division>
<Division number="154" book="ja2">

<Paragraph number="925" type="verses" book="ja2">
_Na majjetha yasaṃ patto,_  
_na byādhe pattasaṃsayaṃ;_  
_Vāyametheva kiccesu,_  
_saṃvare vivarāni ca._  

</Paragraph>

</Division>
<Division number="155" book="ja2">

<Paragraph number="926" type="verses" book="ja2">
_Ye vuddhā abbhatikkantā,_  
_sampattā kālapariyāyaṃ;_  
_Idha dhammaṃ caritvāna,_  
_evaṃte tidivaṃ gatā._  

</Paragraph>

</Division>
<Division number="156" book="ja2">

<Paragraph number="927" type="verses" book="ja2">
_Idaṃ sutvā kāsipati,_  
_dhammamattani pālaya;_  
_Dhataraṭṭhañca muñcāhi,_  
_haṃsānaṃ pavaruttamaṃ”._  

</Paragraph>

</Division>
<Division number="157" book="ja2">

<Paragraph number="928" type="verses" book="ja2">
_“Āharantudakaṃ pajjaṃ,_
