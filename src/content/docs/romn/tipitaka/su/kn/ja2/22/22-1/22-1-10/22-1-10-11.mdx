_“Kassetaṃ mukhamābhāti,_  
_hemaṃvuttattamagginā;_  
_Nikkhaṃva jātarūpassa,_  
_ukkāmukhapahaṃsitaṃ._  

</Paragraph>

</Division>
<Division number="2304" book="ja2">

<Paragraph number="3662" type="verses" book="ja2">
_Ubho sadisapaccaṅgā,_  
_ubho sadisalakkhaṇā;_  
_Jālissa sadiso eko,_  
_ekā kaṇhājinā yathā._  

</Paragraph>

</Division>
<Division number="2305" book="ja2">

<Paragraph number="3663" type="verses" book="ja2">
_Sīhā bilāva nikkhantā,_  
_ubho sampatirūpakā;_  
_Jātarūpamayāyeva,_  
_ime dissanti dārakā._  

</Paragraph>

</Division>
<Division number="2306" book="ja2">

<Paragraph number="3664" type="verses" book="ja2">
_Kuto nu bhavaṃ bhāradvāja,_  
_Ime ānesi dārake;_  
_Ajja raṭṭhaṃ anuppatto,_  
_Kuhiṃ gacchasi brāhmaṇa”._  

</Paragraph>

</Division>
<Division number="2307" book="ja2">

<Paragraph number="3665" type="verses" book="ja2">
_“Mayhaṃ te dārakā deva,_  
_dinnā vittena sañjaya;_  
_Ajja pannarasā ratti,_  
_yato laddhā me dārakā”._  

</Paragraph>

</Division>
<Division number="2308" book="ja2">

<Paragraph number="3666" type="verses" book="ja2">
_“Kena vā vācapeyyena,_  
_sammāñāyena saddahe;_  
_Ko tetaṃ dānamadadā,_  
_puttake dānamuttamaṃ”._  

</Paragraph>

</Division>
<Division number="2309" book="ja2">

<Paragraph number="3667" type="verses" book="ja2">
_“Yo yācataṃ patiṭṭhāsi,_  
_bhūtānaṃ dharaṇīriva;_  
_So me vessantaro rājā,_  
_puttedāsi vane vasaṃ._  

</Paragraph>

</Division>
<Division number="2310" book="ja2">

<Paragraph number="3668" type="verses" book="ja2">
_Yo yācataṃ gatī āsi,_  
_savantīnaṃva sāgaro;_  
_So me vessantaro rājā,_  
_puttedāsi vane vasaṃ”._  

</Paragraph>

</Division>
<Division number="2311" book="ja2">

<Paragraph number="3669" type="verses" book="ja2">
_“Dukkaṭaṃ vata bho raññā,_  
_saddhena gharamesinā;_  
_Kathaṃ nu puttake dajjā,_  
_araññe avaruddhako._  

</Paragraph>

</Division>
<Division number="2312" book="ja2">

<Paragraph number="3670" type="verses" book="ja2">
_Imaṃ bhonto nisāmetha,_  
_yāvantettha samāgatā;_  
_Kathaṃ vessantaro rājā,_  
_puttedāsi vane vasaṃ._  

</Paragraph>

</Division>
<Division number="2313" book="ja2">

<Paragraph number="3671" type="verses" book="ja2">
_Dāsiṃ dāsaṃ ca so dajjā,_  
_assaṃ cassatarīrathaṃ;_  
_Hatthiñca kuñjaraṃ dajja,_  
_kathaṃ so dajja dārake”._  

</Paragraph>

</Division>
<Division number="2314" book="ja2">

<Paragraph number="3672" type="verses" book="ja2">
_“Yassa nassa ghare dāso,_  
_asso cassatarīratho;_  
_Hatthī ca kuñjaro nāgo,_  
_kiṃ so dajjā pitāmaha”._  

</Paragraph>

</Division>
<Division number="2315" book="ja2">

<Paragraph number="3673" type="verses" book="ja2">
_“Dānamassa pasaṃsāma,_  
_na ca nindāma puttakā;_  
_Kathaṃ nu hadayaṃ āsi,_  
_tumhe datvā vanibbake”._  

</Paragraph>

</Division>
<Division number="2316" book="ja2">

<Paragraph number="3674" type="verses" book="ja2">
_“Dukkhassa hadayaṃ āsi,_  
_atho uṇhampi passasi;_  
_Rohinīheva tambakkhī,_  
_pitā assūni vattayi”._  

</Paragraph>

</Division>
<Division number="2317" book="ja2">

<Paragraph number="3675" type="verses" book="ja2">
_Yaṃ taṃ kaṇhājināvoca,_  
_“ayaṃ maṃ tāta brāhmaṇo;_  
_Laṭṭhiyā paṭikoṭeti,_  
_ghare jātaṃva dāsiyaṃ._  

</Paragraph>

</Division>
<Division number="2318" book="ja2">

<Paragraph number="3676" type="verses" book="ja2">
_Na cāyaṃ brāhmaṇo tāta,_  
_dhammikā honti brāhmaṇā;_  
_Yakkho brāhmaṇavaṇṇena,_  
_khādituṃ tāta neti no;_  
_Nīyamāne pisācena,_  
_kiṃ nu tāta udikkhasi”._  

</Paragraph>

</Division>
<Division number="2319" book="ja2">

<Paragraph number="3677" type="verses" book="ja2">
_“Rājaputtī ca vo mātā,_  
_rājaputto ca vo pitā;_  
_Pubbe me aṅkamāruyha,_  
_kiṃ nu tiṭṭhatha ārakā”._  

</Paragraph>

</Division>
<Division number="2320" book="ja2">

<Paragraph number="3678" type="verses" book="ja2">
_“Rājaputtī ca no mātā,_  
_rājaputto ca no pitā;_  
_Dāsā mayaṃ brāhmaṇassa,_  
_tasmā tiṭṭhāma ārakā”._  

</Paragraph>

</Division>
<Division number="2321" book="ja2">

<Paragraph number="3679" type="verses" book="ja2">
_“Mā sammevaṃ avacuttha,_  
_ḍayhate hadayaṃ mama;_  
_Citakāyaṃva me kāyo,_  
_āsane na sukhaṃ labhe._  

</Paragraph>

</Division>
<Division number="2322" book="ja2">

<Paragraph number="3680" type="verses" book="ja2">
_Mā sammevaṃ avacuttha,_  
_bhiyyo sokaṃ janetha maṃ;_  
_Nikkiṇissāmi dabbena,_  
_na vo dāsā bhavissatha._  

</Paragraph>

</Division>
<Division number="2323" book="ja2">

<Paragraph number="3681" type="verses" book="ja2">
_Kimagghiyañhi vo tāta,_  
_brāhmaṇassa pitā adā;_  
_Yathābhūtaṃ me akkhātha,_  
_paṭipādentu brāhmaṇaṃ”._  

</Paragraph>

</Division>
<Division number="2324" book="ja2">

<Paragraph number="3682" type="verses" book="ja2">
_“Sahassagghañhi maṃ tāta,_  
_brāhmaṇassa pitā adā;_  
_Atha kaṇhājinaṃ kaññaṃ,_  
_hatthinā ca satena ca”._  

</Paragraph>

</Division>
<Division number="2325" book="ja2">

<Paragraph number="3683" type="verses" book="ja2">
_“Uṭṭhehi katte taramāno,_  
_brāhmaṇassa avākara;_  
_Dāsisataṃ dāsasataṃ,_  
_gavaṃ hatthusabhaṃ sataṃ;_  
_Jātarūpasahassañca,_  
_puttānaṃ dehi nikkayaṃ”._  

</Paragraph>

</Division>
<Division number="2326" book="ja2">

<Paragraph number="3684" type="verses" book="ja2">
_Tato kattā taramāno,_  
_brāhmaṇassa avākari;_  
_Dāsisataṃ dāsasataṃ,_  
_gavaṃ hatthusabhaṃ sataṃ;_  
_Jātarūpasahassañca,_  
_puttānaṃdāsi nikkayaṃ._  

</Paragraph>

</Division>
<Division number="2327" book="ja2">

<Paragraph number="3685" type="verses" book="ja2">
_Nikkiṇitvā nahāpetvā,_  
_bhojayitvāna dārake;_  
_Samalaṅkaritvā bhaṇḍena,_  
_ucchaṅge upavesayuṃ._  

</Paragraph>

</Division>
<Division number="2328" book="ja2">

<Paragraph number="3686" type="verses" book="ja2">
_Sīsaṃ nhāte sucivatthe,_  
_sabbābharaṇabhūsite;_  
_Rājā aṅke karitvāna,_  
_ayyako paripucchatha._  

</Paragraph>

</Division>
<Division number="2329" book="ja2">

<Paragraph number="3687" type="verses" book="ja2">
_Kuṇḍale ghusite māle,_  
_sabbābharaṇabhūsite;_  
_Rājā aṅke karitvāna,_  
_idaṃ vacanamabravi._  

</Paragraph>

</Division>
<Division number="2330" book="ja2">

<Paragraph number="3688" type="verses" book="ja2">
_“Kacci ubho arogā te,_  
_jāli mātāpitā tava;_  
_Kacci uñchena yāpenti,_  
_kacci mūlaphalā bahū._  

</Paragraph>

</Division>
<Division number="2331" book="ja2">

<Paragraph number="3689" type="verses" book="ja2">

_appameva sarīsapā;_  
_Vane vāḷamigākiṇṇe,_  
_kacci hiṃsā na vijjati”._  

</Paragraph>

</Division>
<Division number="2332" book="ja2">

<Paragraph number="3690" type="verses" book="ja2">

_deva mātāpitā mama;_  
_Atho uñchena yāpenti,_  
_atho mūlaphalā bahū._  

</Paragraph>

</Division>
<Division number="2333" book="ja2">

<Paragraph number="3691" type="verses" book="ja2">

_appameva sarīsapā;_  
_Vane vāḷamigākiṇṇe,_  
_hiṃsā nesaṃ na vijjati._  

</Paragraph>

</Division>
<Division number="2334" book="ja2">

<Paragraph number="3692" type="verses" book="ja2">
_Khaṇantālukalambāni,_  
_bilāni takkalāni ca;_  
_Kolaṃ bhallātakaṃ bellaṃ,_  
_sā no āhatva posati._  

</Paragraph>

</Division>
<Division number="2335" book="ja2">

<Paragraph number="3693" type="verses" book="ja2">
_Yañceva sā āharati,_  
_vanamūlaphalahāriyā;_  
_Taṃ no sabbe samāgantvā,_  
_rattiṃ bhuñjāma no divā._  

</Paragraph>

</Division>
<Division number="2336" book="ja2">

<Paragraph number="3694" type="verses" book="ja2">
_Ammāva no kisā paṇḍu,_  
_āharantī dumapphalaṃ;_  
_Vātātapena sukhumālī,_  
_padumaṃ hatthagatāmiva._  

</Paragraph>

</Division>
<Division number="2337" book="ja2">

<Paragraph number="3695" type="verses" book="ja2">
_Ammāya patanū kesā,_  
_vicarantyā brahāvane;_  
_Vane vāḷamigākiṇṇe,_  
_khaggadīpinisevite._  

</Paragraph>

</Division>
<Division number="2338" book="ja2">

<Paragraph number="3696" type="verses" book="ja2">
_Kesesu jaṭaṃ bandhitvā,_  
_kacche jallamadhārayi;_  
_Cammavāsī chamā seti,_  
_jātavedaṃ namassati._  

</Paragraph>

</Division>
<Division number="2339" book="ja2">

<Paragraph number="3697" type="verses" book="ja2">
_Puttā piyā manussānaṃ,_  
_lokasmiṃ udapajjisuṃ;_  
_Na hi nūnamhākaṃ ayyassa,_  
_putte sneho ajāyatha”._  

</Paragraph>

</Division>
<Division number="2340" book="ja2">

<Paragraph number="3698" type="verses" book="ja2">
_“Dukkaṭañca hi no putta,_  
_bhūnahaccaṃ kataṃ mayā;_  
_Yohaṃ sivīnaṃ vacanā,_  
_pabbājesimadūsakaṃ._  

</Paragraph>

</Division>
<Division number="2341" book="ja2">

<Paragraph number="3699" type="verses" book="ja2">

_dhanaṃ dhaññañca vijjati;_  
_Etu vessantaro rājā,_  
_siviraṭṭhe pasāsatu”._  

</Paragraph>

</Division>
<Division number="2342" book="ja2">

<Paragraph number="3700" type="verses" book="ja2">
_“Na deva mayhaṃ vacanā,_  
_ehiti sivisuttamo;_  
_Sayameva devo gantvā,_  
_siñca bhogehi atrajaṃ”._  

</Paragraph>

</Division>
<Division number="2343" book="ja2">

<Paragraph number="3701" type="verses" book="ja2">

_sañjayo ajjhabhāsatha;_  
_“Hatthī assā rathā pattī,_  
_senā sannāhayantu naṃ;_  
_Negamā ca maṃ anventu,_  
_brāhmaṇā ca purohitā._  

</Paragraph>

</Division>
<Division number="2344" book="ja2">

_nānāvaṇṇehilaṅkatā._  

</Paragraph>

</Division>
<Division number="2345" book="ja2">

_suddhāneke nivāsitā;_  
_Khippamāyantu sannaddhā,_  
_nānāvaṇṇehilaṅkatā._  

</Paragraph>

</Division>
<Division number="2346" book="ja2">

_pabbato gandhamādano;_  
_Nānārukkhehi sañchanno,_  
_mahābhūtagaṇālayo._  

</Paragraph>

</Division>
<Division number="2347" book="ja2">

_disā bhāti pavāti ca;_  
_Khippamāyantu sannaddhā,_  
_disā bhantu pavantu ca._  

</Paragraph>

</Division>
<Division number="2348" book="ja2">

_hemakappanavāsasā._  

</Paragraph>

</Division>
<Division number="2349" book="ja2">

_hatthikkhandhehi dassitā._  

</Paragraph>

</Division>
<Division number="2350" book="ja2">

_yojayantu catuddasa;_  
_Ājānīyāva jātiyā,_  
_sindhavā sīghavāhanā._  

</Paragraph>

</Division>
<Division number="2351" book="ja2">

_illiyācāpadhāribhi;_  
_Khippamāyantu sannaddhā,_  
_assapiṭṭhehilaṅkatā._  

</Paragraph>

</Division>
<Division number="2352" book="ja2">

_yojayantu catuddasa;_

_suvaṇṇacitapakkhare._  

</Paragraph>

</Division>
<Division number="2353" book="ja2">

_cammāni kavacāni ca;_

_rathesu rathajīvino._  

</Paragraph>

<Division number="2354" book="ja2">

_yena maggena ehiti._  

</Paragraph>

</Paragraph>

<Division number="2357" book="ja2">

_Maggamhi patitiṭṭhantu,_  
_yena maggena ehiti._  

</Paragraph>
