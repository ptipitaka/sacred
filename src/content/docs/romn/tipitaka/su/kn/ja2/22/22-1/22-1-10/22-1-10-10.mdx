<Division number="2267" book="ja2">

<Paragraph number="3622" type="verses" book="ja2">
_Tato ratyā vivasāne,_  
_sūriyassuggamanaṃ pati;_  
_Sakko brāhmaṇavaṇṇena,_  
_pāto tesaṃ adissatha._  

</Paragraph>

</Division>
<Division number="2268" book="ja2">

<Paragraph number="3623" type="verses" book="ja2">
_“Kacci nu bhoto kusalaṃ,_  
_kacci bhoto anāmayaṃ;_  
_Kacci uñchena yāpetha,_  
_kacci mūlaphalā bahū._  

</Paragraph>

</Division>
<Division number="2269" book="ja2">

<Paragraph number="3624" type="verses" book="ja2">
_Kacci ḍaṃsā makasā ca,_  
_appameva sarīsapā;_  
_Vane vāḷamigākiṇṇe,_  
_kacci hiṃsā na vijjati”._  

</Paragraph>

</Division>
<Division number="2270" book="ja2">

<Paragraph number="3625" type="verses" book="ja2">
_“Kusalañceva no brahme,_  
_atho brahme anāmayaṃ;_  
_Atho uñchena yāpema,_  
_atho mūlaphalā bahū._  

</Paragraph>

</Division>
<Division number="2271" book="ja2">

<Paragraph number="3626" type="verses" book="ja2">
_Atho ḍaṃsā makasā ca,_  
_appameva sarīsapā;_  
_Vane vāḷamigākiṇṇe,_  
_hiṃsā mayhaṃ na vijjati._  

</Paragraph>

</Division>
<Division number="2272" book="ja2">

<Paragraph number="3627" type="verses" book="ja2">
_Satta no māse vasataṃ,_  
_araññe jīvasokinaṃ;_  
_Idaṃ dutiyaṃ passāma,_  
_brāhmaṇaṃ devavaṇṇinaṃ;_  
_Ādāya veḷuvaṃ daṇḍaṃ,_  
_dhārentaṃ ajinakkhipaṃ._  

</Paragraph>

</Division>
<Division number="2273" book="ja2">

<Paragraph number="3628" type="verses" book="ja2">
_Svāgataṃ te mahābrahme,_  
_atho me adurāgataṃ;_  
_Anto pavisa bhaddante,_  
_pāde pakkhālayassu te._  

</Paragraph>

</Division>
<Division number="2274" book="ja2">

<Paragraph number="3629" type="verses" book="ja2">
_Tindukāni piyālāni,_  
_madhuke kāsumāriyo;_  
_Phalāni khuddakappāni,_  
_bhuñja brahme varaṃ varaṃ._  

</Paragraph>

</Division>
<Division number="2275" book="ja2">

<Paragraph number="3630" type="verses" book="ja2">
_Idampi pānīyaṃ sītaṃ,_  
_ābhataṃ girigabbharā;_  
_Tato piva mahābrahme,_  
_sace tvaṃ abhikaṅkhasi._  

</Paragraph>

</Division>
<Division number="2276" book="ja2">

<Paragraph number="3631" type="verses" book="ja2">
_Atha tvaṃ kena vaṇṇena,_  
_kena vā pana hetunā;_  
_Anuppatto brahāraññaṃ,_  
_taṃ me akkhāhi pucchito”._  

</Paragraph>

</Division>
<Division number="2277" book="ja2">

<Paragraph number="3632" type="verses" book="ja2">
_“Yathā vārivaho pūro,_  
_sabbakālaṃ na khīyati;_  
_Evaṃ taṃ yācitāgacchiṃ,_  
_bhariyaṃ me dehi yācito”._  

</Paragraph>

</Division>
<Division number="2278" book="ja2">

<Paragraph number="3633" type="verses" book="ja2">
_“Dadāmi na vikampāmi,_  
_yaṃ maṃ yācasi brāhmaṇa;_  
_Santaṃ nappaṭiguyhāmi,_  
_dāne me ramatī mano”._  

</Paragraph>

</Division>
<Division number="2279" book="ja2">

<Paragraph number="3634" type="verses" book="ja2">
_Maddiṃ hatthe gahetvāna,_  
_udakassa kamaṇḍaluṃ;_  
_Brāhmaṇassa adā dānaṃ,_  
_sivīnaṃ raṭṭhavaḍḍhano._  

</Paragraph>

</Division>
<Division number="2280" book="ja2">

<Paragraph number="3635" type="verses" book="ja2">
_Tadāsi yaṃ bhiṃsanakaṃ,_  
_tadāsi lomahaṃsanaṃ;_  
_Maddiṃ paricajantassa,_  
_medanī sampakampatha._  

</Paragraph>

</Division>
<Division number="2281" book="ja2">

<Paragraph number="3636" type="verses" book="ja2">
_Nevassa maddī bhākuṭi,_  
_na sandhīyati na rodati;_  
_Pekkhatevassa tuṇhī sā,_  
_“eso jānāti yaṃ varaṃ”._  

</Paragraph>

<Paragraph number="3637" type="verses" book="ja2">
_(Jāliṃ kaṇhājinaṃ dhītaṃ,_  
_maddideviṃ patibbataṃ;_  
_Cajamāno ca cintesiṃ,_  
_bodhiyāyeva kāraṇā._  

</Paragraph>

<Paragraph number="3638" type="verses" book="ja2">
_Na me dessā ubho puttā,_  
_maddī devī na dessiyā;_  
_Sabbaññutaṃ piyaṃ mayhaṃ,_  
_tasmā piye adāsahaṃ.)_  

</Paragraph>

</Division>
<Division number="2282" book="ja2">

<Paragraph number="3639" type="verses" book="ja2">
_“Komārī yassāhaṃ bhariyā,_  
_sāmiko mama issaro;_  
_Yassicche tassa maṃ dajjā,_  
_vikkiṇeyya haneyya vā”._  

</Paragraph>

</Division>
<Division number="2283" book="ja2">

<Paragraph number="3640" type="verses" book="ja2">
_Tesaṃ saṅkappamaññāya,_  
_devindo etadabravi;_  
_“Sabbe jitā te paccūhā,_  
_ye dibbā ye ca mānusā._  

</Paragraph>

</Division>
<Division number="2284" book="ja2">

<Paragraph number="3641" type="verses" book="ja2">
_Ninnāditā te pathavī,_  
_saddo te tidivaṅgato;_  
_Samantā vijjutā āguṃ,_  
_girīnaṃva patissutā._  

</Paragraph>

</Division>
<Division number="2285" book="ja2">

<Paragraph number="3642" type="verses" book="ja2">
_Tassa te anumodanti,_  
_ubho nāradapabbatā;_  
_Indo ca brahmā pajāpati,_  
_somo yamo vessavaṇo;_  
_Sabbe devānumodanti,_  
_dukkarañhi karoti so._  

</Paragraph>

</Division>
<Division number="2286" book="ja2">

<Paragraph number="3643" type="verses" book="ja2">
_Duddadaṃ dadamānānaṃ,_  
_dukkaraṃ kamma kubbataṃ;_  
_Asanto nānukubbanti,_  
_sataṃ dhammo durannayo._  

</Paragraph>

</Division>
<Division number="2287" book="ja2">

<Paragraph number="3644" type="verses" book="ja2">
_Tasmā satañca asataṃ,_  
_nānā hoti ito gati;_  
_Asanto nirayaṃ yanti,_  
_santo saggaparāyaṇā._  

</Paragraph>

</Division>
<Division number="2288" book="ja2">

<Paragraph number="3645" type="verses" book="ja2">
_Yametaṃ kumāre adā,_  
_bhariyaṃ adā vane vasaṃ;_  
_Brahmayānamanokkamma,_  
_sagge te taṃ vipaccatu”._  

</Paragraph>

</Division>
<Division number="2289" book="ja2">

<Paragraph number="3646" type="verses" book="ja2">
_“Dadāmi bhoto bhariyaṃ,_  
_maddiṃ sabbaṅgasobhanaṃ;_  
_Tvañceva maddiyā channo,_  
_maddī ca patinā saha._  

</Paragraph>

</Division>
<Division number="2290" book="ja2">

<Paragraph number="3647" type="verses" book="ja2">
_Yathā payo ca saṅkho ca,_  
_ubho samānavaṇṇino;_  
_Evaṃ tuvañca maddī ca,_  
_samānamanacetasā._  

</Paragraph>

</Division>
<Division number="2291" book="ja2">

<Paragraph number="3648" type="verses" book="ja2">
_Avaruddhettha araññasmiṃ,_  
_ubho sammatha assame;_  
_Khattiyā gottasampannā,_  
_sujātā mātupettito;_  
_Yathā puññāni kayirātha,_  
_dadantā aparāparaṃ._  

</Paragraph>

</Division>
<Division number="2292" book="ja2">

<Paragraph number="3649" type="verses" book="ja2">
_Sakkohamasmi devindo,_  
_āgatosmi tavantike;_  
_Varaṃ varassu rājisi,_  
_vare aṭṭha dadāmi te”._  

</Paragraph>

</Division>
<Division number="2293" book="ja2">

<Paragraph number="3650" type="verses" book="ja2">
_“Varañce me ado sakka,_  
_sabbabhūtānamissara;_  
_Pitā maṃ anumodeyya,_  
_ito pattaṃ sakaṃ gharaṃ;_  
_Āsanena nimanteyya,_  
_paṭhametaṃ varaṃ vare._  

</Paragraph>

</Division>
<Division number="2294" book="ja2">

<Paragraph number="3651" type="verses" book="ja2">
_Purisassa vadhaṃ na roceyyaṃ,_  
_Api kibbisakārakaṃ;_  
_Vajjhaṃ vadhamhā moceyyaṃ,_  
_Dutiyetaṃ varaṃ vare._  

</Paragraph>

</Division>
<Division number="2295" book="ja2">

<Paragraph number="3652" type="verses" book="ja2">
_Ye vuḍḍhā ye ca daharā,_  
_ye ca majjhimaporisā;_  
_Mameva upajīveyyuṃ,_  
_tatiyetaṃ varaṃ vare._  

</Paragraph>

</Division>
<Division number="2296" book="ja2">

<Paragraph number="3653" type="verses" book="ja2">
_Paradāraṃ na gaccheyyaṃ,_  
_sadārapasuto siyaṃ;_  
_Thīnaṃ vasaṃ na gaccheyyaṃ,_  
_catutthetaṃ varaṃ vare._  

</Paragraph>

</Division>
<Division number="2297" book="ja2">

<Paragraph number="3654" type="verses" book="ja2">
_Putto me sakka jāyetha,_  
_so ca dīghāyuko siyā;_  
_Dhammena jine pathaviṃ,_  
_pañcametaṃ varaṃ vare._  

</Paragraph>

</Division>
<Division number="2298" book="ja2">

<Paragraph number="3655" type="verses" book="ja2">
_Tato ratyā vivasāne,_  
_sūriyassuggamanaṃ pati;_  
_Dibbā bhakkhā pātubhaveyyuṃ,_  
_chaṭṭhametaṃ varaṃ vare._  

</Paragraph>

</Division>
<Division number="2299" book="ja2">

<Paragraph number="3656" type="verses" book="ja2">
_Dadato me na khīyetha,_  
_datvā nānutapeyyahaṃ;_  
_Dadaṃ cittaṃ pasādeyyaṃ,_  
_sattametaṃ varaṃ vare._  

</Paragraph>

</Division>
<Division number="2300" book="ja2">

<Paragraph number="3657" type="verses" book="ja2">
_Ito vimuccamānāhaṃ,_  
_saggagāmī visesagū;_  
_Anivatti tato assaṃ,_  
_aṭṭhametaṃ varaṃ vare”._  

</Paragraph>

</Division>
<Division number="2301" book="ja2">

<Paragraph number="3658" type="verses" book="ja2">
_Tassa taṃ vacanaṃ sutvā,_  
_devindo etadabravi;_  
_“Aciraṃ vata te tato,_  
_pitā taṃ daṭṭhumessati”._  

</Paragraph>

</Division>
<Division number="2302" book="ja2">

<Paragraph number="3659" type="verses" book="ja2">
_Idaṃ vatvāna maghavā,_  
_devarājā sujampati;_  
_Vessantare varaṃ datvā,_  
_saggakāyaṃ apakkami._  


</Paragraph>
</Division>

<Paragraph number="3660" type="center" book="ja2">
Sakkapabbaṃ nāma.
