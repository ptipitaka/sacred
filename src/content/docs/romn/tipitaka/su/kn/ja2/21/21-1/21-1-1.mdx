<Paragraph number="771" type="verses" book="ja2">
_“Sumukha anupacinantā,_  
_pakkamanti vihaṅgamā;_  
_Gaccha tuvampi mā kaṅkhi,_  
_natthi baddhe sahāyatā”._  

</Paragraph>

</Division>
<Division number="2" book="ja2">

<Paragraph number="772" type="verses" book="ja2">
_“Gacche vāhaṃ na vā gacche,_  
_na tena amaro siyaṃ;_  
_Sukhitaṃ taṃ upāsitvā,_  
_dukkhitaṃ taṃ kathaṃ jahe._  

</Paragraph>

</Division>
<Division number="3" book="ja2">

<Paragraph number="773" type="verses" book="ja2">
_Maraṇaṃ vā tayā saddhiṃ,_  
_jīvitaṃ vā tayā vinā;_  
_Tadeva maraṇaṃ seyyo,_  
_yañce jīve tayā vinā._  

</Paragraph>

</Division>
<Division number="4" book="ja2">

<Paragraph number="774" type="verses" book="ja2">
_Nesa dhammo mahārāja,_  
_yaṃ taṃ evaṃ gataṃ jahe;_  
_Yā gati tuyhaṃ sā mayhaṃ,_  
_ruccate vihagādhipa”._  

</Paragraph>

</Division>
<Division number="5" book="ja2">

<Paragraph number="775" type="verses" book="ja2">
_“Kā nu pāsena baddhassa,_  
_gati aññā mahānasā;_  
_Sā kathaṃ cetayānassa,_  
_muttassa tava ruccati._  

</Paragraph>

</Division>
<Division number="6" book="ja2">

<Paragraph number="776" type="verses" book="ja2">
_Kaṃ vā tvaṃ passase atthaṃ,_  
_mama tuyhañca pakkhima;_  
_Ñātīnaṃ vāvasiṭṭhānaṃ,_  
_ubhinnaṃ jīvitakkhaye._  

</Paragraph>

</Division>
<Division number="7" book="ja2">

<Paragraph number="777" type="verses" book="ja2">
_Yaṃ na kañcanadepiñcha,_  
_Andhena tamasā gataṃ;_  
_Tādise sañcajaṃ pāṇaṃ,_  
_Kamatthamabhijotaye”._  

</Paragraph>

</Division>
<Division number="8" book="ja2">

<Paragraph number="778" type="verses" book="ja2">
_“Kathaṃ nu patataṃ seṭṭha,_  
_Dhamme atthaṃ na bujjhasi;_  
_Dhammo apacito santo,_  
_Atthaṃ dasseti pāṇinaṃ._  

</Paragraph>

</Division>
<Division number="9" book="ja2">

<Paragraph number="779" type="verses" book="ja2">
_Sohaṃ dhammaṃ apekkhāno,_  
_dhammā catthaṃ samuṭṭhitaṃ;_  
_Bhattiñca tayi sampassaṃ,_  
_nāvakaṅkhāmi jīvitaṃ._  

</Paragraph>

</Division>
<Division number="10" book="ja2">

<Paragraph number="780" type="verses" book="ja2">
_Addhā eso sataṃ dhammo,_  
_yo mitto mittamāpade;_  
_Na caje jīvitassāpi,_  
_hetudhammamanussaraṃ”._  

</Paragraph>

</Division>
<Division number="11" book="ja2">

<Paragraph number="781" type="verses" book="ja2">
_“Svāyaṃ dhammo ca te ciṇṇo,_  
_bhatti ca viditā mayi;_  
_Kāmaṃ karassu mayhetaṃ,_  
_gacchevānumato mayā._  

</Paragraph>

</Division>
<Division number="12" book="ja2">

<Paragraph number="782" type="verses" book="ja2">
_Api tvevaṃ gate kāle,_  
_yaṃ khaṇḍaṃ ñātinaṃ mayā;_  
_Tayā taṃ buddhisampannaṃ,_  
_assa paramasaṃvutaṃ”._  

</Paragraph>

</Division>
<Division number="13" book="ja2">

<Paragraph number="783" type="verses" book="ja2">
_Iccevaṃ mantayantānaṃ,_  
_Ariyānaṃ ariyavuttinaṃ;_  
_Paccadissatha nesādo,_  
_Āturānamivantako._  

</Paragraph>

</Division>
<Division number="14" book="ja2">

<Paragraph number="784" type="verses" book="ja2">
_Te sattumabhisañcikkha,_  
_dīgharattaṃ hitā dijā;_  
_Tuṇhīmāsittha ubhayo,_  
_na sañcalesumāsanā._  

</Paragraph>

</Division>
<Division number="15" book="ja2">

<Paragraph number="785" type="verses" book="ja2">
_Dhataraṭṭhe ca disvāna,_  
_samuḍḍente tato tato;_  
_Abhikkamatha vegena,_  
_dijasattu dijādhipe._  

</Paragraph>

</Division>
<Division number="16" book="ja2">

<Paragraph number="786" type="verses" book="ja2">
_So ca vegenabhikkamma,_  
_āsajja parame dije;_  
_Paccakamittha nesādo,_  
_baddhā iti vicintayaṃ._  

</Paragraph>

</Division>
<Division number="17" book="ja2">

<Paragraph number="787" type="verses" book="ja2">
_Ekaṃva baddhamāsīnaṃ,_  
_abaddhañca punāparaṃ;_  
_Āsajja baddhamāsīnaṃ,_  
_pekkhamānamadīnavaṃ._  

</Paragraph>

</Division>
<Division number="18" book="ja2">

<Paragraph number="788" type="verses" book="ja2">
_Tato so vimatoyeva,_  
_paṇḍare ajjhabhāsatha;_  
_Pavaḍḍhakāye āsīne,_  
_dijasaṅghagaṇādhipe._  

</Paragraph>

</Division>
<Division number="19" book="ja2">

<Paragraph number="789" type="verses" book="ja2">
_“Yannu pāsena mahatā,_  
_baddho na kurute disaṃ;_  
_Atha kasmā abaddho tvaṃ,_  
_balī pakkhi na gacchasi._  

</Paragraph>

</Division>
<Division number="20" book="ja2">

<Paragraph number="790" type="verses" book="ja2">
_Kiṃ nu tyāyaṃ dijo hoti,_  
_mutto baddhaṃ upāsasi;_  
_Ohāya sakuṇā yanti,_  
_kiṃ eko avahīyasi”._  

</Paragraph>

</Division>
<Division number="21" book="ja2">

<Paragraph number="791" type="verses" book="ja2">
_“Rājā me so dijāmitta,_  
_sakhā pāṇasamo ca me;_  
_Neva naṃ vijahissāmi,_  
_yāva kālassa pariyāyaṃ”._  

</Paragraph>

</Division>
<Division number="22" book="ja2">

<Paragraph number="792" type="verses" book="ja2">
_“Kathaṃ panāyaṃ vihaṅgo,_  
_nāddasa pāsamoḍḍitaṃ;_  
_Padañhetaṃ mahantānaṃ,_  
_boddhumarahanti āpadaṃ”._  

</Paragraph>

</Division>
<Division number="23" book="ja2">

<Paragraph number="793" type="verses" book="ja2">
_“Yadā parābhavo hoti,_  
_poso jīvitasaṅkhaye;_  
_Atha jālañca pāsañca,_  
_āsajjāpi na bujjhati._  

</Paragraph>

</Division>
<Division number="24" book="ja2">

<Paragraph number="794" type="verses" book="ja2">
_Api tveva mahāpañña,_  
_pāsā bahuvidhā tatā;_  
_Guyhamāsajja bajjhanti,_  
_athevaṃ jīvitakkhaye”._  

</Paragraph>

</Division>
<Division number="25" book="ja2">

<Paragraph number="795" type="verses" book="ja2">
_“Api nāyaṃ tayā saddhiṃ,_  
_saṃvāsassa sukhudrayo;_  
_Api no anumaññāsi,_  
_api no jīvitaṃ dade”._  

</Paragraph>

</Division>
<Division number="26" book="ja2">

<Paragraph number="796" type="verses" book="ja2">
_“Na ceva me tvaṃ baddhosi,_  
_napi icchāmi te vadhaṃ;_  
_Kāmaṃ khippamito gantvā,_  
_jīva tvaṃ anigho ciraṃ”._  

</Paragraph>

</Division>
<Division number="27" book="ja2">

<Paragraph number="797" type="verses" book="ja2">
_“Nevāhametamicchāmi,_  
_aññatretassa jīvitā;_  
_Sace ekena tuṭṭhosi,_  
_muñcetaṃ mañca bhakkhaya._  

</Paragraph>

</Division>
<Division number="28" book="ja2">

<Paragraph number="798" type="verses" book="ja2">
_Ārohapariṇāhena,_  
_tulyāsmā vayasā ubho;_  
_Na te lābhena jīvatthi,_  
_etena niminā tuvaṃ._  

</Paragraph>

</Division>
<Division number="29" book="ja2">

<Paragraph number="799" type="verses" book="ja2">
_Tadiṅgha samapekkhassu,_  
_hotu giddhi tavamhasu;_  
_Maṃ pubbe bandha pāsena,_  
_pacchā muñca dijādhipaṃ._  

</Paragraph>

</Division>
<Division number="30" book="ja2">

<Paragraph number="800" type="verses" book="ja2">
_Tāvadeva ca te lābho,_  
_katāssa yācanāya ca;_  
_Mitti ca dhataraṭṭhehi,_  
_yāvajīvāya te siyā”._  

</Paragraph>

</Division>
<Division number="31" book="ja2">

<Paragraph number="801" type="verses" book="ja2">
_“Passantu no mahāsaṃghā,_  
_tayā muttaṃ ito gataṃ;_  
_Mittāmaccā ca bhaccā ca,_  
_puttadārā ca bandhavā._  

</Paragraph>

</Division>
<Division number="32" book="ja2">

<Paragraph number="802" type="verses" book="ja2">
_Na ca te tādisā mittā,_  
_bahūnaṃ idha vijjati;_  
_Yathā tvaṃ dhataraṭṭhassa,_  
_pāṇasādhāraṇo sakhā._  

</Paragraph>

</Division>
<Division number="33" book="ja2">

<Paragraph number="803" type="verses" book="ja2">
_So te sahāyaṃ muñcāmi,_  
_hotu rājā tavānugo;_  
_Kāmaṃ khippamito gantvā,_  
_ñātimajjhe virocatha”._  

</Paragraph>

</Division>
<Division number="34" book="ja2">

<Paragraph number="804" type="verses" book="ja2">
_“So patīto pamuttena,_  
_bhattunā bhattugāravo;_  
_Ajjhabhāsatha vakkaṅgo,_  
_vācaṃ kaṇṇasukhaṃ bhaṇaṃ._  

</Paragraph>

</Division>
<Division number="35" book="ja2">

<Paragraph number="805" type="verses" book="ja2">
_Evaṃ luddaka nandassu,_  
_saha sabbehi ñātibhi;_  
_Yathāhamajja nandāmi,_  
_muttaṃ disvā dijādhipaṃ”._  

</Paragraph>

</Division>
<Division number="36" book="ja2">

<Paragraph number="806" type="verses" book="ja2">
_“Ehi taṃ anusikkhāmi,_  
_Yathā tvamapi lacchase;_  
_Lābhaṃ tavāyaṃ dhataraṭṭho,_  
_Pāpaṃ kiñci na dakkhati._  

</Paragraph>

</Division>
<Division number="37" book="ja2">

<Paragraph number="807" type="verses" book="ja2">

_kāje ubhayato ṭhite._  

</Paragraph>

</Division>

</Paragraph>

</Division>

</Paragraph>

</Division>

_kāje ubhayato ṭhite._

<Paragraph number="812" type="verses" book="ja2">
_“Kathaṃ panime vihaṅgā,_  
_tava hatthattamāgatā;_  
_Kathaṃ luddo mahantānaṃ,_  
_issare idha ajjhagā”._  

</Paragraph>

</Division>
<Division number="43" book="ja2">

<Paragraph number="813" type="verses" book="ja2">
_“Vihitā santime pāsā,_  
_pallalesu janādhipa;_  
_Yaṃ yadāyatanaṃ maññe,_  
_dijānaṃ pāṇarodhanaṃ._  

</Paragraph>

</Division>
<Division number="44" book="ja2">

<Paragraph number="814" type="verses" book="ja2">
_Tādisaṃ pāsamāsajja,_  
_haṃsarājā abajjhatha;_  
_Taṃ abaddho upāsīno,_  
_mamāyaṃ ajjhabhāsatha._  

</Paragraph>

</Division>
<Division number="45" book="ja2">

<Paragraph number="815" type="verses" book="ja2">
_Sudukkaraṃ anariyehi,_  
_dahate bhāvamuttamaṃ;_  
_Bhatturatthe parakkanto,_  
_dhammayutto vihaṅgamo._  

</Paragraph>

</Division>
<Division number="46" book="ja2">

<Paragraph number="816" type="verses" book="ja2">
_Attanāyaṃ cajitvāna,_  
_jīvitaṃ jīvitāraho;_  
_Anutthunanto āsīno,_  
_bhattu yācittha jīvitaṃ._  

</Paragraph>

</Division>
<Division number="47" book="ja2">

_pasādamahamajjhagā;_  
_Tato naṃ pāmuciṃ pāsā,_  
_anuññāsiṃ sukhena ca._  

</Paragraph>

<Division number="55" book="ja2">

<Paragraph number="825" type="verses" book="ja2">

_dijo paramadhammiko;_  
_Mādisassa hi luddassa,_  
_janayeyyātha maddavaṃ._  

</Paragraph>

</Division>
<Division number="56" book="ja2">

<Paragraph number="826" type="verses" book="ja2">

_nāññaṃ passāmi edisaṃ;_  
_Sabbasākuṇikāgāme,_  
_taṃ passa manujādhipa”._  

</Paragraph>

</Division>
<Division number="57" book="ja2">

<Paragraph number="827" type="verses" book="ja2">

_pīṭhe sovaṇṇaye subhe;_  
_Ajjhabhāsatha vakkaṅgo,_  
_vācaṃ kaṇṇasukhaṃ bhaṇaṃ._  

</Paragraph>

</Division>
<Division number="58" book="ja2">

<Paragraph number="828" type="verses" book="ja2">

_kacci bhoto anāmayaṃ;_  
_Kacci raṭṭhamidaṃ phītaṃ,_  
_dhammena manusāsasi”._  

</Paragraph>

</Division>
<Division number="59" book="ja2">

<Paragraph number="829" type="verses" book="ja2">

_Atho haṃsa anāmayaṃ;_  
_Atho raṭṭhamidaṃ phītaṃ,_  
_Dhammena manusāsahaṃ”._  

</Paragraph>

</Division>

<Paragraph number="830" type="verses" book="ja2">

_doso koci na vijjati;_  
_Kacci ca te tavatthesu,_  
_nāvakaṅkhanti jīvitaṃ”._  

</Paragraph>

</Division>
<Division number="61" book="ja2">

<Paragraph number="831" type="verses" book="ja2">

_nāvakaṅkhanti jīvitaṃ”._  

</Paragraph>

</Division>
<Division number="62" book="ja2">

<Paragraph number="832" type="verses" book="ja2">

_tava chandavasānugā”._  

</Paragraph>

</Division>
<Division number="63" book="ja2">

<Paragraph number="833" type="verses" book="ja2">

_assavā piyabhāṇinī;_  
_Puttarūpayasūpetā,_  
_mama chandavasānugā._  

</Paragraph>

</Division>

_Dukkhamāpajji vipulaṃ,_  
_tasmiṃ paṭhamamāpade._  

</Paragraph>

</Division>
<Division number="65" book="ja2">

<Paragraph number="835" type="verses" book="ja2">

_Evametesaṃ jammānaṃ,_  
_pātikaṃ bhavati tāvade”._  

</Paragraph>

</Division>
<Division number="66" book="ja2">

<Paragraph number="836" type="verses" book="ja2">

_evamāpadiyā sati;_  
_Na cāyaṃ kiñci rasmāsu,_  
_sattūva samapajjatha._  

</Paragraph>

</Division>
<Division number="67" book="ja2">

_pubbeva ajjhabhāsatha;_  
_Tadāyaṃ sumukhoyeva,_  
_paṇḍito paccabhāsatha._  

</Paragraph>

</Division>

_anuññāsi sukhena ca._  

</Paragraph>

</Division>
<Division number="69" book="ja2">

<Paragraph number="839" type="verses" book="ja2">

_etadatthāya cintitaṃ;_  
_Bhoto sakāsegamanaṃ,_  
_etassa dhanamicchatā”._  

</Paragraph>

</Division>
<Division number="70" book="ja2">

_patīto casmi dassanā;_  
_Eso cāpi bahuṃ vittaṃ,_  
_labhataṃ yāvadicchati”._  

</Paragraph>

</Division>
<Division number="71" book="ja2">

<Paragraph number="841" type="verses" book="ja2">

_bhogehi manujādhipo;_  
_Ajjhabhāsatha vakkaṅgaṃ,_  
_vācaṃ kaṇṇasukhaṃ bhaṇaṃ._  

</Paragraph>

</Division>
<Division number="72" book="ja2">

<Paragraph number="842" type="verses" book="ja2">
