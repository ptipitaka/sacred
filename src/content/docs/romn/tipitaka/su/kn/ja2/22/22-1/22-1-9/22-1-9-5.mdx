<Division number="1458" book="ja2">

<Paragraph number="2796" type="verses" book="ja2">
_“Ehi dāni gamissāma,_  
_dinno no issarena me;_  
_Mamevatthaṃ paṭipajja,_  
_esa dhammo sanantano”._  

</Paragraph>

</Division>
<Division number="1459" book="ja2">

<Paragraph number="2797" type="verses" book="ja2">
_“Jānāmi māṇava tayāhamasmi,_  
_Dinnohamasmi tava issarena;_  
_Tīhañca taṃ vāsayemu agāre,_  
_Yenaddhunā anusāsemu putte”._  

</Paragraph>

</Division>
<Division number="1460" book="ja2">

<Paragraph number="2798" type="verses" book="ja2">
_“Taṃ me tathā hotu vasemu tīhaṃ,_  
_Kurutaṃ bhavajja gharesu kiccaṃ;_  
_Anusāsataṃ puttadāre bhavajja,_  
_Yathā tayī pecca sukhī bhaveyya”._  

</Paragraph>

</Division>
<Division number="1461" book="ja2">

<Paragraph number="2799" type="verses" book="ja2">
_“Sādhū”ti vatvāna pahūtakāmo,_  
_Pakkāmi yakkho vidhurena saddhiṃ;_  
_Taṃ kuñjarājaññahayānuciṇṇaṃ,_  
_Pāvekkhi antepuramariyaseṭṭho._  

</Paragraph>

</Division>
<Division number="1462" book="ja2">

<Paragraph number="2800" type="verses" book="ja2">
_Koñcaṃ mayūrañca piyañca ketaṃ,_  
_Upāgami tattha surammarūpaṃ;_  
_Pahūtabhakkhaṃ bahuannapānaṃ,_  
_Masakkasāraṃ viya vāsavassa._  

</Paragraph>

</Division>
<Division number="1463" book="ja2">

<Paragraph number="2801" type="verses" book="ja2">
_Tattha naccanti gāyanti,_  
_avhāyanti varāvaraṃ;_  
_Accharā viya devesu,_  
_nāriyo samalaṅkatā._  

</Paragraph>

</Division>
<Division number="1464" book="ja2">

<Paragraph number="2802" type="verses" book="ja2">
_Samaṅgikatvā pamadāhi yakkhaṃ,_  
_Annena pānena ca dhammapālo;_  
_Atthattha mevānuvicintayanto,_  
_Pāvekkhi bhariyāya tadā sakāse._  

</Paragraph>

</Division>
<Division number="1465" book="ja2">

<Paragraph number="2803" type="verses" book="ja2">
_Taṃ candanagandharasānulittaṃ,_  
_Suvaṇṇajambonadanikkhasādisaṃ;_  
_Bhariyaṃvacā “ehi suṇohi bhoti,_  
_Puttāni āmantaya tambanette”._  

</Paragraph>

</Division>
<Division number="1466" book="ja2">

<Paragraph number="2804" type="verses" book="ja2">
_Sutvāna vākyaṃ patino anujjā,_  
_Suṇisaṃvaca tambanakhiṃ sunettaṃ;_  
_“Āmantaya vammadharāni cete,_  
_Puttāni indīvarapupphasāme”._  

</Paragraph>

</Division>
<Division number="1467" book="ja2">

<Paragraph number="2805" type="verses" book="ja2">
_Te āgate muddhani dhammapālo,_  
_Cumbitvā putte avikampamāno;_  
_Āmantayitvāna avoca vākyaṃ,_  
_“Dinnāhaṃ raññā idha māṇavassa._  

</Paragraph>

</Division>
<Division number="1468" book="ja2">

<Paragraph number="2806" type="verses" book="ja2">
_Tassajjahaṃ attasukhī vidheyyo,_  
_Ādāya yenicchati tena gacchati;_  
_Ahañca vo sāsitumāgatosmi,_  
_Kathaṃ ahaṃ aparittāya gacche._  

</Paragraph>

</Division>
<Division number="1469" book="ja2">

<Paragraph number="2807" type="verses" book="ja2">
_Sace vo rājā kururaṭṭhavāsī,_  
_Janasandho puccheyya pahūtakāmo;_  
_Kimābhijānātha pure purāṇaṃ,_  
_Kiṃ vo pitā anusāse puratthā._  

</Paragraph>

</Division>
<Division number="1470" book="ja2">

<Paragraph number="2808" type="verses" book="ja2">
_Samāsanā hotha mayāva sabbe,_  
_Konīdha rañño abbhatiko manusso;_  
_Tamañjaliṃ kariya vadetha evaṃ,_  
_Mā hevaṃ deva na hi esa dhammo;_  
_Viyaggharājassa nihīnajacco,_  
_Samāsano deva kathaṃ bhaveyya”._  


</Paragraph>
</Division>

<Paragraph number="2809" type="center" book="ja2">
Lakkhaṇakaṇḍaṃ nāma.
</Paragraph>
