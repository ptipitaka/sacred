<Paragraph number="98" type="verses" book="ja2">
_Atha bravi brahā indo,_  
_vatrabhū jayataṃ pitā;_  
_Devakaññaṃ parābhetvā,_  
_sudhammāyaṃ alambusaṃ._  

</Paragraph>

</Division>
<Division number="96" book="ja2" bookNo="23" page="12">

<Paragraph number="99" type="verses" book="ja2">
_“Misse devā taṃ yācanti,_  
_tāvatiṃsā saindakā;_  
_Isippalobhane gaccha,_  
_isisiṅgaṃ alambuse._  

</Paragraph>

</Division>
<Division number="97" book="ja2" bookNo="23" page="12">

<Paragraph number="100" type="verses" book="ja2">
_Purāyaṃ amhe acceti,_  
_vattavā brahmacariyavā;_  
_Nibbānābhirato vuddho,_  
_tassa maggāni āvara”._  

</Paragraph>

</Division>
<Division number="98" book="ja2" bookNo="23" page="12">

<Paragraph number="101" type="verses" book="ja2">
_“Devarāja kimeva tvaṃ,_  
_mameva tuvaṃ sikkhasi;_  
_Isippalobhane gaccha,_  
_santi aññāpi accharā._  

</Paragraph>

</Division>
<Division number="99" book="ja2" bookNo="23" page="13">

<Paragraph number="102" type="verses" book="ja2">
_Mādisiyo pavarā ceva,_  
_asoke nandane vane;_  
_Tāsampi hotu pariyāyo,_  
_tāpi yantu palobhanā”._  

</Paragraph>

</Division>
<Division number="100" book="ja2" bookNo="23" page="13">

<Paragraph number="103" type="verses" book="ja2">
_“Addhā hi saccaṃ bhaṇasi,_  
_santi aññāpi accharā;_  
_Tādisiyo pavarā ceva,_  
_asoke nandane vane._  

</Paragraph>

</Division>
<Division number="101" book="ja2" bookNo="23" page="13">

<Paragraph number="104" type="verses" book="ja2">
_Na tā evaṃ pajānanti,_  
_pāricariyaṃ pumaṃ gatā;_  
_Yādisaṃ tvaṃ pajānāsi,_  
_nāri sabbaṅgasobhane._  

</Paragraph>

</Division>
<Division number="102" book="ja2" bookNo="23" page="13">

<Paragraph number="105" type="verses" book="ja2">
_Tvameva gaccha kalyāṇi,_  
_itthīnaṃ pavarā casi;_  
_Taveva vaṇṇarūpena,_  
_savasamānayissasi”._  

</Paragraph>

</Division>
<Division number="103" book="ja2" bookNo="23" page="13">

<Paragraph number="106" type="verses" book="ja2">
_“Na vāhaṃ na gamissāmi,_  
_devarājena pesitā;_  
_Vibhemi cetaṃ āsāduṃ,_  
_uggatejo hi brāhmaṇo._  

</Paragraph>

</Division>
<Division number="104" book="ja2" bookNo="23" page="13">

<Paragraph number="107" type="verses" book="ja2">
_Aneke nirayaṃ pattā,_  
_isimāsādiyā janā;_  
_Āpannā mohasaṃsāraṃ,_  
_tasmā lomāni haṃsaye”._  

</Paragraph>

</Division>
<Division number="105" book="ja2" bookNo="23" page="13">

<Paragraph number="108" type="verses" book="ja2">
_Idaṃ vatvāna pakkāmi,_  
_accharā kāmavaṇṇinī;_  
_Missā missitumicchantī,_  
_isisiṅgaṃ alambusā._  

</Paragraph>

</Division>
<Division number="106" book="ja2" bookNo="23" page="13">

<Paragraph number="109" type="verses" book="ja2">
_Sā ca taṃ vanamogayha,_  
_isisiṅgena rakkhitaṃ;_  
_Bimbajālakasañchannaṃ,_  
_samantā addhayojanaṃ._  

</Paragraph>

</Division>
<Division number="107" book="ja2" bookNo="23" page="13">

<Paragraph number="110" type="verses" book="ja2">
_Pātova pātarāsamhi,_  
_udaṇhasamayaṃ pati;_  
_Aggiṭṭhaṃ parimajjantaṃ,_  
_isisiṅgaṃ upāgami._  

</Paragraph>

</Division>
<Division number="108" book="ja2" bookNo="23" page="13">

<Paragraph number="111" type="verses" book="ja2">
_“Kā nu vijjurivābhāsi,_  
_osadhī viya tārakā;_  
_Vicittahatthābharaṇā,_  
_āmuttamaṇikuṇḍalā._  

</Paragraph>

</Division>
<Division number="109" book="ja2" bookNo="23" page="13">

<Paragraph number="112" type="verses" book="ja2">
_Ādiccavaṇṇasaṅkāsā,_  
_hemacandanagandhinī;_  
_Saññatūrū mahāmāyā,_  
_kumārī cārudassanā._  

</Paragraph>

</Division>
<Division number="110" book="ja2" bookNo="23" page="13">

<Paragraph number="113" type="verses" book="ja2">
_Vilaggā mudukā suddhā,_  
_pādā te suppatiṭṭhitā;_  
_Gamanā kāmanīyā te,_  
_harantiyeva me mano._  

</Paragraph>

</Division>
<Division number="111" book="ja2" bookNo="23" page="14">

<Paragraph number="114" type="verses" book="ja2">
_Anupubbāva te ūrū,_  
_nāganāsasamūpamā;_  
_Vimaṭṭhā tuyhaṃ sussoṇī,_  
_akkhassa phalakaṃ yathā._  

</Paragraph>

</Division>
<Division number="112" book="ja2" bookNo="23" page="14">

<Paragraph number="115" type="verses" book="ja2">
_Uppalasseva kiñjakkhā,_  
_nābhi te sādhu saṇṭhitā;_  
_Pūrā kaṇhañjanasseva,_  
_dūrato paṭidissati._  

</Paragraph>

</Division>
<Division number="113" book="ja2" bookNo="23" page="14">

<Paragraph number="116" type="verses" book="ja2">
_Duvidhā jātā urajā,_  
_avaṇṭā sādhupaccudā;_  
_Payodharā apatitā,_  
_aḍḍhalābusamā thanā._  

</Paragraph>

</Division>
<Division number="114" book="ja2" bookNo="23" page="14">

<Paragraph number="117" type="verses" book="ja2">
_Dīghā kambutalābhāsā,_  
_gīvā eṇeyyakā yathā;_  
_Paṇḍarāvaraṇā vaggu,_  
_catutthamanasannibhā._  

</Paragraph>

</Division>
<Division number="115" book="ja2" bookNo="23" page="14">

<Paragraph number="118" type="verses" book="ja2">
_Uddhaggā ca adhaggā ca,_  
_dumaggaparimajjitā;_  
_Duvijā nelasambhūtā,_  
_dantā tava sudassanā._  

</Paragraph>

</Division>
<Division number="116" book="ja2" bookNo="23" page="14">

<Paragraph number="119" type="verses" book="ja2">
_Apaṇḍarā lohitantā,_  
_jiñjūkaphalasannibhā;_  
_Āyatā ca visālā ca,_  
_nettā tava sudassanā._  

</Paragraph>

</Division>
<Division number="117" book="ja2" bookNo="23" page="14">

<Paragraph number="120" type="verses" book="ja2">
_Nātidīghā susammaṭṭhā,_  
_kanakabyā samocitā;_  
_Uttamaṅgaruhā tuyhaṃ,_  
_kesā candanagandhikā._  

</Paragraph>

</Division>
<Division number="118" book="ja2" bookNo="23" page="14">

<Paragraph number="121" type="verses" book="ja2">
_Yāvatā kasigorakkhā,_  
_vāṇijānaṃ ca yā gati;_  
_Isīnañca parakkantaṃ,_  
_saññatānaṃ tapassinaṃ._  

</Paragraph>

</Division>
<Division number="119" book="ja2" bookNo="23" page="14">

<Paragraph number="122" type="verses" book="ja2">
_Na te samasamaṃ passe,_  
_asmiṃ pathavimaṇḍale;_  
_Ko vā tvaṃ kassa vā putto,_  
_kathaṃ jānemu taṃ mayaṃ”._  

</Paragraph>

</Division>
<Division number="120" book="ja2" bookNo="23" page="14">

<Paragraph number="123" type="verses" book="ja2">
_“Na pañhakālo bhaddante,_  
_kassapevaṃ gate sati;_  
_Ehi samma ramissāma,_  
_ubho asmākamassame;_  
_Ehi taṃ upagūhissaṃ,_  
_ratīnaṃ kusalo bhava”._  

</Paragraph>

</Division>
<Division number="121" book="ja2" bookNo="23" page="14">

<Paragraph number="124" type="verses" book="ja2">
_Idaṃ vatvāna pakkāmi,_  
_accharā kāmavaṇṇinī;_  
_Missā missitumicchantī,_  
_isisiṅgaṃ alambusā._  

</Paragraph>

</Division>
<Division number="122" book="ja2" bookNo="23" page="14">

<Paragraph number="125" type="verses" book="ja2">
_So ca vegena nikkhamma,_  
_Chetvā dandhaparakkamaṃ;_  
_Tamuttamāsu veṇīsu,_  
_Ajjhappatto parāmasi._  

</Paragraph>

</Division>
<Division number="123" book="ja2" bookNo="23" page="15">

<Paragraph number="126" type="verses" book="ja2">
_Tamudāvatta kalyāṇī,_  
_palissaji susobhanā;_  
_Cavitamhi brahmacariyā,_  
_yathā taṃ atha tositā._  

</Paragraph>

</Division>
<Division number="124" book="ja2" bookNo="23" page="15">

<Paragraph number="127" type="verses" book="ja2">
_Manasā agamā indaṃ,_  
_vasantaṃ nandane vane;_  
_Tassā saṅkappamaññāya,_  
_maghavā devakuñjaro._  

</Paragraph>

</Division>
<Division number="125" book="ja2" bookNo="23" page="15">

<Paragraph number="128" type="verses" book="ja2">
_Pallaṅkaṃ pahiṇī khippaṃ,_  
_sovaṇṇaṃ sopavāhanaṃ;_  
_Sauttaracchadapaññāsaṃ,_  
_sahassapaṭiyatthataṃ._  

</Paragraph>

</Division>
<Division number="126" book="ja2" bookNo="23" page="15">

<Paragraph number="129" type="verses" book="ja2">
_Tamenaṃ tattha dhāresi,_  
_ure katvāna sobhanā;_  
_Yathā ekamuhuttaṃva,_  
_tīṇi vassāni dhārayi._  

</Paragraph>

</Division>
<Division number="127" book="ja2" bookNo="23" page="15">

<Paragraph number="130" type="verses" book="ja2">
_Vimado tīhi vassehi,_  
_pabujjhitvāna brāhmaṇo;_  
_Addasāsi haritarukkhe,_  
_samantā aggiyāyanaṃ._  

</Paragraph>

</Division>
<Division number="128" book="ja2" bookNo="23" page="15">

<Paragraph number="131" type="verses" book="ja2">
_Navapattavanaṃ phullaṃ,_  
_kokilaggaṇaghositaṃ;_  
_Samantā paviloketvā,_  
_rudaṃ assūni vattayi._  

</Paragraph>

</Division>
<Division number="129" book="ja2" bookNo="23" page="15">

<Paragraph number="132" type="verses" book="ja2">
_“Na juhe na jape mante,_  
_aggihuttaṃ pahāpitaṃ;_  
_Ko nu me pāricariyāya,_  
_pubbe cittaṃ palobhayi._  

</Paragraph>

</Division>
<Division number="130" book="ja2" bookNo="23" page="15">

<Paragraph number="133" type="verses" book="ja2">
_Araññe me viharato,_  
_yo me tejā ha sambhutaṃ;_  
_Nānāratnaparipūraṃ,_  
_nāvaṃva gaṇhi aṇṇave”._  

</Paragraph>

</Division>
<Division number="131" book="ja2" bookNo="23" page="15">

<Paragraph number="134" type="verses" book="ja2">
_“Ahaṃ te pāricariyāya,_  
_devarājena pesitā;_  
_Avadhiṃ cittaṃ cittena,_  
_pamādo tvaṃ na bujjhasi”._  

</Paragraph>

</Division>
<Division number="132" book="ja2" bookNo="23" page="15">

<Paragraph number="135" type="verses" book="ja2">
_“Imāni kira maṃ tāto,_  
_kassapo anusāsati;_  
_Kamalāsadisitthiyo,_  
_tāyo bujjhesi māṇava._  

</Paragraph>

</Division>
<Division number="133" book="ja2" bookNo="23" page="15">

<Paragraph number="136" type="verses" book="ja2">
_Uregaṇḍāyo bujjhesi,_  
_tāyo bujjhesi māṇava;_  
_Iccānusāsi maṃ tāto,_  
_yathā maṃ anukampako._  

</Paragraph>

</Division>
<Division number="134" book="ja2" bookNo="23" page="15">

<Paragraph number="137" type="verses" book="ja2">
_Tassāhaṃ vacanaṃ nākaṃ,_  
_pitu vuddhassa sāsanaṃ;_  
_Araññe nimmanussamhi,_  
_svajja jhāyāmi ekako._  

</Paragraph>

</Division>
<Division number="135" book="ja2" bookNo="23" page="16">

<Paragraph number="138" type="verses" book="ja2">
_Sohaṃ tathā karissāmi,_  
_dhiratthu jīvitena me;_  
_Puna vā tādiso hessaṃ,_  
_maraṇaṃ me bhavissati”._  

</Paragraph>

</Division>
<Division number="136" book="ja2" bookNo="23" page="16">

<Paragraph number="139" type="verses" book="ja2">
_Tassa tejaṃ vīriyañca,_  
_Dhitiṃ ñatvā avaṭṭhitaṃ;_  
_Sirasā aggahī pāde,_  
_Isisiṅgaṃ alambusā._  

</Paragraph>

</Division>
<Division number="137" book="ja2" bookNo="23" page="16">

<Paragraph number="140" type="verses" book="ja2">
_“Mā me kujjha mahāvīra,_  
_mā me kujjha mahāise;_  
_Mahā attho mayā ciṇṇo,_  
_tidasānaṃ yasassinaṃ;_  
_Tayā saṅkampitaṃ āsi,_  
_sabbaṃ devapuraṃ tadā”._  

</Paragraph>

</Division>
<Division number="138" book="ja2" bookNo="23" page="16">

<Paragraph number="141" type="verses" book="ja2">
_“Tāvatiṃsā ca ye devā,_  
_tidasānañca vāsavo;_  
_Tvañca bhadde sukhī hohi,_  
_gaccha kaññe yathāsukhaṃ”._  

</Paragraph>

</Division>
<Division number="139" book="ja2" bookNo="23" page="16">

<Paragraph number="142" type="verses" book="ja2">
_Tassa pāde gahetvāna,_  
_katvā ca naṃ padakkhiṇaṃ;_  
_Añjaliṃ paggahetvāna,_  
_tamhā ṭhānā apakkami._  

</Paragraph>

</Division>
<Division number="140" book="ja2" bookNo="23" page="16">

<Paragraph number="143" type="verses" book="ja2">
_Yo ca tassāsi pallaṅko,_  
_sovaṇṇo sopavāhano;_  
_Sauttaracchadapaññāso,_  
_sahassapaṭiyatthato;_  
_Tameva pallaṅkamāruyha,_  
_agā devāna santike._  

</Paragraph>

</Division>
<Division number="141" book="ja2" bookNo="23" page="16">

<Paragraph number="144" type="verses" book="ja2">
_Tamokkamiva āyantiṃ,_  
_jalantiṃ vijjutaṃ yathā;_  
_Patīto sumano vitto,_  
_devindo adadā varaṃ._  

</Paragraph>

</Division>
<Division number="142" book="ja2" bookNo="23" page="16">

<Paragraph number="145" type="verses" book="ja2">
_“Varañce me ado sakka,_  
_sabbabhūtānamissara;_  
_Nisippalobhikā gacche,_  
_etaṃ sakka varaṃ vare”ti._  


</Paragraph>
</Division>
