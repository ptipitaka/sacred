<Paragraph number="147" type="verses" book="ja2">
_“Ariyāvakāsosi pasannanetto,_  
_Maññe bhavaṃ pabbajito kulamhā;_  
_Kathaṃ nu vittāni pahāya bhoge,_  
_Pabbaji nikkhamma gharā sapañña”._  

</Paragraph>

</Division>
<Division number="144" book="ja2" bookNo="23" page="17">

<Paragraph number="148" type="verses" book="ja2">
_“Sayaṃ vimānaṃ naradeva disvā,_  
_Mahānubhāvassa mahoragassa;_  
_Disvāna puññāna mahāvipākaṃ,_  
_Saddhāyahaṃ pabbajitomhi rāja”._  

</Paragraph>

</Division>
<Division number="145" book="ja2" bookNo="23" page="17">

<Paragraph number="149" type="verses" book="ja2">
_“Na kāmakāmā na bhayā na dosā,_  
_Vācaṃ musā pabbajitā bhaṇanti;_  
_Akkhāhi me pucchito etamatthaṃ,_  
_Sutvāna me jāyihitippasādo”._  

</Paragraph>

</Division>

<Paragraph number="150" type="verses" book="ja2">
_“Vāṇijja raṭṭhādhipa gacchamāno,_  
_Pathe addasāsimhi bhojaputte;_  
_Pavaddhakāyaṃ uragaṃ mahantaṃ,_  
_Ādāya gacchante pamodamāne._  

</Paragraph>

</Division>
<Division number="147" book="ja2" bookNo="23" page="17">

<Paragraph number="151" type="verses" book="ja2">
_Sohaṃ samāgamma janinda tehi,_  
_Pahaṭṭhalomo avacamhi bhīto;_  
_‘Kuhiṃ ayaṃ nīyati bhīmakāyo,_  
_Nāgena kiṃ kāhatha bhojaputtā’._  

</Paragraph>

</Division>
<Division number="148" book="ja2" bookNo="23" page="17">

<Paragraph number="152" type="verses" book="ja2">
_‘Nāgo ayaṃ nīyati bhojanatthā,_  
_Pavaddhakāyo urago mahanto;_  
_Sāduñca thūlañca muduñca maṃsaṃ,_  
_Na tvaṃ rasaññāsi videhaputta._  

</Paragraph>

</Division>
<Division number="149" book="ja2" bookNo="23" page="17">

<Paragraph number="153" type="verses" book="ja2">
_Ito mayaṃ gantvā sakaṃ niketaṃ,_  
_Ādāya satthāni vikopayitvā;_  
_Maṃsāni bhokkhāma pamodamānā,_  
_Mayañhi ve sattavo pannagānaṃ’._  

</Paragraph>

</Division>
<Division number="150" book="ja2" bookNo="23" page="17">

<Paragraph number="154" type="verses" book="ja2">
_‘Sace ayaṃ nīyati bhojanatthā,_  
_Pavaddhakāyo urago mahanto;_  
_Dadāmi vo balibaddāni soḷasa,_  
_Nāgaṃ imaṃ muñcatha bandhanasmā’._  

</Paragraph>

</Division>
<Division number="151" book="ja2" bookNo="23" page="18">

<Paragraph number="155" type="verses" book="ja2">
_‘Addhā hi no bhakkho ayaṃ manāpo,_  
_Bahū ca no uragā bhuttapubbā;_  
_Karoma te taṃ vacanaṃ aḷāra,_  
_Mittañca no hohi videhaputta’._  

</Paragraph>

</Division>
<Division number="152" book="ja2" bookNo="23" page="18">

<Paragraph number="156" type="verses" book="ja2">
_Tadāssu te bandhanā mocayiṃsu,_  
_Yaṃ natthuto paṭimokkassa pāse;_  
_Mutto ca so bandhanā nāgarājā,_  
_Pakkāmi pācīnamukho muhuttaṃ._  

</Paragraph>

</Division>
<Division number="153" book="ja2" bookNo="23" page="18">

<Paragraph number="157" type="verses" book="ja2">
_Gantvāna pācīnamukho muhuttaṃ,_  
_Puṇṇehi nettehi palokayī maṃ;_  
_Tadāssahaṃ piṭṭhito anvagacchiṃ,_  
_Dasaṅguliṃ añjaliṃ paggahetvā._  

</Paragraph>

</Division>
<Division number="154" book="ja2" bookNo="23" page="18">

<Paragraph number="158" type="verses" book="ja2">
_‘Gaccheva kho tvaṃ taramānarūpo,_  
_Mā taṃ amittā punaraggahesuṃ;_  
_Dukkho hi luddehi punā samāgamo,_  
_Adassanaṃ bhojaputtāna gaccha’._  

</Paragraph>

</Division>
<Division number="155" book="ja2" bookNo="23" page="18">

<Paragraph number="159" type="verses" book="ja2">
_Agamāsi so rahadaṃ vippasannaṃ,_  
_Nīlobhāsaṃ ramaṇīyaṃ sutitthaṃ;_  
_Samotataṃ jambuhi vetasāhi,_  
_Pāvekkhi nittiṇṇabhayo patīto._  

</Paragraph>

</Division>
<Division number="156" book="ja2" bookNo="23" page="18">

<Paragraph number="160" type="verses" book="ja2">
_So taṃ pavissa na cirassa nāgo,_  
_Dibbena me pāturahuṃ janinda;_  
_Upaṭṭhahī maṃ pitaraṃva putto,_  
_Hadayaṅgamaṃ kaṇṇasukhaṃ bhaṇanto._  

</Paragraph>

</Division>
<Division number="157" book="ja2" bookNo="23" page="19">

<Paragraph number="161" type="verses" book="ja2">
_‘Tvaṃ mesi mātā ca pitā aḷāra,_  
_Abbhantaro pāṇadado sahāyo;_  
_Sakañca iddhiṃ paṭilābhakosmi,_  
_Aḷāra passa me nivesanāni;_  
_Pahūtabhakkhaṃ bahuannapānaṃ,_  
_Masakkasāraṃ viya vāsavassa._  

</Paragraph>

</Division>
<Division number="158" book="ja2" bookNo="23" page="19">

<Paragraph number="162" type="verses" book="ja2">
_Taṃ bhūmibhāgehi upetarūpaṃ,_  
_Asakkharā ceva mudū subhā ca;_  
_Nīcattiṇā apparajā ca bhūmi,_  
_Pāsādikā yattha jahanti sokaṃ._  

</Paragraph>

</Division>
<Division number="159" book="ja2" bookNo="23" page="19">

<Paragraph number="163" type="verses" book="ja2">
_Anāvakulā veḷuriyūpanīlā,_  
_Catuddisaṃ ambavanaṃ surammaṃ;_  
_Pakkā ca pesī ca phalā suphullā,_  
_Niccotukā dhārayantī phalāni’._  

</Paragraph>

</Division>
<Division number="160" book="ja2" bookNo="23" page="19">

<Paragraph number="164" type="verses" book="ja2">
_Tesaṃ vanānaṃ naradeva majjhe,_  
_Nivesanaṃ bhassarasannikāsaṃ;_  
_Rajataggaḷaṃ sovaṇṇamayaṃ uḷāraṃ,_  
_Obhāsatī vijjurivantalikkhe._  

</Paragraph>

</Division>
<Division number="161" book="ja2" bookNo="23" page="19">

<Paragraph number="165" type="verses" book="ja2">
_Maṇīmayā soṇṇamayā uḷārā,_  
_Anekacittā satataṃ sunimmitā;_  
_Paripūrā kaññāhi alaṅkatābhi,_  
_Suvaṇṇakāyūradharāhi rāja._  

</Paragraph>

</Division>
<Division number="162" book="ja2" bookNo="23" page="19">

<Paragraph number="166" type="verses" book="ja2">
_So saṅkhapālo taramānarūpo,_  
_Pāsādamāruyha anomavaṇṇo;_  
_Sahassathambhaṃ atulānubhāvaṃ,_  
_Yatthassa bhariyā mahesī ahosi._  

</Paragraph>

</Division>
<Division number="163" book="ja2" bookNo="23" page="20">

<Paragraph number="167" type="verses" book="ja2">
_Ekā ca nārī taramānarūpā,_  
_Ādāya veḷuriyamayaṃ mahagghaṃ;_  
_Subhaṃ maṇiṃ jātimantūpapannaṃ,_  
_Acoditā āsanamabbhihāsi._  

</Paragraph>

</Division>
<Division number="164" book="ja2" bookNo="23" page="20">

<Paragraph number="168" type="verses" book="ja2">
_Tato maṃ urago hatthe gahetvā,_  
_Nisīdayī pāmukhaāsanasmiṃ;_  
_‘Idamāsanaṃ atra bhavaṃ nisīdatu,_  
_Bhavañhi me aññataro garūnaṃ’._  

</Paragraph>

</Division>
<Division number="165" book="ja2" bookNo="23" page="20">

<Paragraph number="169" type="verses" book="ja2">
_Aññā ca nārī taramānarūpā,_  
_Ādāya vāriṃ upasaṅkamitvā;_  
_Pādāni pakkhālayī me janinda,_  
_Bhariyāva bhattū patino piyassa._  

</Paragraph>

</Division>
<Division number="166" book="ja2" bookNo="23" page="20">

<Paragraph number="170" type="verses" book="ja2">
_Aparā ca nārī taramānarūpā,_  
_Paggayha sovaṇṇamayāya pātiyā;_  
_Anekasūpaṃ vividhaṃ viyañjanaṃ,_  
_Upanāmayī bhatta manuññarūpaṃ._  

</Paragraph>

</Division>
<Division number="167" book="ja2" bookNo="23" page="20">

<Paragraph number="171" type="verses" book="ja2">
_Turiyehi maṃ bhārata bhuttavantaṃ,_  
_Upaṭṭhahuṃ bhattu mano viditvā;_  
_Tatuttariṃ maṃ nipatī mahantaṃ,_  
_Dibbehi kāmehi anappakehi._  

</Paragraph>

</Division>
<Division number="168" book="ja2" bookNo="23" page="20">

<Paragraph number="172" type="verses" book="ja2">
_‘Bhariyā mametā tisatā aḷāra,_  
_Sabbattamajjhā padumuttarābhā;_  
_Aḷāra etāssu te kāmakārā,_  
_Dadāmi te tā paricārayassu’._  

</Paragraph>

</Division>
<Division number="169" book="ja2" bookNo="23" page="20">

<Paragraph number="173" type="verses" book="ja2">
_Saṃvaccharaṃ dibbarasānubhutvā,_  
_Tadāssuhaṃ uttarimajjhabhāsiṃ;_  
_‘Nāgassidaṃ kinti kathañca laddhaṃ,_  
_Kathajjhagamāsi vimānaseṭṭhaṃ._  

</Paragraph>

</Division>
<Division number="170" book="ja2" bookNo="23" page="21">

<Paragraph number="174" type="verses" book="ja2">
_Adhiccaladdhaṃ pariṇāmajaṃ te,_  
_Sayaṃkataṃ udāhu devehi dinnaṃ;_  
_Pucchāmi taṃ nāgarājetamatthaṃ,_  
_Kathajjhagamāsi vimānaseṭṭhaṃ’._  

</Paragraph>

</Division>
<Division number="171" book="ja2" bookNo="23" page="21">

<Paragraph number="175" type="verses" book="ja2">
_‘Nādhiccaladdhaṃ na pariṇāmajaṃ me,_  
_Na sayaṃkataṃ nāpi devehi dinnaṃ;_  
_Sakehi kammehi apāpakehi,_  
_Puññehi me laddhamidaṃ vimānaṃ’._  

</Paragraph>

</Division>
<Division number="172" book="ja2" bookNo="23" page="21">

<Paragraph number="176" type="verses" book="ja2">
_‘Kiṃ te vataṃ kiṃ pana brahmacariyaṃ,_  
_Kissa suciṇṇassa ayaṃ vipāko;_  
_Akkhāhi me nāgarājetamatthaṃ,_  
_Kathaṃ nu te laddhamidaṃ vimānaṃ’._  

</Paragraph>

</Division>
<Division number="173" book="ja2" bookNo="23" page="21">

<Paragraph number="177" type="verses" book="ja2">
_‘Rājā ahosiṃ magadhānamissaro,_  
_Duyyodhano nāma mahānubhāvo;_  
_So ittaraṃ jīvitaṃ saṃviditvā,_  
_Asassataṃ vipariṇāmadhammaṃ._  

</Paragraph>

</Division>
<Division number="174" book="ja2" bookNo="23" page="21">

<Paragraph number="178" type="verses" book="ja2">
_Annañca pānañca pasannacitto,_  
_Sakkacca dānaṃ vipulaṃ adāsiṃ;_  
_Opānabhūtaṃ me gharaṃ tadāsi,_  
_Santappitā samaṇabrāhmaṇā ca._  

</Paragraph>

</Division>
<Division number="175" book="ja2" bookNo="23" page="21">

<Paragraph number="179" type="verses" book="ja2">
_Mālañca gandhañca vilepanañca,_  
_Padīpiyaṃ yānamupassayañca;_  
_Acchādanaṃ seyyamathannapānaṃ,_  
_Sakkacca dānāni adamha tattha._  

</Paragraph>

</Division>
<Division number="176" book="ja2" bookNo="23" page="22">

<Paragraph number="180" type="verses" book="ja2">
_Taṃ me vataṃ taṃ pana brahmacariyaṃ,_  
_Tassa suciṇṇassa ayaṃ vipāko;_  
_Teneva me laddhamidaṃ vimānaṃ,_  
_Pahūtabhakkhaṃ bahuannapānaṃ;_  
_Naccehi gītehi cupetarūpaṃ,_  
_Ciraṭṭhitikaṃ na ca sassatāyaṃ’._  

</Paragraph>

</Division>
<Division number="177" book="ja2" bookNo="23" page="22">

<Paragraph number="181" type="verses" book="ja2">
_‘Appānubhāvā taṃ mahānubhāvaṃ,_  
_Tejassinaṃ hanti atejavanto;_  
_Kimeva dāṭhāvudha kiṃ paṭicca,_  
_Hatthattamāgacchi vanibbakānaṃ._  

</Paragraph>

</Division>
<Division number="178" book="ja2" bookNo="23" page="22">

<Paragraph number="182" type="verses" book="ja2">
_Bhayaṃ nu te anvagataṃ mahantaṃ,_  
_Tejo nu te nānvagaṃ dantamūlaṃ;_  
_Kimeva dāṭhāvudha kiṃ paṭicca,_  
_Kilesamāpajji vanibbakānaṃ’._  

</Paragraph>

</Division>
<Division number="179" book="ja2" bookNo="23" page="22">

<Paragraph number="183" type="verses" book="ja2">
_‘Na me bhayaṃ anvagataṃ mahantaṃ,_  
_Tejo na sakkā mama tehi hantuṃ;_  
_Satañca dhammāni sukittitāni,_  
_Samuddavelāva duraccayāni._  

</Paragraph>

</Division>
<Division number="180" book="ja2" bookNo="23" page="22">

<Paragraph number="184" type="verses" book="ja2">

_Uposathaṃ niccamupāvasāmi;_  
_Athāgamuṃ soḷasa bhojaputtā,_  
_Rajjuṃ gahetvāna daḷhañca pāsaṃ._  

</Paragraph>

</Division>

<Paragraph number="185" type="verses" book="ja2">

_Nayiṃsu maṃ samparigayha luddā;_  
_Etādisaṃ dukkhamahaṃ titikkhaṃ,_  
_Uposathaṃ appaṭikopayanto’._  

</Paragraph>

_Balena vaṇṇena cupetarūpaṃ;_  
_Siriyā paññāya ca bhāvitosi,_  
_Kiṃ patthayaṃ nāga tapo karosi’._  

</Paragraph>

_Na āyuno cāpi aḷāra hetu;_  
_Manussayoniṃ abhipatthayāno,_  
_Tasmā parakkamma tapo karomi’._  

</Paragraph>

_Alaṅkato kappitakesamassu;_

_Gandhabbarājāva disā pabhāsasi._

_Sabbehi kāmehi samaṅgibhūto;_  
_Pucchāmi taṃ nāgarājetamatthaṃ,_  
_Seyyo ito kena manussaloko’._

_Suddhī va saṃvijjati saṃyamo vā;_

_Kāhāmi jātimaraṇassa antaṃ’._  

</Paragraph>

_Annena pānena upaṭṭhitosmi;_  
_Āmantayitvāna palemi nāga,_  
_Cirappavutthosmi ahaṃ janinda’._  

</Paragraph>

_Niccānusiṭṭhā upatiṭṭhathetaṃ;_  
_Kaccinnu taṃ nābhisapittha koci,_  
_Piyañhi me dassanaṃ tuyhaṃ aḷāra’._

_Putto piyo paṭivihito vaseyya;_  
_Tatopi mayhaṃ idhameva seyyo,_  
_Cittañhi te nāga mayī pasannaṃ’._

</Division>

_Dhanāharo maṇiratanaṃ uḷāraṃ;_  
_Ādāya tvaṃ gaccha sakaṃ niketaṃ,_  
_Laddhā dhanaṃ taṃ maṇimossajassu’._  

</Paragraph>

_Asassatā vipariṇāmadhammā;_  
_Ādīnavaṃ kāmaguṇesu disvā,_  
_Saddhāyahaṃ pabbajitomhi rāja._

_Daharā ca vuddhā ca sarīrabhedā;_  
_Etampi disvā pabbajitomhi rāja,_  
_Apaṇṇakaṃ sāmaññameva seyyo”._  

</Paragraph>
