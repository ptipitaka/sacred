<Paragraph number="2680" type="verses" book="ja2">
_“Paṇḍu kisiyāsi dubbalā,_  
_Vaṇṇarūpaṃ natavedisaṃ pure;_  
_Vimale akkhāhi pucchitā,_  
_Kīdisī tuyhaṃ sarīravedanā”._  

</Paragraph>

</Division>
<Division number="1347" book="ja2">

<Paragraph number="2681" type="verses" book="ja2">
_“Dhammo manujesu mātīnaṃ,_  
_Dohaḷo nāma janinda vuccati;_  
_Dhammāhataṃ nāgakuñjara,_  
_Vidhurassa hadayābhipatthaye”._  

</Paragraph>

</Division>
<Division number="1348" book="ja2">

<Paragraph number="2682" type="verses" book="ja2">
_“Candaṃ kho tvaṃ dohaḷāyasi,_  
_Sūriyaṃ vā atha vāpi mālutaṃ;_  
_Dullabhañhi vidhurassa dassanaṃ,_  
_Ko vidhuramidha mānayissati”._  

</Paragraph>

</Division>
<Division number="1349" book="ja2">

<Paragraph number="2683" type="verses" book="ja2">
_“Kiṃ nu tāta tuvaṃ pajjhāyasi,_  
_Padumaṃ hatthagataṃva te mukhaṃ;_  
_Kiṃ nu dummanarūposi issara,_  
_Mā tvaṃ soci amittatāpana”._  

</Paragraph>

</Division>
<Division number="1350" book="ja2">

<Paragraph number="2684" type="verses" book="ja2">
_“Mātā hi tava irandhati,_  
_Vidhurassa hadayaṃ dhaniyati;_  
_Dullabhañhi vidhurassa dassanaṃ,_  
_Ko vidhuramidha mānayissati._  

</Paragraph>

</Division>
<Division number="1351" book="ja2">

<Paragraph number="2685" type="verses" book="ja2">
_Tassa bhattupariyesanaṃ cara,_  
_Yo vidhuramidha mānayissati”;_  
_Pituno ca sā sutvāna vākyaṃ,_  
_Rattiṃ nikkhamma avassutiṃ cari._  

</Paragraph>

</Division>
<Division number="1352" book="ja2">

<Paragraph number="2686" type="verses" book="ja2">
_“Ke gandhabbe rakkhase ca nāge,_  
_Ke kimpurise cāpi mānuse;_  
_Ke paṇḍite sabbakāmadade,_  
_Dīgharattaṃ bhattā me bhavissati”._  

</Paragraph>

</Division>
<Division number="1353" book="ja2">

<Paragraph number="2687" type="verses" book="ja2">
_“Assāsa hessāmi te pati,_  
_Bhattā te hessāmi anindalocane;_  
_Paññā hi mamaṃ tathāvidhā,_  
_Assāsa hessasi bhariyā mama”._  

</Paragraph>

</Division>
<Division number="1354" book="ja2">

<Paragraph number="2688" type="verses" book="ja2">
_Avacāsi puṇṇakaṃ irandhatī,_  
_Pubbapathānugatena cetasā;_  
_“Ehi gacchāma pitu mamantike,_  
_Esova te etamatthaṃ pavakkhati”._  

</Paragraph>

</Division>
<Division number="1355" book="ja2">

<Paragraph number="2689" type="verses" book="ja2">
_Alaṅkatā suvasanā,_  
_mālinī candanussadā;_  
_Yakkhaṃ hatthe gahetvāna,_  
_pitusantikupāgami._  

</Paragraph>

</Division>
<Division number="1356" book="ja2">

<Paragraph number="2690" type="verses" book="ja2">
_“Nāgavara vaco suṇohi me,_  
_Patirūpaṃ paṭipajja suṅkiyaṃ;_  
_Patthemi ahaṃ irandhatiṃ,_  
_Tāya samaṅgiṃ karohi maṃ tuvaṃ._  

</Paragraph>

</Division>
<Division number="1357" book="ja2">

<Paragraph number="2691" type="verses" book="ja2">
_Sataṃ hatthī sataṃ assā,_  
_sataṃ assatarīrathā;_  
_Sataṃ valabhiyo puṇṇā,_  
_nānāratnassa kevalā;_  
_Te nāga paṭipajjassu,_  
_dhītaraṃ dehirandhatiṃ”._  

</Paragraph>

</Division>
<Division number="1358" book="ja2">

<Paragraph number="2692" type="verses" book="ja2">
_“Yāva āmantaye ñātī,_  
_mitte ca suhadajjane;_  
_Anāmanta kataṃ kammaṃ,_  
_taṃ pacchā anutappati”._  

</Paragraph>

</Division>
<Division number="1359" book="ja2">

<Paragraph number="2693" type="verses" book="ja2">
_Tato so varuṇo nāgo,_  
_pavisitvā nivesanaṃ;_  
_Bhariyaṃ āmantayitvāna,_  
_idaṃ vacanamabravi._  

</Paragraph>

</Division>
<Division number="1360" book="ja2">

<Paragraph number="2694" type="verses" book="ja2">
_“Ayaṃ so puṇṇako yakkho,_  
_yācatī maṃ irandhatiṃ;_  
_Bahunā vittalābhena,_  
_tassa dema piyaṃ mamaṃ”._  

</Paragraph>

</Division>
<Division number="1361" book="ja2">

<Paragraph number="2695" type="verses" book="ja2">
_“Na dhanena na vittena,_  
_Labbhā amhaṃ irandhatī;_  
_Sace ca kho hadayaṃ paṇḍitassa,_  
_Dhammena laddhā idha māhareyya;_  
_Etena vittena kumāri labbhā,_  
_Nāññaṃ dhanaṃ uttari patthayāma”._  

</Paragraph>

</Division>
<Division number="1362" book="ja2">

<Paragraph number="2696" type="verses" book="ja2">
_Tato so varuṇo nāgo,_  
_nikkhamitvā nivesanā;_  
_Puṇṇakāmantayitvāna,_  
_idaṃ vacanamabravi._  

</Paragraph>

</Division>
<Division number="1363" book="ja2">

<Paragraph number="2697" type="verses" book="ja2">
_“Na dhanena na vittena,_  
_Labbhā amhaṃ irandhatī;_  
_Sace tuvaṃ hadayaṃ paṇḍitassa,_  
_Dhammena laddhā idha māharesi;_  
_Etena vittena kumāri labbhā,_  
_Nāññaṃ dhanaṃ uttari patthayāma”._  

</Paragraph>

</Division>
<Division number="1364" book="ja2">

<Paragraph number="2698" type="verses" book="ja2">
_“Yaṃ paṇḍitotyeke vadanti loke,_  
_Tameva bāloti punāhu aññe;_  
_Akkhāhi me vippavadanti ettha,_  
_Kaṃ paṇḍitaṃ nāga tuvaṃ vadesi”._  

</Paragraph>

</Division>
<Division number="1365" book="ja2">

<Paragraph number="2699" type="verses" book="ja2">
_“Korabyarājassa dhanañcayassa,_  
_Yadi te suto vidhuro nāma kattā;_  
_Ānehi taṃ paṇḍitaṃ dhammaladdhā,_  
_Irandhatī padacarā te hotu”._  

</Paragraph>

</Division>
<Division number="1366" book="ja2">

<Paragraph number="2700" type="verses" book="ja2">
_Idañca sutvā varuṇassa vākyaṃ,_  
_Uṭṭhāya yakkho paramappatīto;_  
_Tattheva santo purisaṃ asaṃsi,_  
_“Ānehi ājaññamidheva yuttaṃ._  

</Paragraph>

</Division>
<Division number="1367" book="ja2">

<Paragraph number="2701" type="verses" book="ja2">
_Jātarūpamayā kaṇṇā,_  
_kācamhicamayā khurā;_  
_Jambonadassa pākassa,_  
_suvaṇṇassa uracchado”._  

</Paragraph>

</Division>
<Division number="1368" book="ja2">

<Paragraph number="2702" type="verses" book="ja2">
_Devavāhavahaṃ yānaṃ,_  
_Assamāruyha puṇṇako;_  
_Alaṅkato kappitakesamassu,_  
_Pakkāmi vehāyasamantalikkhe._  

</Paragraph>

</Division>
<Division number="1369" book="ja2">

<Paragraph number="2703" type="verses" book="ja2">
_So puṇṇako kāmarāgena giddho,_  
_Irandhatiṃ nāgakaññaṃ jigīsaṃ;_  
_Gantvāna taṃ bhūtapatiṃ yasassiṃ,_  
_Iccabravī vessavaṇaṃ kuveraṃ._  

</Paragraph>

</Division>
<Division number="1370" book="ja2">

<Paragraph number="2704" type="verses" book="ja2">
_“Bhogavatī nāma mandire,_  
_Vāsā hiraññavatīti vuccati;_  
_Nagare nimmite kañcanamaye,_  
_Maṇḍalassa uragassa niṭṭhitaṃ._  

</Paragraph>

</Division>
<Division number="1371" book="ja2">

<Paragraph number="2705" type="verses" book="ja2">
_Aṭṭālakā oṭṭhagīviyo,_  
_Lohitaṅkassa masāragallino;_  
_Pāsādettha silāmayā,_  
_Sovaṇṇaratanehi chāditā._  

</Paragraph>

</Division>
<Division number="1372" book="ja2">

<Paragraph number="2706" type="verses" book="ja2">
_Ambā tilakā ca jambuyo,_  
_Sattapaṇṇā mucalindaketakā;_  
_Piyaṅgu uddālakā sahā,_  
_Uparibhaddakā sinduvārakā._  

</Paragraph>

</Division>
<Division number="1373" book="ja2">

<Paragraph number="2707" type="verses" book="ja2">
_Campeyyakā nāgamallikā,_  
_Bhaginīmālā atha mettha koliyā;_  
_Ete dumā pariṇāmitā,_  
_Sobhayanti uragassa mandiraṃ._  

</Paragraph>

</Division>
<Division number="1374" book="ja2">

<Paragraph number="2708" type="verses" book="ja2">
_Khajjurettha silāmayā,_  
_Sovaṇṇadhuvapupphitā bahū;_  
_Yattha vasato papātiko,_  
_Nāgarājā varuṇo mahiddhiko._  

</Paragraph>

</Division>
<Division number="1375" book="ja2">

<Paragraph number="2709" type="verses" book="ja2">
_Tassa komārikā bhariyā,_  
_Vimalā kañcanavelliviggahā;_  
_Kālā taruṇāva uggatā,_  
_Pucimandatthanī cārudassanā._  

</Paragraph>

</Division>
<Division number="1376" book="ja2">

<Paragraph number="2710" type="verses" book="ja2">
_Lākhārasarattasucchavī,_  
_Kaṇikārāva nivātapupphitā;_  
_Tidivokacarāva accharā,_  
_Vijjuvabbhaghanā vinissaṭā._  

</Paragraph>

</Division>
<Division number="1377" book="ja2">

<Paragraph number="2711" type="verses" book="ja2">
_Sā dohaḷinī suvimhitā,_  
_Vidhurassa hadayaṃ dhaniyati;_  
_Taṃ tesaṃ demi issara,_  
_Tena te denti irandhatiṃ mamaṃ”._  

</Paragraph>

</Division>
<Division number="1378" book="ja2">

<Paragraph number="2712" type="verses" book="ja2">
_So puṇṇako bhūtapatiṃ yasassiṃ,_  
_Āmantaya vessavaṇaṃ kuveraṃ;_  
_“Tattheva santo purisaṃ asaṃsi,_  
_Ānehi ājaññamidheva yuttaṃ”._  

</Paragraph>

</Division>
<Division number="1379" book="ja2">

<Paragraph number="2713" type="verses" book="ja2">

_kācamhicamayā khurā;_  
_Jambonadassa pākassa,_  
_suvaṇṇassa uracchado._  

</Paragraph>

</Division>
<Division number="1380" book="ja2">

<Paragraph number="2714" type="verses" book="ja2">
_Devavāhavahaṃ yānaṃ,_  
_Assamāruyha puṇṇako;_  
_Alaṅkato kappitakesamassu,_  
_Pakkāmi vehāyasamantalikkhe._  

</Paragraph>

</Division>
<Division number="1381" book="ja2">

<Paragraph number="2715" type="verses" book="ja2">
_So aggamā rājagahaṃ surammaṃ,_  
_Aṅgassa rañño nagaraṃ durāyutaṃ;_  
_Pahūtabhakkhaṃ bahuannapānaṃ,_  
_Masakkasāraṃ viya vāsavassa._  

</Paragraph>

</Division>
<Division number="1382" book="ja2">

<Paragraph number="2716" type="verses" book="ja2">
_Mayūrakoñcāgaṇasampaghuṭṭhaṃ,_  
_Dijābhighuṭṭhaṃ dijasaṅghasevitaṃ;_  
_Nānāsakuntābhirudaṃ suvaṅgaṇaṃ,_  
_Pupphābhikiṇṇaṃ himavaṃva pabbataṃ._  

</Paragraph>

</Division>
<Division number="1383" book="ja2">

<Paragraph number="2717" type="verses" book="ja2">
_So puṇṇako vepulamābhirūhi,_  
_Siluccayaṃ kimpurisānuciṇṇaṃ;_  
_Anvesamāno maṇiratanaṃ uḷāraṃ,_  
_Tamaddasā pabbatakūṭamajjhe._  

</Paragraph>

</Division>
<Division number="1384" book="ja2">

<Paragraph number="2718" type="verses" book="ja2">
_Disvā maṇiṃ pabhassaraṃ jātimantaṃ,_  
_Manoharaṃ maṇiratanaṃ uḷāraṃ;_  
_Daddallamānaṃ yasasā yasassinaṃ,_  
_Obhāsatī vijjurivantalikkhe._  

</Paragraph>

</Division>
<Division number="1385" book="ja2">

<Paragraph number="2719" type="verses" book="ja2">
_Tamaggahī veḷuriyaṃ mahagghaṃ,_  
_Manoharaṃ nāma mahānubhāvaṃ;_  
_Ājaññamāruyha manomavaṇṇo,_  
_Pakkāmi vehāyasamantalikkhe._  

</Paragraph>

</Division>
<Division number="1386" book="ja2">

<Paragraph number="2720" type="verses" book="ja2">
_So aggamā nagaramindapatthaṃ,_  
_Oruyhupāgacchi sabhaṃ kurūnaṃ;_  
_Samāgate ekasataṃ samagge,_  
_Avhettha yakkho avikampamāno._  

</Paragraph>

</Division>
<Division number="1387" book="ja2">

<Paragraph number="2721" type="verses" book="ja2">
_“Ko nīdha raññaṃ varamābhijeti,_  
_Kamābhijeyyāma varaddhanena;_  
_Kamanuttaraṃ ratanavaraṃ jināma,_  
_Ko vāpi no jeti varaddhanena”._  

</Paragraph>

</Division>
<Division number="1388" book="ja2">

<Paragraph number="2722" type="verses" book="ja2">
_“Kuhiṃ nu raṭṭhe tava jātibhūmi,_  
_Na korabyasseva vaco tavedaṃ;_  
_Abhītosi no vaṇṇanibhāya sabbe,_  
_Akkhāhi me nāmañca bandhave ca”._  

</Paragraph>

</Division>
<Division number="1389" book="ja2">

<Paragraph number="2723" type="verses" book="ja2">
_“Kaccāyano māṇavakosmi rāja,_  
_Anūnanāmo iti mavhayanti;_  
_Aṅgesu me ñātayo bandhavā ca,_  
_Akkhena devasmi idhānupatto”._  

</Paragraph>

</Division>
<Division number="1390" book="ja2">

<Paragraph number="2724" type="verses" book="ja2">
_“Kiṃ māṇavassa ratanāni atthi,_  
_Ye taṃ jinanto hare akkhadhutto;_  
_Bahūni rañño ratanāni atthi,_  
_Te tvaṃ daliddo kathamavhayesi”._  

</Paragraph>

</Division>
<Division number="1391" book="ja2">

<Paragraph number="2725" type="verses" book="ja2">
_“Manoharo nāma maṇī mamāyaṃ,_  
_Manoharaṃ maṇiratanaṃ uḷāraṃ;_  
_Imañca ājaññamamittatāpanaṃ,_  
_Etaṃ me jinitvā hare akkhadhutto”._  

</Paragraph>

</Division>
<Division number="1392" book="ja2">

<Paragraph number="2726" type="verses" book="ja2">
_“Eko maṇī māṇava kiṃ karissati,_  
_Ājāniyeko pana kiṃ karissati;_  
_Bahūni rañño maṇiratanāni atthi,_  
_Ājāniyā vātajavā anappakā”._  


</Paragraph>
</Division>

<Paragraph number="2727" type="center" book="ja2">
Dohaḷakaṇḍaṃ nāma.
