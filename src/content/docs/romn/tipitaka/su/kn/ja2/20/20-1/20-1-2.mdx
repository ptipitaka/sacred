_evaṃ jānāhi bhāradha”._  

</Paragraph>

_veyyāvaccaṃ anappakaṃ;_  
_Devamhi vassamānamhi,_  
_anovassaṃ bhavaṃ akā._  

</Paragraph>

</Division>
<Division number="95" book="ja2">

<Paragraph number="678" type="verses" book="ja2">

_sītacchāyaṃ bhavaṃ akā;_  
_Tato amittamajjhesu,_  
_saratāṇaṃ bhavaṃ akā._  

</Paragraph>

</Division>
<Division number="96" book="ja2">

<Paragraph number="679" type="verses" book="ja2">
_Tato phītāni raṭṭhāni,_  
_vasino te bhavaṃ akā;_  
_Tato ekasataṃ khatye,_  
_anuyante bhavaṃ akā._  

</Paragraph>

</Division>
<Division number="97" book="ja2">

<Paragraph number="680" type="verses" book="ja2">
_Patītāssu mayaṃ bhoto,_  
_Vada taṃ bhañjamicchasi;_  
_Hatthiyānaṃ assarathaṃ,_  
_Nāriyo ca alaṅkatā;_  
_Nivesanāni rammāni,_  
_Mayaṃ bhoto dadāmase._  

</Paragraph>

</Division>
<Division number="98" book="ja2">

<Paragraph number="681" type="verses" book="ja2">
_Atha vaṅge vā magadhe,_  
_mayaṃ bhoto dadāmase;_  
_Atha vā assakāvantī,_  
_sumanā damma te mayaṃ._  

</Paragraph>

</Division>
<Division number="99" book="ja2">

<Paragraph number="682" type="verses" book="ja2">
_Upaḍḍhaṃ vāpi rajjassa,_  
_mayaṃ bhoto dadāmase;_  
_Sace te attho rajjena,_  
_anusāsa yadicchasi”._  

</Paragraph>

</Division>
<Division number="100" book="ja2">

<Paragraph number="683" type="verses" book="ja2">

_nagarena dhanena vā;_  
_Athopi janapadena,_  
_attho mayhaṃ na vijjati._  

</Paragraph>

</Division>
<Division number="101" book="ja2">

<Paragraph number="684" type="verses" book="ja2">
_Bhotova raṭṭhe vijite,_  
_araññe atthi assamo;_  
_Pitā mayhaṃ janettī ca,_  
_ubho sammanti assame._  

</Paragraph>

</Division>
<Division number="102" book="ja2">

<Paragraph number="685" type="verses" book="ja2">

_Puññaṃ na labhāmi kātave;_  
_Bhavantaṃ ajjhāvaraṃ katvā,_  
_Soṇaṃ yācemu saṃvaraṃ”._  

</Paragraph>

</Division>
<Division number="103" book="ja2">

<Paragraph number="686" type="verses" book="ja2">
_“Karomi te taṃ vacanaṃ,_  
_yaṃ maṃ bhaṇasi brāhmaṇa;_  
_Etañca kho no akkhāhi,_  
_kīvanto hontu yācakā”._  

</Paragraph>

</Division>
<Division number="104" book="ja2">

_mahāsālā ca brāhmaṇā;_  
_Ime ca khattiyā sabbe,_  
_abhijātā yasassino;_  
_Bhavañca rājā manojo,_  
_alaṃ hessanti yācakā”._  

</Paragraph>

</Division>
<Division number="105" book="ja2">

_rathaṃ sannayha sārathi;_  
_Ābandhanāni gaṇhātha,_  
_pādāsussārayaddhaje;_  
_Assamaṃ taṃ gamissāmi,_  
_yattha sammati kosiyo”._  

</Paragraph>

</Division>

_senāya caturaṅginī;_

_yattha sammati kosiyo._  

</Paragraph>

</Division>

_udahārāya gacchato”._  

</Paragraph>

</Division>

_tāpaso sahitabbato;_  
_Bharāmi mātāpitaro,_  
_rattindivamatandito._  

</Paragraph>

</Division>

_pubbe katamanussaraṃ”._  

</Paragraph>

</Division>

</Paragraph>

</Division>

</Paragraph>

_anusāsitvāna khattiye._  

</Paragraph>

_pitaraṃ paṭibodhayi._  

</Paragraph>

</Division>

_nisīda tvaṃ mahāise”._  

</Paragraph>

</Division>

_taramāno mahāisi;_  
_Assamā nikkhamitvāna,_  
_sadvāramhi upāvisi._  

</Paragraph>

</Division>

_jalantaṃriva tejasā;_

_kosiyo etadabravi._  

</Paragraph>

</Division>

_hāsayantā rathesabhaṃ._  

</Paragraph>

</Division>

_ko eti siriyā jalaṃ._  

</Paragraph>

_ko eti siriyā jalaṃ._  

</Paragraph>

</Division>

_ko eti siriyā jalaṃ._  

</Paragraph>

</Division>

_hatthikkhandhena āyato._  

</Paragraph>

_ko eti siriyā jalaṃ._  

</Paragraph>

_ko eti siriyā jalaṃ._

</Paragraph>

</Paragraph>

</Paragraph>

</Paragraph>
