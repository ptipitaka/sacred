<Division number="1" book="ja2">

<Paragraph number="2" type="verses" book="ja2">
_“Vessantaraṃ taṃ pucchāmi,_  
_sakuṇa bhaddamatthu te;_  
_Rajjaṃ kāretukāmena,_  
_kiṃ su kiccaṃ kataṃ varaṃ”._  

</Paragraph>

</Division>
<Division number="2" book="ja2">

<Paragraph number="3" type="verses" book="ja2">
_“Cirassaṃ vata maṃ tāto,_  
_kaṃso bārāṇasiggaho;_  
_Pamatto appamattaṃ maṃ,_  
_pitā puttaṃ acodayi._  

</Paragraph>

</Division>
<Division number="3" book="ja2">

<Paragraph number="4" type="verses" book="ja2">
_Paṭhameneva vitathaṃ,_  
_kodhaṃ hāsaṃ nivāraye;_  
_Tato kiccāni kāreyya,_  
_taṃ vataṃ āhu khattiya._  

</Paragraph>

</Division>
<Division number="4" book="ja2">

<Paragraph number="5" type="verses" book="ja2">
_Yaṃ tvaṃ tāta tapokammaṃ,_  
_Pubbe katamasaṃsayaṃ;_  
_Ratto duṭṭho ca yaṃ kayirā,_  
_Na taṃ kayirā tato puna._  

</Paragraph>

</Division>
<Division number="5" book="ja2">

<Paragraph number="6" type="verses" book="ja2">
_Khattiyassa pamattassa,_  
_raṭṭhasmiṃ raṭṭhavaḍḍhana;_  
_Sabbe bhogā vinassanti,_  
_rañño taṃ vuccate aghaṃ._  

</Paragraph>

</Division>
<Division number="6" book="ja2">

<Paragraph number="7" type="verses" book="ja2">
_Sirī tāta alakkhī ca,_  
_pucchitā etadabravuṃ;_  
_Uṭṭhānavīriye pose,_  
_ramāhaṃ anusūyake._  

</Paragraph>

</Division>
<Division number="7" book="ja2">

<Paragraph number="8" type="verses" book="ja2">
_Usūyake duhadaye,_  
_purise kammadussake;_  
_Kāḷakaṇṇī mahārāja,_  
_ramati cakkabhañjanī._  

</Paragraph>

</Division>
<Division number="8" book="ja2">

<Paragraph number="9" type="verses" book="ja2">
_So tvaṃ sabbesu suhadayo,_  
_sabbesaṃ rakkhito bhava;_  
_Alakkhiṃ nuda mahārāja,_  
_lakkhyā bhava nivesanaṃ._  

</Paragraph>

</Division>
<Division number="9" book="ja2">

<Paragraph number="10" type="verses" book="ja2">
_Sa lakkhīdhitisampanno,_  
_puriso hi mahaggato;_  
_Amittānaṃ kāsipati,_  
_mūlaṃ aggañca chindati._  

</Paragraph>

</Division>
<Division number="10" book="ja2">

<Paragraph number="11" type="verses" book="ja2">
_Sakkopi hi bhūtapati,_  
_uṭṭhāne nappamajjati;_  
_Sa kalyāṇe dhitiṃ katvā,_  
_uṭṭhāne kurute mano._  

</Paragraph>

</Division>
<Division number="11" book="ja2">

<Paragraph number="12" type="verses" book="ja2">
_Gandhabbā pitaro devā,_  
_sājīvā honti tādino;_  
_Uṭṭhāhato appamajjato,_  
_anutiṭṭhanti devatā._  

</Paragraph>

</Division>
<Division number="12" book="ja2">

<Paragraph number="13" type="verses" book="ja2">
_So appamatto akkuddho,_  
_tāta kiccāni kāraya;_  
_Vāyamassu ca kiccesu,_  
_nālaso vindate sukhaṃ._  

</Paragraph>

</Division>
<Division number="13" book="ja2">

<Paragraph number="14" type="verses" book="ja2">
_Tattheva te vattapadā,_  
_esāva anusāsanī;_  
_Alaṃ mitte sukhāpetuṃ,_  
_amittānaṃ dukhāya ca”._  

</Paragraph>

</Division>
<Division number="14" book="ja2">

<Paragraph number="15" type="verses" book="ja2">
_“Sakkhisi tvaṃ kuṇḍalini,_  
_maññasi khattabandhuni;_  
_Rajjaṃ kāretukāmena,_  
_kiṃ su kiccaṃ kataṃ varaṃ”._  

</Paragraph>

</Division>
<Division number="15" book="ja2">

<Paragraph number="16" type="verses" book="ja2">
_“Dveva tāta padakāni,_  
_yattha sabbaṃ patiṭṭhitaṃ;_  
_Aladdhassa ca yo lābho,_  
_laddhassa cānurakkhaṇā._  

</Paragraph>

</Division>
<Division number="16" book="ja2">

<Paragraph number="17" type="verses" book="ja2">
_Amacce tāta jānāhi,_  
_dhīre atthassa kovide;_  
_Anakkhākitave tāta,_  
_asoṇḍe avināsake._  

</Paragraph>

</Division>
<Division number="17" book="ja2">

<Paragraph number="18" type="verses" book="ja2">
_Yo ca taṃ tāta rakkheyya,_  
_dhanaṃ yañceva te siyā;_  
_Sūtova rathaṃ saṅgaṇhe,_  
_so te kiccāni kāraye._  

</Paragraph>

</Division>
<Division number="18" book="ja2">

<Paragraph number="19" type="verses" book="ja2">
_Susaṅgahitantajano,_  
_sayaṃ vittaṃ avekkhiya;_  
_Nidhiñca iṇadānañca,_  
_na kare parapattiyā._  

</Paragraph>

</Division>
<Division number="19" book="ja2">

<Paragraph number="20" type="verses" book="ja2">
_Sayaṃ āyaṃ vayaṃ jaññā,_  
_sayaṃ jaññā katākataṃ;_  
_Niggaṇhe niggahārahaṃ,_  
_paggaṇhe paggahārahaṃ._  

</Paragraph>

</Division>
<Division number="20" book="ja2">

<Paragraph number="21" type="verses" book="ja2">
_Sayaṃ jānapadaṃ atthaṃ,_  
_anusāsa rathesabha;_  
_Mā te adhammikā yuttā,_  
_dhanaṃ raṭṭhañca nāsayuṃ._  

</Paragraph>

</Division>
<Division number="21" book="ja2">

<Paragraph number="22" type="verses" book="ja2">
_Mā ca vegena kiccāni,_  
_karosi kārayesi vā;_  
_Vegasā hi kataṃ kammaṃ,_  
_mando pacchānutappati._  

</Paragraph>

</Division>
<Division number="22" book="ja2">

<Paragraph number="23" type="verses" book="ja2">
_Mā te adhisare muñca,_  
_subāḷhamadhikopitaṃ;_  
_Kodhasā hi bahū phītā,_  
_kulā akulataṃ gatā._  

</Paragraph>

</Division>
<Division number="23" book="ja2">

<Paragraph number="24" type="verses" book="ja2">
_Mā tāta issaromhīti,_  
_anatthāya patārayi;_  
_Itthīnaṃ purisānañca,_  
_mā te āsi dukhudrayo._  

</Paragraph>

</Division>
<Division number="24" book="ja2">

<Paragraph number="25" type="verses" book="ja2">
_Apetalomahaṃsassa,_  
_rañño kāmānusārino;_  
_Sabbe bhogā vinassanti,_  
_rañño taṃ vuccate aghaṃ._  

</Paragraph>

</Division>
<Division number="25" book="ja2">

<Paragraph number="26" type="verses" book="ja2">
_Tattheva te vattapadā,_  
_esāva anusāsanī;_  
_Dakkhassudāni puññakaro,_  
_asoṇḍo avināsako;_  
_Sīlavāssu mahārāja,_  
_dussīlo vinipātiko”._  

</Paragraph>

</Division>
<Division number="26" book="ja2">

<Paragraph number="27" type="verses" book="ja2">
_“Apucchimha kosiyagottaṃ,_  
_kuṇḍaliniṃ tatheva ca;_  
_Tvaṃ dāni vadehi jambuka,_  
_balānaṃ balamuttamaṃ”._  

</Paragraph>

</Division>
<Division number="27" book="ja2">

<Paragraph number="28" type="verses" book="ja2">
_“Balaṃ pañcavidhaṃ loke,_  
_purisasmiṃ mahaggate;_  
_Tattha bāhubalaṃ nāma,_  
_carimaṃ vuccate balaṃ._  

</Paragraph>

</Division>
<Division number="28" book="ja2">

<Paragraph number="29" type="verses" book="ja2">
_Bhogabalañca dīghāvu,_  
_dutiyaṃ vuccate balaṃ;_  
_Amaccabalañca dīghāvu,_  
_tatiyaṃ vuccate balaṃ._  

</Paragraph>

</Division>
<Division number="29" book="ja2">

<Paragraph number="30" type="verses" book="ja2">
_Abhijaccabalaṃ ceva,_  
_taṃ catutthaṃ asaṃsayaṃ;_  
_Yāni cetāni sabbāni,_  
_adhigaṇhāti paṇḍito._  

</Paragraph>

</Division>
<Division number="30" book="ja2">

<Paragraph number="31" type="verses" book="ja2">
_Taṃ balānaṃ balaṃ seṭṭhaṃ,_  
_aggaṃ paññābalaṃ balaṃ;_  
_Paññābalenupatthaddho,_  
_atthaṃ vindati paṇḍito._  

</Paragraph>

</Division>
<Division number="31" book="ja2">

<Paragraph number="32" type="verses" book="ja2">
_Api ce labhati mando,_  
_phītaṃ dharaṇimuttamaṃ;_  
_Akāmassa pasayhaṃ vā,_  
_añño taṃ paṭipajjati._  

</Paragraph>

</Division>
<Division number="32" book="ja2">

<Paragraph number="33" type="verses" book="ja2">
_Abhijātopi ce hoti,_  
_rajjaṃ laddhāna khattiyo;_  
_Duppañño hi kāsipati,_  
_sabbenapi na jīvati._  

</Paragraph>

</Division>
<Division number="33" book="ja2">

<Paragraph number="34" type="verses" book="ja2">
_Paññāva sutaṃ vinicchinī,_  
_Paññā kitti silokavaḍḍhanī;_  
_Paññāsahito naro idha,_  
_Api dukkhe sukhāni vindati._  

</Paragraph>

</Division>
<Division number="34" book="ja2">

<Paragraph number="35" type="verses" book="ja2">
_Paññañca kho asussūsaṃ,_  
_na koci adhigacchati;_  
_Bahussutaṃ anāgamma,_  
_dhammaṭṭhaṃ avinibbhujaṃ._  

</Paragraph>

</Division>
<Division number="35" book="ja2">

<Paragraph number="36" type="verses" book="ja2">
_Yo ca dhammavibhaṅgaññū,_  
_kāluṭṭhāyī matandito;_  
_Anuṭṭhahati kālena,_  
_kammaphalaṃ tassijjhati._  

</Paragraph>

</Division>
<Division number="36" book="ja2">

<Paragraph number="37" type="verses" book="ja2">
_Anāyatanasīlassa,_  
_anāyatanasevino;_  
_Na nibbindiyakārissa,_  
_sammadattho vipaccati._  

</Paragraph>

</Division>
<Division number="37" book="ja2">

<Paragraph number="38" type="verses" book="ja2">
_Ajjhattañca payuttassa,_  
_tathāyatanasevino;_  
_Anibbindiyakārissa,_  
_sammadattho vipaccati._  

</Paragraph>

</Division>
<Division number="38" book="ja2">

<Paragraph number="39" type="verses" book="ja2">
_Yogappayogasaṅkhātaṃ,_  
_sambhatassānurakkhaṇaṃ;_  
_Tāni tvaṃ tāta sevassu,_  
_mā akammāya randhayi;_  
_Akammunā hi dummedho,_  
_naḷāgāraṃva sīdati”._  

</Paragraph>

</Division>
<Division number="39" book="ja2">

<Paragraph number="40" type="verses" book="ja2">
_“Dhammaṃ cara mahārāja,_  
_mātāpitūsu khattiya;_  
_Idha dhammaṃ caritvāna,_  
_rāja saggaṃ gamissasi._  

</Paragraph>

</Division>
<Division number="40" book="ja2">

<Paragraph number="41" type="verses" book="ja2">
_Dhammaṃ cara mahārāja,_  
_puttadāresu khattiya;_  
_Idha dhammaṃ caritvāna,_  
_rāja saggaṃ gamissasi._  

</Paragraph>

</Division>
<Division number="41" book="ja2">

<Paragraph number="42" type="verses" book="ja2">
_Dhammaṃ cara mahārāja,_  
_mittāmaccesu khattiya;_  
_Idha dhammaṃ caritvāna,_  
_rāja saggaṃ gamissasi._  

</Paragraph>

</Division>
<Division number="42" book="ja2">

<Paragraph number="43" type="verses" book="ja2">
_Dhammaṃ cara mahārāja,_  
_vāhanesu balesu ca;_  
_Idha dhammaṃ caritvāna,_  
_rāja saggaṃ gamissasi._  

</Paragraph>

</Division>
<Division number="43" book="ja2">

<Paragraph number="44" type="verses" book="ja2">
_Dhammaṃ cara mahārāja,_  
_gāmesu nigamesu ca;_  
_ …pe… _  

</Paragraph>

</Division>
<Division number="44" book="ja2">

<Paragraph number="45" type="verses" book="ja2">
_Dhammaṃ cara mahārāja,_  
_raṭṭhesu janapadesu ca;_  
_ …pe… _  

</Paragraph>

</Division>
<Division number="45" book="ja2">

<Paragraph number="46" type="verses" book="ja2">
_Dhammaṃ cara mahārāja,_  
_samaṇabrāhmaṇesu ca;_  
_ …pe… _  

</Paragraph>

</Division>
<Division number="46" book="ja2">

<Paragraph number="47" type="verses" book="ja2">
_Dhammaṃ cara mahārāja,_  
_migapakkhīsu khattiya;_  
_Idha dhammaṃ caritvāna,_  
_rāja saggaṃ gamissasi._  

</Paragraph>

</Division>
<Division number="47" book="ja2">

<Paragraph number="48" type="verses" book="ja2">
_Dhammaṃ cara mahārāja,_  
_dhammo ciṇṇo sukhāvaho;_  
_Idha dhammaṃ caritvāna,_  
_rāja saggaṃ gamissasi._  

</Paragraph>

</Division>
<Division number="48" book="ja2">

<Paragraph number="49" type="verses" book="ja2">
_Dhammaṃ cara mahārāja,_  
_Saindā devā sabrahmakā;_  
_Suciṇṇena divaṃ pattā,_  
_Mā dhammaṃ rāja pāmado._  

</Paragraph>

</Division>
<Division number="49" book="ja2">

<Paragraph number="50" type="verses" book="ja2">
_Tattheva te vattapadā,_  
_Esāva anusāsanī;_  
_Sappaññasevī kalyāṇī,_  
_Samattaṃ sāma taṃ vidū”ti._  


</Paragraph>
</Division>
