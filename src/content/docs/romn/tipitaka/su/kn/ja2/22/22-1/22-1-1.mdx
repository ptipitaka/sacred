<Paragraph number="1315" type="verses" book="ja2">
_“Mā paṇḍiccayaṃ vibhāvaya,_  
_Bālamato bhava sabbapāṇinaṃ;_  
_Sabbo taṃ jano ocināyatu,_  
_Evaṃ tava attho bhavissati”._  

</Paragraph>

</Division>
<Division number="2" book="ja2">

<Paragraph number="1316" type="verses" book="ja2">
_“Karomi te taṃ vacanaṃ,_  
_yaṃ maṃ bhaṇasi devate;_  
_Atthakāmāsi me amma,_  
_hitakāmāsi devate”._  

</Paragraph>

</Division>
<Division number="3" book="ja2">

<Paragraph number="1317" type="verses" book="ja2">
_“Kiṃ nu santaramānova,_  
_kāsuṃ khaṇasi sārathi;_  
_Puṭṭho me samma akkhāhi,_  
_kiṃ kāsuyā karissasi”._  

</Paragraph>

</Division>
<Division number="4" book="ja2">

<Paragraph number="1318" type="verses" book="ja2">
_“Rañño mūgo ca pakkho ca,_  
_putto jāto acetaso;_  
_Somhi raññā samajjhiṭṭho,_  
_puttaṃ me nikhaṇaṃ vane”._  

</Paragraph>

</Division>
<Division number="5" book="ja2">

<Paragraph number="1319" type="verses" book="ja2">
_“Na badhiro na mūgosmi,_  
_na pakkho na ca vīkalo;_  
_Adhammaṃ sārathi kayirā,_  
_mañce tvaṃ nikhaṇaṃ vane._  

</Paragraph>

</Division>
<Division number="6" book="ja2">

<Paragraph number="1320" type="verses" book="ja2">
_Ūrū bāhuñca me passa,_  
_bhāsitañca suṇohi me;_  
_Adhammaṃ sārathi kayirā,_  
_mañce tvaṃ nikhaṇaṃ vane”._  

</Paragraph>

</Division>
<Division number="7" book="ja2">

<Paragraph number="1321" type="verses" book="ja2">
_“Devatā nusi gandhabbo,_  
_adu sakko purindado;_  
_Ko vā tvaṃ kassa vā putto,_  
_kathaṃ jānemu taṃ mayaṃ”._  

</Paragraph>

</Division>
<Division number="8" book="ja2">

<Paragraph number="1322" type="verses" book="ja2">
_“Namhi devo na gandhabbo,_  
_nāpi sakko purindado;_  
_Kāsirañño ahaṃ putto,_  
_yaṃ kāsuyā nikhaññasi._  

</Paragraph>

</Division>
<Division number="9" book="ja2">

<Paragraph number="1323" type="verses" book="ja2">
_Tassa rañño ahaṃ putto,_  
_yaṃ tvaṃ sammūpajīvasi;_  
_Adhammaṃ sārathi kayirā,_  
_mañce tvaṃ nikhaṇaṃ vane._  

</Paragraph>

</Division>
<Division number="10" book="ja2">

<Paragraph number="1324" type="verses" book="ja2">
_Yassa rukkhassa chāyāya,_  
_nisīdeyya sayeyya vā;_  
_Na tassa sākhaṃ bhañjeyya,_  
_mittadubbho hi pāpako._  

</Paragraph>

</Division>
<Division number="11" book="ja2">

<Paragraph number="1325" type="verses" book="ja2">
_Yathā rukkho tathā rājā,_  
_yathā sākhā tathā ahaṃ;_  
_Yathā chāyūpago poso,_  
_evaṃ tvamasi sārathi;_  
_Adhammaṃ sārathi kayirā,_  
_mañce tvaṃ nikhaṇaṃ vane._  

</Paragraph>

</Division>
<Division number="12" book="ja2">

<Paragraph number="1326" type="verses" book="ja2">
_Pahūtabhakkho bhavati,_  
_vippavuttho sakaṃ gharā;_  
_Bahū naṃ upajīvanti,_  
_yo mittānaṃ na dubbhati._  

</Paragraph>

</Division>
<Division number="13" book="ja2">

<Paragraph number="1327" type="verses" book="ja2">
_Yaṃ yaṃ janapadaṃ yāti,_  
_nigame rājadhāniyo;_  
_Sabbattha pūjito hoti,_  
_yo mittānaṃ na dubbhati._  

</Paragraph>

</Division>
<Division number="14" book="ja2">

<Paragraph number="1328" type="verses" book="ja2">
_Nāssa corā pasāhanti,_  
_nātimaññanti khattiyā;_  
_Sabbe amitte tarati,_  
_yo mittānaṃ na dubbhati._  

</Paragraph>

</Division>
<Division number="15" book="ja2">

<Paragraph number="1329" type="verses" book="ja2">
_Akkuddho sagharaṃ eti,_  
_sabhāyaṃ paṭinandito;_  
_Ñātīnaṃ uttamo hoti,_  
_yo mittānaṃ na dubbhati._  

</Paragraph>

</Division>
<Division number="16" book="ja2">

<Paragraph number="1330" type="verses" book="ja2">
_Sakkatvā sakkato hoti,_  
_garu hoti sagāravo;_  
_Vaṇṇakittibhato hoti,_  
_yo mittānaṃ na dubbhati._  

</Paragraph>

</Division>
<Division number="17" book="ja2">

<Paragraph number="1331" type="verses" book="ja2">
_Pūjako labhate pūjaṃ,_  
_vandako paṭivandanaṃ;_  
_Yaso kittiñca pappoti,_  
_yo mittānaṃ na dubbhati._  

</Paragraph>

</Division>
<Division number="18" book="ja2">

<Paragraph number="1332" type="verses" book="ja2">
_Aggi yathā pajjalati,_  
_devatāva virocati;_  
_Siriyā ajahito hoti,_  
_yo mittānaṃ na dubbhati._  

</Paragraph>

</Division>
<Division number="19" book="ja2">

<Paragraph number="1333" type="verses" book="ja2">
_Gāvo tassa pajāyanti,_  
_khette vuttaṃ virūhati;_  
_Vuttānaṃ phalamasnāti,_  
_yo mittānaṃ na dubbhati._  

</Paragraph>

</Division>
<Division number="20" book="ja2">

<Paragraph number="1334" type="verses" book="ja2">
_Darito pabbatāto vā,_  
_rukkhato patito naro;_  
_Cuto patiṭṭhaṃ labhati,_  
_yo mittānaṃ na dubbhati._  

</Paragraph>

</Division>
<Division number="21" book="ja2">

<Paragraph number="1335" type="verses" book="ja2">
_Virūḷhamūlasantānaṃ,_  
_nigrodhamiva māluto;_  
_Amittā nappasāhanti,_  
_yo mittānaṃ na dubbhati”._  

</Paragraph>

</Division>
<Division number="22" book="ja2">

<Paragraph number="1336" type="verses" book="ja2">
_“Ehi taṃ paṭinessāmi,_  
_rājaputta sakaṃ gharaṃ;_  
_Rajjaṃ kārehi bhaddante,_  
_kiṃ araññe karissasi”._  

</Paragraph>

</Division>
<Division number="23" book="ja2">

<Paragraph number="1337" type="verses" book="ja2">
_“Alaṃ me tena rajjena,_  
_ñātakehi dhanena vā;_  
_Yaṃ me adhammacariyāya,_  
_rajjaṃ labbhetha sārathi”._  

</Paragraph>

</Division>
<Division number="24" book="ja2">

<Paragraph number="1338" type="verses" book="ja2">
_“Puṇṇapattaṃ maṃ lābhehi,_  
_rājaputta ito gato;_  
_Pitā mātā ca me dajjuṃ,_  
_rājaputta tayī gate._  

</Paragraph>

</Division>
<Division number="25" book="ja2">

<Paragraph number="1339" type="verses" book="ja2">
_Orodhā ca kumārā ca,_  
_vesiyānā ca brāhmaṇā;_  
_Tepi attamanā dajjuṃ,_  
_rājaputta tayī gate._  

</Paragraph>

</Division>
<Division number="26" book="ja2">

<Paragraph number="1340" type="verses" book="ja2">
_Hatthārohā anīkaṭṭhā,_  
_rathikā pattikārakā;_  
_Tepi attamanā dajjuṃ,_  
_rājaputta tayī gate._  

</Paragraph>

</Division>
<Division number="27" book="ja2">

<Paragraph number="1341" type="verses" book="ja2">
_Bahudhaññā jānapadā,_  
_negamā ca samāgatā;_  
_Upāyanāni me dajjuṃ,_  
_rājaputta tayī gate”._  

</Paragraph>

</Division>
<Division number="28" book="ja2">

<Paragraph number="1342" type="verses" book="ja2">
_“Pitu mātu cahaṃ catto,_  
_raṭṭhassa nigamassa ca;_  
_Atho sabbakumārānaṃ,_  
_natthi mayhaṃ sakaṃ gharaṃ._  

</Paragraph>

</Division>
<Division number="29" book="ja2">

<Paragraph number="1343" type="verses" book="ja2">
_Anuññāto ahaṃ matyā,_  
_sañcatto pitarā mahaṃ;_  
_Ekoraññe pabbajito,_  
_na kāme abhipatthaye._  

</Paragraph>

</Division>
<Division number="30" book="ja2">

<Paragraph number="1344" type="verses" book="ja2">

_phalāsāva samijjhati;_  
_Vipakkabrahmacariyosmi,_  
_evaṃ jānāhi sārathi._  

</Paragraph>

</Division>
<Division number="31" book="ja2">

_nikkhanto akutobhayo”._  

</Paragraph>

</Division>
<Division number="32" book="ja2">

<Paragraph number="1346" type="verses" book="ja2">
_“Evaṃ vaggukatho santo,_  
_visaṭṭhavacano casi;_  
_Kasmā pitu ca mātucca,_  
_santike na bhaṇī tadā”._  

</Paragraph>

</Division>
<Division number="33" book="ja2">

<Paragraph number="1347" type="verses" book="ja2">
_“Nāhaṃ asandhitā pakkho,_  
_na badhiro asotatā;_  
_Nāhaṃ ajivhatā mūgo,_  
_mā maṃ mūgamadhārayi._  

</Paragraph>

</Division>
<Division number="34" book="ja2">

_yattha rajjamakārayiṃ;_  
_Kārayitvā tahiṃ rajjaṃ,_  
_pāpatthaṃ nirayaṃ bhusaṃ._  

</Paragraph>

</Division>
<Division number="35" book="ja2">

<Paragraph number="1349" type="verses" book="ja2">

_Asītivassasahassāni,_  
_nirayamhi apaccisaṃ._  

</Paragraph>

</Division>
<Division number="36" book="ja2">

<Paragraph number="1350" type="verses" book="ja2">

_mā maṃ rajjābhisecayuṃ;_  
_Tasmā pitu ca mātucca,_  
_santike na bhaṇiṃ tadā._  

</Paragraph>

</Division>
<Division number="37" book="ja2">

<Paragraph number="1351" type="verses" book="ja2">

_pitā atthānusāsati;_  
_Ekaṃ hanatha bandhatha,_  
_ekaṃ khārāpatacchikaṃ;_  
_Ekaṃ sūlasmiṃ uppetha,_  
_iccassa manusāsati._  

</Paragraph>

</Division>
<Division number="38" book="ja2">

_vācāyo samudīritā;_  
_Amūgo mūgavaṇṇena,_  
_apakkho pakkhasammato;_  
_Sake muttakarīsasmiṃ,_  
_acchāhaṃ samparipluto._  

</Paragraph>

</Division>

_Tadeva me tvaṃ vacanaṃ,_  
_yācito kattumarahasi._  

</Paragraph>

</Division>

_yāva rājānamānaye;_
