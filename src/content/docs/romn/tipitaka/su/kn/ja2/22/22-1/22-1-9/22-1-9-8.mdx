<Paragraph number="2892" type="verses" book="ja2">
_So tattha gantvāna vicintayanto,_  
_“Uccāvacā cetanakā bhavanti;_  
_Nayimassa jīvena mamatthi kiñci,_  
_Hantvānimaṃ hadayamānayissaṃ”._  

</Paragraph>

</Division>
<Division number="1550" book="ja2" bookNo="23" page="297">

<Paragraph number="2893" type="verses" book="ja2">
_So tattha gantvā pabbatantarasmiṃ,_  
_Anto pavisitvāna paduṭṭhacitto;_  
_Asaṃvutasmiṃ jagatippadese,_  
_Adhosiraṃ dhārayi kātiyāno._  

</Paragraph>

</Division>
<Division number="1551" book="ja2" bookNo="23" page="297">

<Paragraph number="2894" type="verses" book="ja2">
_So lambamāno narake papāte,_  
_Mahabbhaye lomahaṃse vidugge;_  
_Asantasanto kurūnaṃ kattuseṭṭho,_  
_Iccabravi puṇṇakaṃ nāma yakkhaṃ._  

</Paragraph>

</Division>
<Division number="1552" book="ja2" bookNo="23" page="298">

<Paragraph number="2895" type="verses" book="ja2">
_“Ariyāvakāsosi anariyarūpo,_  
_Asaññato saññatasannikāso;_  
_Accāhitaṃ kammaṃ karosi ludraṃ,_  
_Bhāve ca te kusalaṃ natthi kiñci._  

</Paragraph>

</Division>
<Division number="1553" book="ja2" bookNo="23" page="298">

<Paragraph number="2896" type="verses" book="ja2">
_Yaṃ maṃ papātasmiṃ papātumicchasi,_  
_Ko nu tavattho maraṇena mayhaṃ;_  
_Amānusasseva tavajja vaṇṇo,_  
_Ācikkha me tvaṃ katamāsi devatā”._  

</Paragraph>

</Division>
<Division number="1554" book="ja2" bookNo="23" page="298">

<Paragraph number="2897" type="verses" book="ja2">
_“Yadi te suto puṇṇako nāma yakkho,_  
_Rañño kuverassa hi so sajibbo;_  
_Bhūmindharo varuṇo nāma nāgo,_  
_Brahā sucī vaṇṇabalūpapanno._  

</Paragraph>

</Division>
<Division number="1555" book="ja2" bookNo="23" page="298">

<Paragraph number="2898" type="verses" book="ja2">
_Tassānujaṃ dhītaraṃ kāmayāmi,_  
_Irandhatī nāma sā nāgakaññā;_  
_Tassā sumajjhāya piyāya hetu,_  
_Patārayiṃ tuyha vadhāya dhīra”._  

</Paragraph>

</Division>
<Division number="1556" book="ja2" bookNo="23" page="298">

<Paragraph number="2899" type="verses" book="ja2">
_“Mā heva tvaṃ yakkha ahosi mūḷho,_  
_Naṭṭhā bahū duggahītena loke;_  
_Kiṃ te sumajjhāya piyāya kiccaṃ,_  
_Maraṇena me iṅgha suṇomi sabbaṃ”._  

</Paragraph>

</Division>
<Division number="1557" book="ja2" bookNo="23" page="298">

<Paragraph number="2900" type="verses" book="ja2">
_“Mahānubhāvassa mahoragassa,_  
_Dhītukāmo ñātibhatohamasmi;_  
_Taṃ yācamānaṃ sasuro avoca,_  
_Yathā mamaññiṃsu sukāmanītaṃ._  

</Paragraph>

</Division>
<Division number="1558" book="ja2" bookNo="23" page="299">

<Paragraph number="2901" type="verses" book="ja2">
_Dajjemu kho te sutanuṃ sunettaṃ,_  
_Sucimhitaṃ candanalittagattaṃ;_  
_Sace tuvaṃ hadayaṃ paṇḍitassa,_  
_Dhammena laddhā idha māharesi;_  
_Etena vittena kumāri labbhā,_  
_Naññaṃ dhanaṃ uttari patthayāma._  

</Paragraph>

</Division>
<Division number="1559" book="ja2" bookNo="23" page="299">

<Paragraph number="2902" type="verses" book="ja2">
_Evaṃ na mūḷhosmi suṇohi katte,_  
_Na cāpi me duggahitatthi kiñci;_  
_Hadayena te dhammaladdhena nāgā,_  
_Irandhatiṃ nāgakaññaṃ dadanti._  

</Paragraph>

</Division>
<Division number="1560" book="ja2" bookNo="23" page="299">

<Paragraph number="2903" type="verses" book="ja2">
_Tasmā ahaṃ tuyhaṃ vadhāya yutto,_  
_Evaṃ mamattho maraṇena tuyhaṃ;_  
_Idheva taṃ narake pātayitvā,_  
_Hantvāna taṃ hadayamānayissaṃ”._  

</Paragraph>

</Division>
<Division number="1561" book="ja2" bookNo="23" page="299">

<Paragraph number="2904" type="verses" book="ja2">
_“Khippaṃ mamaṃ uddhara kātiyāna,_  
_Hadayena me yadi te atthi kiccaṃ;_  
_Ye kecime sādhunarassa dhammā,_  
_Sabbeva te pātukaromi ajja”._  

</Paragraph>

</Division>
<Division number="1562" book="ja2" bookNo="23" page="299">

<Paragraph number="2905" type="verses" book="ja2">
_So puṇṇako kurūnaṃ kattuseṭṭhaṃ,_  
_Nagamuddhani khippaṃ patiṭṭhapetvā;_  
_Assatthamāsīnaṃ samekkhiyāna,_  
_Paripucchi kattāramanomapaññaṃ._  

</Paragraph>

</Division>
<Division number="1563" book="ja2" bookNo="23" page="299">

<Paragraph number="2906" type="verses" book="ja2">
_“Samuddhato mesi tuvaṃ papātā,_  
_Hadayena te ajja mamatthi kiccaṃ;_  
_Ye kecime sādhunarassa dhammā,_  
_Sabbeva me pātukarohi ajja”._  

</Paragraph>

</Division>
<Division number="1564" book="ja2" bookNo="23" page="299">

<Paragraph number="2907" type="verses" book="ja2">
_“Samuddhato tyasmi ahaṃ papātā,_  
_Hadayena me yadi te atthi kiccaṃ;_  
_Ye kecime sādhunarassa dhammā,_  
_Sabbeva te pātukaromi ajja._  

</Paragraph>

</Division>
<Division number="1565" book="ja2" bookNo="23" page="300">

<Paragraph number="2908" type="verses" book="ja2">
_Yātānuyāyī ca bhavāhi māṇava,_  
_Allañca pāṇiṃ parivajjayassu;_  
_Mā cassu mittesu kadāci dubbhī,_  
_Mā ca vasaṃ asatīnaṃ nigacche”._  

</Paragraph>

</Division>
<Division number="1566" book="ja2" bookNo="23" page="300">

<Paragraph number="2909" type="verses" book="ja2">
_“Kathaṃ nu yātaṃ anuyāyī hoti,_  
_Allañca pāṇiṃ dahate kathaṃ so;_  
_Asatī ca kā ko pana mittadubbho,_  
_Akkhāhi me pucchito etamatthaṃ”._  

</Paragraph>

</Division>
<Division number="1567" book="ja2" bookNo="23" page="300">

<Paragraph number="2910" type="verses" book="ja2">
_“Asanthutaṃ nopi ca diṭṭhapubbaṃ,_  
_Yo āsanenāpi nimantayeyya;_  
_Tasseva atthaṃ puriso kareyya,_  
_Yātānuyāyīti tamāhu paṇḍitā._  

</Paragraph>

</Division>
<Division number="1568" book="ja2" bookNo="23" page="300">

<Paragraph number="2911" type="verses" book="ja2">
_Yassekarattampi ghare vaseyya,_  
_Yatthannapānaṃ puriso labheyya;_  
_Na tassa pāpaṃ manasāpi cintaye,_  
_Adubbhapāṇiṃ dahate mittadubbho._  

</Paragraph>

</Division>
<Division number="1569" book="ja2" bookNo="23" page="300">

<Paragraph number="2912" type="verses" book="ja2">
_Yassa rukkhassa chāyāya,_  
_nisīdeyya sayeyya vā;_  
_Na tassa sākhaṃ bhañjeyya,_  
_mittadubbho hi pāpako._  

</Paragraph>

</Division>
<Division number="1570" book="ja2" bookNo="23" page="300">

<Paragraph number="2913" type="verses" book="ja2">
_Puṇṇampi cemaṃ pathaviṃ dhanena,_  
_Dajjitthiyā puriso sammatāya;_  
_Laddhā khaṇaṃ atimaññeyya tampi,_  
_Tāsaṃ vasaṃ asatīnaṃ na gacche._  

</Paragraph>

</Division>
<Division number="1571" book="ja2" bookNo="23" page="300">

<Paragraph number="2914" type="verses" book="ja2">
_Evaṃ kho yātaṃ anuyāyī hoti,_  
_Allañca pāṇiṃ dahate punevaṃ;_  
_Asatī ca sā so pana mittadubbho,_  
_So dhammiko hohi jahassu adhammaṃ”._  


</Paragraph>
</Division>

<Paragraph number="2915" type="center" book="ja2">
Sādhunaradhammakaṇḍaṃ nāma.
</Paragraph>
