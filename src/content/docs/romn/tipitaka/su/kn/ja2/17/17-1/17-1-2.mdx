<Paragraph number="52" type="verses" book="ja2">
_“Alaṅkatā kuṇḍalino suvatthā,_  
_Veḷuriyamuttātharukhaggabandhā;_  
_Rathesabhā tiṭṭhatha ke nu tumhe,_  
_Kathaṃ vo jānanti manussaloke”._  

</Paragraph>

</Division>
<Division number="51" book="ja2">

<Paragraph number="53" type="verses" book="ja2">
_“Ahamaṭṭhako bhīmaratho panāyaṃ,_  
_Kāliṅgarājā pana uggatoyaṃ;_  
_Susaññatānaṃ isīnaṃ dassanāya,_  
_Idhāgatā pucchitāyemha pañhe”._  

</Paragraph>

</Division>
<Division number="52" book="ja2">

<Paragraph number="54" type="verses" book="ja2">
_“Vehāyasaṃ tiṭṭhasi antalikkhe,_  
_Pathaddhuno pannaraseva cando;_  
_Pucchāmi taṃ yakkha mahānubhāva,_  
_Kathaṃ taṃ jānanti manussaloke”._  

</Paragraph>

</Division>
<Division number="53" book="ja2">

<Paragraph number="55" type="verses" book="ja2">
_“Yamāhu devesu sujampatīti,_  
_Maghavāti taṃ āhu manussaloke;_  
_Sa devarājā idamajja patto,_  
_Susaññatānaṃ isīnaṃ dassanāya”._  

</Paragraph>

</Division>
<Division number="54" book="ja2">

<Paragraph number="56" type="verses" book="ja2">
_“Dūre sutā no isayo samāgatā,_  
_Mahiddhikā iddhiguṇūpapannā;_  
_Vandāmi te ayire pasannacitto,_  
_Ye jīvalokettha manussaseṭṭhā”._  

</Paragraph>

</Division>
<Division number="55" book="ja2">

<Paragraph number="57" type="verses" book="ja2">
_“Gandho isīnaṃ ciradikkhitānaṃ,_  
_Kāyā cuto gacchati mālutena;_  
_Ito paṭikkamma sahassanetta,_  
_Gandho isīnaṃ asuci devarāja”._  

</Paragraph>

</Division>
<Division number="56" book="ja2">

<Paragraph number="58" type="verses" book="ja2">
_“Gandho isīnaṃ ciradikkhitānaṃ,_  
_Kāyā cuto gacchatu mālutena;_  
_Vicitrapupphaṃ surabhiṃva mālaṃ,_  
_Gandhañca etaṃ pāṭikaṅkhāma bhante;_  
_Na hettha devā paṭikkūlasaññino”._  

</Paragraph>

</Division>
<Division number="57" book="ja2">

<Paragraph number="59" type="verses" book="ja2">
_“Purindado bhūtapatī yasassī,_  
_Devānamindo sakko maghavā sujampati;_  
_Sa devarājā asuragaṇappamaddano,_  
_Okāsamākaṅkhati pañha pucchituṃ._  

</Paragraph>

</Division>
<Division number="58" book="ja2">

<Paragraph number="60" type="verses" book="ja2">
_Ko nevimesaṃ idha paṇḍitānaṃ,_  
_Pañhe puṭṭho nipuṇe byākarissati;_  
_Tiṇṇañca raññaṃ manujādhipānaṃ,_  
_Devānamindassa ca vāsavassa”._  

</Paragraph>

</Division>
<Division number="59" book="ja2">

<Paragraph number="61" type="verses" book="ja2">
_“Ayaṃ isi sarabhaṅgo tapassī,_  
_Yato jāto virato methunasmā;_  
_Āceraputto suvinītarūpo,_  
_So nesaṃ pañhāni viyākarissati”._  

</Paragraph>

</Division>
<Division number="60" book="ja2">

<Paragraph number="62" type="verses" book="ja2">
_“Koṇḍañña pañhāni viyākarohi,_  
_Yācanti taṃ isayo sādhurūpā;_  
_Koṇḍañña eso manujesu dhammo,_  
_Yaṃ vuddhamāgacchati esa bhāro”._  

</Paragraph>

</Division>
<Division number="61" book="ja2">

<Paragraph number="63" type="verses" book="ja2">
_“Katāvakāsā pucchantu bhonto,_  
_Yaṃ kiñci pañhaṃ manasābhipatthitaṃ;_  
_Ahañhi taṃ taṃ vo viyākarissaṃ,_  
_Ñatvā sayaṃ lokamimaṃ parañca”._  

</Paragraph>

</Division>
<Division number="62" book="ja2">

<Paragraph number="64" type="verses" book="ja2">
_Tato ca maghavā sakko,_  
_atthadassī purindado;_  
_Apucchi paṭhamaṃ pañhaṃ,_  
_yañcāsi abhipatthitaṃ._  

</Paragraph>

</Division>
<Division number="63" book="ja2">

<Paragraph number="65" type="verses" book="ja2">
_“Kiṃ sū vadhitvā na kadāci socati,_  
_Kissappahānaṃ isayo vaṇṇayanti;_  
_Kassīdha vuttaṃ pharusaṃ khametha,_  
_Akkhāhi me koṇḍañña etamatthaṃ”._  

</Paragraph>

</Division>
<Division number="64" book="ja2">

<Paragraph number="66" type="verses" book="ja2">
_“Kodhaṃ vadhitvā na kadāci socati,_  
_Makkhappahānaṃ isayo vaṇṇayanti;_  
_Sabbesaṃ vuttaṃ pharusaṃ khametha,_  
_Etaṃ khantiṃ uttamamāhu santo”._  

</Paragraph>

</Division>
<Division number="65" book="ja2">

<Paragraph number="67" type="verses" book="ja2">
_“Sakkā ubhinnaṃ vacanaṃ titikkhituṃ,_  
_Sadisassa vā seṭṭhatarassa vāpi;_  
_Kathaṃ nu hīnassa vaco khametha,_  
_Akkhāhi me koṇḍañña etamatthaṃ”._  

</Paragraph>

</Division>
<Division number="66" book="ja2">

<Paragraph number="68" type="verses" book="ja2">
_“Bhayā hi seṭṭhassa vaco khametha,_  
_Sārambhahetū pana sādisassa;_  
_Yo cīdha hīnassa vaco khametha,_  
_Etaṃ khantiṃ uttamamāhu santo”._  

</Paragraph>

</Division>
<Division number="67" book="ja2">

<Paragraph number="69" type="verses" book="ja2">
_“Kathaṃ vijaññā catupattharūpaṃ,_  
_Seṭṭhaṃ sarikkhaṃ athavāpi hīnaṃ;_  
_Virūparūpena caranti santo,_  
_Tasmā hi sabbesaṃ vaco khametha”._  

</Paragraph>

</Division>
<Division number="68" book="ja2">

<Paragraph number="70" type="verses" book="ja2">
_“Na hetamatthaṃ mahatīpi senā,_  
_Sarājikā yujjhamānā labhetha;_  
_Yaṃ khantimā sappuriso labhetha,_  
_Khantī balassūpasamanti verā”._  

</Paragraph>

</Division>
<Division number="69" book="ja2">

<Paragraph number="71" type="verses" book="ja2">
_“Subhāsitaṃ te anumodiyāna,_  
_Aññaṃ taṃ pucchāmi tadiṅgha brūhi;_  
_Yathā ahuṃ daṇḍakī nāḷikero,_  
_Athajjuno kalābu cāpi rājā;_  
_Tesaṃ gatiṃ brūhi supāpakamminaṃ,_  
_Katthūpapannā isinaṃ viheṭhakā”._  

</Paragraph>

</Division>
<Division number="70" book="ja2">

<Paragraph number="72" type="verses" book="ja2">
_“Kisañhi vacchaṃ avakiriya daṇḍakī,_  
_Ucchinnamūlo sajano saraṭṭho;_  
_Kukkuḷanāme nirayamhi paccati,_  
_Tassa phuliṅgāni patanti kāye._  

</Paragraph>

</Division>
<Division number="71" book="ja2">

<Paragraph number="73" type="verses" book="ja2">
_Yo saññate pabbajite aheṭhayi,_  
_Dhammaṃ bhaṇante samaṇe adūsake;_  
_Taṃ nāḷikeraṃ sunakhā parattha,_  
_Saṅgamma khādanti viphandamānaṃ._  

</Paragraph>

</Division>
<Division number="72" book="ja2">

<Paragraph number="74" type="verses" book="ja2">
_Athajjuno niraye sattisūle,_  
_Avaṃsiro patito uddhampādo;_  
_Aṅgīrasaṃ gotamaṃ heṭhayitvā,_  
_Khantiṃ tapassiṃ cirabrahmacāriṃ._  

</Paragraph>

</Division>
<Division number="73" book="ja2">

<Paragraph number="75" type="verses" book="ja2">
_Yo khaṇḍaso pabbajitaṃ achedayi,_  
_Khantiṃ vadantaṃ samaṇaṃ adūsakaṃ;_  
_Kalābuvīciṃ upapajja paccati,_  
_Mahāpatāpaṃ kaṭukaṃ bhayānakaṃ._  

</Paragraph>

</Division>
<Division number="74" book="ja2">

<Paragraph number="76" type="verses" book="ja2">
_Etāni sutvā nirayāni paṇḍito,_  
_Aññāni pāpiṭṭhatarāni cettha;_  
_Dhammaṃ care samaṇabrāhmaṇesu,_  
_Evaṃkaro saggamupeti ṭhānaṃ”._  

</Paragraph>

</Division>
<Division number="75" book="ja2">

<Paragraph number="77" type="verses" book="ja2">
_“Subhāsitaṃ te anumodiyāna,_  
_Aññaṃ taṃ pucchāmi tadiṅgha brūhi;_  
_Kathaṃvidhaṃ sīlavantaṃ vadanti,_  
_Kathaṃvidhaṃ paññavantaṃ vadanti;_  
_Kathaṃvidhaṃ sappurisaṃ vadanti,_  
_Kathaṃvidhaṃ no siri no jahāti”._  

</Paragraph>

</Division>
<Division number="76" book="ja2">

<Paragraph number="78" type="verses" book="ja2">
_“Kāyena vācāya ca yodha saññato,_  
_Manasā ca kiñci na karoti pāpaṃ;_  
_Na attahetū alikaṃ bhaṇeti,_  
_Tathāvidhaṃ sīlavantaṃ vadanti._  

</Paragraph>

</Division>
<Division number="77" book="ja2">

<Paragraph number="79" type="verses" book="ja2">
_Gambhīrapañhaṃ manasābhicintayaṃ,_  
_Nāccāhitaṃ kamma karoti luddaṃ;_  
_Kālāgataṃ atthapadaṃ na riñcati,_  
_Tathāvidhaṃ paññavantaṃ vadanti._  

</Paragraph>

</Division>
<Division number="78" book="ja2">

<Paragraph number="80" type="verses" book="ja2">
_Yo ve kataññū katavedi dhīro,_  
_Kalyāṇamitto daḷhabhatti ca hoti;_  
_Dukhitassa sakkacca karoti kiccaṃ,_  
_Tathāvidhaṃ sappurisaṃ vadanti._  

</Paragraph>

</Division>
<Division number="79" book="ja2">

<Paragraph number="81" type="verses" book="ja2">
_Etehi sabbehi guṇehupeto,_  
_Saddho mudū saṃvibhāgī vadaññū;_  
_Saṅgāhakaṃ sakhilaṃ saṇhavācaṃ,_  
_Tathāvidhaṃ no siri no jahāti”._  

</Paragraph>

</Division>
<Division number="80" book="ja2">

<Paragraph number="82" type="verses" book="ja2">
_“Subhāsitaṃ te anumodiyāna,_  
_Aññaṃ taṃ pucchāmi tadiṅgha brūhi;_  
_Sīlaṃ siriñcāpi satañca dhammaṃ,_  
_Paññañca kaṃ seṭṭhataraṃ vadanti”._  

</Paragraph>

</Division>
<Division number="81" book="ja2">

<Paragraph number="83" type="verses" book="ja2">
_“Paññā hi seṭṭhā kusalā vadanti,_  
_Nakkhattarājāriva tārakānaṃ;_  
_Sīlaṃ sīrī cāpi satañca dhammo,_  
_Anvāyikā paññavato bhavanti”._  

</Paragraph>

</Division>
<Division number="82" book="ja2">

<Paragraph number="84" type="verses" book="ja2">
_“Subhāsitaṃ te anumodiyāna,_  
_Aññaṃ taṃ pucchāmi tadiṅgha brūhi;_  
_Kathaṃkaro kintikaro kimācaraṃ,_  
_Kiṃ sevamāno labhatīdha paññaṃ;_  
_Paññāya dāni paṭipadaṃ vadehi,_  
_Kathaṃkaro paññavā hoti macco”._  

</Paragraph>

</Division>
<Division number="83" book="ja2">

<Paragraph number="85" type="verses" book="ja2">
_“Sevetha vuddhe nipuṇe bahussute,_  
_Uggāhako ca paripucchako siyā;_  
_Suṇeyya sakkacca subhāsitāni,_  
_Evaṃkaro paññavā hoti macco._  

</Paragraph>

</Division>
<Division number="84" book="ja2">

<Paragraph number="86" type="verses" book="ja2">
_Sa paññavā kāmaguṇe avekkhati,_  
_Aniccato dukkhato rogato ca;_  
_Evaṃ vipassī pajahāti chandaṃ,_  
_Dukkhesu kāmesu mahabbhayesu._  

</Paragraph>

</Division>
<Division number="85" book="ja2">

<Paragraph number="87" type="verses" book="ja2">
_Sa vītarāgo pavineyya dosaṃ,_  
_Mettaṃ cittaṃ bhāvaye appamāṇaṃ;_  
_Sabbesu bhūtesu nidhāya daṇḍaṃ,_  
_Anindito brahmamupeti ṭhānaṃ”._  

</Paragraph>

</Division>
<Division number="86" book="ja2">

<Paragraph number="88" type="verses" book="ja2">
_“Mahatthiyaṃ āgamanaṃ ahosi,_  
_Tavamaṭṭhakā bhīmarathassa cāpi;_  
_Kāliṅgarājassa ca uggatassa,_  
_Sabbesa vo kāmarāgo pahīno”._  

</Paragraph>

</Division>
<Division number="87" book="ja2">

<Paragraph number="89" type="verses" book="ja2">
_“Evametaṃ paracittavedi,_  
_Sabbesa no kāmarāgo pahīno;_  
_Karohi okāsamanuggahāya,_  
_Yathā gatiṃ te abhisambhavema”._  

</Paragraph>

</Division>
<Division number="88" book="ja2">

<Paragraph number="90" type="verses" book="ja2">
_“Karomi okāsamanuggahāya,_  
_Tathā hi vo kāmarāgo pahīno;_  
_Pharātha kāyaṃ vipulāya pītiyā,_  
_Yathā gatiṃ me abhisambhavetha”._  

</Paragraph>

</Division>
<Division number="89" book="ja2">

<Paragraph number="91" type="verses" book="ja2">
_“Sabbaṃ karissāma tavānusāsaniṃ,_  
_Yaṃ yaṃ tuvaṃ vakkhasi bhūripañña;_  
_Pharāma kāyaṃ vipulāya pītiyā,_  
_Yathā gatiṃ te abhisambhavema”._  

</Paragraph>

</Division>
<Division number="90" book="ja2">

<Paragraph number="92" type="verses" book="ja2">
_“Katāya vacchassa kisassa pūjā,_  
_Gacchantu bhonto isayo sādhurūpā;_  
_Jhāne ratā hotha sadā samāhitā,_  
_Esā ratī pabbajitassa seṭṭhā”._  

</Paragraph>

</Division>
<Division number="91" book="ja2">

<Paragraph number="93" type="verses" book="ja2">
_Sutvāna gāthā paramatthasaṃhitā,_  
_Subhāsitā isinā paṇḍitena;_  
_Te vedajātā anumodamānā,_  
_Pakkāmu devā devapuraṃ yasassino._  

</Paragraph>

</Division>
<Division number="92" book="ja2">

<Paragraph number="94" type="verses" book="ja2">
_Gāthā imā atthavatī subyañjanā,_  
_Subhāsitā isinā paṇḍitena;_  
_Yo kocimā aṭṭhikatvā suṇeyya,_  
_Labhetha pubbāpariyaṃ visesaṃ;_  
_Laddhāna pubbāpariyaṃ visesaṃ,_  
_Adassanaṃ maccurājassa gacche._  

</Paragraph>

</Division>
<Division number="93" book="ja2">

<Paragraph number="95" type="verses" book="ja2">
_“Sālissaro sāriputto,_  
_meṇḍissaro ca kassapo;_  
_Pabbato anuruddho ca,_  
_kaccāyano ca devalo._  

</Paragraph>

</Division>
<Division number="94" book="ja2">

<Paragraph number="96" type="verses" book="ja2">
_Anusisso ca ānando,_  
_kisavaccho ca kolito;_  
_Nārado udāyitthero,_  
_parisā buddhaparisā;_  
_Sarabhaṅgo lokanātho,_  
_evaṃ dhāretha jātakan”ti._  


</Paragraph>
</Division>
