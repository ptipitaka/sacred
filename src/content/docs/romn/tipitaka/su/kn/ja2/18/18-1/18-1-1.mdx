<Division number="1" book="ja2">

<Paragraph number="255" type="verses" book="ja2">
_“Uddayhate janapado,_  
_raṭṭhañcāpi vinassati;_  
_Ehi niḷinike gaccha,_  
_taṃ me brāhmaṇamānaya”._  

</Paragraph>

</Division>
<Division number="2" book="ja2">

<Paragraph number="256" type="verses" book="ja2">
_“Nāhaṃ dukkhakkhamā rāja,_  
_nāhaṃ addhānakovidā;_  
_Kathaṃ ahaṃ gamissāmi,_  
_vanaṃ kuñjarasevitaṃ”._  

</Paragraph>

</Division>
<Division number="3" book="ja2">

<Paragraph number="257" type="verses" book="ja2">
_“Phītaṃ janapadaṃ gantvā,_  
_hatthinā ca rathena ca;_  
_Dārusaṅghāṭayānena,_  
_evaṃ gaccha niḷinike._  

</Paragraph>

</Division>
<Division number="4" book="ja2">

<Paragraph number="258" type="verses" book="ja2">
_Hatthiassarathe pattī,_  
_gacchevādāya khattiye;_  
_Taveva vaṇṇarūpena,_  
_vasaṃ tamānayissasi”._  

</Paragraph>

</Division>
<Division number="5" book="ja2">

<Paragraph number="259" type="verses" book="ja2">
_“Kadalīdhajapaññāṇo,_  
_ābhujīparivārito;_  
_Eso padissati rammo,_  
_isisiṅgassa assamo._  

</Paragraph>

</Division>
<Division number="6" book="ja2">

<Paragraph number="260" type="verses" book="ja2">
_Eso aggissa saṅkhāto,_  
_eso dhūmo padissati;_  
_Maññe no aggiṃ hāpeti,_  
_isisiṅgo mahiddhiko”._  

</Paragraph>

</Division>
<Division number="7" book="ja2">

<Paragraph number="261" type="verses" book="ja2">
_“Tañca disvāna āyantiṃ,_  
_āmuttamaṇikuṇḍalaṃ;_  
_Isisiṅgo pāvisi bhīto,_  
_assamaṃ paṇṇachādanaṃ._  

</Paragraph>

</Division>
<Division number="8" book="ja2">

<Paragraph number="262" type="verses" book="ja2">
_Assamassa ca sā dvāre,_  
_geṇḍukenassa kīḷati;_  
_Vidaṃsayantī aṅgāni,_  
_guyhaṃ pakāsitāni ca._  

</Paragraph>

</Division>
<Division number="9" book="ja2">

<Paragraph number="263" type="verses" book="ja2">
_Tañca disvāna kīḷantiṃ,_  
_paṇṇasālagato jaṭī;_  
_Assamā nikkhamitvāna,_  
_idaṃ vacanamabravi._  

</Paragraph>

</Division>
<Division number="10" book="ja2">

<Paragraph number="264" type="verses" book="ja2">
_Ambho ko nāma so rukkho,_  
_yassa tevaṃgataṃ phalaṃ;_  
_Dūrepi khittaṃ pacceti,_  
_na taṃ ohāya gacchati”._  

</Paragraph>

</Division>
<Division number="11" book="ja2">

<Paragraph number="265" type="verses" book="ja2">
_“Assamassa mama brahme,_  
_samīpe gandhamādane;_  
_Bahavo tādisā rukkhā,_  
_yassa tevaṃgataṃ phalaṃ;_  
_Dūrepi khittaṃ pacceti,_  
_na maṃ ohāya gacchati”._  

</Paragraph>

</Division>
<Division number="12" book="ja2">

<Paragraph number="266" type="verses" book="ja2">
_“Etū bhavaṃ assamimaṃ adetu,_  
_Pajjañca bhakkhañca paṭiccha dammi;_  
_Idamāsanaṃ atra bhavaṃ nisīdatu,_  
_Ito bhavaṃ mūlaphalāni bhuñjatu._  

</Paragraph>

</Division>
<Division number="13" book="ja2">

<Paragraph number="267" type="verses" book="ja2">
_Kiṃ te idaṃ ūrūnamantarasmiṃ,_  
_Supicchitaṃ kaṇharivappakāsati;_  
_Akkhāhi me pucchito etamatthaṃ,_  
_Kose nu te uttamaṅgaṃ paviṭṭhaṃ”._  

</Paragraph>

</Division>
<Division number="14" book="ja2">

<Paragraph number="268" type="verses" book="ja2">
_“Ahaṃ vane mūlaphalesanaṃ caraṃ,_  
_Āsādayiṃ acchaṃ sughorarūpaṃ;_  
_So maṃ patitvā sahasājjhapatto,_  
_Panujja maṃ abbahi uttamaṅgaṃ._  

</Paragraph>

</Division>
<Division number="15" book="ja2">

<Paragraph number="269" type="verses" book="ja2">
_Svāyaṃ vaṇo khajjati kaṇḍuvāyati,_  
_Sabbañca kālaṃ na labhāmi sātaṃ;_  
_Paho bhavaṃ kaṇḍumimaṃ vinetuṃ,_  
_Kurutaṃ bhavaṃ yācito brāhmaṇatthaṃ”._  

</Paragraph>

</Division>
<Division number="16" book="ja2">

<Paragraph number="270" type="verses" book="ja2">
_“Gambhīrarūpo te vaṇo salohito,_  
_Apūtiko vaṇagandho mahā ca;_  
_Karomi te kiñci kasāyayogaṃ,_  
_Yathā bhavaṃ paramasukhī bhaveyya”._  

</Paragraph>

</Division>
<Division number="17" book="ja2">

<Paragraph number="271" type="verses" book="ja2">
_“Na mantayogā na kasāyayogā,_  
_Na osadhā brahmacāri kamanti;_  
_Yaṃ te mudu tena vinehi kaṇḍuṃ,_  
_Yathā ahaṃ paramasukhī bhaveyyaṃ”._  

</Paragraph>

</Division>
<Division number="18" book="ja2">

<Paragraph number="272" type="verses" book="ja2">
_“Ito nu bhoto katamena assamo,_  
_Kacci bhavaṃ abhiramasi araññe;_  
_Kacci nu te mūlaphalaṃ pahūtaṃ,_  
_Kacci bhavantaṃ na vihiṃsanti vāḷā”._  

</Paragraph>

</Division>
<Division number="19" book="ja2">

<Paragraph number="273" type="verses" book="ja2">
_“Ito ujuṃ uttarāyaṃ disāyaṃ,_  
_Khemānadī himavatā pabhāvī;_  
_Tassā tīre assamo mayha rammo,_  
_Aho bhavaṃ assamaṃ mayhaṃ passe._  

</Paragraph>

</Division>
<Division number="20" book="ja2">

<Paragraph number="274" type="verses" book="ja2">
_Ambā ca sālā tilakā ca jambuyo,_  
_Uddālakā pāṭaliyo ca phullā;_  
_Samantato kimpurisābhigītaṃ,_  
_Aho bhavaṃ assamaṃ mayhaṃ passe._  

</Paragraph>

</Division>
<Division number="21" book="ja2">

<Paragraph number="275" type="verses" book="ja2">
_Tālā ca mūlā ca phalā ca mettha,_  
_Vaṇṇena gandhena upetarūpaṃ;_  
_Taṃ bhūmibhāgehi upetarūpaṃ,_  
_Aho bhavaṃ assamaṃ mayhaṃ passe._  

</Paragraph>

</Division>
<Division number="22" book="ja2">

<Paragraph number="276" type="verses" book="ja2">
_Phalā ca mūlā ca pahūtamettha,_  
_Vaṇṇena gandhena rasenupetā;_  
_Āyanti ca luddakā taṃ padesaṃ,_  
_Mā me tato mūlaphalaṃ ahāsuṃ”._  

</Paragraph>

</Division>
<Division number="23" book="ja2">

<Paragraph number="277" type="verses" book="ja2">
_“Pitā mamaṃ mūlaphalesanaṃ gato,_  
_Idāni āgacchati sāyakāle;_  
_Ubhova gacchāmase assamaṃ taṃ,_  
_Yāva pitā mūlaphalato etu”._  

</Paragraph>

</Division>
<Division number="24" book="ja2">

<Paragraph number="278" type="verses" book="ja2">
_“Aññe bahū isayo sādhurūpā,_  
_Rājīsayo anumagge vasanti;_  
_Teyeva pucchesi mamassamaṃ taṃ,_  
_Te taṃ nayissanti mamaṃ sakāse”._  

</Paragraph>

</Division>
<Division number="25" book="ja2">

<Paragraph number="279" type="verses" book="ja2">
_“Na te kaṭṭhāni bhinnāni,_  
_na te udakamābhataṃ;_  
_Aggīpi te na hāpito,_  
_kiṃ nu mandova jhāyasi._  

</Paragraph>

</Division>
<Division number="26" book="ja2">

<Paragraph number="280" type="verses" book="ja2">
_Bhinnāni kaṭṭhāni huto ca aggi,_  
_Tapanīpi te samitā brahmacārī;_  
_Pīṭhañca mayhaṃ udakañca hoti,_  
_Ramasi tuvaṃ brahmabhūto puratthā._  

</Paragraph>

</Division>
<Division number="27" book="ja2">

<Paragraph number="281" type="verses" book="ja2">
_Abhinnakaṭṭhosi anābhatodako,_  
_Ahāpitaggīsi asiddhabhojano;_  
_Na me tuvaṃ ālapasī mamajja,_  
_Naṭṭhaṃ nu kiṃ cetasikañca dukkhaṃ”._  

</Paragraph>

</Division>
<Division number="28" book="ja2">

<Paragraph number="282" type="verses" book="ja2">
_“Idhāgamā jaṭilo brahmacārī,_  
_Sudassaneyyo sutanū vineti;_  
_Nevātidīgho na panātirasso,_  
_Sukaṇhakaṇhacchadanehi bhoto._  

</Paragraph>

</Division>
<Division number="29" book="ja2">

<Paragraph number="283" type="verses" book="ja2">
_Amassujāto apurāṇavaṇṇī,_  
_Ādhārarūpañca panassa kaṇṭhe;_  
_Dve yamā gaṇḍā ure sujātā,_  
_Suvaṇṇatindukanibhā pabhassarā._  

</Paragraph>

</Division>
<Division number="30" book="ja2">

<Paragraph number="284" type="verses" book="ja2">
_Mukhañca tassa bhusadassaneyyaṃ,_  
_Kaṇṇesu lambanti ca kuñcitaggā;_  
_Te jotare carato māṇavassa,_  
_Suttañca yaṃ saṃyamanaṃ jaṭānaṃ._  

</Paragraph>

</Division>
<Division number="31" book="ja2">

<Paragraph number="285" type="verses" book="ja2">
_Aññā ca tassa saṃyamāni catasso,_  
_Nīlā pītā lohitikā ca setā;_  
_Tā piṃsare carato māṇavassa,_  
_Tiriṭisaṅghāriva pāvusamhi._  

</Paragraph>

</Division>
<Division number="32" book="ja2">

<Paragraph number="286" type="verses" book="ja2">
_Na mikhalaṃ muñjamayaṃ dhāreti,_  
_Na santhare no pana pabbajassa;_  
_Tā jotare jaghanantare vilaggā,_  
_Sateratā vijjurivantalikkhe._  

</Paragraph>

</Division>
<Division number="33" book="ja2">

<Paragraph number="287" type="verses" book="ja2">
_Akhīlakāni ca avaṇṭakāni,_  
_Heṭṭhā nabhyā kaṭisamohitāni;_  
_Aghaṭṭitā niccakīḷaṃ karonti,_  
_Haṃ tāta kiṃrukkhaphalāni tāni._  

</Paragraph>

</Division>
<Division number="34" book="ja2">

<Paragraph number="288" type="verses" book="ja2">
_Jaṭā ca tassa bhusadassaneyyā,_  
_Parosataṃ vellitaggā sugandhā;_  
_Dvedhā siro sādhu vibhattarūpo,_  
_Aho nu kho mayha tathā jaṭāssu._  

</Paragraph>

</Division>
<Division number="35" book="ja2">

<Paragraph number="289" type="verses" book="ja2">
_Yadā ca so pakirati tā jaṭāyo,_  
_Vaṇṇena gandhena upetarūpā;_  
_Nīluppalaṃ vātasameritaṃva,_  
_Tatheva saṃvāti panassamo ayaṃ._  

</Paragraph>

</Division>
<Division number="36" book="ja2">

<Paragraph number="290" type="verses" book="ja2">
_Paṅko ca tassa bhusadassaneyyo,_  
_Netādiso yādiso mayhaṃ kāye;_  
_So vāyatī erito mālutena,_  
_Vanaṃ yathā aggagimhe suphullaṃ._  

</Paragraph>

</Division>
<Division number="37" book="ja2">

<Paragraph number="291" type="verses" book="ja2">
_Nihanti so rukkhaphalaṃ pathabyā,_  
_Sucittarūpaṃ ruciraṃ dassaneyyaṃ;_  
_Khittañca tassa punareti hatthaṃ,_  
_Haṃ tāta kiṃrukkhaphalaṃ nu kho taṃ._  

</Paragraph>

</Division>
<Division number="38" book="ja2">

<Paragraph number="292" type="verses" book="ja2">
_Dantā ca tassa bhusadassaneyyā,_  
_Suddhā samā saṅkhavarūpapannā;_  
_Mano pasādenti vivariyamānā,_  
_Na hi nūna so sākamakhādi tehi._  

</Paragraph>

</Division>
<Division number="39" book="ja2">

<Paragraph number="293" type="verses" book="ja2">
_Akakkasaṃ aggaḷitaṃ muhuṃ muduṃ,_  
_Ujuṃ anuddhataṃ acapalamassa bhāsitaṃ;_  
_Rudaṃ manuññaṃ karavīkasussaraṃ,_  
_Hadayaṅgamaṃ rañjayateva me mano._  

</Paragraph>

</Division>
<Division number="40" book="ja2">

<Paragraph number="294" type="verses" book="ja2">
_Bindussaro nātivisaṭṭhavākyo,_  
_Na nūna sajjhāyamatippayutto;_  
_Icchāmi bho taṃ punadeva daṭṭhuṃ,_  
_Mitto hi me māṇavohu puratthā._  

</Paragraph>

</Division>
<Division number="41" book="ja2">

<Paragraph number="295" type="verses" book="ja2">
_Susandhi sabbattha vimaṭṭhimaṃ vaṇaṃ,_  
_Puthū sujātaṃ kharapattasannitaṃ;_  
_Teneva maṃ uttariyāna māṇavo,_  
_Vivaritaṃ ūruṃ jaghanena piḷayi._  

</Paragraph>

</Division>
<Division number="42" book="ja2">

<Paragraph number="296" type="verses" book="ja2">
_Tapanti ābhanti virocare ca,_  
_Sateratā vijjurivantalikkhe;_  
_Bāhā mudū añjanalomasādisā,_  
_Vicitravaṭṭaṅgulikāssa sobhare._  

</Paragraph>

</Division>
<Division number="43" book="ja2">

<Paragraph number="297" type="verses" book="ja2">
_Akakkasaṅgo na ca dīghalomo,_  
_Nakhāssa dīghā api lohitaggā;_  
_Mudūhi bāhāhi palissajanto,_  
_Kalyāṇarūpo ramayaṃ upaṭṭhahi._  

</Paragraph>

</Division>
<Division number="44" book="ja2">

<Paragraph number="298" type="verses" book="ja2">
_Dumassa tūlūpanibhā pabhassarā,_  
_Suvaṇṇakambutalavaṭṭasucchavi;_  
_Hatthā mudū tehi maṃ samphusitvā,_  
_Ito gato tena maṃ dahanti tāta._  

</Paragraph>

</Division>
<Division number="45" book="ja2">

<Paragraph number="299" type="verses" book="ja2">
_Na nūna so khārividhaṃ ahāsi,_  
_Na nūna so kaṭṭhāni sayaṃ abhañji;_  
_Na nūna so hanti dume kuṭhāriyā,_  
_Na hissa hatthesu khilāni atthi._  

</Paragraph>

</Division>
<Division number="46" book="ja2">

<Paragraph number="300" type="verses" book="ja2">
_Accho ca kho tassa vaṇaṃ akāsi,_  
_So maṃbravi sukhitaṃ maṃ karohi;_  
_Tāhaṃ kariṃ tena mamāsi sokhyaṃ,_  
_So cabravi ‘sukhitosmī’ti brahme._  

</Paragraph>

</Division>
<Division number="47" book="ja2">

<Paragraph number="301" type="verses" book="ja2">
_Ayañca te māluvapaṇṇasanthatā,_  
_Vikiṇṇarūpāva mayā ca tena ca;_  
_Kilantarūpā udake ramitvā,_  
_Punappunaṃ paṇṇakuṭiṃ vajāma._  

</Paragraph>

</Division>
<Division number="48" book="ja2">

<Paragraph number="302" type="verses" book="ja2">
_Na majja mantā paṭibhanti tāta,_  
_Na aggihuttaṃ napi yaññatantaṃ;_  
_Na cāpi te mūlaphalāni bhuñje,_  
_Yāva na passāmi taṃ brahmacāriṃ._  

</Paragraph>

</Division>
<Division number="49" book="ja2">

<Paragraph number="303" type="verses" book="ja2">
_Addhā pajānāsi tuvampi tāta,_  
_Yassaṃ disaṃ vasate brahmacārī;_  
_Taṃ maṃ disaṃ pāpaya tāta khippaṃ,_  
_Mā te ahaṃ amarimassamamhi._  

</Paragraph>

</Division>
<Division number="50" book="ja2">

<Paragraph number="304" type="verses" book="ja2">
_Vicitraphullaṃ hi vanaṃ sutaṃ mayā,_  
_Dijābhighuṭṭhaṃ dijasaṅghasevitaṃ;_  
_Taṃ maṃ vanaṃ pāpaya tāta khippaṃ,_  
_Purā te pāṇaṃ vijahāmi assame”._  

</Paragraph>

</Division>
<Division number="51" book="ja2">

<Paragraph number="305" type="verses" book="ja2">
_“Imasmāhaṃ jotirase vanamhi,_  
_Gandhabbadevaccharasaṅghasevite;_  
_Isīnamāvāse sanantanamhi,_  
_Netādisaṃ aratiṃ pāpuṇetha._  

</Paragraph>

</Division>
<Division number="52" book="ja2">

<Paragraph number="306" type="verses" book="ja2">
_Bhavanti mittāni atho na honti,_  
_Ñātīsu mittesu karonti pemaṃ;_  
_Ayañca jammo kissa vā niviṭṭho,_  
_Yo neva jānāti kutomhi āgato._  

</Paragraph>

</Division>
<Division number="53" book="ja2">

<Paragraph number="307" type="verses" book="ja2">
_Saṃvāsena hi mittāni,_  
_sandhīyanti punappunaṃ;_  
_Sveva mitto asaṅgantu,_  
_asaṃvāsena jīrati._  

</Paragraph>

</Division>
<Division number="54" book="ja2">

<Paragraph number="308" type="verses" book="ja2">
_Sace tuvaṃ dakkhasi brahmacāriṃ,_  
_Sace tuvaṃ sallape brahmacārinā;_  
_Sampannasassaṃva mahodakena,_  
_Tapoguṇaṃ khippamimaṃ pahissasi._  

</Paragraph>

</Division>
<Division number="55" book="ja2">

<Paragraph number="309" type="verses" book="ja2">
_Punapi ce dakkhasi brahmacāriṃ,_  
_Punapi ce sallape brahmacārinā;_  
_Sampannasassaṃva mahodakena,_  
_Usmāgataṃ khippamimaṃ pahissasi._  

</Paragraph>

</Division>
<Division number="56" book="ja2">

<Paragraph number="310" type="verses" book="ja2">
_Bhūtāni hetāni caranti tāta,_  
_Virūparūpena manussaloke;_  
_Na tāni sevetha naro sapañño,_  
_Āsajja naṃ nassati brahmacārī”ti._  


</Paragraph>
</Division>

Niḷinikājātakaṃ paṭhamaṃ.
