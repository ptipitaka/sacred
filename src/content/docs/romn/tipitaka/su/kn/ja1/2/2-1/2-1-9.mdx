<Paragraph number="364" type="verses" book="ja1">
_“Udetayañ cakkhumā ekarājā,_  
_Harissavaṇṇo pathavippabhāso;_  
_Taṃ taṃ namassāmi harissavaṇṇaṃ pathavippabhāsaṃ,_  
_Tayājja guttā viharemu divasaṃ._  

</Paragraph>

<Paragraph number="365" type="verses" book="ja1">
_Ye brāhmaṇā vedagū sabbadhamme,_  
_Te me namo te ca maṃ pālayantu;_  
_Namatthu buddhānaṃ namatthu bodhiyā,_  
_Namo vimuttānaṃ namo vimuttiyā;_  
_Imaṃ so parittaṃ katvā,_  
_Moro carati esanā._  

</Paragraph>

</Division>
<Division number="18" book="ja1">

<Paragraph number="366" type="verses" book="ja1">
_Apetayaṃ cakkhumā ekarājā,_  
_Harissavaṇṇo pathavippabhāso;_  
_Taṃ taṃ namassāmi harissavaṇṇaṃ pathavippabhāsaṃ,_  
_Tayājja guttā viharemu rattiṃ._  

</Paragraph>

<Paragraph number="367" type="verses" book="ja1">
_Ye brāhmaṇā vedagū sabbadhamme,_  
_Te me namo te ca maṃ pālayantu;_  
_Namatthu buddhānaṃ namatthu bodhiyā,_  
_Namo vimuttānaṃ namo vimuttiyā;_  
_Imaṃ so parittaṃ katvā,_  
_Moro vāsamakappayī”ti._  


</Paragraph>
</Division>
