<Paragraph number="2055" type="verses" book="ja1">
_“Vaṇṇavā abhirūposi,_  
_ghano sañjātarohito;_  
_Cakkavāka surūposi,_  
_vippasannamukhindriyo._  

</Paragraph>

</Division>
<Division number="136" book="ja1" bookNo="22" page="59">

<Paragraph number="2056" type="verses" book="ja1">
_Pāṭhīnaṃ pāvusaṃ macchaṃ,_  
_balajaṃ muñjarohitaṃ;_  
_Gaṅgāya tīre nisinno,_  
_evaṃ bhuñjasi bhojanaṃ”._  

</Paragraph>

</Division>
<Division number="137" book="ja1" bookNo="22" page="59">

<Paragraph number="2057" type="verses" book="ja1">
_“Na vāhametaṃ bhuñjāmi,_  
_Jaṅgalānodakāni vā;_  
_Aññatra sevālapaṇakā,_  
_Etaṃ me samma bhojanaṃ”._  

</Paragraph>

</Division>
<Division number="138" book="ja1" bookNo="22" page="59">

<Paragraph number="2058" type="verses" book="ja1">
_“Na vāhametaṃ saddahāmi,_  
_cakkavākassa bhojanaṃ;_  
_Ahampi samma bhuñjāmi,_  
_gāme loṇiyateliyaṃ._  

</Paragraph>

</Division>
<Division number="139" book="ja1" bookNo="22" page="59">

<Paragraph number="2059" type="verses" book="ja1">
_Manussesu kataṃ bhattaṃ,_  
_suciṃ maṃsūpasecanaṃ;_  
_Na ca me tādiso vaṇṇo,_  
_cakkavāka yathā tuvaṃ”._  

</Paragraph>

</Division>
<Division number="140" book="ja1" bookNo="22" page="59">

<Paragraph number="2060" type="verses" book="ja1">
_“Sampassaṃ attani veraṃ,_  
_hiṃsayaṃ mānusiṃ pajaṃ;_  
_Utrasto ghasasī bhīto,_  
_tena vaṇṇo tavediso._  

</Paragraph>

</Division>
<Division number="141" book="ja1" bookNo="22" page="60">

<Paragraph number="2061" type="verses" book="ja1">
_Sabbalokaviruddhosi,_  
_dhaṅka pāpena kammunā;_  
_Laddho piṇḍo na pīṇeti,_  
_tena vaṇṇo tavediso._  

</Paragraph>

</Division>
<Division number="142" book="ja1" bookNo="22" page="60">

<Paragraph number="2062" type="verses" book="ja1">
_Ahampi samma bhuñjāmi,_  
_ahiṃsaṃ sabbapāṇinaṃ;_  
_Appossukko nirāsaṅkī,_  
_asoko akutobhayo._  

</Paragraph>

</Division>
<Division number="143" book="ja1" bookNo="22" page="60">

<Paragraph number="2063" type="verses" book="ja1">
_So karassu ānubhāvaṃ,_  
_vītivattassu sīliyaṃ;_  
_Ahiṃsāya cara loke,_  
_piyo hohisi maṃmiva._  

</Paragraph>

</Division>
<Division number="144" book="ja1" bookNo="22" page="60">

<Paragraph number="2064" type="verses" book="ja1">
_Yo na hanti na ghāteti,_  
_na jināti na jāpaye;_  
_Mettaṃso sabbabhūtesu,_  
_veraṃ tassa na kenacī”ti._  


</Paragraph>
</Division>

<Paragraph number="2065" type="center" book="ja1">
Cakkavākajātakaṃ terasamaṃ.
</Paragraph>
