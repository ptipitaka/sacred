<Paragraph number="1007" type="verses" book="ja1">
_“Samaṇaṃ taṃ maññamāno,_  
_upagacchimasaññataṃ;_  
_So maṃ daṇḍena pāhāsi,_  
_yathā assamaṇo tathā”._  

</Paragraph>

<Division number="98" book="ja1" bookNo="22" page="252">

<Paragraph number="1008" type="verses" book="ja1">
_“Kiṃ te jaṭāhi dummedha,_  
_kiṃ te ajinasāṭiyā;_  
_Abbhantaraṃ te gahanaṃ,_  
_bāhiraṃ parimajjasi”._  

</Paragraph>

</Division>
<Division number="99" book="ja1" bookNo="22" page="252">

<Paragraph number="1009" type="verses" book="ja1">
_“Ehi godha nivattassu,_  
_bhuñja sālīnamodanaṃ;_  
_Telaṃ loṇañca me atthi,_  
_pahūtaṃ mayha pipphali._  

</Paragraph>

</Division>
<Division number="100" book="ja1" bookNo="22" page="252">

<Paragraph number="1010" type="verses" book="ja1">
_Esa bhiyyo pavekkhāmi,_  
_vammikaṃ sataporisaṃ;_  
_Telaṃ loṇañca kittesi,_  
_ahitaṃ mayha pipphalī”ti._  


</Paragraph>
</Division>
