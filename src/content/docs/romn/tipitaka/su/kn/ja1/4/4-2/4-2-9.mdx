<Paragraph number="975" type="verses" book="ja1">
_“Susukhaṃ vata jīvāmi,_  
_labhāmi ceva bhuñjituṃ;_  
_Paripanthe ca tiṭṭhāmi,_  
_kā nu bhante gatī mama”._  

</Paragraph>

<Division number="74" book="ja1" bookNo="22" page="235">

<Paragraph number="976" type="verses" book="ja1">
_“Mano ce te nappaṇamati,_  
_pakkhi pāpassa kammuno;_  
_Abyāvaṭassa bhadrassa,_  
_na pāpamupalimpati”._  

</Paragraph>

</Division>
<Division number="75" book="ja1" bookNo="22" page="235">

<Paragraph number="977" type="verses" book="ja1">
_“Ñātako no nisinnoti,_  
_bahu āgacchate jano;_  
_Paṭicca kammaṃ phusati,_  
_tasmiṃ me saṅkate mano”._  

</Paragraph>

</Division>
<Division number="76" book="ja1" bookNo="22" page="236">

<Paragraph number="978" type="verses" book="ja1">
_“Na paṭicca kammaṃ phusati,_  
_mano ce nappadussati;_  
_Appossukkassa bhadrassa,_  
_na pāpamupalimpatī”ti._  


</Paragraph>
</Division>

<Paragraph number="979" type="center" book="ja1">
Tittirajātakaṃ navamaṃ.
</Paragraph>
