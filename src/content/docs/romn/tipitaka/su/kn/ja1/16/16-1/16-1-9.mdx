<Division number="297" book="ja1">

<Paragraph number="3560" type="verses" book="ja1">
_“Kā vedhamānā girikandarāyaṃ,_  
_Ekā tuvaṃ tiṭṭhasi saṃhitūru;_  
_Puṭṭhāsi me pāṇipameyyamajjhe,_  
_Akkhāhi me nāmañca bandhave ca._  

</Paragraph>

</Division>
<Division number="298" book="ja1">

<Paragraph number="3561" type="verses" book="ja1">
_Obhāsayaṃ vanaṃ rammaṃ,_  
_sīhabyagghanisevitaṃ;_  
_Kā vā tvamasi kalyāṇi,_  
_kassa vā tvaṃ sumajjhime;_  
_Abhivādemi taṃ bhadde,_  
_dānavāhaṃ namatthu te”._  

</Paragraph>

</Division>
<Division number="299" book="ja1">

<Paragraph number="3562" type="verses" book="ja1">
_“Yo putto kāsirājassa,_  
_sotthisenoti taṃ vidū;_  
_Tassāhaṃ sambulā bhariyā,_  
_evaṃ jānāhi dānava;_  
_Abhivādemi taṃ bhante,_  
_sambulāhaṃ namatthu te._  

</Paragraph>

</Division>
<Division number="300" book="ja1">

<Paragraph number="3563" type="verses" book="ja1">
_Vedehaputto bhaddante,_  
_vane vasati āturo;_  
_Tamahaṃ rogasammattaṃ,_  
_ekā ekaṃ upaṭṭhahaṃ._  

</Paragraph>

</Division>
<Division number="301" book="ja1">

<Paragraph number="3564" type="verses" book="ja1">
_Ahañca vanamuñchāya,_  
_Madhumaṃsaṃ migābilaṃ;_  
_Yadāharāmi taṃ bhakkho,_  
_Tassa nūnajja nādhati”._  

</Paragraph>

</Division>
<Division number="302" book="ja1">

<Paragraph number="3565" type="verses" book="ja1">
_“Kiṃ vane rājaputtena,_  
_āturena karissasi;_  
_Sambule pariciṇṇena,_  
_ahaṃ bhattā bhavāmi te”._  

</Paragraph>

</Division>
<Division number="303" book="ja1">

<Paragraph number="3566" type="verses" book="ja1">
_“Sokaṭṭāya durattāya,_  
_kiṃ rūpaṃ vijjate mama;_  
_Aññaṃ pariyesa bhaddante,_  
_abhirūpataraṃ mayā”._  

</Paragraph>

</Division>
<Division number="304" book="ja1">

<Paragraph number="3567" type="verses" book="ja1">
_“Ehimaṃ girimāruyha,_  
_bhariyā me catussatā;_  
_Tāsaṃ tvaṃ pavarā hohi,_  
_sabbakāmasamiddhinī._  

</Paragraph>

</Division>
<Division number="305" book="ja1">

<Paragraph number="3568" type="verses" book="ja1">
_Nūna tārakavaṇṇābhe,_  
_yaṃ kiñci manasicchasi;_  
_Sabbaṃ taṃ pacuraṃ mayhaṃ,_  
_ramassvajja mayā saha._  

</Paragraph>

</Division>
<Division number="306" book="ja1">

<Paragraph number="3569" type="verses" book="ja1">
_No ce tuvaṃ maheseyyaṃ,_  
_Sambule kārayissasi;_  
_Alaṃ tvaṃ pātarāsāya,_  
_Paṇhe bhakkhā bhavissasi”._  

</Paragraph>

</Division>
<Division number="307" book="ja1">

<Paragraph number="3570" type="verses" book="ja1">
_Tañca sattajaṭo luddo,_  
_kaḷāro purisādako;_  
_Vane nāthaṃ apassantiṃ,_  
_sambulaṃ aggahī bhuje._  

</Paragraph>

</Division>
<Division number="308" book="ja1">

<Paragraph number="3571" type="verses" book="ja1">
_Adhipannā pisācena,_  
_luddenāmisacakkhunā;_  
_Sā ca sattuvasaṃ pattā,_  
_patimevānusocati._  

</Paragraph>

</Division>
<Division number="309" book="ja1">

<Paragraph number="3572" type="verses" book="ja1">
_“Na me idaṃ tathā dukkhaṃ,_  
_yaṃ maṃ khādeyya rakkhaso;_  
_Yañca me ayyaputtassa,_  
_mano hessati aññathā._  

</Paragraph>

</Division>
<Division number="310" book="ja1">

<Paragraph number="3573" type="verses" book="ja1">
_Na santi devā pavasanti nūna,_  
_Na hi nūna santi idha lokapālā;_  
_Sahasā karontānamasaññatānaṃ,_  
_Na hi nūna santi paṭisedhitāro”._  

</Paragraph>

</Division>
<Division number="311" book="ja1">

<Paragraph number="3574" type="verses" book="ja1">
_“Itthīnamesā pavarā yasassinī,_  
_Santā samā aggirivuggatejā;_  
_Tañce tuvaṃ rakkhasādesi kaññaṃ,_  
_Muddhā ca hi sattadhā te phaleyya;_  
_Mā tvaṃ dahī muñca patibbatāya”._  

</Paragraph>

</Division>
<Division number="312" book="ja1">

<Paragraph number="3575" type="verses" book="ja1">
_Sā ca assamamāgacchi,_  
_pamuttā purisādakā;_  
_Nīḷaṃ paḷinaṃ sakuṇīva,_  
_gatasiṅgaṃva ālayaṃ._  

</Paragraph>

</Division>
<Division number="313" book="ja1">

<Paragraph number="3576" type="verses" book="ja1">
_Sā tattha paridevesi,_  
_rājaputtī yasassinī;_  
_Sambulā utumattakkhā,_  
_vane nāthaṃ apassantī._  

</Paragraph>

</Division>
<Division number="314" book="ja1">

<Paragraph number="3577" type="verses" book="ja1">
_Samaṇe brāhmaṇe vande,_  
_sampannacaraṇe ise;_  
_Rājaputtaṃ apassantī,_  
_“tumhaṃmhi saraṇaṃ gatā”._  

</Paragraph>

</Division>
<Division number="315" book="ja1">

<Paragraph number="3578" type="verses" book="ja1">
_Vande sīhe ca byagghe ca,_  
_ye ca aññe vane migā;_  
_Rājaputtaṃ apassantī,_  
_“tumhaṃmhi saraṇaṃ gatā”._  

</Paragraph>

</Division>
<Division number="316" book="ja1">

<Paragraph number="3579" type="verses" book="ja1">
_Tiṇā latāni osadhyo,_  
_pabbatāni vanāni ca;_  
_Rājaputtaṃ apassantī,_  
_“tumhaṃmhi saraṇaṃ gatā”._  

</Paragraph>

</Division>
<Division number="317" book="ja1">

<Paragraph number="3580" type="verses" book="ja1">
_Vande indīvarīsāmaṃ,_  
_rattiṃ nakkhattamāliniṃ;_  
_Rājaputtaṃ apassantī,_  
_“tumhaṃmhi saraṇaṃ gatā”._  

</Paragraph>

</Division>
<Division number="318" book="ja1">

<Paragraph number="3581" type="verses" book="ja1">
_Vande bhāgīrathiṃ gaṅgaṃ,_  
_savantīnaṃ paṭiggahaṃ;_  
_Rājaputtaṃ apassantī,_  
_“tumhaṃmhi saraṇaṃ gatā”._  

</Paragraph>

</Division>
<Division number="319" book="ja1">

<Paragraph number="3582" type="verses" book="ja1">
_Vande ahaṃ pabbatarājaseṭṭhaṃ,_  
_Himavantaṃ siluccayaṃ;_  
_Rājaputtaṃ apassantī,_  
_“Tumhaṃmhi saraṇaṃ gatā”._  

</Paragraph>

</Division>
<Division number="320" book="ja1">

<Paragraph number="3583" type="verses" book="ja1">
_“Atisāyaṃ vatāgacchi,_  
_rājaputti yasassini;_  
_Kena nujja samāgacchi,_  
_ko te piyataro mayā”._  

</Paragraph>

</Division>
<Division number="321" book="ja1">

<Paragraph number="3584" type="verses" book="ja1">
_“Idaṃ khohaṃ tadāvocaṃ,_  
_gahitā tena sattunā;_  
_Na me idaṃ tathā dukkhaṃ,_  
_yaṃ maṃ khādeyya rakkhaso;_  
_Yañca me ayyaputtassa,_  
_mano hessati aññathā”._  

</Paragraph>

</Division>
<Division number="322" book="ja1">

<Paragraph number="3585" type="verses" book="ja1">
_“Corīnaṃ bahubuddhīnaṃ,_  
_yāsu saccaṃ sudullabhaṃ;_  
_Thīnaṃ bhāvo durājāno,_  
_macchassevodake gataṃ”._  

</Paragraph>

</Division>
<Division number="323" book="ja1">

<Paragraph number="3586" type="verses" book="ja1">
_“Tathā maṃ saccaṃ pāletu,_  
_pālayissati ce mamaṃ;_  
_Yathāhaṃ nābhijānāmi,_  
_aññaṃ piyataraṃ tayā;_  
_Etena saccavajjena,_  
_byādhi te vūpasammatu”._  

</Paragraph>

</Division>
<Division number="324" book="ja1">

<Paragraph number="3587" type="verses" book="ja1">
_“Ye kuñjarā sattasatā uḷārā,_  
_Rakkhanti rattindivamuyyutāvudhā;_  
_Dhanuggahānañca satāni soḷasa,_  
_Kathaṃvidhe passasi bhadde sattavo”._  

</Paragraph>

</Division>
<Division number="325" book="ja1">

<Paragraph number="3588" type="verses" book="ja1">
_“Alaṅkatāyo padumuttarattacā,_  
_Virāgitā passati haṃsagaggarā;_  
_Tāsaṃ suṇitvā mitagītavāditaṃ,_  
_Na dāni me tāta tathā yathā pure._  

</Paragraph>

</Division>
<Division number="326" book="ja1">

<Paragraph number="3589" type="verses" book="ja1">
_Suvaṇṇasaṅkaccadharā suviggahā,_  
_Alaṅkatā mānusiyaccharūpamā;_  
_Senopiyā tāta aninditaṅgiyo,_  
_Khattiyakaññā paṭilobhayanti naṃ._  

</Paragraph>

</Division>
<Division number="327" book="ja1">

<Paragraph number="3590" type="verses" book="ja1">
_Sace ahaṃ tāta tathā yathā pure,_  
_Patiṃ tamuñchāya punā vane bhare;_  
_Sammānaye maṃ na ca maṃ vimānaye,_  
_Itopi me tāta tato varaṃ siyā._  

</Paragraph>

</Division>
<Division number="328" book="ja1">

<Paragraph number="3591" type="verses" book="ja1">
_Yamannapāne vipulasmi ohite,_  
_Nārī vimaṭṭhābharaṇā alaṅkatā;_  
_Sabbaṅgupetā patino ca appiyā,_  
_Abajjha tassā maraṇaṃ tato varaṃ._  

</Paragraph>

</Division>
<Division number="329" book="ja1">

<Paragraph number="3592" type="verses" book="ja1">
_Api ce daliddā kapaṇā anāḷhiyā,_  
_Kaṭādutīyā patino ca sā piyā;_  
_Sabbaṅgupetāyapi appiyāya,_  
_Ayameva seyyā kapaṇāpi yā piyā”._  

</Paragraph>

</Division>
<Division number="330" book="ja1">

<Paragraph number="3593" type="verses" book="ja1">
_“Sudullabhitthī purisassa yā hitā,_  
_Bhattitthiyā dullabho yo hito ca;_  
_Hitā ca te sīlavatī ca bhariyā,_  
_Janinda dhammaṃ cara sambulāya”._  

</Paragraph>

</Division>
<Division number="331" book="ja1">

<Paragraph number="3594" type="verses" book="ja1">
_“Sace tuvaṃ vipule laddhabhoge,_  
_Issāvatiṇṇā maraṇaṃ upesi;_  
_Ahañca te bhadde imā rājakaññā,_  
_Sabbe te vacanakarā bhavāmā”ti._  


</Paragraph>
</Division>

<Paragraph number="3595" type="center" book="ja1">
Sambulājātakaṃ navamaṃ.
</Paragraph>
