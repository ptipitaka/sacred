<Division number="54" book="ja1">

<Paragraph number="1545" type="verses" book="ja1">
_“Yeme ahaṃ na jānāmi,_  
_aṭṭhisena vanibbake;_  
_Te maṃ saṅgamma yācanti,_  
_kasmā maṃ tvaṃ na yācasi”._  

</Paragraph>

</Division>
<Division number="55" book="ja1">

<Paragraph number="1546" type="verses" book="ja1">
_“Yācako appiyo hoti,_  
_yācaṃ adadamappiyo;_  
_Tasmāhaṃ taṃ na yācāmi,_  
_mā me videssanā ahu”._  

</Paragraph>

</Division>
<Division number="56" book="ja1">

<Paragraph number="1547" type="verses" book="ja1">
_“Yo ve yācanajīvāno,_  
_kāle yācaṃ na yācati;_  
_Parañca puññā dhaṃseti,_  
_attanāpi na jīvati._  

</Paragraph>

</Division>
<Division number="57" book="ja1">

<Paragraph number="1548" type="verses" book="ja1">
_Yo ca yācanajīvāno,_  
_kāle yācañhi yācati;_  
_Parañca puññaṃ labbheti,_  
_attanāpi ca jīvati._  

</Paragraph>

</Division>
<Division number="58" book="ja1">

<Paragraph number="1549" type="verses" book="ja1">
_Na ve dessanti sappaññā,_  
_Disvā yācakamāgate;_  
_Brahmacāri piyo mesi,_  
_Vada tvaṃ bhaññamicchasi”._  

</Paragraph>

</Division>
<Division number="59" book="ja1">

<Paragraph number="1550" type="verses" book="ja1">
_“Na ve yācanti sappaññā,_  
_Dhīro ca veditumarahati;_  
_Uddissa ariyā tiṭṭhanti,_  
_Esā ariyāna yācanā”._  

</Paragraph>

</Division>
<Division number="60" book="ja1">

<Paragraph number="1551" type="verses" book="ja1">
_“Dadāmi te brāhmaṇa rohiṇīnaṃ,_  
_Gavaṃ sahassaṃ saha puṅgavena;_  
_Ariyo hi ariyassa kathaṃ na dajjā,_  
_Sutvāna gāthā tava dhammayuttā”ti._  


</Paragraph>
</Division>

<Paragraph number="1552" type="center" book="ja1">
Aṭṭhisenakajātakaṃ aṭṭhamaṃ.
</Paragraph>
