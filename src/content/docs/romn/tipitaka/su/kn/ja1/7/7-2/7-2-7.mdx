<Paragraph number="1623" type="verses" book="ja1">
_“Ahaṃ dasasataṃbyāmaṃ,_  
_uragamādāya āgato;_  
_Tañca mañca mahākāyaṃ,_  
_dhārayaṃ nappavedhasi._  

</Paragraph>

</Division>
<Division number="122" book="ja1" bookNo="22" page="374">

<Paragraph number="1624" type="verses" book="ja1">
_Athimaṃ khuddakaṃ pakkhiṃ,_  
_Appamaṃsataraṃ mayā;_  
_Dhārayaṃ byathasi bhītā,_  
_Kamatthaṃ koṭasimbali”._  

</Paragraph>

</Division>
<Division number="123" book="ja1" bookNo="22" page="374">

<Paragraph number="1625" type="verses" book="ja1">
_“Maṃsabhakkho tuvaṃ rāja,_  
_phalabhakkho ayaṃ dijo;_  
_Ayaṃ nigrodhabījāni,_  
_pilakkhudumbarāni ca;_  
_Assatthāni ca bhakkhitvā,_  
_khandhe me ohadissati._  

</Paragraph>

</Division>
<Division number="124" book="ja1" bookNo="22" page="374">

<Paragraph number="1626" type="verses" book="ja1">
_Te rukkhā saṃvirūhanti,_  
_mama passe nivātajā;_  
_Te maṃ pariyonandhissanti,_  
_arukkhaṃ maṃ karissare._  

</Paragraph>

</Division>
<Division number="125" book="ja1" bookNo="22" page="374">

<Paragraph number="1627" type="verses" book="ja1">
_Santi aññepi rukkhā se,_  
_mūlino khandhino dumā;_  
_Iminā sakuṇajātena,_  
_bījamāharitā hatā._  

</Paragraph>

</Division>
<Division number="126" book="ja1" bookNo="22" page="375">

<Paragraph number="1628" type="verses" book="ja1">
_Ajjhārūhābhivaḍḍhanti,_  
_brahantampi vanappatiṃ;_  
_Tasmā rāja pavedhāmi,_  
_sampassaṃnāgataṃ bhayaṃ”._  

</Paragraph>

</Division>
<Division number="127" book="ja1" bookNo="22" page="375">

<Paragraph number="1629" type="verses" book="ja1">
_“Saṅkeyya saṅkitabbāni,_  
_rakkheyyānāgataṃ bhayaṃ;_  
_Anāgatabhayā dhīro,_  
_ubho loke avekkhatī”ti._  


</Paragraph>
</Division>

<Paragraph number="1630" type="center" book="ja1">
Koṭasimbalijātakaṃ sattamaṃ.
</Paragraph>
