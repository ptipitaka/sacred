<Paragraph number="3438" type="verses" book="ja1">
_Bārāṇasyaṃ ahū rājā,_  
_kāsīnaṃ raṭṭhavaḍḍhano;_  
_Mittāmaccaparibyūḷho,_  
_agamāsi migājinaṃ._  

</Paragraph>

</Division>
<Division number="179" book="ja1" bookNo="22" page="298">

<Paragraph number="3439" type="verses" book="ja1">
_Tattha brāhmaṇamaddakkhi,_  
_setaṃ citraṃ kilāsinaṃ;_  
_Viddhastaṃ koviḷāraṃva,_  
_kisaṃ dhamanisanthataṃ._  

</Paragraph>

</Division>
<Division number="180" book="ja1" bookNo="22" page="298">

<Paragraph number="3440" type="verses" book="ja1">
_Paramakāruññataṃ pattaṃ,_  
_disvā kicchagataṃ naraṃ;_  
_Avaca byamhito rājā,_  
_“yakkhānaṃ katamo nusi._  

</Paragraph>

</Division>
<Division number="181" book="ja1" bookNo="22" page="298">

<Paragraph number="3441" type="verses" book="ja1">
_Hatthapādā ca te setā,_  
_tato setataraṃ siro;_  
_Gattaṃ kammāsavaṇṇaṃ te,_  
_kilāsabahulo casi._  

</Paragraph>

</Division>
<Division number="182" book="ja1" bookNo="22" page="298">

<Paragraph number="3442" type="verses" book="ja1">
_Vaṭṭanāvaḷi saṅkāsā,_  
_piṭṭhi te ninnatunnatā;_  
_Kāḷapabbāva te aṅgā,_  
_nāññaṃ passāmi edisaṃ._  

</Paragraph>

</Division>
<Division number="183" book="ja1" bookNo="22" page="298">

<Paragraph number="3443" type="verses" book="ja1">
_Ugghaṭṭapādo tasito,_  
_kiso dhamanisanthato;_  
_Chāto ātattarūposi,_  
_kutosi kattha gacchasi._  

</Paragraph>

</Division>
<Division number="184" book="ja1" bookNo="22" page="298">

<Paragraph number="3444" type="verses" book="ja1">
_Duddasī appakārosi,_  
_dubbaṇṇo bhīmadassano;_  
_Janetti yāpi te mātā,_  
_na taṃ iccheyya passituṃ._  

</Paragraph>

</Division>
<Division number="185" book="ja1" bookNo="22" page="298">

<Paragraph number="3445" type="verses" book="ja1">
_Kiṃ kammamakaraṃ pubbe,_  
_Kaṃ avajjhaṃ aghātayi;_  
_Kibbisaṃ yaṃ karitvāna,_  
_Idaṃ dukkhaṃ upāgami”._  

</Paragraph>

</Division>
<Division number="186" book="ja1" bookNo="22" page="298">

<Paragraph number="3446" type="verses" book="ja1">
_“Taggha te ahamakkhissaṃ,_  
_yathāpi kusalo tathā;_  
_Saccavādiñhi lokasmiṃ,_  
_pasaṃsantīdha paṇḍitā._  

</Paragraph>

</Division>
<Division number="187" book="ja1" bookNo="22" page="299">

<Paragraph number="3447" type="verses" book="ja1">
_Eko caraṃ gogaveso,_  
_mūḷho accasariṃ vane;_  
_Araññe irīṇe vivane,_  
_nānākuñjarasevite._  

</Paragraph>

</Division>
<Division number="188" book="ja1" bookNo="22" page="299">

<Paragraph number="3448" type="verses" book="ja1">
_Vāḷamigānucarite,_  
_vippanaṭṭhosmi kānane;_  
_Acariṃ tattha sattāhaṃ,_  
_khuppipāsasamappito._  

</Paragraph>

</Division>
<Division number="189" book="ja1" bookNo="22" page="299">

<Paragraph number="3449" type="verses" book="ja1">
_Tattha tindukamaddakkhiṃ,_  
_visamaṭṭhaṃ bubhukkhito;_  
_Papātamabhilambantaṃ,_  
_sampannaphaladhārinaṃ._  

</Paragraph>

</Division>
<Division number="190" book="ja1" bookNo="22" page="299">

<Paragraph number="3450" type="verses" book="ja1">
_Vātassitāni bhakkhesiṃ,_  
_tāni rucciṃsu me bhusaṃ;_  
_Atitto rukkhamārūhiṃ,_  
_tattha hessāmi āsito._  

</Paragraph>

</Division>
<Division number="191" book="ja1" bookNo="22" page="299">

<Paragraph number="3451" type="verses" book="ja1">
_Ekaṃ me bhakkhitaṃ āsi,_  
_dutiyaṃ abhipatthitaṃ;_  
_Tato sā bhañjatha sākhā,_  
_chinnā pharasunā viya._  

</Paragraph>

</Division>
<Division number="192" book="ja1" bookNo="22" page="299">

<Paragraph number="3452" type="verses" book="ja1">
_Sohaṃ sahāva sākhāhi,_  
_uddhampādo avaṃsiro;_  
_Appatiṭṭhe anālambe,_  
_giriduggasmi pāpataṃ._  

</Paragraph>

</Division>
<Division number="193" book="ja1" bookNo="22" page="299">

<Paragraph number="3453" type="verses" book="ja1">
_Yasmā ca vāri gambhīraṃ,_  
_tasmā na samapajjisaṃ;_  
_Tattha sesiṃ nirānando,_  
_anūnā dasa rattiyo._  

</Paragraph>

</Division>
<Division number="194" book="ja1" bookNo="22" page="299">

<Paragraph number="3454" type="verses" book="ja1">
_Athettha kapi māgañchi,_  
_gonaṅgulo darīcaro;_  
_Sākhāhi sākhaṃ vicaranto,_  
_khādamāno dumapphalaṃ._  

</Paragraph>

</Division>
<Division number="195" book="ja1" bookNo="22" page="299">

<Paragraph number="3455" type="verses" book="ja1">
_So maṃ disvā kisaṃ paṇḍuṃ,_  
_kāruññamakaraṃ mayi;_  
_Ambho ko nāma so ettha,_  
_evaṃ dukkhena aṭṭito._  

</Paragraph>

</Division>
<Division number="196" book="ja1" bookNo="22" page="299">

<Paragraph number="3456" type="verses" book="ja1">
_Manusso amanusso vā,_  
_attānaṃ me pavedaya;_  
_Tassañjaliṃ paṇāmetvā,_  
_idaṃ vacanamabraviṃ._  

</Paragraph>

</Division>
<Division number="197" book="ja1" bookNo="22" page="299">

<Paragraph number="3457" type="verses" book="ja1">
_Manussohaṃ byasampatto,_  
_sā me natthi ito gati;_  
_Taṃ vo vadāmi bhaddaṃ vo,_  
_tvañca me saraṇaṃ bhava._  

</Paragraph>

</Division>
<Division number="198" book="ja1" bookNo="22" page="300">

<Paragraph number="3458" type="verses" book="ja1">
_Garuṃ silaṃ gahetvāna,_  
_vicarī pabbate kapi;_  
_Silāya yoggaṃ katvāna,_  
_nisabho etadabravi._  

</Paragraph>

</Division>
<Division number="199" book="ja1" bookNo="22" page="300">

<Paragraph number="3459" type="verses" book="ja1">
_Ehi me piṭṭhimāruyha,_  
_gīvaṃ gaṇhāhi bāhubhi;_  
_Ahaṃ taṃ uddharissāmi,_  
_giriduggata vegasā._  

</Paragraph>

</Division>
<Division number="200" book="ja1" bookNo="22" page="300">

<Paragraph number="3460" type="verses" book="ja1">
_Tassa taṃ vacanaṃ sutvā,_  
_vānarindassa sirīmato;_  
_Piṭṭhimāruyha dhīrassa,_  
_gīvaṃ bāhāhi aggahiṃ._  

</Paragraph>

</Division>
<Division number="201" book="ja1" bookNo="22" page="332">

<Paragraph number="3461" type="verses" book="ja1">
_So maṃ tato samuṭṭhāsi,_  
_tejassī balavā kapi;_  
_Vihaññamāno kicchena,_  
_giriduggata vegasā._  

</Paragraph>

</Division>
<Division number="202" book="ja1" bookNo="22" page="382">

<Paragraph number="3462" type="verses" book="ja1">
_Uddharitvāna maṃ santo,_  
_nisabho etadabravi;_  
_Iṅgha maṃ samma rakkhassu,_  
_pasupissaṃ muhuttakaṃ._  

</Paragraph>

</Division>
<Division number="203" book="ja1" bookNo="22" page="382">

<Paragraph number="3463" type="verses" book="ja1">
_Sīhā byagghā ca dīpī ca,_  
_acchakokataracchayo;_  
_Te maṃ pamattaṃ hiṃseyyuṃ,_  
_te tvaṃ disvā nivāraya._  

</Paragraph>

</Division>
<Division number="204" book="ja1" bookNo="22" page="382">

<Paragraph number="3464" type="verses" book="ja1">
_Evaṃ me parittātūna,_  
_pasupī so muhuttakaṃ;_  
_Tadāhaṃ pāpikaṃ diṭṭhiṃ,_  
_paṭilacchiṃ ayoniso._  

</Paragraph>

</Division>
<Division number="205" book="ja1" bookNo="22" page="382">

<Paragraph number="3465" type="verses" book="ja1">
_Bhakkho ayaṃ manussānaṃ,_  
_yathā caññe vane migā;_  
_Yaṃ nūnimaṃ vadhitvāna,_  
_chāto khādeyya vānaraṃ._  

</Paragraph>

</Division>
<Division number="206" book="ja1" bookNo="22" page="382">

<Paragraph number="3466" type="verses" book="ja1">
_Asito ca gamissāmi,_  
_maṃsamādāya sambalaṃ;_  
_Kantāraṃ nittharissāmi,_  
_pātheyyaṃ me bhavissati._  

</Paragraph>

</Division>
<Division number="207" book="ja1" bookNo="22" page="382">

<Paragraph number="3467" type="verses" book="ja1">
_Tato silaṃ gahetvāna,_  
_matthakaṃ sannitāḷayiṃ;_  
_Mama gattakilantassa,_  
_pahāro dubbalo ahu._  

</Paragraph>

</Division>
<Division number="208" book="ja1" bookNo="22" page="383">

<Paragraph number="3468" type="verses" book="ja1">
_So ca vegenudappatto,_  
_kapi ruhira makkhito;_  
_Assupuṇṇehi nettehi,_  
_rodanto maṃ udikkhati._  

</Paragraph>

</Division>
<Division number="209" book="ja1" bookNo="22" page="383">

<Paragraph number="3469" type="verses" book="ja1">
_‘Māyyomaṃ kari bhaddante,_  
_tvañca nāmedisaṃ kari;_  
_Tvañca kho nāma dīghāvu,_  
_aññe vāretumarahasi._  

</Paragraph>

</Division>
<Division number="210" book="ja1" bookNo="22" page="383">

<Paragraph number="3470" type="verses" book="ja1">
_Aho vata re purisa,_  
_tāvadukkarakāraka;_  
_Edisā visamā duggā,_  
_papātā uddhato mayā._  

</Paragraph>

</Division>
<Division number="211" book="ja1" bookNo="22" page="383">

<Paragraph number="3471" type="verses" book="ja1">
_Ānīto paralokāva,_  
_dubbheyyaṃ maṃ amaññatha;_  
_Taṃ tena pāpadhammena,_  
_pāpaṃ pāpena cintitaṃ._  

</Paragraph>

</Division>
<Division number="212" book="ja1" bookNo="22" page="383">

<Paragraph number="3472" type="verses" book="ja1">
_Mā heva tvaṃ adhammaṭṭha,_  
_vedanaṃ kaṭukaṃ phusi;_  
_Mā heva pāpakammaṃ taṃ,_  
_phalaṃ veḷuṃva taṃ vadhi._  

</Paragraph>

</Division>
<Division number="213" book="ja1" bookNo="22" page="383">

<Paragraph number="3473" type="verses" book="ja1">
_Tayime natthi vissāso,_  
_pāpadhamma asaññata;_  
_Ehi me piṭṭhito gaccha,_  
_dissamānova santike._  

</Paragraph>

</Division>
<Division number="214" book="ja1" bookNo="22" page="383">

<Paragraph number="3474" type="verses" book="ja1">
_Muttosi hatthā vāḷānaṃ,_  
_pattosi mānusiṃ padaṃ;_  
_Esa maggo adhammaṭṭha,_  
_tena gaccha yathāsukhaṃ’._  

</Paragraph>

</Division>
<Division number="215" book="ja1" bookNo="22" page="383">

<Paragraph number="3475" type="verses" book="ja1">
_Idaṃ vatvā giricaro,_  
_rahade pakkhalya matthakaṃ;_  
_Assūni sampamajjitvā,_  
_tato pabbatamāruhi._  

</Paragraph>

</Division>
<Division number="216" book="ja1" bookNo="22" page="383">

<Paragraph number="3476" type="verses" book="ja1">
_Sohaṃ tenābhisattosmi,_  
_pariḷāhena aṭṭito;_  
_Ḍayhamānena gattena,_  
_vāriṃ pātuṃ upāgamiṃ._  

</Paragraph>

</Division>
<Division number="217" book="ja1" bookNo="22" page="383">

<Paragraph number="3477" type="verses" book="ja1">
_Agginā viya santatto,_  
_rahado ruhiramakkhito;_  
_Pubbalohitasaṅkāso,_  
_sabbo me samapajjatha._  

</Paragraph>

</Division>
<Division number="218" book="ja1" bookNo="22" page="383">

<Paragraph number="3478" type="verses" book="ja1">
_Yāvanto udabindūni,_  
_kāyasmiṃ nipatiṃsu me;_  
_Tāvanto gaṇḍa jāyetha,_  
_addhabeluvasādisā._  

</Paragraph>

</Division>
<Division number="219" book="ja1" bookNo="22" page="383">

<Paragraph number="3479" type="verses" book="ja1">
_Pabhinnā pagghariṃsu me,_  
_kuṇapā pubbalohitā;_  
_Yena yeneva gacchāmi,_  
_gāmesu nigamesu ca._  

</Paragraph>

</Division>
<Division number="220" book="ja1" bookNo="22" page="384">

<Paragraph number="3480" type="verses" book="ja1">
_Daṇḍahatthā nivārenti,_  
_itthiyo purisā ca maṃ;_  
_Okkitā pūtigandhena,_  
_māssu orena āgamā._  

</Paragraph>

</Division>
<Division number="221" book="ja1" bookNo="22" page="384">

<Paragraph number="3481" type="verses" book="ja1">
_Etādisaṃ idaṃ dukkhaṃ,_  
_satta vassāni dāni me;_  
_Anubhomi sakaṃ kammaṃ,_  
_pubbe dukkaṭamattano._  

</Paragraph>

</Division>
<Division number="222" book="ja1" bookNo="22" page="384">

<Paragraph number="3482" type="verses" book="ja1">
_Taṃ vo vadāmi bhaddante,_  
_yāvantettha samāgatā;_  
_Māssu mittāna dubbhittho,_  
_mittadubbho hi pāpako._  

</Paragraph>

</Division>
<Division number="223" book="ja1" bookNo="22" page="384">

<Paragraph number="3483" type="verses" book="ja1">
_Kuṭṭhī kilāsī bhavati,_  
_yo mittānidha dubbhati;_  
_Kāyassa bhedā mittaddu,_  
_nirayaṃ sopapajjatī”ti._  


</Paragraph>
</Division>

<Paragraph number="3484" type="center" book="ja1">
Mahākapijātakaṃ chaṭṭhaṃ.
