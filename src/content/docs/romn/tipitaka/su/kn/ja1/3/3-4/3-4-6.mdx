<Paragraph number="816" type="verses" book="ja1">
_“Mā sālūkassa pihayi,_  
_āturannāni bhuñjati;_  
_Appossukko bhusaṃ khāda,_  
_etaṃ dīghāyulakkhaṇaṃ._  

</Paragraph>

<Division number="107" book="ja1">

<Paragraph number="817" type="verses" book="ja1">
_Idāni so idhāgantvā,_  
_atithī yuttasevako;_  
_Atha dakkhasi sālūkaṃ,_  
_sayantaṃ musaluttaraṃ”._  

</Paragraph>

</Division>
<Division number="108" book="ja1">

<Paragraph number="818" type="verses" book="ja1">
_Vikantaṃ sūkaraṃ disvā,_  
_sayantaṃ musaluttaraṃ;_  
_Jaraggavā vicintesuṃ,_  
_varamhākaṃ bhusāmivāti._  


</Paragraph>
</Division>
