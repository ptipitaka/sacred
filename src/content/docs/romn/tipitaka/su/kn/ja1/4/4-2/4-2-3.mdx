<Paragraph number="945" type="verses" book="ja1">
_“Yo te hatthe ca pāde ca,_  
_kaṇṇanāsañca chedayi;_  
_Tassa kujjha mahāvīra,_  
_mā raṭṭhaṃ vinasā idaṃ”._  

</Paragraph>

</Division>
<Division number="50" book="ja1">

<Paragraph number="946" type="verses" book="ja1">
_“Yo me hatthe ca pāde ca,_  
_kaṇṇanāsañca chedayi;_  
_Ciraṃ jīvatu so rājā,_  
_na hi kujjhanti mādisā”._  

</Paragraph>

</Division>
<Division number="51" book="ja1">

<Paragraph number="947" type="verses" book="ja1">
_Ahū atītamaddhānaṃ,_  
_samaṇo khantidīpano;_  
_Taṃ khantiyāyeva ṭhitaṃ,_  
_kāsirājā achedayi._  

</Paragraph>

</Division>
<Division number="52" book="ja1">

<Paragraph number="948" type="verses" book="ja1">
_Tassa kammapharusassa,_  
_vipāko kaṭuko ahu;_  
_Yaṃ kāsirājā vedesi,_  
_nirayamhi samappitoti._  


</Paragraph>
</Division>

<Paragraph number="949" type="center" book="ja1">
Khantīvādījātakaṃ tatiyaṃ.
</Paragraph>
