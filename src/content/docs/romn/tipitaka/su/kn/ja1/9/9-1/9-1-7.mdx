<Paragraph number="1846" type="verses" book="ja1">
_“Assa indasamo rāja,_  
_accantaṃ ajarāmaro;_  
_Sace tvaṃ yaññaṃ yājeyya,_  
_isiṃ lomasakassapaṃ”._  

</Paragraph>

</Division>
<Division number="61" book="ja1" bookNo="22" page="365">

<Paragraph number="1847" type="verses" book="ja1">
_“Sasamuddapariyāyaṃ,_  
_mahiṃ sāgarakuṇḍalaṃ;_  
_Na icche saha nindāya,_  
_evaṃ seyya vijānahi._  

</Paragraph>

</Division>
<Division number="62" book="ja1" bookNo="22" page="365">

<Paragraph number="1848" type="verses" book="ja1">
_Dhiratthu taṃ yasalābhaṃ,_  
_dhanalābhañca brāhmaṇa;_  
_Yā vutti vinipātena,_  
_adhammacaraṇena vā._  

</Paragraph>

</Division>
<Division number="63" book="ja1" bookNo="22" page="365">

<Paragraph number="1849" type="verses" book="ja1">
_Api ce pattamādāya,_  
_anagāro paribbaje;_  
_Sāyeva jīvikā seyyo,_  
_yā cādhammena esanā._  

</Paragraph>

</Division>
<Division number="64" book="ja1" bookNo="22" page="365">

<Paragraph number="1850" type="verses" book="ja1">
_Api ce pattamādāya,_  
_anagāro paribbaje;_  
_Aññaṃ ahiṃsayaṃ loke,_  
_api rajjena taṃ varaṃ”._  

</Paragraph>

</Division>
<Division number="65" book="ja1" bookNo="22" page="365">

<Paragraph number="1851" type="verses" book="ja1">
_“Balaṃ cando balaṃ suriyo,_  
_balaṃ samaṇabrāhmaṇā;_  
_Balaṃ velā samuddassa,_  
_balātibalamitthiyo._  

</Paragraph>

</Division>
<Division number="66" book="ja1" bookNo="22" page="366">

<Paragraph number="1852" type="verses" book="ja1">
_Yathā uggatapaṃ santaṃ,_  
_isiṃ lomasakassapaṃ;_  
_Pitu atthā candavatī,_  
_vājapeyyaṃ ayājayi”._  

</Paragraph>

</Division>
<Division number="67" book="ja1" bookNo="22" page="366">

<Paragraph number="1853" type="verses" book="ja1">
_“Taṃ lobhapakataṃ kammaṃ,_  
_Kaṭukaṃ kāmahetukaṃ;_  
_Tassa mūlaṃ gavesissaṃ,_  
_Checchaṃ rāgaṃ sabandhanaṃ._  

</Paragraph>

</Division>
<Division number="68" book="ja1" bookNo="22" page="366">

<Paragraph number="1854" type="verses" book="ja1">
_Dhiratthu kāme subahūpi loke,_  
_Tapova seyyo kāmaguṇehi rāja;_  
_Tapo karissāmi pahāya kāme,_  
_Taveva raṭṭhaṃ candavatī ca hotū”ti._  


</Paragraph>
</Division>

<Paragraph number="1855" type="center" book="ja1">
Lomasakassapajātakaṃ sattamaṃ.
</Paragraph>
