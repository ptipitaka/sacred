<Division number="54" book="ja1">

<Paragraph number="2441" type="verses" book="ja1">
_“Dūte te brahme pāhesiṃ,_  
_Gaṅgātīrasmi jhāyato;_  
_Tesaṃ puṭṭho na byākāsi,_  
_Dukkhaṃ guyhamataṃ nu te”._  

</Paragraph>

</Division>
<Division number="55" book="ja1">

<Paragraph number="2442" type="verses" book="ja1">
_“Sace te dukkhamuppajje,_  
_kāsīnaṃ raṭṭhavaḍḍhana;_  
_Mā kho naṃ tassa akkhāhi,_  
_yo taṃ dukkhā na mocaye._  

</Paragraph>

</Division>
<Division number="56" book="ja1">

<Paragraph number="2443" type="verses" book="ja1">
_Yo tassa dukkhajātassa,_  
_ekaṅgamapi bhāgaso;_  
_Vippamoceyya dhammena,_  
_kāmaṃ tassa pavedaya._  

</Paragraph>

</Division>
<Division number="57" book="ja1">

<Paragraph number="2444" type="verses" book="ja1">
_Suvijānaṃ siṅgālānaṃ,_  
_sakuṇānañca vassitaṃ;_  
_Manussavassitaṃ rāja,_  
_dubbijānataraṃ tato._  

</Paragraph>

</Division>
<Division number="58" book="ja1">

<Paragraph number="2445" type="verses" book="ja1">
_Api ce maññatī poso,_  
_ñāti mitto sakhāti vā;_  
_Yo pubbe sumano hutvā,_  
_pacchā sampajjate diso._  

</Paragraph>

</Division>
<Division number="59" book="ja1">

<Paragraph number="2446" type="verses" book="ja1">
_Yo attano dukkhamanānupuṭṭho,_  
_Pavedaye jantu akālarūpe;_  
_Ānandino tassa bhavantimittā,_  
_Hitesino tassa dukhī bhavanti._  

</Paragraph>

</Division>
<Division number="60" book="ja1">

<Paragraph number="2447" type="verses" book="ja1">
_Kālañca ñatvāna tathāvidhassa,_  
_Medhāvīnaṃ ekamanaṃ viditvā;_  
_Akkheyya tibbāni parassa dhīro,_  
_Saṇhaṃ giraṃ atthavatiṃ pamuñce._  

</Paragraph>

</Division>
<Division number="61" book="ja1">

<Paragraph number="2448" type="verses" book="ja1">
_Sace ca jaññā avisayhamattano,_  
_Na te hi mayhaṃ sukhāgamāya;_  
_Ekova tibbāni saheyya dhīro,_  
_Saccaṃ hirottappamapekkhamāno._  

</Paragraph>

</Division>
<Division number="62" book="ja1">

<Paragraph number="2449" type="verses" book="ja1">
_Ahaṃ raṭṭhe vicaranto,_  
_nigame rājadhāniyo;_  
_Bhikkhamāno mahārāja,_  
_ācariyassa dhanatthiko._  

</Paragraph>

</Division>
<Division number="63" book="ja1">

<Paragraph number="2450" type="verses" book="ja1">
_Gahapatī rājapurise,_  
_mahāsāle ca brāhmaṇe;_  
_Alatthaṃ satta nikkhāni,_  
_suvaṇṇassa janādhipa;_  
_Te me naṭṭhā mahārāja,_  
_tasmā socāmahaṃ bhusaṃ._  

</Paragraph>

</Division>
<Division number="64" book="ja1">

<Paragraph number="2451" type="verses" book="ja1">
_Purisā te mahārāja,_  
_manasānuvicintitā;_  
_Nālaṃ dukkhā pamocetuṃ,_  
_tasmā tesaṃ na byāhariṃ._  

</Paragraph>

</Division>
<Division number="65" book="ja1">

<Paragraph number="2452" type="verses" book="ja1">
_Tvañca kho me mahārāja,_  
_manasānuvicintito;_  
_Alaṃ dukkhā pamocetuṃ,_  
_tasmā tuyhaṃ pavedayiṃ”._  

</Paragraph>

</Division>
<Division number="66" book="ja1">

<Paragraph number="2453" type="verses" book="ja1">
_“Tassādāsi pasannatto,_  
_kāsīnaṃ raṭṭhavaḍḍhano;_  
_Jātarūpamaye nikkhe,_  
_suvaṇṇassa catuddasā”ti._  


</Paragraph>
</Division>

<Paragraph number="2454" type="center" book="ja1">
Dūtajātakaṃ pañcamaṃ.
</Paragraph>
