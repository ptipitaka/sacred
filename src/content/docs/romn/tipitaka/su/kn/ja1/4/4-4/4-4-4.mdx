<Division number="133" book="ja1">

<Paragraph number="1054" type="verses" book="ja1">
_“Gavañce taramānānaṃ,_  
_jimhaṃ gacchati puṅgavo;_  
_Sabbā tā jimhaṃ gacchanti,_  
_nette jimhaṃ gate sati._  

</Paragraph>

</Division>
<Division number="134" book="ja1">

<Paragraph number="1055" type="verses" book="ja1">
_Evameva manussesu,_  
_yo hoti seṭṭhasammato;_  
_So ce adhammaṃ carati,_  
_pageva itarā pajā;_  
_Sabbaṃ raṭṭhaṃ dukhaṃ seti,_  
_rājā ce hoti adhammiko._  

</Paragraph>

</Division>
<Division number="135" book="ja1">

<Paragraph number="1056" type="verses" book="ja1">
_Gavañce taramānānaṃ,_  
_ujuṃ gacchati puṅgavo;_  
_Sabbā gāvī ujuṃ yanti,_  
_nette ujuṃ gate sati._  

</Paragraph>

</Division>
<Division number="136" book="ja1">

<Paragraph number="1057" type="verses" book="ja1">
_Evameva manussesu,_  
_yo hoti seṭṭhasammato;_  
_So sace dhammaṃ carati,_  
_pageva itarā pajā;_  
_Sabbaṃ raṭṭhaṃ sukhaṃ seti,_  
_rājā ce hoti dhammiko”ti._  


</Paragraph>
</Division>

<Paragraph number="1058" type="center" book="ja1">
Rājovādajātakaṃ catutthaṃ.
</Paragraph>
