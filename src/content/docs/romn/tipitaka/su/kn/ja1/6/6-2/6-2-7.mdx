<Division number="116" book="ja1">

<Paragraph number="1451" type="verses" book="ja1">
_“Yametaṃ vārijaṃ pupphaṃ,_  
_Adinnaṃ upasiṅghasi;_  
_Ekaṅgametaṃ theyyānaṃ,_  
_Gandhathenosi mārisa”._  

</Paragraph>

</Division>
<Division number="117" book="ja1">

<Paragraph number="1452" type="verses" book="ja1">
_“Na harāmi na bhañjāmi,_  
_ārā siṅghāmi vārijaṃ;_  
_Atha kena nu vaṇṇena,_  
_gandhathenoti vuccati._  

</Paragraph>

</Division>
<Division number="118" book="ja1">

<Paragraph number="1453" type="verses" book="ja1">
_Yoyaṃ bhisāni khaṇati,_  
_puṇḍarīkāni bhañjati;_  
_Evaṃ ākiṇṇakammanto,_  
_kasmā eso na vuccati”._  

</Paragraph>

</Division>
<Division number="119" book="ja1">

<Paragraph number="1454" type="verses" book="ja1">
_“Ākiṇṇaluddo puriso,_  
_dhāticelaṃva makkhito;_  
_Tasmiṃ me vacanaṃ natthi,_  
_tañcārahāmi vattave._  

</Paragraph>

</Division>
<Division number="120" book="ja1">

<Paragraph number="1455" type="verses" book="ja1">
_Anaṅgaṇassa posassa,_  
_niccaṃ sucigavesino;_  
_Vālaggamattaṃ pāpassa,_  
_abbhāmattaṃva khāyati”._  

</Paragraph>

</Division>
<Division number="121" book="ja1">

<Paragraph number="1456" type="verses" book="ja1">
_“Addhā maṃ yakkha jānāsi,_  
_atho maṃ anukampasi;_  
_Punapi yakkha vajjāsi,_  
_yadā passasi edisaṃ”._  

</Paragraph>

</Division>
<Division number="122" book="ja1">

<Paragraph number="1457" type="verses" book="ja1">
_“Neva taṃ upajīvāmi,_  
_napi te bhatakāmhase;_  
_Tvameva bhikkhu jāneyya,_  
_yena gaccheyya suggatin”ti._  


</Paragraph>
</Division>

<Paragraph number="1458" type="center" book="ja1">
Siṅghapupphajātakaṃ sattamaṃ.
</Paragraph>
