<Paragraph number="1161" type="verses" book="ja1">
_“Nayidaṃ niccaṃ bhavitabbaṃ brahmadatta,_  
_Khemaṃ subhikkhaṃ sukhatā ca kāye;_  
_Atthaccaye mā ahu sampamūḷho,_  
_Bhinnaplavo sāgarasseva majjhe._  

</Paragraph>

</Division>
<Division number="15" book="ja1" bookNo="22" page="243">

<Paragraph number="1162" type="verses" book="ja1">
_Yāni karoti puriso,_  
_tāni attani passati;_  
_Kalyāṇakārī kalyāṇaṃ,_  
_pāpakārī ca pāpakaṃ;_  
_Yādisaṃ vapate bījaṃ,_  
_tādisaṃ harate phalaṃ”._  

</Paragraph>

</Division>
<Division number="16" book="ja1" bookNo="22" page="243">

<Paragraph number="1163" type="verses" book="ja1">
_“Idaṃ tadācariyavaco,_  
_pārāsariyo yadabravi;_  
_Mā su tvaṃ akari pāpaṃ,_  
_yaṃ tvaṃ pacchā kataṃ tape._  

</Paragraph>

</Division>
<Division number="17" book="ja1" bookNo="22" page="243">

<Paragraph number="1164" type="verses" book="ja1">
_Ayameva so piṅgiya venasākho,_  
_Yamhi ghātayiṃ khattiyānaṃ sahassaṃ;_  
_Alaṅkate candanasārānulitte,_  
_Tameva dukkhaṃ paccāgataṃ mamaṃ._  

</Paragraph>

</Division>
<Division number="18" book="ja1" bookNo="22" page="243">

<Paragraph number="1165" type="verses" book="ja1">
_Sāmā ca kho candanalittagattā,_  
_Laṭṭhīva sobhañjanakassa uggatā;_  
_Adisvā kālaṃ karissāmi ubbariṃ,_  
_Taṃ me ito dukkhataraṃ bhavissatī”ti._  


</Paragraph>
</Division>

<Paragraph number="1166" type="center" book="ja1">
Venasākhajātakaṃ tatiyaṃ.
</Paragraph>
