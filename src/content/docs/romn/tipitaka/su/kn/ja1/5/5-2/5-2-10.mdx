<Division number="105" book="ja1">

<Paragraph number="1271" type="verses" book="ja1">
_“Haṃso palāsamavaca,_  
_nigrodho samma jāyati;_  
_Aṅkasmiṃ te nisinnova,_  
_so te mammāni checchati”._  

</Paragraph>

</Division>
<Division number="106" book="ja1">

<Paragraph number="1272" type="verses" book="ja1">
_“Vaḍḍhatāmeva nigrodho,_  
_patiṭṭhassa bhavāmahaṃ;_  
_Yathā pitā ca mātā ca,_  
_evaṃ me so bhavissati”._  

</Paragraph>

</Division>
<Division number="107" book="ja1">

<Paragraph number="1273" type="verses" book="ja1">
_“Yaṃ tvaṃ aṅkasmiṃ vaḍḍhesi,_  
_khīrarukkhaṃ bhayānakaṃ;_  
_Āmanta kho taṃ gacchāma,_  
_vuḍḍhi massa na ruccati”._  

</Paragraph>

</Division>
<Division number="108" book="ja1">

<Paragraph number="1274" type="verses" book="ja1">
_“Idāni kho maṃ bhāyeti,_  
_mahānerunidassanaṃ;_  
_Haṃsassa anabhiññāya,_  
_mahā me bhayamāgataṃ”._  

</Paragraph>

</Division>
<Division number="109" book="ja1">

<Paragraph number="1275" type="verses" book="ja1">
_“Na tassa vuḍḍhi kusalappasatthā,_  
_Yo vaḍḍhamāno ghasate patiṭṭhaṃ;_  
_Tassūparodhaṃ parisaṅkamāno,_  
_Patārayī mūlavadhāya dhīro”ti._  


</Paragraph>
</Division>

<Paragraph number="1276" type="center" book="ja1">
Palāsajātakaṃ dasamaṃ.

  
Vaṇṇārohavaggo dutiyo.
</Paragraph>
