<Paragraph number="2265" type="verses" book="ja1">
_“Kasanti vapanti te janā,_  
_Manujā kammaphalūpajīvino;_  
_Nayimassa dīpakassa bhāgino,_  
_Jambudīpā idameva no varaṃ”._  

</Paragraph>

</Division>
<Division number="26" book="ja1" bookNo="22" page="96">

<Paragraph number="2266" type="verses" book="ja1">
_“Tipañcarattūpagatamhi cande,_  
_Vego mahā hehiti sāgarassa;_  
_Uplavissaṃ dīpamimaṃ uḷāraṃ,_  
_Mā vo vadhī gacchatha leṇamaññaṃ”._  

</Paragraph>

</Division>
<Division number="27" book="ja1" bookNo="22" page="96">

<Paragraph number="2267" type="verses" book="ja1">
_“Na jātuyaṃ sāgaravārivego,_  
_Uplavissaṃ dīpamimaṃ uḷāraṃ;_  
_Taṃ me nimittehi bahūhi diṭṭhaṃ,_  
_Mā bhetha kiṃ socatha modathavho._  

</Paragraph>

</Division>
<Division number="28" book="ja1" bookNo="22" page="97">

<Paragraph number="2268" type="verses" book="ja1">
_Pahūtabhakkhaṃ bahuannapānaṃ,_  
_Pattattha āvāsamimaṃ uḷāraṃ;_  
_Na vo bhayaṃ paṭipassāmi kiñci,_  
_Āputtaputtehi pamodathavho”._  

</Paragraph>

</Division>
<Division number="29" book="ja1" bookNo="22" page="97">

<Paragraph number="2269" type="verses" book="ja1">
_“Yo devoyaṃ dakkhiṇāyaṃ disāyaṃ,_  
_Khemanti pakkosati tassa saccaṃ;_  
_Na uttaro vedi bhayābhayassa,_  
_Mā bhetha kiṃ socatha modathavho”._  

</Paragraph>

</Division>
<Division number="30" book="ja1" bookNo="22" page="97">

<Paragraph number="2270" type="verses" book="ja1">
_“Yathā ime vippavadanti yakkhā,_  
_Eko bhayaṃ saṃsati khemameko;_  
_Tadiṅgha mayhaṃ vacanaṃ suṇātha,_  
_Khippaṃ lahuṃ mā vinassimha sabbe._  

</Paragraph>

</Division>
<Division number="31" book="ja1" bookNo="22" page="97">

<Paragraph number="2271" type="verses" book="ja1">
_Sabbe samāgamma karoma nāvaṃ,_  
_Doṇiṃ daḷhaṃ sabbayantūpapannaṃ;_  
_Sace ayaṃ dakkhiṇo saccamāha,_  
_Moghaṃ paṭikkosati uttaroyaṃ;_  
_Sā ceva no hehiti āpadatthā,_  
_Imañca dīpaṃ na pariccajema._  

</Paragraph>

</Division>
<Division number="32" book="ja1" bookNo="22" page="97">

<Paragraph number="2272" type="verses" book="ja1">
_Sace ca kho uttaro saccamāha,_  
_Moghaṃ paṭikkosati dakkhiṇoyaṃ;_  
_Tameva nāvaṃ abhiruyha sabbe,_  
_Evaṃ mayaṃ sotthi taremu pāraṃ._  

</Paragraph>

</Division>
<Division number="33" book="ja1" bookNo="22" page="97">

<Paragraph number="2273" type="verses" book="ja1">
_Na ve sugaṇhaṃ paṭhamena seṭṭhaṃ,_  
_Kaniṭṭhamāpāthagataṃ gahetvā;_  
_Yo cīdha tacchaṃ paviceyya gaṇhati,_  
_Sa ve naro seṭṭhamupeti ṭhānaṃ._  

</Paragraph>

</Division>
<Division number="34" book="ja1" bookNo="22" page="97">

<Paragraph number="2274" type="verses" book="ja1">
_Yathāpi te sāgaravārimajjhe,_  
_Sakammunā sotthi vahiṃsu vāṇijā;_  
_Anāgatatthaṃ paṭivijjhiyāna,_  
_Appampi nācceti sa bhūripañño._  

</Paragraph>

</Division>
<Division number="35" book="ja1" bookNo="22" page="97">

<Paragraph number="2275" type="verses" book="ja1">
_Bālā ca mohena rasānugiddhā,_  
_Anāgataṃ appaṭivijjhiyatthaṃ;_  
_Paccuppanne sīdanti atthajāte,_  
_Samuddamajjhe yathā te manussā._  

</Paragraph>

</Division>
<Division number="36" book="ja1" bookNo="22" page="98">

<Paragraph number="2276" type="verses" book="ja1">
_Anāgataṃ paṭikayirātha kiccaṃ,_  
_‘Mā maṃ kiccaṃ kiccakāle byadhesi’;_  
_Taṃ tādisaṃ paṭikatakiccakāriṃ,_  
_Na taṃ kiccaṃ kiccakāle byadhetī”ti._  


</Paragraph>
</Division>

<Paragraph number="2277" type="center" book="ja1">
Samuddavāṇijajātakaṃ tatiyaṃ.
</Paragraph>
