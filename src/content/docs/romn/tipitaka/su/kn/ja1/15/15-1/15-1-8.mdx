<Paragraph number="3045" type="verses" book="ja1">
_Bhallātiyo nāma ahosi rājā,_  
_Raṭṭhaṃ pahāya migavaṃ acāri so;_  
_Agamā girivaraṃ gandhamādanaṃ,_  
_Supupphitaṃ kimpurisānuciṇṇaṃ._  

</Paragraph>

</Division>
<Division number="187" book="ja1" bookNo="22" page="117">

<Paragraph number="3046" type="verses" book="ja1">
_Sāḷūrasaṃghañca nisedhayitvā,_  
_Dhanuṃ kalāpañca so nikkhipitvā;_  
_Upāgami vacanaṃ vattukāmo,_  
_Yatthaṭṭhitā kimpurisā ahesuṃ._  

</Paragraph>

</Division>
<Division number="188" book="ja1" bookNo="22" page="117">

<Paragraph number="3047" type="verses" book="ja1">
_“Himaccaye hemavatāya tīre,_  
_Kimidhaṭṭhitā mantayavho abhiṇhaṃ;_  
_Pucchāmi vo mānusadehavaṇṇe,_  
_Kathaṃ vo jānanti manussaloke”._  

</Paragraph>

</Division>
<Division number="189" book="ja1" bookNo="22" page="117">

<Paragraph number="3048" type="verses" book="ja1">
_“Mallaṃ giriṃ paṇḍarakaṃ tikūṭaṃ,_  
_Sītodakā anuvicarāma najjo;_  
_Migā manussāva nibhāsavaṇṇā,_  
_Jānanti no kimpurisāti ludda”._  

</Paragraph>

</Division>

_Āliṅgito cāsi piyo piyāya;_  
_Pucchāmi vo mānusadehavaṇṇe,_  
_Kimidha vane rodatha appatītā._  

</Paragraph>

_Āliṅgito cāsi piyo piyāya;_  
_Pucchāmi vo mānusadehavaṇṇe,_  
_Kimidha vane vilapatha appatītā._  

</Paragraph>

_Sukiccharūpaṃ paridevayavho,_  
_Āliṅgito cāsi piyo piyāya;_  
_Pucchāmi vo mānusadehavaṇṇe,_  
_Kimidha vane socatha appatītā”._  

</Paragraph>

</Division>
<Division number="193" book="ja1" bookNo="22" page="118">

<Paragraph number="3052" type="verses" book="ja1">
_“Mayekarattaṃ vippavasimha ludda,_  
_Akāmakā aññamaññaṃ sarantā;_  
_Tamekarattaṃ anutappamānā,_  
_Socāma ‘sā ratti punaṃ na hessati’”._  

</Paragraph>

</Division>
<Division number="194" book="ja1" bookNo="22" page="118">

<Paragraph number="3053" type="verses" book="ja1">
_“Yamekarattaṃ anutappathetaṃ,_  
_Dhanaṃ va naṭṭhaṃ pitaraṃ va petaṃ;_  
_Pucchāmi vo mānusadehavaṇṇe,_  
_Kathaṃ vinā vāsamakappayittha”._  

</Paragraph>

</Division>
<Division number="195" book="ja1" bookNo="22" page="118">

<Paragraph number="3054" type="verses" book="ja1">
_“Yamimaṃ nadiṃ passasi sīghasotaṃ,_  
_Nānādumacchādanaṃ selakūlaṃ;_  
_Taṃ me piyo uttari vassakāle,_  
_Mamañca maññaṃ anubandhatīti._  

</Paragraph>

</Division>
<Division number="196" book="ja1" bookNo="22" page="118">

<Paragraph number="3055" type="verses" book="ja1">
_Ahañca aṅkolakamocināmi,_  
_Atimuttakaṃ sattaliyothikañca;_  
_‘Piyo ca me hehiti mālabhārī,_  
_Ahañca naṃ mālinī ajjhupessaṃ’._  

</Paragraph>

</Division>
<Division number="197" book="ja1" bookNo="22" page="118">

<Paragraph number="3056" type="verses" book="ja1">
_Ahañcidaṃ kuravakamocināmi,_  
_Uddālakā pāṭalisindhuvārakā;_  
_‘Piyo ca me hehiti mālabhārī,_  
_Ahañca naṃ mālinī ajjhupessaṃ’._  

</Paragraph>

</Division>
<Division number="198" book="ja1" bookNo="22" page="118">

<Paragraph number="3057" type="verses" book="ja1">
_Ahañca sālassa supupphitassa,_  
_Oceyya pupphāni karomi mālaṃ;_  
_‘Piyo ca me hehiti mālabhārī,_  
_Ahañca naṃ mālinī ajjhupessaṃ’._  

</Paragraph>

</Division>
<Division number="199" book="ja1" bookNo="22" page="118">

<Paragraph number="3058" type="verses" book="ja1">
_Ahañca sālassa supupphitassa,_  
_Oceyya pupphāni karomi bhāraṃ;_  
_Idañca no hehiti santharatthaṃ,_  
_Yatthajjimaṃ viharissāma rattiṃ._  

</Paragraph>

</Division>
<Division number="200" book="ja1" bookNo="22" page="118">

<Paragraph number="3059" type="verses" book="ja1">
_Ahañca kho agaḷuṃ candanañca,_  
_Silāya piṃsāmi pamattarūpā;_  
_‘Piyo ca me hehiti rositaṅgo,_  
_Ahañca naṃ rositā ajjhupessaṃ’._  

</Paragraph>

</Division>
<Division number="201" book="ja1" bookNo="22" page="300">

<Paragraph number="3060" type="verses" book="ja1">
_Athāgamā salilaṃ sīghasotaṃ,_  
_Nudaṃ sāle salaḷe kaṇṇikāre;_  
_Āpūratha tena muhuttakena,_  
_Sāyaṃ nadī āsi mayā suduttarā._  

</Paragraph>

</Division>
<Division number="202" book="ja1" bookNo="22" page="332">

<Paragraph number="3061" type="verses" book="ja1">
_Ubhosu tīresu mayaṃ tadā ṭhitā,_  
_Sampassantā ubhayo aññamaññaṃ;_  
_Sakimpi rodāma sakiṃ hasāma,_  
_Kicchena no āgamā saṃvarī sā._  

</Paragraph>

</Division>
<Division number="203" book="ja1" bookNo="22" page="333">

<Paragraph number="3062" type="verses" book="ja1">
_Pātova kho uggate sūriyamhi,_  
_Catukkaṃ nadiṃ uttariyāna ludda;_  
_Āliṅgiyā aññamaññaṃ mayaṃ ubho,_  
_Sakimpi rodāma sakiṃ hasāma._  

</Paragraph>

</Division>
<Division number="204" book="ja1" bookNo="22" page="333">

<Paragraph number="3063" type="verses" book="ja1">
_Tīhūnakaṃ sattasatāni ludda,_  
_Yamidha mayaṃ vippavasimha pubbe;_  
_Vassekimaṃ jīvitaṃ bhūmipāla,_  
_Ko nīdha kantāya vinā vaseyya”._  

</Paragraph>

</Division>
<Division number="205" book="ja1" bookNo="22" page="333">

<Paragraph number="3064" type="verses" book="ja1">
_“Āyuñca vo kīvatako nu samma,_  
_Sacepi jānātha vadetha āyuṃ;_  
_Anussavā vuḍḍhato āgamā vā,_  
_Akkhātha me taṃ avikampamānā”._  

</Paragraph>

</Division>
<Division number="206" book="ja1" bookNo="22" page="333">

<Paragraph number="3065" type="verses" book="ja1">
_“Āyuñca no vassasahassaṃ ludda,_  
_Na cantarā pāpako atthi rogo;_  
_Appañca dukkhaṃ sukhameva bhiyyo,_  
_Avītarāgā vijahāma jīvitaṃ”._  

</Paragraph>

</Division>
<Division number="207" book="ja1" bookNo="22" page="333">

<Paragraph number="3066" type="verses" book="ja1">
_“Idañca sutvāna amānusānaṃ,_  
_Bhallātiyo ittara jīvitanti;_  
_Nivattatha na migavaṃ acari,_  
_Adāsi dānāni abhuñji bhoge._  

</Paragraph>

</Division>
<Division number="208" book="ja1" bookNo="22" page="333">

<Paragraph number="3067" type="verses" book="ja1">
_Idañca sutvāna amānusānaṃ,_  
_Sammodatha mā kalahaṃ akattha;_  
_Mā vo tapī attakammāparādho,_  
_Yathāpi te kimpurisekarattaṃ._  

</Paragraph>

</Division>
<Division number="209" book="ja1" bookNo="22" page="333">

<Paragraph number="3068" type="verses" book="ja1">
_Idañca sutvāna amānusānaṃ,_  
_Sammodatha mā vivādaṃ akattha;_  
_Mā vo tapī attakammāparādho,_  
_Yathāpi te kimpurisekarattaṃ”._  

</Paragraph>

</Division>

_Vacanapathaṃ tava atthasaṃhitaṃ;_  
_Muñcaṃ giraṃ nudaseva me daraṃ,_  
_Samaṇa sukhāvaha jīva me ciran”ti._
