<Paragraph number="2146" type="verses" book="ja1">
_“Ekā nisinnā suci saññatūrū,_  
_Pāsādamāruyha aninditaṅgī;_  
_Yācāmi taṃ kinnaranettacakkhu,_  
_Imekarattiṃ ubhayo vasema”._  

</Paragraph>

</Division>
<Division number="38" book="ja1" bookNo="22" page="76">

<Paragraph number="2147" type="verses" book="ja1">
_“Okiṇṇantaraparikhaṃ,_  
_daḷhamaṭṭālakoṭṭhakaṃ;_  
_Rakkhitaṃ khaggahatthehi,_  
_duppavesamidaṃ puraṃ._  

</Paragraph>

</Division>
<Division number="39" book="ja1" bookNo="22" page="76">

<Paragraph number="2148" type="verses" book="ja1">
_Daharassa yuvino cāpi,_  
_āgamo ca na vijjati;_  
_Atha kena nu vaṇṇena,_  
_saṅgamaṃ icchase mayā”._  

</Paragraph>

</Division>
<Division number="40" book="ja1" bookNo="22" page="76">

<Paragraph number="2149" type="verses" book="ja1">
_“Yakkhohamasmi kalyāṇi,_  
_Āgatosmi tavantike;_  
_Tvaṃ maṃ nandaya bhaddante,_  
_Puṇṇakaṃsaṃ dadāmi te”._  

</Paragraph>

</Division>
<Division number="41" book="ja1" bookNo="22" page="76">

<Paragraph number="2150" type="verses" book="ja1">
_“Devaṃ va yakkhaṃ atha vā manussaṃ,_  
_Na patthaye udayamaticca aññaṃ;_  
_Gaccheva tvaṃ yakkha mahānubhāva,_  
_Mā cassu gantvā punarāvajittha”._  

</Paragraph>

</Division>
<Division number="42" book="ja1" bookNo="22" page="76">

<Paragraph number="2151" type="verses" book="ja1">
_“Yā sā rati uttamā kāmabhoginaṃ,_  
_Yaṃ hetu sattā visamaṃ caranti;_  
_Mā taṃ ratiṃ jīyi tuvaṃ sucimhite,_  
_Dadāmi te rūpiyaṃ kaṃsapūraṃ”._  

</Paragraph>

</Division>
<Division number="43" book="ja1" bookNo="22" page="77">

<Paragraph number="2152" type="verses" book="ja1">
_“Nāriṃ naro nijjhapayaṃ dhanena,_  
_Ukkaṃsatī yattha karoti chandaṃ;_  
_Vipaccanīko tava devadhammo,_  
_Paccakkhato thokatarena esi”._  

</Paragraph>

</Division>
<Division number="44" book="ja1" bookNo="22" page="77">

<Paragraph number="2153" type="verses" book="ja1">
_“Āyu ca vaṇṇo ca manussaloke,_  
_Nihīyati manujānaṃ sugatte;_  
_Teneva vaṇṇena dhanampi tuyhaṃ,_  
_Nihīyati jiṇṇatarāsi ajja._  

</Paragraph>

</Division>
<Division number="45" book="ja1" bookNo="22" page="77">

<Paragraph number="2154" type="verses" book="ja1">
_Evaṃ me pekkhamānassa,_  
_rājaputti yasassini;_  
_Hāyateva tava vaṇṇo,_  
_ahorattānamaccaye”._  

</Paragraph>

</Division>
<Division number="46" book="ja1" bookNo="22" page="77">

<Paragraph number="2155" type="verses" book="ja1">
_“Imināva tvaṃ vayasā,_  
_rājaputti sumedhase;_  
_Brahmacariyaṃ careyyāsi,_  
_bhiyyo vaṇṇavatī siyā”._  

</Paragraph>

</Division>
<Division number="47" book="ja1" bookNo="22" page="77">

<Paragraph number="2156" type="verses" book="ja1">
_“Devā na jīranti yathā manussā,_  
_Gattesu tesaṃ valiyo na honti;_  
_Pucchāmi taṃ yakkha mahānubhāva,_  
_Kathaṃ nu devāna sarīradeho”._  

</Paragraph>

</Division>
<Division number="48" book="ja1" bookNo="22" page="77">

<Paragraph number="2157" type="verses" book="ja1">
_“Devā na jīranti yathā manussā,_  
_Gattesu tesaṃ valiyo na honti;_  
_Suve suve bhiyyatarova tesaṃ,_  
_Dibbo ca vaṇṇo vipulā ca bhogā”._  

</Paragraph>

</Division>
<Division number="49" book="ja1" bookNo="22" page="78">

<Paragraph number="2158" type="verses" book="ja1">
_“Kiṃsūdha bhītā janatā anekā,_  
_Maggo ca nekāyatanaṃ pavutto;_  
_Pucchāmi taṃ yakkha mahānubhāva,_  
_Katthaṭṭhito paralokaṃ na bhāye”._  

</Paragraph>

</Division>
<Division number="50" book="ja1" bookNo="22" page="78">

<Paragraph number="2159" type="verses" book="ja1">
_“Vācaṃ manañca paṇidhāya sammā,_  
_Kāyena pāpāni akubbamāno;_  
_Bahunnapānaṃ gharamāvasanto,_  
_Saddho mudū saṃvibhāgī vadaññū;_  
_Saṅgāhako sakhilo saṇhavāco,_  
_Etthaṭṭhito paralokaṃ na bhāye”._  

</Paragraph>

</Division>
<Division number="51" book="ja1" bookNo="22" page="78">

<Paragraph number="2160" type="verses" book="ja1">
_“Anusāsasi maṃ yakkha,_  
_yathā mātā yathā pitā;_  
_Uḷāravaṇṇaṃ pucchāmi,_  
_ko nu tvamasi subrahā”._  

</Paragraph>

</Division>
<Division number="52" book="ja1" bookNo="22" page="78">

<Paragraph number="2161" type="verses" book="ja1">
_“Udayohamasmi kalyāṇi,_  
_saṅgarattā idhāgato;_  
_Āmanta kho taṃ gacchāmi,_  
_muttosmi tava saṅgarā”._  

</Paragraph>

</Division>
<Division number="53" book="ja1" bookNo="22" page="78">

<Paragraph number="2162" type="verses" book="ja1">
_“Sace kho tvaṃ udayosi,_  
_saṅgarattā idhāgato;_  
_Anusāsa maṃ rājaputta,_  
_yathāssa puna saṅgamo”._  

</Paragraph>

</Division>
<Division number="54" book="ja1" bookNo="22" page="78">

<Paragraph number="2163" type="verses" book="ja1">
_“Atipatati vayo khaṇo tatheva,_  
_Ṭhānaṃ natthi dhuvaṃ cavanti sattā;_  
_Parijiyyati addhuvaṃ sarīraṃ,_  
_Udaye mā pamāda carassu dhammaṃ._  

</Paragraph>

</Division>
<Division number="55" book="ja1" bookNo="22" page="78">

<Paragraph number="2164" type="verses" book="ja1">
_Kasiṇā pathavī dhanassa pūrā,_  
_Ekasseva siyā anaññadheyyā;_  
_Tañcāpi jahati avītarāgo,_  
_Udaye mā pamāda carassu dhammaṃ._  

</Paragraph>

</Division>
<Division number="56" book="ja1" bookNo="22" page="78">

<Paragraph number="2165" type="verses" book="ja1">
_Mātā ca pitā ca bhātaro ca,_  
_Bhariyā yāpi dhanena hoti kītā;_  
_Te cāpi jahanti aññamaññaṃ,_  
_Udaye mā pamāda carassu dhammaṃ._  

</Paragraph>

</Division>
<Division number="57" book="ja1" bookNo="22" page="79">

<Paragraph number="2166" type="verses" book="ja1">
_Kāyo parabhojananti ñatvā,_  
_Saṃsāre sugatiñca duggatiñca;_  
_Ittaravāsoti jāniyāna,_  
_Udaye mā pamāda carassu dhammaṃ”._  

</Paragraph>

</Division>
<Division number="58" book="ja1" bookNo="22" page="79">

<Paragraph number="2167" type="verses" book="ja1">
_“Sādhu bhāsatiyaṃ yakkho,_  
_appaṃ maccāna jīvitaṃ;_  
_Kasirañca parittañca,_  
_tañca dukkhena saṃyutaṃ;_  
_Sāhaṃ ekā pabbajissāmi,_  
_hitvā kāsiṃ surundhanan”ti._  


</Paragraph>
</Division>

<Paragraph number="2168" type="center" book="ja1">
Udayajātakaṃ catutthaṃ.
</Paragraph>
