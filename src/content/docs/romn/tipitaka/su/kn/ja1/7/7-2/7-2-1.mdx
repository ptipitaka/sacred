<Paragraph number="1572" type="verses" book="ja1">
_“Hitvā gāmasahassāni,_  
_paripuṇṇāni soḷasa;_  
_Koṭṭhāgārāni phītāni,_  
_sannidhiṃ dāni kubbasi”._  

</Paragraph>

</Division>
<Division number="77" book="ja1" bookNo="22" page="285">

<Paragraph number="1573" type="verses" book="ja1">
_“Hitvā gandhāravisayaṃ,_  
_pahūtadhanadhāriyaṃ;_  
_Pasāsanato nikkhanto,_  
_idha dāni pasāsasi”._  

</Paragraph>

</Division>
<Division number="78" book="ja1" bookNo="22" page="285">

<Paragraph number="1574" type="verses" book="ja1">
_“Dhammaṃ bhaṇāmi vedeha,_  
_adhammo me na ruccati;_  
_Dhammaṃ me bhaṇamānassa,_  
_na pāpamupalimpati”._  

</Paragraph>

</Division>
<Division number="79" book="ja1" bookNo="22" page="285">

<Paragraph number="1575" type="verses" book="ja1">
_“Yena kenaci vaṇṇena,_  
_paro labhati ruppanaṃ;_  
_Mahatthiyampi ce vācaṃ,_  
_na taṃ bhāseyya paṇḍito”._  

</Paragraph>

</Division>
<Division number="80" book="ja1" bookNo="22" page="285">

<Paragraph number="1576" type="verses" book="ja1">
_“Kāmaṃ ruppatu vā mā vā,_  
_bhusaṃva vikirīyatu;_  
_Dhammaṃ me bhaṇamānassa,_  
_na pāpamupalimpati”._  

</Paragraph>

</Division>
<Division number="81" book="ja1" bookNo="22" page="285">

<Paragraph number="1577" type="verses" book="ja1">
_“No ce assa sakā buddhi,_  
_vinayo vā susikkhito;_  
_Vane andhamahiṃsova,_  
_careyya bahuko jano._  

</Paragraph>

</Division>
<Division number="82" book="ja1" bookNo="22" page="285">

<Paragraph number="1578" type="verses" book="ja1">
_Yasmā ca panidhekacce,_  
_āceramhi susikkhitā;_  
_Tasmā vinītavinayā,_  
_caranti susamāhitā”ti._  


</Paragraph>
</Division>

<Paragraph number="1579" type="center" book="ja1">
Gandhārajātakaṃ paṭhamaṃ.
</Paragraph>
