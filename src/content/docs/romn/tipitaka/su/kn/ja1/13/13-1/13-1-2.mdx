<Division number="14" book="ja1">

<Paragraph number="2398" type="verses" book="ja1">
_“Kuṭhārihattho puriso,_  
_vanamogayha tiṭṭhasi;_  
_Puṭṭho me samma akkhāhi,_  
_kiṃ dāruṃ chetumicchasi”._  

</Paragraph>

</Division>
<Division number="15" book="ja1">

<Paragraph number="2399" type="verses" book="ja1">
_“Isso vanāni carasi,_  
_samāni visamāni ca;_  
_Puṭṭho me samma akkhāhi,_  
_kiṃ dāruṃ nemiyā daḷhaṃ”._  

</Paragraph>

</Division>
<Division number="16" book="ja1">

<Paragraph number="2400" type="verses" book="ja1">
_“Neva sālo na khadiro,_  
_nāssakaṇṇo kuto dhavo;_  
_Rukkho ca phandano nāma,_  
_taṃ dāruṃ nemiyā daḷhaṃ”._  

</Paragraph>

</Division>
<Division number="17" book="ja1">

<Paragraph number="2401" type="verses" book="ja1">
_“Kīdisānissa pattāni,_  
_khandho vā pana kīdiso;_  
_Puṭṭho me samma akkhāhi,_  
_yathā jānemu phandanaṃ”._  

</Paragraph>

</Division>
<Division number="18" book="ja1">

<Paragraph number="2402" type="verses" book="ja1">
_“Yassa sākhā palambanti,_  
_namanti na ca bhañjare;_  
_So rukkho phandano nāma,_  
_yassa mūle ahaṃ ṭhito._  

</Paragraph>

</Division>
<Division number="19" book="ja1">

<Paragraph number="2403" type="verses" book="ja1">
_Arānaṃ cakkanābhīnaṃ,_  
_īsānemirathassa ca;_  
_Sabbassa te kammaniyo,_  
_ayaṃ hessati phandano”._  

</Paragraph>

</Division>
<Division number="20" book="ja1">

<Paragraph number="2404" type="verses" book="ja1">
_“Iti phandanarukkhopi,_  
_tāvade ajjhabhāsatha;_  
_Mayhampi vacanaṃ atthi,_  
_bhāradvāja suṇohi me._  

</Paragraph>

</Division>
<Division number="21" book="ja1">

<Paragraph number="2405" type="verses" book="ja1">
_Issassa upakkhandhamhā,_  
_Ukkacca caturaṅgulaṃ;_  
_Tena nemiṃ pasāresi,_  
_Evaṃ daḷhataraṃ siyā._  

</Paragraph>

</Division>
<Division number="22" book="ja1">

<Paragraph number="2406" type="verses" book="ja1">
_Iti phandanarukkhopi,_  
_veraṃ appesi tāvade;_  
_Jātānañca ajātānaṃ,_  
_issānaṃ dukkhamāvahi._  

</Paragraph>

</Division>
<Division number="23" book="ja1">

<Paragraph number="2407" type="verses" book="ja1">
_Iccevaṃ phandano issaṃ,_  
_isso ca pana phandanaṃ;_  
_Aññamaññaṃ vivādena,_  
_aññamaññamaghātayuṃ._  

</Paragraph>

</Division>
<Division number="24" book="ja1">

<Paragraph number="2408" type="verses" book="ja1">
_Evameva manussānaṃ,_  
_vivādo yattha jāyati;_  
_Mayūranaccaṃ naccanti,_  
_yathā te issaphandanā._  

</Paragraph>

</Division>
<Division number="25" book="ja1">

<Paragraph number="2409" type="verses" book="ja1">
_Taṃ vo vadāmi bhaddaṃ vo,_  
_yāvantettha samāgatā;_  
_Sammodatha mā vivadatha,_  
_mā hotha issaphandanā._  

</Paragraph>

</Division>
<Division number="26" book="ja1">

<Paragraph number="2410" type="verses" book="ja1">
_Sāmaggimeva sikkhetha,_  
_buddhehetaṃ pasaṃsitaṃ;_  
_Sāmaggirato dhammaṭṭho,_  
_yogakkhemā na dhaṃsatī”ti._  


</Paragraph>
</Division>

<Paragraph number="2411" type="center" book="ja1">
Phandanajātakaṃ dutiyaṃ.
</Paragraph>
