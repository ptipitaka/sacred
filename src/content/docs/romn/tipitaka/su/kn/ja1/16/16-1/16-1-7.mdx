<Paragraph number="3485" type="verses" book="ja1">
_“Sace vo vuyhamānānaṃ,_  
_sattannaṃ udakaṇṇave;_  
_Manussabalimesāno,_  
_nāvaṃ gaṇheyya rakkhaso;_  
_Anupubbaṃ kathaṃ datvā,_  
_muñcesi dakarakkhasā”._  

</Paragraph>

</Division>
<Division number="225" book="ja1">

<Paragraph number="3486" type="verses" book="ja1">
_“Mātaraṃ paṭhamaṃ dajjaṃ,_  
_Bhariyaṃ datvāna bhātaraṃ;_  
_Tato sahāyaṃ datvāna,_  
_Pañcamaṃ dajjaṃ brāhmaṇaṃ;_  
_Chaṭṭhāhaṃ dajjamattānaṃ,_  
_Neva dajjaṃ mahosadhaṃ”._  

</Paragraph>

</Division>
<Division number="226" book="ja1">

<Paragraph number="3487" type="verses" book="ja1">
_“Posetā te janettī ca,_  
_dīgharattānukampikā;_  
_Chabbhī tayi padussati,_  
_paṇḍitā atthadassinī;_  
_Aññaṃ upanisaṃ katvā,_  
_vadhā taṃ parimocayi._  

</Paragraph>

</Division>
<Division number="227" book="ja1">

<Paragraph number="3488" type="verses" book="ja1">
_Taṃ tādisiṃ pāṇadadiṃ,_  
_orasaṃ gabbhadhāriniṃ;_  
_Mātaraṃ kena dosena,_  
_dajjāsi dakarakkhino”._  

</Paragraph>

</Division>
<Division number="228" book="ja1">

<Paragraph number="3489" type="verses" book="ja1">
_“Daharā viyalaṅkāraṃ,_  
_dhāreti apiḷandhanaṃ;_  
_Dovārike anīkaṭṭhe,_  
_ativelaṃ pajagghati._  

</Paragraph>

</Division>
<Division number="229" book="ja1">

<Paragraph number="3490" type="verses" book="ja1">
_Athopi paṭirājūnaṃ,_  
_sayaṃ dūtāni sāsati;_  
_Mātaraṃ tena dosena,_  
_dajjāhaṃ dakarakkhino”._  

</Paragraph>

</Division>
<Division number="230" book="ja1">

<Paragraph number="3491" type="verses" book="ja1">
_“Itthigumbassa pavarā,_  
_accantaṃ piyabhāṇinī;_  
_Anuggatā sīlavatī,_  
_chāyāva anapāyinī._  

</Paragraph>

</Division>
<Division number="231" book="ja1">

<Paragraph number="3492" type="verses" book="ja1">
_Akkodhanā puññavatī,_  
_paṇḍitā atthadassinī;_  
_Ubbariṃ kena dosena,_  
_dajjāsi dakarakkhino”._  

</Paragraph>

</Division>
<Division number="232" book="ja1">

<Paragraph number="3493" type="verses" book="ja1">
_“Khiḍḍāratisamāpannaṃ,_  
_anatthavasamāgataṃ;_  
_Sā maṃ sakāna puttānaṃ,_  
_ayācaṃ yācate dhanaṃ._  

</Paragraph>

</Division>
<Division number="233" book="ja1">

<Paragraph number="3494" type="verses" book="ja1">
_Sohaṃ dadāmi sāratto,_  
_bahuṃ uccāvacaṃ dhanaṃ;_  
_Suduccajaṃ cajitvāna,_  
_pacchā socāmi dummano;_  
_Ubbariṃ tena dosena,_  
_dajjāhaṃ dakarakkhino”._  

</Paragraph>

</Division>
<Division number="234" book="ja1">

<Paragraph number="3495" type="verses" book="ja1">
_“Yenocitā janapadā,_  
_ānītā ca paṭiggahaṃ;_  
_Ābhataṃ pararajjebhi,_  
_abhiṭṭhāya bahuṃ dhanaṃ._  

</Paragraph>

</Division>
<Division number="235" book="ja1">

<Paragraph number="3496" type="verses" book="ja1">
_Dhanuggahānaṃ pavaraṃ,_  
_sūraṃ tikhiṇamantinaṃ;_  
_Bhātaraṃ kena dosena,_  
_dajjāsi dakarakkhino”._  

</Paragraph>

</Division>
<Division number="236" book="ja1">

<Paragraph number="3497" type="verses" book="ja1">
_“Yenocitā janapadā,_  
_ānītā ca paṭiggahaṃ;_  
_Ābhataṃ pararajjebhi,_  
_abhiṭṭhāya bahuṃ dhanaṃ._  

</Paragraph>

</Division>
<Division number="237" book="ja1">

<Paragraph number="3498" type="verses" book="ja1">
_Dhanuggahānaṃ pavaro,_  
_sūro tikhiṇamanti ca;_  
_Mayāyaṃ sukhito rājā,_  
_atimaññati dārako._  

</Paragraph>

</Division>
<Division number="238" book="ja1">

<Paragraph number="3499" type="verses" book="ja1">
_Upaṭṭhānampi me ayye,_  
_na so eti yathā pure;_  
_Bhātaraṃ tena dosena,_  
_dajjāhaṃ dakarakkhino”._  

</Paragraph>

</Division>
<Division number="239" book="ja1">

<Paragraph number="3500" type="verses" book="ja1">
_“Ekarattena ubhayo,_  
_tvañceva dhanusekha ca;_  
_Ubho jātettha pañcālā,_  
_sahāyā susamāvayā._  

</Paragraph>

</Division>
<Division number="240" book="ja1">

<Paragraph number="3501" type="verses" book="ja1">
_Cariyā taṃ anubandhittho,_  
_ekadukkhasukho tava;_  
_Ussukko te divārattiṃ,_  
_sabbakiccesu byāvaṭo;_  
_Sahāyaṃ kena dosena,_  
_dajjāsi dakarakkhino”._  

</Paragraph>

</Division>
<Division number="241" book="ja1">

<Paragraph number="3502" type="verses" book="ja1">
_“Cariyā maṃ ayaṃ ayye,_  
_pajagghittho mayā saha;_  
_Ajjāpi tena vaṇṇena,_  
_ativelaṃ pajagghati._  

</Paragraph>

</Division>
<Division number="242" book="ja1">

<Paragraph number="3503" type="verses" book="ja1">
_Ubbariyāpihaṃ ayye,_  
_mantayāmi rahogato;_  
_Anāmanto pavisati,_  
_pubbe appaṭivedito._  

</Paragraph>

</Division>
<Division number="243" book="ja1">

<Paragraph number="3504" type="verses" book="ja1">
_Laddhadvāro katokāso,_  
_ahirikaṃ anādaraṃ;_  
_Sahāyaṃ tena dosena,_  
_dajjāhaṃ dakarakkhino”._  

</Paragraph>

</Division>
<Division number="244" book="ja1">

<Paragraph number="3505" type="verses" book="ja1">
_“Kusalo sabbanimittānaṃ,_  
_rutaññū āgatāgamo;_  
_Uppāte supine yutto,_  
_niyyāne ca pavesane._  

</Paragraph>

</Division>
<Division number="245" book="ja1">

<Paragraph number="3506" type="verses" book="ja1">
_Paṭṭho bhūmantalikkhasmiṃ,_  
_Nakkhattapadakovido;_  
_Brāhmaṇaṃ kena dosena,_  
_Dajjāsi dakarakkhino”._  

</Paragraph>

</Division>
<Division number="246" book="ja1">

<Paragraph number="3507" type="verses" book="ja1">
_“Parisāyampi me ayye,_  
_ummīlitvā udikkhati;_  
_Tasmā accabhamuṃ luddaṃ,_  
_dajjāhaṃ dakarakkhino”._  

</Paragraph>

</Division>
<Division number="247" book="ja1">

<Paragraph number="3508" type="verses" book="ja1">
_“Sasamuddapariyāyaṃ,_  
_mahiṃ sāgarakuṇḍalaṃ;_  
_Vasundharaṃ āvasasi,_  
_amaccaparivārito._  

</Paragraph>

</Division>
<Division number="248" book="ja1">

<Paragraph number="3509" type="verses" book="ja1">
_Cāturanto mahāraṭṭho,_  
_vijitāvī mahabbalo;_  
_Pathabyā ekarājāsi,_  
_yaso te vipulaṃ gato._  

</Paragraph>

</Division>
<Division number="249" book="ja1">

<Paragraph number="3510" type="verses" book="ja1">
_Soḷasitthisahassāni,_  
_āmuttamaṇikuṇḍalā;_  
_Nānājanapadā nārī,_  
_devakaññūpamā subhā._  

</Paragraph>

</Division>
<Division number="250" book="ja1">

<Paragraph number="3511" type="verses" book="ja1">
_Evaṃ sabbaṅgasampannaṃ,_  
_sabbakāmasamiddhinaṃ;_  
_Sukhitānaṃ piyaṃ dīghaṃ,_  
_jīvitaṃ āhu khattiya._  

</Paragraph>

</Division>
<Division number="251" book="ja1">

<Paragraph number="3512" type="verses" book="ja1">
_Atha tvaṃ kena vaṇṇena,_  
_kena vā pana hetunā;_  
_Paṇḍitaṃ anurakkhanto,_  
_pāṇaṃ cajasi duccajaṃ”._  

</Paragraph>

</Division>
<Division number="252" book="ja1">

<Paragraph number="3513" type="verses" book="ja1">
_“Yatopi āgato ayye,_  
_mama hatthaṃ mahosadho;_  
_Nābhijānāmi dhīrassa,_  
_anumattampi dukkaṭaṃ._  

</Paragraph>

</Division>
<Division number="253" book="ja1">

<Paragraph number="3514" type="verses" book="ja1">
_Sace ca kismici kāle,_  
_maraṇaṃ me pure siyā;_  
_So me putte paputte ca,_  
_sukhāpeyya mahosadho._  

</Paragraph>

</Division>
<Division number="254" book="ja1">

<Paragraph number="3515" type="verses" book="ja1">
_Anāgataṃ paccuppannaṃ,_  
_sabbamatthampi passati;_  
_Anāparādhakammantaṃ,_  
_na dajjaṃ dakarakkhino”._  

</Paragraph>

</Division>
<Division number="255" book="ja1">

<Paragraph number="3516" type="verses" book="ja1">
_“Idaṃ suṇātha pañcālā,_  
_cūḷaneyyassa bhāsitaṃ;_  
_Paṇḍitaṃ anurakkhanto,_  
_pāṇaṃ cajati duccajaṃ._  

</Paragraph>

</Division>
<Division number="256" book="ja1">

<Paragraph number="3517" type="verses" book="ja1">
_Mātu bhariyāya bhātucca,_  
_sakhino brāhmaṇassa ca;_  
_Attano cāpi pañcālo,_  
_channaṃ cajati jīvitaṃ._  

</Paragraph>

</Division>
<Division number="257" book="ja1">

<Paragraph number="3518" type="verses" book="ja1">
_Evaṃ mahatthikā paññā,_  
_nipuṇā sādhucintinī;_  
_Diṭṭhadhammahitatthāya,_  
_samparāyasukhāya cā”ti._  


</Paragraph>
</Division>

<Paragraph number="3519" type="center" book="ja1">
