<Paragraph number="1950" type="verses" book="ja1">
_“Bahussuto sutadhammosi saṅkha,_  
_Diṭṭhā tayā samaṇabrāhmaṇā ca;_  
_Athakkhaṇe dassayase vilāpaṃ,_  
_Añño nu ko te paṭimantako mayā”._  

</Paragraph>

</Division>
<Division number="40" book="ja1" bookNo="22" page="42">

<Paragraph number="1951" type="verses" book="ja1">
_“Subbhū subhā suppaṭimukkakambu,_  
_Paggayha sovaṇṇamayāya pātiyā;_  
_‘Bhuñjassu bhattaṃ’ iti maṃ vadeti,_  
_Saddhāvittā_  
_Tamahaṃ noti brūmi”._  

</Paragraph>

</Division>
<Division number="41" book="ja1" bookNo="22" page="43">

<Paragraph number="1952" type="verses" book="ja1">
_“Etādisaṃ brāhmaṇa disvāna yakkhaṃ,_  
_Puccheyya poso sukhamāsisāno;_  
_Uṭṭhehi naṃ pañjalikābhipuccha,_  
_Devī nusi tvaṃ uda mānusī nu”._  

</Paragraph>

</Division>
<Division number="42" book="ja1" bookNo="22" page="43">

<Paragraph number="1953" type="verses" book="ja1">
_“Yaṃ tvaṃ sukhenābhisamekkhase maṃ,_  
_Bhuñjassu bhattaṃ iti maṃ vadesi;_  
_Pucchāmi taṃ nāri mahānubhāve,_  
_Devī nusi tvaṃ uda mānusī nu”._  

</Paragraph>

</Division>
<Division number="43" book="ja1" bookNo="22" page="43">

<Paragraph number="1954" type="verses" book="ja1">
_“Devī ahaṃ saṅkha mahānubhāvā,_  
_Idhāgatā sāgaravārimajjhe;_  
_Anukampikā no ca paduṭṭhacittā,_  
_Taveva atthāya idhāgatāsmi._  

</Paragraph>

</Division>
<Division number="44" book="ja1" bookNo="22" page="43">

<Paragraph number="1955" type="verses" book="ja1">
_Idhannapānaṃ sayanāsanañca,_  
_Yānāni nānāvividhāni saṅkha;_  
_Sabbassa tyāhaṃ paṭipādayāmi,_  
_Yaṃ kiñci tuyhaṃ manasābhipatthitaṃ”._  

</Paragraph>

</Division>
<Division number="45" book="ja1" bookNo="22" page="44">

<Paragraph number="1956" type="verses" book="ja1">
_“Yaṃ kiñci yiṭṭhañca hutañca mayhaṃ,_  
_Sabbassa no issarā tvaṃ sugatte;_  
_Susoṇi subbhamu suvilaggamajjhe,_  
_Kissa me kammassa ayaṃ vipāko”._  

</Paragraph>

</Division>
<Division number="46" book="ja1" bookNo="22" page="44">

<Paragraph number="1957" type="verses" book="ja1">
_“Ghamme pathe brāhmaṇa ekabhikkhuṃ,_  
_Ugghaṭṭapādaṃ tasitaṃ kilantaṃ;_  
_Paṭipādayī saṅkha upāhanāni,_  
_Sā dakkhiṇā kāmaduhā tavajja”._  

</Paragraph>

</Division>
<Division number="47" book="ja1" bookNo="22" page="44">

<Paragraph number="1958" type="verses" book="ja1">
_“Sā hotu nāvā phalakūpapannā,_  
_Anavassutā erakavātayuttā;_  
_Aññassa yānassa na hettha bhūmi,_  
_Ajjeva maṃ moḷiniṃ pāpayassu”._  

</Paragraph>

</Division>
<Division number="48" book="ja1" bookNo="22" page="44">

<Paragraph number="1959" type="verses" book="ja1">
_“Sā tattha vittā sumanā patītā,_  
_Nāvaṃ sucittaṃ abhinimminitvā;_  
_Ādāya saṅkhaṃ purisena saddhiṃ,_  
_Upānayī nagaraṃ sādhuramman”ti._  


</Paragraph>
</Division>

<Paragraph number="1960" type="center" book="ja1">
Saṅkhajātakaṃ catutthaṃ.
</Paragraph>
