<Paragraph number="1856" type="verses" book="ja1">
_“Kāsāyavatthe sakuṇe vadāmi,_  
_Duve duve nandamane carante;_  
_Kaṃ aṇḍajaṃ aṇḍajā mānusesu,_  
_Jātiṃ pasaṃsanti tadiṅgha brūtha”._  

</Paragraph>

</Division>
<Division number="70" book="ja1" bookNo="22" page="366">

<Paragraph number="1857" type="verses" book="ja1">
_“Amhe manussesu manussahiṃsa,_  
_Anubbate cakkavāke vadanti;_  
_Kalyāṇabhāvamhe dijesu sammatā,_  
_Abhirūpā vicarāma aṇṇave”._  
_( )_  

</Paragraph>

</Division>
<Division number="71" book="ja1" bookNo="22" page="366">

<Paragraph number="1858" type="verses" book="ja1">
_“Kiṃ aṇṇave kāni phalāni bhuñje,_  
_Maṃsaṃ kuto khādatha cakkavākā;_  
_Kiṃ bhojanaṃ bhuñjatha vo anomā,_  
_Balañca vaṇṇo ca anapparūpā”._  

</Paragraph>

</Division>
<Division number="72" book="ja1" bookNo="22" page="367">

<Paragraph number="1859" type="verses" book="ja1">
_“Na aṇṇave santi phalāni dhaṅka,_  
_Maṃsaṃ kuto khādituṃ cakkavāke;_  
_Sevālabhakkhamha apāṇabhojanā,_  
_Na ghāsahetūpi karoma pāpaṃ”._  

</Paragraph>

</Division>
<Division number="73" book="ja1" bookNo="22" page="367">

<Paragraph number="1860" type="verses" book="ja1">
_“Na me idaṃ ruccati cakkavāka,_  
_Asmiṃ bhave bhojanasannikāso;_  
_Ahosi pubbe tato me aññathā,_  
_Icceva me vimati ettha jātā._  

</Paragraph>

</Division>
<Division number="74" book="ja1" bookNo="22" page="367">

<Paragraph number="1861" type="verses" book="ja1">
_Ahampi maṃsāni phalāni bhuñje,_  
_Annāni ca loṇiyateliyāni;_  
_Rasaṃ manussesu labhāmi bhottuṃ,_  
_Sūrova saṅgāmamukhaṃ vijetvā;_  
_Na ca me tādiso vaṇṇo,_  
_Cakkavāka yathā tava”._  

</Paragraph>

</Division>
<Division number="75" book="ja1" bookNo="22" page="367">

<Paragraph number="1862" type="verses" book="ja1">
_“Asuddhabhakkhosi khaṇānupātī,_  
_Kicchena te labbhati annapānaṃ;_  
_Na tussasī rukkhaphalehi dhaṅka,_  
_Maṃsāni vā yāni susānamajjhe._  

</Paragraph>

</Division>
<Division number="76" book="ja1" bookNo="22" page="367">

<Paragraph number="1863" type="verses" book="ja1">
_Yo sāhasena adhigamma bhoge,_  
_Paribhuñjati dhaṅka khaṇānupātī;_  
_Tato upakkosati naṃ sabhāvo,_  
_Upakkuṭṭho vaṇṇabalaṃ jahāti._  

</Paragraph>

</Division>
<Division number="77" book="ja1" bookNo="22" page="367">

<Paragraph number="1864" type="verses" book="ja1">
_Appampi ce nibbutiṃ bhuñjatī yadi,_  
_Asāhasena aparūpaghātī;_  
_Balañca vaṇṇo ca tadassa hoti,_  
_Na hi sabbo āhāramayena vaṇṇo”ti._  


</Paragraph>
</Division>

<Paragraph number="1865" type="center" book="ja1">
Cakkavākajātakaṃ aṭṭhamaṃ.
</Paragraph>
