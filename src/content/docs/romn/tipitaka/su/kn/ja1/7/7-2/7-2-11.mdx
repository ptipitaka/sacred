_“Āgamissati me pāpaṃ,_  
_āgamissati me bhayaṃ;_  
_Tadā hi calitā sākhā,_  
_manussena migena vā”._  

</Paragraph>

<Division number="155" book="ja1">

<Paragraph number="1661" type="verses" book="ja1">
_“Bhīruyā nūna me kāmo,_  
_avidūre vasantiyā;_  
_Karissati kisaṃ paṇḍuṃ,_  
_sāva sākhā parantapaṃ._  

</Paragraph>

</Division>
<Division number="156" book="ja1">

<Paragraph number="1662" type="verses" book="ja1">
_Socayissati maṃ kantā,_  
_gāme vasamaninditā;_  
_Karissati kisaṃ paṇḍuṃ,_  
_sāva sākhā parantapaṃ._  

</Paragraph>

</Division>
<Division number="157" book="ja1">

<Paragraph number="1663" type="verses" book="ja1">
_Tayā maṃ asitāpaṅgi,_  
_sitāni bhaṇitāni ca;_  
_Kisaṃ paṇḍuṃ karissanti,_  
_sāva sākhā parantapaṃ”._  

</Paragraph>

</Division>
<Division number="158" book="ja1">

<Paragraph number="1664" type="verses" book="ja1">
_“Agamā nūna so saddo,_  
_asaṃsi nūna so tava;_  
_Akkhātaṃ nūna taṃ tena,_  
_yo taṃ sākhamakampayi._  

</Paragraph>

</Division>
<Division number="159" book="ja1">

<Paragraph number="1665" type="verses" book="ja1">
_Idaṃ kho taṃ samāgamma,_  
_mama bālassa cintitaṃ;_  
_Tadā hi calitā sākhā,_  
_manussena migena vā”._  

</Paragraph>

</Division>
<Division number="160" book="ja1">

<Paragraph number="1666" type="verses" book="ja1">
_“Tatheva tvaṃ avedesi,_  
_Avañci pitaraṃ mama;_  
_Hantvā sākhāhi chādento,_  
_Āgamissati me bhayan”ti. (1140)_  


</Paragraph>
</Division>

<Paragraph number="1667" type="center" book="ja1">
Parantapajātakaṃ ekādasamaṃ.

  
Gandhāravaggo dutiyo.



</Paragraph>

<Paragraph number="1668" type="center" book="ja1">
Tassuddānaṃ


</Paragraph>

<Paragraph number="1669" type="verses" book="ja1">
_Varagāma mahākapi bhaggava ca,_  
_Daḷhadhamma sakuñjara kesavaro;_  
_Urago vidhuro puna jāgarataṃ,_  
_Atha kosalādhipa parantapa cāti._  


</Paragraph>

<Paragraph number="1670" type="center" book="ja1">
Atha vagguddānaṃ


</Paragraph>

<Paragraph number="1671" type="verses" book="ja1">
_Atha sattanipātamhi,_  
_vaggaṃ me bhaṇato suṇa;_  
_Kukku ca puna gandhāro,_  
_dveva vuttā mahesināti._  


</Paragraph>
