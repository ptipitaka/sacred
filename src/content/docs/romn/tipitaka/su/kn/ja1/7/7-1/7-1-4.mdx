<Paragraph number="1509" type="verses" book="ja1">
_“Te kathaṃ nu karissanti,_  
_vuddhā giridarīsayā;_  
_Ahaṃ baddhosmi pāsena,_  
_nilīyassa vasaṃ gato”._  

</Paragraph>

</Division>
<Division number="23" book="ja1">

<Paragraph number="1510" type="verses" book="ja1">
_“Kiṃ gijjha paridevasi,_  
_kā nu te paridevanā;_  
_Na me suto vā diṭṭho vā,_  
_bhāsanto mānusiṃ dijo”._  

</Paragraph>

</Division>
<Division number="24" book="ja1">

<Paragraph number="1511" type="verses" book="ja1">
_“Bharāmi mātāpitaro,_  
_vuddhe giridarīsaye;_  
_Te kathaṃ nu karissanti,_  
_ahaṃ vasaṃ gato tava”._  

</Paragraph>

</Division>
<Division number="25" book="ja1">

<Paragraph number="1512" type="verses" book="ja1">
_“Yannu gijjho yojanasataṃ,_  
_kuṇapāni avekkhati;_  
_Kasmā jālañca pāsañca,_  
_āsajjāpi na bujjhasi”._  

</Paragraph>

</Division>
<Division number="26" book="ja1">

<Paragraph number="1513" type="verses" book="ja1">
_“Yadā parābhavo hoti,_  
_poso jīvitasaṅkhaye;_  
_Atha jālañca pāsañca,_  
_āsajjāpi na bujjhati”._  

</Paragraph>

</Division>
<Division number="27" book="ja1">

<Paragraph number="1514" type="verses" book="ja1">
_“Bharassu mātāpitaro,_  
_vuddhe giridarīsaye;_  
_Mayā tvaṃ samanuññāto,_  
_sotthiṃ passāhi ñātake”._  

</Paragraph>

</Division>
<Division number="28" book="ja1">

<Paragraph number="1515" type="verses" book="ja1">
_“Evaṃ luddaka nandassu,_  
_saha sabbehi ñātibhi;_  
_Bharissaṃ mātāpitaro,_  
_vuddhe giridarīsaye”ti._  


</Paragraph>
</Division>

<Paragraph number="1516" type="center" book="ja1">
Mātuposakagijjhajātakaṃ catutthaṃ.
</Paragraph>
