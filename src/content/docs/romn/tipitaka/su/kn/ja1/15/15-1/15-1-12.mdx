<Paragraph number="3178" type="verses" book="ja1">
_“Pañca paṇḍitā samāgatāttha,_  
_Pañhā me paṭibhāti taṃ suṇātha;_  
_Nindiyamatthaṃ pasaṃsiyaṃ vā,_  
_Kassevāvikareyya guyhamatthaṃ”._  

</Paragraph>

</Division>
<Division number="316" book="ja1" bookNo="22" page="347">

<Paragraph number="3179" type="verses" book="ja1">
_“Tvaṃ āvikarohi bhūmipāla,_  
_Bhattā bhārasaho tuvaṃ vade taṃ;_  
_Tava chandarucīni sammasitvā,_  
_Atha vakkhanti janinda pañca dhīrā”._  

</Paragraph>

</Division>
<Division number="317" book="ja1" bookNo="22" page="347">

<Paragraph number="3180" type="verses" book="ja1">
_“Yā sīlavatī anaññatheyyā,_  
_Bhattucchandavasānugā manāpā;_  
_Nindiyamatthaṃ pasaṃsiyaṃ vā,_  
_Bhariyāyāvikareyya guyhamatthaṃ”._  

</Paragraph>

</Division>
<Division number="318" book="ja1" bookNo="22" page="347">

<Paragraph number="3181" type="verses" book="ja1">
_“Yo kicchagatassa āturassa,_  
_Saraṇaṃ hoti gatī parāyanañca;_  
_Nindiyamatthaṃ pasaṃsiyaṃ vā,_  
_Sakhino vāvikareyya guyhamatthaṃ”._  

</Paragraph>

</Division>
<Division number="319" book="ja1" bookNo="22" page="347">

<Paragraph number="3182" type="verses" book="ja1">
_“Jeṭṭho atha majjhimo kaniṭṭho,_  
_Yo ce sīlasamāhito ṭhitatto;_  
_Nindiyamatthaṃ pasaṃsiyaṃ vā,_  
_Bhātu vāvīkareyya guyhamatthaṃ”._  

</Paragraph>

</Division>
<Division number="320" book="ja1" bookNo="22" page="347">

<Paragraph number="3183" type="verses" book="ja1">
_“Yo ve pituhadayassa paddhagū,_  
_Anujāto pitaraṃ anomapañño;_  
_Nindiyamatthaṃ pasaṃsiyaṃ vā,_  
_Puttassāvikareyya guyhamatthaṃ”._  

</Paragraph>

</Division>
<Division number="321" book="ja1" bookNo="22" page="348">

<Paragraph number="3184" type="verses" book="ja1">
_“Mātā dvipadājanindaseṭṭha,_  
_Yā naṃ poseti chandasā piyena;_  
_Nindiyamatthaṃ pasaṃsiyaṃ vā,_  
_Mātuyāvīkareyya guyhamatthaṃ”._  

</Paragraph>

</Division>
<Division number="322" book="ja1" bookNo="22" page="348">

<Paragraph number="3185" type="verses" book="ja1">
_“Guyhassa hi guyhameva sādhu,_  
_Na hi guyhassa pasatthamāvikammaṃ;_  
_Anipphannatā saheyya dhīro,_  
_Nipphannova yathāsukhaṃ bhaṇeyya”._  

</Paragraph>

</Division>
<Division number="323" book="ja1" bookNo="22" page="348">

<Paragraph number="3186" type="verses" book="ja1">
_“Kiṃ tvaṃ vimanosi rājaseṭṭha,_  
_Dvipadajaninda vacanaṃ suṇoma metaṃ;_  
_Kiṃ cintayamāno dummanosi,_  
_Nūna deva aparādho atthi mayhaṃ”._  

</Paragraph>

</Division>
<Division number="324" book="ja1" bookNo="22" page="348">

<Paragraph number="3187" type="verses" book="ja1">
_“‘Paṇhe vajjho mahosadho’ti,_  
_Āṇatto me vadhāya bhūripañño;_  
_Taṃ cintayamāno dummanosmi,_  
_Na hi devī aparādho atthi tuyhaṃ”._  

</Paragraph>

</Division>
<Division number="325" book="ja1" bookNo="22" page="348">

<Paragraph number="3188" type="verses" book="ja1">
_“Abhidosagato dāni ehisi,_  
_Kiṃ sutvā kiṃ saṅkate mano te;_  
_Ko te kimavoca bhūripañña,_  
_Iṅgha vacanaṃ suṇoma brūhi metaṃ”._  

</Paragraph>

</Division>
<Division number="326" book="ja1" bookNo="22" page="348">

<Paragraph number="3189" type="verses" book="ja1">
_“‘Paṇhe vajjho mahosadho’ti,_  
_Yadi te mantayitaṃ janinda dosaṃ;_  
_Bhariyāya rahogato asaṃsi,_  
_Guyhaṃ pātukataṃ sutaṃ mametaṃ”._  

</Paragraph>

</Division>
<Division number="327" book="ja1" bookNo="22" page="349">

<Paragraph number="3190" type="verses" book="ja1">
_“Yaṃ sālavanasmiṃ senako,_  
_Pāpakammaṃ akāsi asabbhirūpaṃ;_  
_Sakhinova rahogato asaṃsi,_  
_Guyhaṃ pātukataṃ sutaṃ mametaṃ._  

</Paragraph>

</Division>
<Division number="328" book="ja1" bookNo="22" page="349">

<Paragraph number="3191" type="verses" book="ja1">
_Pukkusa purisassa te janinda,_  
_Uppanno rogo arājayutto;_  
_Bhātuñca rahogato asaṃsi,_  
_Guyhaṃ pātukataṃ sutaṃ mametaṃ._  

</Paragraph>

</Division>
<Division number="329" book="ja1" bookNo="22" page="349">

<Paragraph number="3192" type="verses" book="ja1">
_Ābādhoyaṃ asabbhirūpo,_  
_Kāmindo naradevena phuṭṭho;_  
_Puttassa rahogato asaṃsi,_  
_Guyhaṃ pātukataṃ sutaṃ mametaṃ._  

</Paragraph>

</Division>
<Division number="330" book="ja1" bookNo="22" page="349">

<Paragraph number="3193" type="verses" book="ja1">
_Aṭṭhavaṅkaṃ maṇiratanaṃ uḷāraṃ,_  
_Sakko te adadā pitāmahassa;_  
_Devindassa gataṃ tadajja hatthaṃ,_  
_Mātuñca rahogato asaṃsi;_  
_Guyhaṃ pātukataṃ sutaṃ mametaṃ._  

</Paragraph>

</Division>
<Division number="331" book="ja1" bookNo="22" page="349">

<Paragraph number="3194" type="verses" book="ja1">
_Guyhassa hi guyhameva sādhu,_  
_Na hi guyhassa pasatthamāvikammaṃ;_  
_Anipphannatā saheyya dhīro,_  
_Nipphannova yathāsukhaṃ bhaṇeyya._  

</Paragraph>

</Division>
<Division number="332" book="ja1" bookNo="22" page="349">

<Paragraph number="3195" type="verses" book="ja1">
_Na guyhamatthaṃ vivareyya,_  
_rakkheyya naṃ yathā nidhiṃ;_  
_Na hi pātukato sādhu,_  
_guyho attho pajānatā._  

</Paragraph>

</Division>
<Division number="333" book="ja1" bookNo="22" page="349">

<Paragraph number="3196" type="verses" book="ja1">
_Thiyā guyhaṃ na saṃseyya,_  
_amittassa ca paṇḍito;_  
_Yo cāmisena saṃhīro,_  
_hadayattheno ca yo naro._  

</Paragraph>

</Division>
<Division number="334" book="ja1" bookNo="22" page="349">

<Paragraph number="3197" type="verses" book="ja1">
_Guyhamatthaṃ asambuddhaṃ,_  
_sambodhayati yo naro;_  
_Mantabhedabhayā tassa,_  
_dāsabhūto titikkhati._  

</Paragraph>

</Division>
<Division number="335" book="ja1" bookNo="22" page="350">

<Paragraph number="3198" type="verses" book="ja1">
_Yāvanto purisassatthaṃ,_  
_guyhaṃ jānanti mantinaṃ;_  
_Tāvanto tassa ubbegā,_  
_tasmā guyhaṃ na vissaje._  

</Paragraph>

</Division>
<Division number="336" book="ja1" bookNo="22" page="350">

<Paragraph number="3199" type="verses" book="ja1">
_Vivicca bhāseyya divā rahassaṃ,_  
_Rattiṃ giraṃ nātivelaṃ pamuñce;_  
_Upassutikā hi suṇanti mantaṃ,_  
_Tasmā manto khippamupeti bhedan”ti._  


</Paragraph>
</Division>

Pañcapaṇḍitajātakaṃ dvādasamaṃ.
