<Paragraph number="682" type="verses" book="ja1">
_“Bhutvā tiṇaparighāsaṃ,_  
_bhutvā ācāmakuṇḍakaṃ;_  
_Etaṃ te bhojanaṃ āsi,_  
_kasmā dāni na bhuñjasi”._  

</Paragraph>

</Division>
<Division number="11" book="ja1">

<Paragraph number="683" type="verses" book="ja1">
_“Yattha posaṃ na jānanti,_  
_jātiyā vinayena vā;_  
_Bahuṃ tattha mahābrahme,_  
_api ācāmakuṇḍakaṃ._  

</Paragraph>

</Division>
<Division number="12" book="ja1">

<Paragraph number="684" type="verses" book="ja1">
_Tvañca kho maṃ pajānāsi,_  
_yādisāyaṃ hayuttamo;_  
_Jānanto jānamāgamma,_  
_na te bhakkhāmi kuṇḍakan”ti._  


</Paragraph>
</Division>

<Paragraph number="685" type="center" book="ja1">
Kuṇḍakakucchisindhavajātakaṃ catutthaṃ.
</Paragraph>
