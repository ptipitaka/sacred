<Paragraph number="1580" type="verses" book="ja1">
_“Attānaṃ saṅkamaṃ katvā,_  
_Yo sotthiṃ samatārayi;_  
_Kiṃ tvaṃ tesaṃ kime tuyhaṃ,_  
_Honti ete mahākapi”._  

</Paragraph>

</Division>
<Division number="84" book="ja1" bookNo="22" page="286">

<Paragraph number="1581" type="verses" book="ja1">
_“Rājāhaṃ issaro tesaṃ,_  
_yūthassa parihārako;_  
_Tesaṃ sokaparetānaṃ,_  
_bhītānaṃ te arindama._  

</Paragraph>

</Division>
<Division number="85" book="ja1" bookNo="22" page="286">

<Paragraph number="1582" type="verses" book="ja1">
_Ullaṅghayitvā attānaṃ,_  
_vissaṭṭhadhanuno sataṃ;_  
_Tato aparapādesu,_  
_daḷhaṃ bandhaṃ latāguṇaṃ._  

</Paragraph>

</Division>
<Division number="86" book="ja1" bookNo="22" page="286">

<Paragraph number="1583" type="verses" book="ja1">
_Chinnabbhamiva vātena,_  
_Nuṇṇo rukkhaṃ upāgamiṃ;_  
_Sohaṃ appabhavaṃ tattha,_  
_Sākhaṃ hatthehi aggahiṃ._  

</Paragraph>

</Division>
<Division number="87" book="ja1" bookNo="22" page="286">

<Paragraph number="1584" type="verses" book="ja1">
_Taṃ maṃ viyāyataṃ santaṃ,_  
_sākhāya ca latāya ca;_  
_Samanukkamantā pādehi,_  
_sotthiṃ sākhāmigā gatā._  

</Paragraph>

</Division>
<Division number="88" book="ja1" bookNo="22" page="286">

<Paragraph number="1585" type="verses" book="ja1">
_Taṃ maṃ na tapate bandho,_  
_mato me na tapessati;_  
_Sukhamāharitaṃ tesaṃ,_  
_yesaṃ rajjamakārayiṃ._  

</Paragraph>

</Division>
<Division number="89" book="ja1" bookNo="22" page="287">

<Paragraph number="1586" type="verses" book="ja1">
_Esā te upamā rāja,_  
_taṃ suṇohi arindama;_  
_Raññā raṭṭhassa yoggassa,_  
_balassa nigamassa ca;_  
_Sabbesaṃ sukhameṭṭhabbaṃ,_  
_khattiyena pajānatā”ti._  


</Paragraph>
</Division>

<Paragraph number="1587" type="center" book="ja1">
Mahākapijātakaṃ dutiyaṃ.
</Paragraph>
