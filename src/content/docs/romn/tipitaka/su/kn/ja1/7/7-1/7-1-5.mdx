<Division number="29" book="ja1">

<Paragraph number="1517" type="verses" book="ja1">
_“Anutīracārī bhaddante,_  
_sahāyamanudhāva maṃ;_  
_Mahā me gahito maccho,_  
_so maṃ harati vegasā”._  

</Paragraph>

</Division>
<Division number="30" book="ja1">

<Paragraph number="1518" type="verses" book="ja1">
_“Gambhīracārī bhaddante,_  
_daḷhaṃ gaṇhāhi thāmasā;_  
_Ahaṃ taṃ uddharissāmi,_  
_supaṇṇo uragāmiva”._  

</Paragraph>

</Division>
<Division number="31" book="ja1">

<Paragraph number="1519" type="verses" book="ja1">
_“Vivādo no samuppanno,_  
_dabbhapuppha suṇohi me;_  
_Samehi medhagaṃ samma,_  
_vivādo vūpasammataṃ”._  

</Paragraph>

</Division>
<Division number="32" book="ja1">

<Paragraph number="1520" type="verses" book="ja1">
_“Dhammaṭṭhohaṃ pure āsiṃ,_  
_bahū aḍḍā me tīritā;_  
_Samemi medhagaṃ samma,_  
_vivādo vūpasammataṃ._  

</Paragraph>

</Division>
<Division number="33" book="ja1">

<Paragraph number="1521" type="verses" book="ja1">
_Anutīracāri naṅguṭṭhaṃ,_  
_sīsaṃ gambhīracārino;_  
_Accāyaṃ majjhimo khaṇḍo,_  
_dhammaṭṭhassa bhavissati”._  

</Paragraph>

</Division>
<Division number="34" book="ja1">

<Paragraph number="1522" type="verses" book="ja1">
_“Cirampi bhakkho abhavissa,_  
_sace na vivademase;_  
_Asīsakaṃ anaṅguṭṭhaṃ,_  
_siṅgālo harati rohitaṃ”._  

</Paragraph>

</Division>
<Division number="35" book="ja1">

<Paragraph number="1523" type="verses" book="ja1">
_“Yathāpi rājā nandeyya,_  
_rajjaṃ laddhāna khattiyo;_  
_Evāhamajja nandāmi,_  
_disvā puṇṇamukhaṃ patiṃ._  

</Paragraph>

</Division>
<Division number="36" book="ja1">

<Paragraph number="1524" type="verses" book="ja1">
_Kathaṃ nu thalajo santo,_  
_udake macchaṃ parāmasi;_  
_Puṭṭho me samma akkhāhi,_  
_kathaṃ adhigataṃ tayā”._  

</Paragraph>

</Division>
<Division number="37" book="ja1">

<Paragraph number="1525" type="verses" book="ja1">
_“Vivādena kisā honti,_  
_vivādena dhanakkhayā;_  
_Jīnā uddā vivādena,_  
_bhuñja māyāvi rohitaṃ._  

</Paragraph>

</Division>
<Division number="38" book="ja1">

<Paragraph number="1526" type="verses" book="ja1">
_Evameva manussesu,_  
_vivādo yattha jāyati;_  
_Dhammaṭṭhaṃ paṭidhāvanti,_  
_so hi nesaṃ vināyako;_  
_Dhanāpi tattha jīyanti,_  
_rājakoso pavaḍḍhatī”ti._  


</Paragraph>
</Division>

<Paragraph number="1527" type="center" book="ja1">
Dabbhapupphajātakaṃ pañcamaṃ.
</Paragraph>
