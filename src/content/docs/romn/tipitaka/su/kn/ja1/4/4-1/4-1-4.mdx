<Division number="13" book="ja1">

<Paragraph number="898" type="verses" book="ja1">
_“Imāni maṃ daddara tāpayanti,_  
_Vācāduruttāni manussaloke;_  
_Maṇḍūkabhakkhā udakantasevī,_  
_Āsīvisaṃ maṃ avisā sapanti”._  

</Paragraph>

</Division>
<Division number="14" book="ja1">

<Paragraph number="899" type="verses" book="ja1">
_“Sakā raṭṭhā pabbājito,_  
_aññaṃ janapadaṃ gato;_  
_Mahantaṃ koṭṭhaṃ kayirātha,_  
_duruttānaṃ nidhetave._  

</Paragraph>

</Division>
<Division number="15" book="ja1">

<Paragraph number="900" type="verses" book="ja1">
_Yattha posaṃ na jānanti,_  
_jātiyā vinayena vā;_  
_Na tattha mānaṃ kayirātha,_  
_vasamaññātake jane._  

</Paragraph>

</Division>
<Division number="16" book="ja1">

<Paragraph number="901" type="verses" book="ja1">
_Videsavāsaṃ vasato,_  
_jātavedasamenapi;_  
_Khamitabbaṃ sapaññena,_  
_api dāsassa tajjitan”ti._  


</Paragraph>
</Division>

<Paragraph number="902" type="center" book="ja1">
Daddarajātakaṃ catutthaṃ.
</Paragraph>
