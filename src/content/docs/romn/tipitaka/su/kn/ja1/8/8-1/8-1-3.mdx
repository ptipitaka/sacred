<Paragraph number="1692" type="verses" book="ja1">
_“Idaṃ suvaṇṇakāyūraṃ,_  
_muttā veḷuriyā bahū;_  
_Sabbaṃ harassu bhaddante,_  
_mañca dāsīti sāvaya”._  

</Paragraph>

</Division>
<Division number="19" book="ja1" bookNo="22" page="311">

<Paragraph number="1693" type="verses" book="ja1">
_“Oropayassu kalyāṇi,_  
_mā bāḷhaṃ paridevasi;_  
_Na cāhaṃ abhijānāmi,_  
_ahantvā dhanamābhataṃ”._  

</Paragraph>

</Division>
<Division number="20" book="ja1" bookNo="22" page="311">

<Paragraph number="1694" type="verses" book="ja1">
_“Yato sarāmi attānaṃ,_  
_yato pattāsmi viññutaṃ;_  
_Na cāhaṃ abhijānāmi,_  
_aññaṃ piyataraṃ tayā._  

</Paragraph>

</Division>
<Division number="21" book="ja1" bookNo="22" page="312">

<Paragraph number="1695" type="verses" book="ja1">
_Ehi taṃ upagūhissaṃ,_  
_karissañca padakkhiṇaṃ;_  
_Na hi dāni puna atthi,_  
_mama tuyhañca saṅgamo”._  

</Paragraph>

</Division>
<Division number="22" book="ja1" bookNo="22" page="312">

<Paragraph number="1696" type="verses" book="ja1">
_“Na hi sabbesu ṭhānesu,_  
_puriso hoti paṇḍito;_  
_Itthīpi paṇḍitā hoti,_  
_tattha tattha vicakkhaṇā._  

</Paragraph>

</Division>
<Division number="23" book="ja1" bookNo="22" page="312">

<Paragraph number="1697" type="verses" book="ja1">
_Na hi sabbesu ṭhānesu,_  
_puriso hoti paṇḍito;_  
_Itthīpi paṇḍitā hoti,_  
_lahuṃ atthaṃ vicintikā._  

</Paragraph>

</Division>
<Division number="24" book="ja1" bookNo="22" page="312">

<Paragraph number="1698" type="verses" book="ja1">
_Lahuñca vata khippañca,_  
_nikaṭṭhe samacetayi;_  
_Migaṃ puṇṇāyateneva,_  
_sulasā sattukaṃ vadhi._  

</Paragraph>

</Division>
<Division number="25" book="ja1" bookNo="22" page="312">

<Paragraph number="1699" type="verses" book="ja1">
_Yodha uppatitaṃ atthaṃ,_  
_na khippamanubujjhati;_  
_So haññati mandamati,_  
_corova girigabbhare._  

</Paragraph>

</Division>
<Division number="26" book="ja1" bookNo="22" page="312">

<Paragraph number="1700" type="verses" book="ja1">
_Yo ca uppatitaṃ atthaṃ,_  
_khippameva nibodhati;_  
_Muccate sattusambādhā,_  
_sulasā sattukāmivā”ti._  


</Paragraph>
</Division>

<Paragraph number="1701" type="center" book="ja1">
Sulasājātakaṃ tatiyaṃ.
</Paragraph>
