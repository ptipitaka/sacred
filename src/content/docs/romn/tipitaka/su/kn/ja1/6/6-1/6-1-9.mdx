<Paragraph number="1390" type="verses" book="ja1">
_“Dhammaṃ caratha ñātayo,_  
_Dhammaṃ caratha bhaddaṃ vo;_  
_Dhammacārī sukhaṃ seti,_  
_Asmiṃ loke paramhi ca”._  

</Paragraph>

</Division>
<Division number="66" book="ja1" bookNo="22" page="264">

<Paragraph number="1391" type="verses" book="ja1">
_“Bhaddako vatayaṃ pakkhī,_  
_dijo paramadhammiko;_  
_Ekapādena tiṭṭhanto,_  
_dhammamevānusāsati”._  

</Paragraph>

</Division>
<Division number="67" book="ja1" bookNo="22" page="264">

<Paragraph number="1392" type="verses" book="ja1">
_“Nāssa sīlaṃ vijānātha,_  
_anaññāya pasaṃsatha;_  
_Bhutvā aṇḍañca potañca,_  
_dhammo dhammoti bhāsati._  

</Paragraph>

</Division>
<Division number="68" book="ja1" bookNo="22" page="264">

<Paragraph number="1393" type="verses" book="ja1">
_Aññaṃ bhaṇati vācāya,_  
_aññaṃ kāyena kubbati;_  
_Vācāya no ca kāyena,_  
_na taṃ dhammaṃ adhiṭṭhito._  

</Paragraph>

</Division>
<Division number="69" book="ja1" bookNo="22" page="264">

<Paragraph number="1394" type="verses" book="ja1">
_Vācāya sakhilo manoviduggo,_  
_Channo kūpasayova kaṇhasappo;_  
_Dhammadhajo gāmanigamāsu sādhu,_  
_Dujjāno purisena bālisena._  

</Paragraph>

</Division>
<Division number="70" book="ja1" bookNo="22" page="265">

<Paragraph number="1395" type="verses" book="ja1">
_Imaṃ tuṇḍehi pakkhehi,_  
_pādā cimaṃ viheṭhatha;_  
_Chavañhimaṃ vināsetha,_  
_nāyaṃ saṃvāsanāraho”ti._  


</Paragraph>
</Division>

<Paragraph number="1396" type="center" book="ja1">
Dhammadhajajātakaṃ navamaṃ.
</Paragraph>
