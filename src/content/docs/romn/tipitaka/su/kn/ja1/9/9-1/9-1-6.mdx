<Paragraph number="1834" type="verses" book="ja1">
_“Bahussutaṃ cittakathiṃ,_  
_gaṅgā vahati pāṭaliṃ;_  
_Vuyhamānaka bhaddante,_  
_ekaṃ me dehi gāthakaṃ”._  

</Paragraph>

</Division>
<Division number="50" book="ja1" bookNo="22" page="363">

<Paragraph number="1835" type="verses" book="ja1">
_“Yena siñcanti dukkhitaṃ,_  
_yena siñcanti āturaṃ;_  
_Tassa majjhe marissāmi,_  
_jātaṃ saraṇato bhayaṃ”._  

</Paragraph>

</Division>
<Division number="51" book="ja1" bookNo="22" page="363">

<Paragraph number="1836" type="verses" book="ja1">
_“Yattha bījāni rūhanti,_  
_sattā yattha patiṭṭhitā;_  
_Sā me sīsaṃ nipīḷeti,_  
_jātaṃ saraṇato bhayaṃ”._  

</Paragraph>

</Division>
<Division number="52" book="ja1" bookNo="22" page="363">

<Paragraph number="1837" type="verses" book="ja1">
_“Yena bhattāni paccanti,_  
_sītaṃ yena vihaññati;_  
_So maṃ ḍahati gattāni,_  
_jātaṃ saraṇato bhayaṃ”._  

</Paragraph>

</Division>
<Division number="53" book="ja1" bookNo="22" page="364">

<Paragraph number="1838" type="verses" book="ja1">
_“Yena bhuttena yāpenti,_  
_puthū brāhmaṇakhattiyā;_  
_So maṃ bhutto byāpādeti,_  
_jātaṃ saraṇato bhayaṃ”._  

</Paragraph>

</Division>
<Division number="54" book="ja1" bookNo="22" page="364">

<Paragraph number="1839" type="verses" book="ja1">
_“Gimhānaṃ pacchime māse,_  
_vātamicchanti paṇḍitā;_  
_So maṃ bhañjati gattāni,_  
_jātaṃ saraṇato bhayaṃ”._  

</Paragraph>

</Division>
<Division number="55" book="ja1" bookNo="22" page="364">

<Paragraph number="1840" type="verses" book="ja1">
_“Yaṃ nissitā jagatiruhaṃ,_  
_svāyaṃ aggiṃ pamuñcati;_  
_Disā bhajatha vakkaṅgā,_  
_jātaṃ saraṇato bhayaṃ”._  

</Paragraph>

</Division>
<Division number="56" book="ja1" bookNo="22" page="364">

<Paragraph number="1841" type="verses" book="ja1">
_“Yamānayiṃ somanassaṃ,_  
_māliniṃ candanussadaṃ;_  
_Sā maṃ gharā nicchubhati,_  
_jātaṃ saraṇato bhayaṃ”._  

</Paragraph>

</Division>
<Division number="57" book="ja1" bookNo="22" page="316">

<Paragraph number="1842" type="verses" book="ja1">
_“Yena jātena nandissaṃ,_  
_yassa ca bhavamicchisaṃ;_  
_So maṃ gharā nicchubhati,_  
_jātaṃ saraṇato bhayaṃ”._  

</Paragraph>

</Division>
<Division number="58" book="ja1" bookNo="22" page="364">

<Paragraph number="1843" type="verses" book="ja1">
_“Suṇantu me jānapadā,_  
_negamā ca samāgatā;_  
_Yatodakaṃ tadādittaṃ,_  
_yato khemaṃ tato bhayaṃ._  

</Paragraph>

</Division>
<Division number="59" book="ja1" bookNo="22" page="364">

<Paragraph number="1844" type="verses" book="ja1">
_Rājā vilumpate raṭṭhaṃ,_  
_brāhmaṇo ca purohito;_  
_Attaguttā viharatha,_  
_jātaṃ saraṇato bhayan”ti._  


</Paragraph>
</Division>

<Paragraph number="1845" type="center" book="ja1">
Padakusalamāṇavajātakaṃ chaṭṭhaṃ.
</Paragraph>
