<Paragraph number="2697" type="verses" book="ja1">
_“Sace hi tyāhaṃ dhanahetu gāhito,_  
_Mā maṃ vadhī jīvagāhaṃ gahetvā;_  
_Rañño ca maṃ samma upantikaṃ nehi,_  
_Maññe dhanaṃ lacchasinapparūpaṃ”._  

</Paragraph>

</Division>
<Division number="144" book="ja1" bookNo="22" page="112">

<Paragraph number="2698" type="verses" book="ja1">
_“Na me ayaṃ tuyha vadhāya ajja,_  
_Samāhito cāpadhure khurappo;_  
_Pāsañca tyāhaṃ adhipātayissaṃ,_  
_Yathāsukhaṃ gacchatu morarājā”._  

</Paragraph>

</Division>
<Division number="145" book="ja1" bookNo="22" page="112">

<Paragraph number="2699" type="verses" book="ja1">
_“Yaṃ satta vassāni mamānubandhi,_  
_Rattindivaṃ khuppipāsaṃ sahanto;_  
_Atha kissa maṃ pāsavasūpanītaṃ,_  
_Pamuttave icchasi bandhanasmā._  

</Paragraph>

</Division>
<Division number="146" book="ja1" bookNo="22" page="112">

<Paragraph number="2700" type="verses" book="ja1">
_Pāṇātipātā virato nusajja,_  
_Abhayaṃ nu te sabbabhūtesu dinnaṃ;_  
_Yaṃ maṃ tuvaṃ pāsavasūpanītaṃ,_  
_Pamuttave icchasi bandhanasmā”._  

</Paragraph>

</Division>
<Division number="147" book="ja1" bookNo="22" page="112">

<Paragraph number="2701" type="verses" book="ja1">
_“Pāṇātipātā viratassa brūhi,_  
_Abhayañca yo sabbabhūtesu deti;_  
_Pucchāmi taṃ morarājetamatthaṃ,_  
_Ito cuto kiṃ labhate sukhaṃ so”._  

</Paragraph>

</Division>
<Division number="148" book="ja1" bookNo="22" page="112">

<Paragraph number="2702" type="verses" book="ja1">
_“Pāṇātipātā viratassa brūmi,_  
_Abhayañca yo sabbabhūtesu deti;_  
_Diṭṭheva dhamme labhate pasaṃsaṃ,_  
_Saggañca so yāti sarīrabhedā”._  

</Paragraph>

</Division>
<Division number="149" book="ja1" bookNo="22" page="112">

<Paragraph number="2703" type="verses" book="ja1">
_“Na santi devā iti āhu eke,_  
_Idheva jīvo vibhavaṃ upeti;_  
_Tathā phalaṃ sukatadukkaṭānaṃ,_  
_Dattupaññattañca vadanti dānaṃ;_  
_Tesaṃ vaco arahataṃ saddahāno,_  
_Tasmā ahaṃ sakuṇe bādhayāmi”._  

</Paragraph>

</Division>
<Division number="150" book="ja1" bookNo="22" page="112">

<Paragraph number="2704" type="verses" book="ja1">
_“Cando ca suriyo ca ubho sudassanā,_  
_Gacchanti obhāsayamantalikkhe;_  
_Imassa lokassa parassa vā te,_  
_Kathaṃ nu te āhu manussaloke”._  

</Paragraph>

</Division>
<Division number="151" book="ja1" bookNo="22" page="173">

<Paragraph number="2705" type="verses" book="ja1">
_“Cando ca suriyo ca ubho sudassanā,_  
_Gacchanti obhāsayamantalikkhe;_  
_Parassa lokassa na te imassa,_  
_Devāti te āhu manussaloke”._  

</Paragraph>

</Division>
<Division number="152" book="ja1" bookNo="22" page="112">

<Paragraph number="2706" type="verses" book="ja1">
_“Ettheva te nīhatā hīnavādā,_  
_Ahetukā ye na vadanti kammaṃ;_  
_Tathā phalaṃ sukatadukkaṭānaṃ,_  
_Dattupaññattaṃ ye ca vadanti dānaṃ”._  

</Paragraph>

</Division>
<Division number="153" book="ja1" bookNo="22" page="113">

<Paragraph number="2707" type="verses" book="ja1">
_“Addhā hi saccaṃ vacanaṃ tavedaṃ,_  
_Kathañhi dānaṃ aphalaṃ bhaveyya;_  
_Tathā phalaṃ sukatadukkaṭānaṃ,_  
_Dattupaññattañca kathaṃ bhaveyya._  

</Paragraph>

</Division>
<Division number="154" book="ja1" bookNo="22" page="113">

<Paragraph number="2708" type="verses" book="ja1">
_Kathaṃkaro kintikaro kimācaraṃ,_  
_Kiṃ sevamāno kena tapoguṇena;_  
_Akkhāhi me morarājetamatthaṃ,_  
_Yathā ahaṃ no nirayaṃ pateyyaṃ”._  

</Paragraph>

</Division>
<Division number="155" book="ja1" bookNo="22" page="113">

<Paragraph number="2709" type="verses" book="ja1">
_“Ye keci atthi samaṇā pathabyā,_  
_Kāsāyavatthā anagāriyā te;_  
_Pātova piṇḍāya caranti kāle,_  
_Vikālacariyā viratā hi santo._  

</Paragraph>

</Division>
<Division number="156" book="ja1" bookNo="22" page="113">

<Paragraph number="2710" type="verses" book="ja1">
_Te tattha kālenupasaṅkamitvā,_  
_Pucchāhi yaṃ te manaso piyaṃ siyā;_  
_Te te pavakkhanti yathāpajānaṃ,_  
_Imassa lokassa parassa catthaṃ”._  

</Paragraph>

</Division>
<Division number="157" book="ja1" bookNo="22" page="113">

<Paragraph number="2711" type="verses" book="ja1">
_“Tacaṃva jiṇṇaṃ urago purāṇaṃ,_  
_Paṇḍūpalāsaṃ harito dumova;_  
_Esappahīno mama luddabhāvo,_  
_Jahāmahaṃ luddakabhāvamajja._  

</Paragraph>

</Division>
<Division number="158" book="ja1" bookNo="22" page="113">

<Paragraph number="2712" type="verses" book="ja1">
_Ye cāpi me sakuṇā atthi baddhā,_  
_Satāninekāni nivesanasmiṃ;_  
_Tesampahaṃ jīvitamajja dammi,_  
_Mokkhañca te pattā sakaṃ niketaṃ”._  

</Paragraph>

</Division>
<Division number="159" book="ja1" bookNo="22" page="113">

<Paragraph number="2713" type="verses" book="ja1">
_“Luddo carī pāsahattho araññe,_  
_Bādhetu morādhipatiṃ yasassiṃ;_  
_Bandhitvā morādhipatiṃ yasassiṃ,_  
_Dukkhā sa pamucci yathāhaṃ pamutto”ti._  


</Paragraph>
