<Division number="97" book="ja1">

<Paragraph number="3355" type="verses" book="ja1">
_“Kiṃ nu socasinuccaṅgi,_  
_paṇḍūsi varavaṇṇini;_  
_Milāyasi visālakkhi,_  
_mālāva parimadditā”._  

</Paragraph>

</Division>
<Division number="98" book="ja1">

<Paragraph number="3356" type="verses" book="ja1">
_“Dohaḷo me mahārāja,_  
_supinantenupajjhagā;_  
_Na so sulabharūpova,_  
_yādiso mama dohaḷo”._  

</Paragraph>

</Division>
<Division number="99" book="ja1">

<Paragraph number="3357" type="verses" book="ja1">
_“Ye keci mānusā kāmā,_  
_idha lokasmi nandane;_  
_Sabbe te pacurā mayhaṃ,_  
_ahaṃ te dammi dohaḷaṃ”._  

</Paragraph>

</Division>
<Division number="100" book="ja1">

<Paragraph number="3358" type="verses" book="ja1">
_“Luddā deva samāyantu,_  
_ye keci vijite tava;_  
_Etesaṃ ahamakkhissaṃ,_  
_yādiso mama dohaḷo”._  

</Paragraph>

</Division>
<Division number="101" book="ja1">

<Paragraph number="3359" type="verses" book="ja1">
_“Ime te luddakā devi,_  
_katahatthā visāradā;_  
_Vanaññū ca migaññū ca,_  
_mamatthe cattajīvitā”._  

</Paragraph>

</Division>
<Division number="102" book="ja1">

<Paragraph number="3360" type="verses" book="ja1">
_“Luddaputtā nisāmetha,_  
_yāvantettha samāgatā;_  
_Chabbisāṇaṃ gajaṃ setaṃ,_  
_addasaṃ supine ahaṃ;_  
_Tassa dantehi me attho,_  
_alābhe natthi jīvitaṃ”._  

</Paragraph>

</Division>
<Division number="103" book="ja1">

<Paragraph number="3361" type="verses" book="ja1">
_“Na no pitūnaṃ na pitāmahānaṃ,_  
_Diṭṭho suto kuñjaro chabbisāṇo;_  
_Yamaddasā supine rājaputtī,_  
_Akkhāhi no yādiso hatthināgo”._  

</Paragraph>

</Division>
<Division number="104" book="ja1">

<Paragraph number="3362" type="verses" book="ja1">
_“Disā catasso vidisā catasso,_  
_Uddhaṃ adho dasa disā imāyo;_  
_Katamaṃ disaṃ tiṭṭhati nāgarājā,_  
_Yamaddasā supine chabbisāṇaṃ”._  

</Paragraph>

</Division>
<Division number="105" book="ja1">

<Paragraph number="3363" type="verses" book="ja1">
_“Ito ujuṃ uttariyaṃ disāyaṃ,_  
_Atikkamma so sattagirī brahante;_  
_Suvaṇṇapasso nāma girī uḷāro,_  
_Supupphito kimpurisānuciṇṇo._  

</Paragraph>

</Division>
<Division number="106" book="ja1">

<Paragraph number="3364" type="verses" book="ja1">
_Āruyha selaṃ bhavanaṃ kinnarānaṃ,_  
_Olokaya pabbatapādamūlaṃ;_  
_Atha dakkhasī meghasamānavaṇṇaṃ,_  
_Nigrodharājaṃ aṭṭhasahassapādaṃ._  

</Paragraph>

</Division>
<Division number="107" book="ja1">

<Paragraph number="3365" type="verses" book="ja1">
_Tatthacchatī kuñjaro chabbisāṇo,_  
_Sabbaseto duppasaho parebhi;_  
_Rakkhanti naṃ aṭṭhasahassanāgā,_  
_Īsādantā vātajavappahārino._  

</Paragraph>

</Division>
<Division number="108" book="ja1">

<Paragraph number="3366" type="verses" book="ja1">
_Tiṭṭhanti te tumūlaṃ passasantā,_  
_Kuppanti vātassapi eritassa;_  
_Manussabhūtaṃ pana tattha disvā,_  
_Bhasmaṃ kareyyuṃ nāssa rajopi tassa”._  

</Paragraph>

</Division>
<Division number="109" book="ja1">

<Paragraph number="3367" type="verses" book="ja1">
_“Bahū hime rājakulamhi santi,_  
_Piḷandhanā jātarūpassa devī;_  
_Muttā maṇī veḷuriyāmayā ca,_  
_Kiṃ kāhasi dantapiḷandhanena;_  
_Māretukāmā kuñjaraṃ chabbisāṇaṃ,_  
_Udāhu ghātessasi luddaputte”._  

</Paragraph>

</Division>
<Division number="110" book="ja1">

<Paragraph number="3368" type="verses" book="ja1">
_“Sā issitā dukkhitā casmi ludda,_  
_Uddhañca sussāmi anussarantī;_  
_Karohi me luddaka etamatthaṃ,_  
_Dassāmi te gāmavarāni pañca”._  

</Paragraph>

</Division>
<Division number="111" book="ja1">

<Paragraph number="3369" type="verses" book="ja1">
_“Katthacchatī kattha mupeti ṭhānaṃ,_  
_Vīthissa kā nhāna gatassa hoti;_  
_Kathañhi so nhāyati nāgarājā,_  
_Kathaṃ vijānemu gatiṃ gajassa”._  

</Paragraph>

</Division>
<Division number="112" book="ja1">

<Paragraph number="3370" type="verses" book="ja1">
_“Tattheva sā pokkharaṇī adūre,_  
_Rammā sutitthā ca mahodikā ca;_  
_Sampupphitā bhamaragaṇānuciṇṇā,_  
_Ettha hi so nhāyati nāgarājā._  

</Paragraph>

</Division>
<Division number="113" book="ja1">

<Paragraph number="3371" type="verses" book="ja1">
_Sīsaṃ nahātuppalamālabhārī,_  
_Sabbaseto puṇḍarīkattacaṅgī;_  
_Āmodamāno gacchati sanniketaṃ,_  
_Purakkhatvā mahesiṃ sabbabhaddaṃ”._  

</Paragraph>

</Division>
<Division number="114" book="ja1">

<Paragraph number="3372" type="verses" book="ja1">
_Tattheva so uggahetvāna vākyaṃ,_  
_Ādāya tūṇiñca dhanuñca luddo;_  
_Vituriyati sattagirī brahante,_  
_Suvaṇṇapassaṃ nāma giriṃ uḷāraṃ._  

</Paragraph>

</Division>
<Division number="115" book="ja1">

<Paragraph number="3373" type="verses" book="ja1">
_Āruyha selaṃ bhavanaṃ kinnarānaṃ,_  
_Olokayī pabbatapādamūlaṃ;_  
_Tatthaddasā meghasamānavaṇṇaṃ,_  
_Nigrodharājaṃ aṭṭhasahassapādaṃ._  

</Paragraph>

</Division>
<Division number="116" book="ja1">

<Paragraph number="3374" type="verses" book="ja1">
_Tatthaddasā kuñjaraṃ chabbisāṇaṃ,_  
_Sabbasetaṃ duppasahaṃ parebhi;_  
_Rakkhanti naṃ aṭṭhasahassanāgā,_  
_Īsādantā vātajavappahārino._  

</Paragraph>

</Division>
<Division number="117" book="ja1">

<Paragraph number="3375" type="verses" book="ja1">
_Tatthaddasā pokkharaṇiṃ adūre,_  
_Rammaṃ sutitthañca mahodikañca;_  
_Sampupphitaṃ bhamaragaṇānuciṇṇaṃ,_  
_Yattha hi so nhāyati nāgarājā._  

</Paragraph>

</Division>
<Division number="118" book="ja1">

<Paragraph number="3376" type="verses" book="ja1">
_Disvāna nāgassa gatiṃ ṭhitiñca,_  
_Vīthissa yā nhānagatassa hoti;_  
_Opātamāgacchi anariyarūpo,_  
_Payojito cittavasānugāya._  

</Paragraph>

</Division>
<Division number="119" book="ja1">

<Paragraph number="3377" type="verses" book="ja1">
_Khaṇitvāna kāsuṃ phalakehi chādayi,_  
_Attānamodhāya dhanuñca luddo;_  
_Passāgataṃ puthusallena nāgaṃ,_  
_Samappayī dukkaṭakammakārī._  

</Paragraph>

</Division>
<Division number="120" book="ja1">

<Paragraph number="3378" type="verses" book="ja1">
_Viddho ca nāgo koñcamanādi ghoraṃ,_  
_Sabbe ca nāgā ninnaduṃ ghorarūpaṃ;_  
_Tiṇañca kaṭṭhañca raṇaṃ karontā,_  
_Dhāviṃsu te aṭṭha disā samantato._  

</Paragraph>

</Division>
<Division number="121" book="ja1">

<Paragraph number="3379" type="verses" book="ja1">
_Vadhissametanti parāmasanto,_  
_Kāsāvamaddakkhi dhajaṃ isīnaṃ;_  
_Dukkhena phuṭṭhassudapādi saññā,_  
_Arahaddhajo sabbhi avajjharūpo._  

</Paragraph>

</Division>
<Division number="122" book="ja1">

<Paragraph number="3380" type="verses" book="ja1">
_“Anikkasāvo kāsāvaṃ,_  
_yo vatthaṃ paridahissati;_  
_Apeto damasaccena,_  
_na so kāsāvamarahati._  

</Paragraph>

</Division>
<Division number="123" book="ja1">

<Paragraph number="3381" type="verses" book="ja1">
_Yo ca vantakasāvassa,_  
_sīlesu susamāhito;_  
_Upeto damasaccena,_  
_sa ve kāsāvamarahati”._  

</Paragraph>

</Division>
<Division number="124" book="ja1">

<Paragraph number="3382" type="verses" book="ja1">
_Samappito puthusallena nāgo,_  
_Aduṭṭhacitto luddakamajjhabhāsi;_  
_“Kimatthayaṃ kissa vā samma hetu,_  
_Mamaṃ vadhī kassa vāyaṃ payogo”._  

</Paragraph>

</Division>
<Division number="125" book="ja1">

<Paragraph number="3383" type="verses" book="ja1">
_“Kāsissa rañño mahesī bhadante,_  
_Sā pūjitā rājakule subhaddā;_  
_Taṃ addasā sā ca mamaṃ asaṃsi,_  
_Dantehi atthoti ca maṃ avoca”._  

</Paragraph>

</Division>
<Division number="126" book="ja1">

<Paragraph number="3384" type="verses" book="ja1">
_“Bahū hi me dantayugā uḷārā,_  
_Ye me pitūnañca pitāmahānaṃ;_  
_Jānāti sā kodhanā rājaputtī,_  
_Vadhatthikā veramakāsi bālā._  

</Paragraph>

</Division>
<Division number="127" book="ja1">

<Paragraph number="3385" type="verses" book="ja1">
_Uṭṭhehi tvaṃ ludda kharaṃ gahetvā,_  
_Dante ime chinda purā marāmi;_  
_Vajjāsi taṃ kodhanaṃ rājaputtiṃ,_  
_‘Nāgo hato handa imassa dantā’”._  

</Paragraph>

</Division>
<Division number="128" book="ja1">

<Paragraph number="3386" type="verses" book="ja1">
_Uṭṭhāya so luddo kharaṃ gahetvā,_  
_Chetvāna dantāni gajuttamassa;_  
_Vaggū subhe appaṭime pathabyā,_  
_Ādāya pakkāmi tato hi khippaṃ._  

</Paragraph>

</Division>
<Division number="129" book="ja1">

<Paragraph number="3387" type="verses" book="ja1">
_Bhayaṭṭitā nāgavadhena aṭṭā,_  
_Ye te nāgā aṭṭha disā vidhāvuṃ;_  
_Adisvāna posaṃ gajapaccamittaṃ,_  
_Paccāgamuṃ yena so nāgarājā._  

</Paragraph>

</Division>
<Division number="130" book="ja1">

<Paragraph number="3388" type="verses" book="ja1">
_Te tattha kanditvā roditvāna nāgā,_  
_Sīse sake paṃsukaṃ okiritvā;_  
_Agamaṃsu te sabbe sakaṃ niketaṃ,_  
_Purakkhatvā mahesiṃ sabbabhaddaṃ._  

</Paragraph>

</Division>
<Division number="131" book="ja1">

<Paragraph number="3389" type="verses" book="ja1">
_Ādāya dantāni gajuttamassa,_  
_Vaggū subhe appaṭime pathabyā;_  
_Suvaṇṇarājīhi samantamodare,_  
_So luddako kāsipuraṃ upāgami;_  
_Upanesi so rājakaññāya dante,_  
_Nāgo hato handa imassa dantā._  

</Paragraph>

</Division>
<Division number="132" book="ja1">

<Paragraph number="3390" type="verses" book="ja1">
_Disvāna dantāni gajuttamassa,_  
_Bhattuppiyassa purimāya jātiyā;_  
_Tattheva tassā hadayaṃ aphāli,_  
_Teneva sā kālamakāsi bālā._  

</Paragraph>

</Division>
<Division number="133" book="ja1">

<Paragraph number="3391" type="verses" book="ja1">
_Sambodhipatto sa mahānubhāvo,_  
_Sitaṃ akāsi parisāya majjhe;_  
_Pucchiṃsu bhikkhū suvimuttacittā,_  
_“Nākāraṇe pātukaronti buddhā._  

</Paragraph>

</Division>
<Division number="134" book="ja1">

<Paragraph number="3392" type="verses" book="ja1">
_Yamaddasātha dahariṃ kumāriṃ,_  
_Kāsāyavatthaṃ anagāriyaṃ carantiṃ;_  
_Sā kho tadā rājakaññā ahosi,_  
_Ahaṃ tadā nāgarājā ahosiṃ._  

</Paragraph>

</Division>
<Division number="135" book="ja1">

<Paragraph number="3393" type="verses" book="ja1">
_Ādāya dantāni gajuttamassa,_  
_Vaggū subhe appaṭime pathabyā;_  
_Yo luddako kāsipuraṃ upāgami,_  
_So kho tadā devadatto ahosi”._  

</Paragraph>

</Division>
<Division number="136" book="ja1">

<Paragraph number="3394" type="verses" book="ja1">
_Anāvasūraṃ cirarattasaṃsitaṃ,_  
_Uccāvacaṃ caritamidaṃ purāṇaṃ;_  
_Vītaddaro vītasoko visallo,_  
_Sayaṃ abhiññāya abhāsi buddho._  

</Paragraph>

</Division>
<Division number="137" book="ja1">

<Paragraph number="3395" type="verses" book="ja1">
_“Ahaṃ vo tena kālena,_  
_ahosiṃ tattha bhikkhavo;_  
_Nāgarājā tadā homi,_  
_evaṃ dhāretha jātakan”ti._  


</Paragraph>
</Division>
