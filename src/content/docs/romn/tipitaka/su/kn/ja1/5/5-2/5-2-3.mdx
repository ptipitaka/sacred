<Paragraph number="1229" type="verses" book="ja1">
_“Hiriṃ tarantaṃ vijigucchamānaṃ,_  
_Tavāhamasmī iti bhāsamānaṃ;_  
_Seyyāni kammāni anādiyantaṃ,_  
_Neso mamanti iti naṃ vijaññā._  

</Paragraph>

</Division>
<Division number="71" book="ja1">

<Paragraph number="1230" type="verses" book="ja1">
_Yañhi kayirā tañhi vade,_  
_yaṃ na kayirā na taṃ vade;_  
_Akarontaṃ bhāsamānaṃ,_  
_parijānanti paṇḍitā._  

</Paragraph>

</Division>
<Division number="72" book="ja1">

<Paragraph number="1231" type="verses" book="ja1">
_Na so mitto yo sadā appamatto,_  
_Bhedāsaṅkī randhamevānupassī;_  
_Yasmiñca setī urasīva putto,_  
_Sa ve mitto yo abhejjo parehi._  

</Paragraph>

</Division>
<Division number="73" book="ja1">

<Paragraph number="1232" type="verses" book="ja1">
_Pāmojjakaraṇaṃ ṭhānaṃ,_  
_pasaṃsāvahanaṃ sukhaṃ;_  
_Phalānisaṃso bhāveti,_  
_vahanto porisaṃ dhuraṃ._  

</Paragraph>

</Division>
<Division number="74" book="ja1">

<Paragraph number="1233" type="verses" book="ja1">
_Pavivekarasaṃ pitvā,_  
_rasaṃ upasamassa ca;_  
_Niddaro hoti nippāpo,_  
_dhammappītirasaṃ pivan”ti._  


</Paragraph>
</Division>
