<Paragraph number="1397" type="verses" book="ja1">
_“Sace brāhmaṇa gacchesi,_  
_sākete ajjunaṃ vanaṃ;_  
_Vajjāsi nandiyaṃ nāma,_  
_puttaṃ asmākamorasaṃ;_  
_Mātā pitā ca te vuddhā,_  
_te taṃ icchanti passituṃ”._  

</Paragraph>

</Division>
<Division number="72" book="ja1" bookNo="22" page="265">

<Paragraph number="1398" type="verses" book="ja1">
_“Bhuttā mayā nivāpāni,_  
_Rājino pānabhojanaṃ;_  
_Taṃ rājapiṇḍaṃ avabhottuṃ,_  
_Nāhaṃ brāhmaṇa mussahe”._  

</Paragraph>

</Division>
<Division number="73" book="ja1" bookNo="22" page="265">

<Paragraph number="1399" type="verses" book="ja1">
_“Odahissāmahaṃ passaṃ,_  
_khurappānissa rājino;_  
_Tadāhaṃ sukhito mutto,_  
_api passeyya mātaraṃ._  

</Paragraph>

</Division>
<Division number="74" book="ja1" bookNo="22" page="265">

<Paragraph number="1400" type="verses" book="ja1">
_Migarājā pure āsiṃ,_  
_kosalassa niketane;_  
_Nandiyo nāma nāmena,_  
_abhirūpo catuppado._  

</Paragraph>

</Division>
<Division number="75" book="ja1" bookNo="22" page="265">

<Paragraph number="1401" type="verses" book="ja1">
_Taṃ maṃ vadhitumāgacchi,_  
_dāyasmiṃ ajjune vane;_  
_Dhanuṃ advejjhaṃ katvāna,_  
_usuṃ sannayha kosalo._  

</Paragraph>

</Division>
<Division number="76" book="ja1" bookNo="22" page="265">

<Paragraph number="1402" type="verses" book="ja1">
_Tassāhaṃ odahiṃ passaṃ,_  
_khurappānissa rājino;_  
_Tadāhaṃ sukhito mutto,_  
_mātaraṃ daṭṭhumāgato”ti._  


</Paragraph>
</Division>

<Paragraph number="1403" type="center" book="ja1">
Nandiyamigarājajātakaṃ dasamaṃ.

  
Avāriyavaggo paṭhamo.



</Paragraph>

<Paragraph number="1404" type="center" book="ja1">
Tassuddānaṃ


</Paragraph>

_Sadarīmukha neru latā ca puna;_  
_Apananda sirī ca sucittavaro,_  
_Atha dhammika nandimigena dasāti._
</Paragraph>
