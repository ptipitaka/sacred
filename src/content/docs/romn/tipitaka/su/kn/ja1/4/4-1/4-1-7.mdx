<Paragraph number="913" type="verses" book="ja1">
_“Acetanaṃ brāhmaṇa assuṇantaṃ,_  
_Jāno ajānantamimaṃ palāsaṃ;_  
_Āraddhaviriyo dhuvaṃ appamatto,_  
_Sukhaseyyaṃ pucchasi kissa hetu”._  

</Paragraph>

</Division>
<Division number="26" book="ja1">

<Paragraph number="914" type="verses" book="ja1">
_“Dūre suto ceva brahā ca rukkho,_  
_Dese ṭhito bhūtanivāsarūpo;_  
_Tasmā namassāmi imaṃ palāsaṃ,_  
_Ye cettha bhūtā te dhanassa hetu”._  

</Paragraph>

</Division>
<Division number="27" book="ja1">

<Paragraph number="915" type="verses" book="ja1">
_“So te karissāmi yathānubhāvaṃ,_  
_Kataññutaṃ brāhmaṇa pekkhamāno;_  
_Kathañhi āgamma sataṃ sakāse,_  
_Moghāni te assu pariphanditāni._  

</Paragraph>

</Division>
<Division number="28" book="ja1">

<Paragraph number="916" type="verses" book="ja1">
_Yo tindukarukkhassa paro pilakkho,_  
_Parivārito pubbayañño uḷāro;_  
_Tassesa mūlasmiṃ nidhi nikhāto,_  
_Adāyādo gaccha taṃ uddharāhī”ti._  


</Paragraph>
</Division>

<Paragraph number="917" type="center" book="ja1">
Palāsajātakaṃ sattamaṃ.
</Paragraph>
