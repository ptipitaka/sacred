<Paragraph number="1597" type="verses" book="ja1">
_“Ahañce daḷhadhammassa,_  
_vahantī nābhirādhayiṃ;_  
_Dharantī urasi sallaṃ,_  
_yuddhe vikkantacārinī._  

</Paragraph>

</Division>
<Division number="99" book="ja1" bookNo="22" page="322">

<Paragraph number="1598" type="verses" book="ja1">
_Nūna rājā na jānāti,_  
_mama vikkamaporisaṃ;_  
_Saṅgāme sukatantāni,_  
_dūtavippahitāni ca._  

</Paragraph>

</Division>
<Division number="100" book="ja1" bookNo="22" page="322">

<Paragraph number="1599" type="verses" book="ja1">
_Sā nūnāhaṃ marissāmi,_  
_abandhu aparāyinī;_  
_Tadā hi kumbhakārassa,_  
_dinnā chakaṇahārikā”._  

</Paragraph>

</Division>
<Division number="101" book="ja1" bookNo="22" page="323">

<Paragraph number="1600" type="verses" book="ja1">
_“Yāvatāsīsatī poso,_  
_tāvadeva pavīṇati;_  
_Atthāpāye jahanti naṃ,_  
_oṭṭhibyādhiṃva khattiyo._  

</Paragraph>

</Division>
<Division number="102" book="ja1" bookNo="22" page="323">

<Paragraph number="1601" type="verses" book="ja1">
_Yo pubbe katakalyāṇo,_  
_katattho nāvabujjhati;_  
_Atthā tassa palujjanti,_  
_ye honti abhipatthitā._  

</Paragraph>

</Division>
<Division number="103" book="ja1" bookNo="22" page="323">

<Paragraph number="1602" type="verses" book="ja1">
_Yo pubbe katakalyāṇo,_  
_katattho manubujjhati;_  
_Atthā tassa pavaḍḍhanti,_  
_ye honti abhipatthitā._  

</Paragraph>

</Division>
<Division number="104" book="ja1" bookNo="22" page="323">

<Paragraph number="1603" type="verses" book="ja1">
_Taṃ vo vadāmi bhaddante,_  
_yāvantettha samāgatā;_  
_Sabbe kataññuno hotha,_  
_ciraṃ saggamhi ṭhassathā”ti._  


</Paragraph>
</Division>

<Paragraph number="1604" type="center" book="ja1">
Daḷhadhammajātakaṃ catutthaṃ.
