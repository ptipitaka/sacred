<Paragraph number="1101" type="verses" book="ja1">
_“Avasimha tavāgāre,_  
_niccaṃ sakkatapūjitā;_  
_Tvameva dānimakari,_  
_handa rāja vajāmahaṃ”._  

</Paragraph>

<Division number="170" book="ja1" bookNo="22" page="379">

<Paragraph number="1102" type="verses" book="ja1">
_“Yo ve kate paṭikate,_  
_kibbise paṭikibbise;_  
_Evaṃ taṃ sammatī veraṃ,_  
_vasa kuntini māgamā”._  

</Paragraph>

</Division>
<Division number="171" book="ja1" bookNo="22" page="379">

<Paragraph number="1103" type="verses" book="ja1">
_“Na katassa ca kattā ca,_  
_metti sandhīyate puna;_  
_Hadayaṃ nānujānāti,_  
_gacchaññeva rathesabha”._  

</Paragraph>

</Division>
<Division number="172" book="ja1" bookNo="22" page="380">

<Paragraph number="1104" type="verses" book="ja1">
_“Katassa ceva kattā ca,_  
_metti sandhīyate puna;_  
_Dhīrānaṃ no ca bālānaṃ,_  
_vasa kuntini māgamā”ti._  


</Paragraph>
</Division>

<Paragraph number="1105" type="center" book="ja1">
Kuntinījātakaṃ tatiyaṃ.
</Paragraph>
