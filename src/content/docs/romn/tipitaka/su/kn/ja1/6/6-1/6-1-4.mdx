<Division number="20" book="ja1">

<Paragraph number="1340" type="verses" book="ja1">
_“Kākolā kākasaṅghā ca,_  
_mayañca patataṃ varā;_  
_Sabbeva sadisā homa,_  
_imaṃ āgamma pabbataṃ._  

</Paragraph>

</Division>
<Division number="21" book="ja1">

<Paragraph number="1341" type="verses" book="ja1">
_Idha sīhā ca byagghā ca,_  
_siṅgālā ca migādhamā;_  
_Sabbeva sadisā honti,_  
_ayaṃ ko nāma pabbato”._  

</Paragraph>

</Division>
<Division number="22" book="ja1">

<Paragraph number="1342" type="verses" book="ja1">
_“Imaṃ nerūti jānanti,_  
_manussā pabbatuttamaṃ;_  
_Idha vaṇṇena sampannā,_  
_vasanti sabbapāṇino”._  

</Paragraph>

</Division>
<Division number="23" book="ja1">

<Paragraph number="1343" type="verses" book="ja1">
_“Amānanā yattha siyā,_  
_santānaṃ vā vimānanā;_  
_Hīnasammānanā vāpi,_  
_na tattha vasatiṃ vase._  

</Paragraph>

</Division>
<Division number="24" book="ja1">

<Paragraph number="1344" type="verses" book="ja1">
_Yatthālaso ca dakkho ca,_  
_sūro bhīru ca pūjiyā;_  
_Na tattha santo vasanti,_  
_avisesakare nare._  

</Paragraph>

</Division>
<Division number="25" book="ja1">

<Paragraph number="1345" type="verses" book="ja1">
_Nāyaṃ neru vibhajati,_  
_hīnaukkaṭṭhamajjhime;_  
_Avisesakaro neru,_  
_handa neruṃ jahāmase”ti._  


</Paragraph>
</Division>

<Paragraph number="1346" type="center" book="ja1">
Nerujātakaṃ catutthaṃ.
</Paragraph>
