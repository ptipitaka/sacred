<Division number="8" book="ja1">

<Paragraph number="1493" type="verses" book="ja1">
_“Yathā cāpo ninnamati,_  
_jiyā cāpi nikūjati;_  
_Haññate nūna manojo,_  
_migarājā sakhā mama._  

</Paragraph>

</Division>
<Division number="9" book="ja1">

<Paragraph number="1494" type="verses" book="ja1">
_Handa dāni vanantāni,_  
_pakkamāmi yathāsukhaṃ;_  
_Netādisā sakhā honti,_  
_labbhā me jīvato sakhā”._  

</Paragraph>

</Division>
<Division number="10" book="ja1">

<Paragraph number="1495" type="verses" book="ja1">
_“Na pāpajanasaṃsevī,_  
_accantaṃ sukhamedhati;_  
_Manojaṃ passa semānaṃ,_  
_giriyassānusāsanī”._  

</Paragraph>

</Division>
<Division number="11" book="ja1">

<Paragraph number="1496" type="verses" book="ja1">
_“Na pāpasampavaṅkena,_  
_mātā puttena nandati;_  
_Manojaṃ passa semānaṃ,_  
_acchannaṃ samhi lohite”._  

</Paragraph>

</Division>
<Division number="12" book="ja1">

<Paragraph number="1497" type="verses" book="ja1">
_“Evamāpajjate poso,_  
_pāpiyo ca nigacchati;_  
_Yo ve hitānaṃ vacanaṃ,_  
_na karoti atthadassinaṃ”._  

</Paragraph>

</Division>
<Division number="13" book="ja1">

<Paragraph number="1498" type="verses" book="ja1">
_“Evañca so hoti tato ca pāpiyo,_  
_Yo uttamo adhamajanūpasevī;_  
_Passuttamaṃ adhamajanūpasevitaṃ,_  
_Migādhipaṃ saravaraveganiddhutaṃ”._  

</Paragraph>

</Division>
<Division number="14" book="ja1">

<Paragraph number="1499" type="verses" book="ja1">
_“Nihīyati puriso nihīnasevī,_  
_Na ca hāyetha kadāci tulyasevī;_  
_Seṭṭhamupagamaṃ udeti khippaṃ,_  
_Tasmāttanā uttaritaraṃ bhajethā”ti._  


</Paragraph>
</Division>

<Paragraph number="1500" type="center" book="ja1">
Manojajātakaṃ dutiyaṃ.
</Paragraph>
