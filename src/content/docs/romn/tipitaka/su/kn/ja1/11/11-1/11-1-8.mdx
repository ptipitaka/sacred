<Paragraph number="2210" type="verses" book="ja1">
_“Jānanto no mahārāja,_  
_tava sīlaṃ janādhipo;_  
_Ime kumāre pūjento,_  
_na taṃ kenaci maññatha._  

</Paragraph>

</Division>
<Division number="98" book="ja1" bookNo="22" page="85">

<Paragraph number="2211" type="verses" book="ja1">
_Tiṭṭhante no mahārāje,_  
_adu deve divaṅgate;_  
_Ñātī taṃ samanuññiṃsu,_  
_sampassaṃ atthamattano._  

</Paragraph>

</Division>
<Division number="99" book="ja1" bookNo="22" page="85">

<Paragraph number="2212" type="verses" book="ja1">
_Kena saṃvara vattena,_  
_sañjāte abhitiṭṭhasi;_  
_Kena taṃ nātivattanti,_  
_ñātisaṅghā samāgatā”._  

</Paragraph>

</Division>
<Division number="100" book="ja1" bookNo="22" page="85">

<Paragraph number="2213" type="verses" book="ja1">
_“Na rājaputta usūyāmi,_  
_samaṇānaṃ mahesinaṃ;_  
_Sakkaccaṃ te namassāmi,_  
_pāde vandāmi tādinaṃ._  

</Paragraph>

</Division>
<Division number="101" book="ja1" bookNo="22" page="85">

<Paragraph number="2214" type="verses" book="ja1">
_Te maṃ dhammaguṇe yuttaṃ,_  
_sussūsamanusūyakaṃ;_  
_Samaṇā manusāsanti,_  
_isī dhammaguṇe ratā._  

</Paragraph>

</Division>
<Division number="102" book="ja1" bookNo="22" page="85">

<Paragraph number="2215" type="verses" book="ja1">
_Tesāhaṃ vacanaṃ sutvā,_  
_samaṇānaṃ mahesinaṃ;_  
_Na kiñci atimaññāmi,_  
_dhamme me nirato mano._  

</Paragraph>

</Division>
<Division number="103" book="ja1" bookNo="22" page="85">

<Paragraph number="2216" type="verses" book="ja1">
_Hatthārohā anīkaṭṭhā,_  
_rathikā pattikārakā;_  
_Tesaṃ nappaṭibandhāmi,_  
_niviṭṭhaṃ bhattavetanaṃ._  

</Paragraph>

</Division>
<Division number="104" book="ja1" bookNo="22" page="85">

<Paragraph number="2217" type="verses" book="ja1">
_Mahāmattā ca me atthi,_  
_mantino paricārakā;_  
_Bārāṇasiṃ voharanti,_  
_bahumaṃsasurodakaṃ._  

</Paragraph>

</Division>
<Division number="105" book="ja1" bookNo="22" page="85">

<Paragraph number="2218" type="verses" book="ja1">
_Athopi vāṇijā phītā,_  
_nānāraṭṭhehi āgatā;_  
_Tesu me vihitā rakkhā,_  
_evaṃ jānāhuposatha”._  

</Paragraph>

</Division>
<Division number="106" book="ja1" bookNo="22" page="86">

<Paragraph number="2219" type="verses" book="ja1">
_“Dhammena kira ñātīnaṃ,_  
_rajjaṃ kārehi saṃvara;_  
_Medhāvī paṇḍito cāsi,_  
_athopi ñātinaṃ hito._  

</Paragraph>

</Division>
<Division number="107" book="ja1" bookNo="22" page="86">

<Paragraph number="2220" type="verses" book="ja1">
_Taṃ taṃ ñātiparibyūḷhaṃ,_  
_nānāratanamocitaṃ;_  
_Amittā nappasahanti,_  
_indaṃva asurādhipo”ti._  


</Paragraph>
</Division>

<Paragraph number="2221" type="center" book="ja1">
Saṃvarajātakaṃ aṭṭhamaṃ.
</Paragraph>
