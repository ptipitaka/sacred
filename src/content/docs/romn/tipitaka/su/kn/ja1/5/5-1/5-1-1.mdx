<Division number="1" book="ja1">

<Paragraph number="1146" type="verses" book="ja1">
_“Jīno rathassaṃ maṇikuṇḍale ca,_  
_Putte ca dāre ca tatheva jīno;_  
_Sabbesu bhogesu asesakesu,_  
_Kasmā na santappasi sokakāle”._  

</Paragraph>

</Division>
<Division number="2" book="ja1">

<Paragraph number="1147" type="verses" book="ja1">
_“Pubbeva maccaṃ vijahanti bhogā,_  
_Macco vā te pubbataraṃ jahāti;_  
_Asassatā bhogino kāmakāmi,_  
_Tasmā na socāmahaṃ sokakāle._  

</Paragraph>

</Division>
<Division number="3" book="ja1">

<Paragraph number="1148" type="verses" book="ja1">
_Udeti āpūrati veti cando,_  
_Atthaṃ tapetvāna paleti sūriyo;_  
_Viditā mayā sattuka lokadhammā,_  
_Tasmā na socāmahaṃ sokakāle”._  

</Paragraph>

</Division>
<Division number="4" book="ja1">

<Paragraph number="1149" type="verses" book="ja1">
_“Alaso gihī kāmabhogī na sādhu,_  
_Asaññato pabbajito na sādhu;_  
_Rājā na sādhu anisammakārī,_  
_Yo paṇḍito kodhano taṃ na sādhu._  

</Paragraph>

</Division>
<Division number="5" book="ja1">

<Paragraph number="1150" type="verses" book="ja1">
_Nisamma khattiyo kayirā,_  
_nānisamma disampati;_  
_Nisammakārino rāja,_  
_yaso kitti ca vaḍḍhatī”ti._  


</Paragraph>
</Division>

<Paragraph number="1151" type="center" book="ja1">
Maṇikuṇḍalajātakaṃ paṭhamaṃ.
</Paragraph>
