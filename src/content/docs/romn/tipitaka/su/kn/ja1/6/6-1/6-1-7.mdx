<Division number="40" book="ja1">

<Paragraph number="1363" type="verses" book="ja1">
_“Kā nu kāḷena vaṇṇena,_  
_na cāpi piyadassanā;_  
_Kā vā tvaṃ kassa vā dhītā,_  
_kathaṃ jānemu taṃ mayaṃ”._  

</Paragraph>

</Division>
<Division number="41" book="ja1">

<Paragraph number="1364" type="verses" book="ja1">
_“Mahārājassahaṃ dhītā,_  
_virūpakkhassa caṇḍiyā;_  
_Ahaṃ kāḷī alakkhikā,_  
_kāḷakaṇṇīti maṃ vidū;_  
_Okāsaṃ yācito dehi,_  
_vasemu tava santike”._  

</Paragraph>

</Division>
<Division number="42" book="ja1">

<Paragraph number="1365" type="verses" book="ja1">
_“Kiṃsīle kiṃsamācāre,_  
_Purise nivisase tuvaṃ;_  
_Puṭṭhā me kāḷi akkhāhi,_  
_Kathaṃ jānemu taṃ mayaṃ”._  

</Paragraph>

</Division>
<Division number="43" book="ja1">

<Paragraph number="1366" type="verses" book="ja1">
_“Makkhī paḷāsī sārambhī,_  
_issukī maccharī saṭho;_  
_So mayhaṃ puriso kanto,_  
_laddhaṃ yassa vinassati._  

</Paragraph>

</Division>
<Division number="44" book="ja1">

<Paragraph number="1367" type="verses" book="ja1">
_Kodhano upanāhī ca,_  
_pisuṇo ca vibhedako;_  
_Kaṇḍakavāco pharuso,_  
_so me kantataro tato._  

</Paragraph>

</Division>
<Division number="45" book="ja1">

<Paragraph number="1368" type="verses" book="ja1">
_Ajja suveti puriso,_  
_sadatthaṃ nāvabujjhati;_  
_Ovajjamāno kuppati,_  
_seyyaṃ so atimaññati._  

</Paragraph>

</Division>
<Division number="46" book="ja1">

<Paragraph number="1369" type="verses" book="ja1">
_Davappaluddho puriso,_  
_sabbamittehi dhaṃsati;_  
_So mayhaṃ puriso kanto,_  
_tasmiṃ homi anāmayā”._  

</Paragraph>

</Division>
<Division number="47" book="ja1">

<Paragraph number="1370" type="verses" book="ja1">
_“Apehi etto tvaṃ kāḷi,_  
_netaṃ amhesu vijjati;_  
_Aññaṃ janapadaṃ gaccha,_  
_nigame rājadhāniyo”._  

</Paragraph>

</Division>
<Division number="48" book="ja1">

<Paragraph number="1371" type="verses" book="ja1">
_“Ahampi kho taṃ jānāmi,_  
_netaṃ tumhesu vijjati;_  
_Santi loke alakkhikā,_  
_saṃgharanti bahuṃ dhanaṃ;_  
_Ahaṃ devo ca me bhātā,_  
_ubho naṃ vidhamāmase”._  

</Paragraph>

</Division>
<Division number="49" book="ja1">

<Paragraph number="1372" type="verses" book="ja1">
_“Kā nu dibbena vaṇṇena,_  
_pathabyā supatiṭṭhitā;_  
_Kā vā tvaṃ kassa vā dhītā,_  
_kathaṃ jānemu taṃ mayaṃ”._  

</Paragraph>

</Division>
<Division number="50" book="ja1">

<Paragraph number="1373" type="verses" book="ja1">
_“Mahārājassahaṃ dhītā,_  
_dhataraṭṭhassa sirīmato;_  
_Ahaṃ sirī ca lakkhī ca,_  
_bhūripaññāti maṃ vidū;_  
_Okāsaṃ yācito dehi,_  
_vasemu tava santike”._  

</Paragraph>

</Division>
<Division number="51" book="ja1">

<Paragraph number="1374" type="verses" book="ja1">
_“Kiṃsīle kiṃsamācāre,_  
_Purise nivisase tuvaṃ;_  
_Puṭṭhā me lakkhi akkhāhi,_  
_Kathaṃ jānemu taṃ mayaṃ”._  

</Paragraph>

</Division>
<Division number="52" book="ja1">

<Paragraph number="1375" type="verses" book="ja1">
_“Yo cāpi sīte atha vāpi uṇhe,_  
_Vātātape ḍaṃsasarīsape ca;_  
_Khudhaṃ pipāsaṃ abhibhuyya sabbaṃ,_  
_Rattindivaṃ yo satataṃ niyutto._  

</Paragraph>

</Division>
<Division number="53" book="ja1">

<Paragraph number="1376" type="verses" book="ja1">
_Kālāgatañca na hāpeti atthaṃ,_  
_So me manāpo nivise ca tamhi;_  
_Akkodhano mittavā cāgavā ca,_  
_Sīlūpapanno asaṭhojubhūto._  

</Paragraph>

</Division>
<Division number="54" book="ja1">

<Paragraph number="1377" type="verses" book="ja1">
_Saṅgāhako sakhilo saṇhavāco,_  
_Mahattapattopi nivātavutti;_  
_Tasmiṃhaṃ pose vipulā bhavāmi,_  
_Ūmi samuddassa yathāpi vaṇṇaṃ._  

</Paragraph>

</Division>
<Division number="55" book="ja1">

<Paragraph number="1378" type="verses" book="ja1">
_Yo cāpi mitte atha vā amitte,_  
_Seṭṭhe sarikkhe atha vāpi hīne;_  
_Atthaṃ carantaṃ atha vā anatthaṃ,_  
_Āvī raho saṅgahameva vatte._  

</Paragraph>

</Division>
<Division number="56" book="ja1">

<Paragraph number="1379" type="verses" book="ja1">
_Vācaṃ na vajjā pharusaṃ kadāci,_  
_Matassa jīvassa ca tassa homi;_  
_Etesaṃ yo aññataraṃ labhitvā,_  
_Kantā sirī majjati appapañño;_  
_Taṃ dittarūpaṃ visamaṃ carantaṃ,_  
_Karīsaṭhānaṃva vivajjayāmi._  

</Paragraph>

</Division>
<Division number="57" book="ja1">

<Paragraph number="1380" type="verses" book="ja1">
_Attanā kurute lakkhiṃ,_  
_alakkhiṃ kurutattanā;_  
_Na hi lakkhiṃ alakkhiṃ vā,_  
_añño aññassa kārako”ti._  


</Paragraph>
</Division>

<Paragraph number="1381" type="center" book="ja1">
Sirikāḷakaṇṇijātakaṃ sattamaṃ.
</Paragraph>
