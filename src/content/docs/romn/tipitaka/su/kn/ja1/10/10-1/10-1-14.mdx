<Paragraph number="2066" type="verses" book="ja1">
_“Saccaṃ kira tvaṃ api bhūripañña,_  
_Yā tādisī sīri dhitī matī ca;_  
_Na tāyatebhāvavasūpanitaṃ,_  
_Yo yavakaṃ bhuñjasi appasūpaṃ”._  

</Paragraph>

</Division>
<Division number="146" book="ja1" bookNo="22" page="61">

<Paragraph number="2067" type="verses" book="ja1">
_“Sukhaṃ dukkhena paripācayanto,_  
_Kālā kālaṃ vicinaṃ chandachanno;_  
_Atthassa dvārāni avāpuranto,_  
_Tenāhaṃ tussāmi yavodanena._  

</Paragraph>

</Division>
<Division number="147" book="ja1" bookNo="22" page="61">

<Paragraph number="2068" type="verses" book="ja1">
_Kālañca ñatvā abhijīhanāya,_  
_Mantehi atthaṃ paripācayitvā;_  
_Vijambhissaṃ sīhavijambhitāni,_  
_Tāyiddhiyā dakkhasi maṃ punāpi”._  

</Paragraph>

</Division>
<Division number="148" book="ja1" bookNo="22" page="61">

<Paragraph number="2069" type="verses" book="ja1">
_“Sukhīpi heke na karonti pāpaṃ,_  
_Avaṇṇasaṃsaggabhayā puneke;_  
_Pahū samāno vipulatthacintī,_  
_Kiṃkāraṇā me na karosi dukkhaṃ”._  

</Paragraph>

</Division>
<Division number="149" book="ja1" bookNo="22" page="61">

<Paragraph number="2070" type="verses" book="ja1">
_“Na paṇḍitā attasukhassa hetu,_  
_Pāpāni kammāni samācaranti;_  
_Dukkhena phuṭṭhā khalitāpi santā,_  
_Chandā ca dosā na jahanti dhammaṃ”._  

</Paragraph>

</Division>
<Division number="150" book="ja1" bookNo="22" page="61">

<Paragraph number="2071" type="verses" book="ja1">
_“Yena kenaci vaṇṇena,_  
_mudunā dāruṇena vā;_  
_Uddhare dīnamattānaṃ,_  
_pacchā dhammaṃ samācare”._  

</Paragraph>

</Division>
<Division number="151" book="ja1" bookNo="22" page="61">

<Paragraph number="2072" type="verses" book="ja1">
_“Yassa rukkhassa chāyāya,_  
_nisīdeyya sayeyya vā;_  
_Na tassa sākhaṃ bhañjeyya,_  
_mittadubbho hi pāpako._  

</Paragraph>

</Division>
<Division number="152" book="ja1" bookNo="22" page="61">

<Paragraph number="2073" type="verses" book="ja1">
_Yassāpi dhammaṃ puriso vijaññā,_  
_Ye cassa kaṅkhaṃ vinayanti santo;_  
_Taṃ hissa dīpañca parāyanañca,_  
_Na tena mettiṃ jarayetha pañño”._  

</Paragraph>

</Division>
<Division number="153" book="ja1" bookNo="22" page="62">

<Paragraph number="2074" type="verses" book="ja1">
_“Alaso gihī kāmabhogī na sādhu,_  
_Asaññato pabbajito na sādhu;_  
_Rājā na sādhu anisammakārī,_  
_Yo paṇḍito kodhano taṃ na sādhu._  

</Paragraph>

</Division>
<Division number="154" book="ja1" bookNo="22" page="62">

<Paragraph number="2075" type="verses" book="ja1">
_Nisamma khattiyo kayirā,_  
_nānisamma disampati;_  
_Nisammakārino rāja,_  
_yaso kitti ca vaḍḍhatī”ti._  


</Paragraph>
</Division>

<Paragraph number="2076" type="center" book="ja1">
Bhūripaññajātakaṃ cuddasamaṃ.
</Paragraph>
