<Division number="92" book="ja1">

<Paragraph number="2008" type="verses" book="ja1">
_“Kiṃ te vataṃ kiṃ pana brahmacariyaṃ,_  
_Kissa suciṇṇassa ayaṃ vipāko;_  
_Akkhāhi me brāhmaṇa etamatthaṃ,_  
_Kasmā nu tumhaṃ daharā na miyyare”._  

</Paragraph>

</Division>
<Division number="93" book="ja1">

<Paragraph number="2009" type="verses" book="ja1">
_“Dhammaṃ carāma na musā bhaṇāma,_  
_Pāpāni kammāni parivajjayāma;_  
_Anariyaṃ parivajjemu sabbaṃ,_  
_Tasmā hi amhaṃ daharā na miyyare._  

</Paragraph>

</Division>
<Division number="94" book="ja1">

<Paragraph number="2010" type="verses" book="ja1">
_Suṇoma dhammaṃ asataṃ satañca,_  
_Na cāpi dhammaṃ asataṃ rocayāma;_  
_Hitvā asante na jahāma sante,_  
_Tasmā hi amhaṃ daharā na miyyare._  

</Paragraph>

</Division>
<Division number="95" book="ja1">

<Paragraph number="2011" type="verses" book="ja1">
_Pubbeva dānā sumanā bhavāma,_  
_Dadampi ve attamanā bhavāma;_  
_Datvāpi ve nānutappāma pacchā,_  
_Tasmā hi amhaṃ daharā na miyyare._  

</Paragraph>

</Division>
<Division number="96" book="ja1">

<Paragraph number="2012" type="verses" book="ja1">
_Samaṇe mayaṃ brāhmaṇe addhike ca,_  
_Vanibbake yācanake dalidde;_  
_Annena pānena abhitappayāma,_  
_Tasmā hi amhaṃ daharā na miyyare._  

</Paragraph>

</Division>
<Division number="97" book="ja1">

<Paragraph number="2013" type="verses" book="ja1">
_Mayañca bhariyaṃ nātikkamāma,_  
_Amhe ca bhariyā nātikkamanti;_  
_Aññatra tāhi brahmacariyaṃ carāma,_  
_Tasmā hi amhaṃ daharā na miyyare._  

</Paragraph>

</Division>
<Division number="98" book="ja1">

<Paragraph number="2014" type="verses" book="ja1">
_Pāṇātipātā viramāma sabbe,_  
_Loke adinnaṃ parivajjayāma;_  
_Amajjapā nopi musā bhaṇāma,_  
_Tasmā hi amhaṃ daharā na miyyare._  

</Paragraph>

</Division>
<Division number="99" book="ja1">

<Paragraph number="2015" type="verses" book="ja1">
_Etāsu ve jāyare suttamāsu,_  
_Medhāvino honti pahūtapaññā;_  
_Bahussutā vedaguno ca honti,_  
_Tasmā hi amhaṃ daharā na miyyare._  

</Paragraph>

</Division>
<Division number="100" book="ja1">

<Paragraph number="2016" type="verses" book="ja1">
_Mātā pitā ca bhaginī bhātaro ca,_  
_Puttā ca dārā ca mayañca sabbe;_  
_Dhammaṃ carāma paralokahetu,_  
_Tasmā hi amhaṃ daharā na miyyare._  

</Paragraph>

</Division>
<Division number="101" book="ja1">

<Paragraph number="2017" type="verses" book="ja1">
_Dāsā ca dāsyo anujīvino ca,_  
_Paricārakā kammakarā ca sabbe;_  
_Dhammaṃ caranti paralokahetu,_  
_Tasmā hi amhaṃ daharā na miyyare”._  

</Paragraph>

</Division>
<Division number="102" book="ja1">

<Paragraph number="2018" type="verses" book="ja1">
_“Dhammo have rakkhati dhammacāriṃ,_  
_Dhammo suciṇṇo sukhamāvahāti;_  
_Esānisaṃso dhamme suciṇṇe,_  
_Na duggatiṃ gacchati dhammacārī._  

</Paragraph>

</Division>
<Division number="103" book="ja1">

<Paragraph number="2019" type="verses" book="ja1">
_Dhammo have rakkhati dhammacāriṃ,_  
_Chattaṃ mahantaṃ viya vassakāle;_  
_Dhammena gutto mama dhammapālo,_  
_Aññassa aṭṭhīni sukhī kumāro”ti._  


</Paragraph>
</Division>

<Paragraph number="2020" type="center" book="ja1">
Mahādhammapālajātakaṃ navamaṃ.
</Paragraph>
