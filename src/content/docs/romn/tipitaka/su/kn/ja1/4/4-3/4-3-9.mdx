<Paragraph number="1027" type="verses" book="ja1">
_“Yaṃ annapānassa pure labhāma,_  
_Taṃ dāni sākhamigameva gacchati;_  
_Gacchāma dāni vanameva rādha,_  
_Asakkatā casma dhanañjayāya”._  

</Paragraph>

</Division>
<Division number="114" book="ja1">

<Paragraph number="1028" type="verses" book="ja1">
_“Lābho alābho yaso ayaso ca,_  
_Nindā pasaṃsā ca sukhañca dukkhaṃ;_  
_Ete aniccā manujesu dhammā,_  
_Mā soci kiṃ socasi poṭṭhapāda”._  

</Paragraph>

</Division>
<Division number="115" book="ja1">

<Paragraph number="1029" type="verses" book="ja1">
_“Addhā tuvaṃ paṇḍitakosi rādha,_  
_Jānāsi atthāni anāgatāni;_  
_Kathaṃ nu sākhāmigaṃ dakkhisāma,_  
_Niddhāvitaṃ rājakulatova jammaṃ”._  

</Paragraph>

</Division>
<Division number="116" book="ja1">

<Paragraph number="1030" type="verses" book="ja1">
_“Cāleti kaṇṇaṃ bhakuṭiṃ karoti,_  
_Muhuṃ muhuṃ bhāyayate kumāre;_  
_Sayameva taṃ kāhati kāḷabāhu,_  
_Yenārakā ṭhassati annapānā”ti._  


</Paragraph>
</Division>

<Paragraph number="1031" type="center" book="ja1">
Kāḷabāhujātakaṃ navamaṃ.
</Paragraph>
