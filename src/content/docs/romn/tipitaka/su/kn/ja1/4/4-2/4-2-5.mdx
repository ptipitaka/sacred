<Paragraph number="955" type="verses" book="ja1">
_“Pharusā vata te vācā,_  
_maṃsaṃ yācanako asi;_  
_Kilomasadisī vācā,_  
_kilomaṃ samma dammi te”._  

</Paragraph>

</Division>
<Division number="58" book="ja1" bookNo="22" page="248">

<Paragraph number="956" type="verses" book="ja1">
_“Aṅgametaṃ manussānaṃ,_  
_bhātā loke pavuccati;_  
_Aṅgassa sadisī vācā,_  
_aṅgaṃ samma dadāmi te”._  

</Paragraph>

</Division>
<Division number="59" book="ja1" bookNo="22" page="234">

<Paragraph number="957" type="verses" book="ja1">
_“Tātāti putto vadamāno,_  
_kampeti hadayaṃ pitu;_  
_Hadayassa sadisī vācā,_  
_hadayaṃ samma dammi te”._  

</Paragraph>

</Division>
<Division number="60" book="ja1" bookNo="22" page="234">

<Paragraph number="958" type="verses" book="ja1">
_“Yassa gāme sakhā natthi,_  
_yathāraññaṃ tatheva taṃ;_  
_Sabbassa sadisī vācā,_  
_sabbaṃ samma dadāmi te”ti._  


</Paragraph>
</Division>

Sabbamaṃsalābhajātakaṃ pañcamaṃ.
</Paragraph>
