<Paragraph number="940" type="verses" book="ja1">
_“Api kassapa mandiyā,_  
_yuvā sapati hanti vā;_  
_Sabbaṃ taṃ khamate dhīro,_  
_paṇḍito taṃ titikkhati._  

</Paragraph>

</Division>
<Division number="46" book="ja1" bookNo="22" page="232">

<Paragraph number="941" type="verses" book="ja1">
_Sacepi santo vivadanti,_  
_khippaṃ sandhīyare puna;_  
_Bālā pattāva bhijjanti,_  
_na te samathamajjhagū._  

</Paragraph>

</Division>
<Division number="47" book="ja1" bookNo="22" page="232">

<Paragraph number="942" type="verses" book="ja1">
_Ete bhiyyo samāyanti,_  
_sandhi tesaṃ na jīrati;_  
_Yo cādhipannaṃ jānāti,_  
_yo ca jānāti desanaṃ._  

</Paragraph>

</Division>
<Division number="48" book="ja1" bookNo="22" page="232">

<Paragraph number="943" type="verses" book="ja1">
_Eso hi uttaritaro,_  
_Bhāravaho dhuraddharo;_  
_Yo paresādhipannānaṃ,_  
_Sayaṃ sandhātumarahatī”ti._  


</Paragraph>
</Division>

<Paragraph number="944" type="center" book="ja1">
Kassapamandiyajātakaṃ dutiyaṃ.
</Paragraph>
