<Paragraph number="508" type="verses" book="ja1">
_“Virūpakkhehi me mettaṃ,_  
_Mettaṃ erāpathehi me;_  
_Chabyāputtehi me mettaṃ,_  
_Mettaṃ kaṇhāgotamakehi ca._  

</Paragraph>

<Paragraph number="509" type="verses" book="ja1">
_Apādakehi me mettaṃ,_  
_mettaṃ dvipādakehi me;_  
_Catuppadehi me mettaṃ,_  
_mettaṃ bahuppadehi me._  

</Paragraph>

<Paragraph number="510" type="verses" book="ja1">
_Mā maṃ apādako hiṃsi,_  
_mā maṃ hiṃsi dvipādako;_  
_Mā maṃ catuppado hiṃsi,_  
_mā maṃ hiṃsi bahuppado._  

</Paragraph>

<Paragraph number="511" type="verses" book="ja1">
_Sabbe sattā sabbe pāṇā,_  
_sabbe bhūtā ca kevalā;_  
_Sabbe bhadrāni passantu,_  
_mā kañci pāpamāgamā._  

</Paragraph>

</Division>
<Division number="106" book="ja1" bookNo="22" page="216">

<Paragraph number="512" type="verses" book="ja1">
_Appamāṇo buddho,_  
_Appamāṇo dhammo;_  
_Appamāṇo saṃgho,_  
_Pamāṇavantāni sarīsapāni;_  
_Ahivicchikasatapadī,_  
_Uṇṇanābhi sarabū mūsikā._  

</Paragraph>

<Paragraph number="513" type="verses" book="ja1">
_Katā me rakkhā katā me parittā,_  
_Paṭikkamantu bhūtāni;_  
_Sohaṃ namo bhagavato,_  
_Namo sattannaṃ sammāsambuddhānan”ti._  


</Paragraph>
</Division>

<Paragraph number="514" type="center" book="ja1">
Khaṇḍajātakaṃ tatiyaṃ.
</Paragraph>
