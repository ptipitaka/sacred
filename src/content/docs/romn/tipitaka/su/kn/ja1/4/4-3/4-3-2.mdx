<Paragraph number="992" type="verses" book="ja1">
_“Duddubhāyati bhaddante,_  
_yasmiṃ dese vasāmahaṃ;_  
_Ahampetaṃ na jānāmi,_  
_kimetaṃ duddubhāyati”._  

</Paragraph>

</Division>
<Division number="86" book="ja1" bookNo="22" page="236">

<Paragraph number="993" type="verses" book="ja1">
_“Beluvaṃ patitaṃ sutvā,_  
_duddubhanti saso javi;_  
_Sasassa vacanaṃ sutvā,_  
_santattā migavāhinī._  

</Paragraph>

</Division>
<Division number="87" book="ja1" bookNo="22" page="237">

<Paragraph number="994" type="verses" book="ja1">
_Appatvā padaviññāṇaṃ,_  
_paraghosānusārino;_  
_Panādaparamā bālā,_  
_te honti parapattiyā._  

</Paragraph>

</Division>
<Division number="88" book="ja1" bookNo="22" page="237">

<Paragraph number="995" type="verses" book="ja1">
_Ye ca sīlena sampannā,_  
_paññāyūpasame ratā;_  
_Ārakā viratā dhīrā,_  
_na honti parapattiyā”ti._  


</Paragraph>
</Division>

Duddubhajātakaṃ dutiyaṃ.
</Paragraph>
