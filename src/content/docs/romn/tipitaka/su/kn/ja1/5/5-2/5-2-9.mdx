<Division number="101" book="ja1">

<Paragraph number="1266" type="verses" book="ja1">
_“Atikkamma ramaṇakaṃ,_  
_sadāmattañca dūbhakaṃ;_  
_Brahmattarañca pāsādaṃ,_  
_kenatthena idhāgato”._  

</Paragraph>

</Division>
<Division number="102" book="ja1">

<Paragraph number="1267" type="verses" book="ja1">
_“Ito bahutarā bhogā,_  
_atra maññe bhavissare;_  
_Iti etāya saññāya,_  
_passa maṃ byasanaṃ gataṃ”._  

</Paragraph>

</Division>
<Division number="103" book="ja1">

<Paragraph number="1268" type="verses" book="ja1">
_“Catubbhi aṭṭhajjhagamā,_  
_aṭṭhāhipi ca soḷasa;_  
_Soḷasāhi ca bāttiṃsa,_  
_atricchaṃ cakkamāsado;_  
_Icchāhatassa posassa,_  
_cakkaṃ bhamati matthake._  

</Paragraph>

</Division>
<Division number="104" book="ja1">

<Paragraph number="1269" type="verses" book="ja1">
_Uparivisālā duppūrā,_  
_icchā visaṭagāminī;_  
_Ye ca taṃ anugijjhanti,_  
_te honti cakkadhārino”ti._  


</Paragraph>
</Division>
