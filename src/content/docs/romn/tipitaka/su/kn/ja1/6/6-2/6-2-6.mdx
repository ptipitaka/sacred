<Paragraph number="1443" type="verses" book="ja1">
_“Dubbaṇṇarūpaṃ tuvamariyavaṇṇī,_  
_Purakkhatvā pañjaliko namassasi;_  
_Seyyo nu te so udavā sarikkho,_  
_Nāmaṃ parassattano cāpi brūhi”._  

</Paragraph>

</Division>
<Division number="110" book="ja1" bookNo="22" page="289">

<Paragraph number="1444" type="verses" book="ja1">
_“Na nāmagottaṃ gaṇhanti rāja,_  
_Sammaggatānujjugatāna devā;_  
_Ahañca te nāmadheyyaṃ vadāmi,_  
_Sakkohamasmī tidasānamindo”._  

</Paragraph>

</Division>
<Division number="111" book="ja1" bookNo="22" page="289">

<Paragraph number="1445" type="verses" book="ja1">
_“Yo disvā bhikkhuṃ caraṇūpapannaṃ,_  
_Purakkhatvā pañjaliko namassati;_  
_Pucchāmi taṃ devarājetamatthaṃ,_  
_Ito cuto kiṃ labhate sukhaṃ so”._  

</Paragraph>

</Division>
<Division number="112" book="ja1" bookNo="22" page="289">

<Paragraph number="1446" type="verses" book="ja1">
_“Yo disvā bhikkhuṃ caraṇūpapannaṃ,_  
_Purakkhatvā pañjaliko namassati;_  
_Diṭṭheva dhamme labhate pasaṃsaṃ,_  
_Saggañca so yāti sarīrabhedā”._  

</Paragraph>

</Division>
<Division number="113" book="ja1" bookNo="22" page="289">

<Paragraph number="1447" type="verses" book="ja1">
_“Lakkhī vata me udapādi ajja,_  
_Yaṃ vāsavaṃ bhūtapatiddasāma;_  
_Bhikkhuñca disvāna tuvañca sakka,_  
_Kāhāmi puññāni anappakāni”._  

</Paragraph>

</Division>
<Division number="114" book="ja1" bookNo="22" page="324">

<Paragraph number="1448" type="verses" book="ja1">
_“Addhā have sevitabbā sapaññā,_  
_Bahussutā ye bahuṭhānacintino;_  
_Bhikkhuñca disvāna mamañca rāja,_  
_Karohi puññāni anappakāni”._  

</Paragraph>

</Division>
<Division number="115" book="ja1" bookNo="22" page="324">

<Paragraph number="1449" type="verses" book="ja1">
_“Akkodhano niccapasannacitto,_  
_Sabbātithīyācayogo bhavitvā;_  
_Nihacca mānaṃ abhivādayissaṃ,_  
_Sutvāna devinda subhāsitānī”ti._  


</Paragraph>
</Division>

<Paragraph number="1450" type="center" book="ja1">
Vijjādharajātakaṃ chaṭṭhaṃ.
</Paragraph>
