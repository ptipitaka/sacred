<Division number="58" book="ja1">

<Paragraph number="1382" type="verses" book="ja1">
_“Sucittapattachadana,_  
_tambacūḷa vihaṅgama;_  
_Oroha dumasākhāya,_  
_mudhā bhariyā bhavāmi te”._  

</Paragraph>

</Division>
<Division number="59" book="ja1">

<Paragraph number="1383" type="verses" book="ja1">
_“Catuppadī tvaṃ kalyāṇi,_  
_dvipadāhaṃ manorame;_  
_Migī pakkhī asaññuttā,_  
_aññaṃ pariyesa sāmikaṃ”._  

</Paragraph>

</Division>
<Division number="60" book="ja1">

<Paragraph number="1384" type="verses" book="ja1">
_“Komārikā te hessāmi,_  
_Mañjukā piyabhāṇinī;_  
_Vinda maṃ ariyena vedena,_  
_Sāvaya maṃ yadicchasi”._  

</Paragraph>

</Division>
<Division number="61" book="ja1">

<Paragraph number="1385" type="verses" book="ja1">
_“Kuṇapādini lohitape,_  
_cori kukkuṭapothini;_  
_Na tvaṃ ariyena vedena,_  
_mamaṃ bhattāramicchasi”._  

</Paragraph>

</Division>
<Division number="62" book="ja1">

<Paragraph number="1386" type="verses" book="ja1">
_“Evampi caturā nārī,_  
_disvāna sadhanaṃ naraṃ;_  
_Nenti saṇhāhi vācāhi,_  
_biḷārī viya kukkuṭaṃ._  

</Paragraph>

</Division>
<Division number="63" book="ja1">

<Paragraph number="1387" type="verses" book="ja1">
_Yo ca uppatitaṃ atthaṃ,_  
_na khippamanubujjhati;_  
_Amittavasamanveti,_  
_pacchā ca anutappati._  

</Paragraph>

</Division>
<Division number="64" book="ja1">

<Paragraph number="1388" type="verses" book="ja1">
_Yo ca uppatitaṃ atthaṃ,_  
_khippameva nibodhati;_  
_Muccate sattusambādhā,_  
_kukkuṭova biḷāriyā”ti._  


</Paragraph>
</Division>

<Paragraph number="1389" type="center" book="ja1">
Kukkuṭajātakaṃ aṭṭhamaṃ.
</Paragraph>
