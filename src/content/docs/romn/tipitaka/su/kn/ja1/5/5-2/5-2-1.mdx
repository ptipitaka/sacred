<Division number="62" book="ja1">

<Paragraph number="1219" type="verses" book="ja1">
_Evañce maṃ viharantaṃ,_  
_subāhu samma dubbhasi;_  
_Na dānāhaṃ tayā saddhiṃ,_  
_saṃvāsamabhirocaye._  

</Paragraph>

</Division>
<Division number="63" book="ja1">

<Paragraph number="1220" type="verses" book="ja1">
_Yo paresaṃ vacanāni,_  
_saddaheyya yathātathaṃ;_  
_Khippaṃ bhijjetha mittasmiṃ,_  
_verañca pasave bahuṃ._  

</Paragraph>

</Division>
<Division number="64" book="ja1">

<Paragraph number="1221" type="verses" book="ja1">

_Bhedāsaṅkī randhamevānupassī;_  
_Yasmiñca setī urasīva putto,_  
_Sa ve mitto yo abhejjo parehī”ti._  


</Paragraph>
