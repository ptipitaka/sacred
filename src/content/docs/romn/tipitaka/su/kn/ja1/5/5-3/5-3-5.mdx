<Paragraph number="1306" type="verses" book="ja1">
_“Idāni khomhi sukhito arogo,_  
_Nikkaṇṭako nippatito kapoto;_  
_Kāhāmi dānī hadayassa tuṭṭhiṃ,_  
_Tathāhimaṃ maṃsasākaṃ baleti”._  

</Paragraph>

<Division number="136" book="ja1">

<Paragraph number="1307" type="verses" book="ja1">
_“Kāyaṃ balākā sikhinī,_  
_corī laṅghipitāmahā;_  
_Oraṃ balāke āgaccha,_  
_caṇḍo me vāyaso sakhā”._  

</Paragraph>

</Division>
<Division number="137" book="ja1">

<Paragraph number="1308" type="verses" book="ja1">
_“Alañhi te jagghitāye,_  
_Mamaṃ disvāna edisaṃ;_  
_Vilūnaṃ sūdaputtena,_  
_Piṭṭhamaṇḍena makkhitaṃ”._  

</Paragraph>

</Division>
<Division number="138" book="ja1">

<Paragraph number="1309" type="verses" book="ja1">
_“Sunhāto suvilittosi,_  
_annapānena tappito;_  
_Kaṇṭhe ca te veḷuriyo,_  
_agamā nu kajaṅgalaṃ”._  

</Paragraph>

</Division>
<Division number="139" book="ja1">

<Paragraph number="1310" type="verses" book="ja1">
_“Mā te mitto amitto vā,_  
_agamāsi kajaṅgalaṃ;_  
_Piñchāni tattha lāyitvā,_  
_kaṇṭhe bandhanti vaṭṭanaṃ”._  

</Paragraph>

</Division>
<Division number="140" book="ja1">

<Paragraph number="1311" type="verses" book="ja1">
_“Punapāpajjasī samma,_  
_Sīlañhi tava tādisaṃ;_  
_Na hi mānusakā bhogā,_  
_Subhuñjā honti pakkhinā”ti. (841)_  


</Paragraph>
</Division>

<Paragraph number="1312" type="center" book="ja1">
Kapotajātakaṃ pañcamaṃ.

  
Aḍḍhavaggo tatiyo.



</Paragraph>

<Paragraph number="1313" type="center" book="ja1">
Tassuddānaṃ


</Paragraph>

<Paragraph number="1314" type="verses" book="ja1">
_Atha vaṇṇa sasīla hiri labhate,_  
_Sumukhā visa sāḷiyamittavaro;_  
_Atha cakka palāsa sarāja sato,_  
_Yava bāla kapotaka pannarasāti._  


</Paragraph>

<Paragraph number="1315" type="center" book="ja1">
Atha vagguddānaṃ


</Paragraph>

<Paragraph number="1316" type="verses" book="ja1">
_Jīnañca vaṇṇaṃ asamaṃvaguppari,_  
_Sudesitā jātakanti santi vīsati;_  
_Mahesino brahmacarittamutta-_  
_Mavoca gāthā atthavatī subyañjanāti._  


</Paragraph>
