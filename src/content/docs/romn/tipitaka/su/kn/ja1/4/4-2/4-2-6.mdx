<Division number="61" book="ja1">

<Paragraph number="960" type="verses" book="ja1">
_“Satta me rohitā macchā,_  
_udakā thalamubbhatā;_  
_Idaṃ brāhmaṇa me atthi,_  
_etaṃ bhutvā vane vasa”._  

</Paragraph>

</Division>
<Division number="62" book="ja1">

<Paragraph number="961" type="verses" book="ja1">
_“Dussa me khettapālassa,_  
_rattibhattaṃ apābhataṃ;_  
_Maṃsasūlā ca dve godhā,_  
_ekañca dadhivārakaṃ;_  
_Idaṃ brāhmaṇa me atthi,_  
_etaṃ bhutvā vane vasa”._  

</Paragraph>

</Division>
<Division number="63" book="ja1">

<Paragraph number="962" type="verses" book="ja1">
_“Ambapakkaṃ dakaṃ sītaṃ,_  
_Sītacchāyā manoramā;_  
_Idaṃ brāhmaṇa me atthi,_  
_Etaṃ bhutvā vane vasa”._  

</Paragraph>

</Division>
<Division number="64" book="ja1">

<Paragraph number="963" type="verses" book="ja1">
_“Na sasassa tilā atthi,_  
_Na muggā napi taṇḍulā;_  
_Iminā agginā pakkaṃ,_  
_Mamaṃ bhutvā vane vasā”ti._  


</Paragraph>
</Division>

<Paragraph number="964" type="center" book="ja1">
Sasapaṇḍitajātakaṃ chaṭṭhaṃ.
