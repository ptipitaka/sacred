<Paragraph number="2759" type="verses" book="ja1">
_“Abbhuto vata lokasmiṃ,_  
_uppajji lomahaṃsano;_  
_Dibbo ratho pāturahu,_  
_vedehassa yasassino”._  

</Paragraph>

</Division>
<Division number="203" book="ja1" bookNo="22" page="300">

<Paragraph number="2760" type="verses" book="ja1">
_Devaputto mahiddhiko,_  
_mātali devasārathi;_  
_Nimantayittha rājānaṃ,_  
_vedehaṃ mithilaggahaṃ._  

</Paragraph>

</Division>
<Division number="204" book="ja1" bookNo="22" page="300">

<Paragraph number="2761" type="verses" book="ja1">
_“Ehimaṃ rathamāruyha,_  
_rājaseṭṭha disampati;_  
_Devā dassanakāmā te,_  
_tāvatiṃsā saindakā;_  
_Saramānā hi te devā,_  
_sudhammāyaṃ samacchare”._  

</Paragraph>

</Division>
<Division number="205" book="ja1" bookNo="22" page="300">

<Paragraph number="2762" type="verses" book="ja1">
_Tato ca rājā sādhino,_  
_vedeho mithilaggaho;_  
_Sahassayuttamāruyha,_  
_agā devāna santike;_  
_Taṃ devā paṭinandiṃsu,_  
_disvā rājānamāgataṃ._  

</Paragraph>

</Division>
<Division number="206" book="ja1" bookNo="22" page="300">

<Paragraph number="2763" type="verses" book="ja1">
_“Svāgataṃ te mahārāja,_  
_atho te adurāgataṃ;_  
_Nisīda dāni rājīsi,_  
_devarājassa santike”._  

</Paragraph>

</Division>
<Division number="207" book="ja1" bookNo="22" page="300">

<Paragraph number="2764" type="verses" book="ja1">
_Sakkopi paṭinandittha,_  
_vedehaṃ mithilaggahaṃ;_  
_Nimantayittha kāmehi,_  
_āsanena ca vāsavo._  

</Paragraph>

</Division>
<Division number="208" book="ja1" bookNo="22" page="301">

<Paragraph number="2765" type="verses" book="ja1">
_“Sādhu khosi anuppatto,_  
_āvāsaṃ vasavattinaṃ;_  
_Vasa devesu rājīsi,_  
_sabbakāmasamiddhisu;_  
_Tāvatiṃsesu devesu,_  
_bhuñja kāme amānuse”._  

</Paragraph>

</Division>
<Division number="209" book="ja1" bookNo="22" page="301">

<Paragraph number="2766" type="verses" book="ja1">
_“Ahaṃ pure saggagato ramāmi,_  
_Naccehi gītehi ca vāditehi;_  
_So dāni ajja na ramāmi sagge,_  
_Āyuṃ nu khīṇo maraṇaṃ nu santike;_  
_Udāhu mūḷhosmi janindaseṭṭha”._  

</Paragraph>

</Division>
<Division number="210" book="ja1" bookNo="22" page="301">

<Paragraph number="2767" type="verses" book="ja1">
_“Na tāyu khīṇaṃ maraṇañca dūre,_  
_Na cāpi mūḷho naravīraseṭṭha;_  
_Tuyhañca puññāni parittakāni,_  
_Yesaṃ vipākaṃ idha vedayittho._  

</Paragraph>

</Division>
<Division number="211" book="ja1" bookNo="22" page="301">

<Paragraph number="2768" type="verses" book="ja1">
_Vasa devānubhāvena,_  
_rājaseṭṭha disampati;_  
_Tāvatiṃsesu devesu,_  
_bhuñja kāme amānuse”._  

</Paragraph>

</Division>
<Division number="212" book="ja1" bookNo="22" page="301">

<Paragraph number="2769" type="verses" book="ja1">
_“Yathā yācitakaṃ yānaṃ,_  
_yathā yācitakaṃ dhanaṃ;_  
_Evaṃsampadamevetaṃ,_  
_yaṃ parato dānapaccayā._  

</Paragraph>

</Division>
<Division number="213" book="ja1" bookNo="22" page="301">

<Paragraph number="2770" type="verses" book="ja1">
_Na cāhametamicchāmi,_  
_yaṃ parato dānapaccayā;_  
_Sayaṃkatāni puññāni,_  
_taṃ me āveṇikaṃ dhanaṃ._  

</Paragraph>

</Division>
<Division number="214" book="ja1" bookNo="22" page="301">

<Paragraph number="2771" type="verses" book="ja1">
_Sohaṃ gantvā manussesu,_  
_kāhāmi kusalaṃ bahuṃ;_  
_Dānena samacariyāya,_  
_saṃyamena damena ca;_  
_Yaṃ katvā sukhito hoti,_  
_na ca pacchānutappati._  

</Paragraph>

</Division>
<Division number="215" book="ja1" bookNo="22" page="301">

<Paragraph number="2772" type="verses" book="ja1">
_Imāni tāni khettāni,_  
_imaṃ nikkhaṃ sukuṇḍalaṃ;_  
_Imā tā haritānūpā,_  
_imā najjo savantiyo._  

</Paragraph>

</Division>
<Division number="216" book="ja1" bookNo="22" page="301">

<Paragraph number="2773" type="verses" book="ja1">
_Imā tā pokkharaṇī rammā,_  
_cakkavākapakūjitā;_  
_Mandālakehi sañchannā,_  
_padumuppalakehi ca;_  
_Yassimāni mamāyiṃsu,_  
_kiṃ nu te disataṃ gatā._  

</Paragraph>

</Division>
<Division number="217" book="ja1" bookNo="22" page="302">

<Paragraph number="2774" type="verses" book="ja1">
_Tānīdha khettāni so bhūmibhāgo,_  
_Teyeva ārāmavanūpacārā;_  
_Tameva mayhaṃ janataṃ apassato,_  
_Suññaṃva me nārada khāyate disā”._  

</Paragraph>

</Division>
<Division number="218" book="ja1" bookNo="22" page="302">

<Paragraph number="2775" type="verses" book="ja1">
_“Diṭṭhā mayā vimānāni,_  
_obhāsentā catuddisā;_  
_Sammukhā devarājassa,_  
_tidasānañca sammukhā._  

</Paragraph>

</Division>
<Division number="219" book="ja1" bookNo="22" page="302">

<Paragraph number="2776" type="verses" book="ja1">
_Vutthaṃ me bhavanaṃ dibyaṃ,_  
_Bhuttā kāmā amānusā;_  
_Tāvatiṃsesu devesu,_  
_Sabbakāmasamiddhisu._  

</Paragraph>

</Division>
<Division number="220" book="ja1" bookNo="22" page="302">

<Paragraph number="2777" type="verses" book="ja1">
_Sohaṃ etādisaṃ hitvā,_  
_puññāyamhi idhāgato;_  
_Dhammameva carissāmi,_  
_nāhaṃ rajjena atthiko._  

</Paragraph>

</Division>
<Division number="221" book="ja1" bookNo="22" page="302">

<Paragraph number="2778" type="verses" book="ja1">
_Adaṇḍāvacaraṃ maggaṃ,_  
_sammāsambuddhadesitaṃ;_  
_Taṃ maggaṃ paṭipajjissaṃ,_  
_yena gacchanti subbatā”ti._  


</Paragraph>
</Division>

<Paragraph number="2779" type="center" book="ja1">
Sādhinajātakaṃ ekādasamaṃ.
</Paragraph>
