<Paragraph number="2680" type="verses" book="ja1">
_“Appossukko dāni tuvaṃ kapota,_  
_Vihaṅgama na tava bhojanattho;_  
_Khudaṃ pipāsaṃ adhivāsayanto,_  
_Kasmā bhavaṃposathiko kapota”._  

</Paragraph>

</Division>
<Division number="128" book="ja1" bookNo="22" page="135">

<Paragraph number="2681" type="verses" book="ja1">
_“Ahaṃ pure giddhigato kapotiyā,_  
_Asmiṃ padesasmimubho ramāma;_  
_Athaggahī sākuṇiko kapotiṃ,_  
_Akāmako tāya vinā ahosiṃ._  

</Paragraph>

</Division>
<Division number="129" book="ja1" bookNo="22" page="135">

<Paragraph number="2682" type="verses" book="ja1">
_Nānābhavā vippayogena tassā,_  
_Manomayaṃ vedana vedayāmi;_  
_Tasmā ahaṃposathaṃ pālayāmi,_  
_Rāgo mamaṃ mā punarāgamāsi”. (1)_  

</Paragraph>

</Division>
<Division number="130" book="ja1" bookNo="22" page="135">

<Paragraph number="2683" type="verses" book="ja1">
_“Anujjugāmī uragā dujivha,_  
_Dāṭhāvudho ghoravisosi sappa;_  
_Khudaṃ pipāsaṃ adhivāsayanto,_  
_Kasmā bhavaṃposathiko nu dīgha._  

</Paragraph>

</Division>
<Division number="131" book="ja1" bookNo="22" page="135">

<Paragraph number="2684" type="verses" book="ja1">
_Usabho ahū balavā gāmikassa,_  
_Calakkakū vaṇṇabalūpapanno;_  
_So maṃ akkami taṃ kupito aḍaṃsiṃ,_  
_Dukkhābhitunno maraṇaṃ upāgā._  

</Paragraph>

</Division>
<Division number="132" book="ja1" bookNo="22" page="135">

<Paragraph number="2685" type="verses" book="ja1">
_Tato janā nikkhamitvāna gāmā,_  
_Kanditvā roditvā apakkamiṃsu;_  
_Tasmā ahaṃposathaṃ pālayāmi,_  
_Kodho mamaṃ mā punarāgamāsi. (2)_  

</Paragraph>

</Division>
<Division number="133" book="ja1" bookNo="22" page="110">

<Paragraph number="2686" type="verses" book="ja1">
_Matāna maṃsāni bahū susāne,_  
_Manuññarūpaṃ tava bhojane taṃ;_  
_Khudaṃ pipāsaṃ adhivāsayanto,_  
_Kasmā bhavaṃposathiko siṅgāla._  

</Paragraph>

</Division>
<Division number="134" book="ja1" bookNo="22" page="110">

<Paragraph number="2687" type="verses" book="ja1">
_Pavisi kucchiṃ mahato gajassa,_  
_Kuṇape rato hatthimaṃsesu giddho;_  
_Uṇho ca vāto tikhiṇā ca rasmiyo,_  
_Te sosayuṃ tassa karīsamaggaṃ._  

</Paragraph>

</Division>
<Division number="135" book="ja1" bookNo="22" page="110">

<Paragraph number="2688" type="verses" book="ja1">
_Kiso ca paṇḍū ca ahaṃ bhadante,_  
_Na me ahū nikkhamanāya maggo;_  
_Mahā ca megho sahasā pavassi,_  
_So temayī tassa karīsamaggaṃ._  

</Paragraph>

</Division>
<Division number="136" book="ja1" bookNo="22" page="111">

<Paragraph number="2689" type="verses" book="ja1">
_Tato ahaṃ nikkhamisaṃ bhadante,_  
_Cando yathā rāhumukhā pamutto;_  
_Tasmā ahaṃposathaṃ pālayāmi,_  
_Lobho mamaṃ mā punarāgamāsi. (3)_  

</Paragraph>

</Division>
<Division number="137" book="ja1" bookNo="22" page="111">

<Paragraph number="2690" type="verses" book="ja1">
_Vammīkathūpasmiṃ kipillikāni,_  
_Nippothayanto tuvaṃ pure carāsi;_  
_Khudaṃ pipāsaṃ adhivāsayanto,_  
_Kasmā bhavaṃposathiko nu accha._  

</Paragraph>

</Division>
<Division number="138" book="ja1" bookNo="22" page="111">

<Paragraph number="2691" type="verses" book="ja1">
_Sakaṃ niketaṃ atihīḷayāno,_  
_Atricchatā mallagāmaṃ agacchiṃ;_  
_Tato janā nikkhamitvāna gāmā,_  
_Kodaṇḍakena paripothayiṃsu maṃ._  

</Paragraph>

</Division>
<Division number="139" book="ja1" bookNo="22" page="111">

<Paragraph number="2692" type="verses" book="ja1">
_So bhinnasīso ruhiramakkhitaṅgo,_  
_Paccāgamāsiṃ sakaṃ niketaṃ;_  
_Tasmā ahaṃposathaṃ pālayāmi,_  
_Atricchatā mā punarāgamāsi”. (4)_  

</Paragraph>

</Division>
<Division number="140" book="ja1" bookNo="22" page="111">

<Paragraph number="2693" type="verses" book="ja1">
_“Yaṃ no apucchittha tuvaṃ bhadante,_  
_Sabbeva byākarimha yathāpajānaṃ;_  
_Mayampi pucchāma tuvaṃ bhadante,_  
_Kasmā bhavaṃposathiko nu brahme”._  

</Paragraph>

</Division>
<Division number="141" book="ja1" bookNo="22" page="111">

<Paragraph number="2694" type="verses" book="ja1">
_“Anūpalitto mama assamamhi,_  
_Paccekabuddho muhuttaṃ nisīdi;_  
_So maṃ avedī gatimāgatiñca,_  
_Nāmañca gottaṃ caraṇañca sabbaṃ._  

</Paragraph>

</Division>
<Division number="142" book="ja1" bookNo="22" page="111">

<Paragraph number="2695" type="verses" book="ja1">
_Evampahaṃ vandi na tassa pāde,_  
_Na cāpi naṃ mānagatena pucchiṃ;_  
_Tasmā ahaṃposathaṃ pālayāmi,_  
_Māno mamaṃ mā punarāgamāsī”ti. (5)_  


</Paragraph>
</Division>

<Paragraph number="2696" type="center" book="ja1">
Pañcuposathikajātakaṃ sattamaṃ.
</Paragraph>
