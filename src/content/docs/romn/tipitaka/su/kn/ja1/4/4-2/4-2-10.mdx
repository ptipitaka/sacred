<Paragraph number="980" type="verses" book="ja1">
_“Succajaṃ vata naccaji,_  
_vācāya adadaṃ giriṃ;_  
_Kiṃ hitassa cajantassa,_  
_vācāya adada pabbataṃ”._  

</Paragraph>

<Division number="78" book="ja1">

<Paragraph number="981" type="verses" book="ja1">
_“Yañhi kayirā tañhi vade,_  
_yaṃ na kayirā na taṃ vade;_  
_Akarontaṃ bhāsamānaṃ,_  
_parijānanti paṇḍitā”._  

</Paragraph>

</Division>
<Division number="79" book="ja1">

<Paragraph number="982" type="verses" book="ja1">
_“Rājaputta namo tyatthu,_  
_sacce dhamme ṭhito casi;_  
_Yassa te byasanaṃ patto,_  
_saccasmiṃ ramate mano”._  

</Paragraph>

</Division>
<Division number="80" book="ja1">

<Paragraph number="983" type="verses" book="ja1">
_“Yā daliddī daliddassa,_  
_aḍḍhā aḍḍhassa kittima;_  
_Sā hissa paramā bhariyā,_  
_sahiraññassa itthiyo”ti._  


</Paragraph>
</Division>

<Paragraph number="984" type="center" book="ja1">
Succajajātakaṃ dasamaṃ.

  
Pucimandavaggo dutiyo.



</Paragraph>

<Paragraph number="985" type="center" book="ja1">
Tassuddānaṃ

<Paragraph number="986" type="verses" book="ja1">
_Atha cora sakassapa khantīvaro,_  
_Dujjīvitatā ca varā pharusā;_  
_Atha sasa matañca vasanta sukhaṃ,_  
_Succajaṃvatanaccajinā ca dasāti._
</Paragraph>
