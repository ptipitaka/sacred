<Division number="112" book="ja1">

<Paragraph number="2968" type="verses" book="ja1">
_“Ayaṃ so luddako eti,_  
_luddarūpo sahāvudho;_  
_Yo no vadhissati ajja,_  
_usunā sattiyā api”._  

</Paragraph>

</Division>
<Division number="113" book="ja1">

<Paragraph number="2969" type="verses" book="ja1">
_“Sā muhuttaṃ palāyitvā,_  
_bhayaṭṭā bhayatajjitā;_  
_Sudukkaraṃ akarā bhīru,_  
_maraṇāyūpanivattatha”._  

</Paragraph>

</Division>
<Division number="114" book="ja1">

<Paragraph number="2970" type="verses" book="ja1">
_“Kiṃ nu teme migā honti,_  
_muttā baddhaṃ upāsare;_  
_Na taṃ cajitumicchanti,_  
_jīvitassapi kāraṇā”._  

</Paragraph>

</Division>
<Division number="115" book="ja1">

<Paragraph number="2971" type="verses" book="ja1">
_“Bhātaro honti me ludda,_  
_sodariyā ekamātukā;_  
_Na maṃ cajitumicchanti,_  
_jīvitassapi kāraṇā”._  

</Paragraph>

</Division>
<Division number="116" book="ja1">

<Paragraph number="2972" type="verses" book="ja1">
_“Te hi nūna marissanti,_  
_andhā apariṇāyakā;_  
_Pañcannaṃ jīvitaṃ dehi,_  
_bhātaraṃ muñca luddaka”._  

</Paragraph>

</Division>
<Division number="117" book="ja1">

<Paragraph number="2973" type="verses" book="ja1">
_“So vo ahaṃ pamokkhāmi,_  
_mātāpettibharaṃ migaṃ;_  
_Nandantu mātāpitaro,_  
_muttaṃ disvā mahāmigaṃ”._  

</Paragraph>

</Division>
<Division number="118" book="ja1">

<Paragraph number="2974" type="verses" book="ja1">
_“Evaṃ luddaka nandassu,_  
_saha sabbehi ñātibhi;_  
_Yathāhamajja nandāmi,_  
_muttaṃ disvā mahāmigaṃ”._  

</Paragraph>

</Division>
<Division number="119" book="ja1">

<Paragraph number="2975" type="verses" book="ja1">
_“Kathaṃ tvaṃ pamokkho āsi,_  
_Upanītasmi jīvite;_  
_Kathaṃ putta amocesi,_  
_Kūṭapāsamha luddako”._  

</Paragraph>

</Division>
<Division number="120" book="ja1">

<Paragraph number="2976" type="verses" book="ja1">
_“Bhaṇaṃ kaṇṇasukhaṃ vācaṃ,_  
_hadayaṅgaṃ hadayassitaṃ;_  
_Subhāsitāhi vācāhi,_  
_cittako maṃ amocayi._  

</Paragraph>

</Division>
<Division number="121" book="ja1">

<Paragraph number="2977" type="verses" book="ja1">
_Bhaṇaṃ kaṇṇasukhaṃ vācaṃ,_  
_hadayaṅgaṃ hadayassitaṃ;_  
_Subhāsitāhi vācāhi,_  
_sutanā maṃ amocayi._  

</Paragraph>

</Division>
<Division number="122" book="ja1">

<Paragraph number="2978" type="verses" book="ja1">
_Sutvā kaṇṇasukhaṃ vācaṃ,_  
_hadayaṅgaṃ hadayassitaṃ;_  
_Subhāsitāni sutvāna,_  
_luddako maṃ amocayi”._  

</Paragraph>

</Division>
<Division number="123" book="ja1">

<Paragraph number="2979" type="verses" book="ja1">
_“Evaṃ ānandito hotu,_  
_saha dārehi luddako;_  
_Yathā mayajja nandāma,_  
_disvā rohaṇamāgataṃ”._  

</Paragraph>

</Division>
<Division number="124" book="ja1">

<Paragraph number="2980" type="verses" book="ja1">
_“Nanu tvaṃ avaca ludda,_  
_‘migacammāni āhariṃ’;_  
_Atha kena nu vaṇṇena,_  
_migacammāni nāhari”._  

</Paragraph>

</Division>
<Division number="125" book="ja1">

<Paragraph number="2981" type="verses" book="ja1">
_“Āgamā ceva hatthatthaṃ,_  
_kūṭapāsañca so migo;_  
_Abajjhi taṃ migarājaṃ,_  
_tañca muttā upāsare._  

</Paragraph>

</Division>
<Division number="126" book="ja1">

<Paragraph number="2982" type="verses" book="ja1">
_Tassa me ahu saṃvego,_  
_abbhuto lomahaṃsano;_  
_Imañcāhaṃ migaṃ haññe,_  
_ajja hissāmi jīvitaṃ”._  

</Paragraph>

</Division>
<Division number="127" book="ja1">

<Paragraph number="2983" type="verses" book="ja1">
_“Kīdisā te migā ludda,_  
_kīdisā dhammikā migā;_  
_Kathaṃvaṇṇā kathaṃsīlā,_  
_bāḷhaṃ kho ne pasaṃsasi”._  

</Paragraph>

</Division>
<Division number="128" book="ja1">

<Paragraph number="2984" type="verses" book="ja1">
_“Odātasiṅgā sucivālā,_  
_jātarūpatacūpamā;_  
_Pādā lohitakā tesaṃ,_  
_añjitakkhā manoramā”._  

</Paragraph>

</Division>
<Division number="129" book="ja1">

<Paragraph number="2985" type="verses" book="ja1">
_“Edisā te migā deva,_  
_edisā dhammikā migā;_  
_Mātāpettibharā deva,_  
_na te so abhihārituṃ”._  

</Paragraph>

</Division>
<Division number="130" book="ja1">

<Paragraph number="2986" type="verses" book="ja1">
_“Dammi nikkhasataṃ ludda,_  
_thūlañca maṇikuṇḍalaṃ;_  
_Catussadañca pallaṅkaṃ,_  
_umāpupphasarinnibhaṃ._  

</Paragraph>

</Division>
<Division number="131" book="ja1">

<Paragraph number="2987" type="verses" book="ja1">
_Dve ca sādisiyo bhariyā,_  
_usabhañca gavaṃ sataṃ;_  
_Dhammena rajjaṃ kāressaṃ,_  
_bahukāro mesi luddaka._  

</Paragraph>

</Division>
<Division number="132" book="ja1">

<Paragraph number="2988" type="verses" book="ja1">
_Kasivāṇijjā iṇadānaṃ,_  
_uñchācariyā ca luddaka;_  
_Etena dāraṃ posehi,_  
_mā pāpaṃ akarī punāti”._  


</Paragraph>
</Division>

Rohaṇamigajātakaṃ pañcamaṃ.
