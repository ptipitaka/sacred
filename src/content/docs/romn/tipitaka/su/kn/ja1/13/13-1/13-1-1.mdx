<Division number="1" book="ja1">

<Paragraph number="2384" type="verses" book="ja1">
_“Ahāsi me ambaphalāni pubbe,_  
_Aṇūni thūlāni ca brahmacāri;_  
_Teheva mantehi na dāni tuyhaṃ,_  
_Dumapphalā pātubhavanti brahme”._  

</Paragraph>

</Division>
<Division number="2" book="ja1">

<Paragraph number="2385" type="verses" book="ja1">
_“Nakkhattayogaṃ paṭimānayāmi,_  
_Khaṇaṃ muhuttañca mante na passaṃ;_  
_Nakkhattayogañca khaṇañca laddhā,_  
_Addhāharissambaphalaṃ pahūtaṃ”._  

</Paragraph>

</Division>
<Division number="3" book="ja1">

<Paragraph number="2386" type="verses" book="ja1">
_“Nakkhattayogaṃ na pure abhāṇi,_  
_Khaṇaṃ muhuttaṃ na pure asaṃsi;_  
_Sayaṃ harī ambaphalaṃ pahūtaṃ,_  
_Vaṇṇena gandhena rasenupetaṃ._  

</Paragraph>

</Division>
<Division number="4" book="ja1">

<Paragraph number="2387" type="verses" book="ja1">
_Mantābhijappena pure hi tuyhaṃ,_  
_Dumapphalā pātubhavanti brahme;_  
_Svājja na pāresi jappampi mantaṃ,_  
_Ayaṃ so ko nāma tavajja dhammo”._  

</Paragraph>

</Division>
<Division number="5" book="ja1">

<Paragraph number="2388" type="verses" book="ja1">
_“Caṇḍālaputto mama sampadāsi,_  
_Dhammena mante pakatiñca saṃsi;_  
_Mā cassu me pucchito nāmagottaṃ,_  
_Guyhittho atthaṃ vijaheyya manto._  

</Paragraph>

</Division>
<Division number="6" book="ja1">

<Paragraph number="2389" type="verses" book="ja1">
_Sohaṃ janindena janamhi puṭṭho,_  
_Makkhābhibhūto alikaṃ abhāṇiṃ;_  
_‘Mantā ime brāhmaṇassā’ti micchā,_  
_Pahīnamanto kapaṇo rudāmi”._  

</Paragraph>

</Division>
<Division number="7" book="ja1">

<Paragraph number="2390" type="verses" book="ja1">
_“Eraṇḍā pucimandā vā,_  
_atha vā pālibhaddakā;_  
_Madhuṃ madhutthiko vinde,_  
_so hi tassa dumuttamo._  

</Paragraph>

</Division>
<Division number="8" book="ja1">

<Paragraph number="2391" type="verses" book="ja1">
_Khattiyā brāhmaṇā vessā,_  
_suddā caṇḍālapukkusā;_  
_Yamhā dhammaṃ vijāneyya,_  
_so hi tassa naruttamo._  

</Paragraph>

</Division>
<Division number="9" book="ja1">

<Paragraph number="2392" type="verses" book="ja1">
_Imassa daṇḍañca vadhañca datvā,_  
_Gale gahetvā khalayātha jammaṃ;_  
_Yo uttamatthaṃ kasirena laddhaṃ,_  
_Mānātimānena vināsayittha”._  

</Paragraph>

</Division>
<Division number="10" book="ja1">

<Paragraph number="2393" type="verses" book="ja1">
_“Yathā samaṃ maññamāno pateyya,_  
_Sobbhaṃ guhaṃ narakaṃ pūtipādaṃ;_  
_Rajjūti vā akkame kaṇhasappaṃ,_  
_Andho yathā jotimadhiṭṭhaheyya;_  
_Evampi maṃ tvaṃ khalitaṃ sapañña,_  
_Pahīnamantassa punappadāhi”._  

</Paragraph>

</Division>
<Division number="11" book="ja1">

<Paragraph number="2394" type="verses" book="ja1">
_“Dhammena mantaṃ tava sampadāsiṃ,_  
_Tuvampi dhammena pariggahesi;_  
_Pakatimpi te attamano asaṃsiṃ,_  
_Dhamme ṭhitaṃ taṃ na jaheyya manto._  

</Paragraph>

</Division>
<Division number="12" book="ja1">

<Paragraph number="2395" type="verses" book="ja1">
_Yo bāla mantaṃ kasirena laddhaṃ,_  
_Yaṃ dullabhaṃ ajja manussaloke;_  
_Kiñcāpi laddhā jīvituṃ appapañño,_  
_Vināsayī alikaṃ bhāsamāno._  

</Paragraph>

</Division>
<Division number="13" book="ja1">

<Paragraph number="2396" type="verses" book="ja1">
_Bālassa mūḷhassa akataññuno ca,_  
_Musā bhaṇantassa asaññatassa;_  
_Mante mayaṃ tādisake na dema,_  
_Kuto mantā gaccha na mayha ruccasī”ti._  


</Paragraph>
</Division>

<Paragraph number="2397" type="center" book="ja1">
Ambajātakaṃ paṭhamaṃ.
</Paragraph>
