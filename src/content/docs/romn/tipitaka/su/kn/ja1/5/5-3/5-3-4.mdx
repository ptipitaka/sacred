<Paragraph number="1298" type="verses" book="ja1">
_“Sabbaṃ bhaṇḍaṃ samādāya,_  
_pāraṃ tiṇṇosi brāhmaṇa;_  
_Paccāgaccha lahuṃ khippaṃ,_  
_mampi tārehi dānito”._  

</Paragraph>

</Division>
<Division number="129" book="ja1" bookNo="22" page="291">

<Paragraph number="1299" type="verses" book="ja1">
_“Asanthutaṃ maṃ cirasanthutena,_  
_Nimīni bhotī addhuvaṃ dhuvena;_  
_Mayāpi bhotī nimineyya aññaṃ,_  
_Ito ahaṃ dūrataraṃ gamissaṃ”._  

</Paragraph>

</Division>
<Division number="130" book="ja1" bookNo="22" page="291">

<Paragraph number="1300" type="verses" book="ja1">
_“Kāyaṃ eḷagalāgumbe,_  
_karoti ahuhāsiyaṃ;_  
_Nayīdha naccagītaṃ vā,_  
_tāḷaṃ vā susamāhitaṃ;_  
_Anamhikāle susoṇi,_  
_kiṃ nu jagghasi sobhane”._  

</Paragraph>

</Division>
<Division number="131" book="ja1" bookNo="22" page="291">

<Paragraph number="1301" type="verses" book="ja1">
_“Siṅgāla bāla dummedha,_  
_appapaññosi jambuka;_  
_Jīno macchañca pesiñca,_  
_kapaṇo viya jhāyasi”._  

</Paragraph>

</Division>
<Division number="132" book="ja1" bookNo="22" page="291">

<Paragraph number="1302" type="verses" book="ja1">
_“Sudassaṃ vajjamaññesaṃ,_  
_attano pana duddasaṃ;_  
_Jīnā patiñca jārañca,_  
_maññe tvaññeva jhāyasi”._  

</Paragraph>

</Division>
<Division number="133" book="ja1" bookNo="22" page="292">

<Paragraph number="1303" type="verses" book="ja1">
_“Evametaṃ migarāja,_  
_yathā bhāsasi jambuka;_  
_Sā nūnāhaṃ ito gantvā,_  
_bhattu hessaṃ vasānugā”._  

</Paragraph>

</Division>
<Division number="134" book="ja1" bookNo="22" page="292">

<Paragraph number="1304" type="verses" book="ja1">
_“Yo hare mattikaṃ thālaṃ,_  
_kaṃsathālampi so hare;_  
_Katañceva tayā pāpaṃ,_  
_punapevaṃ karissasī”ti._  


</Paragraph>
</Division>

<Paragraph number="1305" type="center" book="ja1">
Cūḷadhanuggahajātakaṃ catutthaṃ.
</Paragraph>
