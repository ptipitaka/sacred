<Division number="87" book="ja1">

<Paragraph number="1876" type="verses" book="ja1">
_“Kuto nu āgacchatha bho tayo janā,_  
_Svāgatā etha nisīdathāsane;_  
_Kaccittha bhonto kusalaṃ anāmayaṃ,_  
_Cirassamabbhāgamanaṃ hi vo idha”._  

</Paragraph>

</Division>
<Division number="88" book="ja1">

<Paragraph number="1877" type="verses" book="ja1">
_“Ahameva eko idha majja patto,_  
_Na cāpi me dutiyo koci vijjati;_  
_Kimeva sandhāya te bhāsitaṃ ise,_  
_‘Kuto nu āgacchatha bho tayo janā’”._  

</Paragraph>

</Division>
<Division number="89" book="ja1">

<Paragraph number="1878" type="verses" book="ja1">
_“Tuvañca eko bhariyā ca te piyā,_  
_Samuggapakkhittanikiṇṇamantare;_  
_Sā rakkhitā kucchigatāva te sadā,_  
_Vāyussa puttena sahā tahiṃ ratā”._  

</Paragraph>

</Division>
<Division number="90" book="ja1">

<Paragraph number="1879" type="verses" book="ja1">
_“Saṃviggarūpo isinā viyākato,_  
_So dānavo tattha samuggamuggili;_  
_Addakkhi bhariyaṃ suci māladhāriniṃ,_  
_Vāyussa puttena sahā tahiṃ rataṃ”._  

</Paragraph>

</Division>
<Division number="91" book="ja1">

<Paragraph number="1880" type="verses" book="ja1">
_“Sudiṭṭharūpamuggatapānuvattinā,_  
_Hīnā narā ye pamadāvasaṃ gatā;_  
_Yathā have pāṇarivettha rakkhitā,_  
_Duṭṭhā mayī aññamabhippamodayi._  

</Paragraph>

</Division>
<Division number="92" book="ja1">

<Paragraph number="1881" type="verses" book="ja1">
_Divā ca ratto ca mayā upaṭṭhitā,_  
_Tapassinā jotirivā vane vasaṃ;_  
_Sā dhammamukkamma adhammamācari,_  
_Akiriyarūpo pamadāhi santhavo._  

</Paragraph>

</Division>
<Division number="93" book="ja1">

<Paragraph number="1882" type="verses" book="ja1">
_Sarīramajjhamhi ṭhitātimaññahaṃ,_  
_Mayhaṃ ayanti asatiṃ asaññataṃ;_  
_Sā dhammamukkamma adhammamācari,_  
_Akiriyarūpo pamadāhi santhavo._  

</Paragraph>

</Division>
<Division number="94" book="ja1">

<Paragraph number="1883" type="verses" book="ja1">
_Surakkhitaṃ meti kathaṃ nu vissase,_  
_Anekacittāsu na hatthi rakkhaṇā;_  
_Etā hi pātālapapātasannibhā,_  
_Etthappamatto byasanaṃ nigaccha”ti._  

</Paragraph>

</Division>
<Division number="95" book="ja1">

<Paragraph number="1884" type="verses" book="ja1">
_“Tasmā hi te sukhino vītasokā,_  
_Ye mātugāmehi caranti nissaṭā;_  
_Etaṃ sivaṃ uttamamābhipatthayaṃ,_  
_Na mātugāmehi kareyya santhavanti”._  


</Paragraph>
</Division>

<Paragraph number="1885" type="center" book="ja1">
Samuggajātakaṃ dasamaṃ.
</Paragraph>
