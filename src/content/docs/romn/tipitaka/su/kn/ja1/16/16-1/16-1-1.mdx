<Division number="1" book="ja1">

<Paragraph number="3256" type="verses" book="ja1">
_“Kiṃchando kimadhippāyo,_  
_eko sammasi ghammani;_  
_Kiṃ patthayāno kiṃ esaṃ,_  
_kena atthena brāhmaṇa”._  

</Paragraph>

</Division>
<Division number="2" book="ja1">

<Paragraph number="3257" type="verses" book="ja1">
_“Yathā mahā vāridharo,_  
_kumbho supariṇāhavā;_  
_Tathūpamaṃ ambapakkaṃ,_  
_vaṇṇagandharasuttamaṃ._  

</Paragraph>

</Division>
<Division number="3" book="ja1">

<Paragraph number="3258" type="verses" book="ja1">
_Taṃ vuyhamānaṃ sotena,_  
_disvānāmalamajjhime;_  
_Pāṇīhi naṃ gahetvāna,_  
_agyāyatanamāhariṃ._  

</Paragraph>

</Division>
<Division number="4" book="ja1">

<Paragraph number="3259" type="verses" book="ja1">
_Tato kadalipattesu,_  
_nikkhipitvā sayaṃ ahaṃ;_  
_Satthena naṃ vikappetvā,_  
_khuppipāsaṃ ahāsi me._  

</Paragraph>

</Division>
<Division number="5" book="ja1">

<Paragraph number="3260" type="verses" book="ja1">
_Sohaṃ apetadaratho,_  
_Byantībhūto dukhakkhamo;_  
_Assādaṃ nādhigacchāmi,_  
_Phalesvaññesu kesuci._  

</Paragraph>

</Division>
<Division number="6" book="ja1">

<Paragraph number="3261" type="verses" book="ja1">
_Sosetvā nūna maraṇaṃ,_  
_taṃ mamaṃ āvahissati;_  
_Ambaṃ yassa phalaṃ sādu,_  
_madhuraggaṃ manoramaṃ;_  
_Yamuddhariṃ vuyhamānaṃ,_  
_udadhismā mahaṇṇave._  

</Paragraph>

</Division>
<Division number="7" book="ja1">

<Paragraph number="3262" type="verses" book="ja1">
_Akkhātaṃ te mayā sabbaṃ,_  
_yasmā upavasāmahaṃ;_  
_Rammaṃ pati nisinnosmi,_  
_puthulomāyutā puthu._  

</Paragraph>

</Division>
<Division number="8" book="ja1">

<Paragraph number="3263" type="verses" book="ja1">
_Tvañca kho meva akkhāhi,_  
_attānamapalāyini;_  
_Kā vā tvamasi kalyāṇi,_  
_kissa vā tvaṃ sumajjhime._  

</Paragraph>

</Division>
<Division number="9" book="ja1">

<Paragraph number="3264" type="verses" book="ja1">
_Ruppapaṭṭapalimaṭṭhīva,_  
_byagghīva girisānujā;_  
_Yā santi nāriyo devesu,_  
_devānaṃ paricārikā._  

</Paragraph>

</Division>
<Division number="10" book="ja1">

<Paragraph number="3265" type="verses" book="ja1">
_Yā ca manussalokasmiṃ,_  
_Rūpenānvāgatitthiyo;_  
_Rūpena te sadisī natthi,_  
_Devesu gandhabbamanussaloke;_  
_Puṭṭhāsi me cārupubbaṅgi,_  
_Brūhi nāmañca bandhave”._  

</Paragraph>

</Division>
<Division number="11" book="ja1">

<Paragraph number="3266" type="verses" book="ja1">
_“Yaṃ tvaṃ pati nisinnosi,_  
_rammaṃ brāhmaṇa kosikiṃ;_  
_Sāhaṃ bhusālayāvutthā,_  
_varavārivahoghasā._  

</Paragraph>

</Division>
<Division number="12" book="ja1">

<Paragraph number="3267" type="verses" book="ja1">
_Nānādumagaṇākiṇṇā,_  
_bahukā girikandarā;_  
_Mameva pamukhā honti,_  
_abhisandanti pāvuse._  

</Paragraph>

</Division>
<Division number="13" book="ja1">

<Paragraph number="3268" type="verses" book="ja1">
_Atho bahū vanatodā,_  
_nīlavārivahindharā;_  
_Bahukā nāgavittodā,_  
_abhisandanti vārinā._  

</Paragraph>

</Division>
<Division number="14" book="ja1">

<Paragraph number="3269" type="verses" book="ja1">
_Tā ambajambulabujā,_  
_nīpā tālā cudumbarā;_  
_Bahūni phalajātāni,_  
_āvahanti abhiṇhaso._  

</Paragraph>

</Division>
<Division number="15" book="ja1">

<Paragraph number="3270" type="verses" book="ja1">
_Yaṃ kiñci ubhato tīre,_  
_phalaṃ patati ambuni;_  
_Asaṃsayaṃ taṃ sotassa,_  
_phalaṃ hoti vasānugaṃ._  

</Paragraph>

</Division>
<Division number="16" book="ja1">

<Paragraph number="3271" type="verses" book="ja1">
_Etadaññāya medhāvi,_  
_puthupañña suṇohi me;_  
_Mā rocaya mabhisaṅgaṃ,_  
_paṭisedha janādhipa._  

</Paragraph>

</Division>
<Division number="17" book="ja1">

<Paragraph number="3272" type="verses" book="ja1">
_Na vāhaṃ vaḍḍhavaṃ maññe,_  
_Yaṃ tvaṃ raṭṭhābhivaḍḍhana;_  
_Āceyyamāno rājisi,_  
_Maraṇaṃ abhikaṅkhasi._  

</Paragraph>

</Division>
<Division number="18" book="ja1">

<Paragraph number="3273" type="verses" book="ja1">
_Tassa jānanti pitaro,_  
_gandhabbā ca sadevakā;_  
_Ye cāpi isayo loke,_  
_saññatattā tapassino;_  
_Asaṃsayaṃ tepi jānanti,_  
_paṭṭhabhūtā yasassino”._  

</Paragraph>

</Division>
<Division number="19" book="ja1">

<Paragraph number="3274" type="verses" book="ja1">
_“Evaṃ viditvā vidū sabbadhammaṃ,_  
_Viddhaṃsanaṃ cavanaṃ jīvitassa;_  
_Na cīyatī tassa narassa pāpaṃ,_  
_Sace na ceteti vadhāya tassa._  

</Paragraph>

</Division>
<Division number="20" book="ja1">

<Paragraph number="3275" type="verses" book="ja1">
_Isipūgasamaññāte,_  
_evaṃ lokyā viditā sati;_  
_Anariyaparisambhāse,_  
_pāpakammaṃ jigīsasi._  

</Paragraph>

</Division>
<Division number="21" book="ja1">

<Paragraph number="3276" type="verses" book="ja1">
_Sace ahaṃ marissāmi,_  
_tīre te puthusussoṇi;_  
_Asaṃsayaṃ taṃ asiloko,_  
_mayi pete āgamissati._  

</Paragraph>

</Division>
<Division number="22" book="ja1">

<Paragraph number="3277" type="verses" book="ja1">
_Tasmā hi pāpakaṃ kammaṃ,_  
_rakkhasseva sumajjhime;_  
_Mā taṃ sabbo jano pacchā,_  
_pakuṭṭhāyi mayi mate”._  

</Paragraph>

</Division>
<Division number="23" book="ja1">

<Paragraph number="3278" type="verses" book="ja1">
_“Aññātametaṃ avisayhasāhi,_  
_Attānamambañca dadāmi te taṃ;_  
_So duccaje kāmaguṇe pahāya,_  
_Santiñca dhammañca adhiṭṭhitosi._  

</Paragraph>

</Division>
<Division number="24" book="ja1">

<Paragraph number="3279" type="verses" book="ja1">
_Yo hitvā pubbasaññogaṃ,_  
_pacchā saṃyojane ṭhito;_  
_Adhammañceva carati,_  
_pāpañcassa pavaḍḍhati._  

</Paragraph>

</Division>
<Division number="25" book="ja1">

<Paragraph number="3280" type="verses" book="ja1">
_Ehi taṃ pāpayissāmi,_  
_kāmaṃ appossuko bhava;_  
_Upānayāmi sītasmiṃ,_  
_viharāhi anussuko._  

</Paragraph>

</Division>
<Division number="26" book="ja1">

<Paragraph number="3281" type="verses" book="ja1">
_Taṃ puppharasamattebhi,_  
_vakkaṅgehi arindama;_  
_Koñcā mayūrā diviyā,_  
_kolaṭṭhimadhusāḷikā;_  
_Kūjitā haṃsapūgehi,_  
_kokilettha pabodhare._  

</Paragraph>

</Division>
<Division number="27" book="ja1">

<Paragraph number="3282" type="verses" book="ja1">
_Ambettha vippasākhaggā,_  
_palālakhalasannibhā;_  
_Kosambasaḷalā nīpā,_  
_pakkatālavilambino”._  

</Paragraph>

</Division>
<Division number="28" book="ja1">

<Paragraph number="3283" type="verses" book="ja1">
_“Mālī tiriṭī kāyūrī,_  
_aṅgadī candanussado;_  
_Rattiṃ tvaṃ paricāresi,_  
_divā vedesi vedanaṃ._  

</Paragraph>

</Division>
<Division number="29" book="ja1">

<Paragraph number="3284" type="verses" book="ja1">
_Soḷasitthisahassāni,_  
_yā temā paricārikā;_  
_Evaṃ mahānubhāvosi,_  
_abbhuto lomahaṃsano._  

</Paragraph>

</Division>
<Division number="30" book="ja1">

<Paragraph number="3285" type="verses" book="ja1">
_Kiṃ kammamakarī pubbe,_  
_pāpaṃ attadukhāvahaṃ;_  
_Yaṃ karitvā manussesu,_  
_piṭṭhimaṃsāni khādasi”._  

</Paragraph>

</Division>
<Division number="31" book="ja1">

<Paragraph number="3286" type="verses" book="ja1">
_“Ajjhenāni paṭiggayha,_  
_kāmesu gadhito ahaṃ;_  
_Acariṃ dīghamaddhānaṃ,_  
_paresaṃ ahitāyahaṃ._  

</Paragraph>

</Division>
<Division number="32" book="ja1">

<Paragraph number="3287" type="verses" book="ja1">
_Yo piṭṭhimaṃsiko hoti,_  
_evaṃ ukkacca khādati;_  
_Yathāhaṃ ajja khādāmi,_  
_piṭṭhimaṃsāni attano”ti._  


</Paragraph>
</Division>

<Paragraph number="3288" type="center" book="ja1">
Kiṃchandajātakaṃ paṭhamaṃ.
</Paragraph>
