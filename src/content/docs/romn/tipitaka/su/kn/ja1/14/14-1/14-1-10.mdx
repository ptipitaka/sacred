<Division number="180" book="ja1">

<Paragraph number="2736" type="verses" book="ja1">
_Vāṇijā samitiṃ katvā,_  
_nānāraṭṭhato āgatā;_  
_Dhanāharā pakkamiṃsu,_  
_ekaṃ katvāna gāmaṇiṃ._  

</Paragraph>

</Division>
<Division number="181" book="ja1">

<Paragraph number="2737" type="verses" book="ja1">
_Te taṃ kantāramāgamma,_  
_appabhakkhaṃ anodakaṃ;_  
_Mahānigrodhamaddakkhuṃ,_  
_sītacchāyaṃ manoramaṃ._  

</Paragraph>

</Division>
<Division number="182" book="ja1">

<Paragraph number="2738" type="verses" book="ja1">
_Te ca tattha nisīditvā,_  
_tassa rukkhassa chāyayā;_  
_Vāṇijā samacintesuṃ,_  
_bālā mohena pārutā._  

</Paragraph>

</Division>
<Division number="183" book="ja1">

<Paragraph number="2739" type="verses" book="ja1">
_Allāyate ayaṃ rukkho,_  
_api vārīva sandati;_  
_Iṅghassa purimaṃ sākhaṃ,_  
_mayaṃ chindāma vāṇijā._  

</Paragraph>

</Division>
<Division number="184" book="ja1">

<Paragraph number="2740" type="verses" book="ja1">
_Sā ca chinnāva pagghari,_  
_acchaṃ vāriṃ anāvilaṃ;_  
_Te tattha nhatvā pivitvā,_  
_yāvaticchiṃsu vāṇijā._  

</Paragraph>

</Division>
<Division number="185" book="ja1">

<Paragraph number="2741" type="verses" book="ja1">
_Dutiyaṃ samacintesuṃ,_  
_bālā mohena pārutā;_  
_Iṅghassa dakkhiṇaṃ sākhaṃ,_  
_mayaṃ chindāma vāṇijā._  

</Paragraph>

</Division>
<Division number="186" book="ja1">

<Paragraph number="2742" type="verses" book="ja1">
_Sā ca chinnāva pagghari,_  
_sālimaṃsodanaṃ bahuṃ;_  
_Appodavaṇṇe kummāse,_  
_siṅgiṃ vidalasūpiyo._  

</Paragraph>

</Division>
<Division number="187" book="ja1">

<Paragraph number="2743" type="verses" book="ja1">
_Te tattha bhutvā khāditvā,_  
_yāvaticchiṃsu vāṇijā;_  
_Tatiyaṃ samacintesuṃ,_  
_bālā mohena pārutā;_  
_Iṅghassa pacchimaṃ sākhaṃ,_  
_mayaṃ chindāma vāṇijā._  

</Paragraph>

</Division>
<Division number="188" book="ja1">

<Paragraph number="2744" type="verses" book="ja1">
_Sā ca chinnāva pagghari,_  
_nāriyo samalaṅkatā;_  
_Vicitravatthābharaṇā,_  
_āmuttamaṇikuṇḍalā._  

</Paragraph>

</Division>
<Division number="189" book="ja1">

<Paragraph number="2745" type="verses" book="ja1">
_Api su vāṇijā ekā,_  
_nāriyo paṇṇavīsati;_  
_Samantā parivāriṃsu,_  
_tassa rukkhassa chāyayā._  

</Paragraph>

</Division>
<Division number="190" book="ja1">

<Paragraph number="2746" type="verses" book="ja1">
_Te tāhi paricāretvā,_  
_yāvaticchiṃsu vāṇijā;_  
_Catutthaṃ samacintesuṃ,_  
_bālā mohena pārutā;_  
_Iṅghassa uttaraṃ sākhaṃ,_  
_mayaṃ chindāma vāṇijā._  

</Paragraph>

</Division>
<Division number="191" book="ja1">

<Paragraph number="2747" type="verses" book="ja1">
_Sā ca chinnāva pagghari,_  
_muttā veḷuriyā bahū;_  
_Rajataṃ jātarūpañca,_  
_kuttiyo paṭiyāni ca._  

</Paragraph>

</Division>
<Division number="192" book="ja1">

<Paragraph number="2748" type="verses" book="ja1">
_Kāsikāni ca vatthāni,_  
_uddiyāni ca kambalā;_  
_Te tattha bhāre bandhitvā,_  
_yāvaticchiṃsu vāṇijā._  

</Paragraph>

</Division>
<Division number="193" book="ja1">

<Paragraph number="2749" type="verses" book="ja1">
_Pañcamaṃ samacintesuṃ,_  
_bālā mohena pārutā;_  
_Iṅghassa mūlaṃ chindāma,_  
_api bhiyyo labhāmase._  

</Paragraph>

</Division>
<Division number="194" book="ja1">

<Paragraph number="2750" type="verses" book="ja1">
_Athuṭṭhahi satthavāho,_  
_yācamāno katañjalī;_  
_Nigrodho kiṃ parajjhati,_  
_vāṇijā bhaddamatthu te._  

</Paragraph>

</Division>
<Division number="195" book="ja1">

<Paragraph number="2751" type="verses" book="ja1">
_Vāridā purimā sākhā,_  
_annapānañca dakkhiṇā;_  
_Nāridā pacchimā sākhā,_  
_sabbakāme ca uttarā;_  
_Nigrodho kiṃ parajjhati,_  
_vāṇijā bhaddamatthu te._  

</Paragraph>

</Division>
<Division number="196" book="ja1">

<Paragraph number="2752" type="verses" book="ja1">
_Yassa rukkhassa chāyāya,_  
_nisīdeyya sayeyya vā;_  
_Na tassa sākhaṃ bhañjeyya,_  
_mittadubbho hi pāpako._  

</Paragraph>

</Division>
<Division number="197" book="ja1">

<Paragraph number="2753" type="verses" book="ja1">
_Te ca tassānādiyitvā,_  
_ekassa vacanaṃ bahū;_  
_Nisitāhi kuṭhārīhi,_  
_mūlato naṃ upakkamuṃ._  

</Paragraph>

</Division>
<Division number="198" book="ja1">

<Paragraph number="2754" type="verses" book="ja1">
_Tato nāgā nikkhamiṃsu,_  
_sannaddhā paṇṇavīsati;_  
_Dhanuggahānaṃ tisatā,_  
_chasahassā ca vammino._  

</Paragraph>

</Division>
<Division number="199" book="ja1">

<Paragraph number="2755" type="verses" book="ja1">
_“Ete hanatha bandhatha,_  
_Mā vo muñcittha jīvitaṃ;_  
_Ṭhapetvā satthavāhaṃva,_  
_Sabbe bhasmaṃ karotha ne”._  

</Paragraph>

</Division>
<Division number="200" book="ja1">

<Paragraph number="2756" type="verses" book="ja1">
_“Tasmā hi paṇḍito poso,_  
_sampassaṃ atthamattano;_  
_Lobhassa na vasaṃ gacche,_  
_haneyyārisakaṃ manaṃ._  

</Paragraph>

</Division>
<Division number="201" book="ja1">

<Paragraph number="2757" type="verses" book="ja1">
_Evamādīnavaṃ ñatvā,_  
_taṇhā dukkhassa sambhavaṃ;_  
_Vītataṇho anādāno,_  
_sato bhikkhu paribbaje”ti._  


</Paragraph>
</Division>

<Paragraph number="2758" type="center" book="ja1">
Mahāvāṇijajātakaṃ dasamaṃ.
</Paragraph>
