<Paragraph number="1757" type="verses" book="ja1">
_“Gaṅgā kumudinī santā,_  
_saṅkhavaṇṇā ca kokilā;_  
_Jambu tālaphalaṃ dajjā,_  
_atha nūna tadā siyā._  

</Paragraph>

</Division>
<Division number="78" book="ja1" bookNo="22" page="318">

<Paragraph number="1758" type="verses" book="ja1">
_Yadā kacchapalomānaṃ,_  
_pāvāro tividho siyā;_  
_Hemantikaṃ pāvuraṇaṃ,_  
_atha nūna tadā siyā._  

</Paragraph>

</Division>
<Division number="79" book="ja1" bookNo="22" page="319">

<Paragraph number="1759" type="verses" book="ja1">
_Yadā makasapādānaṃ,_  
_aṭṭālo sukato siyā;_  
_Daḷho ca avikampī ca,_  
_atha nūna tadā siyā._  

</Paragraph>

</Division>
<Division number="80" book="ja1" bookNo="22" page="319">

<Paragraph number="1760" type="verses" book="ja1">
_Yadā sasavisāṇānaṃ,_  
_nisseṇī sukatā siyā;_  
_Saggassārohaṇatthāya,_  
_atha nūna tadā siyā._  

</Paragraph>

</Division>
<Division number="81" book="ja1" bookNo="22" page="319">

<Paragraph number="1761" type="verses" book="ja1">
_Yadā nisseṇimāruyha,_  
_candaṃ khādeyyu mūsikā;_  
_Rāhuñca paripāteyyuṃ,_  
_atha nūna tadā siyā._  

</Paragraph>

</Division>
<Division number="82" book="ja1" bookNo="22" page="319">

<Paragraph number="1762" type="verses" book="ja1">
_Yadā surāghaṭaṃ pitvā,_  
_makkhikā gaṇacāriṇī;_  
_Aṅgāre vāsaṃ kappeyyuṃ,_  
_atha nūna tadā siyā._  

</Paragraph>

</Division>
<Division number="83" book="ja1" bookNo="22" page="319">

<Paragraph number="1763" type="verses" book="ja1">
_Yadā bimboṭṭhasampanno,_  
_gadrabho sumukho siyā;_  
_Kusalo naccagītassa,_  
_atha nūna tadā siyā._  

</Paragraph>

</Division>
<Division number="84" book="ja1" bookNo="22" page="319">

<Paragraph number="1764" type="verses" book="ja1">
_Yadā kākā ulūkā ca,_  
_mantayeyyuṃ rahogatā;_  
_Aññamaññaṃ pihayeyyuṃ,_  
_atha nūna tadā siyā._  

</Paragraph>

</Division>
<Division number="85" book="ja1" bookNo="22" page="319">

<Paragraph number="1765" type="verses" book="ja1">
_Yadā muḷāla pattānaṃ,_  
_chattaṃ thirataraṃ siyā;_  
_Vassassa paṭighātāya,_  
_atha nūna tadā siyā._  

</Paragraph>

</Division>
<Division number="86" book="ja1" bookNo="22" page="320">

<Paragraph number="1766" type="verses" book="ja1">
_Yadā kulako sakuṇo,_  
_pabbataṃ gandhamādanaṃ;_  
_Tuṇḍenādāya gaccheyya,_  
_atha nūna tadā siyā._  

</Paragraph>

</Division>
<Division number="87" book="ja1" bookNo="22" page="320">

<Paragraph number="1767" type="verses" book="ja1">
_Yadā sāmuddikaṃ nāvaṃ,_  
_sayantaṃ savaṭākaraṃ;_  
_Ceṭo ādāya gaccheyya,_  
_atha nūna tadā siyā”ti._  


</Paragraph>
</Division>

<Paragraph number="1768" type="center" book="ja1">
Aṭṭhānajātakaṃ navamaṃ.
</Paragraph>
