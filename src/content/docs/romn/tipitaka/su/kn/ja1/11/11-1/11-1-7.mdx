<Division number="84" book="ja1">

<Paragraph number="2196" type="verses" book="ja1">
_“Etha lakkhaṇa sītā ca,_  
_ubho otarathodakaṃ;_  
_Evāyaṃ bharato āha,_  
_‘rājā dasaratho mato’”._  

</Paragraph>

</Division>
<Division number="85" book="ja1">

<Paragraph number="2197" type="verses" book="ja1">
_“Kena rāmappabhāvena,_  
_socitabbaṃ na socasi;_  
_Pitaraṃ kālakataṃ sutvā,_  
_na taṃ pasahate dukhaṃ”._  

</Paragraph>

</Division>
<Division number="86" book="ja1">

<Paragraph number="2198" type="verses" book="ja1">
_“Yaṃ na sakkā nipāletuṃ,_  
_posena lapataṃ bahuṃ;_  
_Sa kissa viññū medhāvī,_  
_attānamupatāpaye._  

</Paragraph>

</Division>
<Division number="87" book="ja1">

<Paragraph number="2199" type="verses" book="ja1">
_Daharā ca hi vuddhā ca,_  
_ye bālā ye ca paṇḍitā;_  
_Aḍḍhā ceva daliddā ca,_  
_sabbe maccuparāyaṇā._  

</Paragraph>

</Division>
<Division number="88" book="ja1">

<Paragraph number="2200" type="verses" book="ja1">
_Phalānamiva pakkānaṃ,_  
_niccaṃ patanato bhayaṃ;_  
_Evaṃ jātāna maccānaṃ,_  
_nicca maraṇato bhayaṃ._  

</Paragraph>

</Division>
<Division number="89" book="ja1">

<Paragraph number="2201" type="verses" book="ja1">
_Sāyameke na dissanti,_  
_pāto diṭṭhā bahujjanā;_  
_Pāto eke na dissanti,_  
_sāyaṃ diṭṭhā bahujjanā._  

</Paragraph>

</Division>
<Division number="90" book="ja1">

<Paragraph number="2202" type="verses" book="ja1">
_Paridevayamāno ce,_  
_kiñcidatthaṃ udabbahe;_  
_Sammūḷho hiṃsamattānaṃ,_  
_kayirā taṃ vicakkhaṇo._  

</Paragraph>

</Division>
<Division number="91" book="ja1">

<Paragraph number="2203" type="verses" book="ja1">
_Kiso vivaṇṇo bhavati,_  
_hiṃsamattānamattano;_  
_Na tena petā pālenti,_  
_niratthā paridevanā._  

</Paragraph>

</Division>
<Division number="92" book="ja1">

<Paragraph number="2204" type="verses" book="ja1">
_Yathā saraṇamādittaṃ,_  
_vārinā parinibbaye;_  
_Evampi dhīro sutavā,_  
_medhāvī paṇḍito naro;_  
_Khippamuppatitaṃ sokaṃ,_  
_vāto tūlaṃva dhaṃsaye._  

</Paragraph>

</Division>
<Division number="93" book="ja1">

<Paragraph number="2205" type="verses" book="ja1">
_Macco ekova acceti,_  
_ekova jāyate kule;_  
_Saṃyogaparamā tveva,_  
_sambhogā sabbapāṇinaṃ._  

</Paragraph>

</Division>
<Division number="94" book="ja1">

<Paragraph number="2206" type="verses" book="ja1">
_Tasmā hi dhīrassa bahussutassa,_  
_Sampassato lokamimaṃ parañca;_  
_Aññāya dhammaṃ hadayaṃ manañca,_  
_Sokā mahantāpi na tāpayanti._  

</Paragraph>

</Division>
<Division number="95" book="ja1">

<Paragraph number="2207" type="verses" book="ja1">
_Sohaṃ dassañca bhokkhañca,_  
_bharissāmi ca ñātake;_  
_Sesañca pālayissāmi,_  
_kiccametaṃ vijānato._  

</Paragraph>

</Division>
<Division number="96" book="ja1">

<Paragraph number="2208" type="verses" book="ja1">
_Dasa vassasahassāni,_  
_saṭṭhi vassasatāni ca;_  
_Kambugīvo mahābāhu,_  
_rāmo rajjamakārayī”ti._  


</Paragraph>
</Division>

<Paragraph number="2209" type="center" book="ja1">
Dasarathajātakaṃ sattamaṃ.
</Paragraph>
