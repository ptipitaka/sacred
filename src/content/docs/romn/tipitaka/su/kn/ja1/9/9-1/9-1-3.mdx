<Paragraph number="1802" type="verses" book="ja1">
_“Dumo yadā hoti phalūpapanno,_  
_Bhuñjanti naṃ vihaṅgamā sampatantā;_  
_Khīṇanti ñatvāna dumaṃ phalaccaye,_  
_Disodisaṃ yanti tato vihaṅgamā._  

</Paragraph>

</Division>
<Division number="21" book="ja1" bookNo="22" page="359">

<Paragraph number="1803" type="verses" book="ja1">
_Cara cārikaṃ lohitatuṇḍa māmari,_  
_Kiṃ tvaṃ suva sukkhadumamhi jhāyasi;_  
_Tadiṅgha maṃ brūhi vasantasannibha,_  
_Kasmā suva sukkhadumaṃ na riñcasi”._  

</Paragraph>

</Division>
<Division number="22" book="ja1" bookNo="22" page="360">

<Paragraph number="1804" type="verses" book="ja1">
_“Ye ve sakhīnaṃ sakhāro bhavanti,_  
_Pāṇaccaye dukkhasukhesu haṃsa;_  
_Khīṇaṃ akhīṇampi na taṃ jahanti,_  
_Santo sataṃ dhammamanussarantā._  

</Paragraph>

</Division>
<Division number="23" book="ja1" bookNo="22" page="360">

<Paragraph number="1805" type="verses" book="ja1">
_Sohaṃ sataṃ aññatarosmi haṃsa,_  
_Ñātī ca me hoti sakhā ca rukkho;_  
_Taṃ nussahe jīvikattho pahātuṃ,_  
_Khīṇanti ñatvāna na hesa dhammo”._  

</Paragraph>

</Division>
<Division number="24" book="ja1" bookNo="22" page="360">

<Paragraph number="1806" type="verses" book="ja1">
_“Sādhu sakkhi kataṃ hoti,_  
_metti saṃsati santhavo;_  
_Sacetaṃ dhammaṃ rocesi,_  
_pāsaṃsosi vijānataṃ._  

</Paragraph>

</Division>
<Division number="25" book="ja1" bookNo="22" page="360">

<Paragraph number="1807" type="verses" book="ja1">
_So te suva varaṃ dammi,_  
_pattayāna vihaṅgama;_  
_Varaṃ varassu vakkaṅga,_  
_yaṃ kiñci manasicchasi”._  

</Paragraph>

</Division>
<Division number="26" book="ja1" bookNo="22" page="360">

<Paragraph number="1808" type="verses" book="ja1">
_“Varañca me haṃsa bhavaṃ dadeyya,_  
_Ayañca rukkho punarāyuṃ labhetha;_  
_So sākhavā phalimā saṃvirūḷho,_  
_Madhutthiko tiṭṭhatu sobhamāno”._  

</Paragraph>

</Division>
<Division number="27" book="ja1" bookNo="22" page="360">

<Paragraph number="1809" type="verses" book="ja1">
_“Taṃ passa samma phalimaṃ uḷāraṃ,_  
_Sahāva te hotu udumbarena;_  
_So sākhavā phalimā saṃvirūḷho,_  
_Madhutthiko tiṭṭhatu sobhamāno”._  

</Paragraph>

</Division>
<Division number="28" book="ja1" bookNo="22" page="360">

<Paragraph number="1810" type="verses" book="ja1">
_“Evaṃ sakka sukhī hohi,_  
_saha sabbehi ñātibhi;_  
_Yathāhamajja sukhito,_  
_disvāna saphalaṃ dumaṃ”._  

</Paragraph>

</Division>
<Division number="29" book="ja1" bookNo="22" page="360">

<Paragraph number="1811" type="verses" book="ja1">
_“Suvassa ca varaṃ datvā,_  
_katvāna saphalaṃ dumaṃ;_  
_Pakkāmi saha bhariyāya,_  
_devānaṃ nandanaṃ vanan”ti._  


</Paragraph>
</Division>

<Paragraph number="1812" type="center" book="ja1">
Mahāsuvajātakaṃ tatiyaṃ.
</Paragraph>
