<Paragraph number="2352" type="verses" book="ja1">
_“Nādaṭṭhā parato dosaṃ,_  
_aṇuṃthūlāni sabbaso;_  
_Issaro paṇaye daṇḍaṃ,_  
_sāmaṃ appaṭivekkhiya._  

</Paragraph>

</Division>
<Division number="107" book="ja1" bookNo="22" page="106">

<Paragraph number="2353" type="verses" book="ja1">
_Yo ca appaṭivekkhitvā,_  
_daṇḍaṃ kubbati khattiyo;_  
_Sakaṇṭakaṃ so gilati,_  
_jaccandhova samakkhikaṃ._  

</Paragraph>

</Division>
<Division number="108" book="ja1" bookNo="22" page="107">

<Paragraph number="2354" type="verses" book="ja1">
_Adaṇḍiyaṃ daṇḍayati,_  
_daṇḍiyañca adaṇḍiyaṃ;_  
_Andhova visamaṃ maggaṃ,_  
_na jānāti samāsamaṃ._  

</Paragraph>

</Division>
<Division number="109" book="ja1" bookNo="22" page="107">

<Paragraph number="2355" type="verses" book="ja1">
_Yo ca etāni ṭhānāni,_  
_Aṇuṃthūlāni sabbaso;_  
_Sudiṭṭhamanusāseyya,_  
_Sa ve voharitumarahati._  

</Paragraph>

</Division>
<Division number="110" book="ja1" bookNo="22" page="107">

<Paragraph number="2356" type="verses" book="ja1">
_Nekantamudunā sakkā,_  
_Ekantatikhiṇena vā;_  
_Attaṃ mahante ṭhapetuṃ,_  
_Tasmā ubhayamācare._  

</Paragraph>

</Division>
<Division number="111" book="ja1" bookNo="22" page="107">

<Paragraph number="2357" type="verses" book="ja1">
_Paribhūto mudu hoti,_  
_atitikkho ca veravā;_  
_Etañca ubhayaṃ ñatvā,_  
_anumajjhaṃ samācare._  

</Paragraph>

</Division>
<Division number="112" book="ja1" bookNo="22" page="107">

<Paragraph number="2358" type="verses" book="ja1">
_Bahumpi ratto bhāseyya,_  
_duṭṭhopi bahu bhāsati;_  
_Na itthikāraṇā rāja,_  
_puttaṃ ghātetumarahasi”._  

</Paragraph>

</Division>
<Division number="113" book="ja1" bookNo="22" page="107">

<Paragraph number="2359" type="verses" book="ja1">
_“Sabbova loko ekato,_  
_Itthī ca ayamekikā;_  
_Tenāhaṃ paṭipajjissaṃ,_  
_Gacchatha pakkhipatheva taṃ”._  

</Paragraph>

</Division>
<Division number="114" book="ja1" bookNo="22" page="107">

<Paragraph number="2360" type="verses" book="ja1">
_“Anekatāle narake,_  
_gambhīre ca suduttare;_  
_Pātito giriduggasmiṃ,_  
_kena tvaṃ tattha nāmari”._  

</Paragraph>

</Division>
<Division number="115" book="ja1" bookNo="22" page="108">

<Paragraph number="2361" type="verses" book="ja1">
_“Nāgo jātaphaṇo tattha,_  
_thāmavā girisānujo;_  
_Paccaggahi maṃ bhogehi,_  
_tenāhaṃ tattha nāmariṃ”._  

</Paragraph>

</Division>
<Division number="116" book="ja1" bookNo="22" page="108">

<Paragraph number="2362" type="verses" book="ja1">
_“Ehi taṃ paṭinessāmi,_  
_rājaputta sakaṃ gharaṃ;_  
_Rajjaṃ kārehi bhaddante,_  
_kiṃ araññe karissasi”._  

</Paragraph>

</Division>
<Division number="117" book="ja1" bookNo="22" page="108">

<Paragraph number="2363" type="verses" book="ja1">
_“Yathā gilitvā baḷisaṃ,_  
_Uddhareyya salohitaṃ;_  
_Uddharitvā sukhī assa,_  
_Evaṃ passāmi attanaṃ”._  

</Paragraph>

</Division>
<Division number="118" book="ja1" bookNo="22" page="108">

<Paragraph number="2364" type="verses" book="ja1">
_“Kiṃ nu tvaṃ baḷisaṃ brūsi,_  
_kiṃ tvaṃ brūsi salohitaṃ;_  
_Kiṃ nu tvaṃ ubbhataṃ brūsi,_  
_taṃ me akkhāhi pucchito”._  

</Paragraph>

</Division>
<Division number="119" book="ja1" bookNo="22" page="108">

<Paragraph number="2365" type="verses" book="ja1">
_“Kāmāhaṃ baḷisaṃ brūmi,_  
_hatthiassaṃ salohitaṃ;_  
_Cattāhaṃ ubbhataṃ brūmi,_  
_evaṃ jānāhi khattiya”._  

</Paragraph>

</Division>
<Division number="120" book="ja1" bookNo="22" page="108">

<Paragraph number="2366" type="verses" book="ja1">
_“Ciñcamāṇavikā mātā,_  
_devadatto ca me pitā;_  
_Ānando paṇḍito nāgo,_  
_sāriputto ca devatā;_  
_Rājaputto ahaṃ āsiṃ,_  
_evaṃ dhāretha jātakan”ti._  


</Paragraph>
</Division>

<Paragraph number="2367" type="center" book="ja1">
Mahāpadumajātakaṃ navamaṃ.
</Paragraph>
