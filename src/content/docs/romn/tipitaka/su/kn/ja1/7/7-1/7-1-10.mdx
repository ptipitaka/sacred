<Division number="68" book="ja1">

<Paragraph number="1561" type="verses" book="ja1">
_“Dvāsattati gotama puññakammā,_  
_Vasavattino jātijaraṃ atītā;_  
_Ayamantimā vedagū brahmapatti,_  
_Asmābhijappanti janā anekā”._  

</Paragraph>

</Division>
<Division number="69" book="ja1">

<Paragraph number="1562" type="verses" book="ja1">
_“Appañhi etaṃ na hi dīghamāyu,_  
_Yaṃ tvaṃ baka maññasi dīghamāyuṃ;_  
_Sataṃ sahassāni nirabbudānaṃ,_  
_Āyuṃ pajānāmi tavāha brahme”._  

</Paragraph>

</Division>
<Division number="70" book="ja1">

<Paragraph number="1563" type="verses" book="ja1">
_“Anantadassī bhagavāhamasmi,_  
_Jātijjaraṃ sokamupātivatto;_  
_Kiṃ me purāṇaṃ vatasīlavattaṃ,_  
_Ācikkha me taṃ yamahaṃ vijaññaṃ”._  

</Paragraph>

</Division>
<Division number="71" book="ja1">

<Paragraph number="1564" type="verses" book="ja1">
_“Yaṃ tvaṃ apāyesi bahū manusse,_  
_Pipāsite ghammani samparete;_  
_Taṃ te purāṇaṃ vatasīlavattaṃ,_  
_Suttappabuddhova anussarāmi._  

</Paragraph>

</Division>
<Division number="72" book="ja1">

<Paragraph number="1565" type="verses" book="ja1">
_Yaṃ eṇikūlasmi janaṃ gahītaṃ,_  
_Amocayī gayhaka niyyamānaṃ;_  
_Taṃ te purāṇaṃ vatasīlavattaṃ,_  
_Suttappabuddhova anussarāmi._  

</Paragraph>

</Division>
<Division number="73" book="ja1">

<Paragraph number="1566" type="verses" book="ja1">
_Gaṅgāya sotasmiṃ gahītanāvaṃ,_  
_Luddena nāgena manussakappā;_  
_Amocayi tvaṃ balasā pasayha,_  
_Taṃ te purāṇaṃ vatasīlavattaṃ;_  
_Suttappabuddhova anussarāmi._  

</Paragraph>

</Division>
<Division number="74" book="ja1">

<Paragraph number="1567" type="verses" book="ja1">
_Kappo ca te baddhacaro ahosiṃ,_  
_Sambuddhimantaṃ vatinaṃ amaññaṃ;_  
_Taṃ te purāṇaṃ vatasīlavattaṃ,_  
_Suttappabuddhova anussarāmi”._  

</Paragraph>

</Division>
<Division number="75" book="ja1">

<Paragraph number="1568" type="verses" book="ja1">
_“Addhā pajānāsi mametamāyuṃ,_  
_Aññampi jānāsi tathā hi buddho;_  
_Tathā hi tāyaṃ jalitānubhāvo,_  
_Obhāsayaṃ tiṭṭhati brahmalokan”ti._  


</Paragraph>
</Division>

<Paragraph number="1569" type="center" book="ja1">
Bakajātakaṃ dasamaṃ.

  
Kukkuvaggo paṭhamo.



</Paragraph>

<Paragraph number="1570" type="center" book="ja1">
Tassuddānaṃ


</Paragraph>

_Varakaṇṇika cāpavaro sutano,_  
_Atha gijjha sarohitamacchavaro;_  
_Puna paṇṇaka senaka yācanako,_  
_Atha veri sabrahmabakena dasāti._
</Paragraph>
