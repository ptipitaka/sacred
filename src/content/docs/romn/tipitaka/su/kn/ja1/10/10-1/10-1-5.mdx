<Division number="49" book="ja1">

<Paragraph number="1961" type="verses" book="ja1">
_“Yo te imaṃ visālakkhiṃ,_  
_piyaṃ sammhitabhāsiniṃ;_  
_Ādāya balā gaccheyya,_  
_kiṃ nu kayirāsi brāhmaṇa”._  

</Paragraph>

</Division>
<Division number="50" book="ja1">

<Paragraph number="1962" type="verses" book="ja1">
_“Uppajje me na mucceyya,_  
_na me mucceyya jīvato;_  
_Rajaṃva vipulā vuṭṭhi,_  
_khippameva nivāraye”._  

</Paragraph>

</Division>
<Division number="51" book="ja1">

<Paragraph number="1963" type="verses" book="ja1">
_“Yannu pubbe vikatthittho,_  
_balamhiva apassito;_  
_Svajja tuṇhikato dāni,_  
_saṅghāṭiṃ sibbamacchasi”._  

</Paragraph>

</Division>
<Division number="52" book="ja1">

<Paragraph number="1964" type="verses" book="ja1">
_“Uppajji me na muccittha,_  
_na me muccittha jīvato;_  
_Rajaṃva vipulā vuṭṭhi,_  
_khippameva nivārayiṃ”._  

</Paragraph>

</Division>
<Division number="53" book="ja1">

<Paragraph number="1965" type="verses" book="ja1">
_“Kiṃ te uppajji no mucci,_  
_kiṃ te na mucci jīvato;_  
_Rajaṃva vipulā vuṭṭhi,_  
_katamaṃ tvaṃ nivārayi”._  

</Paragraph>

</Division>
<Division number="54" book="ja1">

<Paragraph number="1966" type="verses" book="ja1">
_“Yamhi jāte na passati,_  
_ajāte sādhu passati;_  
_So me uppajji no mucci,_  
_kodho dummedhagocaro._  

</Paragraph>

</Division>
<Division number="55" book="ja1">

<Paragraph number="1967" type="verses" book="ja1">
_Yena jātena nandanti,_  
_amittā dukkhamesino;_  
_So me uppajji no mucci,_  
_kodho dummedhagocaro._  

</Paragraph>

</Division>
<Division number="56" book="ja1">

<Paragraph number="1968" type="verses" book="ja1">
_Yasmiñca jāyamānamhi,_  
_sadatthaṃ nāvabujjhati;_  
_So me uppajji no mucci,_  
_kodho dummedhagocaro._  

</Paragraph>

</Division>
<Division number="57" book="ja1">

<Paragraph number="1969" type="verses" book="ja1">
_Yenābhibhūto kusalaṃ jahāti,_  
_Parakkare vipulañcāpi atthaṃ;_  
_Sa bhīmaseno balavā pamaddī,_  
_Kodho mahārāja na me amuccatha._  

</Paragraph>

</Division>
<Division number="58" book="ja1">

<Paragraph number="1970" type="verses" book="ja1">
_Kaṭṭhasmiṃ matthamānasmiṃ,_  
_Pāvako nāma jāyati;_  
_Tameva kaṭṭhaṃ ḍahati,_  
_Yasmā so jāyate gini._  

</Paragraph>

</Division>
<Division number="59" book="ja1">

<Paragraph number="1971" type="verses" book="ja1">
_Evaṃ mandassa posassa,_  
_bālassa avijānato;_  
_Sārambhā jāyate kodho,_  
_sopi teneva ḍayhati._  

</Paragraph>

</Division>
<Division number="60" book="ja1">

<Paragraph number="1972" type="verses" book="ja1">
_Aggīva tiṇakaṭṭhasmiṃ,_  
_kodho yassa pavaḍḍhati;_  
_Nihīyati tassa yaso,_  
_kāḷapakkheva candimā._  

</Paragraph>

</Division>
<Division number="61" book="ja1">

<Paragraph number="1973" type="verses" book="ja1">
_Anedho dhūmaketūva,_  
_kodho yassūpasammati;_  
_Āpūrati tassa yaso,_  
_sukkapakkheva candimā”ti._  


</Paragraph>
</Division>

<Paragraph number="1974" type="center" book="ja1">
Cūḷabodhijātakaṃ pañcamaṃ.
</Paragraph>
