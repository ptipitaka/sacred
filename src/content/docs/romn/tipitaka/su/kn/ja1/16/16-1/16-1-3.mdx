<Paragraph number="3321" type="verses" book="ja1">
_“Cirassaṃ vata me udapādi ajja,_  
_Bhakkho mahā sattamibhattakāle;_  
_Kutosi ko vāsi tadiṅgha brūhi,_  
_Ācikkha jātiṃ vidito yathāsi”._  

</Paragraph>

</Division>
<Division number="65" book="ja1" bookNo="22" page="182">

<Paragraph number="3322" type="verses" book="ja1">
_“Pañcālarājā migavaṃ paviṭṭho,_  
_Jayaddiso nāma yadissuto te;_  
_Carāmi kacchāni vanāni cāhaṃ,_  
_Pasadaṃ imaṃ khāda mamajja muñca”._  

</Paragraph>

</Division>
<Division number="66" book="ja1" bookNo="22" page="182">

<Paragraph number="3323" type="verses" book="ja1">
_“Seneva tvaṃ paṇasi sassamāno,_  
_Mamesa bhakkho pasado yaṃ vadesi;_  
_Taṃ khādiyāna pasadaṃ jighaññaṃ,_  
_Khādissaṃ pacchā na vilāpakālo”._  

</Paragraph>

</Division>
<Division number="67" book="ja1" bookNo="22" page="182">

<Paragraph number="3324" type="verses" book="ja1">
_“Na catthi mokkho mama nikkayena,_  
_Gantvāna paccāgamanāya paṇhe;_  
_Taṃ saṅgaraṃ brāhmaṇassappadāya,_  
_Saccānurakkhī punarāvajissaṃ”._  

</Paragraph>

</Division>
<Division number="68" book="ja1" bookNo="22" page="183">

<Paragraph number="3325" type="verses" book="ja1">
_“Kiṃ kammajātaṃ anutappate tvaṃ,_  
_Pattaṃ samīpaṃ maraṇassa rāja;_  
_Ācikkha me taṃ api sakkuṇemu,_  
_Anujānituṃ āgamanāya paṇhe”._  

</Paragraph>

</Division>
<Division number="69" book="ja1" bookNo="22" page="183">

<Paragraph number="3326" type="verses" book="ja1">
_“Katā mayā brāhmaṇassa dhanāsā,_  
_Taṃ saṅgaraṃ paṭimukkaṃ na muttaṃ;_  
_Taṃ saṅgaraṃ brāhmaṇassappadāya,_  
_Saccānurakkhī punarāvajissaṃ”._  

</Paragraph>

</Division>
<Division number="70" book="ja1" bookNo="22" page="183">

<Paragraph number="3327" type="verses" book="ja1">
_“Yā te katā brāhmaṇassa dhanāsā,_  
_Taṃ saṅgaraṃ paṭimukkaṃ na muttaṃ;_  
_Taṃ saṅgaraṃ brāhmaṇassappadāya,_  
_Saccānurakkhī punarāvajassu”._  

</Paragraph>

</Division>
<Division number="71" book="ja1" bookNo="22" page="183">

<Paragraph number="3328" type="verses" book="ja1">
_Mutto ca so porisādassa hatthā,_  
_Gantvā sakaṃ mandiraṃ kāmakāmī;_  
_Taṃ saṅgaraṃ brāhmaṇassappadāya,_  
_Āmantayī puttamalīnasattaṃ._  

</Paragraph>

</Division>
<Division number="72" book="ja1" bookNo="22" page="183">

<Paragraph number="3329" type="verses" book="ja1">
_“Ajjeva rajjaṃ abhisiñcayassu,_  
_Dhammaṃ cara sesu paresu cāpi;_  
_Adhammakāro ca te māhu raṭṭhe,_  
_Gacchāmahaṃ porisādassa ñatte”._  

</Paragraph>

</Division>
<Division number="73" book="ja1" bookNo="22" page="183">

<Paragraph number="3330" type="verses" book="ja1">
_“Kiṃ kamma kubbaṃ tava deva pāva,_  
_Nārādhayī taṃ tadicchāmi sotuṃ;_  
_Yamajja rajjamhi udassaye tuvaṃ,_  
_Rajjampi niccheyyaṃ tayā vināhaṃ”._  

</Paragraph>

</Division>
<Division number="74" book="ja1" bookNo="22" page="183">

<Paragraph number="3331" type="verses" book="ja1">
_“Na kammunā vā vacasā va tāta,_  
_Aparādhitohaṃ tuviyaṃ sarāmi;_  
_Sandhiñca katvā purisādakena,_  
_Saccānurakkhī punāhaṃ gamissaṃ”._  

</Paragraph>

</Division>
<Division number="75" book="ja1" bookNo="22" page="183">

<Paragraph number="3332" type="verses" book="ja1">
_“Ahaṃ gamissāmi idheva hohi,_  
_Natthi tato jīvato vippamokkho;_  
_Sace tuvaṃ gacchasiyeva rāja,_  
_Ahampi gacchāmi ubho na homa”._  

</Paragraph>

</Division>
<Division number="76" book="ja1" bookNo="22" page="184">

<Paragraph number="3333" type="verses" book="ja1">
_“Addhā hi tāta satānesa dhammo,_  
_Maraṇā ca me dukkhataraṃ tadassa;_  
_Kammāsapādo taṃ yadā pacitvā,_  
_Pasayha khāde bhidā rukkhasūle”._  

</Paragraph>

</Division>
<Division number="77" book="ja1" bookNo="22" page="184">

<Paragraph number="3334" type="verses" book="ja1">
_“Pāṇena te pāṇamahaṃ nimissaṃ,_  
_Mā tvaṃ agā porisādassa ñatte;_  
_Etañca te pāṇamahaṃ nimissaṃ,_  
_Tasmā mataṃ jīvitassa vaṇṇemi”._  

</Paragraph>

</Division>
<Division number="78" book="ja1" bookNo="22" page="184">

<Paragraph number="3335" type="verses" book="ja1">
_Tato have dhitimā rājaputto,_  
_Vanditvā mātu ca pitu ca pāde;_  
_Dukhinissa mātā nipatā pathabyā,_  
_Pitāssa paggayha bhujāni kandati._  

</Paragraph>

</Division>
<Division number="79" book="ja1" bookNo="22" page="184">

<Paragraph number="3336" type="verses" book="ja1">
_Taṃ gacchantaṃ tāva pitā viditvā,_  
_Parammukho vandati pañjalīko;_  
_“Somo ca rājā varuṇo ca rājā,_  
_Pajāpatī candimā sūriyo ca;_  
_Etehi gutto purisādakamhā,_  
_Anuññāto sotthi paccehi tāta”._  

</Paragraph>

</Division>
<Division number="80" book="ja1" bookNo="22" page="184">

<Paragraph number="3337" type="verses" book="ja1">
_“Yaṃ daṇḍakirañño gatassa mātā,_  
_Rāmassakāsi sotthānaṃ suguttā;_  
_Taṃ te ahaṃ sotthānaṃ karomi,_  
_Etena saccena sarantu devā;_  
_Anuññāto sotthi paccehi putta”._  

</Paragraph>

</Division>
<Division number="81" book="ja1" bookNo="22" page="184">

<Paragraph number="3338" type="verses" book="ja1">
_“Āvī raho vāpi manopadosaṃ,_  
_Nāhaṃ sare jātu malīnasatte;_  
_Etena saccena sarantu devā,_  
_Anuññāto sotthi paccehi bhātika”._  

</Paragraph>

</Division>
<Division number="82" book="ja1" bookNo="22" page="184">

<Paragraph number="3339" type="verses" book="ja1">
_“Yasmā ca me anadhimanosi sāmi,_  
_Na cāpi me manasā appiyosi;_  
_Etena saccena sarantu devā,_  
_Anuññāto sotthi paccehi sāmi”._  

</Paragraph>

</Division>
<Division number="83" book="ja1" bookNo="22" page="184">

<Paragraph number="3340" type="verses" book="ja1">
_“Brahā ujū cārumukho kutosi,_  
_Na maṃ pajānāsi vane vasantaṃ;_  
_Luddaṃ maṃ ñatvā ‘purisādako’ti,_  
_Ko sotthimājānamidhāvajeyya”._  

</Paragraph>

</Division>
<Division number="84" book="ja1" bookNo="22" page="184">

<Paragraph number="3341" type="verses" book="ja1">
_“Jānāmi ludda purisādako tvaṃ,_  
_Na taṃ na jānāmi vane vasantaṃ;_  
_Ahañca puttosmi jayaddisassa,_  
_Mamajja khāda pituno pamokkhā”._  

</Paragraph>

</Division>
<Division number="85" book="ja1" bookNo="22" page="184">

<Paragraph number="3342" type="verses" book="ja1">
_“Jānāmi puttoti jayaddisassa,_  
_Tathā hi vo mukhavaṇṇo ubhinnaṃ;_  
_Sudukkaraññeva kataṃ tavedaṃ,_  
_Yo mattumicche pituno pamokkhā”._  

</Paragraph>

</Division>
<Division number="86" book="ja1" bookNo="22" page="185">

<Paragraph number="3343" type="verses" book="ja1">
_“Na dukkaraṃ kiñci mahettha maññe,_  
_Yo mattumicche pituno pamokkhā;_  
_Mātu ca hetu paraloka gantvā,_  
_Sukhena saggena ca sampayutto”._  

</Paragraph>

</Division>
<Division number="87" book="ja1" bookNo="22" page="185">

<Paragraph number="3344" type="verses" book="ja1">
_“Ahañca kho attano pāpakiriyaṃ,_  
_Āvī raho vāpi sare na jātu;_  
_Saṅkhātajātīmaraṇohamasmi,_  
_Yatheva me idha tathā parattha._  

</Paragraph>

</Division>
<Division number="88" book="ja1" bookNo="22" page="185">

<Paragraph number="3345" type="verses" book="ja1">
_Khādajja maṃ dāni mahānubhāva,_  
_Karassu kiccāni imaṃ sarīraṃ;_  
_Rukkhassa vā te papatāmi aggā,_  
_Chādayamāno mayhaṃ tvamadesi maṃsaṃ”._  

</Paragraph>

</Division>
<Division number="89" book="ja1" bookNo="22" page="185">

<Paragraph number="3346" type="verses" book="ja1">
_“Idañca te ruccati rājaputta,_  
_Cajesi pāṇaṃ pituno pamokkhā;_  
_Tasmā hi so tvaṃ taramānarūpo,_  
_Sambhañja kaṭṭhāni jalehi aggiṃ”._  

</Paragraph>

</Division>
<Division number="90" book="ja1" bookNo="22" page="185">

<Paragraph number="3347" type="verses" book="ja1">
_Tato have dhitimā rājaputto,_  
_Dāruṃ samāhatvā mahantamaggiṃ;_  
_Sandīpayitvā paṭivedayittha,_  
_“Ādīpito dāni mahāyamaggi”._  

</Paragraph>

</Division>
<Division number="91" book="ja1" bookNo="22" page="185">

<Paragraph number="3348" type="verses" book="ja1">
_“Khādajja maṃ dāni pasayhakāri,_  
_Kiṃ maṃ muhuṃ pekkhasi haṭṭhalomo;_  
_Tathā tathā tuyhamahaṃ karomi,_  
_Yathā yathā maṃ chādayamāno adesi”._  

</Paragraph>

</Division>
<Division number="92" book="ja1" bookNo="22" page="185">

<Paragraph number="3349" type="verses" book="ja1">
_“Ko tādisaṃ arahati khāditāye,_  
_Dhamme ṭhitaṃ saccavādiṃ vadaññuṃ;_  
_Muddhāpi tassa viphaleyya sattadhā,_  
_Yo tādisaṃ saccavādiṃ adeyya”._  

</Paragraph>

</Division>
<Division number="93" book="ja1" bookNo="22" page="185">

<Paragraph number="3350" type="verses" book="ja1">
_“Idañhi so brāhmaṇaṃ maññamāno,_  
_Saso avāsesi sake sarīre;_  
_Teneva so candimā devaputto,_  
_Sasatthuto kāmaduhajja yakkha”._  

</Paragraph>

</Division>
<Division number="94" book="ja1" bookNo="22" page="185">

<Paragraph number="3351" type="verses" book="ja1">
_“Cando yathā rāhumukhā pamutto,_  
_Virocate pannaraseva bhāṇumā;_  
_Evaṃ tuvaṃ porisādā pamutto,_  
_Viroca kapile mahānubhāva;_  
_Āmodayaṃ pitaraṃ mātarañca,_  
_Sabbo ca te nandatu ñātipakkho”._  

</Paragraph>

</Division>
<Division number="95" book="ja1" bookNo="22" page="186">

<Paragraph number="3352" type="verses" book="ja1">
_Tato have dhitimā rājaputto,_  
_Katañjalī pariyāya porisādaṃ;_  
_Anuññāto sotthi sukhī arogo,_  
_Paccāgamā kapilamalīnasatto._  

</Paragraph>

</Division>
<Division number="96" book="ja1" bookNo="22" page="198">

<Paragraph number="3353" type="verses" book="ja1">
_Taṃ negamā jānapadā ca sabbe,_  
_Hatthārohā rathikā pattikā ca;_  
_Namassamānā pañjalikā upāgamuṃ,_  
_“Namatthu te dukkarakārakosī”ti._  


</Paragraph>
</Division>
