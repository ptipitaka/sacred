<Division number="37" book="ja1">

<Paragraph number="2278" type="verses" book="ja1">
_“Kāmaṃ kāmayamānassa,_  
_tassa ce taṃ samijjhati;_  
_Addhā pītimano hoti,_  
_laddhā macco yadicchati._  

</Paragraph>

</Division>
<Division number="38" book="ja1">

<Paragraph number="2279" type="verses" book="ja1">
_Kāmaṃ kāmayamānassa,_  
_tassa ce taṃ samijjhati;_  
_Tato naṃ aparaṃ kāme,_  
_ghamme taṇhaṃva vindati._  

</Paragraph>

</Division>
<Division number="39" book="ja1">

<Paragraph number="2280" type="verses" book="ja1">
_Gavaṃva siṅgino siṅgaṃ,_  
_vaḍḍhamānassa vaḍḍhati;_  
_Evaṃ mandassa posassa,_  
_bālassa avijānato;_  
_Bhiyyo taṇhā pipāsā ca,_  
_vaḍḍhamānassa vaḍḍhati._  

</Paragraph>

</Division>
<Division number="40" book="ja1">

<Paragraph number="2281" type="verses" book="ja1">
_Pathabyā sāliyavakaṃ,_  
_gavāssaṃ dāsaporisaṃ;_  
_Datvā ca nālamekassa,_  
_iti vidvā samaṃ care._  

</Paragraph>

</Division>
<Division number="41" book="ja1">

<Paragraph number="2282" type="verses" book="ja1">
_Rājā pasayha pathaviṃ vijitvā,_  
_Sasāgarantaṃ mahimāvasanto;_  
_Oraṃ samuddassa atittarūpo,_  
_Pāraṃ samuddassapi patthayetha._  

</Paragraph>

</Division>
<Division number="42" book="ja1">

<Paragraph number="2283" type="verses" book="ja1">
_Yāva anussaraṃ kāme,_  
_Manasā titti nājjhagā;_  
_Tato nivattā paṭikkamma disvā,_  
_Te ve sutittā ye paññāya tittā._  

</Paragraph>

</Division>
<Division number="43" book="ja1">

<Paragraph number="2284" type="verses" book="ja1">
_Paññāya tittinaṃ seṭṭhaṃ,_  
_na so kāmehi tappati;_  
_Paññāya tittaṃ purisaṃ,_  
_taṇhā na kurute vasaṃ._  

</Paragraph>

</Division>
<Division number="44" book="ja1">

<Paragraph number="2285" type="verses" book="ja1">
_Apacinetheva kāmānaṃ,_  
_appicchassa alolupo;_  
_Samuddamatto puriso,_  
_na so kāmehi tappati._  

</Paragraph>

</Division>
<Division number="45" book="ja1">

<Paragraph number="2286" type="verses" book="ja1">
_Rathakārova cammassa,_  
_parikantaṃ upāhanaṃ;_  
_Yaṃ yaṃ cajati kāmānaṃ,_  
_taṃ taṃ sampajjate sukhaṃ;_  
_Sabbañce sukhamiccheyya,_  
_sabbe kāme pariccaje”._  

</Paragraph>

</Division>
<Division number="46" book="ja1">

<Paragraph number="2287" type="verses" book="ja1">
_“Aṭṭha te bhāsitā gāthā,_  
_sabbā honti sahassiyo;_  
_Paṭigaṇha mahābrahme,_  
_sādhetaṃ tava bhāsitaṃ”._  

</Paragraph>

</Division>
<Division number="47" book="ja1">

<Paragraph number="2288" type="verses" book="ja1">
_“Na me attho sahassehi,_  
_satehi nahutehi vā;_  
_Pacchimaṃ bhāsato gāthaṃ,_  
_kāme me na rato mano”._  

</Paragraph>

</Division>
<Division number="48" book="ja1">

<Paragraph number="2289" type="verses" book="ja1">
_“Bhadrako vatāyaṃ māṇavako,_  
_Sabbalokavidū muni;_  
_Yo imaṃ taṇhaṃ dukkhajananiṃ,_  
_Parijānāti paṇḍito”ti._  


</Paragraph>
</Division>

<Paragraph number="2290" type="center" book="ja1">
Kāmajātakaṃ catutthaṃ.
</Paragraph>
