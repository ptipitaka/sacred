<Division number="1" book="ja1">

<Paragraph number="1909" type="verses" book="ja1">
_“Catudvāramidaṃ nagaraṃ,_  
_āyasaṃ daḷhapākāraṃ;_  
_Oruddhapaṭiruddhosmi,_  
_kiṃ pāpaṃ pakataṃ mayā”._  

</Paragraph>

</Division>
<Division number="2" book="ja1">

<Paragraph number="1910" type="verses" book="ja1">
_“Sabbe apihitā dvārā,_  
_oruddhosmi yathā dijo;_  
_Kimādhikaraṇaṃ yakkha,_  
_cakkābhinihato ahaṃ._  

</Paragraph>

</Division>
<Division number="3" book="ja1">

<Paragraph number="1911" type="verses" book="ja1">
_Laddhā satasahassāni,_  
_atirekāni vīsati;_  
_Anukampakānaṃ ñātīnaṃ,_  
_vacanaṃ samma nākari._  

</Paragraph>

</Division>
<Division number="4" book="ja1">

<Paragraph number="1912" type="verses" book="ja1">
_Laṅghiṃ samuddaṃ pakkhandi,_  
_sāgaraṃ appasiddhikaṃ;_  
_Catubbhi aṭṭhajjhagamā,_  
_aṭṭhāhipi ca soḷasa._  

</Paragraph>

</Division>
<Division number="5" book="ja1">

<Paragraph number="1913" type="verses" book="ja1">
_Soḷasāhi ca bāttiṃsa,_  
_atricchaṃ cakkamāsado;_  
_Icchāhatassa posassa,_  
_cakkaṃ bhamati matthake._  

</Paragraph>

</Division>
<Division number="6" book="ja1">

<Paragraph number="1914" type="verses" book="ja1">
_Uparivisālā duppūrā,_  
_icchā visaṭagāminī;_  
_Ye ca taṃ anugijjhanti,_  
_te honti cakkadhārino._  

</Paragraph>

</Division>
<Division number="7" book="ja1">

<Paragraph number="1915" type="verses" book="ja1">
_Bahubhaṇḍaṃ avahāya,_  
_maggaṃ appaṭivekkhiya;_  
_Yesañcetaṃ asaṅkhātaṃ,_  
_te honti cakkadhārino._  

</Paragraph>

</Division>
<Division number="8" book="ja1">

<Paragraph number="1916" type="verses" book="ja1">
_Kammaṃ samekkhe vipulañca bhogaṃ,_  
_Icchaṃ na seveyya anatthasaṃhitaṃ;_  
_Kareyya vākyaṃ anukampakānaṃ,_  
_Taṃ tādisaṃ nātivatteyya cakkaṃ”._  

</Paragraph>

</Division>
<Division number="9" book="ja1">

<Paragraph number="1917" type="verses" book="ja1">
_“Kīvaciraṃ nu me yakkha,_  
_cakkaṃ sirasi ṭhassati;_  
_Kati vassasahassāni,_  
_taṃ me akkhāhi pucchito”._  

</Paragraph>

</Division>
<Division number="10" book="ja1">

<Paragraph number="1918" type="verses" book="ja1">
_“Atisaro paccasaro,_  
_mittavinda suṇohi me;_  
_Cakkaṃ te sirasi māviddhaṃ,_  
_na taṃ jīvaṃ pamokkhasī”ti._  


</Paragraph>
</Division>

<Paragraph number="1919" type="center" book="ja1">
Catudvārajātakaṃ paṭhamaṃ.
</Paragraph>
