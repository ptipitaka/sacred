<Division number="83" book="ja1">

<Paragraph number="1413" type="verses" book="ja1">
_“Akakkasaṃ apharusaṃ,_  
_kharadhotaṃ supāsiyaṃ;_  
_Sukhumaṃ tikhiṇaggañca,_  
_ko sūciṃ ketumicchati._  

</Paragraph>

</Division>
<Division number="84" book="ja1">

<Paragraph number="1414" type="verses" book="ja1">
_Sumajjañca supāsañca,_  
_anupubbaṃ suvaṭṭitaṃ;_  
_Ghanaghātimaṃ paṭithaddhaṃ,_  
_ko sūciṃ ketumicchati”._  

</Paragraph>

</Division>
<Division number="85" book="ja1">

<Paragraph number="1415" type="verses" book="ja1">
_“Ito dāni patāyanti,_  
_sūciyo baḷisāni ca;_  
_Koyaṃ kammāragāmasmiṃ,_  
_sūciṃ vikketumicchati._  

</Paragraph>

</Division>
<Division number="86" book="ja1">

<Paragraph number="1416" type="verses" book="ja1">
_Ito satthāni gacchanti,_  
_Kammantā vividhā puthū;_  
_Koyaṃ kammāragāmasmiṃ,_  
_Sūciṃ vikketumicchati”._  

</Paragraph>

</Division>
<Division number="87" book="ja1">

<Paragraph number="1417" type="verses" book="ja1">
_“Sūciṃ kammāragāmasmiṃ,_  
_vikketabbā pajānatā;_  
_Ācariyāva jānanti,_  
_kammaṃ sukatadukkaṭaṃ._  

</Paragraph>

</Division>
<Division number="88" book="ja1">

<Paragraph number="1418" type="verses" book="ja1">
_Imañce te pitā bhadde,_  
_Sūciṃ jaññā mayā kataṃ;_  
_Tayā ca maṃ nimanteyya,_  
_Yañcatthaññaṃ ghare dhanan”ti._  


</Paragraph>
</Division>

<Paragraph number="1419" type="center" book="ja1">
Sūcijātakaṃ dutiyaṃ.
</Paragraph>
