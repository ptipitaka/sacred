<Division number="67" book="ja1">

<Paragraph number="2455" type="verses" book="ja1">
_Rājā kāliṅgo cakkavatti,_  
_Dhammena pathavimanusāsaṃ;_  
_Agamā bodhisamīpaṃ,_  
_Nāgena mahānubhāvena._  

</Paragraph>

</Division>
<Division number="68" book="ja1">

<Paragraph number="2456" type="verses" book="ja1">
_Kāliṅgo bhāradvājo ca,_  
_Rājānaṃ kāliṅgaṃ samaṇakolaññaṃ;_  
_Cakkaṃ vattayato pariggahetvā,_  
_Pañjalī idamavoca._  

</Paragraph>

</Division>
<Division number="69" book="ja1">

<Paragraph number="2457" type="verses" book="ja1">
_“Paccoroha mahārāja,_  
_Bhūmibhāgo yathā samaṇuggato;_  
_Idha anadhivarā buddhā,_  
_Abhisambuddhā virocanti._  

</Paragraph>

</Division>
<Division number="70" book="ja1">

<Paragraph number="2458" type="verses" book="ja1">
_Padakkhiṇato āvaṭṭā,_  
_Tiṇalatā asmiṃ bhūmibhāgasmiṃ;_  
_Pathaviyā nābhiyaṃ maṇḍo,_  
_Iti no sutaṃ mante mahārāja._  

</Paragraph>

</Division>
<Division number="71" book="ja1">

<Paragraph number="2459" type="verses" book="ja1">
_Sāgarapariyantāya,_  
_Mediniyā sabbabhūtadharaṇiyā;_  
_Pathaviyā ayaṃ maṇḍo,_  
_Orohitvā namo karohi._  

</Paragraph>

</Division>
<Division number="72" book="ja1">

<Paragraph number="2460" type="verses" book="ja1">
_Ye te bhavanti nāgā ca,_  
_abhijātā ca kuñjarā;_  
_Ettāvatā padesaṃ te,_  
_nāgā neva mupayanti._  

</Paragraph>

</Division>
<Division number="73" book="ja1">

<Paragraph number="2461" type="verses" book="ja1">
_Abhijāto nāgo kāmaṃ,_  
_Pesehi kuñjaraṃ dantiṃ;_  
_Ettāvatā padeso,_  
_Sakkā nāgena mupagantuṃ._  

</Paragraph>

</Division>
<Division number="74" book="ja1">

<Paragraph number="2462" type="verses" book="ja1">
_Taṃ sutvā rājā kāliṅgo,_  
_Veyyañjanikavaco nisāmetvā;_  
_Sampesesi nāgaṃ ñassāma,_  
_Mayaṃ yathimassidaṃ vacanaṃ”._  

</Paragraph>

</Division>
<Division number="75" book="ja1">

<Paragraph number="2463" type="verses" book="ja1">
_Sampesito ca raññā,_  
_Nāgo koñcova abhinaditvāna;_  
_Paṭisakkitvā nisīdi,_  
_Garuṃva bhāraṃ asahamāno._  

</Paragraph>

</Division>
<Division number="76" book="ja1">

<Paragraph number="2464" type="verses" book="ja1">
_Kāliṅgabhāradvājo,_  
_Nāgaṃ khīṇāyukaṃ viditvāna;_  
_Rājānaṃ kāliṅgaṃ,_  
_Taramāno ajjhabhāsittha;_  
_“Aññaṃ saṅkama nāgaṃ,_  
_Nāgo khīṇāyuko mahārāja”._  

</Paragraph>

</Division>
<Division number="77" book="ja1">

<Paragraph number="2465" type="verses" book="ja1">
_Taṃ sutvā kāliṅgo,_  
_taramāno saṅkamī nāgaṃ;_  
_Saṅkanteva raññe,_  
_nāgo tattheva pati bhumyā;_  
_Veyyañjanikavaco,_  
_yathā tathā ahu nāgo._  

</Paragraph>

</Division>
<Division number="78" book="ja1">

<Paragraph number="2466" type="verses" book="ja1">
_Kāliṅgo rājā kāliṅgaṃ,_  
_brāhmaṇaṃ etadavoca;_  
_“Tvameva asi sambuddho,_  
_sabbaññū sabbadassāvī”._  

</Paragraph>

</Division>
<Division number="79" book="ja1">

<Paragraph number="2467" type="verses" book="ja1">
_Taṃ anadhivāsento kāliṅgaṃ,_  
_Brāhmaṇo idamavoca;_  
_“Veyyañjanikā hi mayaṃ,_  
_Buddhā sabbaññuno mahārāja._  

</Paragraph>

</Division>
<Division number="80" book="ja1">

<Paragraph number="2468" type="verses" book="ja1">
_Sabbaññū sabbavidū ca,_  
_Buddhā na lakkhaṇena jānanti;_  
_Āgamabalasā hi mayaṃ,_  
_Buddhā sabbaṃ pajānanti”._  

</Paragraph>

</Division>
<Division number="81" book="ja1">

<Paragraph number="2469" type="verses" book="ja1">
_Mahayitvā sambodhiṃ,_  
_Nānāturiyehi vajjamānehi;_  
_Mālāvilepanaṃ abhiharitvā,_  
_Atha rājā manupāyāsi._  

</Paragraph>

</Division>
<Division number="82" book="ja1">

<Paragraph number="2470" type="verses" book="ja1">
_Saṭṭhi vāhasahassāni,_  
_pupphānaṃ sannipātayi;_  
_Pūjesi rājā kāliṅgo,_  
_bodhimaṇḍamanuttaranti._  


</Paragraph>
</Division>

<Paragraph number="2471" type="center" book="ja1">
Kāliṅgabodhijātakaṃ chaṭṭhaṃ.
</Paragraph>
