<Division number="69" book="ja1">

<Paragraph number="970" type="verses" book="ja1">
_“Yaṃ taṃ vasantasamaye,_  
_kaṇaveresu bhāṇusu;_  
_Sāmaṃ bāhāya pīḷesi,_  
_sā taṃ ārogyamabravi”._  

</Paragraph>

</Division>
<Division number="70" book="ja1">

<Paragraph number="971" type="verses" book="ja1">
_“Ambho na kira saddheyyaṃ,_  
_yaṃ vāto pabbataṃ vahe;_  
_Pabbatañce vahe vāto,_  
_sabbampi pathaviṃ vahe;_  
_Yattha sāmā kālakatā,_  
_sā maṃ ārogyamabravi”._  

</Paragraph>

</Division>
<Division number="71" book="ja1">

<Paragraph number="972" type="verses" book="ja1">
_“Na ceva sā kālakatā,_  
_na ca sā aññamicchati;_  
_Ekabhattikinī sāmā,_  
_tameva abhikaṅkhati”._  

</Paragraph>

</Division>
<Division number="72" book="ja1">

<Paragraph number="973" type="verses" book="ja1">
_“Asanthutaṃ maṃ cirasanthutena,_  
_Nimīni sāmā adhuvaṃ dhuvena;_  
_Mayāpi sāmā nimineyya aññaṃ,_  
_Ito ahaṃ dūrataraṃ gamissan”ti._  


</Paragraph>
</Division>

<Paragraph number="974" type="center" book="ja1">
Kaṇaverajātakaṃ aṭṭhamaṃ.
</Paragraph>
