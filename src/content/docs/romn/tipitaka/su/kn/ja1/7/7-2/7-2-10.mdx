<Paragraph number="1647" type="verses" book="ja1">
_“Na kiratthi anomadassisu,_  
_Pāricariyā buddhesu appikā;_  
_Sukkhāya aloṇikāya ca,_  
_Passaphalaṃ kummāsapiṇḍiyā._  

</Paragraph>

</Division>
<Division number="143" book="ja1" bookNo="22" page="377">

<Paragraph number="1648" type="verses" book="ja1">
_Hatthigavassā cime bahū,_  
_Dhanadhaññaṃ pathavī ca kevalā;_  
_Nāriyo cimā accharūpamā,_  
_Passa phalaṃ kummāsapiṇḍiyā”._  

</Paragraph>

</Division>
<Division number="144" book="ja1" bookNo="22" page="377">

<Paragraph number="1649" type="verses" book="ja1">
_“Abhikkhaṇaṃ rājakuñjara,_  
_Gāthā bhāsasi kosalādhipa;_  
_Pucchāmi taṃ raṭṭhavaḍḍhana,_  
_Bāḷhaṃ pītimano pabhāsasi”._  

</Paragraph>

</Division>
<Division number="145" book="ja1" bookNo="22" page="377">

<Paragraph number="1650" type="verses" book="ja1">
_“Imasmiññeva nagare,_  
_kule aññatare ahuṃ;_  
_Parakammakaro āsiṃ,_  
_bhatako sīlasaṃvuto._  

</Paragraph>

</Division>
<Division number="146" book="ja1" bookNo="22" page="377">

<Paragraph number="1651" type="verses" book="ja1">
_Kammāya nikkhamantohaṃ,_  
_caturo samaṇeddasaṃ;_  
_Ācārasīlasampanne,_  
_sītibhūte anāsave._  

</Paragraph>

</Division>
<Division number="147" book="ja1" bookNo="22" page="377">

<Paragraph number="1652" type="verses" book="ja1">
_Tesu cittaṃ pasādetvā,_  
_nisīdetvā paṇṇasanthate;_  
_Adaṃ buddhāna kummāsaṃ,_  
_pasanno sehi pāṇibhi._  

</Paragraph>

</Division>
<Division number="148" book="ja1" bookNo="22" page="377">

<Paragraph number="1653" type="verses" book="ja1">
_Tassa kammassa kusalassa,_  
_idaṃ me edisaṃ phalaṃ;_  
_Anubhomi idaṃ rajjaṃ,_  
_phītaṃ dharaṇimuttamaṃ”._  

</Paragraph>

</Division>
<Division number="149" book="ja1" bookNo="22" page="377">

<Paragraph number="1654" type="verses" book="ja1">
_“Dadaṃ bhuñja mā ca pamādo,_  
_Cakkaṃ vattaya kosalādhipa;_  
_Mā rāja adhammiko ahu,_  
_Dhammaṃ pālaya kosalādhipa”._  

</Paragraph>

</Division>
<Division number="150" book="ja1" bookNo="22" page="377">

<Paragraph number="1655" type="verses" book="ja1">
_“Sohaṃ tadeva punappunaṃ,_  
_Vaṭumaṃ ācarissāmi sobhane;_  
_Ariyācaritaṃ sukosale,_  
_Arahanto me manāpāva passituṃ._  

</Paragraph>

</Division>
<Division number="151" book="ja1" bookNo="22" page="378">

<Paragraph number="1656" type="verses" book="ja1">
_Devī viya accharūpamā,_  
_Majjhe nārigaṇassa sobhasi;_  
_Kiṃ kammamakāsi bhaddakaṃ,_  
_Kenāsi vaṇṇavatī sukosale”._  

</Paragraph>

</Division>
<Division number="152" book="ja1" bookNo="22" page="378">

<Paragraph number="1657" type="verses" book="ja1">
_“Ambaṭṭhakulassa khattiya,_  
_Dāsyāhaṃ parapesiyā ahuṃ;_  
_Saññatā ca dhammajīvinī,_  
_Sīlavatī ca apāpadassanā._  

</Paragraph>

</Division>
<Division number="153" book="ja1" bookNo="22" page="378">

<Paragraph number="1658" type="verses" book="ja1">
_Uddhaṭabhattaṃ ahaṃ tadā,_  
_Caramānassa adāsi bhikkhuno;_  
_Vittā sumanā sayaṃ ahaṃ,_  
_Tassa kammassa phalaṃ mamedisan”ti._  


</Paragraph>
</Division>

<Paragraph number="1659" type="center" book="ja1">
Kummāsapiṇḍijātakaṃ dasamaṃ.
</Paragraph>
