<Paragraph number="3071" type="verses" book="ja1">
_“Ko taṃ hiṃsati heṭheti,_  
_Kiṃ dummano socasi appatīto;_  
_Kassajja mātāpitaro rudantu,_  
_Kvajja setu nihato pathabyā”._  

</Paragraph>

</Division>
<Division number="212" book="ja1" bookNo="22" page="334">

<Paragraph number="3072" type="verses" book="ja1">
_“Tuṭṭhosmi deva tava dassanena,_  
_Cirassaṃ passāmi taṃ bhūmipāla;_  
_Ahiṃsako reṇumanuppavissa,_  
_Puttena te heṭhayitosmi deva”._  

</Paragraph>

</Division>
<Division number="213" book="ja1" bookNo="22" page="334">

<Paragraph number="3073" type="verses" book="ja1">
_“Āyantu dovārikā khaggabandhā,_  
_Kāsāviyā yantu antepurantaṃ;_  
_Hantvāna taṃ somanassaṃ kumāraṃ,_  
_Chetvāna sīsaṃ varamāharantu”._  

</Paragraph>

</Division>
<Division number="214" book="ja1" bookNo="22" page="334">

<Paragraph number="3074" type="verses" book="ja1">
_“Pesitā rājino dūtā,_  
_kumāraṃ etadabravuṃ;_  
_Issarena vitiṇṇosi,_  
_vadhaṃ pattosi khattiya”._  

</Paragraph>

</Division>
<Division number="215" book="ja1" bookNo="22" page="334">

<Paragraph number="3075" type="verses" book="ja1">
_“Sa rājaputto paridevayanto,_  
_Dasaṅguliṃ añjaliṃ paggahetvā;_  
_Ahampi icchāmi janinda daṭṭhuṃ,_  
_Jīvaṃ maṃ netvā paṭidassayetha”._  

</Paragraph>

</Division>
<Division number="216" book="ja1" bookNo="22" page="334">

<Paragraph number="3076" type="verses" book="ja1">
_Tassa taṃ vacanaṃ sutvā,_  
_rañño puttaṃ adassayuṃ;_  
_Putto ca pitaraṃ disvā,_  
_dūratovajjhabhāsatha._  

</Paragraph>

</Division>
<Division number="217" book="ja1" bookNo="22" page="335">

<Paragraph number="3077" type="verses" book="ja1">
_“Āgacchuṃ dovārikā khaggabandhā,_  
_Kāsāviyā hantu mamaṃ janinda;_  
_Akkhāhi me pucchito etamatthaṃ,_  
_Aparādho ko nidha mamajja atthi”._  

</Paragraph>

</Division>
<Division number="218" book="ja1" bookNo="22" page="335">

<Paragraph number="3078" type="verses" book="ja1">
_“Sāyañca pāto udakaṃ sajāti,_  
_Aggiṃ sadā pāricaratappamatto;_  
_Taṃ tādisaṃ saṃyataṃ brahmacāriṃ,_  
_Kasmā tuvaṃ brūsi gahappatīti”._  

</Paragraph>

</Division>
<Division number="219" book="ja1" bookNo="22" page="335">

<Paragraph number="3079" type="verses" book="ja1">
_“Tālā ca mūlā ca phalā ca deva,_  
_Pariggahā vividhā santimassa;_  
_Te rakkhati gopayatappamatto,_  
_Tasmā ahaṃ brūmi gahappatīti”._  

</Paragraph>

</Division>
<Division number="220" book="ja1" bookNo="22" page="335">

<Paragraph number="3080" type="verses" book="ja1">
_“Saccaṃ kho etaṃ vadasi kumāra,_  
_Pariggahā vividhā santimassa;_  
_Te rakkhati gopayatappamatto,_  
_Sa brāhmaṇo gahapati tena hoti”._  

</Paragraph>

</Division>
<Division number="221" book="ja1" bookNo="22" page="335">

<Paragraph number="3081" type="verses" book="ja1">
_“Suṇantu mayhaṃ parisā samāgatā,_  
_Sanegamā jānapadā ca sabbe;_  
_Bālāyaṃ bālassa vaco nisamma,_  
_Ahetunā ghātayate maṃ janindo._  

</Paragraph>

</Division>
<Division number="222" book="ja1" bookNo="22" page="335">

<Paragraph number="3082" type="verses" book="ja1">
_Daḷhasmi mūle visaṭe virūḷhe,_  
_Dunnikkayo veḷu pasākhajāto;_  
_Vandāmi pādāni tava janinda,_  
_Anujāna maṃ pabbajissāmi deva”._  

</Paragraph>

</Division>
<Division number="223" book="ja1" bookNo="22" page="335">

<Paragraph number="3083" type="verses" book="ja1">
_“Bhuñjassu bhoge vipule kumāra,_  
_Sabbañca te issariyaṃ dadāmi;_  
_Ajjeva tvaṃ kurūnaṃ hohi rājā,_  
_Mā pabbajī pabbajjā hi dukkhā”._  

</Paragraph>

</Division>
<Division number="224" book="ja1" bookNo="22" page="336">

<Paragraph number="3084" type="verses" book="ja1">
_“Kinnūdha deva tavamatthi bhogā,_  
_Pubbevahaṃ devaloke ramissaṃ;_  
_Rūpehi saddehi atho rasehi,_  
_Gandhehi phassehi manoramehi._  

</Paragraph>

</Division>
<Division number="225" book="ja1" bookNo="22" page="336">

<Paragraph number="3085" type="verses" book="ja1">
_Bhuttā ca me bhogā tidivasmiṃ deva,_  
_Parivārito accharānaṃ gaṇena;_  
_Tuvañca bālaṃ paraneyyaṃ viditvā,_  
_Na tādise rājakule vaseyyaṃ”._  

</Paragraph>

</Division>
<Division number="226" book="ja1" bookNo="22" page="336">

<Paragraph number="3086" type="verses" book="ja1">
_“Sacāhaṃ bālo paraneyyo asmi,_  
_Ekāparādhaṃ khama putta mayhaṃ;_  
_Punapi ce edisakaṃ bhaveyya,_  
_Yathāmatiṃ somanassa karohi”._  

</Paragraph>

</Division>
<Division number="227" book="ja1" bookNo="22" page="336">

<Paragraph number="3087" type="verses" book="ja1">
_“Anisamma kataṃ kammaṃ,_  
_anavatthāya cintitaṃ;_  
_Bhesajjasseva vebhaṅgo,_  
_vipāko hoti pāpako._  

</Paragraph>

</Division>
<Division number="228" book="ja1" bookNo="22" page="336">

<Paragraph number="3088" type="verses" book="ja1">
_Nisamma ca kataṃ kammaṃ,_  
_sammāvatthāya cintitaṃ;_  
_Bhesajjasseva sampatti,_  
_vipāko hoti bhadrako._  

</Paragraph>

</Division>
<Division number="229" book="ja1" bookNo="22" page="336">

<Paragraph number="3089" type="verses" book="ja1">
_Alaso gihī kāmabhogī na sādhu,_  
_Asaññato pabbajito na sādhu;_  
_Rājā na sādhu anisammakārī,_  
_Yo paṇḍito kodhano taṃ na sādhu._  

</Paragraph>

</Division>
<Division number="230" book="ja1" bookNo="22" page="336">

<Paragraph number="3090" type="verses" book="ja1">
_Nisamma khattiyo kayirā,_  
_nānisamma disampati;_  
_Nisammakārino rāja,_  
_yaso kitti ca vaḍḍhati._  

</Paragraph>

</Division>
<Division number="231" book="ja1" bookNo="22" page="336">

<Paragraph number="3091" type="verses" book="ja1">
_Nisamma daṇḍaṃ paṇayeyya issaro,_  
_Vegā kataṃ tappati bhūmipāla;_  
_Sammāpaṇīdhī ca narassa atthā,_  
_Anānutappā te bhavanti pacchā._  

</Paragraph>

</Division>
<Division number="232" book="ja1" bookNo="22" page="337">

<Paragraph number="3092" type="verses" book="ja1">
_Anānutappāni hi ye karonti,_  
_Vibhajja kammāyatanāni loke;_  
_Viññuppasatthāni sukhudrayāni,_  
_Bhavanti buddhānumatāni tāni._  

</Paragraph>

</Division>
<Division number="233" book="ja1" bookNo="22" page="337">

<Paragraph number="3093" type="verses" book="ja1">
_Āgacchuṃ dovārikā khaggabandhā,_  
_Kāsāviyā hantu mamaṃ janinda;_  
_Mātuñca aṅkasmimahaṃ nisinno,_  
_Ākaḍḍhito sahasā tehi deva._  

</Paragraph>

</Division>
<Division number="234" book="ja1" bookNo="22" page="337">

<Paragraph number="3094" type="verses" book="ja1">
_Kaṭukañhi sambādhaṃ sukicchaṃ patto,_  
_Madhurampi yaṃ jīvitaṃ laddha rāja;_  
_Kicchenahaṃ ajja vadhā pamutto,_  
_Pabbajjamevābhimanohamasmi”._  

</Paragraph>

</Division>
<Division number="235" book="ja1" bookNo="22" page="337">

<Paragraph number="3095" type="verses" book="ja1">
_“Putto tavāyaṃ taruṇo sudhamme,_  
_Anukampako somanasso kumāro;_  
_Taṃ yācamāno na labhāmi svajja,_  
_Arahasi naṃ yācitave tuvampi”._  

</Paragraph>

</Division>
<Division number="236" book="ja1" bookNo="22" page="337">

<Paragraph number="3096" type="verses" book="ja1">
_“Ramassu bhikkhācariyāya putta,_  
_Nisamma dhammesu paribbajassu;_  
_Sabbesu bhūtesu nidhāya daṇḍaṃ,_  
_Anindito brahmamupehi ṭhānaṃ”._  

</Paragraph>

</Division>
<Division number="237" book="ja1" bookNo="22" page="337">

<Paragraph number="3097" type="verses" book="ja1">
_“Acchera rūpaṃ vata yādisañca,_  
_Dukkhitaṃ maṃ dukkhāpayase sudhamme;_  
_Yācassu puttaṃ iti vuccamānā,_  
_Bhiyyova ussāhayase kumāraṃ”._  

</Paragraph>

</Division>
<Division number="238" book="ja1" bookNo="22" page="337">

<Paragraph number="3098" type="verses" book="ja1">
_“Ye vippamuttā anavajjabhogino,_  
_Parinibbutā lokamimaṃ caranti;_  
_Tamariyamaggaṃ paṭipajjamānaṃ,_  
_Na ussahe vārayituṃ kumāraṃ”._  

</Paragraph>

</Division>
<Division number="239" book="ja1" bookNo="22" page="338">

<Paragraph number="3099" type="verses" book="ja1">
_“Addhā have sevitabbā sapaññā,_  
_Bahussutā ye bahuṭhānacintino;_  
_Yesāyaṃ sutvāna subhāsitāni,_  
_Appossukkā vītasokā sudhammā”ti._  


</Paragraph>
</Division>

<Paragraph number="3100" type="center" book="ja1">
Somanassajātakaṃ navamaṃ.
</Paragraph>
