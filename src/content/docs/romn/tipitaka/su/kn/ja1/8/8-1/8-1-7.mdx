<Paragraph number="1738" type="verses" book="ja1">
_“Yo indriyānaṃ kāmena,_  
_vasaṃ nārada gacchati;_  
_So pariccajjubho loke,_  
_jīvantova visussati._  

</Paragraph>

</Division>
<Division number="61" book="ja1" bookNo="22" page="317">

<Paragraph number="1739" type="verses" book="ja1">
_Sukhassānantaraṃ dukkhaṃ,_  
_dukkhassānantaraṃ sukhaṃ;_  
_Sosi patto sukhā dukkhaṃ,_  
_pāṭikaṅkha varaṃ sukhaṃ._  

</Paragraph>

</Division>
<Division number="62" book="ja1" bookNo="22" page="317">

<Paragraph number="1740" type="verses" book="ja1">
_Kicchakāle kicchasaho,_  
_yo kicchaṃ nātivattati;_  
_Sa kicchantaṃ sukhaṃ dhīro,_  
_yogaṃ samadhigacchati._  

</Paragraph>

</Division>
<Division number="63" book="ja1" bookNo="22" page="317">

<Paragraph number="1741" type="verses" book="ja1">
_Na heva kāmāna kāmā,_  
_nānatthā nātthakāraṇā;_  
_Na katañca niraṅkatvā,_  
_dhammā cavitumarahasi”._  

</Paragraph>

</Division>
<Division number="64" book="ja1" bookNo="22" page="317">

<Paragraph number="1742" type="verses" book="ja1">
_“Dakkhaṃ gahapatī sādhu,_  
_saṃvibhajjañca bhojanaṃ;_  
_Ahāso atthalābhesu,_  
_atthabyāpatti abyatho”._  

</Paragraph>

</Division>
<Division number="65" book="ja1" bookNo="22" page="317">

<Paragraph number="1743" type="verses" book="ja1">
_“Ettāvatetaṃ paṇḍiccaṃ,_  
_api so devilo bravi;_  
_Na yito kiñci pāpiyo,_  
_yo indriyānaṃ vasaṃ vaje”._  

</Paragraph>

</Division>
<Division number="66" book="ja1" bookNo="22" page="317">

<Paragraph number="1744" type="verses" book="ja1">
_“Amittānaṃva hatthatthaṃ,_  
_sivi pappoti māmiva;_  
_Kammaṃ vijjañca dakkheyyaṃ,_  
_vivāhaṃ sīlamaddavaṃ;_  
_Ete ca yase hāpetvā,_  
_nibbatto sehi kammehi._  

</Paragraph>

</Division>
<Division number="67" book="ja1" bookNo="22" page="318">

<Paragraph number="1745" type="verses" book="ja1">
_Sohaṃ sahassajīnova,_  
_abandhu aparāyaṇo;_  
_Ariyadhammā apakkanto,_  
_yathā peto tathevahaṃ._  

</Paragraph>

</Division>
<Division number="68" book="ja1" bookNo="22" page="318">

<Paragraph number="1746" type="verses" book="ja1">
_Sukhakāme dukkhāpetvā,_  
_āpannosmi padaṃ imaṃ;_  
_So sukhaṃ nādhigacchāmi,_  
_ṭhito bhāṇumatāmivā”ti._  


</Paragraph>
</Division>

<Paragraph number="1747" type="center" book="ja1">
Indriyajātakaṃ sattamaṃ.
</Paragraph>
