<Paragraph number="1121" type="verses" book="ja1">
_“Sabbāyasaṃ kūṭamatippamāṇaṃ,_  
_Paggayha yo tiṭṭhasi antalikkhe;_  
_Rakkhāya me tvaṃ vihito nusajja,_  
_Udāhu me cetayase vadhāya”._  

</Paragraph>

</Division>
<Division number="186" book="ja1" bookNo="22" page="381">

<Paragraph number="1122" type="verses" book="ja1">
_“Dūto ahaṃ rājidha rakkhasānaṃ,_  
_Vadhāya tuyhaṃ pahitohamasmi;_  
_Indo ca taṃ rakkhati devarājā,_  
_Tenuttamaṅgaṃ na te phālayāmi”._  

</Paragraph>

</Division>
<Division number="187" book="ja1" bookNo="22" page="381">

<Paragraph number="1123" type="verses" book="ja1">
_“Sace ca maṃ rakkhati devarājā,_  
_Devānamindo maghavā sujampati;_  
_Kāmaṃ pisācā vinadantu sabbe,_  
_Na santase rakkhasiyā pajāya._  

</Paragraph>

</Division>
<Division number="188" book="ja1" bookNo="22" page="381">

<Paragraph number="1124" type="verses" book="ja1">
_Kāmaṃ kandantu kumbhaṇḍā,_  
_Sabbe paṃsupisācakā;_  
_Nālaṃ pisācā yuddhāya,_  
_Mahatī sā vibhiṃsikā”ti._  


</Paragraph>
</Division>

<Paragraph number="1125" type="center" book="ja1">
Ayakūṭajātakaṃ sattamaṃ.
</Paragraph>
