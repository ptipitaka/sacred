<Paragraph number="1896" type="verses" book="ja1">
_“Yo te puttake akhādi,_  
_dinnabhatto adūsake;_  
_Tasmiṃ dāṭhaṃ nipātehi,_  
_mā te muccittha jīvato”._  

</Paragraph>

</Division>
<Division number="106" book="ja1" bookNo="22" page="372">

<Paragraph number="1897" type="verses" book="ja1">
_“Ākiṇṇaluddo puriso,_  
_dhāticelaṃva makkhito;_  
_Padesaṃ taṃ na passāmi,_  
_yattha dāṭhaṃ nipātaye._  

</Paragraph>

</Division>
<Division number="107" book="ja1" bookNo="22" page="372">

<Paragraph number="1898" type="verses" book="ja1">
_Akataññussa posassa,_  
_niccaṃ vivaradassino;_  
_Sabbañce pathaviṃ dajjā,_  
_neva naṃ abhirādhaye”._  

</Paragraph>

</Division>
<Division number="108" book="ja1" bookNo="22" page="372">

<Paragraph number="1899" type="verses" book="ja1">
_“Kiṃ nu subāhu taramānarūpo,_  
_Paccāgatosi saha māṇavena;_  
_Kiṃ kiccamatthaṃ idhamatthi tuyhaṃ,_  
_Akkhāhi me pucchito etamatthaṃ”._  

</Paragraph>

</Division>
<Division number="109" book="ja1" bookNo="22" page="372">

<Paragraph number="1900" type="verses" book="ja1">
_“Yo te sakhā daddaro sādhurūpo,_  
_Tassa vadhaṃ parisaṅkāmi ajja;_  
_Purisassa kammāyatanāni sutvā,_  
_Nāhaṃ sukhiṃ daddaraṃ ajja maññe”._  

</Paragraph>

</Division>
<Division number="110" book="ja1" bookNo="22" page="372">

<Paragraph number="1901" type="verses" book="ja1">
_“Kānissa kammāyatanāni assu,_  
_Purisassa vuttisamodhānatāya;_  
_Kaṃ vā paṭiññaṃ purisassa sutvā,_  
_Parisaṅkasi daddaraṃ māṇavena”._  

</Paragraph>

</Division>
<Division number="111" book="ja1" bookNo="22" page="372">

<Paragraph number="1902" type="verses" book="ja1">
_“Ciṇṇā kaliṅgā caritā vaṇijjā,_  
_Vettācaro saṅkupathopi ciṇṇo;_  
_Naṭehi ciṇṇaṃ saha vākurehi,_  
_Daṇḍena yuddhampi samajjamajjhe._  

</Paragraph>

</Division>
<Division number="112" book="ja1" bookNo="22" page="373">

<Paragraph number="1903" type="verses" book="ja1">
_Baddhā kulīkā mitamāḷhakena,_  
_Akkhā jitā saṃyamo abbhatīto;_  
_Abbāhitaṃ pubbakaṃ aḍḍharattaṃ,_  
_Hatthā daḍḍhā piṇḍapaṭiggahena._  

</Paragraph>

</Division>
<Division number="113" book="ja1" bookNo="22" page="373">

<Paragraph number="1904" type="verses" book="ja1">
_Tānissa kammāyatanāni assu,_  
_Purisassa vuttisamodhānatāya;_  
_Yathā ayaṃ dissati lomapiṇḍo,_  
_Gāvo hatā kiṃ pana daddarassā”ti. (1348)_  


</Paragraph>
</Division>

Daddarajātakaṃ dvādasamaṃ.



</Paragraph>

<Paragraph number="1906" type="center" book="ja1">

</Paragraph>

Tassuddānaṃ


</Paragraph>

<Paragraph number="1908" type="verses" book="ja1">
_Varagijjha samajjana haṃsavaro,_  
_Nidhisavhaya hārita pāṭaliko;_  
_Ajarāmara dhaṅka titikkha kuto,_  
_Atha dvādasa pekkhana daddaribhīti._
</Paragraph>
