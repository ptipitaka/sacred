<Paragraph number="1292" type="verses" book="ja1">
_“Kuhiṃ gatā kattha gatā,_  
_iti lālappatī jano;_  
_Ahameveko jānāmi,_  
_udapāne mūsikā hatā._  

</Paragraph>

<Division number="124" book="ja1" bookNo="22" page="290">

<Paragraph number="1293" type="verses" book="ja1">
_Yañcetaṃ iti cīti ca,_  
_gadrabhova nivattasi;_  
_Udapāne mūsikaṃ hantvā,_  
_yavaṃ bhakkhetumicchasi._  

</Paragraph>

</Division>
<Division number="125" book="ja1" bookNo="22" page="290">

<Paragraph number="1294" type="verses" book="ja1">
_Daharo cāsi dummedha,_  
_paṭhamuppattiko susu;_  
_Dīghañcetaṃ samāsajja,_  
_na te dassāmi jīvitaṃ._  

</Paragraph>

</Division>
<Division number="126" book="ja1" bookNo="22" page="290">

<Paragraph number="1295" type="verses" book="ja1">
_Nāntalikkhabhavanena,_  
_nāṅgaputtapinena vā;_  
_Puttena hi patthayito,_  
_silokehi pamocito._  

</Paragraph>

</Division>
<Division number="127" book="ja1" bookNo="22" page="291">

<Paragraph number="1296" type="verses" book="ja1">
_Sabbaṃ sutamadhīyetha,_  
_hīnamukkaṭṭhamajjhimaṃ;_  
_Sabbassa atthaṃ jāneyya,_  
_na ca sabbaṃ payojaye;_  
_Hoti tādisako kālo,_  
_yattha atthāvahaṃ sutan”ti._  


</Paragraph>
