<Paragraph number="1748" type="verses" book="ja1">
_“Ādittasmiṃ agārasmiṃ,_  
_yaṃ nīharati bhājanaṃ;_  
_Taṃ tassa hoti atthāya,_  
_no ca yaṃ tattha ḍayhati._  

</Paragraph>

</Division>
<Division number="70" book="ja1" bookNo="22" page="318">

<Paragraph number="1749" type="verses" book="ja1">
_Evamādīpito loko,_  
_jarāya maraṇena ca;_  
_Nīharetheva dānena,_  
_dinnaṃ hoti sunīhataṃ”._  

</Paragraph>

</Division>
<Division number="71" book="ja1" bookNo="22" page="318">

<Paragraph number="1750" type="verses" book="ja1">
_“Yo dhammaladdhassa dadāti dānaṃ,_  
_Uṭṭhānaviriyādhigatassa jantu;_  
_Atikkamma so vetaraṇiṃ yamassa,_  
_Dibbāni ṭhānāni upeti macco._  

</Paragraph>

</Division>
<Division number="72" book="ja1" bookNo="22" page="318">

<Paragraph number="1751" type="verses" book="ja1">
_Dānañca yuddhañca samānamāhu,_  
_Appāpi santā bahuke jinanti;_  
_Appampi ce saddahāno dadāti,_  
_Teneva so hoti sukhī parattha._  

</Paragraph>

</Division>
<Division number="73" book="ja1" bookNo="22" page="318">

<Paragraph number="1752" type="verses" book="ja1">
_Viceyya dānaṃ sugatappasatthaṃ,_  
_Ye dakkhiṇeyyā idha jīvaloke;_  
_Etesu dinnāni mahapphalāni,_  
_Bījāni vuttāni yathā sukhette._  

</Paragraph>

</Division>
<Division number="74" book="ja1" bookNo="22" page="318">

<Paragraph number="1753" type="verses" book="ja1">
_Yo pāṇabhūtāni aheṭhayaṃ caraṃ,_  
_Parūpavādā na karoti pāpaṃ;_  
_Bhīruṃ pasaṃsanti na tattha sūraṃ,_  
_Bhayā hi santo na karonti pāpaṃ._  

</Paragraph>

</Division>
<Division number="75" book="ja1" bookNo="22" page="318">

<Paragraph number="1754" type="verses" book="ja1">
_Hīnena brahmacariyena,_  
_khattiye upapajjati;_  
_Majjhimena ca devattaṃ,_  
_uttamena visujjhati._  

</Paragraph>

</Division>
<Division number="76" book="ja1" bookNo="22" page="318">

<Paragraph number="1755" type="verses" book="ja1">
_Addhā hi dānaṃ bahudhā pasatthaṃ,_  
_Dānā ca kho dhammapadaṃva seyyo;_  
_Pubbeva hi pubbatareva santo,_  
_Nibbānamevajjhagamuṃ sapaññā”ti._  


</Paragraph>
</Division>

<Paragraph number="1756" type="center" book="ja1">
Ādittajātakaṃ aṭṭhamaṃ.
</Paragraph>
