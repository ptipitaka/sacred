<Division number="36" book="ja1">

<Paragraph number="1712" type="verses" book="ja1">
_“Aṅgārajātā pathavī,_  
_kukkuḷānugatā mahī;_  
_Atha gāyasi vattāni,_  
_na taṃ tapati ātapo._  

</Paragraph>

</Division>
<Division number="37" book="ja1">

<Paragraph number="1713" type="verses" book="ja1">
_Uddhaṃ tapati ādicco,_  
_adho tapati vālukā;_  
_Atha gāyasi vattāni,_  
_na taṃ tapati ātapo”._  

</Paragraph>

</Division>
<Division number="38" book="ja1">

<Paragraph number="1714" type="verses" book="ja1">
_“Na maṃ tapati ātapo,_  
_ātapā tapayanti maṃ;_  
_Atthā hi vividhā rāja,_  
_te tapanti na ātapo”._  

</Paragraph>

</Division>
<Division number="39" book="ja1">

<Paragraph number="1715" type="verses" book="ja1">
_“Addasaṃ kāma te mūlaṃ,_  
_saṅkappā kāma jāyasi;_  
_Na taṃ saṅkappayissāmi,_  
_evaṃ kāma na hehisi”._  

</Paragraph>

</Division>
<Division number="40" book="ja1">

<Paragraph number="1716" type="verses" book="ja1">
_“Appāpi kāmā na alaṃ,_  
_bahūhipi na tappati;_  
_Ahahā bālalapanā,_  
_parivajjetha jaggato”._  

</Paragraph>

</Division>
<Division number="41" book="ja1">

<Paragraph number="1717" type="verses" book="ja1">
_“Appassa kammassa phalaṃ mamedaṃ,_  
_Udayo ajjhāgamā mahattapattaṃ;_  
_Suladdhalābho vata māṇavassa,_  
_Yo pabbajī kāmarāgaṃ pahāya”._  

</Paragraph>

</Division>
<Division number="42" book="ja1">

<Paragraph number="1718" type="verses" book="ja1">
_“Tapasā pajahanti pāpakammaṃ,_  
_Tapasā nhāpitakumbhakārabhāvaṃ;_  
_Tapasā abhibhuyya gaṅgamāla,_  
_Nāmenālapasajja brahmadattaṃ”._  

</Paragraph>

</Division>
<Division number="43" book="ja1">

<Paragraph number="1719" type="verses" book="ja1">
_“Sandiṭṭhikameva ‘amma’ passatha,_  
_Khantīsoraccassa ayaṃ vipāko;_  
_Yo sabbajanassa vanditohu,_  
_Taṃ vandāma sarājikā samaccā”._  

</Paragraph>

</Division>
<Division number="44" book="ja1">

<Paragraph number="1720" type="verses" book="ja1">
_“Mā kiñci avacuttha gaṅgamālaṃ,_  
_Muninaṃ monapathesu sikkhamānaṃ;_  
_Eso hi atari aṇṇavaṃ,_  
_Yaṃ taritvā caranti vītasokā”ti._  


</Paragraph>
</Division>

<Paragraph number="1721" type="center" book="ja1">
Gaṅgamālajātakaṃ pañcamaṃ.
</Paragraph>
