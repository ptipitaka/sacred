<Division number="135" book="ja1">

<Paragraph number="1639" type="verses" book="ja1">
_“Kodha jāgarataṃ sutto,_  
_kodha suttesu jāgaro;_  
_Ko mametaṃ vijānāti,_  
_ko taṃ paṭibhaṇāti me”._  

</Paragraph>

</Division>
<Division number="136" book="ja1">

<Paragraph number="1640" type="verses" book="ja1">
_“Ahaṃ jāgarataṃ sutto,_  
_ahaṃ suttesu jāgaro;_  
_Ahametaṃ vijānāmi,_  
_ahaṃ paṭibhaṇāmi te”._  

</Paragraph>

</Division>
<Division number="137" book="ja1">

<Paragraph number="1641" type="verses" book="ja1">
_“Kathaṃ jāgarataṃ sutto,_  
_kathaṃ suttesu jāgaro;_  
_Kathaṃ etaṃ vijānāsi,_  
_kathaṃ paṭibhaṇāsi me”._  

</Paragraph>

</Division>
<Division number="138" book="ja1">

<Paragraph number="1642" type="verses" book="ja1">
_“Ye dhammaṃ nappajānanti,_  
_saṃyamoti damoti ca;_  
_Tesu suppamānesu,_  
_ahaṃ jaggāmi devate._  

</Paragraph>

</Division>
<Division number="139" book="ja1">

<Paragraph number="1643" type="verses" book="ja1">
_Yesaṃ rāgo ca doso ca,_  
_avijjā ca virājitā;_  
_Tesu jāgaramānesu,_  
_ahaṃ suttosmi devate._  

</Paragraph>

</Division>
<Division number="140" book="ja1">

<Paragraph number="1644" type="verses" book="ja1">
_Evaṃ jāgarataṃ sutto,_  
_evaṃ suttesu jāgaro;_  
_Evametaṃ vijānāmi,_  
_evaṃ paṭibhaṇāmi te”._  

</Paragraph>

</Division>
<Division number="141" book="ja1">

<Paragraph number="1645" type="verses" book="ja1">
_“Sādhu jāgarataṃ sutto,_  
_sādhu suttesu jāgaro;_  
_Sādhumetaṃ vijānāsi,_  
_sādhu paṭibhaṇāsi me”ti._  


</Paragraph>
</Division>

<Paragraph number="1646" type="center" book="ja1">
Jāgarajātakaṃ navamaṃ.
</Paragraph>
