<Paragraph number="862" type="verses" book="ja1">
_“Ucce sakuṇa ḍemāna,_  
_pattayāna vihaṅgama;_  
_Vajjāsi kho tvaṃ vāmūruṃ,_  
_ciraṃ kho sā karissati._  

</Paragraph>

</Division>
<Division number="140" book="ja1" bookNo="22" page="220">

<Paragraph number="863" type="verses" book="ja1">
_Idaṃ kho sā na jānāti,_  
_asiṃ sattiñca oḍḍitaṃ;_  
_Sā caṇḍī kāhati kodhaṃ,_  
_taṃ me tapati no idaṃ._  

</Paragraph>

</Division>

<Paragraph number="864" type="verses" book="ja1">
_Esa uppalasannāho,_  
_nikkhañcussīsakohitaṃ;_  
_Kāsikañca muduṃ vatthaṃ,_  
_tappetu dhanikā piyā”ti._  


</Paragraph>
</Division>
