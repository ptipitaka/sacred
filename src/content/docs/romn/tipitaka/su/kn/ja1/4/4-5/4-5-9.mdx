<Division number="193" book="ja1">

<Paragraph number="1131" type="verses" book="ja1">
_“Neva itthīsu sāmaññaṃ,_  
_nāpi bhakkhesu sārathi;_  
_Athassa sandhibhedassa,_  
_passa yāva sucintitaṃ._  

</Paragraph>

</Division>
<Division number="194" book="ja1">

<Paragraph number="1132" type="verses" book="ja1">
_Asi tikkhova maṃsamhi,_  
_pesuññaṃ parivattati;_  
_Yatthūsabhañca sīhañca,_  
_bhakkhayanti migādhamā._  

</Paragraph>

</Division>
<Division number="195" book="ja1">

<Paragraph number="1133" type="verses" book="ja1">
_Imaṃ so sayanaṃ seti,_  
_yamimaṃ passasi sārathi;_  
_Yo vācaṃ sandhibhedassa,_  
_pisuṇassa nibodhati._  

</Paragraph>

</Division>
<Division number="196" book="ja1">

<Paragraph number="1134" type="verses" book="ja1">
_Te janā sukhamedhanti,_  
_narā saggagatāriva;_  
_Ye vācaṃ sandhibhedassa,_  
_nāvabodhanti sārathī”ti._  


</Paragraph>
</Division>

<Paragraph number="1135" type="center" book="ja1">
Sandhibhedajātakaṃ navamaṃ.
</Paragraph>
