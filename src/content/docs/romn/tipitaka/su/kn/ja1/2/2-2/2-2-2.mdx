<Paragraph number="377" type="verses" book="ja1">
_“Na santhavasmā paramatthi pāpiyo,_  
_Yo santhavo kāpurisena hoti;_  
_Santappito sappinā pāyasena,_  
_Kicchākataṃ paṇṇakuṭiṃ adayhi._  

</Paragraph>

<Division number="24" book="ja1">

<Paragraph number="378" type="verses" book="ja1">
_Na santhavasmā paramatthi seyyo,_  
_Yo santhavo sappurisena hoti;_  
_Sīhassa byagghassa ca dīpino ca,_  
_Sāmā mukhaṃ lehati santhavenā”ti._  


</Paragraph>
</Division>
