<Paragraph number="3017" type="verses" book="ja1">
_Migaluddo mahārājā,_  
_pañcālānaṃ rathesabho;_  
_Nikkhanto saha senāya,_  
_ogaṇo vanamāgamā._  

</Paragraph>

</Division>
<Division number="160" book="ja1">

<Paragraph number="3018" type="verses" book="ja1">
_Tatthaddasā araññasmiṃ,_  
_takkarānaṃ kuṭiṃ kataṃ;_  
_Tassā kuṭiyā nikkhamma,_  
_suvo luddāni bhāsati._  

</Paragraph>

</Division>
<Division number="161" book="ja1">

<Paragraph number="3019" type="verses" book="ja1">
_“Sampannavāhano poso,_  
_yuvā sammaṭṭhakuṇḍalo;_  
_Sobhati lohituṇhīso,_  
_divā sūriyova bhāsati._  

</Paragraph>

</Division>
<Division number="162" book="ja1">

<Paragraph number="3020" type="verses" book="ja1">
_Majjhanhike sampatike,_  
_sutto rājā sasārathi;_  
_Handassābharaṇaṃ sabbaṃ,_  
_gaṇhāma sāhasā mayaṃ._  

</Paragraph>

</Division>
<Division number="163" book="ja1">

<Paragraph number="3021" type="verses" book="ja1">
_Nisīthepi raho dāni,_  
_Sutto rājā sasārathi;_  
_Ādāya vatthaṃ maṇikuṇḍalañca,_  
_Hantvāna sākhāhi avattharāma”._  

</Paragraph>

</Division>
<Division number="164" book="ja1">

<Paragraph number="3022" type="verses" book="ja1">
_“Kiṃ nu ummattarūpova,_  
_sattigumba pabhāsasi;_  
_Durāsadā hi rājāno,_  
_aggi pajjalito yathā”._  

</Paragraph>

</Division>
<Division number="165" book="ja1">

<Paragraph number="3023" type="verses" book="ja1">
_“Atha tvaṃ patikolamba,_  
_matto thullāni gajjasi;_  
_Mātari mayhaṃ naggāya,_  
_kiṃ nu tvaṃ vijigucchase”._  

</Paragraph>

</Division>
<Division number="166" book="ja1">

<Paragraph number="3024" type="verses" book="ja1">
_“Uṭṭhehi samma taramāno,_  
_rathaṃ yojehi sārathi;_  
_Sakuṇo me na ruccati,_  
_aññaṃ gacchāma assamaṃ”._  

</Paragraph>

</Division>
<Division number="167" book="ja1">

<Paragraph number="3025" type="verses" book="ja1">
_“Yutto ratho mahārāja,_  
_yutto ca balavāhano;_  
_Adhitiṭṭha mahārāja,_  
_aññaṃ gacchāma assamaṃ”._  

</Paragraph>

</Division>
<Division number="168" book="ja1">

<Paragraph number="3026" type="verses" book="ja1">
_“Ko numeva gatā sabbe,_  
_ye asmiṃ paricārakā;_  
_Esa gacchati pañcālo,_  
_mutto tesaṃ adassanā._  

</Paragraph>

</Division>
<Division number="169" book="ja1">

<Paragraph number="3027" type="verses" book="ja1">
_Kodaṇḍakāni gaṇhatha,_  
_sattiyo tomarāni ca;_  
_Esa gacchati pañcālo,_  
_mā vo muñcittha jīvataṃ”._  

</Paragraph>

</Division>
<Division number="170" book="ja1">

<Paragraph number="3028" type="verses" book="ja1">
_Athāparo paṭinandittha,_  
_suvo lohitatuṇḍako;_  
_“Svāgataṃ te mahārāja,_  
_atho te adurāgataṃ;_  
_Issarosi anuppatto,_  
_yaṃ idhatthi pavedaya._  

</Paragraph>

</Division>
<Division number="171" book="ja1">

<Paragraph number="3029" type="verses" book="ja1">
_Tindukāni piyālāni,_  
_madhuke kāsumāriyo;_  
_Phalāni khuddakappāni,_  
_bhuñja rāja varaṃ varaṃ._  

</Paragraph>

</Division>
<Division number="172" book="ja1">

<Paragraph number="3030" type="verses" book="ja1">
_Idampi pānīyaṃ sītaṃ,_  
_ābhataṃ girigabbharā;_  
_Tato piva mahārāja,_  
_sace tvaṃ abhikaṅkhasi._  

</Paragraph>

</Division>
<Division number="173" book="ja1">

<Paragraph number="3031" type="verses" book="ja1">
_Araññaṃ uñchāya gatā,_  
_ye asmiṃ paricārakā;_  
_Sayaṃ uṭṭhāya gaṇhavho,_  
_hatthā me natthi dātave”._  

</Paragraph>

</Division>
<Division number="174" book="ja1">

<Paragraph number="3032" type="verses" book="ja1">
_“Bhaddako vatayaṃ pakkhī,_  
_dijo paramadhammiko;_  
_Atheso itaro pakkhī,_  
_suvo luddāni bhāsati._  

</Paragraph>

</Division>
<Division number="175" book="ja1">

<Paragraph number="3033" type="verses" book="ja1">
_‘Etaṃ hanatha bandhatha,_  
_Mā vo muñcittha jīvataṃ’;_  
_Iccevaṃ vilapantassa,_  
_Sotthiṃ pattosmi assamaṃ”._  

</Paragraph>

</Division>
<Division number="176" book="ja1">

<Paragraph number="3034" type="verses" book="ja1">
_“Bhātarosma mahārāja,_  
_sodariyā ekamātukā;_  
_Ekarukkhasmiṃ saṃvaḍḍhā,_  
_nānākhettagatā ubho._  

</Paragraph>

</Division>
<Division number="177" book="ja1">

<Paragraph number="3035" type="verses" book="ja1">
_Sattigumbo ca corānaṃ,_  
_ahañca isīnaṃ idha;_  
_Asataṃ so, sataṃ ahaṃ,_  
_tena dhammena no vinā”._  

</Paragraph>

</Division>
<Division number="178" book="ja1">

<Paragraph number="3036" type="verses" book="ja1">
_“Tattha vadho ca bandho ca,_  
_nikatī vañcanāni ca;_  
_Ālopā sāhasākārā,_  
_tāni so tattha sikkhati._  

</Paragraph>

</Division>
<Division number="179" book="ja1">

<Paragraph number="3037" type="verses" book="ja1">
_Idha saccañca dhammo ca,_  
_Ahiṃsā saṃyamo damo;_  
_Āsanūdakadāyīnaṃ,_  
_Aṅke vaddhosmi bhāradha”._  

</Paragraph>

</Division>
<Division number="180" book="ja1">

<Paragraph number="3038" type="verses" book="ja1">
_“Yaṃ yañhi rāja bhajati,_  
_santaṃ vā yadi vā asaṃ;_  
_Sīlavantaṃ visīlaṃ vā,_  
_vasaṃ tasseva gacchati._  

</Paragraph>

</Division>
<Division number="181" book="ja1">

<Paragraph number="3039" type="verses" book="ja1">
_Yādisaṃ kurute mittaṃ,_  
_yādisaṃ cūpasevati;_  
_Sopi tādisako hoti,_  
_sahavāso hi tādiso._  

</Paragraph>

</Division>
<Division number="182" book="ja1">

<Paragraph number="3040" type="verses" book="ja1">
_Sevamāno sevamānaṃ,_  
_samphuṭṭho samphusaṃ paraṃ;_  
_Saro diddho kalāpaṃva,_  
_alittamupalimpati;_  
_Upalepabhayā dhīro,_  
_neva pāpasakhā siyā._  

</Paragraph>

</Division>
<Division number="183" book="ja1">

<Paragraph number="3041" type="verses" book="ja1">
_Pūtimacchaṃ kusaggena,_  
_yo naro upanayhati;_  
_Kusāpi pūti vāyanti,_  
_evaṃ bālūpasevanā._  

</Paragraph>

</Division>
<Division number="184" book="ja1">

<Paragraph number="3042" type="verses" book="ja1">
_Tagarañca palāsena,_  
_yo naro upanayhati;_  
_Pattāpi surabhi vāyanti,_  
_evaṃ dhīrūpasevanā._  

</Paragraph>

</Division>
<Division number="185" book="ja1">

<Paragraph number="3043" type="verses" book="ja1">
_Tasmā pattapuṭasseva,_  
_ñatvā sampākamattano;_  
_Asante nopaseveyya,_  
_sante seveyya paṇḍito;_  
_Asanto nirayaṃ nenti,_  
_santo pāpenti suggatin”ti._  


</Paragraph>
</Division>

<Paragraph number="3044" type="center" book="ja1">
Sattigumbajātakaṃ sattamaṃ.
