<Division number="26" book="ja1">

<Paragraph number="1347" type="verses" book="ja1">
_“Āsāvatī nāma latā,_  
_jātā cittalatāvane;_  
_Tassā vassasahassena,_  
_ekaṃ nibbattate phalaṃ._  

</Paragraph>

</Division>
<Division number="27" book="ja1">

<Paragraph number="1348" type="verses" book="ja1">
_Taṃ devā payirupāsanti,_  
_tāva dūraphalaṃ satiṃ;_  
_Āsīseva tuvaṃ rāja,_  
_āsā phalavatī sukhā._  

</Paragraph>

</Division>
<Division number="28" book="ja1">

<Paragraph number="1349" type="verses" book="ja1">
_Āsīsateva so pakkhī,_  
_āsīsateva so dijo;_  
_Tassa cāsā samijjhati,_  
_tāva dūragatā satī;_  
_Āsīseva tuvaṃ rāja,_  
_āsā phalavatī sukhā”._  

</Paragraph>

</Division>
<Division number="29" book="ja1">

<Paragraph number="1350" type="verses" book="ja1">
_“Sampesi kho maṃ vācāya,_  
_na ca sampesi kammunā;_  
_Mālā sereyyakasseva,_  
_vaṇṇavantā agandhikā._  

</Paragraph>

</Division>
<Division number="30" book="ja1">

<Paragraph number="1351" type="verses" book="ja1">
_Aphalaṃ madhuraṃ vācaṃ,_  
_yo mittesu pakubbati;_  
_Adadaṃ avissajaṃ bhogaṃ,_  
_sandhi tenassa jīrati._  

</Paragraph>

</Division>
<Division number="31" book="ja1">

<Paragraph number="1352" type="verses" book="ja1">
_Yañhi kayirā tañhi vade,_  
_yaṃ na kayirā na taṃ vade;_  
_Akarontaṃ bhāsamānaṃ,_  
_parijānanti paṇḍitā._  

</Paragraph>

</Division>
<Division number="32" book="ja1">

<Paragraph number="1353" type="verses" book="ja1">
_Balañca vata me khīṇaṃ,_  
_pātheyyañca na vijjati;_  
_Saṅke pāṇūparodhāya,_  
_handa dāni vajāmahaṃ”._  

</Paragraph>

</Division>
<Division number="33" book="ja1">

<Paragraph number="1354" type="verses" book="ja1">
_“Etadeva hi me nāmaṃ,_  
_Yaṃ nāmasmi rathesabha;_  
_Āgamehi mahārāja,_  
_Pitaraṃ āmantayāmahan”ti._  


</Paragraph>
</Division>

<Paragraph number="1355" type="center" book="ja1">
Āsaṅkajātakaṃ pañcamaṃ.
</Paragraph>
