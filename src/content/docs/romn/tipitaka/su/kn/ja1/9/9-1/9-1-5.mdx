<Paragraph number="1824" type="verses" book="ja1">
_“Sutaṃ metaṃ mahābrahme,_  
_kāme bhuñjati hārito;_  
_Kaccetaṃ vacanaṃ tucchaṃ,_  
_kacci suddho iriyyasi”._  

</Paragraph>

</Division>
<Division number="41" book="ja1" bookNo="22" page="362">

<Paragraph number="1825" type="verses" book="ja1">
_“Evametaṃ mahārāja,_  
_yathā te vacanaṃ sutaṃ;_  
_Kummaggaṃ paṭipannosmi,_  
_mohaneyyesu mucchito”._  

</Paragraph>

</Division>
<Division number="42" book="ja1" bookNo="22" page="362">

<Paragraph number="1826" type="verses" book="ja1">
_“Adu paññā kimatthiyā,_  
_nipuṇā sādhucintinī;_  
_Yāya uppatitaṃ rāgaṃ,_  
_kiṃ mano na vinodaye”._  

</Paragraph>

</Division>
<Division number="43" book="ja1" bookNo="22" page="362">

<Paragraph number="1827" type="verses" book="ja1">
_“Cattārome mahārāja,_  
_loke atibalā bhusā;_  
_Rāgo doso mado moho,_  
_yattha paññā na gādhati”._  

</Paragraph>

</Division>
<Division number="44" book="ja1" bookNo="22" page="362">

<Paragraph number="1828" type="verses" book="ja1">
_“Arahā sīlasampanno,_  
_suddho carati hārito;_  
_Medhāvī paṇḍito ceva,_  
_iti no sammato bhavaṃ”._  

</Paragraph>

</Division>
<Division number="45" book="ja1" bookNo="22" page="362">

<Paragraph number="1829" type="verses" book="ja1">
_“Medhāvīnampi hiṃsanti,_  
_isiṃ dhammaguṇe rataṃ;_  
_Vitakkā pāpakā rāja,_  
_subhā rāgūpasaṃhitā”._  

</Paragraph>

</Division>
<Division number="46" book="ja1" bookNo="22" page="363">

<Paragraph number="1830" type="verses" book="ja1">
_“Uppannāyaṃ sarīrajo,_  
_Rāgo vaṇṇavidūsano tava;_  
_Taṃ pajaha bhaddamatthu te,_  
_Bahunnāsi medhāvisammato”._  

</Paragraph>

</Division>
<Division number="47" book="ja1" bookNo="22" page="363">

<Paragraph number="1831" type="verses" book="ja1">
_“Te andhakārake kāme,_  
_Bahudukkhe mahāvise;_  
_Tesaṃ mūlaṃ gavesissaṃ,_  
_Checchaṃ rāgaṃ sabandhanaṃ”._  

</Paragraph>

</Division>
<Division number="48" book="ja1" bookNo="22" page="363">

<Paragraph number="1832" type="verses" book="ja1">
_“Idaṃ vatvāna hārito,_  
_isi saccaparakkamo;_  
_Kāmarāgaṃ virājetvā,_  
_brahmalokūpago ahū”ti._  


</Paragraph>
</Division>

<Paragraph number="1833" type="center" book="ja1">
Haritacajātakaṃ pañcamaṃ.
