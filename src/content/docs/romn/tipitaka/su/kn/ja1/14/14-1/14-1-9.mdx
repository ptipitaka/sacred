<Paragraph number="2715" type="verses" book="ja1">
_“Yadesamānā vicarimha,_  
_pabbatāni vanāni ca;_  
_Anvesaṃ vicariṃ ñātī,_  
_teme adhigatā mayā._  

</Paragraph>

</Division>
<Division number="161" book="ja1" bookNo="22" page="114">

<Paragraph number="2716" type="verses" book="ja1">
_Bahuñcidaṃ mūlaphalaṃ,_  
_bhakkho cāyaṃ anappako;_  
_Rammā cimā girīnajjo,_  
_phāsuvāso bhavissati._  

</Paragraph>

</Division>
<Division number="162" book="ja1" bookNo="22" page="114">

<Paragraph number="2717" type="verses" book="ja1">
_Idhevāhaṃ vasissāmi,_  
_saha sabbehi ñātibhi;_  
_Appossukko nirāsaṅkī,_  
_asoko akutobhayo”._  

</Paragraph>

</Division>
<Division number="163" book="ja1" bookNo="22" page="114">

<Paragraph number="2718" type="verses" book="ja1">
_“Aññampi leṇaṃ pariyesa,_  
_sattu no idha vijjati;_  
_So taccha sūkare hanti,_  
_idhāgantvā varaṃ varaṃ”._  

</Paragraph>

</Division>
<Division number="164" book="ja1" bookNo="22" page="114">

<Paragraph number="2719" type="verses" book="ja1">
_“Ko numhākaṃ idha sattu,_  
_ko ñātī susamāgate;_  
_Duppadhaṃse padhaṃseti,_  
_taṃ me akkhātha pucchitā”._  

</Paragraph>

</Division>
<Division number="165" book="ja1" bookNo="22" page="115">

<Paragraph number="2720" type="verses" book="ja1">
_“Uddhaggarājī migarājā,_  
_balī dāṭhāvudho migo;_  
_So taccha sūkare hanti,_  
_idhāgantvā varaṃ varaṃ”._  

</Paragraph>

</Division>
<Division number="166" book="ja1" bookNo="22" page="115">

<Paragraph number="2721" type="verses" book="ja1">
_“Na no dāṭhā na vijjanti,_  
_balaṃ kāye samohitaṃ;_  
_Sabbe samaggā hutvāna,_  
_vasaṃ kāhāma ekakaṃ”._  

</Paragraph>

</Division>
<Division number="167" book="ja1" bookNo="22" page="115">

<Paragraph number="2722" type="verses" book="ja1">
_“Hadayaṅgamaṃ kaṇṇasukhaṃ,_  
_Vācaṃ bhāsasi tacchaka;_  
_Yopi yuddhe palāyeyya,_  
_Tampi pacchā hanāmase”._  

</Paragraph>

</Division>
<Division number="168" book="ja1" bookNo="22" page="115">

<Paragraph number="2723" type="verses" book="ja1">
_“Pāṇātipātā virato nu ajja,_  
_Abhayaṃ nu te sabbabhūtesu dinnaṃ;_  
_Dāṭhā nu te migavadhāya na santi,_  
_Yo saṃghapatto kapaṇova jhāyasi”._  

</Paragraph>

</Division>
<Division number="169" book="ja1" bookNo="22" page="115">

<Paragraph number="2724" type="verses" book="ja1">
_“Na me dāṭhā na vijjanti,_  
_Balaṃ kāye samohitaṃ;_  
_Ñātī ca disvāna sāmaggī ekato,_  
_Tasmā ca jhāyāmi vanamhi ekako._  

</Paragraph>

</Division>
<Division number="170" book="ja1" bookNo="22" page="115">

<Paragraph number="2725" type="verses" book="ja1">
_Imassudaṃ yanti disodisaṃ pure,_  
_Bhayaṭṭitā leṇagavesino puthu;_  
_Te dāni saṅgamma vasanti ekato,_  
_Yatthaṭṭhitā duppasahajja te mayā._  

</Paragraph>

</Division>
<Division number="171" book="ja1" bookNo="22" page="115">

<Paragraph number="2726" type="verses" book="ja1">
_Pariṇāyakasampannā,_  
_sahitā ekavādino;_  
_Te maṃ samaggā hiṃseyyuṃ,_  
_tasmā nesaṃ na patthaye”._  

</Paragraph>

</Division>
<Division number="172" book="ja1" bookNo="22" page="115">

<Paragraph number="2727" type="verses" book="ja1">
_“Ekova indo asure jināti,_  
_Ekova seno hanti dije pasayha;_  
_Ekova byaggho migasaṅghapatto,_  
_Varaṃ varaṃ hanti balañhi tādisaṃ”._  

</Paragraph>

</Division>
<Division number="173" book="ja1" bookNo="22" page="115">

<Paragraph number="2728" type="verses" book="ja1">
_“Na heva indo na seno,_  
_napi byaggho migādhipo;_  
_Samagge sahite ñātī,_  
_na byagghe kurute vase”._  

</Paragraph>

</Division>
<Division number="174" book="ja1" bookNo="22" page="115">

<Paragraph number="2729" type="verses" book="ja1">
_“Kumbhīlakā sakuṇakā,_  
_saṃghino gaṇacārino;_  
_Sammodamānā ekajjhaṃ,_  
_uppatanti ḍayanti ca._  

</Paragraph>

</Division>
<Division number="175" book="ja1" bookNo="22" page="116">

<Paragraph number="2730" type="verses" book="ja1">
_Tesañca ḍayamānānaṃ,_  
_ekettha apasakkati;_  
_Tañca seno nitāḷeti,_  
_veyyagghiyeva sā gati”._  

</Paragraph>

</Division>
<Division number="176" book="ja1" bookNo="22" page="116">

<Paragraph number="2731" type="verses" book="ja1">
_“Ussāhito jaṭilena,_  
_luddenāmisacakkhunā;_  
_Dāṭhī dāṭhīsu pakkhandi,_  
_maññamāno yathā pure”._  

</Paragraph>

</Division>
<Division number="177" book="ja1" bookNo="22" page="116">

<Paragraph number="2732" type="verses" book="ja1">
_“Sādhu sambahulā ñātī,_  
_api rukkhā araññajā;_  
_Sūkarehi samaggehi,_  
_byaggho ekāyane hato”._  

</Paragraph>

</Division>
<Division number="178" book="ja1" bookNo="22" page="116">

<Paragraph number="2733" type="verses" book="ja1">
_Brāhmaṇañceva byagghañca,_  
_ubho hantvāna sūkarā._  
_Ānandino pamuditā,_  
_mahānādaṃ panādisuṃ._  

</Paragraph>

</Division>
<Division number="179" book="ja1" bookNo="22" page="116">

<Paragraph number="2734" type="verses" book="ja1">
_Te su udumbaramūlasmiṃ,_  
_sūkarā susamāgatā;_  
_Tacchakaṃ abhisiñciṃsu,_  
_“tvaṃ no rājāsi issaro”ti._  


</Paragraph>
</Division>

<Paragraph number="2735" type="center" book="ja1">
Tacchasūkarajātakaṃ navamaṃ.
</Paragraph>
