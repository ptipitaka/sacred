<Division number="101" book="ja1">

<Paragraph number="1012" type="verses" book="ja1">
_“Kāyena yo nāvahare,_  
_vācāya na musā bhaṇe;_  
_Yaso laddhā na majjeyya,_  
_sa ve kakkārumarahati”._  

</Paragraph>

</Division>
<Division number="102" book="ja1">

<Paragraph number="1013" type="verses" book="ja1">
_“Dhammena vittameseyya,_  
_na nikatyā dhanaṃ hare;_  
_Bhoge laddhā na majjeyya,_  
_sa ve kakkārumarahati”._  

</Paragraph>

</Division>
<Division number="103" book="ja1">

<Paragraph number="1014" type="verses" book="ja1">
_“Yassa cittaṃ ahāliddaṃ,_  
_saddhā ca avirāginī;_  
_Eko sāduṃ na bhuñjeyya,_  
_sa ve kakkārumarahati”._  

</Paragraph>

</Division>
<Division number="104" book="ja1">

<Paragraph number="1015" type="verses" book="ja1">
_“Sammukhā vā tirokkhā vā,_  
_Yo sante na paribhāsati;_  
_Yathāvādī tathākārī,_  
_Sa ve kakkārumarahatī”ti._  


</Paragraph>
</Division>

Kakkārujātakaṃ chaṭṭhaṃ.
