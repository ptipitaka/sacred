<Paragraph number="2107" type="verses" book="ja1">
_“Tassa nāgassa vippavāsena,_  
_Virūḷhā sallakī ca kuṭajā ca;_  
_Kuruvindakaravīrā bhisasāmā ca,_  
_Nivāte pupphitā ca kaṇikārā._  

</Paragraph>

</Division>
<Division number="2" book="ja1" bookNo="22" page="71">

<Paragraph number="2108" type="verses" book="ja1">
_Kocideva suvaṇṇakāyurā,_  
_Nāgarājaṃ bharanti piṇḍena;_  
_Yattha rājā rājakumāro vā,_  
_Kavacamabhihessati achambhito”._  

</Paragraph>

</Division>
<Division number="3" book="ja1" bookNo="22" page="93">

<Paragraph number="2109" type="verses" book="ja1">
_“Gaṇhāhi nāga kabaḷaṃ,_  
_mā nāga kisako bhava;_  
_Bahūni rājakiccāni,_  
_tāni nāga karissasi”._  

</Paragraph>

</Division>
<Division number="4" book="ja1" bookNo="22" page="71">

<Paragraph number="2110" type="verses" book="ja1">
_“Sā nūnasā kapaṇikā,_  
_andhā apariṇāyikā;_  
_Khāṇuṃ pādena ghaṭṭeti,_  
_giriṃ caṇḍoraṇaṃ pati”._  

</Paragraph>

</Division>
<Division number="5" book="ja1" bookNo="22" page="71">

<Paragraph number="2111" type="verses" book="ja1">
_“Kā nu te sā mahānāga,_  
_andhā apariṇāyikā;_  
_Khāṇuṃ pādena ghaṭṭeti,_  
_giriṃ caṇḍoraṇaṃ pati”._  

</Paragraph>

</Division>
<Division number="6" book="ja1" bookNo="22" page="71">

<Paragraph number="2112" type="verses" book="ja1">
_“Mātā me sā mahārāja,_  
_andhā apariṇāyikā;_  
_Khāṇuṃ pādena ghaṭṭeti,_  
_giriṃ caṇḍoraṇaṃ pati”._  

</Paragraph>

</Division>
<Division number="7" book="ja1" bookNo="22" page="71">

<Paragraph number="2113" type="verses" book="ja1">
_“Muñcathetaṃ mahānāgaṃ,_  
_yoyaṃ bharati mātaraṃ;_  
_Sametu mātarā nāgo,_  
_saha sabbehi ñātibhi”._  

</Paragraph>

</Division>
<Division number="8" book="ja1" bookNo="22" page="72">

<Paragraph number="2114" type="verses" book="ja1">
_“Mutto ca bandhanā nāgo,_  
_muttamādāya kuñjaro;_  
_Muhuttaṃ assāsayitvā,_  
_agamā yena pabbato._  

</Paragraph>

</Division>
<Division number="9" book="ja1" bookNo="22" page="72">

<Paragraph number="2115" type="verses" book="ja1">
_Tato so naḷiniṃ gantvā,_  
_sītaṃ kuñjarasevitaṃ;_  
_Soṇḍāyūdakamāhatvā,_  
_mātaraṃ abhisiñcatha”._  

</Paragraph>

</Division>
<Division number="10" book="ja1" bookNo="22" page="72">

<Paragraph number="2116" type="verses" book="ja1">
_“Koyaṃ anariyo devo,_  
_akālenapi vassati;_  
_Gato me atrajo putto,_  
_yo mayhaṃ paricārako”._  

</Paragraph>

</Division>
<Division number="11" book="ja1" bookNo="22" page="72">

<Paragraph number="2117" type="verses" book="ja1">
_“Uṭṭhehi amma kiṃ sesi,_  
_āgato tyāhamatrajo;_  
_Muttomhi kāsirājena,_  
_vedehena yasassinā”._  

</Paragraph>

</Division>
<Division number="12" book="ja1" bookNo="22" page="72">

<Paragraph number="2118" type="verses" book="ja1">
_“Ciraṃ jīvatu so rājā,_  
_kāsīnaṃ raṭṭhavaḍḍhano;_  
_Yo me puttaṃ pamocesi,_  
_sadā vuddhāpacāyikan”ti._  


</Paragraph>
</Division>

Mātuposakajātakaṃ paṭhamaṃ.
