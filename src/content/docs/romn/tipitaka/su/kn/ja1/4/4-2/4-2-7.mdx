<Division number="65" book="ja1">

<Paragraph number="965" type="verses" book="ja1">
_“Mataṃ mataṃ evaṃ rodatha,_  
_Na hi taṃ rodatha yo marissati;_  
_Sabbepi sarīradhārino,_  
_Anupubbena jahanti jīvitaṃ._  

</Paragraph>

</Division>
<Division number="66" book="ja1">

<Paragraph number="966" type="verses" book="ja1">
_Devamanussā catuppadā,_  
_Pakkhigaṇā uragā ca bhogino;_  
_Samhi sarīre anissarā,_  
_Ramamānāva jahanti jīvitaṃ._  

</Paragraph>

</Division>
<Division number="67" book="ja1">

<Paragraph number="967" type="verses" book="ja1">
_Evaṃ calitaṃ asaṇṭhitaṃ,_  
_Sukhadukkhaṃ manujesvapekkhiya;_  
_Kanditaruditaṃ niratthakaṃ,_  
_Kiṃ vo sokagaṇābhikīrare._  

</Paragraph>

</Division>
<Division number="68" book="ja1">

<Paragraph number="968" type="verses" book="ja1">
_Dhuttā ca soṇḍā akatā,_  
_bālā sūrā ayogino;_  
_Dhīraṃ maññanti bāloti,_  
_ye dhammassa akovidā”ti._  


</Paragraph>
</Division>

<Paragraph number="969" type="center" book="ja1">
Matarodanajātakaṃ sattamaṃ.
</Paragraph>
