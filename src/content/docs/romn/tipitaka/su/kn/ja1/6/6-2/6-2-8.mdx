<Division number="123" book="ja1">

<Paragraph number="1459" type="verses" book="ja1">
_“Susukhaṃ vata jīvanti,_  
_ye janā vighāsādino;_  
_Diṭṭheva dhamme pāsaṃsā,_  
_samparāye ca suggatī”._  

</Paragraph>

</Division>
<Division number="124" book="ja1">

<Paragraph number="1460" type="verses" book="ja1">
_“Sukassa bhāsamānassa,_  
_na nisāmetha paṇḍitā;_  
_Idaṃ suṇātha sodariyā,_  
_amhevāyaṃ pasaṃsati”._  

</Paragraph>

</Division>
<Division number="125" book="ja1">

<Paragraph number="1461" type="verses" book="ja1">
_“Nāhaṃ tumhe pasaṃsāmi,_  
_kuṇapādā suṇātha me;_  
_Ucchiṭṭhabhojino tumhe,_  
_na tumhe vighāsādino”._  

</Paragraph>

</Division>
<Division number="126" book="ja1">

<Paragraph number="1462" type="verses" book="ja1">
_“Sattavassā pabbajitā,_  
_mejjhāraññe sikhaṇḍino;_  
_Vighāseneva yāpentā,_  
_mayañce bhoto gārayhā;_  
_Ke nu bhoto pasaṃsiyā”._  

</Paragraph>

</Division>
<Division number="127" book="ja1">

<Paragraph number="1463" type="verses" book="ja1">
_“Tumhe sīhānaṃ byagghānaṃ,_  
_vāḷānañcāvasiṭṭhakaṃ;_  
_Ucchiṭṭheneva yāpentā,_  
_maññivho vighāsādino._  

</Paragraph>

</Division>
<Division number="128" book="ja1">

<Paragraph number="1464" type="verses" book="ja1">
_Ye brāhmaṇassa samaṇassa,_  
_aññassa vā vanibbino;_  
_Datvāva sesaṃ bhuñjanti,_  
_te janā vighāsādino”ti._  


</Paragraph>
</Division>

<Paragraph number="1465" type="center" book="ja1">
Vighāsādajātakaṃ aṭṭhamaṃ.
</Paragraph>
