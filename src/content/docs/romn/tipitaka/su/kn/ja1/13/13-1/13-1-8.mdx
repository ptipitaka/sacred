<Division number="104" book="ja1">

<Paragraph number="2494" type="verses" book="ja1">
_“Ahameva dubbhāsitaṃ bhāsi bālo,_  
_Bhekovaraññe ahimavhāyamāno;_  
_Takkāriye sobbhamimaṃ patāmi,_  
_Na kireva sādhu ativelabhāṇī”._  

</Paragraph>

</Division>
<Division number="105" book="ja1">

<Paragraph number="2495" type="verses" book="ja1">
_“Pappoti macco ativelabhāṇī,_  
_Bandhaṃ vadhaṃ sokapariddavañca;_  
_Attānameva garahāsi ettha,_  
_Ācera yaṃ taṃ nikhaṇanti sobbhe”._  

</Paragraph>

</Division>
<Division number="106" book="ja1">

<Paragraph number="2496" type="verses" book="ja1">
_“Kimevahaṃ tuṇḍilamanupucchiṃ,_  
_Kareyya saṃ bhātaraṃ kāḷikāyaṃ;_  
_Naggovahaṃ vatthayugañca jīno,_  
_Ayampi attho bahutādisova”._  

</Paragraph>

</Division>
<Division number="107" book="ja1">

<Paragraph number="2497" type="verses" book="ja1">
_“Yo yujjhamānānamayujjhamāno,_  
_Meṇḍantaraṃ accupatī kuliṅgo;_  
_So piṃsito meṇḍasirehi tattha,_  
_Ayampi attho bahutādisova._  

</Paragraph>

</Division>
<Division number="108" book="ja1">

<Paragraph number="2498" type="verses" book="ja1">
_Caturo janā potthakamaggahesuṃ,_  
_Ekañca posaṃ anurakkhamānā;_  
_Sabbeva te bhinnasirā sayiṃsu,_  
_Ayampi attho bahutādisova._  

</Paragraph>

</Division>
<Division number="109" book="ja1">

<Paragraph number="2499" type="verses" book="ja1">
_Ajā yathā veḷugumbasmiṃ baddhā,_  
_Avakkhipantī asimajjhagacchi;_  
_Teneva tassā galakāvakantaṃ,_  
_Ayampi attho bahutādisova”._  

</Paragraph>

</Division>
<Division number="110" book="ja1">

<Paragraph number="2500" type="verses" book="ja1">
_“Ime na devā na gandhabbaputtā,_  
_Migā ime atthavasaṃ gatā me;_  
_Ekañca naṃ sāyamāse pacantu,_  
_Ekaṃ punappātarāse pacantu”._  

</Paragraph>

</Division>
<Division number="111" book="ja1">

<Paragraph number="2501" type="verses" book="ja1">
_“Sataṃ sahassāni dubhāsitāni,_  
_Kalampi nāgghanti subhāsitassa;_  
_Dubbhāsitaṃ saṅkamāno kileso,_  
_Tasmā tuṇhī kimpurisā na bālyā”._  

</Paragraph>

</Division>
<Division number="112" book="ja1">

<Paragraph number="2502" type="verses" book="ja1">
_“Yā mesā byāhāsi pamuñcathetaṃ,_  
_Giriñca naṃ himavantaṃ nayantu;_  
_Imañca kho dentu mahānasāya,_  
_Pātova naṃ pātarāse pacantu”._  

</Paragraph>

</Division>
<Division number="113" book="ja1">

<Paragraph number="2503" type="verses" book="ja1">
_“Pajjunnanāthā pasavo,_  
_pasunāthā ayaṃ pajā;_  
_Tvaṃ nāthosi mahārāja,_  
_nāthohaṃ bhariyāya me;_  
_Dvinnamaññataraṃ ñatvā,_  
_mutto gaccheyya pabbataṃ._  

</Paragraph>

</Division>
<Division number="114" book="ja1">

<Paragraph number="2504" type="verses" book="ja1">
_Na ve nindā suparivajjayetha,_  
_Nānā janā sevitabbā janinda;_  
_Yeneva eko labhate pasaṃsaṃ,_  
_Teneva añño labhate ninditāraṃ._  

</Paragraph>

</Division>
<Division number="115" book="ja1">

<Paragraph number="2505" type="verses" book="ja1">
_Sabbo loko paricitto aticitto,_  
_Sabbo loko cittavā samhi citte;_  
_Paccekacittā puthu sabbasattā,_  
_Kassīdha cittassa vasena vatte”._  

</Paragraph>

</Division>
<Division number="116" book="ja1">

<Paragraph number="2506" type="verses" book="ja1">
_“Tuṇhī ahū kimpuriso sabhariyo,_  
_Yo dāni byāhāsi bhayassa bhīto;_  
_So dāni mutto sukhito arogo,_  
_Vācākirevatthavatī narānan”ti._  


</Paragraph>
</Division>

<Paragraph number="2507" type="center" book="ja1">
Takkāriyajātakaṃ aṭṭhamaṃ.
</Paragraph>
