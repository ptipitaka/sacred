<Division number="83" book="ja1">

<Paragraph number="2938" type="verses" book="ja1">
_“Paññāyupetaṃ siriyā vihīnaṃ,_  
_Yasassinaṃ vāpi apetapaññaṃ;_  
_Pucchāmi taṃ senaka etamatthaṃ,_  
_Kamettha seyyo kusalā vadanti”._  

</Paragraph>

</Division>
<Division number="84" book="ja1">

<Paragraph number="2939" type="verses" book="ja1">
_“Dhīrā ca bālā ca have janinda,_  
_Sippūpapannā ca asippino ca;_  
_Sujātimantopi ajātimassa,_  
_Yasassino pesakarā bhavanti;_  
_Etampi disvāna ahaṃ vadāmi,_  
_Pañño nihīno sirīmāva seyyo”._  

</Paragraph>

</Division>
<Division number="85" book="ja1">

<Paragraph number="2940" type="verses" book="ja1">
_“Tuvampi pucchāmi anomapañña,_  
_Mahosadha kevaladhammadassi;_  
_Bālaṃ yasassiṃ paṇḍitaṃ appabhogaṃ,_  
_Kamettha seyyo kusalā vadanti”._  

</Paragraph>

</Division>
<Division number="86" book="ja1">

<Paragraph number="2941" type="verses" book="ja1">
_“Pāpāni kammāni karoti bālo,_  
_Idhameva seyyo iti maññamāno;_  
_Idhalokadassī paralokamadassī,_  
_Ubhayattha bālo kalimaggahesi;_  
_Etampi disvāna ahaṃ vadāmi,_  
_Paññova seyyo na yasassi bālo”._  

</Paragraph>

</Division>
<Division number="87" book="ja1">

<Paragraph number="2942" type="verses" book="ja1">
_“Na sippametaṃ vidadhāti bhogaṃ,_  
_Na bandhuvā na sarīravaṇṇo yo;_  
_Passeḷamūgaṃ sukhamedhamānaṃ,_  
_Sirī hi naṃ bhajate goravindaṃ;_  
_Etampi disvāna ahaṃ vadāmi,_  
_Pañño nihīno sirīmāva seyyo”._  

</Paragraph>

</Division>
<Division number="88" book="ja1">

<Paragraph number="2943" type="verses" book="ja1">
_“Laddhā sukhaṃ majjati appapañño,_  
_Dukkhena phuṭṭhopi pamohameti;_  
_Āgantunā dukkhasukhena phuṭṭho,_  
_Pavedhati vāricarova ghamme;_  
_Etampi disvāna ahaṃ vadāmi,_  
_Paññova seyyo na yasassi bālo”._  

</Paragraph>

</Division>
<Division number="89" book="ja1">

<Paragraph number="2944" type="verses" book="ja1">
_“Dumaṃ yathā sāduphalaṃ araññe,_  
_Samantato samabhisaranti pakkhī;_  
_Evampi aḍḍhaṃ sadhanaṃ sabhogaṃ,_  
_Bahujjano bhajati atthahetu;_  
_Etampi disvāna ahaṃ vadāmi,_  
_Pañño nihīno sirīmāva seyyo”._  

</Paragraph>

</Division>
<Division number="90" book="ja1">

<Paragraph number="2945" type="verses" book="ja1">
_“Na sādhu balavā bālo,_  
_Sāhasā vindate dhanaṃ;_  
_Kandantametaṃ dummedhaṃ,_  
_Kaḍḍhanti nirayaṃ bhusaṃ;_  
_Etampi disvāna ahaṃ vadāmi,_  
_Paññova seyyo na yasassi bālo”._  

</Paragraph>

</Division>
<Division number="91" book="ja1">

<Paragraph number="2946" type="verses" book="ja1">
_“Yā kāci najjo gaṅgamabhissavanti,_  
_Sabbāva tā nāmagottaṃ jahanti;_  
_Gaṅgā samuddaṃ paṭipajjamānā,_  
_Na khāyate iddhiṃ paññopi loke;_  
_Etampi disvāna ahaṃ vadāmi,_  
_Pañño nihīno sirīmāva seyyo”._  

</Paragraph>

</Division>
<Division number="92" book="ja1">

<Paragraph number="2947" type="verses" book="ja1">
_“Yametamakkhā udadhiṃ mahantaṃ,_  
_Savanti najjo sabbakālamasaṅkhyaṃ;_  
_So sāgaro niccamuḷāravego,_  
_Velaṃ na acceti mahāsamuddo._  

</Paragraph>

</Division>
<Division number="93" book="ja1">

<Paragraph number="2948" type="verses" book="ja1">
_Evampi bālassa pajappitāni,_  
_Paññaṃ na acceti sirī kadāci;_  
_Etampi disvāna ahaṃ vadāmi,_  
_Paññova seyyo na yasassi bālo”._  

</Paragraph>

</Division>
<Division number="94" book="ja1">

<Paragraph number="2949" type="verses" book="ja1">
_“Asaññato cepi paresamatthaṃ,_  
_Bhaṇāti sandhānagato yasassī;_  
_Tasseva taṃ rūhati ñātimajjhe,_  
_Sirī hi naṃ kārayate na paññā;_  
_Etampi disvāna ahaṃ vadāmi,_  
_Pañño nihīno sirīmāva seyyo”._  

</Paragraph>

</Division>
<Division number="95" book="ja1">

<Paragraph number="2950" type="verses" book="ja1">
_“Parassa vā attano vāpi hetu,_  
_Bālo musā bhāsati appapañño;_  
_So nindito hoti sabhāya majjhe,_  
_Pacchāpi so duggatigāmī hoti;_  
_Etampi disvāna ahaṃ vadāmi,_  
_Paññova seyyo na yasassi bālo”._  

</Paragraph>

</Division>
<Division number="96" book="ja1">

<Paragraph number="2951" type="verses" book="ja1">
_“Atthampi ce bhāsati bhūripañño,_  
_Anāḷhiyo appadhano daliddo;_  
_Na tassa taṃ rūhati ñātimajjhe,_  
_Sirī ca paññāṇavato na hoti;_  
_Etampi disvāna ahaṃ vadāmi,_  
_Pañño nihīno sirīmāva seyyo”._  

</Paragraph>

</Division>
<Division number="97" book="ja1">

<Paragraph number="2952" type="verses" book="ja1">
_“Parassa vā attano vāpi hetu,_  
_Na bhāsati alikaṃ bhūripañño;_  
_So pūjito hoti sabhāya majjhe,_  
_Pacchāpi so suggatigāmī hoti;_  
_Etampi disvāna ahaṃ vadāmi,_  
_Paññova seyyo na yasassi bālo”._  

</Paragraph>

</Division>
<Division number="98" book="ja1">

<Paragraph number="2953" type="verses" book="ja1">
_“Hatthī gavassā maṇikuṇḍalā ca,_  
_Thiyo ca iddhesu kulesu jātā;_  
_Sabbāva tā upabhogā bhavanti,_  
_Iddhassa posassa aniddhimanto;_  
_Etampi disvāna ahaṃ vadāmi,_  
_Pañño nihīno sirīmāva seyyo”._  

</Paragraph>

</Division>
<Division number="99" book="ja1">

<Paragraph number="2954" type="verses" book="ja1">
_“Asaṃvihitakammantaṃ,_  
_Bālaṃ dummedhamantinaṃ;_  
_Sirī jahati dummedhaṃ,_  
_Jiṇṇaṃva urago tacaṃ;_  
_Etampi disvāna ahaṃ vadāmi,_  
_Paññova seyyo na yasassi bālo”._  

</Paragraph>

</Division>
<Division number="100" book="ja1">

<Paragraph number="2955" type="verses" book="ja1">
_“Pañca paṇḍitā mayaṃ bhaddante,_  
_Sabbe pañjalikā upaṭṭhitā;_  
_Tvaṃ no abhibhuyya issarosi,_  
_Sakkova bhūtapati devarājā;_  
_Etampi disvāna ahaṃ vadāmi,_  
_Pañño nihīno sirīmāva seyyo”._  

</Paragraph>

</Division>
<Division number="101" book="ja1">

<Paragraph number="2956" type="verses" book="ja1">
_“Dāsova paññassa yasassi bālo,_  
_Atthesu jātesu tathāvidhesu;_  
_Yaṃ paṇḍito nipuṇaṃ saṃvidheti,_  
_Sammohamāpajjati tattha bālo;_  
_Etampi disvāna ahaṃ vadāmi,_  
_Paññova seyyo na yasassi bālo._  

</Paragraph>

</Division>
<Division number="102" book="ja1">

<Paragraph number="2957" type="verses" book="ja1">
_Addhā hi paññāva sataṃ pasatthā,_  
_Kantā sirī bhogaratā manussā;_  
_Ñāṇañca buddhānamatulyarūpaṃ,_  
_Paññaṃ na acceti sirī kadāci”._  

</Paragraph>

</Division>
<Division number="103" book="ja1">

<Paragraph number="2958" type="verses" book="ja1">
_“Yaṃ taṃ apucchimha akittayī no,_  
_Mahosadha kevaladhammadassī;_  
_Gavaṃ sahassaṃ usabhañca nāgaṃ,_  
_Ājaññayutte ca rathe dasa ime;_  
_Pañhassa veyyākaraṇena tuṭṭho,_  
_Dadāmi te gāmavarāni soḷasā”ti._  


</Paragraph>
</Division>
