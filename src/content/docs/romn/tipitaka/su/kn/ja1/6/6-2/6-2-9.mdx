<Paragraph number="1466" type="verses" book="ja1">
_“Paṇītaṃ bhuñjase bhattaṃ,_  
_sappitelañca mātula;_  
_Atha kena nu vaṇṇena,_  
_kiso tvamasi vāyasa”._  

</Paragraph>

</Division>
<Division number="130" book="ja1" bookNo="22" page="325">

<Paragraph number="1467" type="verses" book="ja1">
_“Amittamajjhe vasato,_  
_tesu āmisamesato;_  
_Niccaṃ ubbiggahadayassa,_  
_kuto kākassa daḷhiyaṃ._  

</Paragraph>

</Division>
<Division number="131" book="ja1" bookNo="22" page="325">

<Paragraph number="1468" type="verses" book="ja1">
_Niccaṃ ubbegino kākā,_  
_dhaṅkā pāpena kammunā;_  
_Laddho piṇḍo na pīṇeti,_  
_kiso tenasmi vaṭṭaka._  

</Paragraph>

</Division>
<Division number="132" book="ja1" bookNo="22" page="325">

<Paragraph number="1469" type="verses" book="ja1">
_Lūkhāni tiṇabījāni,_  
_appasnehāni bhuñjasi;_  
_Atha kena nu vaṇṇena,_  
_thūlo tvamasi vaṭṭaka”._  

</Paragraph>

</Division>
<Division number="133" book="ja1" bookNo="22" page="326">

<Paragraph number="1470" type="verses" book="ja1">
_“Appicchā appacintāya,_  
_adūragamanena ca;_  
_Laddhāladdhena yāpento,_  
_thūlo tenasmi vāyasa._  

</Paragraph>

</Division>
<Division number="134" book="ja1" bookNo="22" page="326">

<Paragraph number="1471" type="verses" book="ja1">
_Appicchassa hi posassa,_  
_appacintasukhassa ca;_  
_Susaṅgahitamānassa,_  
_vuttī susamudānayā”ti._  


</Paragraph>
</Division>

<Paragraph number="1472" type="center" book="ja1">
Vaṭṭakajātakaṃ navamaṃ.
</Paragraph>
