<Paragraph number="1673" type="verses" book="ja1">
_“Odātavatthā suci allakesā,_  
_Kaccāni kiṃ kumbhimadhissayitvā;_  
_Piṭṭhā tilā dhovasi taṇḍulāni,_  
_Tilodano hehiti kissahetu”._  

</Paragraph>

</Division>
<Division number="2" book="ja1" bookNo="22" page="309">

<Paragraph number="1674" type="verses" book="ja1">
_“Na kho ayaṃ brāhmaṇa bhojanatthā,_  
_Tilodano hehiti sādhupakko;_  
_Dhammo mato tassa pahuttamajja,_  
_Ahaṃ karissāmi susānamajjhe”._  

</Paragraph>

</Division>
<Division number="3" book="ja1" bookNo="22" page="358">

<Paragraph number="1675" type="verses" book="ja1">
_“Anuvicca kaccāni karohi kiccaṃ,_  
_Dhammo mato ko nu taveva saṃsi;_  
_Sahassanetto atulānubhāvo,_  
_Na miyyatī dhammavaro kadāci”._  

</Paragraph>

</Division>
<Division number="4" book="ja1" bookNo="22" page="309">

<Paragraph number="1676" type="verses" book="ja1">
_“Daḷhappamāṇaṃ mama ettha brahme,_  
_Dhammo mato natthi mamettha kaṅkhā;_  
_Yeyeva dāni pāpā bhavanti,_  
_Te teva dāni sukhitā bhavanti._  

</Paragraph>

</Division>
<Division number="5" book="ja1" bookNo="22" page="309">

<Paragraph number="1677" type="verses" book="ja1">
_Suṇisā hi mayhaṃ vañjhā ahosi,_  
_Sā maṃ vadhitvāna vijāyi puttaṃ;_  
_Sā dāni sabbassa kulassa issarā,_  
_Ahaṃ panamhi apaviddhā ekikā”._  

</Paragraph>

</Division>
<Division number="6" book="ja1" bookNo="22" page="309">

<Paragraph number="1678" type="verses" book="ja1">
_“Jīvāmi vohaṃ na matohamasmi,_  
_Taveva atthāya idhāgatosmi;_  
_Yā taṃ vadhitvāna vijāyi puttaṃ,_  
_Sahāva puttena karomi bhasmaṃ”._  

</Paragraph>

</Division>
<Division number="7" book="ja1" bookNo="22" page="310">

<Paragraph number="1679" type="verses" book="ja1">
_“Evañca te ruccati devarāja,_  
_Mameva atthāya idhāgatosi;_  
_Ahañca putto suṇisā ca nattā,_  
_Sammodamānā gharamāvasema”._  

</Paragraph>

</Division>
<Division number="8" book="ja1" bookNo="22" page="310">

<Paragraph number="1680" type="verses" book="ja1">
_“Evañca te ruccati kātiyāni,_  
_Hatāpi santā na jahāsi dhammaṃ;_  
_Tuvañca putto suṇisā ca nattā,_  
_Sammodamānā gharamāvasetha._  

</Paragraph>

</Division>
<Division number="9" book="ja1" bookNo="22" page="310">

<Paragraph number="1681" type="verses" book="ja1">
_Sā kātiyānī suṇisāya saddhiṃ,_  
_Sammodamānā gharamāvasittha;_  
_Putto ca nattā ca upaṭṭhahiṃsu,_  
_Devānamindena adhiggahītā”ti._  


</Paragraph>
</Division>

<Paragraph number="1682" type="center" book="ja1">
Kaccānijātakaṃ paṭhamaṃ.
</Paragraph>
