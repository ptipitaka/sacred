<Division number="30" book="ja1">

<Paragraph number="1813" type="verses" book="ja1">
_“Santi rukkhā haripattā,_  
_dumā nekaphalā bahū;_  
_Kasmā nu sukkhe koḷāpe,_  
_suvassa nirato mano”._  

</Paragraph>

</Division>
<Division number="31" book="ja1">

<Paragraph number="1814" type="verses" book="ja1">
_“Phalassa upabhuñjimhā,_  
_nekavassagaṇe bahū;_  
_Aphalampi viditvāna,_  
_sāva metti yathā pure”._  

</Paragraph>

</Division>
<Division number="32" book="ja1">

<Paragraph number="1815" type="verses" book="ja1">
_“Sukkhañca rukkhaṃ koḷāpaṃ,_  
_opattamaphalaṃ dumaṃ;_  
_Ohāya sakuṇā yanti,_  
_kiṃ dosaṃ passase dija”._  

</Paragraph>

</Division>
<Division number="33" book="ja1">

<Paragraph number="1816" type="verses" book="ja1">
_“Ye phalatthā sambhajanti,_  
_aphaloti jahanti naṃ;_  
_Attatthapaññā dummedhā,_  
_te honti pakkhapātino”._  

</Paragraph>

</Division>
<Division number="34" book="ja1">

<Paragraph number="1817" type="verses" book="ja1">
_“Sādhu sakkhi kataṃ hoti,_  
_metti saṃsati santhavo;_  
_Sacetaṃ dhammaṃ rocesi,_  
_pāsaṃsosi vijānataṃ._  

</Paragraph>

</Division>
<Division number="35" book="ja1">

<Paragraph number="1818" type="verses" book="ja1">
_So te suva varaṃ dammi,_  
_pattayāna vihaṅgama;_  
_Varaṃ varassu vakkaṅga,_  
_yaṃ kiñci manasicchasi”._  

</Paragraph>

</Division>
<Division number="36" book="ja1">

<Paragraph number="1819" type="verses" book="ja1">
_“Api nāma naṃ passeyyaṃ,_  
_sapattaṃ saphalaṃ dumaṃ;_  
_Daliddova nidhiṃ laddhā,_  
_nandeyyāhaṃ punappunaṃ”._  

</Paragraph>

</Division>
<Division number="37" book="ja1">

<Paragraph number="1820" type="verses" book="ja1">
_“Tato amatamādāya,_  
_abhisiñci mahīruhaṃ;_  
_Tassa sākhā virūhiṃsu,_  
_sītacchāyā manoramā”._  

</Paragraph>

</Division>
<Division number="38" book="ja1">

<Paragraph number="1821" type="verses" book="ja1">
_“Evaṃ sakka sukhī hohi,_  
_saha sabbehi ñātibhi;_  
_Yathāhamajja sukhito,_  
_disvāna saphalaṃ dumaṃ._  

</Paragraph>

</Division>
<Division number="39" book="ja1">

<Paragraph number="1822" type="verses" book="ja1">
_Suvassa ca varaṃ datvā,_  
_katvāna saphalaṃ dumaṃ;_  
_Pakkāmi saha bhariyāya,_  
_devānaṃ nandanaṃ vanan”ti._  


</Paragraph>
</Division>

<Paragraph number="1823" type="center" book="ja1">
Cūḷasuvajātakaṃ catutthaṃ.
</Paragraph>
