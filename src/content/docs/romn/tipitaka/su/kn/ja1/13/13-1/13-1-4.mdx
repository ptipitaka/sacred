<Division number="40" book="ja1">

<Paragraph number="2426" type="verses" book="ja1">
_“Na te kaṭṭhāni bhinnāni,_  
_na te udakamābhataṃ;_  
_Aggīpi te na hāpito,_  
_kiṃ nu mandova jhāyasi”._  

</Paragraph>

</Division>
<Division number="41" book="ja1">

<Paragraph number="2427" type="verses" book="ja1">
_“Na ussahe vane vatthuṃ,_  
_kassapāmantayāmi taṃ;_  
_Dukkho vāso araññasmiṃ,_  
_raṭṭhaṃ icchāmi gantave._  

</Paragraph>

</Division>
<Division number="42" book="ja1">

<Paragraph number="2428" type="verses" book="ja1">
_Yathā ahaṃ ito gantvā,_  
_Yasmiṃ janapade vasaṃ;_  
_Ācāraṃ brahme sikkheyyaṃ,_  
_Taṃ dhammaṃ anusāsa maṃ”._  

</Paragraph>

</Division>
<Division number="43" book="ja1">

<Paragraph number="2429" type="verses" book="ja1">
_“Sace araññaṃ hitvāna,_  
_vanamūlaphalāni ca;_  
_Raṭṭhe rocayase vāsaṃ,_  
_taṃ dhammaṃ nisāmehi me._  

</Paragraph>

</Division>
<Division number="44" book="ja1">

<Paragraph number="2430" type="verses" book="ja1">
_Visaṃ mā paṭisevittho,_  
_papātaṃ parivajjaya;_  
_Paṅke ca mā visīdittho,_  
_yatto cāsīvise care”._  

</Paragraph>

</Division>
<Division number="45" book="ja1">

<Paragraph number="2431" type="verses" book="ja1">
_“Kiṃ nu visaṃ papāto vā,_  
_paṅko vā brahmacārinaṃ;_  
_Kaṃ tvaṃ āsīvisaṃ brūsi,_  
_taṃ me akkhāhi pucchito”._  

</Paragraph>

</Division>
<Division number="46" book="ja1">

<Paragraph number="2432" type="verses" book="ja1">
_“Āsavo tāta lokasmiṃ,_  
_surā nāma pavuccati;_  
_Manuñño surabhī vaggu,_  
_sādu khuddarasūpamo;_  
_Visaṃ tadāhu ariyā se,_  
_brahmacariyassa nārada._  

</Paragraph>

</Division>
<Division number="47" book="ja1">

<Paragraph number="2433" type="verses" book="ja1">
_Itthiyo tāta lokasmiṃ,_  
_pamattaṃ pamathenti tā;_  
_Haranti yuvino cittaṃ,_  
_tūlaṃ bhaṭṭhaṃva māluto;_  
_Papāto eso akkhāto,_  
_brahmacariyassa nārada._  

</Paragraph>

</Division>
<Division number="48" book="ja1">

<Paragraph number="2434" type="verses" book="ja1">
_Lābho siloko sakkāro,_  
_pūjā parakulesu ca;_  
_Paṅko eso ca akkhāto,_  
_brahmacariyassa nārada._  

</Paragraph>

</Division>
<Division number="49" book="ja1">

<Paragraph number="2435" type="verses" book="ja1">
_Sasatthā tāta rājāno,_  
_āvasanti mahiṃ imaṃ;_  
_Te tādise manussinde,_  
_mahante tāta nārada._  

</Paragraph>

</Division>
<Division number="50" book="ja1">

<Paragraph number="2436" type="verses" book="ja1">
_Issarānaṃ adhipatīnaṃ,_  
_na tesaṃ pādato care;_  
_Āsīvisoti akkhāto,_  
_brahmacariyassa nārada._  

</Paragraph>

</Division>
<Division number="51" book="ja1">

<Paragraph number="2437" type="verses" book="ja1">
_Bhattattho bhattakāle ca,_  
_yaṃ gehamupasaṅkame;_  
_Yadettha kusalaṃ jaññā,_  
_tattha ghāsesanaṃ care._  

</Paragraph>

</Division>
<Division number="52" book="ja1">

<Paragraph number="2438" type="verses" book="ja1">
_Pavisitvā parakulaṃ,_  
_pānatthaṃ bhojanāya vā;_  
_Mitaṃ khāde mitaṃ bhuñje,_  
_na ca rūpe manaṃ kare._  

</Paragraph>

</Division>
<Division number="53" book="ja1">

<Paragraph number="2439" type="verses" book="ja1">
_Goṭṭhaṃ majjaṃ kirāṭañca,_  
_sabhā nikiraṇāni ca;_  
_Ārakā parivajjehi,_  
_yānīva visamaṃ pathan”ti._  


</Paragraph>
</Division>

<Paragraph number="2440" type="center" book="ja1">
Cūḷanāradajātakaṃ catutthaṃ.
</Paragraph>
