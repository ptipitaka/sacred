<Division number="122" book="ja1" bookNo="22" page="88">

<Paragraph number="2369" type="verses" book="ja1">
_“Na naṃ umhayate disvā,_  
_na ca naṃ paṭinandati;_  
_Cakkhūni cassa na dadāti,_  
_paṭilomañca vattati._  

</Paragraph>

</Division>
<Division number="123" book="ja1" bookNo="22" page="88">

<Paragraph number="2370" type="verses" book="ja1">
_Amitte tassa bhajati,_  
_mitte tassa na sevati;_  
_Vaṇṇakāme nivāreti,_  
_akkosante pasaṃsati._  

</Paragraph>

</Division>
<Division number="124" book="ja1" bookNo="22" page="88">

<Paragraph number="2371" type="verses" book="ja1">
_Guyhañca tassa nakkhāti,_  
_tassa guyhaṃ na gūhati;_  
_Kammaṃ tassa na vaṇṇeti,_  
_paññassa nappasaṃsati._  

</Paragraph>

</Division>
<Division number="125" book="ja1" bookNo="22" page="88">

<Paragraph number="2372" type="verses" book="ja1">
_Abhave nandati tassa,_  
_bhave tassa na nandati;_  
_Accheraṃ bhojanaṃ laddhā,_  
_tassa nuppajjate sati;_  
_Tato naṃ nānukampati,_  
_aho sopi labheyyito._  

</Paragraph>

</Division>
<Division number="126" book="ja1" bookNo="22" page="88">

<Paragraph number="2373" type="verses" book="ja1">
_Iccete soḷasākārā,_  
_amittasmiṃ patiṭṭhitā;_  
_Yehi amittaṃ jāneyya,_  
_disvā sutvā ca paṇḍito”._  

</Paragraph>

</Division>

</Paragraph>

</Division>
<Division number="128" book="ja1" bookNo="22" page="89">

<Paragraph number="2375" type="verses" book="ja1">
_Pavutthaṃ tassa sarati,_  
_āgataṃ abhinandati;_  
_Tato kelāyito hoti,_  
_vācāya paṭinandati._  

</Paragraph>

</Division>
<Division number="129" book="ja1" bookNo="22" page="89">

<Paragraph number="2376" type="verses" book="ja1">
_Mitte tasseva bhajati,_  
_amitte tassa na sevati;_  
_Akkosante nivāreti,_  
_vaṇṇakāme pasaṃsati._  

</Paragraph>

</Division>
<Division number="130" book="ja1" bookNo="22" page="89">

<Paragraph number="2377" type="verses" book="ja1">
_Guyhañca tassa akkhāti,_  
_tassa guyhañca gūhati;_  
_Kammañca tassa vaṇṇeti,_  
_paññaṃ tassa pasaṃsati._  

</Paragraph>

</Division>
<Division number="131" book="ja1" bookNo="22" page="89">

<Paragraph number="2378" type="verses" book="ja1">

_abhave tassa na nandati;_  
_Accheraṃ bhojanaṃ laddhā,_  
_tassa uppajjate sati;_  
_Tato naṃ anukampati,_  
_aho sopi labheyyito._  

</Paragraph>

</Division>

_Mittasmiṃ suppatiṭṭhitā;_  
_Yehi mittañca jāneyya,_  
_Disvā sutvā ca paṇḍito”ti. (1779)_  


</Paragraph>
