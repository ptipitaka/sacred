<Paragraph number="1126" type="verses" book="ja1">
_“Araññā gāmamāgamma,_  
_kiṃsīlaṃ kiṃvataṃ ahaṃ;_  
_Purisaṃ tāta seveyyaṃ,_  
_taṃ me akkhāhi pucchito”._  

</Paragraph>

<Division number="190" book="ja1">

<Paragraph number="1127" type="verses" book="ja1">
_“Yo taṃ vissāsaye tāta,_  
_vissāsañca khameyya te;_  
_Sussūsī ca titikkhī ca,_  
_taṃ bhajehi ito gato._  

</Paragraph>

</Division>
<Division number="191" book="ja1">

<Paragraph number="1128" type="verses" book="ja1">
_Yassa kāyena vācāya,_  
_manasā natthi dukkaṭaṃ;_  
_Urasīva patiṭṭhāya,_  
_taṃ bhajehi ito gato._  

</Paragraph>

</Division>

_purisaṃ rāgavirāginaṃ;_  
_Tādisaṃ tāta mā sevi,_  
_nimmanussampi ce siyā”ti._  


</Paragraph>
