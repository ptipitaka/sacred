<Division number="13" book="ja1">

<Paragraph number="2120" type="verses" book="ja1">
_“Suṇohi mayhaṃ vacanaṃ janinda,_  
_Atthena juṇhamhi idhānupatto;_  
_Na brāhmaṇe addhike tiṭṭhamāne,_  
_Gantabbamāhu dvipadindaseṭṭha”._  

</Paragraph>

</Division>
<Division number="14" book="ja1">

<Paragraph number="2121" type="verses" book="ja1">
_“Suṇomi tiṭṭhāmi vadehi brahme,_  
_Yenāsi atthena idhānupatto;_  
_Kaṃ vā tvamatthaṃ mayi patthayāno,_  
_Idhāgamā brahme tadiṅgha brūhi”._  

</Paragraph>

</Division>
<Division number="15" book="ja1">

<Paragraph number="2122" type="verses" book="ja1">
_“Dadāhi me gāmavarāni pañca,_  
_Dāsīsataṃ satta gavaṃsatāni;_  
_Parosahassañca suvaṇṇanikkhe,_  
_Bhariyā ca me sādisī dve dadāhi._  

</Paragraph>

</Division>
<Division number="16" book="ja1">

<Paragraph number="2123" type="verses" book="ja1">
_Tapo nu te brāhmaṇa bhiṃsarūpo,_  
_Mantā nu te brāhmaṇa cittarūpā;_  
_Yakkhā nu te assavā santi keci,_  
_Atthaṃ vā me abhijānāsi kattaṃ._  

</Paragraph>

</Division>
<Division number="17" book="ja1">

<Paragraph number="2124" type="verses" book="ja1">
_Na me tapo atthi na cāpi mantā,_  
_Yakkhāpi me assavā natthi keci;_  
_Atthampi te nābhijānāmi kattaṃ,_  
_Pubbe ca kho saṅgatimattamāsi._  

</Paragraph>

</Division>
<Division number="18" book="ja1">

<Paragraph number="2125" type="verses" book="ja1">
_Paṭhamaṃ idaṃ dassanaṃ jānato me,_  
_Na tābhijānāmi ito puratthā;_  
_Akkhāhi me pucchito etamatthaṃ,_  
_Kadā kuhiṃ vā ahu saṅgamo no._  

</Paragraph>

</Division>
<Division number="19" book="ja1">

<Paragraph number="2126" type="verses" book="ja1">
_Gandhārarājassa puramhi ramme,_  
_Avasimhase takkasīlāyaṃ deva;_  
_Tatthandhakāramhi timīsikāyaṃ,_  
_Aṃsena aṃsaṃ samaghaṭṭayimha._  

</Paragraph>

</Division>
<Division number="20" book="ja1">

<Paragraph number="2127" type="verses" book="ja1">
_Te tattha ṭhatvāna ubho janinda,_  
_Sārāṇiyaṃ vītisārayimha tattha;_  
_Sāyeva no saṅgatimattamāsi,_  
_Tato na pacchā na pure ahosi._  

</Paragraph>

</Division>
<Division number="21" book="ja1">

<Paragraph number="2128" type="verses" book="ja1">
_Yadā kadāci manujesu brahme,_  
_Samāgamo sappurisena hoti;_  
_Na paṇḍitā saṅgatisanthavāni,_  
_Pubbe kataṃ vāpi vināsayanti._  

</Paragraph>

</Division>
<Division number="22" book="ja1">

<Paragraph number="2129" type="verses" book="ja1">
_Bālāva kho saṅgatisanthavāni,_  
_Pubbe kataṃ vāpi vināsayanti;_  
_Bahumpi bālesu kataṃ vinassati,_  
_Tathā hi bālā akataññurūpā._  

</Paragraph>

</Division>
<Division number="23" book="ja1">

<Paragraph number="2130" type="verses" book="ja1">
_Dhīrā ca kho saṅgatisanthavāni,_  
_Pubbe kataṃ vāpi na nāsayanti;_  
_Appampi dhīresu kataṃ na nassati,_  
_Tathā hi dhīrā sukataññurūpā._  

</Paragraph>

</Division>
<Division number="24" book="ja1">

<Paragraph number="2131" type="verses" book="ja1">
_Dadāmi te gāmavarāni pañca,_  
_Dāsīsataṃ satta gavaṃsatāni;_  
_Parosahassañca suvaṇṇanikkhe,_  
_Bhariyā ca te sādisī dve dadāmi._  

</Paragraph>

</Division>
<Division number="25" book="ja1">

<Paragraph number="2132" type="verses" book="ja1">
_Evaṃ sataṃ hoti samecca rāja,_  
_Nakkhattarājāriva tārakānaṃ;_  
_Āpūratī kāsipatī tathāhaṃ,_  
_Tayāpi me saṅgamo ajja laddho”ti._  


</Paragraph>
</Division>

Juṇhajātakaṃ dutiyaṃ.
</Paragraph>
