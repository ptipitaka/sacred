<Paragraph number="2291" type="verses" book="ja1">
_“Dasa khalu imāni ṭhānāni,_  
_Yāni pubbe akaritvā;_  
_Sa pacchā manutappati,_  
_Iccevāha janasandho._  

</Paragraph>

</Division>
<Division number="50" book="ja1" bookNo="22" page="99">

<Paragraph number="2292" type="verses" book="ja1">
_Aladdhā vittaṃ tappati,_  
_pubbe asamudānitaṃ;_  
_Na pubbe dhanamesissaṃ,_  
_iti pacchānutappati._  

</Paragraph>

</Division>
<Division number="51" book="ja1" bookNo="22" page="99">

<Paragraph number="2293" type="verses" book="ja1">
_Sakyarūpaṃ pure santaṃ,_  
_mayā sippaṃ na sikkhitaṃ;_  
_Kicchā vutti asippassa,_  
_iti pacchānutappati._  

</Paragraph>

</Division>
<Division number="52" book="ja1" bookNo="22" page="100">

<Paragraph number="2294" type="verses" book="ja1">
_Kūṭavedī pure āsiṃ,_  
_pisuṇo piṭṭhimaṃsiko;_  
_Caṇḍo ca pharuso cāpi,_  
_iti pacchānutappati._  

</Paragraph>

</Division>
<Division number="53" book="ja1" bookNo="22" page="100">

<Paragraph number="2295" type="verses" book="ja1">
_Pāṇātipātī pure āsiṃ,_  
_luddo cāpi anāriyo;_  
_Bhūtānaṃ nāpacāyissaṃ,_  
_iti pacchānutappati._  

</Paragraph>

</Division>
<Division number="54" book="ja1" bookNo="22" page="100">

<Paragraph number="2296" type="verses" book="ja1">
_Bahūsu vata santīsu,_  
_anāpādāsu itthisu;_  
_Paradāraṃ asevissaṃ,_  
_iti pacchānutappati._  

</Paragraph>

</Division>
<Division number="55" book="ja1" bookNo="22" page="100">

<Paragraph number="2297" type="verses" book="ja1">
_Bahumhi vata santamhi,_  
_annapāne upaṭṭhite;_  
_Na pubbe adadaṃ dānaṃ,_  
_iti pacchānutappati._  

</Paragraph>

</Division>
<Division number="56" book="ja1" bookNo="22" page="100">

<Paragraph number="2298" type="verses" book="ja1">
_Mātaraṃ pitarañcāpi,_  
_jiṇṇakaṃ gatayobbanaṃ;_  
_Pahu santo na posissaṃ,_  
_iti pacchānutappati._  

</Paragraph>

</Division>
<Division number="57" book="ja1" bookNo="22" page="100">

<Paragraph number="2299" type="verses" book="ja1">
_Ācariyamanusatthāraṃ,_  
_sabbakāmarasāharaṃ;_  
_Pitaraṃ atimaññissaṃ,_  
_iti pacchānutappati._  

</Paragraph>

</Division>
<Division number="58" book="ja1" bookNo="22" page="100">

<Paragraph number="2300" type="verses" book="ja1">
_Samaṇe brāhmaṇe cāpi,_  
_sīlavante bahussute;_  
_Na pubbe payirupāsissaṃ,_  
_iti pacchānutappati._  

</Paragraph>

</Division>
<Division number="59" book="ja1" bookNo="22" page="100">

<Paragraph number="2301" type="verses" book="ja1">
_Sādhu hoti tapo ciṇṇo,_  
_santo ca payirupāsito;_  
_Na ca pubbe tapo ciṇṇo,_  
_iti pacchānutappati._  

</Paragraph>

</Division>
<Division number="60" book="ja1" bookNo="22" page="100">

<Paragraph number="2302" type="verses" book="ja1">
_Yo ca etāni ṭhānāni,_  
_yoniso paṭipajjati;_  
_Karaṃ purisakiccāni,_  
_sa pacchā nānutappatī”ti._  


</Paragraph>
</Division>

<Paragraph number="2303" type="center" book="ja1">
Janasandhajātakaṃ pañcamaṃ.
</Paragraph>
