<Division number="103" book="ja1">

<Paragraph number="1436" type="verses" book="ja1">
_“Sakuṇo mayhako nāma,_  
_girisānudarīcaro;_  
_Pakkaṃ pipphalimāruyha,_  
_mayhaṃ mayhanti kandati._  

</Paragraph>

</Division>
<Division number="104" book="ja1">

<Paragraph number="1437" type="verses" book="ja1">
_Tassevaṃ vilapantassa,_  
_dijasaṅghā samāgatā;_  
_Bhutvāna pipphaliṃ yanti,_  
_vilapatveva so dijo._  

</Paragraph>

</Division>
<Division number="105" book="ja1">

<Paragraph number="1438" type="verses" book="ja1">
_Evameva idhekacco,_  
_saṅgharitvā bahuṃ dhanaṃ;_  
_Nevattano na ñātīnaṃ,_  
_yathodhiṃ paṭipajjati._  

</Paragraph>

</Division>
<Division number="106" book="ja1">

<Paragraph number="1439" type="verses" book="ja1">
_Na so acchādanaṃ bhattaṃ,_  
_na mālaṃ na vilepanaṃ;_  
_Anubhoti sakiṃ kiñci,_  
_na saṅgaṇhāti ñātake._  

</Paragraph>

</Division>
<Division number="107" book="ja1">

<Paragraph number="1440" type="verses" book="ja1">
_Tassevaṃ vilapantassa,_  
_mayhaṃ mayhanti rakkhato;_  
_Rājāno atha vā corā,_  
_dāyādā ye va appiyā;_  
_Dhanamādāya gacchanti,_  
_vilapatveva so naro._  

</Paragraph>

</Division>
<Division number="108" book="ja1">

<Paragraph number="1441" type="verses" book="ja1">
_Dhīro bhoge adhigamma,_  
_saṅgaṇhāti ca ñātake;_  
_Tena so kittiṃ pappoti,_  
_pecca sagge pamodatī”ti._  


</Paragraph>
</Division>

<Paragraph number="1442" type="center" book="ja1">
Mayhakajātakaṃ pañcamaṃ.
</Paragraph>
