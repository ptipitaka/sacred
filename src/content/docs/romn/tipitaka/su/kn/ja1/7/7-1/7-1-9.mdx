<Paragraph number="1553" type="verses" book="ja1">
_“Yattha verī nivasati,_  
_na vase tattha paṇḍito;_  
_Ekarattaṃ dirattaṃ vā,_  
_dukkhaṃ vasati verisu._  

</Paragraph>

</Division>
<Division number="62" book="ja1" bookNo="22" page="283">

<Paragraph number="1554" type="verses" book="ja1">
_Diso ve lahucittassa,_  
_posassānuvidhīyato;_  
_Ekassa kapino hetu,_  
_yūthassa anayo kato._  

</Paragraph>

</Division>
<Division number="63" book="ja1" bookNo="22" page="283">

<Paragraph number="1555" type="verses" book="ja1">
_Bālova paṇḍitamānī,_  
_yūthassa parihārako;_  
_Sacittassa vasaṃ gantvā,_  
_sayethāyaṃ yathā kapi._  

</Paragraph>

</Division>
<Division number="64" book="ja1" bookNo="22" page="283">

<Paragraph number="1556" type="verses" book="ja1">
_Na sādhu balavā bālo,_  
_yūthassa parihārako;_  
_Ahito bhavati ñātīnaṃ,_  
_sakuṇānaṃva cetako._  

</Paragraph>

</Division>
<Division number="65" book="ja1" bookNo="22" page="283">

<Paragraph number="1557" type="verses" book="ja1">
_Dhīrova balavā sādhu,_  
_yūthassa parihārako;_  
_Hito bhavati ñātīnaṃ,_  
_tidasānaṃva vāsavo._  

</Paragraph>

</Division>
<Division number="66" book="ja1" bookNo="22" page="283">

<Paragraph number="1558" type="verses" book="ja1">
_Yo ca sīlañca paññañca,_  
_sutañcattani passati;_  
_Ubhinnamatthaṃ carati,_  
_attano ca parassa ca._  

</Paragraph>

</Division>
<Division number="67" book="ja1" bookNo="22" page="283">

<Paragraph number="1559" type="verses" book="ja1">
_Tasmā tuleyya mattānaṃ,_  
_sīlapaññāsutāmiva;_  
_Gaṇaṃ vā parihare dhīro,_  
_eko vāpi paribbaje”ti._  


</Paragraph>
</Division>

<Paragraph number="1560" type="center" book="ja1">
Kapijātakaṃ navamaṃ.
</Paragraph>
