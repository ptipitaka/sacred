<Division number="125" book="ja1">

<Paragraph number="2044" type="verses" book="ja1">
_“Apacantāpi dicchanti,_  
_santo laddhāna bhojanaṃ;_  
_Kimeva tvaṃ pacamāno,_  
_yaṃ na dajjā na taṃ samaṃ._  

</Paragraph>

</Division>
<Division number="126" book="ja1">

<Paragraph number="2045" type="verses" book="ja1">
_Maccherā ca pamādā ca,_  
_evaṃ dānaṃ na diyyati;_  
_Puññaṃ ākaṅkhamānena,_  
_deyyaṃ hoti vijānatā._  

</Paragraph>

</Division>
<Division number="127" book="ja1">

<Paragraph number="2046" type="verses" book="ja1">
_Yasseva bhīto na dadāti maccharī,_  
_Tadevādadato bhayaṃ;_  
_Jighacchā ca pipāsā ca,_  
_Yassa bhāyati maccharī;_  
_Tameva bālaṃ phusati,_  
_Asmiṃ loke paramhi ca._  

</Paragraph>

</Division>
<Division number="128" book="ja1">

<Paragraph number="2047" type="verses" book="ja1">
_Tasmā vineyya maccheraṃ,_  
_dajjā dānaṃ malābhibhū;_  
_Puññāni paralokasmiṃ,_  
_patiṭṭhā honti pāṇinaṃ”._  

</Paragraph>

</Division>
<Division number="129" book="ja1">

<Paragraph number="2048" type="verses" book="ja1">
_“Duddadaṃ dadamānānaṃ,_  
_dukkaraṃ kamma kubbataṃ;_  
_Asanto nānukubbanti,_  
_sataṃ dhammo durannayo._  

</Paragraph>

</Division>
<Division number="130" book="ja1">

<Paragraph number="2049" type="verses" book="ja1">
_Tasmā satañca asataṃ,_  
_nānā hoti ito gati;_  
_Asanto nirayaṃ yanti,_  
_santo saggaparāyaṇā”._  

</Paragraph>

</Division>
<Division number="131" book="ja1">

<Paragraph number="2050" type="verses" book="ja1">
_“Appasmeke pavecchanti,_  
_bahuneke na dicchare;_  
_Appasmā dakkhiṇā dinnā,_  
_sahassena samaṃ mitā”._  

</Paragraph>

</Division>
<Division number="132" book="ja1">

<Paragraph number="2051" type="verses" book="ja1">
_“Dhammaṃ care yopi samuñchakaṃ care,_  
_Dārañca posaṃ dadamappakasmiṃ;_  
_Sataṃ sahassānaṃ sahassayāginaṃ,_  
_Kalampi nāgghanti tathāvidhassa te”._  

</Paragraph>

</Division>
<Division number="133" book="ja1">

<Paragraph number="2052" type="verses" book="ja1">
_“Kenesa yañño vipulo mahagghato,_  
_Samena dinnassa na agghameti;_  
_Kathaṃ sataṃ sahassānaṃ sahassayāginaṃ,_  
_Kalampi nāgghanti tathāvidhassa te”._  

</Paragraph>

</Division>
<Division number="134" book="ja1">

<Paragraph number="2053" type="verses" book="ja1">
_“Dadanti heke visame niviṭṭhā,_  
_Chetvā vadhitvā atha socayitvā;_  
_Sā dakkhiṇā assumukhā sadaṇḍā,_  
_Samena dinnassa na agghameti;_  
_Evaṃ sataṃ sahassānaṃ sahassayāginaṃ,_  
_Kalampi nāgghanti tathāvidhassa te”ti._  


</Paragraph>
</Division>

<Paragraph number="2054" type="center" book="ja1">
Bilārakosiyajātakaṃ dvādasamaṃ.
</Paragraph>
