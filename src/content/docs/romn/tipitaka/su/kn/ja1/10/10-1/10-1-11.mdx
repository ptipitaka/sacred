<Division number="115" book="ja1">

<Paragraph number="2033" type="verses" book="ja1">
_“Alaṅkato maṭṭhakuṇḍalī,_  
_Māladhārī haricandanussado;_  
_Bāhā paggayha kandasi,_  
_Vanamajjhe kiṃ dukkhito tuvaṃ”._  

</Paragraph>

</Division>
<Division number="116" book="ja1">

<Paragraph number="2034" type="verses" book="ja1">
_“Sovaṇṇamayo pabhassaro,_  
_Uppanno rathapañjaro mama;_  
_Tassa cakkayugaṃ na vindāmi,_  
_Tena dukkhena jahāmi jīvitaṃ”._  

</Paragraph>

</Division>
<Division number="117" book="ja1">

<Paragraph number="2035" type="verses" book="ja1">
_“Sovaṇṇamayaṃ maṇīmayaṃ,_  
_Lohamayaṃ atha rūpiyāmayaṃ;_  
_Pāvada rathaṃ karissāmi te,_  
_Cakkayugaṃ paṭipādayāmi taṃ”._  

</Paragraph>

</Division>
<Division number="118" book="ja1">

<Paragraph number="2036" type="verses" book="ja1">
_“So māṇavo tassa pāvadi,_  
_Candasūriyā ubhayettha bhātaro;_  
_Sovaṇṇamayo ratho mama,_  
_Tena cakkayugena sobhati”._  

</Paragraph>

</Division>
<Division number="119" book="ja1">

<Paragraph number="2037" type="verses" book="ja1">
_“Bālo kho tvaṃsi māṇava,_  
_Yo tvaṃ patthayase apatthiyaṃ;_  
_Maññāmi tuvaṃ marissasi,_  
_Na hi tvaṃ lacchasi candasūriye”._  

</Paragraph>

</Division>
<Division number="120" book="ja1">

<Paragraph number="2038" type="verses" book="ja1">
_“Gamanāgamanampi dissati,_  
_Vaṇṇadhātu ubhayettha vīthiyo;_  
_Peto pana neva dissati,_  
_Ko nu kho kandataṃ bālyataro”._  

</Paragraph>

</Division>
<Division number="121" book="ja1">

<Paragraph number="2039" type="verses" book="ja1">
_“Saccaṃ kho vadesi māṇava,_  
_Ahameva kandataṃ bālyataro;_  
_Candaṃ viya dārako rudaṃ,_  
_Petaṃ kālakatābhipatthaye._  

</Paragraph>

</Division>
<Division number="122" book="ja1">

<Paragraph number="2040" type="verses" book="ja1">
_Ādittaṃ vata maṃ santaṃ,_  
_ghatasittaṃva pāvakaṃ;_  
_Vārinā viya osiñcaṃ,_  
_sabbaṃ nibbāpaye daraṃ._  

</Paragraph>

</Division>
<Division number="123" book="ja1">

<Paragraph number="2041" type="verses" book="ja1">
_Abbahī vata me sallaṃ,_  
_yamāsi hadayassitaṃ;_  
_Yo me sokaparetassa,_  
_puttasokaṃ apānudi._  

</Paragraph>

</Division>
<Division number="124" book="ja1">

<Paragraph number="2042" type="verses" book="ja1">
_Sohaṃ abbūḷhasallosmi,_  
_vītasoko anāvilo;_  
_Na socāmi na rodāmi,_  
_tava sutvāna māṇavā”ti._  


</Paragraph>
</Division>

<Paragraph number="2043" type="center" book="ja1">
Maṭṭhakuṇḍalījātakaṃ ekādasamaṃ.
</Paragraph>
