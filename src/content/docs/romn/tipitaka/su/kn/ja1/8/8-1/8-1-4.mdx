<Paragraph number="1702" type="verses" book="ja1">
_“Bhusamhi kuddhoti avekkhiyāna,_  
_Na tāva daṇḍaṃ paṇayeyya issaro;_  
_Aṭṭhānaso appatirūpamattano,_  
_Parassa dukkhāni bhusaṃ udīraye._  

</Paragraph>

</Division>
<Division number="28" book="ja1" bookNo="22" page="313">

<Paragraph number="1703" type="verses" book="ja1">
_Yato ca jāneyya pasādamattano,_  
_Atthaṃ niyuñjeyya parassa dukkaṭaṃ;_  
_Tadāyamatthoti sayaṃ avekkhiya,_  
_Athassa daṇḍaṃ sadisaṃ nivesaye._  

</Paragraph>

</Division>
<Division number="29" book="ja1" bookNo="22" page="313">

<Paragraph number="1704" type="verses" book="ja1">
_Na cāpi jhāpeti paraṃ na attanaṃ,_  
_Amucchito yo nayate nayānayaṃ;_  
_Yo daṇḍadhāro bhavatīdha issaro,_  
_Sa vaṇṇagutto siriyā na dhaṃsati._  

</Paragraph>

</Division>
<Division number="30" book="ja1" bookNo="22" page="313">

<Paragraph number="1705" type="verses" book="ja1">
_Ye khattiyā se anisammakārino,_  
_Paṇenti daṇḍaṃ sahasā pamucchitā;_  
_Avaṇṇasaṃyutā jahanti jīvitaṃ,_  
_Ito vimuttāpi ca yanti duggatiṃ._  

</Paragraph>

</Division>
<Division number="31" book="ja1" bookNo="22" page="313">

<Paragraph number="1706" type="verses" book="ja1">
_Dhamme ca ye ariyappavedite ratā,_  
_Anuttarā te vacasā manasā kammunā ca;_  
_Te santisoraccasamādhisaṇṭhitā,_  
_Vajanti lokaṃ dubhayaṃ tathāvidhā._  

</Paragraph>

</Division>
<Division number="32" book="ja1" bookNo="22" page="313">

<Paragraph number="1707" type="verses" book="ja1">
_Rājāhamasmi narapamadānamissaro,_  
_Sacepi kujjhāmi ṭhapemi attanaṃ;_  
_Nisedhayanto janataṃ tathāvidhaṃ,_  
_Paṇemi daṇḍaṃ anukampa yoniso”._  

</Paragraph>

</Division>
<Division number="33" book="ja1" bookNo="22" page="313">

<Paragraph number="1708" type="verses" book="ja1">
_“Sirī ca lakkhī ca taveva khattiya,_  
_Janādhipa mā vijahi kudācanaṃ;_  
_Akkodhano niccapasannacitto,_  
_Anīgho tuvaṃ vassasatāni pālaya._  

</Paragraph>

</Division>
<Division number="34" book="ja1" bookNo="22" page="313">

<Paragraph number="1709" type="verses" book="ja1">
_Guṇehi etehi upeta khattiya,_  
_Ṭhitamariyavattī suvaco akodhano;_  
_Sukhī anuppīḷa pasāsamediniṃ,_  
_Ito vimuttopi ca yāhi suggatiṃ._  

</Paragraph>

</Division>
<Division number="35" book="ja1" bookNo="22" page="313">

<Paragraph number="1710" type="verses" book="ja1">
_Evaṃ sunītena subhāsitena,_  
_Dhammena ñāyena upāyaso nayaṃ;_  
_Nibbāpaye saṅkhubhitaṃ mahājanaṃ,_  
_Mahāva megho salilena medinin”ti._  


</Paragraph>
</Division>

<Paragraph number="1711" type="center" book="ja1">
Sumaṅgalajātakaṃ catutthaṃ.
</Paragraph>
