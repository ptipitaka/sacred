<Division number="105" book="ja1">

<Paragraph number="1605" type="verses" book="ja1">
_“Yo maṃ pure paccuḍḍeti,_  
_araññe dūramāyato;_  
_So na dissati mātaṅgo,_  
_somadatto kuhiṃ gato._  

</Paragraph>

</Division>
<Division number="106" book="ja1">

<Paragraph number="1606" type="verses" book="ja1">
_Ayaṃ vā so mato seti,_  
_allasiṅgaṃva vacchito;_  
_Bhumyā nipatito seti,_  
_amarā vata kuñjaro”._  

</Paragraph>

</Division>
<Division number="107" book="ja1">

<Paragraph number="1607" type="verses" book="ja1">
_“Anagāriyupetassa,_  
_vippamuttassa te sato;_  
_Samaṇassa na taṃ sādhu,_  
_yaṃ petamanusocasi”._  

</Paragraph>

</Division>
<Division number="108" book="ja1">

<Paragraph number="1608" type="verses" book="ja1">
_“Saṃvāsena have sakka,_  
_manussassa migassa vā;_  
_Hadaye jāyate pemaṃ,_  
_taṃ na sakkā asocituṃ”._  

</Paragraph>

</Division>
<Division number="109" book="ja1">

<Paragraph number="1609" type="verses" book="ja1">
_“Mataṃ marissaṃ rodanti,_  
_ye rudanti lapanti ca;_  
_Tasmā tvaṃ isi mā rodi,_  
_roditaṃ moghamāhu santo._  

</Paragraph>

</Division>
<Division number="110" book="ja1">

<Paragraph number="1610" type="verses" book="ja1">
_Kanditena have brahme,_  
_mato peto samuṭṭhahe;_  
_Sabbe saṅgamma rodāma,_  
_aññamaññassa ñātake”._  

</Paragraph>

</Division>
<Division number="111" book="ja1">

<Paragraph number="1611" type="verses" book="ja1">
_“Ādittaṃ vata maṃ santaṃ,_  
_ghatasittaṃva pāvakaṃ;_  
_Vārinā viya osiñcaṃ,_  
_sabbaṃ nibbāpaye daraṃ._  

</Paragraph>

</Division>
<Division number="112" book="ja1">

<Paragraph number="1612" type="verses" book="ja1">
_Abbahī vata me sallaṃ,_  
_yamāsi hadayassitaṃ;_  
_Yo me sokaparetassa,_  
_puttasokaṃ apānudi._  

</Paragraph>

</Division>
<Division number="113" book="ja1">

<Paragraph number="1613" type="verses" book="ja1">
_Sohaṃ abbūḷhasallosmi,_  
_vītasoko anāvilo;_  
_Na socāmi na rodāmi,_  
_tava sutvāna vāsavā”ti._  


</Paragraph>
</Division>

<Paragraph number="1614" type="center" book="ja1">
Somadattajātakaṃ pañcamaṃ.
</Paragraph>
