<Paragraph number="3520" type="verses" book="ja1">
_“Vikiṇṇavācaṃ aniguyha mantaṃ,_  
_Asaññataṃ aparicakkhitāraṃ;_  
_Bhayaṃ tamanveti sayaṃ abodhaṃ,_  
_Nāgaṃ yathā paṇḍarakaṃ supaṇṇo._  

</Paragraph>

</Division>
<Division number="259" book="ja1" bookNo="22" page="387">

<Paragraph number="3521" type="verses" book="ja1">
_Yo guyhamantaṃ parirakkhaṇeyyaṃ,_  
_Mohā naro saṃsati hāsamāno;_  
_Taṃ bhinnamantaṃ bhayamanveti khippaṃ,_  
_Nāgaṃ yathā paṇḍarakaṃ supaṇṇo._  

</Paragraph>

</Division>
<Division number="260" book="ja1" bookNo="22" page="387">

<Paragraph number="3522" type="verses" book="ja1">
_Nānumitto garuṃ atthaṃ,_  
_guyhaṃ veditumarahati;_  
_Sumitto ca asambuddhaṃ,_  
_sambuddhaṃ vā anatthavā._  

</Paragraph>

</Division>
<Division number="261" book="ja1" bookNo="22" page="388">

<Paragraph number="3523" type="verses" book="ja1">
_Vissāsamāpajjimahaṃ acelaṃ,_  
_Samaṇo ayaṃ sammato bhāvitatto;_  
_Tassāhamakkhiṃ vivariṃ guyhamatthaṃ,_  
_Atītamattho kapaṇaṃ rudāmi._  

</Paragraph>

</Division>
<Division number="262" book="ja1" bookNo="22" page="388">

<Paragraph number="3524" type="verses" book="ja1">
_Tassāhaṃ paramaṃ brahme guyhaṃ,_  
_Vācañhi maṃ nāsakkhiṃ saṃyametuṃ;_  
_Tappakkhato hi bhayamāgataṃ mamaṃ,_  
_Atītamattho kapaṇaṃ rudāmi._  

</Paragraph>

</Division>
<Division number="263" book="ja1" bookNo="22" page="388">

<Paragraph number="3525" type="verses" book="ja1">
_Yo ve naro suhadaṃ maññamāno,_  
_Guyhamatthaṃ saṃsati dukkulīne;_  
_Dosā bhayā athavā rāgarattā,_  
_Pallatthito bālo asaṃsayaṃ so._  

</Paragraph>

</Division>
<Division number="264" book="ja1" bookNo="22" page="388">

<Paragraph number="3526" type="verses" book="ja1">
_Tirokkhavāco asataṃ paviṭṭho,_  
_Yo saṅgatīsu mudīreti vākyaṃ;_  
_Āsīviso dummukhotyāhu taṃ naraṃ,_  
_Ārā ārā saṃyame tādisamhā._  

</Paragraph>

</Division>
<Division number="265" book="ja1" bookNo="22" page="388">

<Paragraph number="3527" type="verses" book="ja1">
_Annaṃ pānaṃ kāsikacandanañca,_  
_Manāpitthiyo mālamucchādanañca;_  
_Ohāya gacchāmase sabbakāme,_  
_Supaṇṇa pāṇūpagatāva tyamhā”._  

</Paragraph>

</Division>
<Division number="266" book="ja1" bookNo="22" page="388">

<Paragraph number="3528" type="verses" book="ja1">
_“Ko nīdha tiṇṇaṃ garahaṃ upeti,_  
_Asmiṃdhaloke pāṇabhū nāgarāja;_  
_Samaṇo supaṇṇo athavā tvameva,_  
_Kiṃ kāraṇā paṇḍarakaggahīto”._  

</Paragraph>

</Division>
<Division number="267" book="ja1" bookNo="22" page="389">

<Paragraph number="3529" type="verses" book="ja1">
_“Samaṇoti me sammatatto ahosi,_  
_Piyo ca me manasā bhāvitatto;_  
_Tassāhamakkhiṃ vivariṃ guyhamatthaṃ,_  
_Atītamattho kapaṇaṃ rudāmi”._  

</Paragraph>

</Division>
<Division number="268" book="ja1" bookNo="22" page="389">

<Paragraph number="3530" type="verses" book="ja1">
_“Na catthi satto amaro pathabyā,_  
_Paññāvidhā natthi na ninditabbā;_  
_Saccena dhammena dhitiyā damena,_  
_Alabbhamabyāharatī naro idha._  

</Paragraph>

</Division>
<Division number="269" book="ja1" bookNo="22" page="389">

<Paragraph number="3531" type="verses" book="ja1">
_Mātāpitā paramā bandhavānaṃ,_  
_Nāssa tatiyo anukampakatthi;_  
_Tesampi guyhaṃ paramaṃ na saṃse,_  
_Mantassa bhedaṃ parisaṅkamāno._  

</Paragraph>

</Division>
<Division number="270" book="ja1" bookNo="22" page="389">

<Paragraph number="3532" type="verses" book="ja1">
_Mātāpitā bhaginī bhātaro ca,_  
_Sahāyā vā yassa honti sapakkhā;_  
_Tesampi guyhaṃ paramaṃ na saṃse,_  
_Mantassa bhedaṃ parisaṅkamāno._  

</Paragraph>

</Division>
<Division number="271" book="ja1" bookNo="22" page="389">

<Paragraph number="3533" type="verses" book="ja1">
_Bhariyā ce purisaṃ vajjā,_  
_Komārī piyabhāṇinī;_  
_Puttarūpayasūpetā,_  
_Ñātisaṅghapurakkhatā;_  
_Tassāpi guyhaṃ paramaṃ na saṃse,_  
_Mantassa bhedaṃ parisaṅkamāno._  

</Paragraph>

</Division>
<Division number="272" book="ja1" bookNo="22" page="389">

<Paragraph number="3534" type="verses" book="ja1">
_Na guyhamatthaṃ vivareyya,_  
_Rakkheyya naṃ yathā nidhiṃ;_  
_Na hi pātukato sādhu,_  
_Guyho attho pajānatā._  

</Paragraph>

</Division>
<Division number="273" book="ja1" bookNo="22" page="389">

<Paragraph number="3535" type="verses" book="ja1">
_Thiyā guyhaṃ na saṃseyya,_  
_amittassa ca paṇḍito;_  
_Yo cāmisena saṃhīro,_  
_hadayattheno ca yo naro._  

</Paragraph>

</Division>
<Division number="274" book="ja1" bookNo="22" page="389">

<Paragraph number="3536" type="verses" book="ja1">
_Guyhamatthaṃ asambuddhaṃ,_  
_sambodhayati yo naro;_  
_Mantabhedabhayā tassa,_  
_dāsabhūto titikkhati._  

</Paragraph>

</Division>
<Division number="275" book="ja1" bookNo="22" page="390">

<Paragraph number="3537" type="verses" book="ja1">
_Yāvanto purisassatthaṃ,_  
_guyhaṃ jānanti mantinaṃ;_  
_Tāvanto tassa ubbegā,_  
_tasmā guyhaṃ na vissaje._  

</Paragraph>

</Division>
<Division number="276" book="ja1" bookNo="22" page="390">

<Paragraph number="3538" type="verses" book="ja1">
_Vivicca bhāseyya divā rahassaṃ,_  
_Rattiṃ giraṃ nātivelaṃ pamuñce;_  
_Upassutikā hi suṇanti mantaṃ,_  
_Tasmā manto khippamupeti bhedaṃ._  

</Paragraph>

</Division>
<Division number="277" book="ja1" bookNo="22" page="390">

<Paragraph number="3539" type="verses" book="ja1">
_Yathāpi assa nagaraṃ mahantaṃ,_  
_Advārakaṃ āyasaṃ bhaddasālaṃ;_  
_Samantakhātāparikhāupetaṃ,_  
_Evampi me te idha guyhamantā._  

</Paragraph>

</Division>
<Division number="278" book="ja1" bookNo="22" page="390">

<Paragraph number="3540" type="verses" book="ja1">
_Ye guyhamantā avikiṇṇavācā,_  
_Daḷhā sadatthesu narā dujivha;_  
_Ārā amittā byavajanti tehi,_  
_Āsīvisā vā riva sattusaṅghā”._  

</Paragraph>

</Division>
<Division number="279" book="ja1" bookNo="22" page="390">

<Paragraph number="3541" type="verses" book="ja1">
_“Hitvā gharaṃ pabbajito acelo,_  
_Naggo muṇḍo carati ghāsahetu;_  
_Tamhi nu kho vivariṃ guyhamatthaṃ,_  
_Atthā ca dhammā ca apaggatamhā._  

</Paragraph>

</Division>
<Division number="280" book="ja1" bookNo="22" page="390">

<Paragraph number="3542" type="verses" book="ja1">
_Kathaṃkaro hoti supaṇṇarāja,_  
_Kiṃsīlo kena vatena vattaṃ;_  
_Samaṇo caraṃ hitvā mamāyitāni,_  
_Kathaṃkaro saggamupeti ṭhānaṃ”._  

</Paragraph>

</Division>
<Division number="281" book="ja1" bookNo="22" page="390">

<Paragraph number="3543" type="verses" book="ja1">
_“Hiriyā titikkhāya damenupeto,_  
_Akkodhano pesuṇiyaṃ pahāya;_  
_Samaṇo caraṃ hitvā mamāyitāni,_  
_Evaṃkaro saggamupeti ṭhānaṃ”._  

</Paragraph>

</Division>
<Division number="282" book="ja1" bookNo="22" page="391">

<Paragraph number="3544" type="verses" book="ja1">
_“Mātāva puttaṃ taruṇaṃ tanujjaṃ,_  
_Samphassatā sabbagattaṃ phareti;_  
_Evampi me tvaṃ pāturahu dijinda,_  
_Mātāva puttaṃ anukampamāno”._  

</Paragraph>

</Division>
<Division number="283" book="ja1" bookNo="22" page="391">

<Paragraph number="3545" type="verses" book="ja1">
_“Handajja tvaṃ muñca vadhā dujivha,_  
_Tayo hi puttā na hi añño atthi;_  
_Antevāsī dinnako atrajo ca,_  
_Rajjassu puttaññataro me ahosi”._  

</Paragraph>

</Division>
<Division number="284" book="ja1" bookNo="22" page="391">

<Paragraph number="3546" type="verses" book="ja1">
_Icceva vākyaṃ visajjī supaṇṇo,_  
_Bhumyaṃ patiṭṭhāya dijo dujivhaṃ;_  
_“Muttajja tvaṃ sabbabhayātivatto,_  
_Thalūdake hohi mayābhigutto._  

</Paragraph>

</Division>
<Division number="285" book="ja1" bookNo="22" page="391">

<Paragraph number="3547" type="verses" book="ja1">
_Ātaṅkinaṃ yathā kusalo bhisakko,_  
_Pipāsitānaṃ rahadova sīto;_  
_Vesmaṃ yathā himasītaṭṭitānaṃ,_  
_Evampi te saraṇamahaṃ bhavāmi”._  

</Paragraph>

</Division>
<Division number="286" book="ja1" bookNo="22" page="391">

<Paragraph number="3548" type="verses" book="ja1">
_“Sandhiṃ katvā amittena,_  
_aṇḍajena jalābuja;_  
_Vivariya dāṭhaṃ sesi,_  
_kuto taṃ bhayamāgataṃ”._  

</Paragraph>

</Division>
<Division number="287" book="ja1" bookNo="22" page="391">

<Paragraph number="3549" type="verses" book="ja1">
_“Saṅketheva amittasmiṃ,_  
_mittasmimpi na vissase;_  
_Abhayā bhayamuppannaṃ,_  
_api mūlāni kantati._  

</Paragraph>

</Division>
<Division number="288" book="ja1" bookNo="22" page="391">

<Paragraph number="3550" type="verses" book="ja1">
_Kathaṃ nu vissase tyamhi,_  
_yenāsi kalaho kato;_  
_Niccayattena ṭhātabbaṃ,_  
_so disabbhi na rajjati._  

</Paragraph>

</Division>
<Division number="289" book="ja1" bookNo="22" page="391">

<Paragraph number="3551" type="verses" book="ja1">
_Vissāsaye na ca taṃ vissayeyya,_  
_Asaṅkito saṅkito ca bhaveyya;_  
_Tathā tathā viññū parakkameyya,_  
_Yathā yathā bhāvaṃ paro na jaññā”._  

</Paragraph>

</Division>
<Division number="290" book="ja1" bookNo="22" page="392">

<Paragraph number="3552" type="verses" book="ja1">
_Te devavaṇṇā sukhumālarūpā,_  
_Ubho samā sujayā puññakhandhā;_  
_Upāgamuṃ karampiyaṃ acelaṃ,_  
_Missībhūtā assavāhāva nāgā._  

</Paragraph>

</Division>
<Division number="291" book="ja1" bookNo="22" page="392">

<Paragraph number="3553" type="verses" book="ja1">
_Tato have paṇḍarako acelaṃ,_  
_Sayamevupāgamma idaṃ avoca;_  
_“Muttajjahaṃ sabbabhayātivatto,_  
_Na hi nūna tuyhaṃ manaso piyamhā”._  

</Paragraph>

</Division>
<Division number="292" book="ja1" bookNo="22" page="392">

<Paragraph number="3554" type="verses" book="ja1">
_“Piyo hi me āsi supaṇṇarājā,_  
_Asaṃsayaṃ paṇḍarakena saccaṃ;_  
_So rāgarattova akāsimetaṃ,_  
_Pāpakammaṃ sampajāno na mohā”._  

</Paragraph>

</Division>
<Division number="293" book="ja1" bookNo="22" page="392">

<Paragraph number="3555" type="verses" book="ja1">
_“Na me piyaṃ appiyaṃ vāpi hoti,_  
_Sampassato lokamimaṃ parañca;_  
_Susaññatānañhi viyañjanena,_  
_Asaññato lokamimaṃ carāsi._  

</Paragraph>

</Division>
<Division number="294" book="ja1" bookNo="22" page="392">

<Paragraph number="3556" type="verses" book="ja1">
_Ariyāvakāsosi anariyovāsi,_  
_Asaññato saññatasannikāso;_  
_Kaṇhābhijātikosi anariyarūpo,_  
_Pāpaṃ bahuṃ duccaritaṃ acāri._  

</Paragraph>

</Division>
<Division number="295" book="ja1" bookNo="22" page="392">

<Paragraph number="3557" type="verses" book="ja1">
_Aduṭṭhassa tuvaṃ dubbhi,_  
_dubbhī ca pisuṇo casi;_  
_Etena saccavajjena,_  
_muddhā te phalatu sattadhā”._  

</Paragraph>

</Division>
<Division number="296" book="ja1" bookNo="22" page="392">

<Paragraph number="3558" type="verses" book="ja1">
_“Tasmā hi mittānaṃ na dubbhitabbaṃ,_  
_Mittadubbhā pāpiyo natthi añño;_  
_Āsittasatto nihato pathabyā,_  
_Indassa vākyena hi saṃvaro hato”ti._  


</Paragraph>
</Division>

<Paragraph number="3559" type="center" book="ja1">
Paṇḍaranāgarājajātakaṃ aṭṭhamaṃ.
