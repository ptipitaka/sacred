<Paragraph number="828" type="verses" book="ja1">
_“Nānāchandā mahārāja,_  
_ekāgāre vasāmase;_  
_Ahaṃ gāmavaraṃ icche,_  
_brāhmaṇī ca gavaṃ sataṃ”._  

</Paragraph>

</Division>
<Division number="116" book="ja1">

<Paragraph number="829" type="verses" book="ja1">
_Putto ca ājaññarathaṃ,_  
_kaññā ca maṇikuṇḍalaṃ;_  
_Yā cesā puṇṇikā jammī,_  
_udukkhalaṃbhikaṅkhati._  

</Paragraph>

</Division>
<Division number="117" book="ja1">

<Paragraph number="830" type="verses" book="ja1">
_“Brāhmaṇassa gāmavaraṃ,_  
_brāhmaṇiyā gavaṃ sataṃ;_  
_Puttassa ājaññarathaṃ,_  
_kaññāya maṇikuṇḍalaṃ;_  
_Yañcetaṃ puṇṇikaṃ jammiṃ,_  
_paṭipādethudukkhalan”ti._  


</Paragraph>
</Division>

<Paragraph number="831" type="center" book="ja1">
Nānāchandajātakaṃ navamaṃ.
</Paragraph>
