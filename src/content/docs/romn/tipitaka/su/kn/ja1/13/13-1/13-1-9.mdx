<Paragraph number="2508" type="verses" book="ja1">
_“Tassa gāmavaraṃ dammi,_  
_Nāriyo ca alaṅkatā;_  
_Yo metaṃ migamakkhāti,_  
_Migānaṃ migamuttamaṃ”._  

</Paragraph>

</Division>
<Division number="118" book="ja1" bookNo="22" page="134">

<Paragraph number="2509" type="verses" book="ja1">
_“Mayhaṃ gāmavaraṃ dehi,_  
_nāriyo ca alaṅkatā;_  
_Ahaṃ te migamakkhissaṃ,_  
_migānaṃ migamuttamaṃ._  

</Paragraph>

</Division>
<Division number="119" book="ja1" bookNo="22" page="134">

<Paragraph number="2510" type="verses" book="ja1">
_Etasmiṃ vanasaṇḍasmiṃ,_  
_ambā sālā ca pupphitā;_  
_Indagopakasañchannā,_  
_ettheso tiṭṭhate migo”._  

</Paragraph>

</Division>
<Division number="120" book="ja1" bookNo="22" page="134">

<Paragraph number="2511" type="verses" book="ja1">
_“Dhanuṃ advejjhaṃ katvāna,_  
_usuṃ sannayhupāgami;_  
_Migo ca disvā rājānaṃ,_  
_dūrato ajjhabhāsatha._  

</Paragraph>

</Division>
<Division number="121" book="ja1" bookNo="22" page="109">

<Paragraph number="2512" type="verses" book="ja1">
_Āgamehi mahārāja,_  
_mā maṃ vijjhi rathesabha;_  
_Ko nu te idamakkhāsi,_  
_ettheso tiṭṭhate migo”._  

</Paragraph>

</Division>
<Division number="122" book="ja1" bookNo="22" page="109">

<Paragraph number="2513" type="verses" book="ja1">
_“Esa pāpacaro poso,_  
_samma tiṭṭhati ārakā;_  
_Soyaṃ me idamakkhāsi,_  
_ettheso tiṭṭhate migo”._  

</Paragraph>

</Division>
<Division number="123" book="ja1" bookNo="22" page="109">

<Paragraph number="2514" type="verses" book="ja1">
_“Saccaṃ kireva māhaṃsu,_  
_narā ekacciyā idha;_  
_Kaṭṭhaṃ niplavitaṃ seyyo,_  
_na tvevekacciyo naro”._  

</Paragraph>

</Division>
<Division number="124" book="ja1" bookNo="22" page="109">

<Paragraph number="2515" type="verses" book="ja1">
_“Kiṃ nu ruru garahasi migānaṃ,_  
_Kiṃ pakkhīnaṃ kiṃ pana mānusānaṃ;_  
_Bhayañhi maṃ vindatinapparūpaṃ,_  
_Sutvāna taṃ mānusiṃ bhāsamānaṃ”._  

</Paragraph>

</Division>
<Division number="125" book="ja1" bookNo="22" page="109">

<Paragraph number="2516" type="verses" book="ja1">
_“Yamuddhariṃ vāhane vuyhamānaṃ,_  
_Mahodake salile sīghasote;_  
_Tatonidānaṃ bhayamāgataṃ mama,_  
_Dukkho have rāja asabbhi saṅgamo”._  

</Paragraph>

</Division>
<Division number="126" book="ja1" bookNo="22" page="109">

<Paragraph number="2517" type="verses" book="ja1">
_“Sohaṃ catuppattamimaṃ vihaṅgamaṃ,_  
_Tanucchidaṃ hadaye ossajāmi;_  
_Hanāmi taṃ mittadubbhiṃ akiccakāriṃ,_  
_Yo tādisaṃ kammakataṃ na jāne”._  

</Paragraph>

</Division>
<Division number="127" book="ja1" bookNo="22" page="109">

<Paragraph number="2518" type="verses" book="ja1">
_“Dhīrassa bālassa have janinda,_  
_Santo vadhaṃ nappasaṃsanti jātu;_  
_Kāmaṃ gharaṃ gacchatu pāpadhammo,_  
_Yañcassa bhaṭṭhaṃ tadetassa dehi;_  
_Ahañca te kāmakaro bhavāmi”._  

</Paragraph>

</Division>
<Division number="128" book="ja1" bookNo="22" page="109">

<Paragraph number="2519" type="verses" book="ja1">
_“Addhā rurū aññataro sataṃ so,_  
_Yo dubbhato mānusassa na dubbhi;_  
_Kāmaṃ gharaṃ gacchatu pāpadhammo,_  
_Yañcassa bhaṭṭhaṃ tadetassa dammi;_  
_Ahañca te kāmacāraṃ dadāmi”._  

</Paragraph>

</Division>
<Division number="129" book="ja1" bookNo="22" page="110">

<Paragraph number="2520" type="verses" book="ja1">
_“Suvijānaṃ siṅgālānaṃ,_  
_sakuṇānañca vassitaṃ;_  
_Manussavassitaṃ rāja,_  
_dubbijānataraṃ tato._  

</Paragraph>

</Division>
<Division number="130" book="ja1" bookNo="22" page="110">

<Paragraph number="2521" type="verses" book="ja1">
_Api ce maññatī poso,_  
_ñāti mitto sakhāti vā;_  
_Yo pubbe sumano hutvā,_  
_pacchā sampajjate diso”._  

</Paragraph>

</Division>
<Division number="131" book="ja1" bookNo="22" page="110">

<Paragraph number="2522" type="verses" book="ja1">
_“Samāgatā jānapadā,_  
_negamā ca samāgatā;_  
_Migā sassāni khādanti,_  
_taṃ devo paṭisedhatu”._  

</Paragraph>

</Division>
<Division number="132" book="ja1" bookNo="22" page="110">

<Paragraph number="2523" type="verses" book="ja1">
_“Kāmaṃ janapado māsi,_  
_raṭṭhañcāpi vinassatu;_  
_Na tvevāhaṃ ruruṃ dubbhe,_  
_datvā abhayadakkhiṇaṃ._  

</Paragraph>

</Division>
<Division number="133" book="ja1" bookNo="22" page="89">

<Paragraph number="2524" type="verses" book="ja1">
_Mā me janapado āsi,_  
_raṭṭhañcāpi vinassatu;_  
_Na tvevāhaṃ migarājassa,_  
_varaṃ datvā musā bhaṇe”ti._  


</Paragraph>
</Division>

<Paragraph number="2525" type="center" book="ja1">
Rurumigarājajātakaṃ navamaṃ.
</Paragraph>
