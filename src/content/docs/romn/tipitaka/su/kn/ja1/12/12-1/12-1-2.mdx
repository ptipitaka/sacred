<Division number="13" book="ja1">

<Paragraph number="2252" type="verses" book="ja1">
_“Kā tvaṃ suddhehi vatthehi,_  
_aghe vehāyasaṃ ṭhitā;_  
_Kena tyāssūni vattanti,_  
_kuto taṃ bhayamāgataṃ”._  

</Paragraph>

</Division>
<Division number="14" book="ja1">

<Paragraph number="2253" type="verses" book="ja1">
_“Taveva deva vijite,_  
_bhaddasāloti maṃ vidū;_  
_Saṭṭhi vassasahassāni,_  
_tiṭṭhato pūjitassa me._  

</Paragraph>

</Division>
<Division number="15" book="ja1">

<Paragraph number="2254" type="verses" book="ja1">
_Kārayantā nagarāni,_  
_agāre ca disampati;_  
_Vividhe cāpi pāsāde,_  
_na maṃ te accamaññisuṃ;_  
_Yatheva maṃ te pūjesuṃ,_  
_tatheva tvampi pūjaya”._  

</Paragraph>

</Division>
<Division number="16" book="ja1">

<Paragraph number="2255" type="verses" book="ja1">
_“Taṃ ivāhaṃ na passāmi,_  
_thūlaṃ kāyena te dumaṃ;_  
_Ārohapariṇāhena,_  
_abhirūposi jātiyā._  

</Paragraph>

</Division>
<Division number="17" book="ja1">

<Paragraph number="2256" type="verses" book="ja1">
_Pāsādaṃ kārayissāmi,_  
_ekatthambhaṃ manoramaṃ;_  
_Tattha taṃ upanessāmi,_  
_ciraṃ te yakkha jīvitaṃ”._  

</Paragraph>

</Division>
<Division number="18" book="ja1">

<Paragraph number="2257" type="verses" book="ja1">
_“Evaṃ cittaṃ udapādi,_  
_sarīrena vinābhāvo;_  
_Puthuso maṃ vikantitvā,_  
_khaṇḍaso avakantatha._  

</Paragraph>

</Division>
<Division number="19" book="ja1">

<Paragraph number="2258" type="verses" book="ja1">
_Agge ca chetvā majjhe ca,_  
_pacchā mūlamhi chindatha;_  
_Evaṃ me chijjamānassa,_  
_na dukkhaṃ maraṇaṃ siyā”._  

</Paragraph>

</Division>
<Division number="20" book="ja1">

<Paragraph number="2259" type="verses" book="ja1">
_“Hatthapādaṃ yathā chinde,_  
_Kaṇṇanāsañca jīvato;_  
_Tato pacchā siro chinde,_  
_Taṃ dukkhaṃ maraṇaṃ siyā._  

</Paragraph>

</Division>
<Division number="21" book="ja1">

<Paragraph number="2260" type="verses" book="ja1">
_Sukhaṃ nu khaṇḍaso chinnaṃ,_  
_bhaddasāla vanappati;_  
_Kiṃhetu kiṃ upādāya,_  
_khaṇḍaso chinnamicchasi”._  

</Paragraph>

</Division>
<Division number="22" book="ja1">

<Paragraph number="2261" type="verses" book="ja1">
_“Yañca hetumupādāya,_  
_hetuṃ dhammūpasaṃhitaṃ;_  
_Khaṇḍaso chinnamicchāmi,_  
_mahārāja suṇohi me._  

</Paragraph>

</Division>
<Division number="23" book="ja1">

<Paragraph number="2262" type="verses" book="ja1">
_Ñātī me sukhasaṃvaddhā,_  
_mama passe nivātajā;_  
_Tepihaṃ upahiṃseyyaṃ,_  
_paresaṃ asukhocitaṃ”._  

</Paragraph>

</Division>
<Division number="24" book="ja1">

<Paragraph number="2263" type="verses" book="ja1">
_“Ceteyyarūpaṃ cetesi,_  
_bhaddasāla vanappati;_  
_Hitakāmosi ñātīnaṃ,_  
_abhayaṃ samma dammi te”ti._  


</Paragraph>
</Division>

<Paragraph number="2264" type="center" book="ja1">
Bhaddasālajātakaṃ dutiyaṃ.
</Paragraph>
