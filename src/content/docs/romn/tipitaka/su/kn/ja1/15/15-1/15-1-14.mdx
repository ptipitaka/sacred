<Paragraph number="3228" type="verses" book="ja1">
_“Yamekarattiṃ paṭhamaṃ,_  
_gabbhe vasati māṇavo;_  
_Abbhuṭṭhitova so yāti,_  
_sagacchaṃ na nivattati._  

</Paragraph>

</Division>
<Division number="364" book="ja1">

<Paragraph number="3229" type="verses" book="ja1">
_Na yujjhamānā na balenavassitā,_  
_Narā na jīranti na cāpi miyyare;_  
_Sabbaṃ hidaṃ jātijarāyupaddutaṃ,_  
_Taṃ me matī hoti carāmi dhammaṃ._  

</Paragraph>

</Division>
<Division number="365" book="ja1">

<Paragraph number="3230" type="verses" book="ja1">
_Caturaṅginiṃ senaṃ subhiṃsarūpaṃ,_  
_Jayanti raṭṭhādhipatī pasayha;_  
_Na maccuno jayitumussahanti,_  
_Taṃ me matī hoti carāmi dhammaṃ._  

</Paragraph>

</Division>
<Division number="366" book="ja1">

<Paragraph number="3231" type="verses" book="ja1">
_Hatthīhi assehi rathehi pattibhi,_  
_Parivāritā muccare ekacceyyā;_  
_Na maccuno muccitumussahanti,_  
_Taṃ me matī hoti carāmi dhammaṃ._  

</Paragraph>

</Division>
<Division number="367" book="ja1">

<Paragraph number="3232" type="verses" book="ja1">
_Hatthīhi assehi rathehi pattibhi,_  
_Sūrā pabhañjanti padhaṃsayanti;_  
_Na maccuno bhañjitumussahanti,_  
_Taṃ me matī hoti carāmi dhammaṃ._  

</Paragraph>

</Division>
<Division number="368" book="ja1">

<Paragraph number="3233" type="verses" book="ja1">
_Mattā gajā bhinnagaḷā pabhinnā,_  
_Nagarāni maddanti janaṃ hananti;_  
_Na maccuno madditumussahanti,_  
_Taṃ me matī hoti carāmi dhammaṃ._  

</Paragraph>

</Division>
<Division number="369" book="ja1">

<Paragraph number="3234" type="verses" book="ja1">
_Issāsino katahatthāpi vīrā,_  
_Dūrepātī akkhaṇavedhinopi;_  
_Na maccuno vijjhitumussahanti,_  
_Taṃ me matī hoti carāmi dhammaṃ._  

</Paragraph>

</Division>
<Division number="370" book="ja1">

<Paragraph number="3235" type="verses" book="ja1">
_Sarāni khīyanti saselakānanā,_  
_Sabbaṃ hidaṃ khīyati dīghamantaraṃ;_  
_Sabbaṃ hidaṃ bhañjare kālapariyāyaṃ,_  
_Taṃ me matī hoti carāmi dhammaṃ._  

</Paragraph>

</Division>
<Division number="371" book="ja1">

<Paragraph number="3236" type="verses" book="ja1">
_Sabbesa mevaṃ hi narāna nārinaṃ,_  
_Calācalaṃ pāṇabhunodha jīvitaṃ;_  
_Paṭova dhuttassa dumova kūlajo,_  
_Taṃ me matī hoti carāmi dhammaṃ._  

</Paragraph>

</Division>
<Division number="372" book="ja1">

<Paragraph number="3237" type="verses" book="ja1">
_Dumapphalāneva patanti māṇavā,_  
_Daharā ca vuddhā ca sarīrabhedā;_  
_Nāriyo narā majjhimaporisā ca,_  
_Taṃ me matī hoti carāmi dhammaṃ._  

</Paragraph>

</Division>
<Division number="373" book="ja1">

<Paragraph number="3238" type="verses" book="ja1">
_Nāyaṃ vayo tārakarājasannibho,_  
_Yadabbhatītaṃ gatameva dāni taṃ;_  
_Jiṇṇassa hī natthi ratī kuto sukhaṃ,_  
_Taṃ me matī hoti carāmi dhammaṃ._  

</Paragraph>

</Division>
<Division number="374" book="ja1">

<Paragraph number="3239" type="verses" book="ja1">
_Yakkhā pisācā athavāpi petā,_  
_Kupitā te assasanti manusse;_  
_Na maccuno assasitussahanti,_  
_Taṃ me matī hoti carāmi dhammaṃ._  

</Paragraph>

</Division>
<Division number="375" book="ja1">

<Paragraph number="3240" type="verses" book="ja1">
_Yakkhe pisāce athavāpi pete,_  
_Kupitepi te nijjhapanaṃ karonti;_  
_Na maccuno nijjhapanaṃ karonti,_  
_Taṃ me matī hoti carāmi dhammaṃ._  

</Paragraph>

</Division>
<Division number="376" book="ja1">

<Paragraph number="3241" type="verses" book="ja1">
_Aparādhake dūsake heṭhake ca,_  
_Rājāno daṇḍenti viditvāna dosaṃ;_  
_Na maccuno daṇḍayitussahanti,_  
_Taṃ me matī hoti carāmi dhammaṃ._  

</Paragraph>

</Division>
<Division number="377" book="ja1">

<Paragraph number="3242" type="verses" book="ja1">
_Aparādhakā dūsakā heṭhakā ca,_  
_Labhanti te rājino nijjhapetuṃ;_  
_Na maccuno nijjhapanaṃ karonti,_  
_Taṃ me matī hoti carāmi dhammaṃ._  

</Paragraph>

</Division>
<Division number="378" book="ja1">

<Paragraph number="3243" type="verses" book="ja1">
_Na khattiyoti na ca brāhmaṇoti,_  
_Na aḍḍhakā balavā tejavāpi;_  
_Na maccurājassa apekkhamatthi,_  
_Taṃ me matī hoti carāmi dhammaṃ._  

</Paragraph>

</Division>
<Division number="379" book="ja1">

<Paragraph number="3244" type="verses" book="ja1">
_Sīhā ca byagghā ca athopi dīpiyo,_  
_Pasayha khādanti vipphandamānaṃ;_  
_Na maccuno khāditumussahanti,_  
_Taṃ me matī hoti carāmi dhammaṃ._  

</Paragraph>

</Division>
<Division number="380" book="ja1">

<Paragraph number="3245" type="verses" book="ja1">
_Māyākārā raṅgamajjhe karontā,_  
_Mohenti cakkhūni janassa tāvade;_  
_Na maccuno mohayitussahanti,_  
_Taṃ me matī hoti carāmi dhammaṃ._  

</Paragraph>

</Division>
<Division number="381" book="ja1">

<Paragraph number="3246" type="verses" book="ja1">
_Āsīvisā kupitā uggatejā,_  
_Ḍaṃsanti mārentipi te manusse;_  
_Na maccuno ḍaṃsitumussahanti,_  
_Taṃ me matī hoti carāmi dhammaṃ._  

</Paragraph>

</Division>
<Division number="382" book="ja1">

<Paragraph number="3247" type="verses" book="ja1">
_Āsīvisā kupitā yaṃ ḍaṃsanti,_  
_Tikicchakā tesa visaṃ hananti;_  
_Na maccuno daṭṭhavisaṃ hananti,_  
_Taṃ me matī hoti carāmi dhammaṃ._  

</Paragraph>

</Division>
<Division number="383" book="ja1">

<Paragraph number="3248" type="verses" book="ja1">
_Dhammantarī vettaraṇī ca bhojo,_  
_Visāni hantvāna bhujaṅgamānaṃ;_  
_Suyyanti te kālakatā tatheva,_  
_Taṃ me matī hoti carāmi dhammaṃ._  

</Paragraph>

</Division>
<Division number="384" book="ja1">

<Paragraph number="3249" type="verses" book="ja1">
_Vijjādharā ghoramadhīyamānā,_  
_Adassanaṃ osadhehi vajanti;_  
_Na maccurājassa vajantadassanaṃ,_  
_Taṃ me matī hoti carāmi dhammaṃ._  

</Paragraph>

</Division>
<Division number="385" book="ja1">

<Paragraph number="3250" type="verses" book="ja1">
_Dhammo have rakkhati dhammacāriṃ,_  
_Dhammo suciṇṇo sukhamāvahāti;_  
_Esānisaṃso dhamme suciṇṇe,_  
_Na duggatiṃ gacchati dhammacārī._  

</Paragraph>

</Division>
<Division number="386" book="ja1">

_Ubho samavipākino;_  
_Adhammo nirayaṃ neti,_  
_Dhammo pāpeti suggatin”ti. (2605)_  


</Paragraph>
