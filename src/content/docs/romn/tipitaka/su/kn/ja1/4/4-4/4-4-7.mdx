<Paragraph number="1069" type="verses" book="ja1">
_“Na te pīṭhamadāyimhā,_  
_na pānaṃ napi bhojanaṃ;_  
_Brahmacāri khamassu me,_  
_etaṃ passāmi accayaṃ”._  

</Paragraph>

<Division number="146" book="ja1">

<Paragraph number="1070" type="verses" book="ja1">
_“Nevābhisajjāmi na cāpi kuppe,_  
_Na cāpi me appiyamāsi kiñci;_  
_Athopi me āsi manovitakko,_  
_Etādiso nūna kulassa dhammo”._  

</Paragraph>

</Division>
<Division number="147" book="ja1">

<Paragraph number="1071" type="verses" book="ja1">
_“Esasmākaṃ kule dhammo,_  
_pitupitāmaho sadā;_  
_Āsanaṃ udakaṃ pajjaṃ,_  
_sabbetaṃ nipadāmase._  

</Paragraph>

</Division>
<Division number="148" book="ja1">

<Paragraph number="1072" type="verses" book="ja1">
_Esasmākaṃ kule dhammo,_  
_pitupitāmaho sadā;_  
_Sakkaccaṃ upatiṭṭhāma,_  
_uttamaṃ viya ñātakan”ti._  


</Paragraph>
</Division>

<Paragraph number="1073" type="center" book="ja1">
Pīṭhajātakaṃ sattamaṃ.
</Paragraph>
