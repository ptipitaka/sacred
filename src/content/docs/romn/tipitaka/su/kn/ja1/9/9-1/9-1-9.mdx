<Paragraph number="1866" type="verses" book="ja1">
_“Sutitikkhaṃ araññamhi,_  
_pantamhi sayanāsane;_  
_Ye ca gāme titikkhanti,_  
_te uḷāratarā tayā”._  

</Paragraph>

</Division>
<Division number="79" book="ja1" bookNo="22" page="368">

<Paragraph number="1867" type="verses" book="ja1">
_“Araññā gāmamāgamma,_  
_kiṃsīlaṃ kiṃvataṃ ahaṃ;_  
_Purisaṃ tāta seveyyaṃ,_  
_taṃ me akkhāhi pucchito”._  

</Paragraph>

</Division>
<Division number="80" book="ja1" bookNo="22" page="368">

<Paragraph number="1868" type="verses" book="ja1">
_“Yo te vissāsaye tāta,_  
_vissāsañca khameyya te;_  
_Sussūsī ca titikkhī ca,_  
_taṃ bhajehi ito gato._  

</Paragraph>

</Division>
<Division number="81" book="ja1" bookNo="22" page="368">

<Paragraph number="1869" type="verses" book="ja1">
_Yassa kāyena vācāya,_  
_manasā natthi dukkaṭaṃ;_  
_Urasīva patiṭṭhāya,_  
_taṃ bhajehi ito gato._  

</Paragraph>

</Division>
<Division number="82" book="ja1" bookNo="22" page="368">

<Paragraph number="1870" type="verses" book="ja1">
_Yo ca dhammena carati,_  
_carantopi na maññati;_  
_Visuddhakāriṃ sappaññaṃ,_  
_taṃ bhajehi ito gato._  

</Paragraph>

</Division>
<Division number="83" book="ja1" bookNo="22" page="368">

<Paragraph number="1871" type="verses" book="ja1">
_Haliddirāgaṃ kapicittaṃ,_  
_purisaṃ rāgavirāginaṃ;_  
_Tādisaṃ tāta mā sevi,_  
_nimmanussampi ce siyā._  

</Paragraph>

</Division>
<Division number="84" book="ja1" bookNo="22" page="368">

<Paragraph number="1872" type="verses" book="ja1">
_Āsīvisaṃva kupitaṃ,_  
_mīḷhalittaṃ mahāpathaṃ;_  
_Ārakā parivajjehi,_  
_yānīva visamaṃ pathaṃ._  

</Paragraph>

</Division>
<Division number="85" book="ja1" bookNo="22" page="369">

<Paragraph number="1873" type="verses" book="ja1">
_Anatthā tāta vaḍḍhanti,_  
_bālaṃ accupasevato;_  
_Māssu bālena saṃgacchi,_  
_amitteneva sabbadā._  

</Paragraph>

</Division>
<Division number="86" book="ja1" bookNo="22" page="369">

<Paragraph number="1874" type="verses" book="ja1">
_Taṃ tāhaṃ tāta yācāmi,_  
_karassu vacanaṃ mama;_  
_Māssu bālena saṃgacchi,_  
_dukkho bālehi saṅgamo”ti._  


</Paragraph>
</Division>

<Paragraph number="1875" type="center" book="ja1">
Haliddirāgajātakaṃ navamaṃ.
</Paragraph>
