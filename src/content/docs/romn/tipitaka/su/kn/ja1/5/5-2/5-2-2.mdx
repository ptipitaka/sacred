<Division number="65" book="ja1">

<Paragraph number="1223" type="verses" book="ja1">
_“Sīlaṃ seyyo sutaṃ seyyo,_  
_iti me saṃsayo ahu;_  
_Sīlameva sutā seyyo,_  
_iti me natthi saṃsayo._  

</Paragraph>

</Division>
<Division number="66" book="ja1">

<Paragraph number="1224" type="verses" book="ja1">
_Moghā jāti ca vaṇṇo ca,_  
_sīlameva kiruttamaṃ;_  
_Sīlena anupetassa,_  
_sutenattho na vijjati._  

</Paragraph>

</Division>
<Division number="67" book="ja1">

<Paragraph number="1225" type="verses" book="ja1">
_Khattiyo ca adhammaṭṭho,_  
_vesso cādhammanissito;_  
_Te pariccajjubho loke,_  
_upapajjanti duggatiṃ._  

</Paragraph>

</Division>
<Division number="68" book="ja1">

<Paragraph number="1226" type="verses" book="ja1">
_Khattiyā brāhmaṇā vessā,_  
_suddā caṇḍālapukkusā;_  
_Idha dhammaṃ caritvāna,_  
_bhavanti tidive samā._  

</Paragraph>

</Division>
<Division number="69" book="ja1">

<Paragraph number="1227" type="verses" book="ja1">
_Na vedā samparāyāya,_  
_Na jāti nāpi bandhavā;_  
_Sakañca sīlaṃ saṃsuddhaṃ,_  
_Samparāyāya sukhāya cā”ti._  


</Paragraph>
</Division>

<Paragraph number="1228" type="center" book="ja1">
Sīlavīmaṃsajātakaṃ dutiyaṃ.
</Paragraph>
