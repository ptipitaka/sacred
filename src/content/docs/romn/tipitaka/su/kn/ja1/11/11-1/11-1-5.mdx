<Division number="67" book="ja1">

<Paragraph number="2177" type="verses" book="ja1">
_“Dhiratthu subahū kāme,_  
_duggandhe bahukaṇṭake;_  
_Ye ahaṃ paṭisevanto,_  
_nālabhiṃ tādisaṃ sukhaṃ”._  

</Paragraph>

</Division>
<Division number="68" book="ja1">

<Paragraph number="2178" type="verses" book="ja1">
_“Mahassādā sukhā kāmā,_  
_Natthi kāmā paraṃ sukhaṃ;_  
_Ye kāme paṭisevanti,_  
_Saggaṃ te upapajjare”._  

</Paragraph>

</Division>
<Division number="69" book="ja1">

<Paragraph number="2179" type="verses" book="ja1">
_“Appassādā dukhā kāmā,_  
_natthi kāmā paraṃ dukhaṃ;_  
_Ye kāme paṭisevanti,_  
_nirayaṃ te upapajjare._  

</Paragraph>

</Division>
<Division number="70" book="ja1">

<Paragraph number="2180" type="verses" book="ja1">
_Asī yathā sunisito,_  
_nettiṃsova supāyiko;_  
_Sattīva urasi khittā,_  
_kāmā dukkhatarā tato._  

</Paragraph>

</Division>
<Division number="71" book="ja1">

<Paragraph number="2181" type="verses" book="ja1">
_Aṅgārānaṃva jalitaṃ,_  
_kāsuṃ sādhikaporisaṃ;_  
_Phālaṃva divasantattaṃ,_  
_kāmā dukkhatarā tato._  

</Paragraph>

</Division>

_Tambaloha vilīnaṃva,_  
_kāmā dukkhatarā tato”ti._  


</Paragraph>
