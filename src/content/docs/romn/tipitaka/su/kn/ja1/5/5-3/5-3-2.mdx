<Division number="116" book="ja1">

<Paragraph number="1284" type="verses" book="ja1">
_“Agārā paccupetassa,_  
_anagārassa te sato;_  
_Samaṇassa na taṃ sādhu,_  
_yaṃ petamanusocasi”._  

</Paragraph>

</Division>
<Division number="117" book="ja1">

<Paragraph number="1285" type="verses" book="ja1">
_“Saṃvāsena have sakka,_  
_manussassa migassa vā;_  
_Hadaye jāyate pemaṃ,_  
_na taṃ sakkā asocituṃ”._  

</Paragraph>

</Division>
<Division number="118" book="ja1">

<Paragraph number="1286" type="verses" book="ja1">
_“Mataṃ marissaṃ rodanti,_  
_ye rudanti lapanti ca;_  
_Tasmā tvaṃ isi mā rodi,_  
_roditaṃ moghamāhu santo._  

</Paragraph>

</Division>
<Division number="119" book="ja1">

<Paragraph number="1287" type="verses" book="ja1">
_Roditena have brahme,_  
_mato peto samuṭṭhahe;_  
_Sabbe saṅgamma rodāma,_  
_aññamaññassa ñātake”._  

</Paragraph>

</Division>
<Division number="120" book="ja1">

<Paragraph number="1288" type="verses" book="ja1">
_“Ādittaṃ vata maṃ santaṃ,_  
_ghatasittaṃva pāvakaṃ;_  
_Vārinā viya osiñcaṃ,_  
_sabbaṃ nibbāpaye daraṃ._  

</Paragraph>

</Division>
<Division number="121" book="ja1">

<Paragraph number="1289" type="verses" book="ja1">
_Abbahi vata me sallaṃ,_  
_yamāsi hadayassitaṃ;_  
_Yo me sokaparetassa,_  
_puttasokaṃ apānudi._  

</Paragraph>

</Division>
<Division number="122" book="ja1">

<Paragraph number="1290" type="verses" book="ja1">
_Sohaṃ abbūḷhasallosmi,_  
_vītasoko anāvilo;_  
_Na socāmi na rodāmi,_  
_tava sutvāna vāsavā”ti._  


</Paragraph>
</Division>

<Paragraph number="1291" type="center" book="ja1">
Migapotakajātakaṃ dutiyaṃ.
</Paragraph>
