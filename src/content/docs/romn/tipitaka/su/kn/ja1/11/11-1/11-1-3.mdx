<Paragraph number="2134" type="verses" book="ja1">
_“Yasokaro puññakarohamasmi,_  
_Sadātthuto samaṇabrāhmaṇānaṃ;_  
_Maggāraho devamanussapūjito,_  
_Dhammo ahaṃ dehi adhamma maggaṃ”._  

</Paragraph>

</Division>
<Division number="27" book="ja1" bookNo="22" page="74">

<Paragraph number="2135" type="verses" book="ja1">
_“Adhammayānaṃ daḷhamāruhitvā,_  
_Asantasanto balavāhamasmi;_  
_Sa kissa hetumhi tavajja dajjaṃ,_  
_Maggaṃ ahaṃ dhamma adinnapubbaṃ”._  

</Paragraph>

</Division>
<Division number="28" book="ja1" bookNo="22" page="74">

<Paragraph number="2136" type="verses" book="ja1">
_“Dhammo have pāturahosi pubbe,_  
_Pacchā adhammo udapādi loke;_  
_Jeṭṭho ca seṭṭho ca sanantano ca,_  
_Uyyāhi jeṭṭhassa kaniṭṭha maggā”._  

</Paragraph>

</Division>
<Division number="29" book="ja1" bookNo="22" page="74">

<Paragraph number="2137" type="verses" book="ja1">
_“Na yācanāya napi pātirūpā,_  
_Na arahatā tehaṃ dadeyyaṃ maggaṃ;_  
_Yuddhañca no hotu ubhinnamajja,_  
_Yuddhamhi yo jessati tassa maggo”._  

</Paragraph>

</Division>
<Division number="30" book="ja1" bookNo="22" page="75">

<Paragraph number="2138" type="verses" book="ja1">
_“Sabbā disā anuvisaṭohamasmi,_  
_Mahabbalo amitayaso atulyo;_  
_Guṇehi sabbehi upetarūpo,_  
_Dhammo adhamma tvaṃ kathaṃ vijessasi._  

</Paragraph>

</Division>
<Division number="31" book="ja1" bookNo="22" page="75">

<Paragraph number="2139" type="verses" book="ja1">
_Lohena ve haññati jātarūpaṃ,_  
_Na jātarūpena hananti lohaṃ;_  
_Sace adhammo hañchati dhammamajja,_  
_Ayo suvaṇṇaṃ viya dassaneyyaṃ._  

</Paragraph>

</Division>
<Division number="32" book="ja1" bookNo="22" page="75">

<Paragraph number="2140" type="verses" book="ja1">
_Sace tuvaṃ yuddhabalo adhamma,_  
_Na tuyha vuḍḍhā ca garū ca atthi;_  
_Maggañca te dammi piyāppiyena,_  
_Vācāduruttānipi te khamāmi”._  

</Paragraph>

</Division>
<Division number="33" book="ja1" bookNo="22" page="75">

<Paragraph number="2141" type="verses" book="ja1">
_“Idañca sutvā vacanaṃ adhammo,_  
_Avaṃsiro patito uddhapādo;_  
_‘Yuddhatthiko ce na labhāmi yuddhaṃ’,_  
_Ettāvatā hoti hato adhammo._  

</Paragraph>

</Division>
<Division number="34" book="ja1" bookNo="22" page="75">

<Paragraph number="2142" type="verses" book="ja1">
_Khantībalo yuddhabalaṃ vijetvā,_  
_Hantvā adhammaṃ nihanitva bhūmyā;_  
_Pāyāsi vitto abhiruyha sandanaṃ,_  
_Maggeneva atibalo saccanikkamo._  

</Paragraph>

</Division>
<Division number="35" book="ja1" bookNo="22" page="76">

<Paragraph number="2143" type="verses" book="ja1">
_Mātā pitā samaṇabrāhmaṇā ca,_  
_Asammānitā yassa sake agāre;_  
_Idheva nikkhippa sarīradehaṃ,_  
_Kāyassa bhedā nirayaṃ vajanti te;_  
_Yathā adhammo patito avaṃsiro._  

</Paragraph>

</Division>
<Division number="36" book="ja1" bookNo="22" page="76">

<Paragraph number="2144" type="verses" book="ja1">
_Mātā pitā samaṇabrāhmaṇā ca,_  
_Susammānitā yassa sake agāre;_  
_Idheva nikkhippa sarīradehaṃ,_  
_Kāyassa bhedā sugatiṃ vajanti te;_  
_Yathāpi dhammo abhiruyha sandanan”ti._  


</Paragraph>
</Division>

<Paragraph number="2145" type="center" book="ja1">
Dhammadevaputtajātakaṃ tatiyaṃ.
</Paragraph>
