<Division number="1" book="ja1">

<Paragraph number="2548" type="verses" book="ja1">
_“Sampannaṃ sālikedāraṃ,_  
_suvā bhuñjanti kosiya;_  
_Paṭivedemi te brahme,_  
_na ne vāretumussahe._  

</Paragraph>

</Division>
<Division number="2" book="ja1">

<Paragraph number="2549" type="verses" book="ja1">
_Eko ca tattha sakuṇo,_  
_yo nesaṃ sabbasundaro;_  
_Bhutvā sāliṃ yathākāmaṃ,_  
_tuṇḍenādāya gacchati”._  

</Paragraph>

</Division>
<Division number="3" book="ja1">

<Paragraph number="2550" type="verses" book="ja1">
_“Oḍḍentu vāḷapāsāni,_  
_yathā bajjhetha so dijo;_  
_Jīvañca naṃ gahetvāna,_  
_ānayehi mamantike”._  

</Paragraph>

</Division>
<Division number="4" book="ja1">

<Paragraph number="2551" type="verses" book="ja1">
_“Ete bhutvā pivitvā ca,_  
_pakkamanti vihaṅgamā;_  
_Eko baddhosmi pāsena,_  
_kiṃ pāpaṃ pakataṃ mayā”._  

</Paragraph>

</Division>
<Division number="5" book="ja1">

<Paragraph number="2552" type="verses" book="ja1">
_“Udaraṃ nūna aññesaṃ,_  
_suva accodaraṃ tava;_  
_Bhutvā sāliṃ yathākāmaṃ,_  
_tuṇḍenādāya gacchasi._  

</Paragraph>

</Division>
<Division number="6" book="ja1">

<Paragraph number="2553" type="verses" book="ja1">
_Koṭṭhaṃ nu tattha pūresi,_  
_suva veraṃ nu te mayā;_  
_Puṭṭho me samma akkhāhi,_  
_kuhiṃ sāliṃ nidāhasi”._  

</Paragraph>

</Division>
<Division number="7" book="ja1">

<Paragraph number="2554" type="verses" book="ja1">
_“Na me veraṃ tayā saddhiṃ,_  
_koṭṭho mayhaṃ na vijjati;_  
_Iṇaṃ muñcāmiṇaṃ dammi,_  
_sampatto koṭasimbaliṃ;_  
_Nidhimpi tattha nidahāmi,_  
_evaṃ jānāhi kosiya”._  

</Paragraph>

</Division>
<Division number="8" book="ja1">

<Paragraph number="2555" type="verses" book="ja1">
_“Kīdisaṃ te iṇadānaṃ,_  
_iṇamokkho ca kīdiso;_  
_Nidhinidhānamakkhāhi,_  
_atha pāsā pamokkhasi”._  

</Paragraph>

</Division>
<Division number="9" book="ja1">

<Paragraph number="2556" type="verses" book="ja1">
_“Ajātapakkhā taruṇā,_  
_puttakā mayha kosiya;_  
_Te maṃ bhatā bharissanti,_  
_tasmā tesaṃ iṇaṃ dade._  

</Paragraph>

</Division>
<Division number="10" book="ja1">

<Paragraph number="2557" type="verses" book="ja1">
_Mātā pitā ca me vuddhā,_  
_jiṇṇakā gatayobbanā;_  
_Tesaṃ tuṇḍena hātūna,_  
_muñce pubbakataṃ iṇaṃ._  

</Paragraph>

</Division>
<Division number="11" book="ja1">

<Paragraph number="2558" type="verses" book="ja1">
_Aññepi tattha sakuṇā,_  
_khīṇapakkhā sudubbalā;_  
_Tesaṃ puññatthiko dammi,_  
_taṃ nidhiṃ āhu paṇḍitā._  

</Paragraph>

</Division>
<Division number="12" book="ja1">

<Paragraph number="2559" type="verses" book="ja1">
_Īdisaṃ me iṇadānaṃ,_  
_iṇamokkho ca īdiso;_  
_Nidhinidhānamakkhāmi,_  
_evaṃ jānāhi kosiya”._  

</Paragraph>

</Division>
<Division number="13" book="ja1">

<Paragraph number="2560" type="verses" book="ja1">
_“Bhaddako vatayaṃ pakkhī,_  
_dijo paramadhammiko;_  
_Ekaccesu manussesu,_  
_ayaṃ dhammo na vijjati._  

</Paragraph>

</Division>
<Division number="14" book="ja1">

<Paragraph number="2561" type="verses" book="ja1">
_Bhuñja sāliṃ yathākāmaṃ,_  
_saha sabbehi ñātibhi;_  
_Punāpi suva passemu,_  
_piyaṃ me tava dassanaṃ”._  

</Paragraph>

</Division>
<Division number="15" book="ja1">

<Paragraph number="2562" type="verses" book="ja1">
_“Bhuttañca pītañca tavassamamhi,_  
_Ratī ca no kosiya te sakāse;_  
_Nikkhittadaṇḍesu dadāhi dānaṃ,_  
_Jiṇṇe ca mātāpitaro bharassu”._  

</Paragraph>

</Division>
<Division number="16" book="ja1">

<Paragraph number="2563" type="verses" book="ja1">
_“Lakkhī vata me udapādi ajja,_  
_Yo addasāsiṃ pavaraṃ dijānaṃ;_  
_Suvassa sutvāna subhāsitāni,_  
_Kāhāmi puññāni anappakāni”._  

</Paragraph>

</Division>
<Division number="17" book="ja1">

<Paragraph number="2564" type="verses" book="ja1">
_So kosiyo attamano udaggo,_  
_Annañca pānañcabhisaṅkharitvā;_  
_Annena pānena pasannacitto,_  
_Santappayi samaṇabrāhmaṇe cāti._  


</Paragraph>
</Division>

<Paragraph number="2565" type="center" book="ja1">
Sālikedārajātakaṃ paṭhamaṃ.
</Paragraph>
