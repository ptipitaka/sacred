<Paragraph number="1318" type="verses" book="ja1">
_“Māsu kujjha bhūmipati,_  
_māsu kujjha rathesabha;_  
_Kuddhaṃ appaṭikujjhanto,_  
_rājā raṭṭhassa pūjito._  

</Paragraph>

</Division>
<Division number="2" book="ja1" bookNo="22" page="257">

<Paragraph number="1319" type="verses" book="ja1">
_Gāme vā yadi vāraññe,_  
_ninne vā yadi vā thale;_  
_Sabbattha anusāsāmi,_  
_māsu kujjha rathesabha”._  

</Paragraph>

</Division>
<Division number="3" book="ja1" bookNo="22" page="276">

<Paragraph number="1320" type="verses" book="ja1">
_“Avāriyapitā nāma,_  
_ahu gaṅgāya nāviko;_  
_Pubbe janaṃ tāretvāna,_  
_pacchā yācati vetanaṃ;_  
_Tenassa bhaṇḍanaṃ hoti,_  
_na ca bhogehi vaḍḍhati”._  

</Paragraph>

</Division>
<Division number="4" book="ja1" bookNo="22" page="257">

<Paragraph number="1321" type="verses" book="ja1">
_“Atiṇṇaṃyeva yācassu,_  
_apāraṃ tāta nāvika;_  
_Añño hi tiṇṇassa mano,_  
_añño hoti pāresino._  

</Paragraph>

</Division>
<Division number="5" book="ja1" bookNo="22" page="257">

<Paragraph number="1322" type="verses" book="ja1">
_Gāme vā yadi vāraññe,_  
_ninne vā yadi vā thale;_  
_Sabbattha anusāsāmi,_  
_māsu kujjhittha nāvika”._  

</Paragraph>

</Division>
<Division number="6" book="ja1" bookNo="22" page="258">

<Paragraph number="1323" type="verses" book="ja1">
_“Yāyevānusāsaniyā,_  
_rājā gāmavaraṃ adā;_  
_Tāyevānusāsaniyā,_  
_nāviko paharī mukhaṃ._  

</Paragraph>

</Division>
<Division number="7" book="ja1" bookNo="22" page="258">

<Paragraph number="1324" type="verses" book="ja1">
_Bhattaṃ bhinnaṃ hatā bhariyā,_  
_Gabbho ca patito chamā;_  
_Migova jātarūpena,_  
_Na tenatthaṃ abandhi sū”ti._  


</Paragraph>
</Division>
