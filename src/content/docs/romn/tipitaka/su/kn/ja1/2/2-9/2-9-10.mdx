<Paragraph number="629" type="verses" book="ja1">
_“Sabbo jano hiṃsito piṅgalena,_  
_Tasmiṃ mate paccayā vedayanti;_  
_Piyo nu te āsi akaṇhanetto,_  
_Kasmā nu tvaṃ rodasi dvārapāla”._  

</Paragraph>

</Division>
<Division number="180" book="ja1" bookNo="22" page="329">

<Paragraph number="630" type="verses" book="ja1">
_“Na me piyo āsi akaṇhanetto,_  
_Bhāyāmi paccāgamanāya tassa;_  
_Ito gato hiṃseyya maccurājaṃ,_  
_So hiṃsito āneyya puna idha”._  

</Paragraph>

</Division>

<Paragraph number="631" type="verses" book="ja1">
_“Daḍḍho vāhasahassehi,_  
_sitto ghaṭasatehi so;_  
_Parikkhatā ca sā bhūmi,_  
_mā bhāyi nāgamissatī”ti._  


</Paragraph>
</Division>
