<Paragraph number="1536" type="verses" book="ja1">
_“Vibbhantacitto kupitindriyosi,_  
_Nettehi te vārigaṇā savanti;_  
_Kiṃ te naṭṭhaṃ kiṃ pana patthayāno,_  
_Idhāgamā brahme tadiṅgha brūhi”._  

</Paragraph>

</Division>
<Division number="47" book="ja1" bookNo="22" page="280">

<Paragraph number="1537" type="verses" book="ja1">
_“Miyyetha bhariyā vajato mamajja,_  
_Agacchato maraṇamāha yakkho;_  
_Etena dukkhena pavedhitosmi,_  
_Akkhāhi me senaka etamatthaṃ”._  

</Paragraph>

</Division>
<Division number="48" book="ja1" bookNo="22" page="281">

<Paragraph number="1538" type="verses" book="ja1">
_“Bahūni ṭhānāni vicintayitvā,_  
_Yamettha vakkhāmi tadeva saccaṃ;_  
_Maññāmi te brāhmaṇa sattubhastaṃ,_  
_Ajānato kaṇhasappo paviṭṭho._  

</Paragraph>

</Division>
<Division number="49" book="ja1" bookNo="22" page="281">

<Paragraph number="1539" type="verses" book="ja1">
_Ādāya daṇḍaṃ parisumbha bhastaṃ,_  
_Passeḷamūgaṃ uragaṃ dujivhaṃ;_  
_Chindajja kaṅkhaṃ vicikicchitāni,_  
_Bhujaṅgamaṃ passa pamuñca bhastaṃ”._  

</Paragraph>

</Division>
<Division number="50" book="ja1" bookNo="22" page="281">

<Paragraph number="1540" type="verses" book="ja1">
_Saṃviggarūpo parisāya majjhe,_  
_So brāhmaṇo sattubhastaṃ pamuñci;_  
_Atha nikkhami urago uggatejo,_  
_Āsīviso sappo phaṇaṃ karitvā._  

</Paragraph>

</Division>
<Division number="51" book="ja1" bookNo="22" page="281">

<Paragraph number="1541" type="verses" book="ja1">
_“Suladdhalābhā janakassa rañño,_  
_Yo passatī senakaṃ sādhupaññaṃ;_  
_Vivaṭṭachaddo nusi sabbadassī,_  
_Ñāṇaṃ nu te brāhmaṇa bhiṃsarūpaṃ”._  

</Paragraph>

</Division>
<Division number="52" book="ja1" bookNo="22" page="281">

<Paragraph number="1542" type="verses" book="ja1">
_“Imāni me sattasatāni atthi,_  
_Gaṇhāhi sabbāni dadāmi tuyhaṃ;_  
_Tayā hi me jīvitamajja laddhaṃ,_  
_Athopi bhariyāya makāsi sotthiṃ”._  

</Paragraph>

</Division>
<Division number="53" book="ja1" bookNo="22" page="281">

<Paragraph number="1543" type="verses" book="ja1">
_“Na paṇḍitā vetanamādiyanti,_  
_Citrāhi gāthāhi subhāsitāhi;_  
_Itopi te brahme dadantu vittaṃ,_  
_Ādāya tvaṃ gaccha sakaṃ niketan”ti._  


</Paragraph>
</Division>

<Paragraph number="1544" type="center" book="ja1">
Sattubhastajātakaṃ sattamaṃ.
</Paragraph>
