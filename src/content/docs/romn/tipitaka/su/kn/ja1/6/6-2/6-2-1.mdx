<Division number="77" book="ja1">

<Paragraph number="1406" type="verses" book="ja1">
_“Saccaṃ kirevamāhaṃsu,_  
_vastaṃ bāloti paṇḍitā;_  
_Passa bālo rahokammaṃ,_  
_āvikubbaṃ na bujjhati”._  

</Paragraph>

</Division>
<Division number="78" book="ja1">

<Paragraph number="1407" type="verses" book="ja1">
_“Tvaṃ khopi samma bālosi,_  
_kharaputta vijānahi;_  
_Rajjuyā hi parikkhitto,_  
_vaṅkoṭṭho ohitomukho._  

</Paragraph>

</Division>
<Division number="79" book="ja1">

<Paragraph number="1408" type="verses" book="ja1">
_Aparampi samma te bālyaṃ,_  
_yo mutto na palāyasi;_  
_So ca bālataro samma,_  
_yaṃ tvaṃ vahasi senakaṃ”._  

</Paragraph>

</Division>
<Division number="80" book="ja1">

<Paragraph number="1409" type="verses" book="ja1">
_“Yannu samma ahaṃ bālo,_  
_ajarāja vijānahi;_  
_Atha kena senako bālo,_  
_taṃ me akkhāhi pucchito”._  

</Paragraph>

</Division>
<Division number="81" book="ja1">

<Paragraph number="1410" type="verses" book="ja1">
_“Uttamatthaṃ labhitvāna,_  
_bhariyāya yo padassati;_  
_Tena jahissatattānaṃ,_  
_sā cevassa na hessati”._  

</Paragraph>

</Division>
<Division number="82" book="ja1">

<Paragraph number="1411" type="verses" book="ja1">
_“Na ve piyammeti janinda tādiso,_  
_Attaṃ niraṃkatvā piyāni sevati;_  
_Attāva seyyo paramā ca seyyo,_  
_Labbhā piyā ocitatthena pacchā”ti._  


</Paragraph>
</Division>

<Paragraph number="1412" type="center" book="ja1">
Kharaputtajātakaṃ paṭhamaṃ.
</Paragraph>
