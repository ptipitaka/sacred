<Paragraph number="1420" type="verses" book="ja1">
_“Navachannake dāni diyyati,_  
_Puṇṇāyaṃ doṇi suvāminī ṭhitā;_  
_Bahuke jane pāsapāṇike,_  
_No ca kho me paṭibhāti bhuñjituṃ”._  

</Paragraph>

</Division>
<Division number="90" book="ja1" bookNo="22" page="267">

<Paragraph number="1421" type="verses" book="ja1">
_“Tasasi bhamasi leṇamicchasi,_  
_Attāṇosi kuhiṃ gamissasi;_  
_Appossukko bhuñja tuṇḍila,_  
_Maṃsatthāya hi positāmhase._  

</Paragraph>

</Division>
<Division number="91" book="ja1" bookNo="22" page="267">

<Paragraph number="1422" type="verses" book="ja1">
_Ogaha rahadaṃ akaddamaṃ,_  
_Sabbaṃ sedamalaṃ pavāhaya;_  
_Gaṇhāhi navaṃ vilepanaṃ,_  
_Yassa gandho na kadāci chijjati”._  

</Paragraph>

</Division>
<Division number="92" book="ja1" bookNo="22" page="267">

<Paragraph number="1423" type="verses" book="ja1">
_“Katamo rahado akaddamo,_  
_Kiṃsu sedamalanti vuccati;_  
_Katamañca navaṃ vilepanaṃ,_  
_Yassa gandho na kadāci chijjati”._  

</Paragraph>

</Division>
<Division number="93" book="ja1" bookNo="22" page="267">

<Paragraph number="1424" type="verses" book="ja1">
_“Dhammo rahado akaddamo,_  
_Pāpaṃ sedamalanti vuccati;_  
_Sīlañca navaṃ vilepanaṃ,_  
_Tassa gandho na kadāci chijjati._  

</Paragraph>

</Division>
<Division number="94" book="ja1" bookNo="22" page="267">

<Paragraph number="1425" type="verses" book="ja1">
_Nandanti sarīraghātino,_  
_Na ca nandanti sarīradhārino;_  
_Puṇṇāya ca puṇṇamāsiyā,_  
_Ramamānāva jahanti jīvitan”ti._  


</Paragraph>
</Division>

<Paragraph number="1426" type="center" book="ja1">
Tuṇḍilajātakaṃ tatiyaṃ.
</Paragraph>
