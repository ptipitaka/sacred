<Paragraph number="1986" type="verses" book="ja1">
_“Na vāhametaṃ jānāmi,_  
_ko vāyaṃ kassa vāti vā;_  
_Yathā sākho cari evaṃ,_  
_nigrodha kinti maññasi._  

</Paragraph>

</Division>
<Division number="73" book="ja1" bookNo="22" page="48">

<Paragraph number="1987" type="verses" book="ja1">
_Tato galavinītena,_  
_purisā nīhariṃsu maṃ;_  
_Datvā mukhapahārāni,_  
_sākhassa vacanaṅkarā._  

</Paragraph>

</Division>
<Division number="74" book="ja1" bookNo="22" page="48">

<Paragraph number="1988" type="verses" book="ja1">
_Etādisaṃ dummatinā,_  
_akataññuna dubbhinā;_  
_Kataṃ anariyaṃ sākhena,_  
_sakhinā te janādhipa”._  

</Paragraph>

</Division>
<Division number="75" book="ja1" bookNo="22" page="49">

<Paragraph number="1989" type="verses" book="ja1">
_“Na vāhametaṃ jānāmi,_  
_napi me koci saṃsati;_  
_Yaṃ me tvaṃ samma akkhāsi,_  
_sākhena kāraṇaṃ kataṃ._  

</Paragraph>

</Division>
<Division number="76" book="ja1" bookNo="22" page="49">

<Paragraph number="1990" type="verses" book="ja1">
_Sakhīnaṃ sājīvakaro,_  
_mama sākhassa cūbhayaṃ;_  
_Tvaṃ nosissariyaṃ dātā,_  
_manussesu mahantataṃ;_  
_Tayāmā labbhitā iddhī,_  
_ettha me natthi saṃsayo._  

</Paragraph>

</Division>
<Division number="77" book="ja1" bookNo="22" page="49">

<Paragraph number="1991" type="verses" book="ja1">
_Yathāpi bījamaggimhi,_  
_ḍayhati na virūhati;_  
_Evaṃ kataṃ asappurise,_  
_nassati na virūhati._  

</Paragraph>

</Division>
<Division number="78" book="ja1" bookNo="22" page="49">

<Paragraph number="1992" type="verses" book="ja1">
_Kataññumhi ca posamhi,_  
_Sīlavante ariyavuttine;_  
_Sukhette viya bījāni,_  
_Kataṃ tamhi na nassati”._  

</Paragraph>

</Division>
<Division number="79" book="ja1" bookNo="22" page="49">

<Paragraph number="1993" type="verses" book="ja1">
_“Imaṃ jammaṃ nekatikaṃ,_  
_asappurisacintakaṃ;_  
_Hanantu sākhaṃ sattīhi,_  
_nāssa icchāmi jīvitaṃ”._  

</Paragraph>

</Division>
<Division number="80" book="ja1" bookNo="22" page="49">

<Paragraph number="1994" type="verses" book="ja1">
_“Khamatassa mahārāja,_  
_Pāṇā na paṭiānayā;_  
_Khama deva asappurisassa,_  
_Nāssa icchāmahaṃ vadhaṃ”._  

</Paragraph>

</Division>
<Division number="81" book="ja1" bookNo="22" page="50">

<Paragraph number="1995" type="verses" book="ja1">
_“Nigrodhameva seveyya,_  
_na sākhamupasaṃvase;_  
_Nigrodhasmiṃ mataṃ seyyo,_  
_yañce sākhasmi jīvitan”ti._  


</Paragraph>
</Division>

<Paragraph number="1996" type="center" book="ja1">
Nigrodhajātakaṃ sattamaṃ.
</Paragraph>
