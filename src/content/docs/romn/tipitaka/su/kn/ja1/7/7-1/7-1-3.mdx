<Division number="15" book="ja1">

<Paragraph number="1501" type="verses" book="ja1">
_“Rājā te bhattaṃ pāhesi,_  
_Suciṃ maṃsūpasecanaṃ;_  
_Maghadevasmiṃ adhivatthe,_  
_Ehi nikkhamma bhuñjasu”._  

</Paragraph>

</Division>
<Division number="16" book="ja1">

<Paragraph number="1502" type="verses" book="ja1">
_“Ehi māṇava orena,_  
_bhikkhamādāya sūpitaṃ;_  
_Tvañca māṇava bhikkhā ca,_  
_ubho bhakkhā bhavissatha”._  

</Paragraph>

</Division>
<Division number="17" book="ja1">

<Paragraph number="1503" type="verses" book="ja1">
_“Appakena tuvaṃ yakkha,_  
_thullamatthaṃ jahissasi;_  
_Bhikkhaṃ te nāharissanti,_  
_janā maraṇasaññino._  

</Paragraph>

</Division>
<Division number="18" book="ja1">

<Paragraph number="1504" type="verses" book="ja1">
_Laddhāya yakkhā tava niccabhikkhaṃ,_  
_Suciṃ paṇītaṃ rasasā upetaṃ;_  
_Bhikkhañca te āhariyo naro idha,_  
_Sudullabho hehiti bhakkhite mayi”._  

</Paragraph>

</Division>
<Division number="19" book="ja1">

<Paragraph number="1505" type="verses" book="ja1">
_“Mameva sutano attho,_  
_yathā bhāsasi māṇava;_  
_Mayā tvaṃ samanuññāto,_  
_sotthiṃ passāhi mātaraṃ._  

</Paragraph>

</Division>
<Division number="20" book="ja1">

<Paragraph number="1506" type="verses" book="ja1">
_Khaggaṃ chattañca pātiñca,_  
_gacchamādāya māṇava;_  
_Sotthiṃ passatu te mātā,_  
_tvañca passāhi mātaraṃ”._  

</Paragraph>

</Division>
<Division number="21" book="ja1">

<Paragraph number="1507" type="verses" book="ja1">
_“Evaṃ yakkha sukhī hohi,_  
_saha sabbehi ñātibhi;_  
_Dhanañca me adhigataṃ,_  
_rañño ca vacanaṃ katan”ti._  


</Paragraph>
</Division>

<Paragraph number="1508" type="center" book="ja1">
Sutanujātakaṃ tatiyaṃ.
</Paragraph>
