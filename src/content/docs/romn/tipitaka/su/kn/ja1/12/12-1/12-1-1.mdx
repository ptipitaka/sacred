<Paragraph number="2239" type="verses" book="ja1">
_“Luddhānaṃ lahucittānaṃ,_  
_akataññūna dubbhinaṃ;_  
_Nādevasatto puriso,_  
_thīnaṃ saddhātumarahati._  

</Paragraph>

</Division>
<Division number="2" book="ja1" bookNo="22" page="93">

<Paragraph number="2240" type="verses" book="ja1">
_Na tā pajānanti kataṃ na kiccaṃ,_  
_Na mātaraṃ pitaraṃ bhātaraṃ vā;_  
_Anariyā samatikkantadhammā,_  
_Sasseva cittassa vasaṃ vajanti._  

</Paragraph>

</Division>
<Division number="3" book="ja1" bookNo="22" page="120">

<Paragraph number="2241" type="verses" book="ja1">
_Cirānuvutthampi piyaṃ manāpaṃ,_  
_Anukampakaṃ pāṇasamampi bhattuṃ;_  
_Āvāsu kiccesu ca naṃ jahanti,_  
_Tasmāhamitthīnaṃ na vissasāmi._  

</Paragraph>

</Division>
<Division number="4" book="ja1" bookNo="22" page="93">

<Paragraph number="2242" type="verses" book="ja1">
_Thīnañhi cittaṃ yathā vānarassa,_  
_Kannappakannaṃ yathā rukkhachāyā;_  
_Calācalaṃ hadayamitthiyānaṃ,_  
_Cakkassa nemi viya parivattati._  

</Paragraph>

</Division>
<Division number="5" book="ja1" bookNo="22" page="93">

<Paragraph number="2243" type="verses" book="ja1">
_Yadā tā passanti samekkhamānā,_  
_Ādeyyarūpaṃ purisassa vittaṃ;_  
_Saṇhāhi vācāhi nayanti menaṃ,_  
_Kambojakā jalajeneva assaṃ._  

</Paragraph>

</Division>
<Division number="6" book="ja1" bookNo="22" page="94">

<Paragraph number="2244" type="verses" book="ja1">
_Yadā na passanti samekkhamānā,_  
_Ādeyyarūpaṃ purisassa vittaṃ;_  
_Samantato naṃ parivajjayanti,_  
_Tiṇṇo nadīpāragatova kullaṃ._  

</Paragraph>

</Division>
<Division number="7" book="ja1" bookNo="22" page="94">

<Paragraph number="2245" type="verses" book="ja1">
_Silesūpamā sikhiriva sabbabhakkhā,_  
_Tikkhamāyā nadīriva sīghasotā;_  
_Sevanti hetā piyamappiyañca,_  
_Nāvā yathā orakūlaṃ parañca._  

</Paragraph>

</Division>
<Division number="8" book="ja1" bookNo="22" page="94">

<Paragraph number="2246" type="verses" book="ja1">
_Na tā ekassa na dvinnaṃ,_  
_āpaṇova pasārito;_  
_Yo tā mayhanti maññeyya,_  
_vātaṃ jālena bādhaye._  

</Paragraph>

</Division>
<Division number="9" book="ja1" bookNo="22" page="94">

<Paragraph number="2247" type="verses" book="ja1">
_Yathā nadī ca pantho ca,_  
_pānāgāraṃ sabhā papā;_  
_Evaṃ lokitthiyo nāma,_  
_velā tāsaṃ na vijjati._  

</Paragraph>

</Division>
<Division number="10" book="ja1" bookNo="22" page="94">

<Paragraph number="2248" type="verses" book="ja1">
_Ghatāsanasamā etā,_  
_kaṇhasappasirūpamā;_  
_Gāvo bahitiṇasseva,_  
_omasanti varaṃ varaṃ._  

</Paragraph>

</Division>
<Division number="11" book="ja1" bookNo="22" page="94">

<Paragraph number="2249" type="verses" book="ja1">
_Ghatāsanaṃ kuñjaraṃ kaṇhasappaṃ,_  
_Muddhābhisittaṃ pamadā ca sabbā;_  
_Ete naro niccayato bhajetha,_  
_Tesaṃ have dubbidu sabbabhāvo._  

</Paragraph>

</Division>
<Division number="12" book="ja1" bookNo="22" page="94">

<Paragraph number="2250" type="verses" book="ja1">
_Naccantavaṇṇā na bahūnaṃ kantā,_  
_Na dakkhiṇā pamadā sevitabbā;_  
_Na parassa bhariyā na dhanassa hetu,_  
_Etitthiyo pañca na sevitabbā”ti._  


</Paragraph>
</Division>

<Paragraph number="2251" type="center" book="ja1">
Cūḷakuṇālajātakaṃ paṭhamaṃ.
</Paragraph>
