<Division number="78" book="ja1">

<Paragraph number="2629" type="verses" book="ja1">
_“Assaṃ gavaṃ rajataṃ jātarūpaṃ,_  
_Bhariyañca so idha labhataṃ manāpaṃ;_  
_Puttehi dārehi samaṅgi hotu,_  
_Bhisāni te brāhmaṇa yo ahāsi”._  

</Paragraph>

</Division>
<Division number="79" book="ja1">

<Paragraph number="2630" type="verses" book="ja1">
_“Mālañca so kāsikacandanañca,_  
_Dhāretu puttassa bahū bhavantu;_  
_Kāmesu tibbaṃ kurutaṃ apekkhaṃ,_  
_Bhisāni te brāhmaṇa yo ahāsi”._  

</Paragraph>

</Division>
<Division number="80" book="ja1">

<Paragraph number="2631" type="verses" book="ja1">
_“Pahūtadhañño kasimā yasassī,_  
_Putte gihī dhanimā sabbakāme;_  
_Vayaṃ apassaṃ gharamāvasātu,_  
_Bhisāni te brāhmaṇa yo ahāsi._  

</Paragraph>

</Division>
<Division number="81" book="ja1">

<Paragraph number="2632" type="verses" book="ja1">
_So khattiyo hotu pasayhakārī,_  
_Rājābhirājā balavā yasassī;_  
_Sa cāturantaṃ mahimāvasātu,_  
_Bhisāni te brāhmaṇa yo ahāsi._  

</Paragraph>

</Division>
<Division number="82" book="ja1">

<Paragraph number="2633" type="verses" book="ja1">
_So brāhmaṇo hotu avītarāgo,_  
_Muhuttanakkhattapathesu yutto;_  
_Pūjetu naṃ raṭṭhapatī yasassī,_  
_Bhisāni te brāhmaṇa yo ahāsi._  

</Paragraph>

</Division>
<Division number="83" book="ja1">

<Paragraph number="2634" type="verses" book="ja1">
_Ajjhāyakaṃ sabbasamantavedaṃ,_  
_Tapassīnaṃ maññatu sabbaloko;_  
_Pūjentu naṃ jānapadā samecca,_  
_Bhisāni te brāhmaṇa yo ahāsi._  

</Paragraph>

</Division>
<Division number="84" book="ja1">

<Paragraph number="2635" type="verses" book="ja1">
_Catussadaṃ gāmavaraṃ samiddhaṃ,_  
_Dinnañhi so bhuñjatu vāsavena;_  
_Avītarāgo maraṇaṃ upetu,_  
_Bhisāni te brāhmaṇa yo ahāsi._  

</Paragraph>

</Division>
<Division number="85" book="ja1">

<Paragraph number="2636" type="verses" book="ja1">
_So gāmaṇī hotu sahāyamajjhe,_  
_Naccehi gītehi pamodamāno;_  
_So rājato byasana mālattha kiñci,_  
_Bhisāni te brāhmaṇa yo ahāsi._  

</Paragraph>

</Division>
<Division number="86" book="ja1">

<Paragraph number="2637" type="verses" book="ja1">
_Yaṃ ekarājā pathaviṃ vijetvā,_  
_Itthīsahassāna ṭhapetu aggaṃ;_  
_Sīmantinīnaṃ pavarā bhavātu,_  
_Bhisāni te brāhmaṇa yā ahāsi._  

</Paragraph>

</Division>
<Division number="87" book="ja1">

<Paragraph number="2638" type="verses" book="ja1">
_Isīnañhi sā sabbasamāgatānaṃ,_  
_Bhuñjeyya sāduṃ avikampamānā;_  
_Carātu lābhena vikatthamānā,_  
_Bhisāni te brāhmaṇa yā ahāsi._  

</Paragraph>

</Division>
<Division number="88" book="ja1">

<Paragraph number="2639" type="verses" book="ja1">
_Āvāsiko hotu mahāvihāre,_  
_Navakammiko hotu gajaṅgalāyaṃ;_  
_Ālokasandhiṃ divasaṃ karotu,_  
_Bhisāni te brāhmaṇa yo ahāsi._  

</Paragraph>

</Division>
<Division number="89" book="ja1">

<Paragraph number="2640" type="verses" book="ja1">
_So bajjhatū pāsasatehi chabbhi,_  
_Rammā vanā niyyatu rājadhāniṃ;_  
_Tuttehi so haññatu pācanehi,_  
_Bhisāni te brāhmaṇa yo ahāsi._  

</Paragraph>

</Division>
<Division number="90" book="ja1">

<Paragraph number="2641" type="verses" book="ja1">
_Alakkamālī tipukaṇṇaviddho,_  
_Laṭṭhīhato sappamukhaṃ upetu;_  
_Sakacchabandho visikhaṃ carātu,_  
_Bhisāni te brāhmaṇa yo ahāsi”._  

</Paragraph>

</Division>
<Division number="91" book="ja1">

<Paragraph number="2642" type="verses" book="ja1">
_“Yo ve anaṭṭhaṃva naṭṭhanti cāha,_  
_Kāmeva so labhataṃ bhuñjatañca;_  
_Agāramajjhe maraṇaṃ upetu,_  
_Yo vā bhonto saṅkati kañcideva”._  

</Paragraph>

</Division>
<Division number="92" book="ja1">

<Paragraph number="2643" type="verses" book="ja1">
_“Yadesamānā vicaranti loke,_  
_Iṭṭhañca kantañca bahūnametaṃ;_  
_Piyaṃ manuññaṃ cidha jīvaloke,_  
_Kasmā isayo nappasaṃsanti kāme”._  

</Paragraph>

</Division>
<Division number="93" book="ja1">

<Paragraph number="2644" type="verses" book="ja1">
_“Kāmesu ve haññare bajjhare ca,_  
_Kāmesu dukkhañca bhayañca jātaṃ;_  
_Kāmesu bhūtādhipatī pamattā,_  
_Pāpāni kammāni karonti mohā._  

</Paragraph>

</Division>
<Division number="94" book="ja1">

<Paragraph number="2645" type="verses" book="ja1">
_Te pāpadhammā pasavetva pāpaṃ,_  
_Kāyassa bhedā nirayaṃ vajanti;_  
_Ādīnavaṃ kāmaguṇesu disvā,_  
_Tasmā isayo nappasaṃsanti kāme”._  

</Paragraph>

</Division>
<Division number="95" book="ja1">

<Paragraph number="2646" type="verses" book="ja1">
_“Vīmaṃsamāno isino bhisāni,_  
_Tīre gahetvāna thale nidhesiṃ;_  
_Suddhā apāpā isayo vasanti,_  
_Etāni te brahmacārī bhisāni”._  

</Paragraph>

</Division>
<Division number="96" book="ja1">

<Paragraph number="2647" type="verses" book="ja1">
_“Na te naṭā no pana kīḷaneyyā,_  
_Na bandhavā no pana te sahāyā;_  
_Kismiṃ vupatthambha sahassanetta,_  
_Isīhi tvaṃ kīḷasi devarāja”._  

</Paragraph>

</Division>
<Division number="97" book="ja1">

<Paragraph number="2648" type="verses" book="ja1">
_“Ācariyo mesi pitā ca mayhaṃ,_  
_Esā patiṭṭhā khalitassa brahme;_  
_Ekāparādhaṃ khama bhūripañña,_  
_Na paṇḍitā kodhabalā bhavanti”._  

</Paragraph>

</Division>
<Division number="98" book="ja1">

<Paragraph number="2649" type="verses" book="ja1">
_“Suvāsitaṃ isinaṃ ekarattaṃ,_  
_Yaṃ vāsavaṃ bhūtapatiddasāma;_  
_Sabbeva bhonto sumanā bhavantu,_  
_Yaṃ brāhmaṇo paccupādī bhisāni”._  

</Paragraph>

</Division>
<Division number="99" book="ja1">

<Paragraph number="2650" type="verses" book="ja1">
_“Ahañca sāriputto ca,_  
_moggallāno ca kassapo;_  
_Anuruddho puṇṇo ānando,_  
_tadāsuṃ satta bhātaro._  

</Paragraph>

</Division>
<Division number="100" book="ja1">

<Paragraph number="2651" type="verses" book="ja1">
_Bhaginī uppalavaṇṇā ca,_  
_dāsī khujjuttarā tadā;_  
_Citto gahapati dāso,_  
_yakkho sātāgiro tadā._  

</Paragraph>

</Division>
<Division number="101" book="ja1">

<Paragraph number="2652" type="verses" book="ja1">
_Pālileyyo tadā nāgo,_  
_madhudo seṭṭhavānaro;_  
_Kāḷudāyī tadā sakko,_  
_evaṃ dhāretha jātakanti”._  


</Paragraph>
</Division>

<Paragraph number="2653" type="center" book="ja1">
Bhisajātakaṃ pañcamaṃ.
</Paragraph>
