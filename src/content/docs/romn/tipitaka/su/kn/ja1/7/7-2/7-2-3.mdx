<Paragraph number="1588" type="verses" book="ja1">
_“Ambāhamaddaṃ vanamantarasmiṃ,_  
_Nīlobhāsaṃ phalitaṃ saṃvirūḷhaṃ;_  
_Tamaddasaṃ phalahetu vibhaggaṃ,_  
_Taṃ disvā bhikkhācariyaṃ carāmi._  

</Paragraph>

</Division>
<Division number="91" book="ja1" bookNo="22" page="287">

<Paragraph number="1589" type="verses" book="ja1">
_Selaṃ sumaṭṭhaṃ naravīraniṭṭhitaṃ,_  
_Nārī yugaṃ dhārayi appasaddaṃ;_  
_Dutiyañca āgamma ahosi saddo,_  
_Taṃ disvā bhikkhācariyaṃ carāmi._  

</Paragraph>

</Division>
<Division number="92" book="ja1" bookNo="22" page="287">

<Paragraph number="1590" type="verses" book="ja1">
_Dijā dijaṃ kuṇapamāharantaṃ,_  
_Ekaṃ samānaṃ bahukā samecca;_  
_Āhārahetū paripātayiṃsu,_  
_Taṃ disvā bhikkhācariyaṃ carāmi._  

</Paragraph>

</Division>
<Division number="93" book="ja1" bookNo="22" page="287">

<Paragraph number="1591" type="verses" book="ja1">
_Usabhāhamaddaṃ yūthassa majjhe,_  
_Calakkakuṃ vaṇṇabalūpapannaṃ;_  
_Tamaddasaṃ kāmahetu vitunnaṃ,_  
_Taṃ disvā bhikkhācariyaṃ carāmi”._  

</Paragraph>

</Division>
<Division number="94" book="ja1" bookNo="22" page="287">

<Paragraph number="1592" type="verses" book="ja1">
_“Karaṇḍako kaliṅgānaṃ,_  
_gandhārānañca naggaji;_  
_Nimirājā videhānaṃ,_  
_pañcālānañca dummukho;_  
_Ete raṭṭhāni hitvāna,_  
_pabbajiṃsu akiñcanā._  

</Paragraph>

</Division>
<Division number="95" book="ja1" bookNo="22" page="287">

<Paragraph number="1593" type="verses" book="ja1">
_Sabbepime devasamā samāgatā,_  
_Aggī yathā pajjalito tathevime;_  
_Ahampi eko carissāmi bhaggavi,_  
_Hitvāna kāmāni yathodhikāni”._  

</Paragraph>

</Division>
<Division number="96" book="ja1" bookNo="22" page="322">

<Paragraph number="1594" type="verses" book="ja1">
_“Ayameva kālo na hi añño atthi,_  
_Anusāsitā me na bhaveyya pacchā;_  
_Ahampi ekā carissāmi bhaggava,_  
_Sakuṇīva muttā purisassa hatthā”._  

</Paragraph>

</Division>
<Division number="97" book="ja1" bookNo="22" page="322">

<Paragraph number="1595" type="verses" book="ja1">
_“Āmaṃ pakkañca jānanti,_  
_atho loṇaṃ aloṇakaṃ;_  
_Tamahaṃ disvāna pabbajiṃ,_  
_careva tvaṃ carāmahan”ti._  


</Paragraph>
</Division>

<Paragraph number="1596" type="center" book="ja1">
Kumbhakārajātakaṃ tatiyaṃ.
</Paragraph>
