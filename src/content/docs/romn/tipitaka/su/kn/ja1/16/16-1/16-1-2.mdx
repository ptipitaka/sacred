<Paragraph number="3289" type="verses" book="ja1">
_“Ko pāturāsī tidivā nabhamhi,_  
_Obhāsayaṃ saṃvariṃ candimāva;_  
_Gattehi te rasmiyo niccharanti,_  
_Sateratā vijjurivantalikkhe._  

</Paragraph>

</Division>
<Division number="34" book="ja1" bookNo="22" page="179">

<Paragraph number="3290" type="verses" book="ja1">
_So chinnavātaṃ kamasī aghamhi,_  
_Vehāyasaṃ gacchasi tiṭṭhasī ca;_  
_Iddhī nu te vatthukatā subhāvitā,_  
_Anaddhagūnaṃ api devatānaṃ._  

</Paragraph>

</Division>
<Division number="35" book="ja1" bookNo="22" page="179">

<Paragraph number="3291" type="verses" book="ja1">
_Vehāyasaṃ gammamāgamma tiṭṭhasi,_  
_Kumbhaṃ kiṇāthāti yametamatthaṃ;_  
_Ko vā tuvaṃ kissa vā tāya kumbho,_  
_Akkhāhi me brāhmaṇa etamatthaṃ”._  

</Paragraph>

</Division>
<Division number="36" book="ja1" bookNo="22" page="179">

<Paragraph number="3292" type="verses" book="ja1">
_“Na sappikumbho napi telakumbho,_  
_Na phāṇitassa na madhussa kumbho;_  
_Kumbhassa vajjāni anappakāni,_  
_Dose bahū kumbhagate suṇātha._  

</Paragraph>

</Division>
<Division number="37" book="ja1" bookNo="22" page="179">

<Paragraph number="3293" type="verses" book="ja1">
_Gaḷeyya yaṃ pitvā pate papātaṃ,_  
_Sobbhaṃ guhaṃ candaniyoḷigallaṃ;_  
_Bahumpi bhuñjeyya abhojaneyyaṃ,_  
_Tassā puṇṇaṃ kumbhamimaṃ kiṇātha._  

</Paragraph>

</Division>
<Division number="38" book="ja1" bookNo="22" page="180">

<Paragraph number="3294" type="verses" book="ja1">
_Yaṃ pitva cittasmimanesamāno,_  
_Āhiṇḍatī goriva bhakkhasādī;_  
_Anāthamāno upagāyati naccati ca,_  
_Tassā puṇṇaṃ kumbhamimaṃ kiṇātha._  

</Paragraph>

</Division>
<Division number="39" book="ja1" bookNo="22" page="180">

<Paragraph number="3295" type="verses" book="ja1">
_Yaṃ ve pivitvā acelova naggo,_  
_Careyya gāme visikhantarāni;_  
_Sammūḷhacitto ativelasāyī,_  
_Tassā puṇṇaṃ kumbhamimaṃ kiṇātha._  

</Paragraph>

</Division>
<Division number="40" book="ja1" bookNo="22" page="180">

<Paragraph number="3296" type="verses" book="ja1">
_Yaṃ pitva uṭṭhāya pavedhamāno,_  
_Sīsañca bāhuñca pacālayanto;_  
_So naccatī dārukaṭallakova,_  
_Tassā puṇṇaṃ kumbhamimaṃ kiṇātha._  

</Paragraph>

</Division>
<Division number="41" book="ja1" bookNo="22" page="180">

<Paragraph number="3297" type="verses" book="ja1">
_Yaṃ ve pivitvā aggidaḍḍhā sayanti,_  
_Atho sigālehipi khāditāse;_  
_Bandhaṃ vadhaṃ bhogajāniñcupenti,_  
_Tassā puṇṇaṃ kumbhamimaṃ kiṇātha._  

</Paragraph>

</Division>
<Division number="42" book="ja1" bookNo="22" page="180">

<Paragraph number="3298" type="verses" book="ja1">
_Yaṃ pitva bhāseyya abhāsaneyyaṃ,_  
_Sabhāyamāsīno apetavattho;_  
_Sammakkhito vantagato byasanno,_  
_Tassā puṇṇaṃ kumbhamimaṃ kiṇātha._  

</Paragraph>

</Division>
<Division number="43" book="ja1" bookNo="22" page="180">

<Paragraph number="3299" type="verses" book="ja1">
_Yaṃ ve pivitvā ukkaṭṭho āvilakkho,_  
_Mameva sabbā pathavīti maññe;_  
_Na me samo cāturantopi rājā,_  
_Tassā puṇṇaṃ kumbhamimaṃ kiṇātha._  

</Paragraph>

</Division>
<Division number="44" book="ja1" bookNo="22" page="180">

<Paragraph number="3300" type="verses" book="ja1">
_Mānātimānā kalahāni pesuṇī,_  
_Dubbaṇṇinī naggayinī palāyinī;_  
_Corāna dhuttāna gatī niketo,_  
_Tassā puṇṇaṃ kumbhamimaṃ kiṇātha._  

</Paragraph>

</Division>
<Division number="45" book="ja1" bookNo="22" page="180">

<Paragraph number="3301" type="verses" book="ja1">
_Iddhāni phītāni kulāni assu,_  
_Anekasāhassadhanāni loke;_  
_Ucchinnadāyajjakatānimāya,_  
_Tassā puṇṇaṃ kumbhamimaṃ kiṇātha._  

</Paragraph>

</Division>
<Division number="46" book="ja1" bookNo="22" page="181">

<Paragraph number="3302" type="verses" book="ja1">
_Dhaññaṃ dhanaṃ rajataṃ jātarūpaṃ,_  
_Khettaṃ gavaṃ yattha vināsayanti;_  
_Ucchedanī vittavataṃ kulānaṃ,_  
_Tassā puṇṇaṃ kumbhamimaṃ kiṇātha._  

</Paragraph>

</Division>
<Division number="47" book="ja1" bookNo="22" page="181">

<Paragraph number="3303" type="verses" book="ja1">
_Yaṃ ve pitvā dittarūpova poso,_  
_Akkosati mātaraṃ pitarañca;_  
_Sassumpi gaṇheyya athopi suṇhaṃ,_  
_Tassā puṇṇaṃ kumbhamimaṃ kiṇātha._  

</Paragraph>

</Division>
<Division number="48" book="ja1" bookNo="22" page="181">

<Paragraph number="3304" type="verses" book="ja1">
_Yaṃ ve pitvā dittarūpāva nārī,_  
_Akkosatī sasuraṃ sāmikañca;_  
_Dāsampi gaṇhe paricārakampi,_  
_Tassā puṇṇaṃ kumbhamimaṃ kiṇātha._  

</Paragraph>

</Division>
<Division number="49" book="ja1" bookNo="22" page="181">

<Paragraph number="3305" type="verses" book="ja1">
_Yaṃ ve pivitvāna haneyya poso,_  
_Dhamme ṭhitaṃ samaṇaṃ brāhmaṇaṃ vā;_  
_Gacche apāyampi tatonidānaṃ,_  
_Tassā puṇṇaṃ kumbhamimaṃ kiṇātha._  

</Paragraph>

</Division>
<Division number="50" book="ja1" bookNo="22" page="181">

<Paragraph number="3306" type="verses" book="ja1">
_Yaṃ ve pivitvā duccaritaṃ caranti,_  
_Kāyena vācāya ca cetasā ca;_  
_Nirayaṃ vajanti duccaritaṃ caritvā,_  
_Tassā puṇṇaṃ kumbhamimaṃ kiṇātha._  

</Paragraph>

</Division>
<Division number="51" book="ja1" bookNo="22" page="181">

<Paragraph number="3307" type="verses" book="ja1">
_Yaṃ yācamānā na labhanti pubbe,_  
_Bahuṃ hiraññampi pariccajantā;_  
_So taṃ pivitvā alikaṃ bhaṇāti,_  
_Tassā puṇṇaṃ kumbhamimaṃ kiṇātha._  

</Paragraph>

</Division>
<Division number="52" book="ja1" bookNo="22" page="181">

<Paragraph number="3308" type="verses" book="ja1">
_Yaṃ ve pitvā pesane pesiyanto,_  
_Accāyike karaṇīyamhi jāte;_  
_Atthampi so nappajānāti vutto,_  
_Tassā puṇṇaṃ kumbhamimaṃ kiṇātha._  

</Paragraph>

</Division>
<Division number="53" book="ja1" bookNo="22" page="181">

<Paragraph number="3309" type="verses" book="ja1">
_Hirīmanāpi ahirīkabhāvaṃ,_  
_Pātuṃ karonti madanāya mattā;_  
_Dhīrāpi santā bahukaṃ bhaṇanti,_  
_Tassā puṇṇaṃ kumbhamimaṃ kiṇātha._  

</Paragraph>

</Division>
<Division number="54" book="ja1" bookNo="22" page="181">

<Paragraph number="3310" type="verses" book="ja1">
_Yaṃ ve pitvā ekathūpā sayanti,_  
_Anāsakā thaṇḍiladukkhaseyyaṃ;_  
_Dubbaṇṇiyaṃ āyasakyañcupenti,_  
_Tassā puṇṇaṃ kumbhamimaṃ kiṇātha._  

</Paragraph>

</Division>
<Division number="55" book="ja1" bookNo="22" page="181">

<Paragraph number="3311" type="verses" book="ja1">
_Yaṃ ve pitvā pattakhandhā sayanti,_  
_Gāvo kuṭahatāva na hi vāruṇiyā;_  
_Vego narena susahoriva,_  
_Tassā puṇṇaṃ kumbhamimaṃ kiṇātha._  

</Paragraph>

</Division>
<Division number="56" book="ja1" bookNo="22" page="181">

<Paragraph number="3312" type="verses" book="ja1">
_Yaṃ manussā vivajjenti,_  
_sappaṃ ghoravisamiva;_  
_Taṃ loke visasamānaṃ,_  
_ko naro pātumarahati._  

</Paragraph>

</Division>
<Division number="57" book="ja1" bookNo="22" page="162">

<Paragraph number="3313" type="verses" book="ja1">
_Yaṃ ve pitvā andhakaveṇḍaputtā,_  
_Samuddatīre paricārayantā;_  
_Upakkamuṃ musalehaññamaññaṃ,_  
_Tassā puṇṇaṃ kumbhamimaṃ kiṇātha._  

</Paragraph>

</Division>
<Division number="58" book="ja1" bookNo="22" page="193">

<Paragraph number="3314" type="verses" book="ja1">
_Yaṃ ve pitvā pubbadevā pamattā,_  
_Tidivā cutā sassatiyā samāya;_  
_Taṃ tādisaṃ majjamimaṃ niratthakaṃ,_  
_Jānaṃ mahārāja kathaṃ piveyya._  

</Paragraph>

</Division>
<Division number="59" book="ja1" bookNo="22" page="182">

<Paragraph number="3315" type="verses" book="ja1">
_Nayimasmiṃ kumbhasmiṃ dadhi vā madhu vā,_  
_Evaṃ abhiññāya kiṇāhi rāja;_  
_Evañhimaṃ kumbhagatā mayā te,_  
_Akkhātarūpaṃ tava sabbamitta”._  

</Paragraph>

</Division>
<Division number="60" book="ja1" bookNo="22" page="182">

<Paragraph number="3316" type="verses" book="ja1">
_“Na me pitā vā athavāpi mātā,_  
_Etādisā yādisako tuvaṃsi;_  
_Hitānukampī paramatthakāmo,_  
_Sohaṃ karissaṃ vacanaṃ tavajja._  

</Paragraph>

</Division>
<Division number="61" book="ja1" bookNo="22" page="182">

<Paragraph number="3317" type="verses" book="ja1">
_Dadāmi te gāmavarāni pañca,_  
_Dāsīsataṃ satta gavaṃsatāni;_  
_Ājaññayutte ca rathe dasa ime,_  
_Ācariyo hosi mamatthakāmo”._  

</Paragraph>

</Division>
<Division number="62" book="ja1" bookNo="22" page="182">

<Paragraph number="3318" type="verses" book="ja1">
_“Taveva dāsīsatamatthu rāja,_  
_Gāmā ca gāvo ca taveva hontu;_  
_Ājaññayuttā ca rathā taveva,_  
_Sakkohamasmī tidasānamindo._  

</Paragraph>

</Division>
<Division number="63" book="ja1" bookNo="22" page="182">

<Paragraph number="3319" type="verses" book="ja1">
_Maṃsodanaṃ sappipāyāsaṃ bhuñja,_  
_Khādassu ca tvaṃ madhumāsapūve;_  
_Evaṃ tuvaṃ dhammarato janinda,_  
_Anindito saggamupehi ṭhānan”ti._  


</Paragraph>
</Division>

<Paragraph number="3320" type="center" book="ja1">
Kumbhajātakaṃ dutiyaṃ.
</Paragraph>
