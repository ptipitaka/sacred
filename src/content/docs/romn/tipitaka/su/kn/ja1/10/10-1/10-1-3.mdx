<Paragraph number="1934" type="verses" book="ja1">
_“Yo kopaneyye na karoti kopaṃ,_  
_Na kujjhati sappuriso kadāci;_  
_Kuddhopi so nāvikaroti kopaṃ,_  
_Taṃ ve naraṃ samaṇamāhu loke”._  

</Paragraph>

</Division>
<Division number="25" book="ja1" bookNo="22" page="40">

<Paragraph number="1935" type="verses" book="ja1">
_“Ūnūdaro yo sahate jighacchaṃ,_  
_Danto tapassī mitapānabhojano;_  
_Āhārahetu na karoti pāpaṃ,_  
_Taṃ ve naraṃ samaṇamāhu loke”._  

</Paragraph>

</Division>
<Division number="26" book="ja1" bookNo="22" page="41">

<Paragraph number="1936" type="verses" book="ja1">
_“Khiḍḍaṃ ratiṃ vippajahitvāna sabbaṃ,_  
_Na cālikaṃ bhāsasi kiñci loke;_  
_Vibhūsaṭṭhānā virato methunasmā,_  
_Taṃ ve naraṃ samaṇamāhu loke”._  

</Paragraph>

</Division>
<Division number="27" book="ja1" bookNo="22" page="41">

<Paragraph number="1937" type="verses" book="ja1">
_“Pariggahaṃ lobhadhammañca sabbaṃ,_  
_Yo ve pariññāya pariccajeti;_  
_Dantaṃ ṭhitattaṃ amamaṃ nirāsaṃ,_  
_Taṃ ve naraṃ samaṇamāhu loke”._  

</Paragraph>

</Division>
<Division number="28" book="ja1" bookNo="22" page="41">

<Paragraph number="1938" type="verses" book="ja1">
_“Pucchāma kattāramanomapaññaṃ,_  
_Kathāsu no viggaho atthi jāto;_  
_Chindajja kaṅkhaṃ vicikicchitāni,_  
_Tadajja kaṅkhaṃ vitaremu sabbe”._  

</Paragraph>

</Division>
<Division number="29" book="ja1" bookNo="22" page="41">

<Paragraph number="1939" type="verses" book="ja1">
_“Ye paṇḍitā atthadasā bhavanti,_  
_Bhāsanti te yoniso tattha kāle;_  
_Kathaṃ nu kathānaṃ abhāsitānaṃ,_  
_Atthaṃ nayeyyuṃ kusalā janindā._  

</Paragraph>

</Division>
<Division number="30" book="ja1" bookNo="22" page="41">

<Paragraph number="1940" type="verses" book="ja1">
_Kathaṃ have bhāsati nāgarājā,_  
_Garuḷo pana venateyyo kimāha;_  
_Gandhabbarājā pana kiṃ vadesi,_  
_Kathaṃ pana kurūnaṃ rājaseṭṭho”._  

</Paragraph>

</Division>
<Division number="31" book="ja1" bookNo="22" page="41">

<Paragraph number="1941" type="verses" book="ja1">
_“Khantiṃ have bhāsati nāgarājā,_  
_Appāhāraṃ garuḷo venateyyo;_  
_Gandhabbarājā rativippahānaṃ,_  
_Akiñcanaṃ kurūnaṃ rājaseṭṭho”._  

</Paragraph>

</Division>
<Division number="32" book="ja1" bookNo="22" page="41">

<Paragraph number="1942" type="verses" book="ja1">
_“Sabbāni etāni subhāsitāni,_  
_Na hettha dubbhāsitamatthi kiñci;_  
_Yasmiñca etāni patiṭṭhitāni,_  
_Arāva nābhyā susamohitāni;_  
_Catubbhi dhammehi samaṅgibhūtaṃ,_  
_Taṃ ve naraṃ samaṇamāhu loke”._  

</Paragraph>

</Division>
<Division number="33" book="ja1" bookNo="22" page="42">

<Paragraph number="1943" type="verses" book="ja1">
_“Tuvañhi seṭṭho tvamanuttarosi,_  
_Tvaṃ dhammagū dhammavidū sumedho;_  
_Paññāya pañhaṃ samadhiggahetvā,_  
_Acchecchi dhīro vicikicchitāni;_  
_Acchecchi kaṅkhaṃ vicikicchitāni,_  
_Cundo yathā nāgadantaṃ kharena._  

</Paragraph>

</Division>
<Division number="34" book="ja1" bookNo="22" page="42">

<Paragraph number="1944" type="verses" book="ja1">
_Nīluppalābhaṃ vimalaṃ anagghaṃ,_  
_Vatthaṃ idaṃ dhūmasamānavaṇṇaṃ;_  
_Pañhassa veyyākaraṇena tuṭṭho,_  
_Dadāmi te dhammapūjāya dhīra._  

</Paragraph>

</Division>
<Division number="35" book="ja1" bookNo="22" page="42">

<Paragraph number="1945" type="verses" book="ja1">
_Suvaṇṇamālaṃ satapattaphullitaṃ,_  
_Sakesaraṃ ratnasahassamaṇḍitaṃ;_  
_Pañhassa veyyākaraṇena tuṭṭho,_  
_Dadāmi te dhammapūjāya dhīra._  

</Paragraph>

</Division>
<Division number="36" book="ja1" bookNo="22" page="42">

<Paragraph number="1946" type="verses" book="ja1">
_Maṇiṃ anagghaṃ ruciraṃ pabhassaraṃ,_  
_Kaṇṭhāvasattaṃ maṇibhūsitaṃ me;_  
_Pañhassa veyyākaraṇena tuṭṭho,_  
_Dadāmi te dhammapūjāya dhīra._  

</Paragraph>

</Division>
<Division number="37" book="ja1" bookNo="22" page="42">

<Paragraph number="1947" type="verses" book="ja1">
_Gavaṃ sahassaṃ usabhañca nāgaṃ,_  
_Ājaññayutte ca rathe dasa ime;_  
_Pañhassa veyyākaraṇena tuṭṭho,_  
_Dadāmi te gāmavarāni soḷasa”._  

</Paragraph>

</Division>
<Division number="38" book="ja1" bookNo="22" page="42">

<Paragraph number="1948" type="verses" book="ja1">
_“Sāriputto tadā nāgo,_  
_supaṇṇo pana kolito;_  
_Gandhabbarājā anuruddho,_  
_rājā ānandapaṇḍito;_  
_Vidhuro bodhisatto ca,_  
_evaṃ dhāretha jātakan”ti._  


</Paragraph>
</Division>

Catuposathiyajātakaṃ tatiyaṃ.
