<Paragraph number="2612" type="verses" book="ja1">
_“Kharājinā jaṭilā paṅkadantā,_  
_Dummakkharūpā ye mantaṃ jappanti;_  
_Kaccinnu te mānusake payoge,_  
_Idaṃ vidū parimuttā apāyā”._  

</Paragraph>

</Division>
<Division number="63" book="ja1" bookNo="22" page="144">

<Paragraph number="2613" type="verses" book="ja1">
_“Pāpāni kammāni karetha rāja,_  
_Bahussuto ce na careyya dhammaṃ;_  
_Sahassavedopi na taṃ paṭicca,_  
_Dukkhā pamucce caraṇaṃ apatvā”._  

</Paragraph>

</Division>
<Division number="64" book="ja1" bookNo="22" page="144">

<Paragraph number="2614" type="verses" book="ja1">
_“Sahassavedopi na taṃ paṭicca,_  
_Dukkhā pamucce caraṇaṃ apatvā;_  
_Maññāmi vedā aphalā bhavanti,_  
_Sasaṃyamaṃ caraṇaññeva saccaṃ”._  

</Paragraph>

</Division>
<Division number="65" book="ja1" bookNo="22" page="144">

<Paragraph number="2615" type="verses" book="ja1">
_“Na heva vedā aphalā bhavanti,_  
_Sasaṃyamaṃ caraṇaññeva saccaṃ;_  
_Kittiñhi pappoti adhicca vede,_  
_Santiṃ puṇāti caraṇena danto”._  

</Paragraph>

</Division>
<Division number="66" book="ja1" bookNo="22" page="144">

<Paragraph number="2616" type="verses" book="ja1">
_“Bhaccā mātā pitā bandhū,_  
_yena jāto sayeva so;_  
_Uddālako ahaṃ bhoto,_  
_sottiyākulavaṃsako”._  

</Paragraph>

</Division>
<Division number="67" book="ja1" bookNo="22" page="144">

<Paragraph number="2617" type="verses" book="ja1">
_“Kathaṃ bho brāhmaṇo hoti,_  
_kathaṃ bhavati kevalī;_  
_Kathañca parinibbānaṃ,_  
_dhammaṭṭho kinti vuccati”._  

</Paragraph>

</Division>
<Division number="68" book="ja1" bookNo="22" page="144">

<Paragraph number="2618" type="verses" book="ja1">
_“Niraṃkatvā aggimādāya brāhmaṇo,_  
_Āpo siñcaṃ yajaṃ usseti yūpaṃ;_  
_Evaṃkaro brāhmaṇo hoti khemī,_  
_Dhamme ṭhitaṃ tena amāpayiṃsu”._  

</Paragraph>

</Division>
<Division number="69" book="ja1" bookNo="22" page="145">

<Paragraph number="2619" type="verses" book="ja1">
_“Na suddhi secanenatthi,_  
_nāpi kevalī brāhmaṇo;_  
_Na khantī nāpi soraccaṃ,_  
_nāpi so parinibbuto”._  

</Paragraph>

</Division>
<Division number="70" book="ja1" bookNo="22" page="145">

<Paragraph number="2620" type="verses" book="ja1">
_“Kathaṃ so brāhmaṇo hoti,_  
_kathaṃ bhavati kevalī;_  
_Kathañca parinibbānaṃ,_  
_dhammaṭṭho kinti vuccati”._  

</Paragraph>

</Division>
<Division number="71" book="ja1" bookNo="22" page="145">

<Paragraph number="2621" type="verses" book="ja1">
_“Akhettabandhū amamo nirāso,_  
_Nillobhapāpo bhavalobhakhīṇo;_  
_Evaṃkaro brāhmaṇo hoti khemī,_  
_Dhamme ṭhitaṃ tena amāpayiṃsu”._  

</Paragraph>

</Division>

_suddā caṇḍālapukkusā;_  
_Sabbeva soratā dantā,_  
_sabbeva parinibbutā;_  
_Sabbesaṃ sītibhūtānaṃ,_  
_atthi seyyotha pāpiyo”._  

</Paragraph>

_Natthi seyyotha pāpiyo;_  
_Panatthaṃ carasi brahmaññaṃ,_  
_Sottiyākulavaṃsataṃ”._  

</Paragraph>

_“Nānārattehi vatthehi,_  
_vimānaṃ bhavati chāditaṃ;_  
_Na tesaṃ chāyā vatthānaṃ,_  
_so rāgo anupajjatha._  

</Paragraph>

_dhammamaññāya subbatā”ti._
