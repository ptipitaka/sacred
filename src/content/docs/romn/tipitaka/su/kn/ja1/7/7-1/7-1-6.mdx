<Paragraph number="1528" type="verses" book="ja1">
_“Paṇṇakaṃ tikhiṇadhāraṃ,_  
_Asiṃ sampannapāyinaṃ;_  
_Parisāyaṃ puriso gilati,_  
_Kiṃ dukkarataraṃ tato;_  
_Yadaññaṃ dukkaraṃ ṭhānaṃ,_  
_Taṃ me akkhāhi pucchito”._  

</Paragraph>

</Division>
<Division number="40" book="ja1" bookNo="22" page="279">

<Paragraph number="1529" type="verses" book="ja1">
_“Gileyya puriso lobhā,_  
_asiṃ sampannapāyinaṃ;_  
_Yo ca vajjā dadāmīti,_  
_taṃ dukkarataraṃ tato;_  
_Sabbaññaṃ sukaraṃ ṭhānaṃ,_  
_evaṃ jānāhi maddava”._  

</Paragraph>

</Division>
<Division number="41" book="ja1" bookNo="22" page="279">

<Paragraph number="1530" type="verses" book="ja1">
_“Byākāsi āyuro pañhaṃ,_  
_atthaṃ dhammassa kovido;_  
_Pukkusaṃ dāni pucchāmi,_  
_kiṃ dukkarataraṃ tato;_  
_Yadaññaṃ dukkaraṃ ṭhānaṃ,_  
_taṃ me akkhāhi pucchito”._  

</Paragraph>

</Division>
<Division number="42" book="ja1" bookNo="22" page="280">

<Paragraph number="1531" type="verses" book="ja1">
_“Na vācamupajīvanti,_  
_aphalaṃ giramudīritaṃ;_  
_Yo ca datvā avākayirā,_  
_taṃ dukkarataraṃ tato;_  
_Sabbaññaṃ sukaraṃ ṭhānaṃ,_  
_evaṃ jānāhi maddava”._  

</Paragraph>

</Division>
<Division number="43" book="ja1" bookNo="22" page="280">

<Paragraph number="1532" type="verses" book="ja1">
_“Byākāsi pukkuso pañhaṃ,_  
_atthaṃ dhammassa kovido;_  
_Senakaṃ dāni pucchāmi,_  
_kiṃ dukkarataraṃ tato;_  
_Yadaññaṃ dukkaraṃ ṭhānaṃ,_  
_taṃ me akkhāhi pucchito”._  

</Paragraph>

</Division>
<Division number="44" book="ja1" bookNo="22" page="280">

<Paragraph number="1533" type="verses" book="ja1">
_“Dadeyya puriso dānaṃ,_  
_appaṃ vā yadi vā bahuṃ;_  
_Yo ca datvā nānutappe,_  
_taṃ dukkarataraṃ tato;_  
_Sabbaññaṃ sukaraṃ ṭhānaṃ,_  
_evaṃ jānāhi maddava”._  

</Paragraph>

</Division>
<Division number="45" book="ja1" bookNo="22" page="280">

<Paragraph number="1534" type="verses" book="ja1">
_“Byākāsi āyuro pañhaṃ,_  
_atho pukkusaporiso;_  
_Sabbe pañhe atibhoti,_  
_yathā bhāsati senako”ti._  


</Paragraph>
</Division>

Paṇṇakajātakaṃ chaṭṭhaṃ.
