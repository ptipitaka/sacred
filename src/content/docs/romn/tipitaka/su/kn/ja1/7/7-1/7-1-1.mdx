<Paragraph number="1485" type="verses" book="ja1">
_“Diyaḍḍhakukkū udayena kaṇṇikā,_  
_Vidatthiyo aṭṭha parikkhipanti naṃ;_  
_Sā siṃsapā sāramayā apheggukā,_  
_Kuhiṃ ṭhitā upparito na dhaṃsati”._  

</Paragraph>

</Division>
<Division number="2" book="ja1" bookNo="22" page="276">

<Paragraph number="1486" type="verses" book="ja1">
_“Yā tiṃsati sāramayā anujjukā,_  
_Parikiriya gopānasiyo samaṃ ṭhitā;_  
_Tāhi susaṅgahitā balasā pīḷitā,_  
_Samaṃ ṭhitā upparito na dhaṃsati._  

</Paragraph>

</Division>
<Division number="3" book="ja1" bookNo="22" page="309">

<Paragraph number="1487" type="verses" book="ja1">
_Evampi mittehi daḷhehi paṇḍito,_  
_Abhejjarūpehi sucīhi mantibhi;_  
_Susaṅgahīto siriyā na dhaṃsati,_  
_Gopānasī bhāravahāva kaṇṇikā._  

</Paragraph>

</Division>
<Division number="4" book="ja1" bookNo="22" page="276">

<Paragraph number="1488" type="verses" book="ja1">
_Kharattacaṃ bellaṃ yathāpi satthavā,_  
_Anāmasantopi karoti tittakaṃ;_  
_Samāharaṃ sāduṃ karoti patthiva,_  
_Asāduṃ kayirā tanubandhamuddharaṃ._  

</Paragraph>

</Division>
<Division number="5" book="ja1" bookNo="22" page="276">

<Paragraph number="1489" type="verses" book="ja1">
_Evampi gāmanigamesu paṇḍito,_  
_Asāhasaṃ rājadhanāni saṃgharaṃ;_  
_Dhammānuvattī paṭipajjamāno,_  
_Sa phāti kayirā aviheṭhayaṃ paraṃ._  

</Paragraph>

</Division>
<Division number="6" book="ja1" bookNo="22" page="276">

<Paragraph number="1490" type="verses" book="ja1">
_Odātamūlaṃ sucivārisambhavaṃ,_  
_Jātaṃ yathā pokkharaṇīsu ambujaṃ;_  
_Padumaṃ yathā agginikāsiphālimaṃ,_  
_Na kaddamo na rajo na vāri limpati._  

</Paragraph>

</Division>
<Division number="7" book="ja1" bookNo="22" page="276">

<Paragraph number="1491" type="verses" book="ja1">
_Evampi vohārasuciṃ asāhasaṃ,_  
_Visuddhakammantamapetapāpakaṃ;_  
_Na limpati kammakilesa tādiso,_  
_Jātaṃ yathā pokkharaṇīsu ambujan”ti._  


</Paragraph>
</Division>

<Paragraph number="1492" type="center" book="ja1">
Kukkujātakaṃ paṭhamaṃ.
</Paragraph>
