<Division number="95" book="ja1">

<Paragraph number="1259" type="verses" book="ja1">
_“Amittahatthatthagatā,_  
_tacasārasamappitā;_  
_Pasannamukhavaṇṇāttha,_  
_kasmā tumhe na socatha”._  

</Paragraph>

</Division>
<Division number="96" book="ja1">

<Paragraph number="1260" type="verses" book="ja1">
_“Na socanāya paridevanāya,_  
_Atthova labbho api appakopi;_  
_Socantamenaṃ dukhitaṃ viditvā,_  
_Paccatthikā attamanā bhavanti._  

</Paragraph>

</Division>
<Division number="97" book="ja1">

<Paragraph number="1261" type="verses" book="ja1">
_Yato ca kho paṇḍito āpadāsu,_  
_Na vedhatī atthavinicchayaññū;_  
_Paccatthikāssa dukhitā bhavanti,_  
_Disvā mukhaṃ avikāraṃ purāṇaṃ._  

</Paragraph>

</Division>
<Division number="98" book="ja1">

<Paragraph number="1262" type="verses" book="ja1">
_Jappena mantena subhāsitena,_  
_Anuppadānena paveṇiyā vā;_  
_Yathā yathā yattha labhetha atthaṃ,_  
_Tathā tathā tattha parakkameyya._  

</Paragraph>

</Division>
<Division number="99" book="ja1">

<Paragraph number="1263" type="verses" book="ja1">
_Yato ca jāneyya alabbhaneyyo,_  
_Mayā va aññena vā esa attho;_  
_Asocamāno adhivāsayeyya,_  
_Kammaṃ daḷhaṃ kinti karomi dānī”ti._  


</Paragraph>
</Division>

<Paragraph number="1264" type="center" book="ja1">
Tacasārajātakaṃ aṭṭhamaṃ.
</Paragraph>
