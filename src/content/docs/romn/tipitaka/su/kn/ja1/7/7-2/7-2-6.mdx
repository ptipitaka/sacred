<Paragraph number="1615" type="verses" book="ja1">
_“Kāḷāni kesāni pure ahesuṃ,_  
_Jātāni sīsamhi yathāpadese;_  
_Tānajja setāni susīma disvā,_  
_Dhammaṃ cara brahmacariyassa kālo”._  

</Paragraph>

</Division>
<Division number="115" book="ja1" bookNo="22" page="373">

<Paragraph number="1616" type="verses" book="ja1">
_“Mameva deva palitaṃ na tuyhaṃ,_  
_Mameva sīsaṃ mama uttamaṅgaṃ;_  
_‘Atthaṃ karissan’ti musā abhāṇiṃ,_  
_Ekāparādhaṃ khama rājaseṭṭha._  

</Paragraph>

</Division>
<Division number="116" book="ja1" bookNo="22" page="373">

<Paragraph number="1617" type="verses" book="ja1">
_Daharo tuvaṃ dassaniyosi rāja,_  
_Paṭhamuggato hosi yathā kaḷīro;_  
_Rajjañca kārehi mamañca passa,_  
_Mā kālikaṃ anudhāvī janinda”._  

</Paragraph>

</Division>
<Division number="117" book="ja1" bookNo="22" page="373">

<Paragraph number="1618" type="verses" book="ja1">
_“Passāmi vohaṃ dahariṃ kumāriṃ,_  
_Sāmaṭṭhapassaṃ sutanuṃ sumajjhaṃ;_  
_Kāḷappavāḷāva pavellamānā,_  
_Palobhayantīva naresu gacchati._  

</Paragraph>

</Division>
<Division number="118" book="ja1" bookNo="22" page="373">

<Paragraph number="1619" type="verses" book="ja1">
_Tamena passāmiparena nāriṃ,_  
_Āsītikaṃ nāvutikaṃ va jaccā;_  
_Daṇḍaṃ gahetvāna pavedhamānaṃ,_  
_Gopānasībhoggasamaṃ carantiṃ._  

</Paragraph>

</Division>
<Division number="119" book="ja1" bookNo="22" page="374">

<Paragraph number="1620" type="verses" book="ja1">
_Sohaṃ tamevānuvicintayanto,_  
_Eko sayāmi sayanassa majjhe;_  
_‘Ahampi evaṃ’ iti pekkhamāno,_  
_Na gahe rame brahmacariyassa kālo._  

</Paragraph>

</Division>
<Division number="120" book="ja1" bookNo="22" page="374">

<Paragraph number="1621" type="verses" book="ja1">
_Rajjuvālambanī cesā,_  
_Yā gehe vasato rati;_  
_Evampi chetvāna vajanti dhīrā,_  
_Anapekkhino kāmasukhaṃ pahāyā”ti._  


</Paragraph>
</Division>

<Paragraph number="1622" type="center" book="ja1">
Susīmajātakaṃ chaṭṭhaṃ.
</Paragraph>
