<Paragraph number="2593" type="verses" book="ja1">
_“Ukkā cilācā bandhanti dīpe,_  
_Pajā mamaṃ khādituṃ patthayanti;_  
_Mittaṃ sahāyañca vadehi senaka,_  
_Ācikkha ñātibyasanaṃ dijānaṃ”._  

</Paragraph>

</Division>
<Division number="45" book="ja1" bookNo="22" page="142">

<Paragraph number="2594" type="verses" book="ja1">
_“Dijo dijānaṃ pavarosi pakkhima,_  
_Ukkusarāja saraṇaṃ taṃ upema;_  
_Pajā mamaṃ khādituṃ patthayanti,_  
_Luddā cilācā bhava me sukhāya”._  

</Paragraph>

</Division>
<Division number="46" book="ja1" bookNo="22" page="142">

<Paragraph number="2595" type="verses" book="ja1">
_“Mittaṃ sahāyañca karonti paṇḍitā,_  
_Kāle akāle sukhamesamānā;_  
_Karomi te senaka etamatthaṃ,_  
_Ariyo hi ariyassa karoti kiccaṃ”._  

</Paragraph>

</Division>
<Division number="47" book="ja1" bookNo="22" page="142">

<Paragraph number="2596" type="verses" book="ja1">
_“Yaṃ hoti kiccaṃ anukampakena,_  
_Ariyassa ariyena kataṃ tayīdaṃ;_  
_Attānurakkhī bhava mā adayhi,_  
_Lacchāma putte tayi jīvamāne”._  

</Paragraph>

</Division>
<Division number="48" book="ja1" bookNo="22" page="142">

<Paragraph number="2597" type="verses" book="ja1">
_“Taveva rakkhāvaraṇaṃ karonto,_  
_Sarīrabhedāpi na santasāmi;_  
_Karonti heke sakhīnaṃ sakhāro,_  
_Pāṇaṃ cajantā satamesa dhammo”._  

</Paragraph>

</Division>
<Division number="49" book="ja1" bookNo="22" page="142">

<Paragraph number="2598" type="verses" book="ja1">
_“Sudukkaraṃ kammamakāsi,_  
_aṇḍajāyaṃ vihaṅgamo;_  
_Atthāya kuraro putte,_  
_aḍḍharatte anāgate”._  

</Paragraph>

</Division>
<Division number="50" book="ja1" bookNo="22" page="142">

<Paragraph number="2599" type="verses" book="ja1">
_“Cutāpi heke khalitā sakammunā,_  
_Mittānukampāya patiṭṭhahanti;_  
_Puttā mamaṭṭā gatimāgatosmi,_  
_Atthaṃ caretho mama vāricara”._  

</Paragraph>

</Division>
<Division number="51" book="ja1" bookNo="22" page="142">

<Paragraph number="2600" type="verses" book="ja1">
_“Dhanena dhaññena ca attanā ca,_  
_Mittaṃ sahāyañca karonti paṇḍitā;_  
_Karomi te senaka etamatthaṃ,_  
_Ariyo hi ariyassa karoti kiccaṃ”._  

</Paragraph>

</Division>
<Division number="52" book="ja1" bookNo="22" page="143">

<Paragraph number="2601" type="verses" book="ja1">
_“Appossukko tāta tuvaṃ nisīda,_  
_Putto pitu carati atthacariyaṃ;_  
_Ahaṃ carissāmi tavetamatthaṃ,_  
_Senassa putte paritāyamāno”._  

</Paragraph>

</Division>
<Division number="53" book="ja1" bookNo="22" page="143">

<Paragraph number="2602" type="verses" book="ja1">
_“Addhā hi tāta satamesa dhammo,_  
_Putto pitu yaṃ care atthacariyaṃ;_  
_Appeva maṃ disvāna pavaḍḍhakāyaṃ,_  
_Senassa puttā na viheṭhayeyyuṃ”._  

</Paragraph>

</Division>
<Division number="54" book="ja1" bookNo="22" page="143">

<Paragraph number="2603" type="verses" book="ja1">
_“Pasū manussā migavīraseṭṭha,_  
_Bhayaṭṭitā seṭṭhamupabbajanti;_  
_Puttā mamaṭṭā gatimāgatosmi,_  
_Tvaṃ nosi rājā bhava me sukhāya”._  

</Paragraph>

</Division>
<Division number="55" book="ja1" bookNo="22" page="143">

<Paragraph number="2604" type="verses" book="ja1">
_“Karomi te senaka etamatthaṃ,_  
_Āyāmi te taṃ disataṃ vadhāya;_  
_Kathañhi viññū pahu sampajāno,_  
_Na vāyame attajanassa guttiyā”._  

</Paragraph>

</Division>
<Division number="56" book="ja1" bookNo="22" page="143">

<Paragraph number="2605" type="verses" book="ja1">
_“Mittañca kayirātha suhadayañca,_  
_Ayirañca kayirātha sukhāgamāya;_  
_Nivatthakocova sarebhihantvā,_  
_Modāma puttehi samaṅgibhūtā._  

</Paragraph>

</Division>
<Division number="57" book="ja1" bookNo="22" page="143">

<Paragraph number="2606" type="verses" book="ja1">
_Sakamittassa kammena,_  
_sahāyassāpalāyino;_  
_Kūjantamupakūjanti,_  
_lomasā hadayaṅgamaṃ._  

</Paragraph>

</Division>
<Division number="58" book="ja1" bookNo="22" page="162">

<Paragraph number="2607" type="verses" book="ja1">
_Mittaṃ sahāyaṃ adhigamma paṇḍito,_  
_So bhuñjatī putta pasuṃ dhanaṃ vā;_  
_Ahañca puttā ca patī ca mayhaṃ,_  
_Mittānukampāya samaṅgibhūtā._  

</Paragraph>

</Division>
<Division number="59" book="ja1" bookNo="22" page="144">

<Paragraph number="2608" type="verses" book="ja1">
_Rājavatā sūravatā ca attho,_  
_Sampannasakhissa bhavanti hete;_  
_So mittavā yasavā uggatatto,_  
_Asmiṃdhaloke modati kāmakāmī._  

</Paragraph>

</Division>
<Division number="60" book="ja1" bookNo="22" page="144">

<Paragraph number="2609" type="verses" book="ja1">
_Karaṇīyāni mittāni,_  
_daliddenāpi senaka;_  
_Passa mittānukampāya,_  
_samaggamhā sañātake._  

</Paragraph>

</Division>
<Division number="61" book="ja1" bookNo="22" page="144">

<Paragraph number="2610" type="verses" book="ja1">
_Sūrena balavantena,_  
_yo mitte kurute dijo;_  
_Evaṃ so sukhito hoti,_  
_yathāhaṃ tvañca senakāti”._  


</Paragraph>
</Division>

<Paragraph number="2611" type="center" book="ja1">
Mahāukkusajātakaṃ tatiyaṃ.
