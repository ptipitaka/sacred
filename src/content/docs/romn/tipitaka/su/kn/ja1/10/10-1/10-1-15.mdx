<Division number="155" book="ja1">

<Paragraph number="2077" type="verses" book="ja1">
_“Kiṃsu naro jappamadhiccakāle,_  
_Kaṃ vā vijjaṃ katamaṃ vā sutānaṃ;_  
_So macco asmiñca paramhi loke,_  
_Kathaṃ karo sotthānena gutto”._  

</Paragraph>

</Division>
<Division number="156" book="ja1">

<Paragraph number="2078" type="verses" book="ja1">
_“Yassa devā pitaro ca sabbe,_  
_Sarīsapā sabbabhūtāni cāpi;_  
_Mettāya niccaṃ apacitāni honti,_  
_Bhūtesu ve sotthānaṃ tadāhu”._  

</Paragraph>

</Division>
<Division number="157" book="ja1">

<Paragraph number="2079" type="verses" book="ja1">
_“Yo sabbalokassa nivātavutti,_  
_Itthīpumānaṃ sahadārakānaṃ;_  
_Khantā duruttānamappaṭikūlavādī,_  
_Adhivāsanaṃ sotthānaṃ tadāhu._  

</Paragraph>

</Division>
<Division number="158" book="ja1">

<Paragraph number="2080" type="verses" book="ja1">
_Yo nāvajānāti sahāyamitte,_  
_Sippena kulyāhi dhanena jaccā;_  
_Rucipañño atthakāle matīmā,_  
_Sahāyesu ve sotthānaṃ tadāhu._  

</Paragraph>

</Division>
<Division number="159" book="ja1">

<Paragraph number="2081" type="verses" book="ja1">
_Mittāni ve yassa bhavanti santo,_  
_Saṃvissatthā avisaṃvādakassa;_  
_Na mittadubbhī saṃvibhāgī dhanena,_  
_Mittesu ve sotthānaṃ tadāhu._  

</Paragraph>

</Division>
<Division number="160" book="ja1">

<Paragraph number="2082" type="verses" book="ja1">
_Yassa bhariyā tulyavayā samaggā,_  
_Anubbatā dhammakāmā pajātā;_  
_Koliniyā sīlavatī patibbatā,_  
_Dāresu ve sotthānaṃ tadāhu._  

</Paragraph>

</Division>
<Division number="161" book="ja1">

<Paragraph number="2083" type="verses" book="ja1">
_Yassa rājā bhūtapati yasassī,_  
_Jānāti soceyyaṃ parakkamañca;_  
_Advejjhatā suhadayaṃ mamanti,_  
_Rājūsu ve sotthānaṃ tadāhu._  

</Paragraph>

</Division>
<Division number="162" book="ja1">

<Paragraph number="2084" type="verses" book="ja1">
_Annañca pānañca dadāti saddho,_  
_Mālañca gandhañca vilepanañca;_  
_Pasannacitto anumodamāno,_  
_Saggesu ve sotthānaṃ tadāhu._  

</Paragraph>

</Division>
<Division number="163" book="ja1">

<Paragraph number="2085" type="verses" book="ja1">
_Yamariyadhammena punanti vuddhā,_  
_Ārādhitā samacariyāya santo;_  
_Bahussutā isayo sīlavanto,_  
_Arahantamajjhe sotthānaṃ tadāhu._  

</Paragraph>

</Division>
<Division number="164" book="ja1">

<Paragraph number="2086" type="verses" book="ja1">
_Etāni kho sotthānāni loke,_  
_Viññuppasatthāni sukhudrayāni;_  
_Tānīdha sevetha naro sapañño,_  
_Na hi maṅgale kiñcanamatthi saccan”ti._  


</Paragraph>
</Division>

<Paragraph number="2087" type="center" book="ja1">
Mahāmaṅgalajātakaṃ pannarasamaṃ.
</Paragraph>
