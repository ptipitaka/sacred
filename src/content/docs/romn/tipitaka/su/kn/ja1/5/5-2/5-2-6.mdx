<Paragraph number="1247" type="verses" book="ja1">
_“Madhuvaṇṇaṃ madhurasaṃ,_  
_madhugandhaṃ visaṃ ahu;_  
_Gumbiyo ghāsamesāno,_  
_araññe odahī visaṃ._  

</Paragraph>

</Division>
<Division number="86" book="ja1" bookNo="22" page="250">

<Paragraph number="1248" type="verses" book="ja1">
_Madhu iti maññamānā,_  
_ye taṃ visamakhādisuṃ;_  
_Tesaṃ taṃ kaṭukaṃ āsi,_  
_maraṇaṃ tenupāgamuṃ._  

</Paragraph>

</Division>
<Division number="87" book="ja1" bookNo="22" page="250">

<Paragraph number="1249" type="verses" book="ja1">
_Ye ca kho paṭisaṅkhāya,_  
_visaṃ taṃ parivajjayuṃ;_  
_Te āturesu sukhitā,_  
_ḍayhamānesu nibbutā._  

</Paragraph>

</Division>
<Division number="88" book="ja1" bookNo="22" page="250">

<Paragraph number="1250" type="verses" book="ja1">
_Evameva manussesu,_  
_visaṃ kāmā samohitā;_  
_Āmisaṃ bandhanañcetaṃ,_  
_maccuveso guhāsayo._  

</Paragraph>

</Division>
<Division number="89" book="ja1" bookNo="22" page="251">

<Paragraph number="1251" type="verses" book="ja1">
_Evameva ime kāme,_  
_āturā paricārike;_  
_Ye sadā parivajjenti,_  
_saṅgaṃ loke upaccagun”ti._  


</Paragraph>
</Division>

<Paragraph number="1252" type="center" book="ja1">
Gumbiyajātakaṃ chaṭṭhaṃ.
</Paragraph>
