<Paragraph number="2853" type="verses" book="ja1">
_“Kuto nu āgacchasi dummavāsī,_  
_Otallako paṃsupisācakova;_  
_Saṅkāracoḷaṃ paṭimuñca kaṇṭhe,_  
_Ko re tuvaṃ hosi adakkhiṇeyyo”._  

</Paragraph>

<Division number="2" book="ja1" bookNo="22" page="155">

<Paragraph number="2854" type="verses" book="ja1">
_“Annaṃ tavedaṃ pakataṃ yasassi,_  
_Taṃ khajjare bhuñjare piyyare ca;_  
_Jānāsi maṃ tvaṃ paradattūpajīviṃ,_  
_Uttiṭṭhapiṇḍaṃ labhataṃ sapāko”._  

</Paragraph>

</Division>
<Division number="3" book="ja1" bookNo="22" page="175">

<Paragraph number="2855" type="verses" book="ja1">
_“Annaṃ mamedaṃ pakataṃ brāhmaṇānaṃ,_  
_Attatthāya saddahato mamedaṃ;_  
_Apehi etto kimidhaṭṭhitosi,_  
_Na mādisā tuyhaṃ dadanti jamma”._  

</Paragraph>

</Division>
<Division number="4" book="ja1" bookNo="22" page="155">

<Paragraph number="2856" type="verses" book="ja1">
_“Thale ca ninne ca vapanti bījaṃ,_  
_Anūpakhette phalamāsamānā;_  
_Etāya saddhāya dadāhi dānaṃ,_  
_Appeva ārādhaye dakkhiṇeyye”._  

</Paragraph>

</Division>
<Division number="5" book="ja1" bookNo="22" page="155">

<Paragraph number="2857" type="verses" book="ja1">
_“Khettāni mayhaṃ viditāni loke,_  
_Yesāhaṃ bījāni patiṭṭhapemi;_  
_Ye brāhmaṇā jātimantūpapannā,_  
_Tānīdha khettāni supesalāni”._  

</Paragraph>

</Division>
<Division number="6" book="ja1" bookNo="22" page="156">

<Paragraph number="2858" type="verses" book="ja1">
_“Jātimado ca atimānitā ca,_  
_Lobho ca doso ca mado ca moho;_  
_Ete aguṇā yesu ca santi sabbe,_  
_Tānīdha khettāni apesalāni._  

</Paragraph>

</Division>
<Division number="7" book="ja1" bookNo="22" page="156">

<Paragraph number="2859" type="verses" book="ja1">
_Jātimado ca atimānitā ca,_  
_Lobho ca doso ca mado ca moho;_  
_Ete aguṇā yesu na santi sabbe,_  
_Tānīdha khettāni supesalāni”._  

</Paragraph>

</Division>
<Division number="8" book="ja1" bookNo="22" page="156">

<Paragraph number="2860" type="verses" book="ja1">
_“Kvettha gatā upajotiyo ca,_  
_Upajjhāyo ca atha vā gaṇḍakucchi;_  
_Imassa daṇḍañca vadhañca datvā,_  
_Gale gahetvā khalayātha jammaṃ”._  

</Paragraph>

</Division>
<Division number="9" book="ja1" bookNo="22" page="156">

<Paragraph number="2861" type="verses" book="ja1">
_“Giriṃ nakhena khaṇasi,_  
_ayo dantehi khādasi;_  
_Jātavedaṃ padahasi,_  
_yo isiṃ paribhāsasi”._  

</Paragraph>

</Division>
<Division number="10" book="ja1" bookNo="22" page="156">

<Paragraph number="2862" type="verses" book="ja1">
_“Idaṃ vatvāna mātaṅgo,_  
_isi saccaparakkamo;_  
_Antalikkhasmiṃ pakkāmi,_  
_brāhmaṇānaṃ udikkhataṃ”._  

</Paragraph>

</Division>
<Division number="11" book="ja1" bookNo="22" page="156">

<Paragraph number="2863" type="verses" book="ja1">
_“Āvellitaṃ piṭṭhito uttamaṅgaṃ,_  
_Bāhuṃ pasāreti akammaneyyaṃ;_  
_Setāni akkhīni yathā matassa,_  
_Ko me imaṃ puttamakāsi evaṃ”._  

</Paragraph>

</Division>
<Division number="12" book="ja1" bookNo="22" page="156">

<Paragraph number="2864" type="verses" book="ja1">
_“Idhāgamā samaṇo dummavāsī,_  
_Otallako paṃsupisācakova;_  
_Saṅkāracoḷaṃ paṭimuñca kaṇṭhe,_  
_So te imaṃ puttamakāsi evaṃ”._  

</Paragraph>

</Division>
<Division number="13" book="ja1" bookNo="22" page="156">

<Paragraph number="2865" type="verses" book="ja1">
_“Katamaṃ disaṃ agamā bhūripañño,_  
_Akkhātha me māṇavā etamatthaṃ;_  
_Gantvāna taṃ paṭikaremu accayaṃ,_  
_Appeva naṃ putta labhemu jīvitaṃ”._  

</Paragraph>

</Division>
<Division number="14" book="ja1" bookNo="22" page="157">

<Paragraph number="2866" type="verses" book="ja1">
_“Vehāyasaṃ agamā bhūripañño,_  
_Pathaddhuno pannaraseva cando;_  
_Api cāpi so purimadisaṃ agacchi,_  
_Saccappaṭiñño isi sādhurūpo”._  

</Paragraph>

</Division>

<Division number="16" book="ja1" bookNo="22" page="157">

<Paragraph number="2868" type="verses" book="ja1">
_“Yakkhā have santi mahānubhāvā,_  
_Anvāgatā isayo sādhurūpā;_  
_Te duṭṭhacittaṃ kupitaṃ viditvā,_  
_Yakkhā hi te puttamakaṃsu evaṃ”._  

</Paragraph>

</Division>
<Division number="17" book="ja1" bookNo="22" page="157">

<Paragraph number="2869" type="verses" book="ja1">
_“Yakkhā ca me puttamakaṃsu evaṃ,_  
_Tvaññeva me mā kuddho brahmacāri;_  
_Tumheva pāde saraṇaṃ gatāsmi,_  
_Anvāgatā puttasokena bhikkhu”._  

</Paragraph>

</Division>
<Division number="18" book="ja1" bookNo="22" page="157">

<Paragraph number="2870" type="verses" book="ja1">
_“Tadeva hi etarahi ca mayhaṃ,_  
_Manopadoso na mamatthi koci;_  
_Putto ca te vedamadena matto,_  
_Atthaṃ na jānāti adhicca vede”._  

</Paragraph>

</Division>
<Division number="19" book="ja1" bookNo="22" page="157">

<Paragraph number="2871" type="verses" book="ja1">
_“Addhā have bhikkhu muhuttakena,_  
_Sammuyhateva purisassa saññā;_  
_Ekāparādhaṃ khama bhūripañña,_  
_Na paṇḍitā kodhabalā bhavanti”._  

</Paragraph>

</Division>
<Division number="20" book="ja1" bookNo="22" page="157">

<Paragraph number="2872" type="verses" book="ja1">
_“Idañca mayhaṃ uttiṭṭhapiṇḍaṃ,_  
_Tava maṇḍabyo bhuñjatu appapañño;_  
_Yakkhā ca te naṃ na viheṭhayeyyuṃ,_  
_Putto ca te hessati so arogo”._  

</Paragraph>

</Division>
<Division number="21" book="ja1" bookNo="22" page="157">

<Paragraph number="2873" type="verses" book="ja1">
_“Maṇḍabya bālosi parittapañño,_  
_Yo puññakhettānamakovidosi;_  
_Mahakkasāvesu dadāsi dānaṃ,_  
_Kiliṭṭhakammesu asaññatesu._  

</Paragraph>

</Division>
<Division number="22" book="ja1" bookNo="22" page="158">

<Paragraph number="2874" type="verses" book="ja1">
_Jaṭā ca kesā ajinā nivatthā,_  
_Jarūdapānaṃva mukhaṃ parūḷhaṃ;_  
_Pajaṃ imaṃ passatha dummarūpaṃ,_  
_Na jaṭājinaṃ tāyati appapaññaṃ._  

</Paragraph>

</Division>

_avijjā ca virājitā;_  
_Khīṇāsavā arahanto,_  
_tesu dinnaṃ mahapphalan”ti._  


</Paragraph>
