<Paragraph number="2184" type="verses" book="ja1">
_“Mittāmaccaparibyūḷhaṃ,_  
_ahaṃ vande rathesabhaṃ;_  
_Pabbajissāmahaṃ rāja,_  
_taṃ devo anumaññatu”._  

</Paragraph>

</Division>
<Division number="74" book="ja1" bookNo="22" page="81">

<Paragraph number="2185" type="verses" book="ja1">
_“Sace te ūnaṃ kāmehi,_  
_ahaṃ paripūrayāmi te;_  
_Yo taṃ hiṃsati vāremi,_  
_mā pabbaja yudhañcaya”._  

</Paragraph>

</Division>
<Division number="75" book="ja1" bookNo="22" page="81">

<Paragraph number="2186" type="verses" book="ja1">
_“Na matthi ūnaṃ kāmehi,_  
_hiṃsitā me na vijjati;_  
_Dīpañca kātumicchāmi,_  
_yaṃ jarā nābhikīrati”._  

</Paragraph>

</Division>
<Division number="76" book="ja1" bookNo="22" page="81">

<Paragraph number="2187" type="verses" book="ja1">
_“Putto vā pitaraṃ yāce,_  
_pitā vā puttamorasaṃ;_  
_Negamo taṃ yāce tāta,_  
_mā pabbaja yudhañcaya”._  

</Paragraph>

</Division>
<Division number="77" book="ja1" bookNo="22" page="81">

<Paragraph number="2188" type="verses" book="ja1">
_“Mā maṃ deva nivārehi,_  
_pabbajantaṃ rathesabha;_  
_Māhaṃ kāmehi sammatto,_  
_jarāya vasamanvagū”._  

</Paragraph>

</Division>
<Division number="78" book="ja1" bookNo="22" page="81">

<Paragraph number="2189" type="verses" book="ja1">
_“Ahaṃ taṃ tāta yācāmi,_  
_ahaṃ putta nivāraye;_  
_Ciraṃ taṃ daṭṭhumicchāmi,_  
_mā pabbaja yudhañcaya”._  

</Paragraph>

</Division>
<Division number="79" book="ja1" bookNo="22" page="82">

<Paragraph number="2190" type="verses" book="ja1">
_“Ussāvova tiṇaggamhi,_  
_sūriyuggamanaṃ pati;_  
_Evamāyu manussānaṃ,_  
_mā maṃ amma nivāraya”._  

</Paragraph>

</Division>
<Division number="80" book="ja1" bookNo="22" page="82">

<Paragraph number="2191" type="verses" book="ja1">
_“Taramāno imaṃ yānaṃ,_  
_āropetu rathesabha;_  
_Mā me mātā tarantassa,_  
_antarāyakarā ahu”._  

</Paragraph>

</Division>
<Division number="81" book="ja1" bookNo="22" page="82">

<Paragraph number="2192" type="verses" book="ja1">
_“Abhidhāvatha bhaddante,_  
_suññaṃ hessati rammakaṃ;_  
_Yudhañcayo anuññāto,_  
_sabbadattena rājinā._  

</Paragraph>

</Division>
<Division number="82" book="ja1" bookNo="22" page="82">

<Paragraph number="2193" type="verses" book="ja1">
_Yohu seṭṭho sahassassa,_  
_yuvā kañcanasannibho;_  
_Soyaṃ kumāro pabbajito,_  
_kāsāyavasano balī”._  

</Paragraph>

</Division>
<Division number="83" book="ja1" bookNo="22" page="82">

<Paragraph number="2194" type="verses" book="ja1">
_“Ubho kumārā pabbajitā,_  
_yudhañcayo yudhiṭṭhilo;_  
_Pahāya mātāpitaro,_  
_saṅgaṃ chetvāna maccuno”ti._  


</Paragraph>
</Division>

<Paragraph number="2195" type="center" book="ja1">
Yudhañcayajātakaṃ chaṭṭhaṃ.
</Paragraph>
