<Paragraph number="1235" type="verses" book="ja1">
_“Ko nu santamhi pajjote,_  
_aggipariyesanaṃ caraṃ;_  
_Addakkhi ratti khajjotaṃ,_  
_jātavedaṃ amaññatha._  

</Paragraph>

</Division>
<Division number="76" book="ja1" bookNo="22" page="249">

<Paragraph number="1236" type="verses" book="ja1">
_Svassa gomayacuṇṇāni,_  
_abhimatthaṃ tiṇāni ca;_  
_Viparītāya saññāya,_  
_nāsakkhi pajjaletave._  

</Paragraph>

</Division>
<Division number="77" book="ja1" bookNo="22" page="249">

<Paragraph number="1237" type="verses" book="ja1">
_Evampi anupāyena,_  
_atthaṃ na labhate migo;_  
_Visāṇato gavaṃ dohaṃ,_  
_yattha khīraṃ na vindati._  

</Paragraph>

</Division>
<Division number="78" book="ja1" bookNo="22" page="250">

<Paragraph number="1238" type="verses" book="ja1">
_Vividhehi upāyehi,_  
_atthaṃ papponti māṇavā;_  
_Niggahena amittānaṃ,_  
_mittānaṃ paggahena ca._  

</Paragraph>

</Division>
<Division number="79" book="ja1" bookNo="22" page="250">

<Paragraph number="1239" type="verses" book="ja1">
_Senāmokkhapalābhena,_  
_vallabhānaṃ nayena ca;_  
_Jagatiṃ jagatipālā,_  
_āvasanti vasundharan”ti._  


</Paragraph>
</Division>

<Paragraph number="1240" type="center" book="ja1">
Khajjopanakajātakaṃ catutthaṃ.
</Paragraph>
