<Paragraph number="1039" type="verses" book="ja1">
_“Yo ve kāle asampatte,_  
_ativelaṃ pabhāsati;_  
_Evaṃ so nihato seti,_  
_kokilāyiva atrajo._

<Division number="122" book="ja1" bookNo="22" page="271">

<Paragraph number="1040" type="verses" book="ja1">
_Na hi satthaṃ sunisitaṃ,_  
_visaṃ halāhalāmiva;_  
_Evaṃ nikaṭṭhe pāteti,_  
_vācā dubbhāsitā yathā._  

</Paragraph>

</Division>
<Division number="123" book="ja1" bookNo="22" page="271">

<Paragraph number="1041" type="verses" book="ja1">
_Tasmā kāle akāle vā,_  
_vācaṃ rakkheyya paṇḍito;_  
_Nātivelaṃ pabhāseyya,_  
_api attasamamhi vā._  

</Paragraph>

</Division>

<Paragraph number="1042" type="verses" book="ja1">
_Yo ca kāle mitaṃ bhāse,_  
_matipubbo vicakkhaṇo;_  
_Sabbe amitte ādeti,_  
_supaṇṇo uragāmivā”ti._  


</Paragraph>
</Division>
