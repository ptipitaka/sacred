<Paragraph number="2829" type="verses" book="ja1">
_“Sukhumālarūpaṃ disvā,_  
_raṭṭhā vivanamāgataṃ;_  
_Kūṭāgāravarūpetaṃ,_  
_mahāsayanamupāsitaṃ._  

</Paragraph>

</Division>
<Division number="271" book="ja1" bookNo="22" page="307">

<Paragraph number="2830" type="verses" book="ja1">
_Tassa te pemakenāhaṃ,_  
_adāsiṃ vaḍḍhamodanaṃ;_  
_Sālīnaṃ vicitaṃ bhattaṃ,_  
_suciṃ maṃsūpasecanaṃ._  

</Paragraph>

</Division>
<Division number="272" book="ja1" bookNo="22" page="307">

<Paragraph number="2831" type="verses" book="ja1">
_Taṃ tvaṃ bhattaṃ paṭiggayha,_  
_brāhmaṇassa adāsayi;_  
_Attānaṃ anasitvāna,_  
_koyaṃ dhammo namatthu te”._  

</Paragraph>

</Division>
<Division number="273" book="ja1" bookNo="22" page="307">

<Paragraph number="2832" type="verses" book="ja1">
_“Ācariyo brāhmaṇo mayhaṃ,_  
_kiccākiccesu byāvaṭo;_  
_Garu ca āmantanīyo ca,_  
_dātumarahāmi bhojanaṃ”._  

</Paragraph>

</Division>
<Division number="274" book="ja1" bookNo="22" page="307">

<Paragraph number="2833" type="verses" book="ja1">
_“Brāhmaṇaṃ dāni pucchāmi,_  
_gotamaṃ rājapūjitaṃ;_  
_Rājā te bhattaṃ pādāsi,_  
_suciṃ maṃsūpasecanaṃ._  

</Paragraph>

</Division>
<Division number="275" book="ja1" bookNo="22" page="307">

<Paragraph number="2834" type="verses" book="ja1">
_Taṃ tvaṃ bhattaṃ paṭiggayha,_  
_isissa bhojanaṃ adā;_  
_Akhettaññūsi dānassa,_  
_koyaṃ dhammo namatthu te”._  

</Paragraph>

</Division>
<Division number="276" book="ja1" bookNo="22" page="307">

<Paragraph number="2835" type="verses" book="ja1">
_“Bharāmi puttadāre ca,_  
_gharesu gadhito ahaṃ;_  
_Bhuñje mānusake kāme,_  
_anusāsāmi rājino._  

</Paragraph>

</Division>
<Division number="277" book="ja1" bookNo="22" page="307">

<Paragraph number="2836" type="verses" book="ja1">
_Āraññikassa isino,_  
_cirarattaṃ tapassino;_  
_Vuḍḍhassa bhāvitattassa,_  
_dātumarahāmi bhojanaṃ”._  

</Paragraph>

</Division>
<Division number="278" book="ja1" bookNo="22" page="307">

<Paragraph number="2837" type="verses" book="ja1">
_“Isiñca dāni pucchāmi,_  
_kisaṃ dhamanisanthataṃ;_  
_Parūḷhakacchanakhalomaṃ,_  
_paṅkadantaṃ rajassiraṃ._  

</Paragraph>

</Division>
<Division number="279" book="ja1" bookNo="22" page="307">

<Paragraph number="2838" type="verses" book="ja1">
_Eko araññe viharasi,_  
_nāvakaṅkhasi jīvitaṃ;_  
_Bhikkhu kena tayā seyyo,_  
_yassa tvaṃ bhojanaṃ adā”._  

</Paragraph>

</Division>
<Division number="280" book="ja1" bookNo="22" page="307">

<Paragraph number="2839" type="verses" book="ja1">
_“Khaṇantālukalambāni,_  
_bilālitakkalāni ca;_  
_Dhunaṃ sāmākanīvāraṃ,_  
_saṃghāriyaṃ pasāriyaṃ._  

</Paragraph>

</Division>
<Division number="281" book="ja1" bookNo="22" page="307">

<Paragraph number="2840" type="verses" book="ja1">
_Sākaṃ bhisaṃ madhuṃ maṃsaṃ,_  
_Badarāmalakāni ca;_  
_Tāni āharitvā bhuñjāmi,_  
_Atthi me so pariggaho._  

</Paragraph>

</Division>
<Division number="282" book="ja1" bookNo="22" page="308">

<Paragraph number="2841" type="verses" book="ja1">
_Pacanto apacantassa,_  
_amamassa sakiñcano;_  
_Anādānassa sādāno,_  
_dātumarahāmi bhojanaṃ”._  

</Paragraph>

</Division>
<Division number="283" book="ja1" bookNo="22" page="308">

<Paragraph number="2842" type="verses" book="ja1">
_“Bhikkhuñca dāni pucchāmi,_  
_tuṇhīmāsīna subbataṃ;_  
_Isi te bhattaṃ pādāsi,_  
_suciṃ maṃsūpasecanaṃ._  

</Paragraph>

</Division>
<Division number="284" book="ja1" bookNo="22" page="308">

<Paragraph number="2843" type="verses" book="ja1">
_Taṃ tvaṃ bhattaṃ paṭiggayha,_  
_tuṇhī bhuñjasi ekako;_  
_Nāññaṃ kañci nimantesi,_  
_koyaṃ dhammo namatthu te”._  

</Paragraph>

</Division>
<Division number="285" book="ja1" bookNo="22" page="308">

<Paragraph number="2844" type="verses" book="ja1">
_“Na pacāmi na pācemi,_  
_na chindāmi na chedaye;_  
_Taṃ maṃ akiñcanaṃ ñatvā,_  
_sabbapāpehi ārataṃ._  

</Paragraph>

</Division>
<Division number="286" book="ja1" bookNo="22" page="308">

<Paragraph number="2845" type="verses" book="ja1">
_Vāmena bhikkhamādāya,_  
_dakkhiṇena kamaṇḍaluṃ;_  
_Isi me bhattaṃ pādāsi,_  
_suciṃ maṃsūpasecanaṃ._  

</Paragraph>

</Division>
<Division number="287" book="ja1" bookNo="22" page="308">

<Paragraph number="2846" type="verses" book="ja1">
_Ete hi dātumarahanti,_  
_samamā sapariggahā;_  
_Paccanīkamahaṃ maññe,_  
_yo dātāraṃ nimantaye”._  

</Paragraph>

</Division>
<Division number="288" book="ja1" bookNo="22" page="308">

<Paragraph number="2847" type="verses" book="ja1">
_“Atthāya vata me ajja,_  
_idhāgacchi rathesabho;_  
_Sohaṃ ajja pajānāmi,_  
_yattha dinnaṃ mahapphalaṃ._  

</Paragraph>

</Division>
<Division number="289" book="ja1" bookNo="22" page="308">

<Paragraph number="2848" type="verses" book="ja1">
_Raṭṭhesu giddhā rājāno,_  
_Kiccākiccesu brāhmaṇā;_  
_Isī mūlaphale giddhā,_  
_Vippamuttā ca bhikkhavo”ti. (2219)_  


</Paragraph>
</Division>

Tassuddānaṃ

_Bhisajāta mahesi kapotavaro;_  
_Atha mora satacchaka vāṇijako,_  
_Atha rāja sabrāhmaṇa bhikkhaparanti._
