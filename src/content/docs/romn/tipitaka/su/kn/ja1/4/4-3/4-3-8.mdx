<Paragraph number="1022" type="verses" book="ja1">
_“Bahūnaṃ vijjatī bhotī,_  
_tehi me kiṃ bhavissati;_  
_Tasmā etaṃ na socāmi,_  
_piyaṃ sammillahāsiniṃ._  

</Paragraph>

<Division number="110" book="ja1" bookNo="22" page="254">

<Paragraph number="1023" type="verses" book="ja1">
_Taṃ tañce anusoceyya,_  
_yaṃ yaṃ tassa na vijjati;_  
_Attānamanusoceyya,_  
_sadā maccuvasaṃ pataṃ._  

</Paragraph>

</Division>

<Paragraph number="1024" type="verses" book="ja1">
_Na heva ṭhitaṃ nāsīnaṃ,_  
_Na sayānaṃ na paddhaguṃ;_  
_Yāva byāti nimisati,_  
_Tatrāpi rasatī vayo._  

</Paragraph>

</Division>

_vinābhāve asaṃsaye;_  
_Bhūtaṃ sesaṃ dayitabbaṃ,_  
_vītaṃ ananusociyan”ti._  


</Paragraph>
