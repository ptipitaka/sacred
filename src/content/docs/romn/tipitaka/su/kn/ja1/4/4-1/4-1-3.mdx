<Division number="9" book="ja1">

<Paragraph number="893" type="verses" book="ja1">
_“Anuttare kāmaguṇe samiddhe,_  
_Bhutvāna pubbe vasī ekarāja;_  
_So dāni dugge narakamhi khitto,_  
_Nappajjahe vaṇṇabalaṃ purāṇaṃ”._  

</Paragraph>

</Division>
<Division number="10" book="ja1">

<Paragraph number="894" type="verses" book="ja1">
_“Pubbeva khantī ca tapo ca mayhaṃ,_  
_Sampatthitā dubbhisena ahosi;_  
_Taṃ dāni laddhāna kathaṃ nu rāja,_  
_Jahe ahaṃ vaṇṇabalaṃ purāṇaṃ._  

</Paragraph>

</Division>
<Division number="11" book="ja1">

<Paragraph number="895" type="verses" book="ja1">
_Sabbā kirevaṃ pariniṭṭhitāni,_  
_Yasassinaṃ paññavantaṃ visayha;_  
_Yaso ca laddhā purimaṃ uḷāraṃ,_  
_Nappajjahe vaṇṇabalaṃ purāṇaṃ._  

</Paragraph>

</Division>
<Division number="12" book="ja1">

<Paragraph number="896" type="verses" book="ja1">
_Panujja dukkhena sukhaṃ janinda,_  
_Sukhena vā dukkhamasayhasāhi;_  
_Ubhayattha santo abhinibbutattā,_  
_Sukhe ca dukkhe ca bhavanti tulyā”ti._  


</Paragraph>
</Division>

<Paragraph number="897" type="center" book="ja1">
Ekarājajātakaṃ tatiyaṃ.
</Paragraph>
