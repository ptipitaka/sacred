<Paragraph number="3146" type="verses" book="ja1">
_Brahmalokā cavitvāna,_  
_devaputto mahiddhiko;_  
_Rañño putto udapādi,_  
_sabbakāmasamiddhisu._  

</Paragraph>

</Division>
<Division number="285" book="ja1" bookNo="22" page="344">

<Paragraph number="3147" type="verses" book="ja1">
_Kāmā vā kāmasaññā vā,_  
_brahmaloke na vijjati;_  
_Svāssu tāyeva saññāya,_  
_kāmehi vijigucchatha._  

</Paragraph>

</Division>
<Division number="286" book="ja1" bookNo="22" page="344">

<Paragraph number="3148" type="verses" book="ja1">
_Tassa cantepure āsi,_  
_jhānāgāraṃ sumāpitaṃ;_  
_So tattha paṭisallīno,_  
_eko rahasi jhāyatha._  

</Paragraph>

</Division>
<Division number="287" book="ja1" bookNo="22" page="344">

<Paragraph number="3149" type="verses" book="ja1">
_Sa rājā paridevesi,_  
_puttasokena aṭṭito;_  
_“Ekaputto cayaṃ mayhaṃ,_  
_na ca kāmāni bhuñjati._  

</Paragraph>

</Division>
<Division number="288" book="ja1" bookNo="22" page="344">

<Paragraph number="3150" type="verses" book="ja1">
_Ko nu khvettha upāyo so,_  
_ko vā jānāti kiñcanaṃ;_  
_Yo me puttaṃ palobheyya,_  
_yathā kāmāni patthaye”._  

</Paragraph>

</Division>
<Division number="289" book="ja1" bookNo="22" page="344">

<Paragraph number="3151" type="verses" book="ja1">
_Ahu kumārī tattheva,_  
_vaṇṇarūpasamāhitā;_  
_Kusalā naccagītassa,_  
_vādite ca padakkhiṇā._  

</Paragraph>

</Division>
<Division number="290" book="ja1" bookNo="22" page="344">

<Paragraph number="3152" type="verses" book="ja1">
_Sā tattha upasaṅkamma,_  
_rājānaṃ etadabravi;_  
_“Ahaṃ kho naṃ palobheyyaṃ,_  
_sace bhattā bhavissati”._  

</Paragraph>

</Division>
<Division number="291" book="ja1" bookNo="22" page="344">

<Paragraph number="3153" type="verses" book="ja1">
_Taṃ tathāvādiniṃ rājā,_  
_kumāriṃ etadabravi;_  
_“Tvaññeva naṃ palobhehi,_  
_tava bhattā bhavissati”._  

</Paragraph>

</Division>
<Division number="292" book="ja1" bookNo="22" page="345">

<Paragraph number="3154" type="verses" book="ja1">
_Sā ca antepuraṃ gantvā,_  
_bahuṃ kāmupasaṃhitaṃ;_  
_Hadayaṅgamā pemanīyā,_  
_citrā gāthā abhāsatha._  

</Paragraph>

</Division>
<Division number="293" book="ja1" bookNo="22" page="345">

<Paragraph number="3155" type="verses" book="ja1">
_Tassā ca gāyamānāya,_  
_saddaṃ sutvāna nāriyā;_  
_Kāmacchandassa uppajji,_  
_janaṃ so paripucchatha._  

</Paragraph>

</Division>
<Division number="294" book="ja1" bookNo="22" page="345">

<Paragraph number="3156" type="verses" book="ja1">
_“Kasseso saddo ko vā so,_  
_bhaṇati uccāvacaṃ bahuṃ;_  
_Hadayaṅgamaṃ pemanīyaṃ,_  
_aho kaṇṇasukhaṃ mama”._  

</Paragraph>

</Division>
<Division number="295" book="ja1" bookNo="22" page="345">

<Paragraph number="3157" type="verses" book="ja1">
_“Esā kho pamadā deva,_  
_khiḍḍā esā anappikā;_  
_Sace tvaṃ kāme bhuñjeyya,_  
_bhiyyo bhiyyo chādeyyu taṃ”._  

</Paragraph>

</Division>
<Division number="296" book="ja1" bookNo="22" page="345">

<Paragraph number="3158" type="verses" book="ja1">
_“Iṅgha āgacchatorena,_  
_avidūramhi gāyatu;_  
_Assamassa samīpamhi,_  
_santike mayhaṃ gāyatu”._  

</Paragraph>

</Division>
<Division number="297" book="ja1" bookNo="22" page="345">

<Paragraph number="3159" type="verses" book="ja1">
_Tirokuṭṭamhi gāyitvā,_  
_jhānāgāramhi pāvisi;_  
_Bandhi naṃ anupubbena,_  
_āraññamiva kuñjaraṃ._  

</Paragraph>

</Division>
<Division number="298" book="ja1" bookNo="22" page="345">

<Paragraph number="3160" type="verses" book="ja1">
_Tassa kāmarasaṃ ñatvā,_  
_issādhammo ajāyatha;_  
_“Ahameva kāme bhuñjeyyaṃ,_  
_mā añño puriso ahu”._  

</Paragraph>

</Division>
<Division number="299" book="ja1" bookNo="22" page="345">

<Paragraph number="3161" type="verses" book="ja1">
_Tato asiṃ gahetvāna,_  
_purise hantuṃ upakkami;_  
_“Ahameveko bhuñjissaṃ,_  
_mā añño puriso siyā”._  

</Paragraph>

</Division>
<Division number="300" book="ja1" bookNo="22" page="345">

<Paragraph number="3162" type="verses" book="ja1">
_Tato jānapadā sabbe,_  
_vikkandiṃsu samāgatā;_  
_“Putto tyāyaṃ mahārāja,_  
_janaṃ heṭhetyadūsakaṃ”._  

</Paragraph>

</Division>
<Division number="301" book="ja1" bookNo="22" page="345">

<Paragraph number="3163" type="verses" book="ja1">
_Tañca rājā vivāhesi,_  
_samhā raṭṭhā ca khattiyo;_  
_“Yāvatā vijitaṃ mayhaṃ,_  
_na te vatthabba tāvade”._  

</Paragraph>

</Division>
<Division number="302" book="ja1" bookNo="22" page="345">

<Paragraph number="3164" type="verses" book="ja1">
_Tato so bhariyamādāya,_  
_samuddaṃ upasaṅkami;_  
_Paṇṇasālaṃ karitvāna,_  
_vanamuñchāya pāvisi._  

</Paragraph>

</Division>
<Division number="303" book="ja1" bookNo="22" page="345">

<Paragraph number="3165" type="verses" book="ja1">
_Athettha isi māgacchi,_  
_samuddaṃ uparūpari;_  
_So tassa gehaṃ pāvekkhi,_  
_bhattakāle upaṭṭhite._  

</Paragraph>

</Division>
<Division number="304" book="ja1" bookNo="22" page="346">

<Paragraph number="3166" type="verses" book="ja1">
_Tañca bhariyā palobhesi,_  
_passa yāva sudāruṇaṃ;_  
_Cuto so brahmacariyamhā,_  
_iddhiyā parihāyatha._  

</Paragraph>

</Division>
<Division number="305" book="ja1" bookNo="22" page="346">

<Paragraph number="3167" type="verses" book="ja1">
_Rājaputto ca uñchāto,_  
_vanamūlaphalaṃ bahuṃ;_  
_Sāyaṃ kājena ādāya,_  
_assamaṃ upasaṅkami._  

</Paragraph>

</Division>
<Division number="306" book="ja1" bookNo="22" page="346">

<Paragraph number="3168" type="verses" book="ja1">
_Isī ca khattiyaṃ disvā,_  
_samuddaṃ upasaṅkami;_  
_“Vehāyasaṃ gamissan”ti,_  
_sīdate so mahaṇṇave._  

</Paragraph>

</Division>
<Division number="307" book="ja1" bookNo="22" page="346">

<Paragraph number="3169" type="verses" book="ja1">
_Khattiyo ca isiṃ disvā,_  
_sīdamānaṃ mahaṇṇave;_  
_Tasseva anukampāya,_  
_imā gāthā abhāsatha._  

</Paragraph>

</Division>
<Division number="308" book="ja1" bookNo="22" page="346">

<Paragraph number="3170" type="verses" book="ja1">
_“Abhijjamāne vārismiṃ,_  
_sayaṃ āgamma iddhiyā;_  
_Missībhāvitthiyā gantvā,_  
_saṃsīdasi mahaṇṇave._  

</Paragraph>

</Division>
<Division number="309" book="ja1" bookNo="22" page="346">

<Paragraph number="3171" type="verses" book="ja1">
_Āvaṭṭanī mahāmāyā,_  
_brahmacariyavikopanā;_  
_Sīdanti naṃ viditvāna,_  
_ārakā parivajjaye._  

</Paragraph>

</Division>
<Division number="310" book="ja1" bookNo="22" page="346">

<Paragraph number="3172" type="verses" book="ja1">
_Analā mudusambhāsā,_  
_duppūrā tā nadīsamā;_  
_Sīdanti naṃ viditvāna,_  
_ārakā parivajjaye._  

</Paragraph>

</Division>
<Division number="311" book="ja1" bookNo="22" page="346">

<Paragraph number="3173" type="verses" book="ja1">
_Yaṃ etā upasevanti,_  
_chandasā vā dhanena vā;_  
_Jātavedova saṃ ṭhānaṃ,_  
_khippaṃ anudahanti naṃ”._  

</Paragraph>

</Division>
<Division number="312" book="ja1" bookNo="22" page="346">

<Paragraph number="3174" type="verses" book="ja1">
_Khattiyassa vaco sutvā,_  
_isissa nibbidā ahu;_  
_Laddhā porāṇakaṃ maggaṃ,_  
_gacchate so vihāyasaṃ._  

</Paragraph>

</Division>
<Division number="313" book="ja1" bookNo="22" page="346">

<Paragraph number="3175" type="verses" book="ja1">
_Khattiyo ca isiṃ disvā,_  
_gacchamānaṃ vihāyasaṃ;_  
_Saṃvegaṃ alabhī dhīro,_  
_pabbajjaṃ samarocayi._  

</Paragraph>

</Division>
<Division number="314" book="ja1" bookNo="22" page="346">

<Paragraph number="3176" type="verses" book="ja1">
_Tato so pabbajitvāna,_  
_kāmarāgaṃ virājayi;_  
_Kāmarāgaṃ virājetvā,_  
_brahmalokūpago ahūti._  


</Paragraph>
</Division>

<Paragraph number="3177" type="center" book="ja1">
Mahāpalobhanajātakaṃ ekādasamaṃ.
</Paragraph>
