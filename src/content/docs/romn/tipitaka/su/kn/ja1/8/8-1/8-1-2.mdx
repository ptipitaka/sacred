<Paragraph number="1683" type="verses" book="ja1">
_“Idaṃ pure ninnamāhu,_  
_bahumacchaṃ mahodakaṃ;_  
_Āvāso bakarājassa,_  
_pettikaṃ bhavanaṃ mama;_  
_Tyajja bhekena yāpema,_  
_okaṃ na vijahāmase”._  

</Paragraph>

</Division>
<Division number="11" book="ja1" bookNo="22" page="310">

<Paragraph number="1684" type="verses" book="ja1">
_“Ko dutiyaṃ asīlissa,_  
_bandharassakkhi bhecchati;_  
_Ko me putte kulāvakaṃ,_  
_mañca sotthiṃ karissati”._  

</Paragraph>

</Division>
<Division number="12" book="ja1" bookNo="22" page="310">

<Paragraph number="1685" type="verses" book="ja1">
_“Sabbā parikkhayā pheggu,_  
_yāva tassā gatī ahu;_  
_Khīṇabhakkho mahārāja,_  
_sāre na ramatī ghuṇo”._  

</Paragraph>

</Division>
<Division number="13" book="ja1" bookNo="22" page="310">

<Paragraph number="1686" type="verses" book="ja1">
_“Sā nūnāhaṃ ito gantvā,_  
_rañño muttā nivesanā;_  
_Attānaṃ ramayissāmi,_  
_dumasākhaniketinī”._  

</Paragraph>

</Division>
<Division number="14" book="ja1" bookNo="22" page="311">

<Paragraph number="1687" type="verses" book="ja1">
_“So nūnāhaṃ ito gantvā,_  
_rañño mutto nivesanā;_  
_Aggodakāni pissāmi,_  
_yūthassa purato vajaṃ”._  

</Paragraph>

</Division>
<Division number="15" book="ja1" bookNo="22" page="311">

<Paragraph number="1688" type="verses" book="ja1">
_“Taṃ maṃ kāmehi sammattaṃ,_  
_rattaṃ kāmesu mucchitaṃ;_  
_Ānayī bharato luddo,_  
_bāhiko bhaddamatthu te”._  

</Paragraph>

</Division>
<Division number="16" book="ja1" bookNo="22" page="311">

<Paragraph number="1689" type="verses" book="ja1">
_“Andhakāratimisāyaṃ,_  
_tuṅge uparipabbate;_  
_Sā maṃ saṇhena mudunā,_  
_mā pādaṃ khali yasmani”._  

</Paragraph>

</Division>
<Division number="17" book="ja1" bookNo="22" page="311">

<Paragraph number="1690" type="verses" book="ja1">
_“Asaṃsayaṃ jātikhayantadassī,_  
_Na gabbhaseyyaṃ punarāvajissaṃ;_  
_Ayamantimā pacchimā gabbhaseyyā,_  
_Khīṇo me saṃsāro punabbhavāyā”ti._  


</Paragraph>
</Division>

<Paragraph number="1691" type="center" book="ja1">
Aṭṭhasaddajātakaṃ dutiyaṃ.
</Paragraph>
