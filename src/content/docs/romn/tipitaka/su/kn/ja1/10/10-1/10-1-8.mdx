<Division number="82" book="ja1">

<Paragraph number="1997" type="verses" book="ja1">
_“Na takkalā santi na āluvāni,_  
_Na biḷāliyo na kaḷambāni tāta;_  
_Eko araññamhi susānamajjhe,_  
_Kimatthiko tāta khaṇāsi kāsuṃ”._  

</Paragraph>

</Division>
<Division number="83" book="ja1">

<Paragraph number="1998" type="verses" book="ja1">
_“Pitāmaho tāta sudubbalo te,_  
_Anekabyādhīhi dukhena phuṭṭho;_  
_Tamajjahaṃ nikhaṇissāmi sobbhe,_  
_Na hissa taṃ jīvitaṃ rocayāmi”._  

</Paragraph>

</Division>
<Division number="84" book="ja1">

<Paragraph number="1999" type="verses" book="ja1">
_“Saṅkappametaṃ paṭiladdha pāpakaṃ,_  
_Accāhitaṃ kamma karosi luddaṃ;_  
_Mayāpi tāta paṭilacchase tuvaṃ,_  
_Etādisaṃ kamma jarūpanīto;_  
_Taṃ kullavattaṃ anuvattamāno,_  
_Ahampi taṃ nikhaṇissāmi sobbhe”._  

</Paragraph>

</Division>
<Division number="85" book="ja1">

<Paragraph number="2000" type="verses" book="ja1">
_“Pharusāhi vācāhi pakubbamāno,_  
_Āsajja maṃ tvaṃ vadase kumāra;_  
_Putto mamaṃ orasako samāno,_  
_Ahitānukampī mama tvaṃsi putta”._  

</Paragraph>

</Division>
<Division number="86" book="ja1">

<Paragraph number="2001" type="verses" book="ja1">
_“Na tāhaṃ tāta ahitānukampī,_  
_Hitānukampī te ahampi tāta;_  
_Pāpañca taṃ kamma pakubbamānaṃ,_  
_Arahāmi no vārayituṃ tato”._  

</Paragraph>

</Division>
<Division number="87" book="ja1">

<Paragraph number="2002" type="verses" book="ja1">
_“Yo mātaraṃ vā pitaraṃ saviṭṭha,_  
_Adūsake hiṃsati pāpadhammo;_  
_Kāyassa bhedā abhisamparāyaṃ,_  
_Asaṃsayaṃ so nirayaṃ upeti._  

</Paragraph>

</Division>
<Division number="88" book="ja1">

<Paragraph number="2003" type="verses" book="ja1">
_Yo mātaraṃ vā pitaraṃ saviṭṭha,_  
_Annena pānena upaṭṭhahāti;_  
_Kāyassa bhedā abhisamparāyaṃ,_  
_Asaṃsayaṃ so sugatiṃ upeti._  

</Paragraph>

</Division>
<Division number="89" book="ja1">

<Paragraph number="2004" type="verses" book="ja1">
_Na me tvaṃ putta ahitānukampī,_  
_Hitānukampī me tvaṃsi putta;_  
_Ahañca taṃ mātarā vuccamāno,_  
_Etādisaṃ kamma karomi luddaṃ”._  

</Paragraph>

</Division>
<Division number="90" book="ja1">

<Paragraph number="2005" type="verses" book="ja1">
_“Yā te sā bhariyā anariyarūpā,_  
_Mātā mamesā sakiyā janetti;_  
_Niddhāpaye tañca sakā agārā,_  
_Aññampi te sā dukhamāvaheyya._  

</Paragraph>

</Division>
<Division number="91" book="ja1">

<Paragraph number="2006" type="verses" book="ja1">
_Yā te sā bhariyā anariyarūpā,_  
_Mātā mamesā sakiyā janetti;_  
_Dantā kareṇūva vasūpanītā,_  
_Sā pāpadhammā punarāvajātū”ti._  


</Paragraph>
</Division>

<Paragraph number="2007" type="center" book="ja1">
Takkalajātakaṃ aṭṭhamaṃ.
</Paragraph>
