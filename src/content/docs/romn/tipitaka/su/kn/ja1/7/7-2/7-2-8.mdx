<Paragraph number="1631" type="verses" book="ja1">
_“Rājā apucchi vidhuraṃ,_  
_dhammakāmo yudhiṭṭhilo;_  
_Api brāhmaṇa jānāsi,_  
_ko eko bahu socati”._  

</Paragraph>

</Division>
<Division number="129" book="ja1" bookNo="22" page="375">

<Paragraph number="1632" type="verses" book="ja1">
_“Brāhmaṇo ajayūthena,_  
_pahūtedho vane vasaṃ;_  
_Dhūmaṃ akāsi vāseṭṭho,_  
_rattindivamatandito._  

</Paragraph>

</Division>
<Division number="130" book="ja1" bookNo="22" page="375">

<Paragraph number="1633" type="verses" book="ja1">
_Tassa taṃ dhūmagandhena,_  
_sarabhā makasaḍḍitā;_  
_Vassāvāsaṃ upāgacchuṃ,_  
_dhūmakārissa santike._  

</Paragraph>

</Division>
<Division number="131" book="ja1" bookNo="22" page="375">

<Paragraph number="1634" type="verses" book="ja1">
_Sarabhesu manaṃ katvā,_  
_ajā so nāvabujjhatha;_  
_Āgacchantī vajantī vā,_  
_tassa tā vinasuṃ ajā._  

</Paragraph>

</Division>
<Division number="132" book="ja1" bookNo="22" page="376">

<Paragraph number="1635" type="verses" book="ja1">
_Sarabhā sarade kāle,_  
_pahīnamakase vane;_  
_Pāvisuṃ giriduggāni,_  
_nadīnaṃ pabhavāni ca._  

</Paragraph>

</Division>
<Division number="133" book="ja1" bookNo="22" page="376">

<Paragraph number="1636" type="verses" book="ja1">
_Sarabhe ca gate disvā,_  
_ajā ca vibhavaṃ gatā;_  
_Kiso ca vivaṇṇo cāsi,_  
_paṇḍurogī ca brāhmaṇo._  

</Paragraph>

</Division>
<Division number="134" book="ja1" bookNo="22" page="376">

<Paragraph number="1637" type="verses" book="ja1">
_Evaṃ yo saṃ niraṃkatvā,_  
_āgantuṃ kurute piyaṃ;_  
_So eko bahu socati,_  
_dhūmakārīva brāhmaṇo”ti._  


</Paragraph>
</Division>

<Paragraph number="1638" type="center" book="ja1">
Dhūmakārijātakaṃ aṭṭhamaṃ.
</Paragraph>
