<Division number="34" book="ja1">

<Paragraph number="1184" type="verses" book="ja1">
_“Eko araññe girikandarāyaṃ,_  
_Paggayha paggayha silaṃ pavecchasi;_  
_Punappunaṃ santaramānarūpo,_  
_Koraṇḍiya ko nu tava yidhattho”._  

</Paragraph>

</Division>
<Division number="35" book="ja1">

<Paragraph number="1185" type="verses" book="ja1">
_“Ahañhimaṃ sāgara sevitantaṃ,_  
_Samaṃ karissāmi yathāpi pāṇi;_  
_Vikiriya sānūni ca pabbatāni ca,_  
_Tasmā silaṃ dariyā pakkhipāmi”._  

</Paragraph>

</Division>
<Division number="36" book="ja1">

<Paragraph number="1186" type="verses" book="ja1">
_“Nayimaṃ mahiṃ arahati pāṇikappaṃ,_  
_Samaṃ manusso karaṇāya meko;_  
_Maññāmimaññeva dariṃ jigīsaṃ,_  
_Koraṇḍiya hāhasi jīvalokaṃ”._  

</Paragraph>

</Division>
<Division number="37" book="ja1">

<Paragraph number="1187" type="verses" book="ja1">
_“Sace ahaṃ bhūtadharaṃ na sakkā,_  
_Samaṃ manusso karaṇāya meko;_  
_Evameva tvaṃ brahme ime manusse,_  
_Nānādiṭṭhike nānayissasi te”._  

</Paragraph>

</Division>
<Division number="38" book="ja1">

<Paragraph number="1188" type="verses" book="ja1">
_“Saṃkhittarūpena bhavaṃ mamatthaṃ,_  
_Akkhāsi koraṇḍiya evametaṃ;_  
_Yathā na sakkā pathavī samāyaṃ,_  
_Kattuṃ manussena tathā manussā”ti._  


</Paragraph>
</Division>

<Paragraph number="1189" type="center" book="ja1">
Koraṇḍiyajātakaṃ chaṭṭhaṃ.
</Paragraph>
