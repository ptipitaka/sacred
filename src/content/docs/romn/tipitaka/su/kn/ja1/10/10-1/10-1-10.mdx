<Division number="104" book="ja1">

<Paragraph number="2021" type="verses" book="ja1">
_“Nāsmase katapāpamhi,_  
_nāsmase alikavādine;_  
_Nāsmase attatthapaññamhi,_  
_atisantepi nāsmase._  

</Paragraph>

</Division>
<Division number="105" book="ja1">

<Paragraph number="2022" type="verses" book="ja1">
_Bhavanti heke purisā,_  
_gopipāsikajātikā;_  
_Ghasanti maññe mittāni,_  
_vācāya na ca kammunā._  

</Paragraph>

</Division>
<Division number="106" book="ja1">

<Paragraph number="2023" type="verses" book="ja1">
_Sukkhañjalipaggahitā,_  
_vācāya paliguṇṭhitā;_  
_Manussapheggū nāsīde,_  
_yasmiṃ natthi kataññutā._  

</Paragraph>

</Division>
<Division number="107" book="ja1">

<Paragraph number="2024" type="verses" book="ja1">
_Na hi aññaññacittānaṃ,_  
_itthīnaṃ purisāna vā;_  
_Nānāvikatvā saṃsaggaṃ,_  
_tādisampi ca nāsmase._  

</Paragraph>

</Division>
<Division number="108" book="ja1">

<Paragraph number="2025" type="verses" book="ja1">
_Anariyakammamokkantaṃ,_  
_athetaṃ sabbaghātinaṃ;_  
_Nisitaṃva paṭicchannaṃ,_  
_tādisampi ca nāsmase._  

</Paragraph>

</Division>
<Division number="109" book="ja1">

<Paragraph number="2026" type="verses" book="ja1">
_Mittarūpenidhekacce,_  
_sākhalyena acetasā;_  
_Vividhehi upāyanti,_  
_tādisampi ca nāsmase._  

</Paragraph>

</Division>
<Division number="110" book="ja1">

<Paragraph number="2027" type="verses" book="ja1">
_Āmisaṃ vā dhanaṃ vāpi,_  
_yattha passati tādiso;_  
_Dubbhiṃ karoti dummedho,_  
_tañca hantvāna gacchati”._  

</Paragraph>

</Division>
<Division number="111" book="ja1">

<Paragraph number="2028" type="verses" book="ja1">
_“Mittarūpena bahavo,_  
_channā sevanti sattavo;_  
_Jahe kāpurise hete,_  
_kukkuṭo viya senakaṃ._  

</Paragraph>

</Division>
<Division number="112" book="ja1">

<Paragraph number="2029" type="verses" book="ja1">
_Yo ca uppatitaṃ atthaṃ,_  
_na khippamanubujjhati;_  
_Amittavasamanveti,_  
_pacchā ca anutappati._  

</Paragraph>

</Division>
<Division number="113" book="ja1">

<Paragraph number="2030" type="verses" book="ja1">
_Yo ca uppatitaṃ atthaṃ,_  
_khippameva nibodhati;_  
_Muccate sattusambādhā,_  
_kukkuṭo viya senakā._  

</Paragraph>

</Division>
<Division number="114" book="ja1">

<Paragraph number="2031" type="verses" book="ja1">
_Taṃ tādisaṃ kūṭamivoḍḍitaṃ vane,_  
_Adhammikaṃ niccavidhaṃsakārinaṃ;_  
_Ārā vivajjeyya naro vicakkhaṇo,_  
_Senaṃ yathā kukkuṭo vaṃsakānane”ti._  


</Paragraph>
</Division>

<Paragraph number="2032" type="center" book="ja1">
Kukkuṭajātakaṃ dasamaṃ.
</Paragraph>
