<Paragraph number="1079" type="verses" book="ja1">
_“Adassanena morassa,_  
_sikhino mañjubhāṇino;_  
_Kākaṃ tattha apūjesuṃ,_  
_maṃsena ca phalena ca._  

</Paragraph>

</Division>
<Division number="154" book="ja1" bookNo="22" page="327">

<Paragraph number="1080" type="verses" book="ja1">
_Yadā ca sarasampanno,_  
_moro bāverumāgamā;_  
_Atha lābho ca sakkāro,_  
_vāyasassa ahāyatha._  

</Paragraph>

</Division>
<Division number="155" book="ja1" bookNo="22" page="327">

<Paragraph number="1081" type="verses" book="ja1">
_Yāva nuppajjatī buddho,_  
_dhammarājā pabhaṅkaro;_  
_Tāva aññe apūjesuṃ,_  
_puthū samaṇabrāhmaṇe._  

</Paragraph>

</Division>
<Division number="156" book="ja1" bookNo="22" page="327">

<Paragraph number="1082" type="verses" book="ja1">
_Yadā ca sarasampanno,_  
_buddho dhammaṃ adesayi;_  
_Atha lābho ca sakkāro,_  
_titthiyānaṃ ahāyathā”ti._  


</Paragraph>
</Division>

<Paragraph number="1083" type="center" book="ja1">
Bāverujātakaṃ navamaṃ.
</Paragraph>
