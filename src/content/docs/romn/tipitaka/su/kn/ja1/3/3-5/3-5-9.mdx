<Paragraph number="870" type="verses" book="ja1">
_“Pure tuvaṃ sīlavataṃ sakāse,_  
_Okkantikaṃ kīḷasi assamamhi;_  
_Karohare makkaṭiyāni makkaṭa,_  
_Na taṃ mayaṃ sīlavataṃ ramāma”._  

</Paragraph>

</Division>
<Division number="146" book="ja1" bookNo="22" page="294">

<Paragraph number="871" type="verses" book="ja1">
_“Sutā hi mayhaṃ paramā visuddhi,_  
_Komāraputtassa bahussutassa;_  
_Mā dāni maṃ maññi tuvaṃ yathā pure,_  
_Jhānānuyutto viharāmi āvuso”._  

</Paragraph>

</Division>
<Division number="147" book="ja1" bookNo="22" page="294">

<Paragraph number="872" type="verses" book="ja1">
_“Sacepi selasmi vapeyya bījaṃ,_  
_Devo ca vasse na hi taṃ virūḷhe;_  
_Sutā hi te sā paramā visuddhi,_  
_Ārā tuvaṃ makkaṭa jhānabhūmiyā”ti._  


</Paragraph>
</Division>

<Paragraph number="873" type="center" book="ja1">
Komāraputtajātakaṃ navamaṃ.
</Paragraph>
