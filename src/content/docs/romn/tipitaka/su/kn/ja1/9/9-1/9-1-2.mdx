<Division number="10" book="ja1">

<Paragraph number="1791" type="verses" book="ja1">
_“Puthusaddo samajano,_  
_na bālo koci maññatha;_  
_Saṃghasmiṃ bhijjamānasmiṃ,_  
_nāññaṃ bhiyyo amaññaruṃ._  

</Paragraph>

</Division>
<Division number="11" book="ja1">

<Paragraph number="1792" type="verses" book="ja1">
_Parimuṭṭhā paṇḍitābhāsā,_  
_vācāgocarabhāṇino;_  
_Yāvicchanti mukhāyāmaṃ,_  
_yena nītā na taṃ vidū._  

</Paragraph>

</Division>
<Division number="12" book="ja1">

<Paragraph number="1793" type="verses" book="ja1">
_Akkocchi maṃ avadhi maṃ,_  
_ajini maṃ ahāsi me;_  
_Ye ca taṃ upanayhanti,_  
_veraṃ tesaṃ na sammati._  

</Paragraph>

</Division>
<Division number="13" book="ja1">

<Paragraph number="1794" type="verses" book="ja1">
_Akkocchi maṃ avadhi maṃ,_  
_ajini maṃ ahāsi me;_  
_Ye ca taṃ nupanayhanti,_  
_veraṃ tesūpasammati._  

</Paragraph>

</Division>
<Division number="14" book="ja1">

<Paragraph number="1795" type="verses" book="ja1">
_Na hi verena verāni,_  
_sammantīdha kudācanaṃ;_  
_Averena ca sammanti,_  
_esa dhammo sanantano._  

</Paragraph>

</Division>
<Division number="15" book="ja1">

<Paragraph number="1796" type="verses" book="ja1">
_Pare ca na vijānanti,_  
_mayamettha yamāmase;_  
_Ye ca tattha vijānanti,_  
_tato sammanti medhagā._  

</Paragraph>

</Division>
<Division number="16" book="ja1">

<Paragraph number="1797" type="verses" book="ja1">
_Aṭṭhicchinnā pāṇaharā,_  
_gavāssa dhanahārino;_  
_Raṭṭhaṃ vilumpamānānaṃ,_  
_tesampi hoti saṅgati;_  
_Kasmā tumhāka no siyā._  

</Paragraph>

</Division>
<Division number="17" book="ja1">

<Paragraph number="1798" type="verses" book="ja1">
_Sace labhetha nipakaṃ sahāyaṃ,_  
_Saddhiṃ caraṃ sādhuvihāridhīraṃ;_  
_Abhibhuyya sabbāni parissayāni,_  
_Careyya tenattamano satīmā._  

</Paragraph>

</Division>
<Division number="18" book="ja1">

<Paragraph number="1799" type="verses" book="ja1">
_No ce labhetha nipakaṃ sahāyaṃ,_  
_Saddhiṃ caraṃ sādhuvihāridhīraṃ;_  
_Rājāva raṭṭhaṃ vijitaṃ pahāya,_  
_Eko care mātaṅgaraññeva nāgo._  

</Paragraph>

</Division>
<Division number="19" book="ja1">

<Paragraph number="1800" type="verses" book="ja1">
_Ekassa caritaṃ seyyo,_  
_Natthi bāle sahāyatā;_  
_Eko care na pāpāni kayirā,_  
_Appossukko mātaṅgaraññeva nāgo”ti._  


</Paragraph>
</Division>

<Paragraph number="1801" type="center" book="ja1">
Kosambiyajātakaṃ dutiyaṃ.
</Paragraph>
