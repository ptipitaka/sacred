<Paragraph number="935" type="verses" book="ja1">
_“Uṭṭhehi cora kiṃ sesi,_  
_ko attho supanena te;_  
_Mā taṃ gahesuṃ rājāno,_  
_gāme kibbisakārakaṃ”._  

</Paragraph>

</Division>
<Division number="42" book="ja1" bookNo="22" page="231">

<Paragraph number="936" type="verses" book="ja1">
_“Yannu coraṃ gahessanti,_  
_gāme kibbisakārakaṃ;_  
_Kiṃ tattha pucimandassa,_  
_vane jātassa tiṭṭhato”._  

</Paragraph>

</Division>
<Division number="43" book="ja1" bookNo="22" page="232">

<Paragraph number="937" type="verses" book="ja1">
_“Na tvaṃ assattha jānāsi,_  
_mama corassa cantaraṃ;_  
_Coraṃ gahetvā rājāno,_  
_gāme kibbisakārakaṃ;_  
_Appenti nimbasūlasmiṃ,_  
_tasmiṃ me saṅkate mano”._  

</Paragraph>

</Division>
<Division number="44" book="ja1" bookNo="22" page="232">

<Paragraph number="938" type="verses" book="ja1">
_“Saṅkeyya saṅkitabbāni,_  
_rakkheyyānāgataṃ bhayaṃ;_  
_Anāgatabhayā dhīro,_  
_ubho loke avekkhatī”ti._  


</Paragraph>
</Division>

<Paragraph number="939" type="center" book="ja1">
Pucimandajātakaṃ paṭhamaṃ.
</Paragraph>
