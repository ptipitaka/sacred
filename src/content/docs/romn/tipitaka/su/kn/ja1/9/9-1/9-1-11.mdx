<Paragraph number="1886" type="verses" book="ja1">
_“Na kho me ruccati āḷi,_  
_pūtimaṃsassa pekkhanā;_  
_Etādisā sakhārasmā,_  
_ārakā parivajjaye”._  

</Paragraph>

</Division>
<Division number="97" book="ja1" bookNo="22" page="371">

<Paragraph number="1887" type="verses" book="ja1">
_“Ummattikā ayaṃ veṇī,_  
_vaṇṇeti patino sakhiṃ;_  
_Pajjhāyi paṭigacchantiṃ,_  
_āgataṃ meṇḍamātaraṃ”._  

</Paragraph>

</Division>
<Division number="98" book="ja1" bookNo="22" page="371">

<Paragraph number="1888" type="verses" book="ja1">
_“Tvaṃ khosi samma ummatto,_  
_dummedho avicakkhaṇo;_  
_Yo tvaṃ matālayaṃ katvā,_  
_akālena vipekkhasi”._  

</Paragraph>

</Division>
<Division number="99" book="ja1" bookNo="22" page="371">

<Paragraph number="1889" type="verses" book="ja1">
_“Na akāle vipekkheyya,_  
_kāle pekkheyya paṇḍito;_  
_Pūtimaṃsova pajjhāyi,_  
_yo akāle vipekkhati”._  

</Paragraph>

</Division>
<Division number="100" book="ja1" bookNo="22" page="371">

<Paragraph number="1890" type="verses" book="ja1">
_“Piyaṃ kho āḷi me hotu,_  
_puṇṇapattaṃ dadāhi me;_  
_Pati sañjīvito mayhaṃ,_  
_eyyāsi piyapucchikā”._  

</Paragraph>

</Division>
<Division number="101" book="ja1" bookNo="22" page="371">

<Paragraph number="1891" type="verses" book="ja1">
_“Piyaṃ kho āḷi te hotu,_  
_puṇṇapattaṃ dadāmi te;_  
_Mahatā parivārena,_  
_essaṃ kayirāhi bhojanaṃ”._  

</Paragraph>

</Division>
<Division number="102" book="ja1" bookNo="22" page="371">

<Paragraph number="1892" type="verses" book="ja1">
_“Kīdiso tuyhaṃ parivāro,_  
_yesaṃ kāhāmi bhojanaṃ;_  
_Kiṃnāmakā ca te sabbe,_  
_taṃ me akkhāhi pucchitā”._  

</Paragraph>

</Division>
<Division number="103" book="ja1" bookNo="22" page="371">

<Paragraph number="1893" type="verses" book="ja1">
_“Māliyo caturakkho ca,_  
_piṅgiyo atha jambuko;_  
_Ediso mayhaṃ parivāro,_  
_tesaṃ kayirāhi bhojanaṃ”._  

</Paragraph>

</Division>
<Division number="104" book="ja1" bookNo="22" page="371">

<Paragraph number="1894" type="verses" book="ja1">
_“Nikkhantāya agārasmā,_  
_bhaṇḍakampi vinassati;_  
_Ārogyaṃ āḷino vajjaṃ,_  
_idheva vasa māgamā”ti._  


</Paragraph>
</Division>

<Paragraph number="1895" type="center" book="ja1">
Pūtimaṃsajātakaṃ ekādasamaṃ.
</Paragraph>
