<Paragraph number="2304" type="verses" book="ja1">
_“Kaṇho kaṇho ca ghoro ca,_  
_sukkadāṭho pabhāsavā;_  
_Baddho pañcahi rajjūhi,_  
_kiṃ ravi sunakho tava”._  

</Paragraph>

</Division>
<Division number="62" book="ja1" bookNo="22" page="101">

<Paragraph number="2305" type="verses" book="ja1">
_“Nāyaṃ migānamatthāya,_  
_usīnaka bhavissati;_  
_Manussānaṃ anayo hutvā,_  
_tadā kaṇho pamokkhati”._  

</Paragraph>

</Division>
<Division number="63" book="ja1" bookNo="22" page="101">

<Paragraph number="2306" type="verses" book="ja1">
_“Pattahatthā samaṇakā,_  
_muṇḍā saṅghāṭipārutā;_  
_Naṅgalehi kasissanti,_  
_tadā kaṇho pamokkhati._  

</Paragraph>

</Division>
<Division number="64" book="ja1" bookNo="22" page="101">

<Paragraph number="2307" type="verses" book="ja1">
_Tapassiniyo pabbajitā,_  
_muṇḍā saṅghāṭipārutā;_  
_Yadā loke gamissanti,_  
_tadā kaṇho pamokkhati._  

</Paragraph>

</Division>
<Division number="65" book="ja1" bookNo="22" page="101">

<Paragraph number="2308" type="verses" book="ja1">
_Dīghottaroṭṭhā jaṭilā,_  
_paṅkadantā rajassirā;_  
_Iṇaṃ codāya gacchanti,_  
_tadā kaṇho pamokkhati._  

</Paragraph>

</Division>
<Division number="66" book="ja1" bookNo="22" page="101">

<Paragraph number="2309" type="verses" book="ja1">
_Adhicca vede sāvittiṃ,_  
_yaññatantañca brāhmaṇā;_  
_Bhatikāya yajissanti,_  
_tadā kaṇho pamokkhati._  

</Paragraph>

</Division>
<Division number="67" book="ja1" bookNo="22" page="101">

<Paragraph number="2310" type="verses" book="ja1">
_Mātaraṃ pitarañcāpi,_  
_jiṇṇakaṃ gatayobbanaṃ;_  
_Pahū santo na bharanti,_  
_tadā kaṇho pamokkhati._  

</Paragraph>

</Division>
<Division number="68" book="ja1" bookNo="22" page="101">

<Paragraph number="2311" type="verses" book="ja1">
_Mātaraṃ pitarañcāpi,_  
_jiṇṇakaṃ gatayobbanaṃ;_  
_Bālā tumheti vakkhanti,_  
_tadā kaṇho pamokkhati._  

</Paragraph>

</Division>
<Division number="69" book="ja1" bookNo="22" page="102">

<Paragraph number="2312" type="verses" book="ja1">
_Ācariyabhariyaṃ sakhiṃ,_  
_mātulāniṃ pitucchakiṃ;_  
_Yadā loke gamissanti,_  
_tadā kaṇho pamokkhati._  

</Paragraph>

</Division>
<Division number="70" book="ja1" bookNo="22" page="102">

<Paragraph number="2313" type="verses" book="ja1">
_Asicammaṃ gahetvāna,_  
_khaggaṃ paggayha brāhmaṇā;_  
_Panthaghātaṃ karissanti,_  
_tadā kaṇho pamokkhati._  

</Paragraph>

</Division>
<Division number="71" book="ja1" bookNo="22" page="102">

<Paragraph number="2314" type="verses" book="ja1">
_Sukkacchavī vedhaverā,_  
_thūlabāhū apātubhā;_  
_Mittabhedaṃ karissanti,_  
_tadā kaṇho pamokkhati._  

</Paragraph>

</Division>
<Division number="72" book="ja1" bookNo="22" page="102">

<Paragraph number="2315" type="verses" book="ja1">
_Māyāvino nekatikā,_  
_asappurisacintakā;_  
_Yadā loke bhavissanti,_  
_tadā kaṇho pamokkhatī”ti._  


</Paragraph>
</Division>

Mahākaṇhajātakaṃ chaṭṭhaṃ.
</Paragraph>
