<Paragraph number="3101" type="verses" book="ja1">
_“Kā nu vijjurivābhāsi,_  
_Osadhī viya tārakā;_  
_Devatā nusi gandhabbī,_  
_Na taṃ maññāmi mānusiṃ”._  

</Paragraph>

</Division>
<Division number="241" book="ja1">

<Paragraph number="3102" type="verses" book="ja1">
_“Namhi devī na gandhabbī,_  
_na mahārāja mānusī;_  
_Nāgakaññāsmi bhaddante,_  
_atthenamhi idhāgatā”._  

</Paragraph>

</Division>
<Division number="242" book="ja1">

<Paragraph number="3103" type="verses" book="ja1">
_“Vibbhantacittā kupitindriyāsi,_  
_Nettehi te vārigaṇā savanti;_  
_Kiṃ te naṭṭhaṃ kiṃ pana patthayānā,_  
_Idhāgatā nāri tadiṅgha brūhi”._  

</Paragraph>

</Division>
<Division number="243" book="ja1">

<Paragraph number="3104" type="verses" book="ja1">
_“Yamuggatejo uragoti cāhu,_  
_Nāgoti naṃ āhu janā janinda;_  
_Tamaggahī puriso jīvikattho,_  
_Taṃ bandhanā muñca patī mameso”._  

</Paragraph>

</Division>
<Division number="244" book="ja1">

<Paragraph number="3105" type="verses" book="ja1">
_“Kathaṃ nvayaṃ balaviriyūpapanno,_  
_Hatthattamāgacchi vanibbakassa;_  
_Akkhāhi me nāgakaññe tamatthaṃ,_  
_Kathaṃ vijānemu gahītanāgaṃ”._  

</Paragraph>

</Division>
<Division number="245" book="ja1">

<Paragraph number="3106" type="verses" book="ja1">
_“Nagarampi nāgo bhasmaṃ kareyya,_  
_Tathā hi so balaviriyūpapanno;_  
_Dhammañca nāgo apacāyamāno,_  
_Tasmā parakkamma tapo karoti._  

</Paragraph>

</Division>
<Division number="246" book="ja1">

<Paragraph number="3107" type="verses" book="ja1">
_Cātuddasiṃ pañcadasiñca rāja,_  
_Catuppathe sammati nāgarājā;_  
_Tamaggahī puriso jīvikattho,_  
_Taṃ bandhanā muñca patī mameso”._  

</Paragraph>

</Division>
<Division number="247" book="ja1">

<Paragraph number="3108" type="verses" book="ja1">

_āmuttamaṇikuṇḍalā;_  
_Vārigehasayā nārī,_  
_tāpi taṃ saraṇaṃ gatā._  

</Paragraph>

</Division>
<Division number="248" book="ja1">

_Gāmena nikkhena gavaṃ satena;_  
_Ossaṭṭhakāyo urago carātu,_  
_Puññatthiko muñcatu bandhanasmā”._  

</Paragraph>

</Division>
<Division number="249" book="ja1">

<Paragraph number="3110" type="verses" book="ja1">

_Gāmena nikkhena gavaṃ satena;_  
_Ossaṭṭhakāyo urago carātu,_  
_Puññatthiko muñcatu bandhanasmā._  

</Paragraph>

</Division>
<Division number="250" book="ja1">

<Paragraph number="3111" type="verses" book="ja1">
_Dammi nikkhasataṃ ludda,_  
_thūlañca maṇikuṇḍalaṃ;_  
_Catussadañca pallaṅkaṃ,_  
_umāpupphasarinnibhaṃ._  

</Paragraph>

</Division>
<Division number="251" book="ja1">

<Paragraph number="3112" type="verses" book="ja1">
_Dve ca sādisiyo bhariyā,_  
_Usabhañca gavaṃ sataṃ;_  
_Ossaṭṭhakāyo urago carātu,_  
_Puññatthiko muñcatu bandhanasmā”._  

</Paragraph>

</Division>
<Division number="252" book="ja1">

<Paragraph number="3113" type="verses" book="ja1">
_“Vināpi dānā tava vacanaṃ janinda,_  
_Muñcemu naṃ uragaṃ bandhanasmā;_  
_Ossaṭṭhakāyo urago carātu,_  
_Puññatthiko muñcatu bandhanasmā”._  

</Paragraph>

</Division>
<Division number="253" book="ja1">

<Paragraph number="3114" type="verses" book="ja1">
_Mutto campeyyako nāgo,_  
_rājānaṃ etadabravi;_  
_“Namo te kāsirājatthu,_  
_namo te kāsivaḍḍhana;_  
_Añjaliṃ te paggaṇhāmi,_  
_passeyyaṃ me nivesanaṃ”._  

</Paragraph>

</Division>
<Division number="254" book="ja1">

<Paragraph number="3115" type="verses" book="ja1">
_“Addhā hi dubbissasametamāhu,_  
_Yaṃ mānuso vissase amānusamhi;_  
_Sace ca maṃ yācasi etamatthaṃ,_  
_Dakkhemu te nāga nivesanāni”._  

</Paragraph>

</Division>
<Division number="255" book="ja1">

<Paragraph number="3116" type="verses" book="ja1">
_“Sacepi vāto girimāvaheyya,_  
_Cando ca suriyo ca chamā pateyyuṃ;_  
_Sabbā ca najjo paṭisotaṃ vajeyyuṃ,_  
_Na tvevahaṃ rāja musā bhaṇeyyaṃ._  

</Paragraph>

</Division>
<Division number="256" book="ja1">

<Paragraph number="3117" type="verses" book="ja1">
_Nabhaṃ phaleyya udadhīpi susse,_  
_Saṃvaṭṭaye bhūtadharā vasundharā;_  
_Siluccayo meru samūlamuppate,_  
_Na tvevahaṃ rāja musā bhaṇeyyaṃ”._  

</Paragraph>

</Division>
<Division number="257" book="ja1">

<Paragraph number="3118" type="verses" book="ja1">

</Paragraph>

</Division>
<Division number="258" book="ja1">

<Paragraph number="3119" type="verses" book="ja1">
_Tumhe khottha ghoravisā uḷārā,_  
_Mahātejā khippakopī ca hotha;_  
_Maṃkāraṇā bandhanasmā pamutto,_  
_Arahasi no jānituye katāni”._  

</Paragraph>

</Division>
<Division number="259" book="ja1">

<Paragraph number="3120" type="verses" book="ja1">
_“So paccataṃ niraye ghorarūpe,_  
_Mā kāyikaṃ sātamalattha kiñci;_  
_Peḷāya baddho maraṇaṃ upetu,_  
_Yo tādisaṃ kammakataṃ na jāne”._  

</Paragraph>

</Division>
<Division number="260" book="ja1">

<Paragraph number="3121" type="verses" book="ja1">
_“Saccappaṭiññā tava mesa hotu,_  
_Akkodhano hohi anupanāhī;_  
_Sabbañca te nāgakulaṃ supaṇṇā,_  
_Aggiṃva gimhesu vivajjayantu”._  

</Paragraph>

</Division>
<Division number="261" book="ja1">

<Paragraph number="3122" type="verses" book="ja1">
_“Anukampasī nāgakulaṃ janinda,_  
_Mātā yathā suppiyaṃ ekaputtaṃ;_  
_Ahañca te nāgakulena saddhiṃ,_  
_Kāhāmi veyyāvaṭikaṃ uḷāraṃ”._  

</Paragraph>

</Division>
<Division number="262" book="ja1">

<Paragraph number="3123" type="verses" book="ja1">
_“Yojentu ve rājarathe sucitte,_  
_Kambojake assatare sudante;_  
_Nāge ca yojentu suvaṇṇakappane,_  
_Dakkhemu nāgassa nivesanāni”._  

</Paragraph>

</Division>
<Division number="263" book="ja1">

<Paragraph number="3124" type="verses" book="ja1">
_Bherī mudiṅgā paṇavā ca saṅkhā,_  
_Avajjayiṃsu uggasenassa rañño;_  
_Pāyāsi rājā bahusobhamāno,_  
_Purakkhato nārigaṇassa majjhe._  

</Paragraph>

</Division>
<Division number="264" book="ja1">

<Paragraph number="3125" type="verses" book="ja1">
_Suvaṇṇacitakaṃ bhūmiṃ,_  
_addakkhi kāsivaḍḍhano;_  
_Sovaṇṇamayapāsāde,_  
_veḷuriyaphalakatthate._  

</Paragraph>

</Division>
<Division number="265" book="ja1">

<Paragraph number="3126" type="verses" book="ja1">
_Sa rājā pāvisi byamhaṃ,_  
_campeyyassa nivesanaṃ;_  
_Ādiccavaṇṇasannibhaṃ,_  
_kaṃsavijju pabhassaraṃ._  

</Paragraph>

</Division>
<Division number="266" book="ja1">

<Paragraph number="3127" type="verses" book="ja1">
_Nānārukkhehi sañchannaṃ,_  
_nānāgandhasamīritaṃ;_  
_So pāvekkhi kāsirājā,_  
_campeyyassa nivesanaṃ._  

</Paragraph>

</Division>
<Division number="267" book="ja1">

<Paragraph number="3128" type="verses" book="ja1">
_Paviṭṭhasmiṃ kāsiraññe,_  
_campeyyassa nivesanaṃ;_  
_Dibbā tūriyā pavajjiṃsu,_  
_nāgakaññā ca naccisuṃ._  

</Paragraph>

</Division>
<Division number="268" book="ja1">

<Paragraph number="3129" type="verses" book="ja1">
_Taṃ nāgakaññā caritaṃ gaṇena,_  
_Anvāruhī kāsirājā pasanno;_  
_Nisīdi sovaṇṇamayamhi pīṭhe,_  
_Sāpassaye candanasāralitte._  

</Paragraph>

</Division>
<Division number="269" book="ja1">

<Paragraph number="3130" type="verses" book="ja1">
_So tattha bhutvā ca atho ramitvā,_  
_Campeyyakaṃ kāsirājā avoca;_  
_“Vimānaseṭṭhāni imāni tuyhaṃ,_  
_Ādiccavaṇṇāni pabhassarāni;_  
_Netādisaṃ atthi manussaloke,_  
_Kiṃ patthayaṃ nāga tapo karosi._  

</Paragraph>

</Division>
<Division number="270" book="ja1">

<Paragraph number="3131" type="verses" book="ja1">
_Tā kambukāyūradharā suvatthā,_  
_Vaṭṭaṅgulī tambatalūpapannā;_  
_Paggayha pāyenti anomavaṇṇā,_  
_Netādisaṃ atthi manussaloke;_  
_Kiṃ patthayaṃ nāga tapo karosi._  

</Paragraph>

</Division>
<Division number="271" book="ja1">

<Paragraph number="3132" type="verses" book="ja1">
_Najjo ca temā puthulomamacchā,_  
_Āṭā sakuntābhirudā sutitthā;_  
_Netādisaṃ atthi manussaloke,_  
_Kiṃ patthayaṃ nāga tapo karosi._  

</Paragraph>

</Division>
<Division number="272" book="ja1">

<Paragraph number="3133" type="verses" book="ja1">
_Koñcā mayūrā diviyā ca haṃsā,_  
_Vaggussarā kokilā sampatanti;_  
_Netādisaṃ atthi manussaloke,_  
_Kiṃ patthayaṃ nāga tapo karosi._  

</Paragraph>

</Division>
<Division number="273" book="ja1">

<Paragraph number="3134" type="verses" book="ja1">
_Ambā ca sālā tilakā ca jambuyo,_  
_Uddālakā pāṭaliyo ca phullā;_  
_Netādisaṃ atthi manussaloke,_  
_Kiṃ patthayaṃ nāga tapo karosi._  

</Paragraph>

</Division>
<Division number="274" book="ja1">

<Paragraph number="3135" type="verses" book="ja1">
_Imā ca te pokkharañño samantato,_  
_Dibbā ca gandhā satataṃ pavāyanti;_  
_Netādisaṃ atthi manussaloke,_  
_Kiṃ patthayaṃ nāga tapo karosi”._  

</Paragraph>

</Division>
<Division number="275" book="ja1">

<Paragraph number="3136" type="verses" book="ja1">
_“Na puttahetu na dhanassa hetu,_  
_Na āyuno cāpi janinda hetu;_  
_Manussayoniṃ abhipatthayāno,_  
_Tasmā parakkamma tapo karomi”._  

</Paragraph>

</Division>
<Division number="276" book="ja1">

<Paragraph number="3137" type="verses" book="ja1">
_“Tvaṃ lohitakkho vihatantaraṃso,_  
_Alaṅkato kappitakesamassu;_  
_Surosito lohitacandanena,_  
_Gandhabbarājāva disā pabhāsasi._  

</Paragraph>

</Division>
<Division number="277" book="ja1">

<Paragraph number="3138" type="verses" book="ja1">
_Deviddhipattosi mahānubhāvo,_  
_Sabbehi kāmehi samaṅgibhūto;_  
_Pucchāmi taṃ nāgarājetamatthaṃ,_  
_Seyyo ito kena manussaloko”._  

</Paragraph>

</Division>
<Division number="278" book="ja1">

<Paragraph number="3139" type="verses" book="ja1">
_“Janinda nāññatra manussalokā,_  
_Suddhī va saṃvijjati saṃyamo vā;_  
_Ahañca laddhāna manussayoniṃ,_  
_Kāhāmi jātimaraṇassa antaṃ”._  

</Paragraph>

</Division>
<Division number="279" book="ja1">

_Bahussutā ye bahuṭhānacintino;_  
_Nāriyo ca disvāna tuvañca nāga,_  
_Kāhāmi puññāni anappakāni”._  

</Paragraph>

</Division>
<Division number="280" book="ja1">

_Bahussutā ye bahuṭhānacintino;_  
_Nāriyo ca disvāna mamañca rāja,_  
_Karohi puññāni anappakāni”._  

</Paragraph>

</Division>
<Division number="281" book="ja1">

_Rāsī suvaṇṇassa ca tālamattā;_  
_Ito haritvāna suvaṇṇagharāni,_  
_Karassu rūpiyapākāraṃ karontu._  

</Paragraph>

</Division>
<Division number="282" book="ja1">

_Veḷuriyamissāni ito haritvā;_  
_Antepure bhūmiyaṃ santharantu,_  
_Nikkaddamā hehiti nīrajā ca._  

</Paragraph>
