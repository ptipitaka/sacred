<Paragraph number="1044" type="verses" book="ja1">
_“Api hantvā hato brūti,_  
_jetvā jitoti bhāsati;_  
_Pubbamakkhāyino rāja,_  
_aññadatthu na saddahe._  

</Paragraph>

</Division>
<Division number="126" book="ja1" bookNo="22" page="271">

<Paragraph number="1045" type="verses" book="ja1">
_Tasmā paṇḍitajātiyo,_  
_suṇeyya itarassapi;_  
_Ubhinnaṃ vacanaṃ sutvā,_  
_yathā dhammo tathā kare._  

</Paragraph>

</Division>
<Division number="127" book="ja1" bookNo="22" page="271">

<Paragraph number="1046" type="verses" book="ja1">
_Alaso gihī kāmabhogī na sādhu,_  
_Asaññato pabbajito na sādhu;_  
_Rājā na sādhu anisammakārī,_  
_Yo paṇḍito kodhano taṃ na sādhu._  

</Paragraph>

</Division>
<Division number="128" book="ja1" bookNo="22" page="271">

<Paragraph number="1047" type="verses" book="ja1">
_Nisamma khattiyo kayirā,_  
_nānisamma disampati;_  
_Nisammakārino rāja,_  
_yaso kitti ca vaḍḍhatī”ti._  


</Paragraph>
</Division>

<Paragraph number="1048" type="center" book="ja1">
Rathalaṭṭhijātakaṃ dutiyaṃ.
</Paragraph>
