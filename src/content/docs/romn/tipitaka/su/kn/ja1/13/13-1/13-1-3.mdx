<Division number="27" book="ja1">

<Paragraph number="2412" type="verses" book="ja1">
_“Idheva haṃsa nipata,_  
_piyaṃ me tava dassanaṃ;_  
_Issarosi anuppatto,_  
_yamidhatthi pavedaya”._  

</Paragraph>

</Division>
<Division number="28" book="ja1">

<Paragraph number="2413" type="verses" book="ja1">
_“Savanena ekassa piyā bhavanti,_  
_Disvā panekassa viyeti chando;_  
_Disvā ca sutvā ca piyā bhavanti,_  
_Kaccinnu me pīyasi dassanena._  

</Paragraph>

</Division>
<Division number="29" book="ja1">

<Paragraph number="2414" type="verses" book="ja1">
_Savanena piyo mesi,_  
_bhiyyo cāgamma dassanaṃ;_  
_Evaṃ piyadassano me,_  
_vasa haṃsa mamantike”._  

</Paragraph>

</Division>
<Division number="30" book="ja1">

<Paragraph number="2415" type="verses" book="ja1">
_“Vaseyyāma tavāgāre,_  
_niccaṃ sakkatapūjitā;_  
_Matto ca ekadā vajje,_  
_‘haṃsarājaṃ pacantu me’”._  

</Paragraph>

</Division>
<Division number="31" book="ja1">

<Paragraph number="2416" type="verses" book="ja1">
_“Dhiratthu taṃ majjapānaṃ,_  
_yaṃ me piyataraṃ tayā;_  
_Na cāpi majjaṃ pissāmi,_  
_yāva me vacchasī ghare”._  

</Paragraph>

</Division>
<Division number="32" book="ja1">

<Paragraph number="2417" type="verses" book="ja1">
_“Suvijānaṃ siṅgālānaṃ,_  
_sakuṇānañca vassitaṃ;_  
_Manussavassitaṃ rāja,_  
_dubbijānataraṃ tato._  

</Paragraph>

</Division>
<Division number="33" book="ja1">

<Paragraph number="2418" type="verses" book="ja1">
_Api ce maññatī poso,_  
_ñāti mitto sakhāti vā;_  
_Yo pubbe sumano hutvā,_  
_pacchā sampajjate diso._  

</Paragraph>

</Division>
<Division number="34" book="ja1">

<Paragraph number="2419" type="verses" book="ja1">
_Yasmiṃ mano nivisati,_  
_avidūre sahāpi so;_  
_Santikepi hi so dūre,_  
_yasmiṃ nāvisate mano._  

</Paragraph>

</Division>
<Division number="35" book="ja1">

<Paragraph number="2420" type="verses" book="ja1">
_Antopi so hoti pasannacitto,_  
_Pāraṃ samuddassa pasannacitto;_  
_Antopi so hoti paduṭṭhacitto,_  
_Pāraṃ samuddassa paduṭṭhacitto._  

</Paragraph>

</Division>
<Division number="36" book="ja1">

<Paragraph number="2421" type="verses" book="ja1">
_Saṃvasantā vivasanti,_  
_ye disā te rathesabha;_  
_Ārā santo saṃvasanti,_  
_manasā raṭṭhavaḍḍhana._  

</Paragraph>

</Division>
<Division number="37" book="ja1">

<Paragraph number="2422" type="verses" book="ja1">
_Aticiraṃ nivāsena,_  
_piyo bhavati appiyo;_  
_Āmanta kho taṃ gacchāma,_  
_purā te homa appiyā”._  

</Paragraph>

</Division>
<Division number="38" book="ja1">

<Paragraph number="2423" type="verses" book="ja1">
_“Evañce yācamānānaṃ,_  
_Añjaliṃ nāvabujjhasi;_  
_Paricārakānaṃ sataṃ,_  
_Vacanaṃ na karosi no;_  
_Evaṃ taṃ abhiyācāma,_  
_Puna kayirāsi pariyāyaṃ”._  

</Paragraph>

</Division>
<Division number="39" book="ja1">

<Paragraph number="2424" type="verses" book="ja1">
_“Evañce no viharataṃ,_  
_antarāyo na hessati;_  
_Tuyhañcāpi mahārāja,_  
_mayhañca raṭṭhavaḍḍhana;_  
_Appeva nāma passemu,_  
_ahorattānamaccaye”ti._  


</Paragraph>
</Division>

<Paragraph number="2425" type="center" book="ja1">
Javanahaṃsajātakaṃ tatiyaṃ.
</Paragraph>
