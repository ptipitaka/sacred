<Paragraph number="1111" type="verses" book="ja1">
_“Vanaṃ yadaggi dahati,_  
_pāvako kaṇhavattanī;_  
_Kathaṃ karosi pacalaka,_  
_evaṃ dandhaparakkamo”._  

</Paragraph>

<Division number="178" book="ja1">

<Paragraph number="1112" type="verses" book="ja1">
_“Bahūni rukkhachiddāni,_  
_pathabyā vivarāni ca;_  
_Tāni ce nābhisambhoma,_  
_hoti no kālapariyāyo”._  

</Paragraph>

</Division>
<Division number="179" book="ja1">

<Paragraph number="1113" type="verses" book="ja1">
_“Yo dandhakāle tarati,_  
_taraṇīye ca dandhati;_  
_Sukkhapaṇṇaṃva akkamma,_  
_atthaṃ bhañjati attano._  

</Paragraph>

</Division>
<Division number="180" book="ja1">

<Paragraph number="1114" type="verses" book="ja1">
_Yo dandhakāle dandheti,_  
_taraṇīye ca tārayi;_  
_Sasīva rattiṃ vibhajaṃ,_  
_tassattho paripūratī”ti._  


</Paragraph>
</Division>

Gajakumbhajātakaṃ pañcamaṃ.
</Paragraph>
