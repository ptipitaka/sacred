<Division number="94" book="ja1">

<Paragraph number="2339" type="verses" book="ja1">
_“Yesaṃ na kadāci bhūtapubbaṃ,_  
_Sakhyaṃ sattapadampimasmi loke;_  
_Jātā amittā duve sahāyā,_  
_Paṭisandhāya caranti kissa hetu”._  

</Paragraph>

</Division>
<Division number="95" book="ja1">

<Paragraph number="2340" type="verses" book="ja1">
_“Yadi me ajja pātarāsakāle,_  
_Pañhaṃ na sakkuṇeyyātha vattumetaṃ;_  
_Raṭṭhā pabbājayissāmi vo sabbe,_  
_Na hi mattho duppaññajātikehi”._  

</Paragraph>

</Division>
<Division number="96" book="ja1">

<Paragraph number="2341" type="verses" book="ja1">
_“Mahājanasamāgamamhi ghore,_  
_Janakolāhalasaṅgamamhi jāte;_  
_Vikkhittamanā anekacittā,_  
_Pañhaṃ na sakkuṇoma vattumetaṃ._  

</Paragraph>

</Division>
<Division number="97" book="ja1">

<Paragraph number="2342" type="verses" book="ja1">
_Ekaggacittāva ekamekā,_  
_Rahasi gatā atthaṃ nicintayitvā;_  
_Paviveke sammasitvāna dhīrā,_  
_Atha vakkhanti janinda etamatthaṃ”._  

</Paragraph>

</Division>
<Division number="98" book="ja1">

<Paragraph number="2343" type="verses" book="ja1">
_“Uggaputtarājaputtiyānaṃ,_  
_Urabbhassa maṃsaṃ piyaṃ manāpaṃ;_  
_Na sunakhassa te adenti maṃsaṃ,_  
_Atha meṇḍassa suṇena sakhyamassa”._  

</Paragraph>

</Division>
<Division number="99" book="ja1">

<Paragraph number="2344" type="verses" book="ja1">
_“Cammaṃ vihananti eḷakassa,_  
_Assapiṭṭhattharassukhassa hetu;_  
_Na ca te sunakhassa attharanti,_  
_Atha meṇḍassa suṇena sakhyamassa”._  

</Paragraph>

</Division>
<Division number="100" book="ja1">

<Paragraph number="2345" type="verses" book="ja1">
_“Āvellitasiṅgiko hi meṇḍo,_  
_Na ca sunakhassa visāṇakāni atthi;_  
_Tiṇabhakkho maṃsabhojano ca,_  
_Atha meṇḍassa suṇena sakhyamassa”._  

</Paragraph>

</Division>
<Division number="101" book="ja1">

<Paragraph number="2346" type="verses" book="ja1">
_“Tiṇamāsi palāsamāsi meṇḍo,_  
_Na ca sunakho tiṇamāsi no palāsaṃ;_  
_Gaṇheyya suṇo sasaṃ biḷāraṃ,_  
_Atha meṇḍassa suṇena sakhyamassa”._  

</Paragraph>

</Division>
<Division number="102" book="ja1">

<Paragraph number="2347" type="verses" book="ja1">
_“Aṭṭhaḍḍhapado catuppadassa,_  
_Meṇḍo aṭṭhanakho adissamāno;_  
_Chādiyamāharatī ayaṃ imassa,_  
_Maṃsaṃ āharatī ayaṃ amussa._  

</Paragraph>

</Division>
<Division number="103" book="ja1">

<Paragraph number="2348" type="verses" book="ja1">
_Pāsādavaragato videhaseṭṭho,_  
_Vitihāraṃ aññamaññabhojanānaṃ;_  
_Addakkhi kira sakkhikaṃ janindo,_  
_Bubhukkassa puṇṇaṃmukhassa cetaṃ”._  

</Paragraph>

</Division>
<Division number="104" book="ja1">

<Paragraph number="2349" type="verses" book="ja1">
_“Lābhā vata me anapparūpā,_  
_Yassa medisā paṇḍitā kulamhi;_  
_Pañhassa gambhīragataṃ nipuṇamatthaṃ,_  
_Paṭivijjhanti subhāsitena dhīrā._  

</Paragraph>

</Division>
<Division number="105" book="ja1">

<Paragraph number="2350" type="verses" book="ja1">
_Assatarirathañca ekamekaṃ,_  
_Phītaṃ gāmavarañca ekamekaṃ;_  
_Sabbesaṃ vo dammi paṇḍitānaṃ,_  
_Paramappatītamano subhāsitenā”ti._  


</Paragraph>
</Division>

<Paragraph number="2351" type="center" book="ja1">
Meṇḍakapañhajātakaṃ aṭṭhamaṃ.
</Paragraph>
