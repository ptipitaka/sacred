<Paragraph number="1203" type="verses" book="ja1">
_“Vikkama re haripāda,_  
_vikkama re mahāmiga;_  
_Chinda vārattikaṃ pāsaṃ,_  
_nāhaṃ ekā vane rame”._  

</Paragraph>

</Division>
<Division number="51" book="ja1">

<Paragraph number="1204" type="verses" book="ja1">
_“Vikkamāmi na pāremi,_  
_bhūmiṃ sumbhāmi vegasā;_  
_Daḷho vārattiko pāso,_  
_pādaṃ me parikantati”._  

</Paragraph>

</Division>
<Division number="52" book="ja1">

<Paragraph number="1205" type="verses" book="ja1">
_“Attharassu palāsāni,_  
_asiṃ nibbāha luddaka;_  
_Paṭhamaṃ maṃ vadhitvāna,_  
_hana pacchā mahāmigaṃ”._  

</Paragraph>

</Division>
<Division number="53" book="ja1">

<Paragraph number="1206" type="verses" book="ja1">
_“Na me sutaṃ vā diṭṭhaṃ vā,_  
_Bhāsantiṃ mānusiṃ migiṃ;_  
_Tvañca bhadde sukhī hohi,_  
_Eso cāpi mahāmigo”._  

</Paragraph>

</Division>
<Division number="54" book="ja1">

<Paragraph number="1207" type="verses" book="ja1">
_“Evaṃ luddaka nandassu,_  
_saha sabbehi ñātibhi;_  
_Yathāhamajja nandāmi,_  
_muttaṃ disvā mahāmigan”ti._  


</Paragraph>
</Division>

<Paragraph number="1208" type="center" book="ja1">
Suvaṇṇamigajātakaṃ navamaṃ.
</Paragraph>
