<Division number="6" book="ja1">

<Paragraph number="1152" type="verses" book="ja1">
_“Kiṃ nu santaramānova,_  
_lāyitvā haritaṃ tiṇaṃ;_  
_Khāda khādāti lapasi,_  
_gatasattaṃ jaraggavaṃ._  

</Paragraph>

</Division>
<Division number="7" book="ja1">

<Paragraph number="1153" type="verses" book="ja1">
_Na hi annena pānena,_  
_mato goṇo samuṭṭhahe;_  
_Tvañca tucchaṃ vilapasi,_  
_yathā taṃ dummatī tathā”._  

</Paragraph>

</Division>
<Division number="8" book="ja1">

<Paragraph number="1154" type="verses" book="ja1">
_“Tatheva tiṭṭhati sīsaṃ,_  
_hatthapādā ca vāladhi;_  
_Sotā tatheva tiṭṭhanti,_  
_maññe goṇo samuṭṭhahe._  

</Paragraph>

</Division>
<Division number="9" book="ja1">

<Paragraph number="1155" type="verses" book="ja1">
_Nevayyakassa sīsañca,_  
_hatthapādā ca dissare;_  
_Rudaṃ mattikathūpasmiṃ,_  
_nanu tvaññeva dummati”._  

</Paragraph>

</Division>
<Division number="10" book="ja1">

<Paragraph number="1156" type="verses" book="ja1">
_“Ādittaṃ vata maṃ santaṃ,_  
_ghatasittaṃva pāvakaṃ;_  
_Vārinā viya osiñcaṃ,_  
_sabbaṃ nibbāpaye daraṃ._  

</Paragraph>

</Division>
<Division number="11" book="ja1">

<Paragraph number="1157" type="verses" book="ja1">
_Abbahī vata me sallaṃ,_  
_yamāsi hadayassitaṃ;_  
_Yo me sokaparetassa,_  
_pitusokaṃ apānudi._  

</Paragraph>

</Division>
<Division number="12" book="ja1">

<Paragraph number="1158" type="verses" book="ja1">
_Sohaṃ abbūḷhasallosmi,_  
_vītasoko anāvilo;_  
_Na socāmi na rodāmi,_  
_tava sutvāna māṇava._  

</Paragraph>

</Division>
<Division number="13" book="ja1">

<Paragraph number="1159" type="verses" book="ja1">
_Evaṃ karonti sappaññā,_  
_ye honti anukampakā;_  
_Vinivattenti sokamhā,_  
_sujāto pitaraṃ yathā”ti._  


</Paragraph>
</Division>

<Paragraph number="1160" type="center" book="ja1">
Sujātajātakaṃ dutiyaṃ.
</Paragraph>
