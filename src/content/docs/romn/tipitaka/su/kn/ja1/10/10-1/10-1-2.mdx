<Paragraph number="1921" type="verses" book="ja1">

_Yasmiṃ pāpāni kammāni,_  
_sa ve kaṇho sujampati”._  

</Paragraph>

<Paragraph number="1922" type="verses" book="ja1">
_“Etasmiṃ te sulapite,_  
_patirūpe subhāsite;_  
_Varaṃ brāhmaṇa te dammi,_  
_yaṃ kiñci manasicchasi”._  

</Paragraph>

</Division>
<Division number="14" book="ja1">

<Paragraph number="1923" type="verses" book="ja1">
_“Varañce me ado sakka,_  
_sabbabhūtānamissara;_  
_Sunikkodhaṃ suniddosaṃ,_  
_nillobhaṃ vuttimattano;_  
_Nisnehamabhikaṅkhāmi,_  
_ete me caturo vare”._  

</Paragraph>

</Division>
<Division number="15" book="ja1">

<Paragraph number="1924" type="verses" book="ja1">
_“Kiṃ nu kodhe vā dose vā,_  
_lobhe snehe ca brāhmaṇa;_  
_Ādīnavaṃ tvaṃ passasi,_  
_taṃ me akkhāhi pucchito”._  

</Paragraph>

</Division>
<Division number="16" book="ja1">

<Paragraph number="1925" type="verses" book="ja1">
_“Appo hutvā bahu hoti,_  
_vaḍḍhate so akhantijo;_  
_Āsaṅgī bahupāyāso,_  
_tasmā kodhaṃ na rocaye._  

</Paragraph>

</Division>
<Division number="17" book="ja1">

<Paragraph number="1926" type="verses" book="ja1">
_Duṭṭhassa pharusā vācā,_  
_parāmāso anantarā;_  
_Tato pāṇi tato daṇḍo,_  
_satthassa paramā gati;_  
_Doso kodhasamuṭṭhāno,_  
_tasmā dosaṃ na rocaye._  

</Paragraph>

</Division>
<Division number="18" book="ja1">

<Paragraph number="1927" type="verses" book="ja1">
_Ālopa sāhasākārā,_  
_nikatī vañcanāni ca;_  
_Dissanti lobhadhammesu,_  
_tasmā lobhaṃ na rocaye._  

</Paragraph>

</Division>

_Te bhusaṃ upatāpenti,_  
_tasmā snehaṃ na rocaye”._  

</Paragraph>

</Division>
