<Division number="29" book="ja1">

<Paragraph number="1178" type="verses" book="ja1">
_“Aññe socanti rodanti,_  
_aññe assumukhā janā;_  
_Pasannamukhavaṇṇosi,_  
_kasmā ghaṭa na socasi”._  

</Paragraph>

</Division>
<Division number="30" book="ja1">

<Paragraph number="1179" type="verses" book="ja1">
_“Nābbhatītaharo soko,_  
_nānāgatasukhāvaho;_  
_Tasmā dhaṅka na socāmi,_  
_natthi soke dutīyatā._  

</Paragraph>

</Division>
<Division number="31" book="ja1">

<Paragraph number="1180" type="verses" book="ja1">
_Socaṃ paṇḍu kiso hoti,_  
_bhattañcassa na ruccati;_  
_Amittā sumanā honti,_  
_sallaviddhassa ruppato._  

</Paragraph>

</Division>
<Division number="32" book="ja1">

<Paragraph number="1181" type="verses" book="ja1">
_Gāme vā yadi vāraññe,_  
_ninne vā yadi vā thale;_  
_Ṭhitaṃ maṃ nāgamissati,_  
_evaṃ diṭṭhapado ahaṃ._  

</Paragraph>

</Division>
<Division number="33" book="ja1">

<Paragraph number="1182" type="verses" book="ja1">
_Yassattā nālamekova,_  
_sabbakāmarasāharo;_  
_Sabbāpi pathavī tassa,_  
_na sukhaṃ āvahissatī”ti._  


</Paragraph>
</Division>

<Paragraph number="1183" type="center" book="ja1">
Ghaṭajātakaṃ pañcamaṃ.
</Paragraph>
