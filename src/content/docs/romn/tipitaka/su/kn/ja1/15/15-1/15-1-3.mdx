<Division number="52" book="ja1">

<Paragraph number="2906" type="verses" book="ja1">
_“Dūre apassaṃ therova,_  
_cakkhuṃ yācitumāgato;_  
_Ekanettā bhavissāma,_  
_cakkhuṃ me dehi yācito”._  

</Paragraph>

</Division>
<Division number="53" book="ja1">

<Paragraph number="2907" type="verses" book="ja1">
_“Kenānusiṭṭho idha māgatosi,_  
_Vanibbaka cakkhupathāni yācituṃ;_  
_Suduccajaṃ yācasi uttamaṅgaṃ,_  
_Yamāhu nettaṃ purisena duccajaṃ”._  

</Paragraph>

</Division>
<Division number="54" book="ja1">

<Paragraph number="2908" type="verses" book="ja1">
_“Yamāhu devesu sujampatīti,_  
_Maghavāti naṃ āhu manussaloke;_  
_Tenānusiṭṭho idha māgatosmi,_  
_Vanibbako cakkhupathāni yācituṃ._  

</Paragraph>

</Division>
<Division number="55" book="ja1">

<Paragraph number="2909" type="verses" book="ja1">
_Vanibbato mayha vaniṃ anuttaraṃ,_  
_Dadāhi te cakkhupathāni yācito;_  
_Dadāhi me cakkhupathaṃ anuttaraṃ,_  
_Yamāhu nettaṃ purisena duccajaṃ”._  

</Paragraph>

</Division>
<Division number="56" book="ja1">

<Paragraph number="2910" type="verses" book="ja1">
_“Yena atthena āgacchi,_  
_yamatthamabhipatthayaṃ;_  
_Te te ijjhantu saṅkappā,_  
_labha cakkhūni brāhmaṇa._  

</Paragraph>

</Division>
<Division number="57" book="ja1">

<Paragraph number="2911" type="verses" book="ja1">
_Ekaṃ te yācamānassa,_  
_Ubhayāni dadāmahaṃ;_  
_Sa cakkhumā gaccha janassa pekkhato,_  
_Yadicchase tvaṃ tadate samijjhatu”._  

</Paragraph>

</Division>
<Division number="58" book="ja1">

<Paragraph number="2912" type="verses" book="ja1">
_“Mā no deva adā cakkhuṃ,_  
_mā no sabbe parākari;_  
_Dhanaṃ dehi mahārāja,_  
_muttā veḷuriyā bahū._  

</Paragraph>

</Division>
<Division number="59" book="ja1">

<Paragraph number="2913" type="verses" book="ja1">
_Yutte deva rathe dehi,_  
_ājānīye calaṅkate;_  
_Nāge dehi mahārāja,_  
_hemakappanavāsase._  

</Paragraph>

</Division>
<Division number="60" book="ja1">

<Paragraph number="2914" type="verses" book="ja1">
_Yathā taṃ sivayo sabbe,_  
_sayoggā sarathā sadā;_  
_Samantā parikireyyuṃ,_  
_evaṃ dehi rathesabha”._  

</Paragraph>

</Division>
<Division number="61" book="ja1">

<Paragraph number="2915" type="verses" book="ja1">
_“Yo ve dassanti vatvāna,_  
_adāne kurute mano;_  
_Bhūmyaṃ so patitaṃ pāsaṃ,_  
_gīvāyaṃ paṭimuñcati._  

</Paragraph>

</Division>
<Division number="62" book="ja1">

<Paragraph number="2916" type="verses" book="ja1">
_Yo ve dassanti vatvāna,_  
_adāne kurute mano;_  
_Pāpā pāpataro hoti,_  
_sampatto yamasādhanaṃ._  

</Paragraph>

</Division>
<Division number="63" book="ja1">

<Paragraph number="2917" type="verses" book="ja1">
_Yañhi yāce tañhi dade,_  
_yaṃ na yāce na taṃ dade;_  
_Svāhaṃ tameva dassāmi,_  
_yaṃ maṃ yācati brāhmaṇo”._  

</Paragraph>

</Division>
<Division number="64" book="ja1">

<Paragraph number="2918" type="verses" book="ja1">
_“Āyuṃ nu vaṇṇaṃ nu sukhaṃ balaṃ nu,_  
_Kiṃ patthayāno nu janinda desi;_  
_Kathañhi rājā sivinaṃ anuttaro,_  
_Cakkhūni dajjā paralokahetu”._  

</Paragraph>

</Division>
<Division number="65" book="ja1">

<Paragraph number="2919" type="verses" book="ja1">
_“Na vāhametaṃ yasasā dadāmi,_  
_Na puttamicche na dhanaṃ na raṭṭhaṃ;_  
_Satañca dhammo carito purāṇo,_  
_Icceva dāne ramate mano mama”._  
_( )_  

</Paragraph>

</Division>
<Division number="66" book="ja1">

<Paragraph number="2920" type="verses" book="ja1">
_“Sakhā ca mitto ca mamāsi sīvika,_  
_Susikkhito sādhu karohi me vaco;_  
_Uddharitvā cakkhūni mamaṃ jigīsato,_  
_Hatthesu ṭhapehi vanibbakassa”._  

</Paragraph>

</Division>
<Division number="67" book="ja1">

<Paragraph number="2921" type="verses" book="ja1">
_Codito sivirājena,_  
_sīviko vacanaṅkaro;_  
_Rañño cakkhūnuddharitvā,_  
_brāhmaṇassūpanāmayi;_  
_Sacakkhu brāhmaṇo āsi,_  
_andho rājā upāvisi._  

</Paragraph>

</Division>
<Division number="68" book="ja1">

<Paragraph number="2922" type="verses" book="ja1">
_Tato so katipāhassa,_  
_uparūḷhesu cakkhusu;_  
_Sūtaṃ āmantayī rājā,_  
_sivīnaṃ raṭṭhavaḍḍhano._  

</Paragraph>

</Division>
<Division number="69" book="ja1">

<Paragraph number="2923" type="verses" book="ja1">
_“Yojehi sārathi yānaṃ,_  
_yuttañca paṭivedaya;_  
_Uyyānabhūmiṃ gacchāma,_  
_pokkharañño vanāni ca”._  

</Paragraph>

</Division>
<Division number="70" book="ja1">

<Paragraph number="2924" type="verses" book="ja1">
_So ca pokkharaṇītīre,_  
_pallaṅkena upāvisi;_  
_Tassa sakko pāturahu,_  
_devarājā sujampati._  

</Paragraph>

</Division>
<Division number="71" book="ja1">

<Paragraph number="2925" type="verses" book="ja1">
_“Sakkohamasmi devindo,_  
_āgatosmi tavantike;_  
_Varaṃ varassu rājīsi,_  
_yaṃ kiñci manasicchasi”._  

</Paragraph>

</Division>
<Division number="72" book="ja1">

<Paragraph number="2926" type="verses" book="ja1">
_“Pahūtaṃ me dhanaṃ sakka,_  
_balaṃ koso canappako;_  
_Andhassa me sato dāni,_  
_maraṇaññeva ruccati”._  

</Paragraph>

</Division>
<Division number="73" book="ja1">

<Paragraph number="2927" type="verses" book="ja1">
_“Yāni saccāni dvipadinda,_  
_tāni bhāsassu khattiya;_  
_Saccaṃ te bhaṇamānassa,_  
_puna cakkhu bhavissati”._  

</Paragraph>

</Division>
<Division number="74" book="ja1">

<Paragraph number="2928" type="verses" book="ja1">
_“Ye maṃ yācitumāyanti,_  
_nānāgottā vanibbakā;_  
_Yopi maṃ yācate tattha,_  
_sopi me manaso piyo;_  
_Etena saccavajjena,_  
_cakkhu me upapajjatha._  

</Paragraph>

</Division>
<Division number="75" book="ja1">

<Paragraph number="2929" type="verses" book="ja1">
_Yaṃ maṃ so yācituṃ āgā,_  
_dehi cakkhunti brāhmaṇo;_  
_Tassa cakkhūni pādāsiṃ,_  
_brāhmaṇassa vanibbato._  

</Paragraph>

</Division>
<Division number="76" book="ja1">

<Paragraph number="2930" type="verses" book="ja1">
_Bhiyyo maṃ āvisī pīti,_  
_somanassañcanappakaṃ;_  
_Etena saccavajjena,_  
_dutiyaṃ me upapajjatha”._  

</Paragraph>

</Division>
<Division number="77" book="ja1">

<Paragraph number="2931" type="verses" book="ja1">
_“Dhammena bhāsitā gāthā,_  
_sivīnaṃ raṭṭhavaḍḍhana;_  
_Etāni tava nettāni,_  
_dibbāni paṭidissare._  

</Paragraph>

</Division>
<Division number="78" book="ja1">

<Paragraph number="2932" type="verses" book="ja1">
_Tirokuṭṭaṃ tiroselaṃ,_  
_samatiggayha pabbataṃ;_  
_Samantā yojanasataṃ,_  
_dassanaṃ anubhontu te”._  

</Paragraph>

</Division>
<Division number="79" book="ja1">

<Paragraph number="2933" type="verses" book="ja1">
_“Ko nīdha vittaṃ na dadeyya yācito,_  
_Api visiṭṭhaṃ supiyampi attano;_  
_Tadiṅgha sabbe sivayo samāgatā,_  
_Dibbāni nettāni mamajja passatha._  

</Paragraph>

</Division>
<Division number="80" book="ja1">

<Paragraph number="2934" type="verses" book="ja1">
_Tirokuṭṭaṃ tiroselaṃ,_  
_samatiggayha pabbataṃ;_  
_Samantā yojanasataṃ,_  
_dassanaṃ anubhonti me._  

</Paragraph>

</Division>
<Division number="81" book="ja1">

<Paragraph number="2935" type="verses" book="ja1">
_Na cāgamattā paramatthi kiñci,_  
_Maccānaṃ idha jīvite;_  
_Datvāna mānusaṃ cakkhuṃ,_  
_Laddhaṃ me cakkhuṃ amānusaṃ._  

</Paragraph>

</Division>
<Division number="82" book="ja1">

<Paragraph number="2936" type="verses" book="ja1">
_Etampi disvā sivayo,_  
_Detha dānāni bhuñjatha;_  
_Datvā ca bhutvā ca yathānubhāvaṃ,_  
_Aninditā saggamupetha ṭhānan”ti._  


</Paragraph>
</Division>

<Paragraph number="2937" type="center" book="ja1">
Sivijātakaṃ tatiyaṃ.
