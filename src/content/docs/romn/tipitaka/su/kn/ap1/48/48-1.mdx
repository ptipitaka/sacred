<Paragraph number="6066" type="verses" book="ap1">
_“Suvaṇṇavaṇṇaṃ sambuddhaṃ,_  
_Āhutīnaṃ paṭiggahaṃ;_  
_Vipinaggena gacchantaṃ,_  
_Addasaṃ lokanāyakaṃ._  

</Paragraph>

</Division>
<Division number="2" book="ap1" bookNo="20" page="429">

<Paragraph number="6067" type="verses" book="ap1">
_Naḷamālaṃ gahetvāna,_  
_nikkhamanto ca tāvade;_  
_Tatthaddasāsiṃ sambuddhaṃ,_  
_oghatiṇṇamanāsavaṃ._  

</Paragraph>

</Division>
<Division number="3" book="ap1" bookNo="20" page="429">

<Paragraph number="6068" type="verses" book="ap1">
_Pasannacitto sumano,_  
_naḷamālamapūjayiṃ;_  
_Dakkhiṇeyyaṃ mahāvīraṃ,_  
_sabbalokānukampakaṃ._  

</Paragraph>

</Division>
<Division number="4" book="ap1" bookNo="20" page="429">

<Paragraph number="6069" type="verses" book="ap1">
_Ekatiṃse ito kappe,_  
_yaṃ mālamabhiropayiṃ;_  
_Duggatiṃ nābhijānāmi,_  
_buddhapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="5" book="ap1" bookNo="20" page="429">

<Paragraph number="6070" type="verses" book="ap1">
_Kilesā jhāpitā mayhaṃ,_  
_bhavā sabbe samūhatā;_  
_Nāgova bandhanaṃ chetvā,_  
_viharāmi anāsavo._  

</Paragraph>

</Division>
<Division number="6" book="ap1" bookNo="20" page="429">

<Paragraph number="6071" type="verses" book="ap1">
_Svāgataṃ vata me āsi,_  
_mama buddhassa santike;_  
_Tisso vijjā anuppattā,_  
_kataṃ buddhassa sāsanaṃ._  

</Paragraph>

</Division>
<Division number="7" book="ap1" bookNo="20" page="429">

<Paragraph number="6072" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_vimokkhāpi ca aṭṭhime;_  
_Chaḷabhiññā sacchikatā,_  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="6073" book="ap1">
Itthaṃ sudaṃ āyasmā naḷamāliyo thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="6074" type="center" book="ap1">
Naḷamāliyattherassāpadānaṃ paṭhamaṃ.
</Paragraph>
