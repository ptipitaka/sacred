<Paragraph number="5486" type="verses" book="ap1">
_“Nagare haṃsavatiyā,_  
_ahosiṃ tiṇahārako;_  
_Tiṇahārena jīvāmi,_  
_tena posemi dārake._  

</Paragraph>

</Division>
<Division number="16" book="ap1" bookNo="20" page="430">

<Paragraph number="5487" type="verses" book="ap1">
_Padumuttaro nāma jino,_  
_sabbadhammāna pāragū;_  
_Tamandhakāraṃ nāsetvā,_  
_uppajji lokanāyako._  

</Paragraph>

</Division>
<Division number="17" book="ap1" bookNo="20" page="430">

<Paragraph number="5488" type="verses" book="ap1">
_Sake ghare nisīditvā,_  
_evaṃ cintesi tāvade;_  
_‘Buddho loke samuppanno,_  
_deyyadhammo na vijjati._  

</Paragraph>

</Division>
<Division number="18" book="ap1" bookNo="20" page="430">

<Paragraph number="5489" type="verses" book="ap1">
_Idaṃ me sāṭakaṃ ekaṃ,_  
_natthi me koci dāyako;_  
_Dukkho nirayasamphasso,_  
_ropayissāmi dakkhiṇaṃ’._  

</Paragraph>

</Division>
<Division number="19" book="ap1" bookNo="20" page="430">

<Paragraph number="5490" type="verses" book="ap1">
_Evāhaṃ cintayitvāna,_  
_sakaṃ cittaṃ pasādayiṃ;_  
_Ekaṃ dussaṃ gahetvāna,_  
_buddhaseṭṭhassadāsahaṃ._  

</Paragraph>

</Division>
<Division number="20" book="ap1" bookNo="20" page="430">

<Paragraph number="5491" type="verses" book="ap1">
_Ekaṃ dussaṃ daditvāna,_  
_ukkuṭṭhiṃ sampavattayiṃ;_  
_‘Yadi buddho tuvaṃ vīra,_  
_tārehi maṃ mahāmuni’._  

</Paragraph>

</Division>
<Division number="21" book="ap1" bookNo="20" page="430">

<Paragraph number="5492" type="verses" book="ap1">
_Padumuttaro lokavidū,_  
_āhutīnaṃ paṭiggaho;_  
_Mama dānaṃ pakittento,_  
_akā me anumodanaṃ._  

</Paragraph>

</Division>
<Division number="22" book="ap1" bookNo="20" page="430">

<Paragraph number="5493" type="verses" book="ap1">
_‘Iminā ekadussena,_  
_cetanāpaṇidhīhi ca;_  
_Kappasatasahassāni,_  
_vinipātaṃ na gacchasi._  

</Paragraph>

</Division>
<Division number="23" book="ap1" bookNo="20" page="430">

<Paragraph number="5494" type="verses" book="ap1">
_Chattiṃsakkhattuṃ devindo,_  
_devarajjaṃ karissasi;_  
_Tettiṃsakkhattuṃ rājā ca,_  
_cakkavattī bhavissasi._  

</Paragraph>

</Division>
<Division number="24" book="ap1" bookNo="20" page="431">

<Paragraph number="5495" type="verses" book="ap1">
_Padesarajjaṃ vipulaṃ,_  
_gaṇanāto asaṅkhiyaṃ;_  
_Devaloke manusse vā,_  
_saṃsaranto tuvaṃ bhave._  

</Paragraph>

</Division>
<Division number="25" book="ap1" bookNo="20" page="431">

<Paragraph number="5496" type="verses" book="ap1">
_Rūpavā guṇasampanno,_  
_anavakkantadehavā;_  
_Akkhobhaṃ amitaṃ dussaṃ,_  
_labhissasi yadicchakaṃ’._  

</Paragraph>

</Division>
<Division number="26" book="ap1" bookNo="20" page="431">

<Paragraph number="5497" type="verses" book="ap1">
_Idaṃ vatvāna sambuddho,_  
_jalajuttamanāmako;_  
_Nabhaṃ abbhuggamī vīro,_  
_haṃsarājāva ambare._  

</Paragraph>

</Division>
<Division number="27" book="ap1" bookNo="20" page="431">

<Paragraph number="5498" type="verses" book="ap1">
_Yaṃ yaṃ yonupapajjāmi,_  
_devattaṃ atha mānusaṃ;_  
_Bhoge me ūnatā natthi,_  
_ekadussassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="28" book="ap1" bookNo="20" page="431">

<Paragraph number="5499" type="verses" book="ap1">
_Paduddhāre paduddhāre,_  
_dussaṃ nibbattate mamaṃ;_  
_Heṭṭhā dussamhi tiṭṭhāmi,_  
_uparicchadanaṃ mama._  

</Paragraph>

</Division>
<Division number="29" book="ap1" bookNo="20" page="431">

<Paragraph number="5500" type="verses" book="ap1">
_Cakkavāḷaṃ upādāya,_  
_sakānanaṃ sapabbataṃ;_  
_Icchamāno cahaṃ ajja,_  
_dussehacchādayeyya taṃ._  

</Paragraph>

</Division>
<Division number="30" book="ap1" bookNo="20" page="431">

<Paragraph number="5501" type="verses" book="ap1">
_Teneva ekadussena,_  
_saṃsaranto bhavābhave;_  
_Suvaṇṇavaṇṇo hutvāna,_  
_saṃsarāmi bhavābhave._  

</Paragraph>

</Division>
<Division number="31" book="ap1" bookNo="20" page="431">

<Paragraph number="5502" type="verses" book="ap1">
_Vipākaṃ ekadussassa,_  
_nājjhagaṃ katthacikkhayaṃ;_  
_Ayaṃ me antimā jāti,_  
_vipaccati idhāpi me._  

</Paragraph>

</Division>
<Division number="32" book="ap1" bookNo="20" page="431">

<Paragraph number="5503" type="verses" book="ap1">
_Satasahassito kappe,_  
_yaṃ dussamadadiṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_ekadussassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="33" book="ap1" bookNo="20" page="431">

<Paragraph number="5504" type="verses" book="ap1">
_Kilesā jhāpitā mayhaṃ,_  
_ …pe… _  
_viharāmi anāsavo._  

</Paragraph>

</Division>
<Division number="34" book="ap1" bookNo="20" page="432">

<Paragraph number="5505" type="verses" book="ap1">
_Svāgataṃ vata me āsi,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ._  

</Paragraph>

</Division>
<Division number="35" book="ap1" bookNo="20" page="432">

<Paragraph number="5506" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="5507" book="ap1">
Itthaṃ sudaṃ āyasmā ekadussadāyako thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="5508" type="center" book="ap1">
Ekadussadāyakattherassāpadānaṃ dutiyaṃ.
</Paragraph>
