<Division number="72" book="ap1">

<Paragraph number="6153" type="verses" book="ap1">
_Pānadhiṃ sukataṃ gayha,_  
_addhānaṃ paṭipajjahaṃ;_  
_Tatthaddasāsiṃ sambuddhaṃ,_  
_pattikaṃ cārudassanaṃ._  

</Paragraph>

</Division>
<Division number="73" book="ap1">

<Paragraph number="6154" type="verses" book="ap1">
_Sakaṃ cittaṃ pasādetvā,_  
_nīharitvāna pānadhiṃ;_  
_Pādamūle ṭhapetvāna,_  
_idaṃ vacanamabraviṃ._  

</Paragraph>

</Division>
<Division number="74" book="ap1">

<Paragraph number="6155" type="verses" book="ap1">
_‘Abhirūha mahāvīra,_  
_sugatinda vināyaka;_  
_Ito phalaṃ labhissāmi,_  
_so me attho samijjhatu’._  

</Paragraph>

</Division>
<Division number="75" book="ap1">

<Paragraph number="6156" type="verses" book="ap1">
_Anomadassī bhagavā,_  
_lokajeṭṭho narāsabho;_  
_Pānadhiṃ abhirūhitvā,_  
_idaṃ vacanamabravi._  

</Paragraph>

</Division>
<Division number="76" book="ap1">

<Paragraph number="6157" type="verses" book="ap1">
_‘Yo pānadhiṃ me adāsi,_  
_pasanno sehi pāṇibhi;_  
_Tamahaṃ kittayissāmi,_  
_suṇātha mama bhāsato’._  

</Paragraph>

</Division>
<Division number="77" book="ap1">

<Paragraph number="6158" type="verses" book="ap1">
_Buddhassa giramaññāya,_  
_sabbe devā samāgatā;_  
_Udaggacittā sumanā,_  
_vedajātā katañjalī._  

</Paragraph>

</Division>
<Division number="78" book="ap1">

<Paragraph number="6159" type="verses" book="ap1">
_‘Pānadhīnaṃ padānena,_  
_sukhitoyaṃ bhavissati;_  
_Pañcapaññāsakkhattuñca,_  
_devarajjaṃ karissati._  

</Paragraph>

</Division>
<Division number="79" book="ap1">

<Paragraph number="6160" type="verses" book="ap1">
_Sahassakkhattuṃ rājā ca,_  
_cakkavattī bhavissati;_  
_Padesarajjaṃ vipulaṃ,_  
_gaṇanāto asaṅkhiyaṃ._  

</Paragraph>

</Division>
<Division number="80" book="ap1">

<Paragraph number="6161" type="verses" book="ap1">
_Aparimeyye ito kappe,_  
_okkākakulasambhavo;_  
_Gotamo nāma gottena,_  
_satthā loke bhavissati._  

</Paragraph>

</Division>
<Division number="81" book="ap1">

<Paragraph number="6162" type="verses" book="ap1">
_Tassa dhammesu dāyādo,_  
_oraso dhammanimmito;_  
_Sabbāsave pariññāya,_  
_nibbāyissatināsavo._  

</Paragraph>

</Division>
<Division number="82" book="ap1">

<Paragraph number="6163" type="verses" book="ap1">
_Devaloke manusse vā,_  
_nibbattissati puññavā;_  
_Devayānapaṭibhāgaṃ,_  
_yānaṃ paṭilabhissati’._  

</Paragraph>

</Division>
<Division number="83" book="ap1">

<Paragraph number="6164" type="verses" book="ap1">
_Pāsādā sivikā vayhaṃ,_  
_hatthino samalaṅkatā;_  
_Rathā vājaññasaṃyuttā,_  
_sadā pātubhavanti me._  

</Paragraph>

</Division>
<Division number="84" book="ap1">

<Paragraph number="6165" type="verses" book="ap1">
_Agārā nikkhamantopi,_  
_rathena nikkhamiṃ ahaṃ;_  
_Kesesu chijjamānesu,_  
_arahattamapāpuṇiṃ._  

</Paragraph>

</Division>
<Division number="85" book="ap1">

<Paragraph number="6166" type="verses" book="ap1">
_Lābhā mayhaṃ suladdhaṃ me,_  
_vāṇijjaṃ suppayojitaṃ;_  
_Datvāna pānadhiṃ ekaṃ,_  
_pattomhi acalaṃ padaṃ._  

</Paragraph>

</Division>
<Division number="86" book="ap1">

<Paragraph number="6167" type="verses" book="ap1">
_Aparimeyye ito kappe,_  
_yaṃ pānadhimadāsahaṃ;_  
_Duggatiṃ nābhijānāmi,_  
_pānadhissa idaṃ phalaṃ._  

</Paragraph>

</Division>

_ …pe… _  
_viharāmi anāsavo._  

</Paragraph>

</Division>

_ …pe… _
