<Paragraph number="5249" type="verses" book="ap1">
_“Sumedho nāma sambuddho,_  
_aggo kāruṇiko muni;_  
_Vivekakāmo lokaggo,_  
_himavantamupāgami._  

</Paragraph>

</Division>
<Division number="2" book="ap1" bookNo="20" page="429">

<Paragraph number="5250" type="verses" book="ap1">
_Ajjhogāhetvā himavaṃ,_  
_sumedho lokanāyako;_  
_Pallaṅkaṃ ābhujitvāna,_  
_nisīdi purisuttamo._  

</Paragraph>

</Division>
<Division number="3" book="ap1" bookNo="20" page="429">

<Paragraph number="5251" type="verses" book="ap1">
_Samādhiṃ so samāpanno,_  
_sumedho lokanāyako;_  
_Sattarattindivaṃ buddho,_  
_nisīdi purisuttamo._  

</Paragraph>

</Division>
<Division number="4" book="ap1" bookNo="20" page="429">

<Paragraph number="5252" type="verses" book="ap1">
_Khāribhāraṃ gahetvāna,_  
_vanamajjhogahiṃ ahaṃ;_  
_Tatthaddasāsiṃ sambuddhaṃ,_  
_oghatiṇṇamanāsavaṃ._  

</Paragraph>

</Division>
<Division number="5" book="ap1" bookNo="20" page="429">

<Paragraph number="5253" type="verses" book="ap1">
_Sammajjaniṃ gahetvāna,_  
_sammajjitvāna assamaṃ;_  
_Catudaṇḍe ṭhapetvāna,_  
_akāsiṃ maṇḍapaṃ tadā._  

</Paragraph>

</Division>
<Division number="6" book="ap1" bookNo="20" page="429">

<Paragraph number="5254" type="verses" book="ap1">
_Sālapupphaṃ āharitvā,_  
_maṇḍapaṃ chādayiṃ ahaṃ;_  
_Pasannacitto sumano,_  
_abhivandiṃ tathāgataṃ._  

</Paragraph>

</Division>
<Division number="7" book="ap1" bookNo="20" page="429">

<Paragraph number="5255" type="verses" book="ap1">
_Yaṃ vadanti sumedhoti,_  
_bhūripaññaṃ sumedhasaṃ;_  
_Bhikkhusaṃghe nisīditvā,_  
_imā gāthā abhāsatha._  

</Paragraph>

</Division>
<Division number="8" book="ap1" bookNo="20" page="429">

<Paragraph number="5256" type="verses" book="ap1">
_Buddhassa giramaññāya,_  
_sabbe devā samāgamuṃ;_  
_Asaṃsayaṃ buddhaseṭṭho,_  
_dhammaṃ deseti cakkhumā._  

</Paragraph>

</Division>
<Division number="9" book="ap1" bookNo="20" page="429">

<Paragraph number="5257" type="verses" book="ap1">
_Sumedho nāma sambuddho,_  
_āhutīnaṃ paṭiggaho;_  
_Devasaṅghe nisīditvā,_  
_imā gāthā abhāsatha._  

</Paragraph>

</Division>
<Division number="10" book="ap1" bookNo="20" page="429">

<Paragraph number="5258" type="verses" book="ap1">
_‘Yo me sattāhaṃ maṇḍapaṃ,_  
_dhārayī sālachāditaṃ;_  
_Tamahaṃ kittayissāmi,_  
_suṇātha mama bhāsato._  

</Paragraph>

</Division>
<Division number="11" book="ap1" bookNo="20" page="429">

<Paragraph number="5259" type="verses" book="ap1">
_Devabhūto manusso vā,_  
_hemavaṇṇo bhavissati;_  
_Pahūtabhogo hutvāna,_  
_kāmabhogī bhavissati._  

</Paragraph>

</Division>
<Division number="12" book="ap1" bookNo="20" page="430">

<Paragraph number="5260" type="verses" book="ap1">
_Saṭṭhi nāgasahassāni,_  
_sabbālaṅkārabhūsitā;_  
_Suvaṇṇakacchā mātaṅgā,_  
_hemakappanavāsasā._  

</Paragraph>

</Division>
<Division number="13" book="ap1" bookNo="20" page="430">

<Paragraph number="5261" type="verses" book="ap1">
_Ārūḷhā gāmaṇīyehi,_  
_tomaraṅkusapāṇibhi;_  
_Sāyaṃ pāto upaṭṭhānaṃ,_  
_āgamissantimaṃ naraṃ;_  
_Tehi nāgehi parivuto,_  
_ramissati ayaṃ naro._  

</Paragraph>

</Division>
<Division number="14" book="ap1" bookNo="20" page="430">

<Paragraph number="5262" type="verses" book="ap1">
_Saṭṭhi assasahassāni,_  
_sabbālaṅkārabhūsitā;_  
_Ājānīyāva jātiyā,_  
_sindhavā sīghavāhino._  

</Paragraph>

</Division>
<Division number="15" book="ap1" bookNo="20" page="430">

<Paragraph number="5263" type="verses" book="ap1">
_Ārūḷhā gāmaṇīyehi,_  
_illiyācāpadhāribhi;_  
_Parivāressantimaṃ niccaṃ,_  
_buddhapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="16" book="ap1" bookNo="20" page="430">

<Paragraph number="5264" type="verses" book="ap1">
_Saṭṭhi rathasahassāni,_  
_sabbālaṅkārabhūsitā;_  
_Dīpā athopi veyagghā,_  
_sannaddhā ussitaddhajā._  

</Paragraph>

</Division>
<Division number="17" book="ap1" bookNo="20" page="430">

<Paragraph number="5265" type="verses" book="ap1">
_Ārūḷhā gāmaṇīyehi,_  
_cāpahatthehi vammibhi;_  
_Parivāressantimaṃ niccaṃ,_  
_buddhapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="18" book="ap1" bookNo="20" page="430">

<Paragraph number="5266" type="verses" book="ap1">
_Saṭṭhi gāmasahassāni,_  
_paripuṇṇāni sabbaso;_  
_Pahūtadhanadhaññāni,_  
_susamiddhāni sabbaso;_  
_Sadā pātubhavissanti,_  
_buddhapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="19" book="ap1" bookNo="20" page="430">

<Paragraph number="5267" type="verses" book="ap1">
_Hatthī assā rathā pattī,_  
_senā ca caturaṅginī;_  
_Parivāressantimaṃ niccaṃ,_  
_buddhapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="20" book="ap1" bookNo="20" page="430">

<Paragraph number="5268" type="verses" book="ap1">
_Aṭṭhārase kappasate,_  
_devaloke ramissati;_  
_Sahassakkhattuṃ rājā ca,_  
_cakkavattī bhavissati._  

</Paragraph>

</Division>
<Division number="21" book="ap1" bookNo="20" page="430">

<Paragraph number="5269" type="verses" book="ap1">
_Satānaṃ tīṇikkhattuñca,_  
_devarajjaṃ karissati;_  
_Padesarajjaṃ vipulaṃ,_  
_gaṇanāto asaṅkhiyaṃ._  

</Paragraph>

</Division>
<Division number="22" book="ap1" bookNo="20" page="430">

<Paragraph number="5270" type="verses" book="ap1">
_Tiṃsakappasahassamhi,_  
_okkākakulasambhavo;_  
_Gotamo nāma gottena,_  
_satthā loke bhavissati._  

</Paragraph>

</Division>
<Division number="23" book="ap1" bookNo="20" page="430">

<Paragraph number="5271" type="verses" book="ap1">
_Tassa dhammesu dāyādo,_  
_oraso dhammanimmito;_  
_Sabbāsave pariññāya,_  
_viharissatināsavo’._  

</Paragraph>

</Division>
<Division number="24" book="ap1" bookNo="20" page="431">

<Paragraph number="5272" type="verses" book="ap1">
_Tiṃsakappasahassamhi,_  
_addasaṃ lokanāyakaṃ;_  
_Etthantaramupādāya,_  
_gavesiṃ amataṃ padaṃ._  

</Paragraph>

</Division>
<Division number="25" book="ap1" bookNo="20" page="431">

<Paragraph number="5273" type="verses" book="ap1">
_Lābhā mayhaṃ suladdhaṃ me,_  
_yamahaññāsi sāsanaṃ;_  
_Tisso vijjā anuppattā,_  
_kataṃ buddhassa sāsanaṃ._  

</Paragraph>

</Division>
<Division number="26" book="ap1" bookNo="20" page="431">

<Paragraph number="5274" type="verses" book="ap1">
_Namo te purisājañña,_  
_namo te purisuttama;_  
_Tava ñāṇaṃ pakittetvā,_  
_pattomhi acalaṃ padaṃ._  

</Paragraph>

</Division>
<Division number="27" book="ap1" bookNo="20" page="431">

<Paragraph number="5275" type="verses" book="ap1">
_Yaṃ yaṃ yonupapajjāmi,_  
_devattaṃ atha mānusaṃ;_  
_Sabbattha sukhito homi,_  
_phalaṃ me ñāṇakittane._  

</Paragraph>

</Division>
<Division number="28" book="ap1" bookNo="20" page="431">

<Paragraph number="5276" type="verses" book="ap1">
_Idaṃ pacchimakaṃ mayhaṃ,_  
_carimo vattate bhavo;_  
_Nāgova bandhanaṃ chetvā,_  
_viharāmi anāsavo._  

</Paragraph>

</Division>
<Division number="29" book="ap1" bookNo="20" page="431">

<Paragraph number="5277" type="verses" book="ap1">
_Kilesā jhāpitā mayhaṃ,_  
_bhavā sabbe samūhatā;_  
_Nāgova bandhanaṃ chetvā,_  
_viharāmi anāsavo._  

</Paragraph>

</Division>
<Division number="30" book="ap1" bookNo="20" page="431">

<Paragraph number="5278" type="verses" book="ap1">
_Svāgataṃ vata me āsi,_  
_mama buddhassa santike;_  
_Tisso vijjā anuppattā,_  
_kataṃ buddhassa sāsanaṃ._  

</Paragraph>

</Division>
<Division number="31" book="ap1" bookNo="20" page="431">

<Paragraph number="5279" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_vimokkhāpi ca aṭṭhime;_  
_Chaḷabhiññā sacchikatā,_  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="5280" book="ap1">
Itthaṃ sudaṃ āyasmā bhaddālitthero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="5281" type="center" book="ap1">
Bhaddālittherassāpadānaṃ paṭhamaṃ.
</Paragraph>
