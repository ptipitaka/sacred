<Division number="14" book="ap1">

<Paragraph number="2527" type="verses" book="ap1">
_Rattikaṃ pupphitaṃ disvā,_  
_kuṭajaṃ dharaṇīruhaṃ;_  
_Samūlaṃ paggahetvāna,_  
_upanesiṃ mahesino._  

</Paragraph>

</Division>
<Division number="15" book="ap1">

<Paragraph number="2528" type="verses" book="ap1">
_Ekanavutito kappe,_  
_yaṃ pupphamabhiropayiṃ;_  
_Duggatiṃ nābhijānāmi,_  
_pupphadānassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="16" book="ap1">

<Paragraph number="2529" type="verses" book="ap1">
_Ito ca aṭṭhame kappe,_  
_suppasannasanāmako;_  
_Sattaratanasampanno,_  
_rājāhosiṃ mahabbalo._  

</Paragraph>

</Division>

_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="2531" book="ap1">

</Paragraph>
