<Paragraph number="3777" type="verses" book="ap1">
_“Padumuttarabhagavato,_  
_lokajeṭṭhassa tādino;_  
_Vipine cetiyaṃ āsi,_  
_vāḷamigasamākule._  

</Paragraph>

</Division>
<Division number="54" book="ap1" bookNo="20" page="310">

<Paragraph number="3778" type="verses" book="ap1">
_Na koci visahi gantuṃ,_  
_cetiyaṃ abhivandituṃ;_  
_Tiṇakaṭṭhalatonaddhaṃ,_  
_paluggaṃ āsi cetiyaṃ._  

</Paragraph>

</Division>
<Division number="55" book="ap1" bookNo="20" page="310">

<Paragraph number="3779" type="verses" book="ap1">
_Vanakammiko tadā āsiṃ,_  
_pitumātumatenahaṃ;_  
_Addasaṃ vipine thūpaṃ,_  
_luggaṃ tiṇalatākulaṃ._  

</Paragraph>

</Division>
<Division number="56" book="ap1" bookNo="20" page="310">

<Paragraph number="3780" type="verses" book="ap1">
_Disvānāhaṃ buddhathūpaṃ,_  
_garucittaṃ upaṭṭhahiṃ;_  
_Buddhaseṭṭhassa thūpoyaṃ,_  
_paluggo acchatī vane._  

</Paragraph>

</Division>
<Division number="57" book="ap1" bookNo="20" page="310">

<Paragraph number="3781" type="verses" book="ap1">
_Nacchannaṃ nappatirūpaṃ,_  
_jānantassa guṇāguṇaṃ;_  
_Buddhathūpaṃ asodhetvā,_  
_aññaṃ kammaṃ payojaye._  

</Paragraph>

</Division>
<Division number="58" book="ap1" bookNo="20" page="310">

<Paragraph number="3782" type="verses" book="ap1">
_Tiṇakaṭṭhañca valliñca,_  
_sodhayitvāna cetiye;_  
_Vanditvā aṭṭha vārāni,_  
_paṭikuṭiko agacchahaṃ._  

</Paragraph>

</Division>
<Division number="59" book="ap1" bookNo="20" page="310">

<Paragraph number="3783" type="verses" book="ap1">
_Tena kammena sukatena,_  
_cetanāpaṇidhīhi ca;_  
_Jahitvā mānusaṃ dehaṃ,_  
_tāvatiṃsamagacchahaṃ._  

</Paragraph>

</Division>
<Division number="60" book="ap1" bookNo="20" page="310">

<Paragraph number="3784" type="verses" book="ap1">
_Tattha me sukataṃ byamhaṃ,_  
_sovaṇṇaṃ sapabhassaraṃ;_  
_Saṭṭhiyojanamubbiddhaṃ,_  
_tiṃsayojanavitthataṃ._  

</Paragraph>

</Division>
<Division number="61" book="ap1" bookNo="20" page="310">

<Paragraph number="3785" type="verses" book="ap1">
_Tisatāni ca vārāni,_  
_devarajjamakārayiṃ;_  
_Pañcavīsatikkhattuñca,_  
_cakkavattī ahosahaṃ._  

</Paragraph>

</Division>
<Division number="62" book="ap1" bookNo="20" page="310">

<Paragraph number="3786" type="verses" book="ap1">
_Bhavābhave saṃsaranto,_  
_mahābhogaṃ labhāmahaṃ;_  
_Bhoge me ūnatā natthi,_  
_sodhanāya idaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="63" book="ap1" bookNo="20" page="310">

<Paragraph number="3787" type="verses" book="ap1">
_Sivikā hatthikhandhena,_  
_vipine gacchato mama;_  
_Yaṃ yaṃ disāhaṃ gacchāmi,_  
_saraṇaṃ sampate vanaṃ._  

</Paragraph>

</Division>
<Division number="64" book="ap1" bookNo="20" page="310">

<Paragraph number="3788" type="verses" book="ap1">
_Khāṇuṃ vā kaṇṭakaṃ vāpi,_  
_nāhaṃ passāmi cakkhunā;_  
_Puññakammena saṃyutto,_  
_sayamevāpanīyare._  

</Paragraph>

</Division>
<Division number="65" book="ap1" bookNo="20" page="310">

<Paragraph number="3789" type="verses" book="ap1">
_Kuṭṭhaṃ gaṇḍo kilāso ca,_  
_apamāro vitacchikā;_  
_Daddu kacchu ca me natthi,_  
_sodhanāya idaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="66" book="ap1" bookNo="20" page="310">

<Paragraph number="3790" type="verses" book="ap1">
_Aññampi me acchariyaṃ,_  
_buddhathūpassa sodhane;_  
_Nābhijānāmi me kāye,_  
_jātaṃ piḷakabindukaṃ._  

</Paragraph>

</Division>
<Division number="67" book="ap1" bookNo="20" page="311">

<Paragraph number="3791" type="verses" book="ap1">
_Aññampi me acchariyaṃ,_  
_buddhathūpamhi sodhite;_  
_Duve bhave saṃsarāmi,_  
_devatte atha mānuse._  

</Paragraph>

</Division>
<Division number="68" book="ap1" bookNo="20" page="311">

<Paragraph number="3792" type="verses" book="ap1">
_Aññampi me acchariyaṃ,_  
_buddhathūpamhi sodhite;_  
_Suvaṇṇavaṇṇo sabbattha,_  
_sappabhāso bhavāmahaṃ._  

</Paragraph>

</Division>
<Division number="69" book="ap1" bookNo="20" page="311">

<Paragraph number="3793" type="verses" book="ap1">
_Aññampi me acchariyaṃ,_  
_buddhathūpamhi sodhite;_  
_Amanāpaṃ vivajjati,_  
_manāpaṃ upatiṭṭhati._  

</Paragraph>

</Division>
<Division number="70" book="ap1" bookNo="20" page="311">

<Paragraph number="3794" type="verses" book="ap1">
_Aññampi me acchariyaṃ,_  
_buddhathūpamhi sodhite;_  
_Visuddhaṃ hoti me cittaṃ,_  
_ekaggaṃ susamāhitaṃ._  

</Paragraph>

</Division>
<Division number="71" book="ap1" bookNo="20" page="311">

<Paragraph number="3795" type="verses" book="ap1">
_Aññampi me acchariyaṃ,_  
_buddhathūpamhi sodhite;_  
_Ekāsane nisīditvā,_  
_arahattamapāpuṇiṃ._  

</Paragraph>

</Division>
<Division number="72" book="ap1" bookNo="20" page="311">

<Paragraph number="3796" type="verses" book="ap1">
_Satasahassito kappe,_  
_yaṃ kammamakariṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_sodhanāya idaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="73" book="ap1" bookNo="20" page="311">

<Paragraph number="3797" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="3798" book="ap1">
Itthaṃ sudaṃ āyasmā pabhaṅkaro thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="3799" type="center" book="ap1">
Pabhaṅkarattherassāpadānaṃ chaṭṭhaṃ.
