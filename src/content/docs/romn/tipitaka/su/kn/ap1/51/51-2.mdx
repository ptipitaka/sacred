<Division number="39" book="ap1">

<Paragraph number="6608" type="verses" book="ap1">
_“Nagare haṃsavatiyā,_  
_kumbhakāro ahosahaṃ;_  
_Addasaṃ virajaṃ buddhaṃ,_  
_oghatiṇṇamanāsavaṃ._  

</Paragraph>

</Division>
<Division number="40" book="ap1">

<Paragraph number="6609" type="verses" book="ap1">
_Sukataṃ mattikāpattaṃ,_  
_buddhaseṭṭhassadāsahaṃ;_  
_Pattaṃ datvā bhagavato,_  
_ujubhūtassa tādino._  

</Paragraph>

</Division>
<Division number="41" book="ap1">

<Paragraph number="6610" type="verses" book="ap1">
_Bhave nibbattamānohaṃ,_  
_soṇṇathāle labhāmahaṃ;_  
_Rūpimaye ca sovaṇṇe,_  
_taṭṭike ca maṇīmaye._  

</Paragraph>

</Division>
<Division number="42" book="ap1">

<Paragraph number="6611" type="verses" book="ap1">
_Pātiyo paribhuñjāmi,_  
_puññakammassidaṃ phalaṃ;_  
_Yasānañca dhanānañca,_  
_aggabhūto ca homahaṃ._  

</Paragraph>

</Division>
<Division number="43" book="ap1">

<Paragraph number="6612" type="verses" book="ap1">
_Yathāpi bhaddake khette,_  
_bījaṃ appampi ropitaṃ;_  
_Sammādhāraṃ pavecchante,_  
_phalaṃ toseti kassakaṃ._  

</Paragraph>

</Division>
<Division number="44" book="ap1">

<Paragraph number="6613" type="verses" book="ap1">
_Tathevidaṃ pattadānaṃ,_  
_buddhakhettamhi ropitaṃ;_  
_Pītidhāre pavassante,_  
_phalaṃ maṃ tosayissati._  

</Paragraph>

</Division>
<Division number="45" book="ap1">

<Paragraph number="6614" type="verses" book="ap1">
_Yāvatā khettā vijjanti,_  
_saṃghāpi ca gaṇāpi ca;_  
_Buddhakhettasamo natthi,_  
_sukhado sabbapāṇinaṃ._  

</Paragraph>

</Division>
<Division number="46" book="ap1">

<Paragraph number="6615" type="verses" book="ap1">
_Namo te purisājañña,_  
_namo te purisuttama;_  
_Ekapattaṃ daditvāna,_  
_pattomhi acalaṃ padaṃ._  

</Paragraph>

</Division>
<Division number="47" book="ap1">

<Paragraph number="6616" type="verses" book="ap1">
_Ekanavutito kappe,_  
_yaṃ pattamadadiṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_pattadānassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="48" book="ap1">

<Paragraph number="6617" type="verses" book="ap1">
_Kilesā jhāpitā mayhaṃ,_  
_ …pe… _  
_viharāmi anāsavo._  

</Paragraph>

</Division>
<Division number="49" book="ap1">

<Paragraph number="6618" type="verses" book="ap1">
_Svāgataṃ vata me āsi,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ._  

</Paragraph>

</Division>
<Division number="50" book="ap1">

<Paragraph number="6619" type="verses" book="ap1">

_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="6620" book="ap1">

</Paragraph>
