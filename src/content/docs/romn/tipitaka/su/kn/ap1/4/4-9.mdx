<Division number="102" book="ap1">

<Paragraph number="1190" type="verses" book="ap1">
_“Siddhatthassa bhagavato,_  
_sīhāsanassa sammukhā;_  
_Dhammacakkaṃ me ṭhapitaṃ,_  
_sukataṃ viññuvaṇṇitaṃ._  

</Paragraph>

</Division>
<Division number="103" book="ap1">

<Paragraph number="1191" type="verses" book="ap1">
_Cāruvaṇṇova sobhāmi,_  
_sayoggabalavāhano;_  
_Parivārenti maṃ niccaṃ,_  
_anuyantā bahujjanā._  

</Paragraph>

</Division>
<Division number="104" book="ap1">

<Paragraph number="1192" type="verses" book="ap1">
_Saṭṭhitūriyasahassehi,_  
_paricāremahaṃ sadā;_  
_Parivārena sobhāmi,_  
_puññakammassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="105" book="ap1">

<Paragraph number="1193" type="verses" book="ap1">
_Catunnavutito kappe,_  
_yaṃ cakkaṃ ṭhapayiṃ ahaṃ;_  
_Duggatiṃ nābhijānāmi,_  
_dhammacakkassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="106" book="ap1">

<Paragraph number="1194" type="verses" book="ap1">
_Ito ekādase kappe,_  
_aṭṭhāsiṃsu janādhipā;_  
_Sahassarājanāmena,_  
_cakkavattī mahabbalā._  

</Paragraph>

</Division>
<Division number="107" book="ap1">

<Paragraph number="1195" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="1196" book="ap1">
Itthaṃ sudaṃ āyasmā dhammacakkiko thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="1197" type="center" book="ap1">
Dhammacakkikattherassāpadānaṃ navamaṃ.
</Paragraph>
