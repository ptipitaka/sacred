<Division number="91" book="ap1">

<Paragraph number="2160" type="verses" book="ap1">
_“Suvaṇṇavaṇṇo sambuddho,_  
_vipassī dakkhiṇāraho;_  
_Purakkhato sāvakehi,_  
_ārāmā abhinikkhami._  

</Paragraph>

</Division>
<Division number="92" book="ap1">

<Paragraph number="2161" type="verses" book="ap1">
_Disvānahaṃ buddhaseṭṭhaṃ,_  
_sabbaññuṃ tamanāsakaṃ;_  
_Pasannacitto sumano,_  
_gaṇṭhipupphaṃ apūjayiṃ._  

</Paragraph>

</Division>
<Division number="93" book="ap1">

<Paragraph number="2162" type="verses" book="ap1">
_Tena cittappasādena,_  
_dvipadindassa tādino;_  
_Haṭṭho haṭṭhena cittena,_  
_puna vandiṃ tathāgataṃ._  

</Paragraph>

</Division>
<Division number="94" book="ap1">

<Paragraph number="2163" type="verses" book="ap1">
_Ekanavutito kappe,_  
_yaṃ pupphamabhiropayiṃ;_  
_Duggatiṃ nābhijānāmi,_  
_buddhapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="95" book="ap1">

<Paragraph number="2164" type="verses" book="ap1">
_Ekatālīsito kappe,_  
_caraṇo nāma khattiyo;_  
_Sattaratanasampanno,_  
_cakkavattī mahabbalo._  

</Paragraph>

</Division>
<Division number="96" book="ap1">

<Paragraph number="2165" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="2166" book="ap1">
Itthaṃ sudaṃ āyasmā gaṇṭhipupphiyo thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="2167" type="center" book="ap1">
Gaṇṭhipupphiyattherassāpadānaṃ navamaṃ.
</Paragraph>
