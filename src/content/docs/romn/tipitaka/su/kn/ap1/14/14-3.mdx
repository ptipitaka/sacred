<Paragraph number="2202" type="verses" book="ap1">
_“Candabhāgānadītīre,_  
_ahosiṃ kinnaro tadā;_  
_Pupphabhakkho cahaṃ āsiṃ,_  
_pupphānivasano tathā._  

</Paragraph>

</Division>
<Division number="18" book="ap1" bookNo="20" page="181">

<Paragraph number="2203" type="verses" book="ap1">
_Atthadassī tu bhagavā,_  
_lokajeṭṭho narāsabho;_  
_Vipinaggena niyyāsi,_  
_haṃsarājāva ambare._  

</Paragraph>

</Division>
<Division number="19" book="ap1" bookNo="20" page="181">

<Paragraph number="2204" type="verses" book="ap1">
_Namo te purisājañña,_  
_cittaṃ te suvisodhitaṃ;_  
_Pasannamukhavaṇṇosi,_  
_vippasannamukhindriyo._  

</Paragraph>

</Division>
<Division number="20" book="ap1" bookNo="20" page="182">

<Paragraph number="2205" type="verses" book="ap1">
_Orohitvāna ākāsā,_  
_bhūripañño sumedhaso;_  
_Saṅghāṭiṃ pattharitvāna,_  
_pallaṅkena upāvisi._  

</Paragraph>

</Division>
<Division number="21" book="ap1" bookNo="20" page="182">

<Paragraph number="2206" type="verses" book="ap1">
_Vilīnaṃ candanādāya,_  
_agamāsiṃ jinantikaṃ;_  
_Pasannacitto sumano,_  
_buddhassa abhiropayiṃ._  

</Paragraph>

</Division>
<Division number="22" book="ap1" bookNo="20" page="182">

<Paragraph number="2207" type="verses" book="ap1">
_Abhivādetvāna sambuddhaṃ,_  
_lokajeṭṭhaṃ narāsabhaṃ;_  
_Pāmojjaṃ janayitvāna,_  
_pakkāmiṃ uttarāmukho._  

</Paragraph>

</Division>
<Division number="23" book="ap1" bookNo="20" page="182">

<Paragraph number="2208" type="verses" book="ap1">
_Aṭṭhārase kappasate,_  
_candanaṃ yaṃ apūjayiṃ;_  
_Duggatiṃ nābhijānāmi,_  
_buddhapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="24" book="ap1" bookNo="20" page="182">

<Paragraph number="2209" type="verses" book="ap1">
_Catuddase kappasate,_  
_ito āsiṃsu te tayo;_  
_Rohaṇī nāma nāmena,_  
_cakkavattī mahabbalā._  

</Paragraph>

</Division>
<Division number="25" book="ap1" bookNo="20" page="182">

<Paragraph number="2210" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="2211" book="ap1">
Itthaṃ sudaṃ āyasmā candanapūjanako thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="2212" type="center" book="ap1">
Candanapūjanakattherassāpadānaṃ tatiyaṃ.

  
Aṭṭhamabhāṇavāraṃ.
</Paragraph>
