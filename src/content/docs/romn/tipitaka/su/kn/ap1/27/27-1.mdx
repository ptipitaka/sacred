<Division number="1" book="ap1">

<Paragraph number="3158" type="verses" book="ap1">
_“Suvaṇṇavaṇṇaṃ siddhatthaṃ,_  
_Gacchantaṃ antarāpaṇe;_  
_Jalajagge duve gayha,_  
_Upāgacchiṃ narāsabhaṃ._  

</Paragraph>

</Division>
<Division number="2" book="ap1">

<Paragraph number="3159" type="verses" book="ap1">
_Ekañca pupphaṃ pādesu,_  
_buddhaseṭṭhassa nikkhipiṃ;_  
_Ekañca pupphaṃ paggayha,_  
_ākāse ukkhipiṃ ahaṃ._  

</Paragraph>

</Division>
<Division number="3" book="ap1">

<Paragraph number="3160" type="verses" book="ap1">
_Catunnavutito kappe,_  
_yaṃ pupphamabhiropayiṃ;_  
_Duggatiṃ nābhijānāmi,_  
_pupphadānassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="4" book="ap1">

<Paragraph number="3161" type="verses" book="ap1">
_Ito chattiṃsakappamhi,_  
_eko āsiṃ mahīpati;_  
_Antalikkhakaro nāma,_  
_cakkavattī mahabbalo._  

</Paragraph>

</Division>
<Division number="5" book="ap1">

<Paragraph number="3162" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_vimokkhāpi ca aṭṭhime;_  
_Chaḷabhiññā sacchikatā,_  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="3163" book="ap1">
Itthaṃ sudaṃ āyasmā ākāsukkhipiyo thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="3164" type="center" book="ap1">
Ākāsukkhipiyattherassāpadānaṃ paṭhamaṃ.
</Paragraph>
