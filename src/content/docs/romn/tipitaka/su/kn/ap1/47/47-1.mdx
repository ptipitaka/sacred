<Division number="1" book="ap1">

<Paragraph number="5970" type="verses" book="ap1">
_“Parinibbute bhagavati,_  
_jalajuttamanāmake;_  
_Āropitamhi citake,_  
_sālapupphamapūjayiṃ._  

</Paragraph>

</Division>
<Division number="2" book="ap1">

<Paragraph number="5971" type="verses" book="ap1">
_Satasahassito kappe,_  
_Yaṃ pupphamabhiropayiṃ;_  
_Duggatiṃ nābhijānāmi,_  
_Citapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="3" book="ap1">

<Paragraph number="5972" type="verses" book="ap1">
_Kilesā jhāpitā mayhaṃ,_  
_bhavā sabbe samūhatā;_  
_Nāgova bandhanaṃ chetvā,_  
_viharāmi anāsavo._  

</Paragraph>

</Division>
<Division number="4" book="ap1">

<Paragraph number="5973" type="verses" book="ap1">
_Svāgataṃ vata me āsi,_  
_mama buddhassa santike;_  
_Tisso vijjā anuppattā,_  
_kataṃ buddhassa sāsanaṃ._  

</Paragraph>

</Division>
<Division number="5" book="ap1">

<Paragraph number="5974" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_vimokkhāpi ca aṭṭhime;_  
_Chaḷabhiññā sacchikatā,_  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="5975" book="ap1">
Itthaṃ sudaṃ āyasmā sālakusumiyo thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="5976" type="center" book="ap1">
Sālakusumiyattherassāpadānaṃ paṭhamaṃ.
</Paragraph>
