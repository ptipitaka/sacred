<Paragraph number="1964" type="verses" book="ap1">
_“Ubhinnaṃ devarājūnaṃ,_  
_saṅgāmo samupaṭṭhito;_  
_Ahosi samupabyūḷho,_  
_mahāghoso avattatha._  

</Paragraph>

</Division>
<Division number="21" book="ap1" bookNo="20" page="163">

<Paragraph number="1965" type="verses" book="ap1">
_Padumuttaro lokavidū,_  
_āhutīnaṃ paṭiggaho;_  
_Antalikkhe ṭhito satthā,_  
_saṃvejesi mahājanaṃ._  

</Paragraph>

</Division>
<Division number="22" book="ap1" bookNo="20" page="163">

<Paragraph number="1966" type="verses" book="ap1">
_Sabbe devā attamanā,_  
_nikkhittakavacāvudhā;_  
_Sambuddhaṃ abhivādetvā,_  
_ekaggāsiṃsu tāvade._  

</Paragraph>

</Division>
<Division number="23" book="ap1" bookNo="20" page="163">

<Paragraph number="1967" type="verses" book="ap1">
_Mayhaṃ saṅkappamaññāya,_  
_vācāsabhimudīrayi;_  
_Anukampako lokavidū,_  
_nibbāpesi mahājanaṃ._  

</Paragraph>

</Division>
<Division number="24" book="ap1" bookNo="20" page="163">

<Paragraph number="1968" type="verses" book="ap1">
_Paduṭṭhacitto manujo,_  
_ekapāṇaṃ viheṭhayaṃ;_  
_Tena cittappadosena,_  
_apāyaṃ upapajjati._  

</Paragraph>

</Division>
<Division number="25" book="ap1" bookNo="20" page="163">

<Paragraph number="1969" type="verses" book="ap1">
_Saṅgāmasīse nāgova,_  
_bahū pāṇe viheṭhayaṃ;_  
_Nibbāpetha sakaṃ cittaṃ,_  
_mā haññittho punappunaṃ._  

</Paragraph>

</Division>
<Division number="26" book="ap1" bookNo="20" page="163">

<Paragraph number="1970" type="verses" book="ap1">
_Dvinnampi yakkharājūnaṃ,_  
_senā sā vimhitā ahu;_  
_Saraṇañca upāgacchuṃ,_  
_lokajeṭṭhaṃ sutādinaṃ._  

</Paragraph>

</Division>
<Division number="27" book="ap1" bookNo="20" page="163">

<Paragraph number="1971" type="verses" book="ap1">
_Saññāpetvāna janataṃ,_  
_padamuddhari cakkhumā;_  
_Pekkhamānova devehi,_  
_pakkāmi uttarāmukho._  

</Paragraph>

</Division>
<Division number="28" book="ap1" bookNo="20" page="163">

<Paragraph number="1972" type="verses" book="ap1">
_Paṭhamaṃ saraṇaṃ gacchiṃ,_  
_dvipadindassa tādino;_  
_Kappānaṃ satasahassaṃ,_  
_duggatiṃ nupapajjahaṃ._  

</Paragraph>

</Division>
<Division number="29" book="ap1" bookNo="20" page="163">

<Paragraph number="1973" type="verses" book="ap1">
_Mahādundubhināmā ca,_  
_soḷasāsuṃ rathesabhā;_  
_Tiṃsakappasahassamhi,_  
_rājāno cakkavattino._  

</Paragraph>

</Division>
<Division number="30" book="ap1" bookNo="20" page="163">

<Paragraph number="1974" type="verses" book="ap1">

_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="1975" book="ap1">

</Paragraph>
