<Division number="1" book="ap1">

<Paragraph number="4177" type="verses" book="ap1">
_“Sataraṃsi nāma bhagavā,_  
_sayambhū aparājito;_  
_Vivekakāmo sambuddho,_  
_gocarāyābhinikkhami._  

</Paragraph>

</Division>
<Division number="2" book="ap1">

<Paragraph number="4178" type="verses" book="ap1">
_Phalahattho ahaṃ disvā,_  
_upagacchiṃ narāsabhaṃ;_  
_Pasannacitto sumano,_  
_avaṭaṃ adadiṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="3" book="ap1">

<Paragraph number="4179" type="verses" book="ap1">
_Catunnavutito kappe,_  
_yaṃ phalamadadiṃ ahaṃ;_  
_Duggatiṃ nābhijānāmi,_  
_phaladānassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="4" book="ap1">

<Paragraph number="4180" type="verses" book="ap1">
_Kilesā jhāpitā mayhaṃ,_  
_bhavā sabbe samūhatā;_  
_Nāgova bandhanaṃ chetvā,_  
_viharāmi anāsavo._  

</Paragraph>

</Division>
<Division number="5" book="ap1">

<Paragraph number="4181" type="verses" book="ap1">
_Svāgataṃ vata me āsi,_  
_mama buddhassa santike;_  
_Tisso vijjā anuppattā,_  
_kataṃ buddhassa sāsanaṃ._  

</Paragraph>

</Division>
<Division number="6" book="ap1">

<Paragraph number="4182" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_vimokkhāpi ca aṭṭhime;_  
_Chaḷabhiññā sacchikatā,_  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="4183" book="ap1">
Itthaṃ sudaṃ āyasmā avaṭaphaladāyako thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="4184" type="center" book="ap1">
Avaṭaphaladāyakattherassāpadānaṃ paṭhamaṃ.
</Paragraph>
