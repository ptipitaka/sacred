* Tumbaka-karaṇḍa

<Division number="162" book="ap1" bookNo="20" page="365">

<Paragraph number="4459" type="verses" book="ap1">
_Tumbake ca karaṇḍe ca,_  
_datvā buddhe gaṇuttame;_  
_Dasānisaṃse anubhomi,_  
_kammānucchavike mama._  

</Paragraph>

</Division>
<Division number="163" book="ap1" bookNo="20" page="365">

<Paragraph number="4460" type="verses" book="ap1">
_Sugutto sukhasamaṅgī,_  
_Mahāyaso tathāgati;_  
_Vipattivigato sukhumālo,_  
_Sabbītiparivajjito._  

</Paragraph>

</Division>
<Division number="164" book="ap1" bookNo="20" page="365">

<Paragraph number="4461" type="verses" book="ap1">
_Vipule ca guṇe lābhī,_  
_samāva calanā mama;_  
_Suvivajjitaubbego,_  
_tumbake ca karaṇḍake._  

</Paragraph>

</Division>
<Division number="165" book="ap1" bookNo="20" page="365">

<Paragraph number="4462" type="verses" book="ap1">
_Labhāmi caturo vaṇṇe,_  
_hatthissaratanāni ca;_  
_Tāni me na vinassanti,_  
_tumbadāne idaṃ phalaṃ._
</Paragraph>
</Division>
