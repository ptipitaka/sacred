<Division number="111" book="ap1">

<Paragraph number="3841" type="verses" book="ap1">
_“Padumuttaro nāma jino,_  
_sabbadhammāna pāragū;_  
_Catusaccaṃ pakāsento,_  
_santāresi bahuṃ janaṃ._  

</Paragraph>

</Division>
<Division number="112" book="ap1">

<Paragraph number="3842" type="verses" book="ap1">
_Ahaṃ tena samayena,_  
_jaṭilo uggatāpano;_  
_Dhunanto vākacīrāni,_  
_gacchāmi ambare tadā._  

</Paragraph>

</Division>
<Division number="113" book="ap1">

<Paragraph number="3843" type="verses" book="ap1">
_Buddhaseṭṭhassa upari,_  
_gantuṃ na visahāmahaṃ;_  
_Pakkhīva selamāsajja,_  
_gamanaṃ na labhāmahaṃ._  

</Paragraph>

</Division>
<Division number="114" book="ap1">

<Paragraph number="3844" type="verses" book="ap1">
_Na me idaṃ bhūtapubbaṃ,_  
_iriyassa vikopanaṃ;_  
_Dake yathā ummujjitvā,_  
_evaṃ gacchāmi ambare._  

</Paragraph>

</Division>
<Division number="115" book="ap1">

<Paragraph number="3845" type="verses" book="ap1">
_Uḷārabhūto manujo,_  
_heṭṭhāsīno bhavissati;_  
_Handa menaṃ gavesissaṃ,_  
_api atthaṃ labheyyahaṃ._  

</Paragraph>

</Division>
<Division number="116" book="ap1">

<Paragraph number="3846" type="verses" book="ap1">
_Orohanto antalikkhā,_  
_saddamassosi satthuno;_  
_Aniccataṃ kathentassa,_  
_tamahaṃ uggahiṃ tadā._  

</Paragraph>

</Division>
<Division number="117" book="ap1">

<Paragraph number="3847" type="verses" book="ap1">
_Aniccasaññamuggayha,_  
_agamāsiṃ mamassamaṃ;_  
_Yāvatāyuṃ vasitvāna,_  
_tattha kālaṅkato ahaṃ._  

</Paragraph>

</Division>
<Division number="118" book="ap1">

<Paragraph number="3848" type="verses" book="ap1">
_Carime vattamānamhi,_  
_Taṃ dhammasavanaṃ sariṃ;_  
_Tena kammena sukatena,_  
_Tāvatiṃsamagacchahaṃ._  

</Paragraph>

</Division>
<Division number="119" book="ap1">

<Paragraph number="3849" type="verses" book="ap1">
_Tiṃsakappasahassāni,_  
_devaloke ramiṃ ahaṃ;_  
_Ekapaññāsakkhattuñca,_  
_devarajjamakārayiṃ._  

</Paragraph>

</Division>
<Division number="120" book="ap1">

<Paragraph number="3850" type="verses" book="ap1">
_Ekasattatikkhattuñca,_  
_cakkavattī ahosahaṃ;_  
_Padesarajjaṃ vipulaṃ,_  
_gaṇanāto asaṅkhiyaṃ._  

</Paragraph>

</Division>
<Division number="121" book="ap1">

<Paragraph number="3851" type="verses" book="ap1">
_Pitugehe nisīditvā,_  
_samaṇo bhāvitindriyo;_  
_Gāthāya paridīpento,_  
_aniccatamudāhari._  

</Paragraph>

</Division>
<Division number="122" book="ap1">

<Paragraph number="3852" type="verses" book="ap1">
_Anussarāmi taṃ saññaṃ,_  
_saṃsaranto bhavābhave;_  
_Na koṭiṃ paṭivijjhāmi,_  
_nibbānaṃ accutaṃ padaṃ._  

</Paragraph>

</Division>
<Division number="123" book="ap1">

<Paragraph number="3853" type="verses" book="ap1">
_Aniccā vata saṅkhārā,_  
_uppādavayadhammino;_  
_Uppajjitvā nirujjhanti,_  
_tesaṃ vūpasamo sukho._  

</Paragraph>

</Division>
<Division number="124" book="ap1">

<Paragraph number="3854" type="verses" book="ap1">
_Saha gāthaṃ suṇitvāna,_  
_pubbakammaṃ anussariṃ;_  
_Ekāsane nisīditvā,_  
_arahattamapāpuṇiṃ._  

</Paragraph>

</Division>
<Division number="125" book="ap1">

<Paragraph number="3855" type="verses" book="ap1">
_Jātiyā sattavassohaṃ,_  
_arahattamapāpuṇiṃ;_  
_Upasampādayi buddho,_  
_guṇamaññāya cakkhumā._  

</Paragraph>

</Division>
<Division number="126" book="ap1">

<Paragraph number="3856" type="verses" book="ap1">
_Dārakova ahaṃ santo,_  
_karaṇīyaṃ samāpayiṃ;_  
_Kiṃ me karaṇīyaṃ ajja,_  
_sakyaputtassa sāsane._  

</Paragraph>

</Division>
<Division number="127" book="ap1">

<Paragraph number="3857" type="verses" book="ap1">
_Satasahassito kappe,_  
_yaṃ kammamakariṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_saddhammasavane phalaṃ._  

</Paragraph>

</Division>
<Division number="128" book="ap1">

<Paragraph number="3858" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="3859" book="ap1">
Itthaṃ sudaṃ āyasmā dhammasavaniyo thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="3860" type="center" book="ap1">
Dhammasavaniyattherassāpadānaṃ navamaṃ.
</Paragraph>
