<Paragraph number="2028" type="verses" book="ap1">
_“Atthadassimhi sugate,_  
_nibbute samanantarā;_  
_Yakkhayoniṃ upapajjiṃ,_  
_yasaṃ patto cahaṃ tadā._  

</Paragraph>

</Division>
<Division number="73" book="ap1" bookNo="20" page="168">

<Paragraph number="2029" type="verses" book="ap1">
_Dulladdhaṃ vata me āsi,_  
_duppabhātaṃ duruṭṭhitaṃ;_  
_Yaṃ me bhoge vijjamāne,_  
_parinibbāyi cakkhumā._  

</Paragraph>

</Division>
<Division number="74" book="ap1" bookNo="20" page="168">

<Paragraph number="2030" type="verses" book="ap1">
_Mama saṅkappamaññāya,_  
_sāgaro nāma sāvako;_  
_Mamuddharitukāmo so,_  
_āgacchi mama santikaṃ._  

</Paragraph>

</Division>
<Division number="75" book="ap1" bookNo="20" page="168">

<Paragraph number="2031" type="verses" book="ap1">
_Kiṃ nu socasi mā bhāyi,_  
_cara dhammaṃ sumedhasa;_  
_Anuppadinnā buddhena,_  
_sabbesaṃ bījasampadā._  

</Paragraph>

</Division>
<Division number="76" book="ap1" bookNo="20" page="168">

<Paragraph number="2032" type="verses" book="ap1">
_Yo ce pūjeyya sambuddhaṃ,_  
_tiṭṭhantaṃ lokanāyakaṃ;_  
_Dhātuṃ sāsapamattampi,_  
_nibbutassāpi pūjaye._  

</Paragraph>

</Division>
<Division number="77" book="ap1" bookNo="20" page="168">

<Paragraph number="2033" type="verses" book="ap1">
_Same cittappasādamhi,_  
_samaṃ puññaṃ mahaggataṃ;_  
_Tasmā thūpaṃ karitvāna,_  
_pūjehi jinadhātuyo._  

</Paragraph>

</Division>
<Division number="78" book="ap1" bookNo="20" page="168">

<Paragraph number="2034" type="verses" book="ap1">
_Sāgarassa vaco sutvā,_  
_buddhathūpaṃ akāsahaṃ;_  
_Pañcavasse paricariṃ,_  
_munino thūpamuttamaṃ._  

</Paragraph>

</Division>
<Division number="79" book="ap1" bookNo="20" page="168">

<Paragraph number="2035" type="verses" book="ap1">
_Tena kammena dvipadinda,_  
_lokajeṭṭha narāsabha;_  
_Sampattiṃ anubhotvāna,_  
_arahattamapāpuṇiṃ._  

</Paragraph>

</Division>
<Division number="80" book="ap1" bookNo="20" page="168">

<Paragraph number="2036" type="verses" book="ap1">
_Bhūripaññā ca cattāro,_  
_sattakappasate ito;_  
_Sattaratanasampannā,_  
_cakkavattī mahabbalā._  

</Paragraph>

</Division>
<Division number="81" book="ap1" bookNo="20" page="168">

<Paragraph number="2037" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="2038" book="ap1">
Itthaṃ sudaṃ āyasmā paccupaṭṭhānasaññako thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="2039" type="center" book="ap1">
Paccupaṭṭhānasaññakattherassāpadānaṃ navamaṃ.
</Paragraph>
