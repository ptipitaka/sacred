<Division number="97" book="ap1">

<Paragraph number="806" type="verses" book="ap1">
_“Padumuttarassa bhagavato,_  
_Lokajeṭṭhassa tādino;_  
_Varanāgo mayā dinno,_  
_Īsādanto urūḷhavā._  

</Paragraph>

</Division>
<Division number="98" book="ap1">

<Paragraph number="807" type="verses" book="ap1">
_Setacchatto pasobhito,_  
_sakappano sahatthipo;_  
_Agghāpetvāna taṃ sabbaṃ,_  
_saṃghārāmaṃ akārayiṃ._  

</Paragraph>

</Division>
<Division number="99" book="ap1">

<Paragraph number="808" type="verses" book="ap1">
_Catupaññāsasahassāni,_  
_pāsāde kārayiṃ ahaṃ;_  
_Mahoghadānaṃ karitvāna,_  
_niyyādesiṃ mahesino._  

</Paragraph>

</Division>
<Division number="100" book="ap1">

<Paragraph number="809" type="verses" book="ap1">
_Anumodi mahāvīro,_  
_sayambhū aggapuggalo;_  
_Sabbe jane hāsayanto,_  
_desesi amataṃ padaṃ._  

</Paragraph>

</Division>
<Division number="101" book="ap1">

<Paragraph number="810" type="verses" book="ap1">
_Taṃ me buddho viyākāsi,_  
_jalajuttaranāmako;_  
_Bhikkhusaṃghe nisīditvā,_  
_imā gāthā abhāsatha._  

</Paragraph>

</Division>
<Division number="102" book="ap1">

<Paragraph number="811" type="verses" book="ap1">
_‘Catupaññāsasahassāni,_  
_pāsāde kārayī ayaṃ;_  
_Kathayissāmi vipākaṃ,_  
_suṇotha mama bhāsato._  

</Paragraph>

</Division>
<Division number="103" book="ap1">

<Paragraph number="812" type="verses" book="ap1">
_Aṭṭhārasasahassāni,_  
_kūṭāgārā bhavissare;_  
_Byamhuttamamhi nibbattā,_  
_sabbasoṇṇamayā ca te._  

</Paragraph>

</Division>
<Division number="104" book="ap1">

<Paragraph number="813" type="verses" book="ap1">
_Paññāsakkhattuṃ devindo,_  
_devarajjaṃ karissati;_  
_Aṭṭhapaññāsakkhattuñca,_  
_cakkavattī bhavissati._  

</Paragraph>

</Division>
<Division number="105" book="ap1">

<Paragraph number="814" type="verses" book="ap1">
_Kappasatasahassamhi,_  
_okkākakulasambhavo;_  
_Gotamo nāma gottena,_  
_satthā loke bhavissati._  

</Paragraph>

</Division>
<Division number="106" book="ap1">

<Paragraph number="815" type="verses" book="ap1">
_Devalokā cavitvāna,_  
_sukkamūlena codito;_  
_Aḍḍhe kule mahābhoge,_  
_nibbattissati tāvade._  

</Paragraph>

</Division>
<Division number="107" book="ap1">

<Paragraph number="816" type="verses" book="ap1">
_So pacchā pabbajitvāna,_  
_sukkamūlena codito;_  
_Raṭṭhapāloti nāmena,_  
_hessati satthu sāvako._  

</Paragraph>

</Division>
<Division number="108" book="ap1">

<Paragraph number="817" type="verses" book="ap1">
_Padhānapahitatto so,_  
_upasanto nirūpadhi;_  
_Sabbāsave pariññāya,_  
_nibbāyissatināsavo’._  

</Paragraph>

</Division>
<Division number="109" book="ap1">

<Paragraph number="818" type="verses" book="ap1">
_Uṭṭhāya abhinikkhamma,_  
_jahitā bhogasampadā;_  
_Kheḷapiṇḍeva bhogamhi,_  
_pemaṃ mayhaṃ na vijjati._  

</Paragraph>

</Division>
<Division number="110" book="ap1">

<Paragraph number="819" type="verses" book="ap1">
_Vīriyaṃ me dhuradhorayhaṃ,_  
_yogakkhemādhivāhanaṃ;_  
_Dhāremi antimaṃ dehaṃ,_  
_sammāsambuddhasāsane._  

</Paragraph>

</Division>
<Division number="111" book="ap1">

<Paragraph number="820" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_vimokkhāpi ca aṭṭhime;_  
_Chaḷabhiññā sacchikatā,_  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="821" book="ap1">
Itthaṃ sudaṃ āyasmā raṭṭhapālo thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="822" type="center" book="ap1">
Raṭṭhapālattherassāpadānaṃ aṭṭhamaṃ.
</Paragraph>
