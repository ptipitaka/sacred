<Paragraph number="4096" type="verses" book="ap1">
_“Pāṭaliṃ haritaṃ disvā,_  
_pādapaṃ dharaṇīruhaṃ;_  
_Ekaṃsaṃ añjaliṃ katvā,_  
_avandiṃ pāṭaliṃ ahaṃ._  

</Paragraph>

</Division>
<Division number="2" book="ap1" bookNo="20" page="335">

<Paragraph number="4097" type="verses" book="ap1">
_Añjaliṃ paggahetvāna,_  
_garuṃ katvāna mānasaṃ;_  
_Antosuddhaṃ bahisuddhaṃ,_  
_suvimuttamanāsavaṃ._  

</Paragraph>

</Division>
<Division number="3" book="ap1" bookNo="20" page="335">

<Paragraph number="4098" type="verses" book="ap1">
_Vipassiṃ lokamahitaṃ,_  
_karuṇāñāṇasāgaraṃ;_  
_Sammukhā viya sambuddhaṃ,_  
_avandiṃ pāṭaliṃ ahaṃ._  

</Paragraph>

</Division>
<Division number="4" book="ap1" bookNo="20" page="335">

<Paragraph number="4099" type="verses" book="ap1">
_Ekanavutito kappe,_  
_yaṃ bodhimabhivandahaṃ;_  
_Duggatiṃ nābhijānāmi,_  
_vandanāya idaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="5" book="ap1" bookNo="20" page="335">

<Paragraph number="4100" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_vimokkhāpi ca aṭṭhime;_  
_Chaḷabhiññā sacchikatā,_  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="4101" book="ap1">
Itthaṃ sudaṃ āyasmā bodhivandako thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="4102" type="center" book="ap1">
Bodhivandakattherassāpadānaṃ paṭhamaṃ.
</Paragraph>
