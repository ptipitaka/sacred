<Paragraph number="1446" type="verses" book="ap1">
_“Padumuttarabuddhassa,_  
_bodhiyā pādaputtame;_  
_Haṭṭho haṭṭhena cittena,_  
_dhajamāropayiṃ ahaṃ._  

</Paragraph>

</Division>
<Division number="58" book="ap1" bookNo="20" page="120">

<Paragraph number="1447" type="verses" book="ap1">
_Patitapattāni gaṇhitvā,_  
_bahiddhā chaḍḍayiṃ ahaṃ;_  
_Antosuddhaṃ bahisuddhaṃ,_  
_adhimuttamanāsavaṃ._  

</Paragraph>

</Division>
<Division number="59" book="ap1" bookNo="20" page="120">

<Paragraph number="1448" type="verses" book="ap1">
_Sammukhā viya sambuddhaṃ,_  
_avandiṃ bodhimuttamaṃ;_  
_Padumuttaro lokavidū,_  
_āhutīnaṃ paṭiggaho._  

</Paragraph>

</Division>
<Division number="60" book="ap1" bookNo="20" page="120">

<Paragraph number="1449" type="verses" book="ap1">
_Bhikkhusaṃghe ṭhito satthā,_  
_imā gāthā abhāsatha;_  
_‘Iminā dhajadānena,_  
_upaṭṭhānena cūbhayaṃ._  

</Paragraph>

</Division>
<Division number="61" book="ap1" bookNo="20" page="120">

<Paragraph number="1450" type="verses" book="ap1">
_Kappānaṃ satasahassaṃ,_  
_duggatiṃ so na gacchati;_  
_Devesu devasobhagyaṃ,_  
_anubhossatinappakaṃ._  

</Paragraph>

</Division>
<Division number="62" book="ap1" bookNo="20" page="121">

<Paragraph number="1451" type="verses" book="ap1">
_Anekasatakkhattuñca,_  
_rājā raṭṭhe bhavissati;_  
_Uggato nāma nāmena,_  
_cakkavattī bhavissati._  

</Paragraph>

</Division>
<Division number="63" book="ap1" bookNo="20" page="121">

<Paragraph number="1452" type="verses" book="ap1">
_Sampattiṃ anubhotvāna,_  
_sukkamūlena codito;_  
_Gotamassa bhagavato,_  
_sāsanebhiramissati’._  

</Paragraph>

</Division>
<Division number="64" book="ap1" bookNo="20" page="121">

<Paragraph number="1453" type="verses" book="ap1">
_Padhānapahitattomhi,_  
_upasanto nirūpadhi;_  
_Dhāremi antimaṃ dehaṃ,_  
_sammāsambuddhasāsane._  

</Paragraph>

</Division>
<Division number="65" book="ap1" bookNo="20" page="121">

<Paragraph number="1454" type="verses" book="ap1">
_Ekapaññāsasahasse,_  
_kappe uggatasavhayo;_  
_Paññāsasatasahasse,_  
_khattiyo meghasavhayo._  

</Paragraph>

</Division>
<Division number="66" book="ap1" bookNo="20" page="121">

<Paragraph number="1455" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="1456" book="ap1">
Itthaṃ sudaṃ āyasmā dhajadāyako thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="1457" type="center" book="ap1">
Dhajadāyakattherassāpadānaṃ aṭṭhamaṃ.
</Paragraph>
