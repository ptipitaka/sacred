<Paragraph number="1352" type="verses" book="ap1">
_“Siddhatthassa bhagavato,_  
_lokajeṭṭhassa tādino;_  
_Agghiyaṃ kārayitvāna,_  
_jātipupphehi chādayiṃ._  

</Paragraph>

</Division>
<Division number="126" book="ap1" bookNo="20" page="113">

<Paragraph number="1353" type="verses" book="ap1">
_Niṭṭhāpetvāna taṃ pupphaṃ,_  
_buddhassa upanāmayiṃ;_  
_Pupphāvasesaṃ paggayha,_  
_buddhassa abhiropayiṃ._  

</Paragraph>

</Division>
<Division number="127" book="ap1" bookNo="20" page="113">

<Paragraph number="1354" type="verses" book="ap1">
_Kañcanagghiyasaṅkāsaṃ,_  
_buddhaṃ lokagganāyakaṃ;_  
_Pasannacitto sumano,_  
_pupphagghiyamupānayiṃ._  

</Paragraph>

</Division>
<Division number="128" book="ap1" bookNo="20" page="113">

<Paragraph number="1355" type="verses" book="ap1">
_Vitiṇṇakaṅkho sambuddho,_  
_tiṇṇoghehi purakkhato;_  
_Bhikkhusaṃghe nisīditvā,_  
_imā gāthā abhāsatha._  

</Paragraph>

</Division>
<Division number="129" book="ap1" bookNo="20" page="113">

<Paragraph number="1356" type="verses" book="ap1">
_‘Dibbagandhaṃ pavāyantaṃ,_  
_yo me pupphagghiyaṃ adā;_  
_Tamahaṃ kittayissāmi,_  
_suṇātha mama bhāsato._  

</Paragraph>

</Division>
<Division number="130" book="ap1" bookNo="20" page="113">

<Paragraph number="1357" type="verses" book="ap1">
_Ito cuto ayaṃ poso,_  
_devasaṅghapurakkhato;_  
_Jātipupphehi parikiṇṇo,_  
_devalokaṃ gamissati._  

</Paragraph>

</Division>
<Division number="131" book="ap1" bookNo="20" page="113">

<Paragraph number="1358" type="verses" book="ap1">
_Ubbiddhaṃ bhavanaṃ tassa,_  
_Sovaṇṇañca maṇīmayaṃ;_  
_Byamhaṃ pātubhavissati,_  
_Puññakammappabhāvitaṃ._  

</Paragraph>

</Division>
<Division number="132" book="ap1" bookNo="20" page="113">

<Paragraph number="1359" type="verses" book="ap1">
_Catusattatikkhattuṃ so,_  
_devarajjaṃ karissati;_  
_Anubhossati sampattiṃ,_  
_accharāhi purakkhato._  

</Paragraph>

</Division>
<Division number="133" book="ap1" bookNo="20" page="113">

<Paragraph number="1360" type="verses" book="ap1">
_Pathabyā rajjaṃ tisataṃ,_  
_vasudhaṃ āvasissati;_  
_Pañcasattatikkhattuñca,_  
_cakkavattī bhavissati._  

</Paragraph>

</Division>
<Division number="134" book="ap1" bookNo="20" page="113">

<Paragraph number="1361" type="verses" book="ap1">
_Dujjayo nāma nāmena,_  
_hessati manujādhipo;_  
_Anubhotvāna taṃ puññaṃ,_  
_sakakammaṃ apassito._  

</Paragraph>

</Division>
<Division number="135" book="ap1" bookNo="20" page="114">

<Paragraph number="1362" type="verses" book="ap1">
_Vinipātaṃ agantvāna,_  
_manussattaṃ gamissati;_  
_Hiraññaṃ tassa nicitaṃ,_  
_koṭisatamanappakaṃ._  

</Paragraph>

</Division>
<Division number="136" book="ap1" bookNo="20" page="114">

<Paragraph number="1363" type="verses" book="ap1">
_Nibbattissati yonimhi,_  
_brāhmaṇe so bhavissati;_  
_Vaṅgantassa suto dhīmā,_  
_sāriyā oraso piyo._  

</Paragraph>

</Division>
<Division number="137" book="ap1" bookNo="20" page="114">

<Paragraph number="1364" type="verses" book="ap1">
_So ca pacchā pabbajitvā,_  
_aṅgīrasassa sāsane;_  
_Cūḷacundoti nāmena,_  
_hessati satthu sāvako._  

</Paragraph>

</Division>
<Division number="138" book="ap1" bookNo="20" page="114">

<Paragraph number="1365" type="verses" book="ap1">
_Sāmaṇerova so santo,_  
_khīṇāsavo bhavissati;_  
_Sabbāsave pariññāya,_  
_nibbāyissatināsavo’._  

</Paragraph>

</Division>
<Division number="139" book="ap1" bookNo="20" page="114">

<Paragraph number="1366" type="verses" book="ap1">
_Upaṭṭhahiṃ mahāvīraṃ,_  
_aññe ca pesale bahū;_  
_Bhātaraṃ me cupaṭṭhāsiṃ,_  
_uttamatthassa pattiyā._  

</Paragraph>

</Division>
<Division number="140" book="ap1" bookNo="20" page="114">

<Paragraph number="1367" type="verses" book="ap1">
_Bhātaraṃ me upaṭṭhitvā,_  
_dhātuṃ pattamhi ohiya;_  
_Sambuddhaṃ upanāmesiṃ,_  
_lokajeṭṭhaṃ narāsabhaṃ._  

</Paragraph>

</Division>
<Division number="141" book="ap1" bookNo="20" page="114">

<Paragraph number="1368" type="verses" book="ap1">
_Ubho hatthehi paggayha,_  
_buddho loke sadevake;_  
_Sandassayanto taṃ dhātuṃ,_  
_kittayi aggasāvakaṃ._  

</Paragraph>

</Division>
<Division number="142" book="ap1" bookNo="20" page="114">

<Paragraph number="1369" type="verses" book="ap1">
_Cittañca suvimuttaṃ me,_  
_saddhā mayhaṃ patiṭṭhitā;_  
_Sabbāsave pariññāya,_  
_viharāmi anāsavo._  

</Paragraph>

</Division>
<Division number="143" book="ap1" bookNo="20" page="114">

<Paragraph number="1370" type="verses" book="ap1">
_Paṭisambhidānuppattā,_  
_Vimokkhāpi ca phassitā;_  
_Chaḷabhiññā sacchikatā,_  
_Kataṃ buddhassa sāsanaṃ”. (1246)_  

</Paragraph>

<Paragraph number="1371" book="ap1">
Itthaṃ sudaṃ āyasmā cundo thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="1372" type="center" book="ap1">
Cundattherassāpadānaṃ dasamaṃ.



</Paragraph>

<Paragraph number="1373" type="center" book="ap1">
Upālivaggo pañcamo.



</Paragraph>

<Paragraph number="1374" type="center" book="ap1">
Tassuddānaṃ


</Paragraph>

<Paragraph number="1375" type="verses" book="ap1">
_Upāli soṇo bhaddiyo,_  
_sanniṭṭhāpakahatthiyo;_  
_Chadanaṃ seyyacaṅkamaṃ,_  
_subhaddo cundasavhayo;_  
_Gāthāsataṃ satālīsaṃ,_  
_catasso ca taduttari._
</Paragraph>
