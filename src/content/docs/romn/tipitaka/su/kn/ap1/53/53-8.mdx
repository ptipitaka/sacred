<Paragraph number="6859" type="verses" book="ap1">
_“Ahaṃ pure bodhipattaṃ,_  
_ujjhitaṃ cetiyaṅgaṇe;_  
_Taṃ gahetvāna chaḍḍesiṃ,_  
_alabhiṃ vīsatīguṇe._  

</Paragraph>

</Division>
<Division number="49" book="ap1" bookNo="20" page="433">

<Paragraph number="6860" type="verses" book="ap1">
_Tassa kammassa tejena,_  
_saṃsaranto bhavābhave;_  
_Duve bhave saṃsarāmi,_  
_devatte cāpi mānuse._  

</Paragraph>

</Division>
<Division number="50" book="ap1" bookNo="20" page="433">

<Paragraph number="6861" type="verses" book="ap1">
_Devalokā cavitvāna,_  
_āgantvā mānusaṃ bhavaṃ;_  
_Duve kule pajāyāmi,_  
_khattiye cāpi brāhmaṇe._  

</Paragraph>

</Division>
<Division number="51" book="ap1" bookNo="20" page="433">

<Paragraph number="6862" type="verses" book="ap1">
_Aṅgapaccaṅgasampanno,_  
_ārohapariṇāhavā;_  
_Abhirūpo suci homi,_  
_sampuṇṇaṅgo anūnako._  

</Paragraph>

</Division>
<Division number="52" book="ap1" bookNo="20" page="433">

<Paragraph number="6863" type="verses" book="ap1">
_Devaloke manusse vā,_  
_jāto vā yattha katthaci;_  
_Bhave suvaṇṇavaṇṇo ca,_  
_uttattakanakūpamo._  

</Paragraph>

</Division>
<Division number="53" book="ap1" bookNo="20" page="433">

<Paragraph number="6864" type="verses" book="ap1">
_Mudukā maddavā sniddhā,_  
_sukhumā sukumārikā;_  
_Chavi me sabbadā hoti,_  
_bodhipatte suchaḍḍite._  

</Paragraph>

</Division>
<Division number="54" book="ap1" bookNo="20" page="433">

<Paragraph number="6865" type="verses" book="ap1">
_Yato kutoci gatīsu,_  
_sarīre samudāgate;_  
_Na limpati rajojallaṃ,_  
_vipāko pattachaḍḍite._  

</Paragraph>

</Division>
<Division number="55" book="ap1" bookNo="20" page="433">

<Paragraph number="6866" type="verses" book="ap1">
_Uṇhe vātātape tassa,_  
_aggitāpena vā pana;_  
_Gatte sedā na muccanti,_  
_vipāko pattachaḍḍite._  

</Paragraph>

</Division>
<Division number="56" book="ap1" bookNo="20" page="433">

<Paragraph number="6867" type="verses" book="ap1">
_Kuṭṭhaṃ gaṇḍo kilāso ca,_  
_tilakā piḷakā tathā;_  
_Na honti kāye daddu ca,_  
_vipāko pattachaḍḍite._  

</Paragraph>

</Division>
<Division number="57" book="ap1" bookNo="20" page="433">

<Paragraph number="6868" type="verses" book="ap1">
_Aparampi guṇaṃ tassa,_  
_nibbattati bhavābhave;_  
_Rogā na honti kāyasmiṃ,_  
_vipāko pattachaḍḍite._  

</Paragraph>

</Division>
<Division number="58" book="ap1" bookNo="20" page="433">

<Paragraph number="6869" type="verses" book="ap1">
_Aparampi guṇaṃ tassa,_  
_nibbattati bhavābhave;_  
_Na hoti cittajā pīḷā,_  
_vipāko pattachaḍḍite._  

</Paragraph>

</Division>
<Division number="59" book="ap1" bookNo="20" page="434">

<Paragraph number="6870" type="verses" book="ap1">
_Aparampi guṇaṃ tassa,_  
_nibbattati bhavābhave;_  
_Amittā na bhavantassa,_  
_vipāko pattachaḍḍite._  

</Paragraph>

</Division>
<Division number="60" book="ap1" bookNo="20" page="434">

<Paragraph number="6871" type="verses" book="ap1">
_Aparampi guṇaṃ tassa,_  
_nibbattati bhavābhave;_  
_Anūnabhogo bhavati,_  
_vipāko pattachaḍḍite._  

</Paragraph>

</Division>
<Division number="61" book="ap1" bookNo="20" page="434">

<Paragraph number="6872" type="verses" book="ap1">
_Aparampi guṇaṃ tassa,_  
_nibbattati bhavābhave;_  
_Aggirājūhi corehi,_  
_na hoti udake bhayaṃ._  

</Paragraph>

</Division>
<Division number="62" book="ap1" bookNo="20" page="434">

<Paragraph number="6873" type="verses" book="ap1">
_Aparampi guṇaṃ tassa,_  
_nibbattati bhavābhave;_  
_Dāsidāsā anucarā,_  
_honti cittānuvattakā._  

</Paragraph>

</Division>
<Division number="63" book="ap1" bookNo="20" page="434">

<Paragraph number="6874" type="verses" book="ap1">
_Yamhi āyuppamāṇamhi,_  
_jāyate mānuse bhave;_  
_Tato na hāyate āyu,_  
_tiṭṭhate yāvatāyukaṃ._  

</Paragraph>

</Division>
<Division number="64" book="ap1" bookNo="20" page="434">

<Paragraph number="6875" type="verses" book="ap1">
_Abbhantarā ca bāhirā,_  
_negamā ca saraṭṭhakā;_  
_Nuyuttā honti sabbepi,_  
_vuddhikāmā sukhicchakā._  

</Paragraph>

</Division>
<Division number="65" book="ap1" bookNo="20" page="434">

<Paragraph number="6876" type="verses" book="ap1">
_Bhogavā yasavā homi,_  
_sirimā ñātipakkhavā;_  
_Apetabhayasantāso,_  
_bhavehaṃ sabbato bhave._  

</Paragraph>

</Division>
<Division number="66" book="ap1" bookNo="20" page="434">

<Paragraph number="6877" type="verses" book="ap1">
_Devā manussā asurā,_  
_gandhabbā yakkharakkhasā;_  
_Sabbe te parirakkhanti,_  
_bhave saṃsarato sadā._  

</Paragraph>

</Division>
<Division number="67" book="ap1" bookNo="20" page="434">

<Paragraph number="6878" type="verses" book="ap1">
_Devaloke manusse ca,_  
_anubhotvā ubho yase;_  
_Avasāne ca nibbānaṃ,_  
_sivaṃ patto anuttaraṃ._  

</Paragraph>

</Division>
<Division number="68" book="ap1" bookNo="20" page="434">

<Paragraph number="6879" type="verses" book="ap1">
_Sambuddhamuddisitvāna,_  
_bodhiṃ vā tassa satthuno;_  
_Yo puññaṃ pasave poso,_  
_tassa kiṃ nāma dullabhaṃ._  

</Paragraph>

</Division>
<Division number="69" book="ap1" bookNo="20" page="434">

<Paragraph number="6880" type="verses" book="ap1">
_Magge phale āgame ca,_  
_jhānābhiññāguṇesu ca;_  
_Aññesaṃ adhiko hutvā,_  
_nibbāyāmi anāsavo._  

</Paragraph>

</Division>
<Division number="70" book="ap1" bookNo="20" page="434">

<Paragraph number="6881" type="verses" book="ap1">
_Purehaṃ bodhiyā pattaṃ,_  
_chaḍḍetvā haṭṭhamānaso;_  
_Imehi vīsataṅgehi,_  
_samaṅgī homi sabbadā._  

</Paragraph>

</Division>
<Division number="71" book="ap1" bookNo="20" page="434">

<Paragraph number="6882" type="verses" book="ap1">
_Kilesā jhāpitā mayhaṃ,_  
_ …pe… _  
_viharāmi anāsavo._  

</Paragraph>

</Division>
<Division number="72" book="ap1" bookNo="20" page="435">

<Paragraph number="6883" type="verses" book="ap1">
_Svāgataṃ vata me āsi,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ._  

</Paragraph>

</Division>
<Division number="73" book="ap1" bookNo="20" page="435">

<Paragraph number="6884" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="6885" book="ap1">
Itthaṃ sudaṃ āyasmā bodhisammajjako thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="6886" type="center" book="ap1">
Bodhisammajjakattherassāpadānaṃ aṭṭhamaṃ.
