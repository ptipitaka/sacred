<Division number="79" book="ap1">

<Paragraph number="5331" type="verses" book="ap1">
_“Jātiṃ jarañca maraṇaṃ,_  
_paccavekkhiṃ ahaṃ tadā;_  
_Ekako abhinikkhamma,_  
_pabbajiṃ anagāriyaṃ._  

</Paragraph>

</Division>
<Division number="80" book="ap1">

<Paragraph number="5332" type="verses" book="ap1">
_Caramānonupubbena,_  
_gaṅgātīraṃ upāgamiṃ;_  
_Tatthaddasāsiṃ pathaviṃ,_  
_gaṅgātīre samunnataṃ._  

</Paragraph>

</Division>
<Division number="81" book="ap1">

<Paragraph number="5333" type="verses" book="ap1">
_Assamaṃ tattha māpetvā,_  
_vasāmi assame ahaṃ;_  
_Sukato caṅkamo mayhaṃ,_  
_nānādijagaṇāyuto._  

</Paragraph>

</Division>
<Division number="82" book="ap1">

<Paragraph number="5334" type="verses" book="ap1">
_Mamupenti ca vissatthā,_  
_kūjanti ca manoharaṃ;_  
_Ramamāno saha tehi,_  
_vasāmi assame ahaṃ._  

</Paragraph>

</Division>
<Division number="83" book="ap1">

<Paragraph number="5335" type="verses" book="ap1">
_Mama assamasāmantā,_  
_migarājā catukkamo;_  
_Āsayā abhinikkhamma,_  
_gajji so asanī viya._  

</Paragraph>

</Division>
<Division number="84" book="ap1">

<Paragraph number="5336" type="verses" book="ap1">
_Nadite migarāje ca,_  
_hāso me udapajjatha;_  
_Migarājaṃ gavesanto,_  
_addasaṃ lokanāyakaṃ._  

</Paragraph>

</Division>
<Division number="85" book="ap1">

<Paragraph number="5337" type="verses" book="ap1">
_Disvānāhaṃ devadevaṃ,_  
_tissaṃ lokagganāyakaṃ;_  
_Haṭṭho haṭṭhena cittena,_  
_pūjayiṃ nāgakesaraṃ._  

</Paragraph>

</Division>
<Division number="86" book="ap1">

<Paragraph number="5338" type="verses" book="ap1">
_Uggacchantaṃva sūriyaṃ,_  
_sālarājaṃva pupphitaṃ;_  
_Osadhiṃva virocantaṃ,_  
_santhaviṃ lokanāyakaṃ._  

</Paragraph>

</Division>
<Division number="87" book="ap1">

<Paragraph number="5339" type="verses" book="ap1">
_‘Tava ñāṇena sabbaññu,_  
_mocesimaṃ sadevakaṃ;_  
_Tavaṃ ārādhayitvāna,_  
_jātiyā parimuccare._  

</Paragraph>

</Division>
<Division number="88" book="ap1">

<Paragraph number="5340" type="verses" book="ap1">
_Adassanena sabbaññu,_  
_buddhānaṃ sabbadassinaṃ;_  
_Patantivīcinirayaṃ,_  
_rāgadosehi ophuṭā._  

</Paragraph>

</Division>
<Division number="89" book="ap1">

<Paragraph number="5341" type="verses" book="ap1">
_Tava dassanamāgamma,_  
_sabbaññu lokanāyaka;_  
_Pamuccanti bhavā sabbā,_  
_phusanti amataṃ padaṃ._  

</Paragraph>

</Division>
<Division number="90" book="ap1">

<Paragraph number="5342" type="verses" book="ap1">
_Yadā buddhā cakkhumanto,_  
_uppajjanti pabhaṅkarā;_  
_Kilese jhāpayitvāna,_  
_ālokaṃ dassayanti te’._  

</Paragraph>

</Division>
<Division number="91" book="ap1">

<Paragraph number="5343" type="verses" book="ap1">
_Kittayitvāna sambuddhaṃ,_  
_tissaṃ lokagganāyakaṃ;_  
_Haṭṭho haṭṭhena cittena,_  
_tiṇasūlaṃ apūjayiṃ._  

</Paragraph>

</Division>
<Division number="92" book="ap1">

<Paragraph number="5344" type="verses" book="ap1">
_Mama saṅkappamaññāya,_  
_tisso lokagganāyako;_  
_Sakāsane nisīditvā,_  
_imā gāthā abhāsatha._  

</Paragraph>

</Division>
<Division number="93" book="ap1">

<Paragraph number="5345" type="verses" book="ap1">
_‘Yo maṃ pupphehi chādesi,_  
_pasanno sehi pāṇibhi;_  
_Tamahaṃ kittayissāmi,_  
_suṇātha mama bhāsato._  

</Paragraph>

</Division>
<Division number="94" book="ap1">

<Paragraph number="5346" type="verses" book="ap1">
_Pañcavīsatikkhattuṃ so,_  
_devarajjaṃ karissati;_  
_Pañcasattatikkhattuñca,_  
_cakkavattī bhavissati._  

</Paragraph>

</Division>
<Division number="95" book="ap1">

<Paragraph number="5347" type="verses" book="ap1">
_Padesarajjaṃ vipulaṃ,_  
_gaṇanāto asaṅkhiyaṃ;_  
_Tassa kammanissandena,_  
_pupphānaṃ pūjanāya ca._  

</Paragraph>

</Division>
<Division number="96" book="ap1">

<Paragraph number="5348" type="verses" book="ap1">
_Sīsaṃnhāto cayaṃ poso,_  
_pupphamākaṅkhate yadi;_  
_Puññakammena saṃyuttaṃ,_  
_purato pātubhavissati._  

</Paragraph>

</Division>
<Division number="97" book="ap1">

<Paragraph number="5349" type="verses" book="ap1">
_Yaṃ yaṃ icchati kāmehi,_  
_taṃ taṃ pātubhavissati;_  
_Saṅkappaṃ paripūretvā,_  
_nibbāyissatināsavo’._  


</Paragraph>
</Division>

<Paragraph number="5350" type="center" book="ap1">
Aṭṭhārasamaṃ bhāṇavāraṃ.


</Paragraph>

<Division number="98" book="ap1">

<Paragraph number="5351" type="verses" book="ap1">
_Kilese jhāpayitvāna,_  
_sampajāno patissato;_  
_Ekāsane nisīditvā,_  
_arahattamapāpuṇiṃ._  

</Paragraph>

</Division>
<Division number="99" book="ap1">

<Paragraph number="5352" type="verses" book="ap1">
_Caṅkamanto nipajjanto,_  
_nisinno uda vā ṭhito;_  
_Buddhaseṭṭhaṃ saritvāna,_  
_viharāmi ahaṃ sadā._  

</Paragraph>

</Division>
<Division number="100" book="ap1">

<Paragraph number="5353" type="verses" book="ap1">
_Cīvare piṇḍapāte ca,_  
_paccaye sayanāsane;_  
_Tattha me ūnatā natthi,_  
_buddhapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="101" book="ap1">

<Paragraph number="5354" type="verses" book="ap1">
_So dāni patto amataṃ,_  
_santaṃ padamanuttaraṃ;_  
_Sabbāsave pariññāya,_  
_viharāmi anāsavo._  

</Paragraph>

</Division>
<Division number="102" book="ap1">

<Paragraph number="5355" type="verses" book="ap1">
_Dvenavute ito kappe,_  
_yaṃ pupphamabhipūjayiṃ;_  
_Duggatiṃ nābhijānāmi,_  
_buddhapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="103" book="ap1">

<Paragraph number="5356" type="verses" book="ap1">
_Kilesā jhāpitā mayhaṃ,_  
_ …pe… _  
_viharāmi anāsavo._  

</Paragraph>

</Division>
<Division number="104" book="ap1">

<Paragraph number="5357" type="verses" book="ap1">

_ …pe… _  
_kataṃ buddhassa sāsanaṃ._  

</Paragraph>

</Division>
<Division number="105" book="ap1">

<Paragraph number="5358" type="verses" book="ap1">

_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="5359" book="ap1">

</Paragraph>
</Division>
