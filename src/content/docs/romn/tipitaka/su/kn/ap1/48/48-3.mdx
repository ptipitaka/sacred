_“Kosiyo nāma bhagavā,_  
_cittakūṭe vasī tadā;_  
_Jhāyī jhānarato buddho,_  
_vivekābhirato muni._  

</Paragraph>

<Division number="31" book="ap1" bookNo="20" page="431">

<Paragraph number="6100" type="verses" book="ap1">
_Ajjhogāhetvā himavantaṃ,_  
_Nārīgaṇapurakkhato;_  
_Addasaṃ kosiyaṃ buddhaṃ,_  
_Puṇṇamāyeva candimaṃ._  

</Paragraph>

</Division>
<Division number="32" book="ap1" bookNo="20" page="431">

<Paragraph number="6101" type="verses" book="ap1">
_Ukkāsate gahetvāna,_  
_parivāresahaṃ tadā;_  
_Sattarattindivaṃ ṭhatvā,_  
_aṭṭhamena agacchahaṃ._  

</Paragraph>

</Division>
<Division number="33" book="ap1" bookNo="20" page="431">

<Paragraph number="6102" type="verses" book="ap1">
_Vuṭṭhitaṃ kosiyaṃ buddhaṃ,_  
_sayambhuṃ aparājitaṃ;_  
_Pasannacitto vanditvā,_  
_ekaṃ bhikkhaṃ adāsahaṃ._  

</Paragraph>

</Division>
<Division number="34" book="ap1" bookNo="20" page="432">

<Paragraph number="6103" type="verses" book="ap1">
_Tena kammena dvipadinda,_  
_lokajeṭṭha narāsabha;_  
_Uppajjiṃ tusite kāye,_  
_ekabhikkhāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="35" book="ap1" bookNo="20" page="432">

<Paragraph number="6104" type="verses" book="ap1">
_Divasañceva rattiñca,_  
_āloko hoti me sadā;_  
_Samantā yojanasataṃ,_  
_obhāsena pharāmahaṃ._  

</Paragraph>

</Division>
<Division number="36" book="ap1" bookNo="20" page="432">

<Paragraph number="6105" type="verses" book="ap1">
_Pañcapaññāsakappamhi,_  
_cakkavattī ahosahaṃ;_  
_Cāturanto vijitāvī,_  
_jambumaṇḍassa issaro._  

</Paragraph>

</Division>
<Division number="37" book="ap1" bookNo="20" page="432">

<Paragraph number="6106" type="verses" book="ap1">
_Tadā me nagaraṃ āsi,_  
_iddhaṃ phītaṃ sunimmitaṃ;_  
_Tiṃsayojanamāyāmaṃ,_  
_vitthārena ca vīsati._  

</Paragraph>

</Division>
<Division number="38" book="ap1" bookNo="20" page="432">

<Paragraph number="6107" type="verses" book="ap1">
_Sobhanaṃ nāma nagaraṃ,_  
_vissakammena māpitaṃ;_  
_Dasasaddāvivittaṃ taṃ,_  
_sammatāḷasamāhitaṃ._  

</Paragraph>

</Division>
<Division number="39" book="ap1" bookNo="20" page="432">

<Paragraph number="6108" type="verses" book="ap1">
_Na tamhi nagare atthi,_  
_vallikaṭṭhañca mattikā;_  
_Sabbasoṇṇamayaṃyeva,_  
_jotate niccakālikaṃ._  

</Paragraph>

</Division>
<Division number="40" book="ap1" bookNo="20" page="432">

<Paragraph number="6109" type="verses" book="ap1">
_Catupākāraparikkhittaṃ,_  
_tayo āsuṃ maṇimayā;_  
_Vemajjhe tālapantī ca,_  
_vissakammena māpitā._  

</Paragraph>

</Division>
<Division number="41" book="ap1" bookNo="20" page="432">

<Paragraph number="6110" type="verses" book="ap1">
_Dasasahassapokkharañño,_  
_padumuppalachāditā;_  
_Puṇḍarīkehi sañchannā,_  
_nānāgandhasamīritā._  

</Paragraph>

</Division>

_yaṃ ukkaṃ dhārayiṃ ahaṃ;_  
_Duggatiṃ nābhijānāmi,_  
_ukkādhārassidaṃ phalaṃ._  

</Paragraph>

</Division>
