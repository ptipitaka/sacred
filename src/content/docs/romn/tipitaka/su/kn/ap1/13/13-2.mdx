<Paragraph number="2065" type="verses" book="ap1">
_“Himavantassāvidūre,_  
_kukkuro nāma pabbato;_  
_Vemajjhe tassa vasati,_  
_brāhmaṇo mantapāragū._  

</Paragraph>

</Division>
<Division number="11" book="ap1" bookNo="20" page="171">

<Paragraph number="2066" type="verses" book="ap1">
_Pañca sissasahassāni,_  
_parivārenti maṃ sadā;_  
_Pubbuṭṭhāyī ca te āsuṃ,_  
_mantesu ca visāradā._  

</Paragraph>

</Division>
<Division number="12" book="ap1" bookNo="20" page="171">

<Paragraph number="2067" type="verses" book="ap1">
_Buddho loke samuppanno,_  
_taṃ vijānātha no bhavaṃ;_  
_Asītibyañjanānassa,_  
_bāttiṃsavaralakkhaṇā._  

</Paragraph>

</Division>
<Division number="13" book="ap1" bookNo="20" page="171">

<Paragraph number="2068" type="verses" book="ap1">
_Byāmappabho jinavaro,_  
_ādiccova virocati;_  
_Sissānaṃ vacanaṃ sutvā,_  
_brāhmaṇo mantapāragū._  

</Paragraph>

</Division>
<Division number="14" book="ap1" bookNo="20" page="171">

<Paragraph number="2069" type="verses" book="ap1">
_Assamā abhinikkhamma,_  
_disaṃ pucchati sissake;_  
_Yamhi dese mahāvīro,_  
_vasati lokanāyako._  

</Paragraph>

</Division>
<Division number="15" book="ap1" bookNo="20" page="171">

<Paragraph number="2070" type="verses" book="ap1">
_Tāhaṃ disaṃ namassissaṃ,_  
_jinaṃ appaṭipuggalaṃ;_  
_Udaggacitto sumano,_  
_pūjesiṃ taṃ tathāgataṃ._  

</Paragraph>

</Division>
<Division number="16" book="ap1" bookNo="20" page="171">

<Paragraph number="2071" type="verses" book="ap1">
_Etha sissā gamissāma,_  
_dakkhissāma tathāgataṃ;_  
_Vanditvā satthuno pāde,_  
_sossāma jinasāsanaṃ._  

</Paragraph>

</Division>
<Division number="17" book="ap1" bookNo="20" page="171">

<Paragraph number="2072" type="verses" book="ap1">
_Ekāhaṃ abhinikkhamma,_  
_byādhiṃ paṭilabhiṃ ahaṃ;_  
_Byādhinā pīḷito santo,_  
_sālaṃ vāsayituṃ gamiṃ._  

</Paragraph>

</Division>
<Division number="18" book="ap1" bookNo="20" page="171">

<Paragraph number="2073" type="verses" book="ap1">
_Sabbe sisse samānetvā,_  
_apucchiṃ te tathāgataṃ;_  
_Kīdisaṃ lokanāthassa,_  
_guṇaṃ paramabuddhino._  

</Paragraph>

</Division>
<Division number="19" book="ap1" bookNo="20" page="171">

<Paragraph number="2074" type="verses" book="ap1">
_Te me puṭṭhā viyākaṃsu,_  
_Yathā dassāvino tathā;_  
_Sakkaccaṃ buddhaseṭṭhaṃ taṃ,_  
_Desesuṃ mama sammukhā._  

</Paragraph>

</Division>
<Division number="20" book="ap1" bookNo="20" page="171">

<Paragraph number="2075" type="verses" book="ap1">
_Tesāhaṃ vacanaṃ sutvā,_  
_sakaṃ cittaṃ pasādayiṃ;_  
_Pupphehi thūpaṃ katvāna,_  
_tattha kālaṅkato ahaṃ._  

</Paragraph>

</Division>
<Division number="21" book="ap1" bookNo="20" page="171">

<Paragraph number="2076" type="verses" book="ap1">
_Te me sarīraṃ jhāpetvā,_  
_agamuṃ buddhasantikaṃ;_  
_Añjaliṃ paggahetvāna,_  
_satthāramabhivādayuṃ._  

</Paragraph>

</Division>
<Division number="22" book="ap1" bookNo="20" page="171">

<Paragraph number="2077" type="verses" book="ap1">
_Pupphehi thūpaṃ katvāna,_  
_sugatassa mahesino;_  
_Kappānaṃ satasahassaṃ,_  
_duggatiṃ nupapajjahaṃ._  

</Paragraph>

</Division>
<Division number="23" book="ap1" bookNo="20" page="172">

<Paragraph number="2078" type="verses" book="ap1">
_Cattālīsasahassamhi,_  
_kappe soḷasa khattiyā;_  
_Nāmenaggisamā nāma,_  
_cakkavattī mahabbalā._  

</Paragraph>

</Division>
<Division number="24" book="ap1" bookNo="20" page="172">

<Paragraph number="2079" type="verses" book="ap1">
_Vīsakappasahassamhi,_  
_rājāno cakkavattino;_  
_Ghatāsanasanāmāva,_  
_aṭṭhattiṃsa mahīpatī._  

</Paragraph>

</Division>
<Division number="25" book="ap1" bookNo="20" page="172">

<Paragraph number="2080" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="2081" book="ap1">
Itthaṃ sudaṃ āyasmā pupphathūpiyo thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="2082" type="center" book="ap1">
Pupphathūpiyattherassāpadānaṃ dutiyaṃ.
</Paragraph>
