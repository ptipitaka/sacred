<Paragraph number="5509" type="verses" book="ap1">
_“Himavantassāvidūre,_  
_gosito nāma pabbato;_  
_Assamo sukato mayhaṃ,_  
_paṇṇasālā sumāpitā._  

</Paragraph>

</Division>
<Division number="37" book="ap1" bookNo="20" page="432">

<Paragraph number="5510" type="verses" book="ap1">
_Nārado nāma nāmena,_  
_kassapo iti maṃ vidū;_  
_Suddhimaggaṃ gavesanto,_  
_vasāmi gosite tadā._  

</Paragraph>

</Division>
<Division number="38" book="ap1" bookNo="20" page="432">

<Paragraph number="5511" type="verses" book="ap1">
_Padumuttaro nāma jino,_  
_sabbadhammāna pāragū;_  
_Vivekakāmo sambuddho,_  
_agañchi anilañjasā._  

</Paragraph>

</Division>
<Division number="39" book="ap1" bookNo="20" page="432">

<Paragraph number="5512" type="verses" book="ap1">
_Vanagge gacchamānassa,_  
_disvā raṃsiṃ mahesino;_  
_Kaṭṭhamañcaṃ paññāpetvā,_  
_ajinañca apatthariṃ._  

</Paragraph>

</Division>
<Division number="40" book="ap1" bookNo="20" page="432">

<Paragraph number="5513" type="verses" book="ap1">
_Āsanaṃ paññapetvāna,_  
_sire katvāna añjaliṃ;_  
_Somanassaṃ paveditvā,_  
_idaṃ vacanamabraviṃ._  

</Paragraph>

</Division>
<Division number="41" book="ap1" bookNo="20" page="432">

<Paragraph number="5514" type="verses" book="ap1">
_‘Sallakatto mahāvīra,_  
_āturānaṃ tikicchako;_  
_Mamaṃ rogaparetassa,_  
_tikicchaṃ dehi nāyaka._  

</Paragraph>

</Division>
<Division number="42" book="ap1" bookNo="20" page="432">

<Paragraph number="5515" type="verses" book="ap1">
_Kallatthikā ye passanti,_  
_buddhaseṭṭha tuvaṃ mune;_  
_Dhuvatthasiddhiṃ papponti,_  
_etesaṃ ajaro bhave._  

</Paragraph>

</Division>
<Division number="43" book="ap1" bookNo="20" page="432">

<Paragraph number="5516" type="verses" book="ap1">
_Na me deyyadhammo atthi,_  
_pavattaphalabhojihaṃ;_  
_Idaṃ me āsanaṃ atthi,_  
_nisīda kaṭṭhamañcake’._  

</Paragraph>

</Division>
<Division number="44" book="ap1" bookNo="20" page="432">

<Paragraph number="5517" type="verses" book="ap1">
_Nisīdi tattha bhagavā,_  
_asambhītova kesarī;_  
_Muhuttaṃ vītināmetvā,_  
_idaṃ vacanamabravi._  

</Paragraph>

</Division>
<Division number="45" book="ap1" bookNo="20" page="432">

<Paragraph number="5518" type="verses" book="ap1">
_‘Vissattho hohi mā bhāyi,_  
_laddho jotiraso tayā;_  
_Yaṃ tuyhaṃ patthitaṃ sabbaṃ,_  
_paripūrissatināgate._  

</Paragraph>

</Division>
<Division number="46" book="ap1" bookNo="20" page="433">

<Paragraph number="5519" type="verses" book="ap1">
_Na moghaṃ taṃ kataṃ tuyhaṃ,_  
_puññakkhette anuttare;_  
_Sakkā uddharituṃ attā,_  
_yassa cittaṃ paṇīhitaṃ._  

</Paragraph>

</Division>
<Division number="47" book="ap1" bookNo="20" page="433">

<Paragraph number="5520" type="verses" book="ap1">
_Imināsanadānena,_  
_cetanāpaṇidhīhi ca;_  
_Kappasatasahassāni,_  
_vinipātaṃ na gacchasi._  

</Paragraph>

</Division>
<Division number="48" book="ap1" bookNo="20" page="433">

<Paragraph number="5521" type="verses" book="ap1">
_Paññāsakkhattuṃ devindo,_  
_devarajjaṃ karissasi;_  
_Asītikkhattuṃ rājā ca,_  
_cakkavattī bhavissasi._  

</Paragraph>

</Division>
<Division number="49" book="ap1" bookNo="20" page="433">

<Paragraph number="5522" type="verses" book="ap1">
_Padesarajjaṃ vipulaṃ,_  
_gaṇanāto asaṅkhiyaṃ;_  
_Sabbattha sukhito hutvā,_  
_saṃsāre saṃsarissasi’._  

</Paragraph>

</Division>
<Division number="50" book="ap1" bookNo="20" page="433">

<Paragraph number="5523" type="verses" book="ap1">
_Idaṃ vatvāna sambuddho,_  
_jalajuttamanāmako;_  
_Nabhaṃ abbhuggamī vīro,_  
_haṃsarājāva ambare._  

</Paragraph>

</Division>
<Division number="51" book="ap1" bookNo="20" page="433">

<Paragraph number="5524" type="verses" book="ap1">
_Hatthiyānaṃ assayānaṃ,_  
_sarathaṃ sandamānikaṃ;_  
_Labhāmi sabbamevetaṃ,_  
_ekāsanassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="52" book="ap1" bookNo="20" page="433">

<Paragraph number="5525" type="verses" book="ap1">
_Kānanaṃ pavisitvāpi,_  
_yadā icchāmi āsanaṃ;_  
_Mama saṅkappamaññāya,_  
_pallaṅko upatiṭṭhati._  

</Paragraph>

</Division>
<Division number="53" book="ap1" bookNo="20" page="433">

<Paragraph number="5526" type="verses" book="ap1">
_Vārimajjhagato santo,_  
_yadā icchāmi āsanaṃ;_  
_Mama saṅkappamaññāya,_  
_pallaṅko upatiṭṭhati._  

</Paragraph>

</Division>
<Division number="54" book="ap1" bookNo="20" page="433">

<Paragraph number="5527" type="verses" book="ap1">
_Yaṃ yaṃ yonupapajjāmi,_  
_devattaṃ atha mānusaṃ;_  
_Pallaṅkasatasahassāni,_  
_parivārenti maṃ sadā._  

</Paragraph>

</Division>
<Division number="55" book="ap1" bookNo="20" page="433">

<Paragraph number="5528" type="verses" book="ap1">
_Duve bhave saṃsarāmi,_  
_devatte atha mānuse;_  
_Duve kule pajāyāmi,_  
_khattiye atha brāhmaṇe._  

</Paragraph>

</Division>
<Division number="56" book="ap1" bookNo="20" page="433">

<Paragraph number="5529" type="verses" book="ap1">
_Ekāsanaṃ daditvāna,_  
_puññakkhette anuttare;_  
_Dhammapallaṅkamādāya,_  
_viharāmi anāsavo._  

</Paragraph>

</Division>
<Division number="57" book="ap1" bookNo="20" page="433">

<Paragraph number="5530" type="verses" book="ap1">
_Satasahassito kappe,_  
_yaṃ dānamadadiṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_ekāsanassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="58" book="ap1" bookNo="20" page="433">

<Paragraph number="5531" type="verses" book="ap1">
_Kilesā jhāpitā mayhaṃ,_  
_ …pe… _  
_viharāmi anāsavo._  

</Paragraph>

</Division>
<Division number="59" book="ap1" bookNo="20" page="434">

<Paragraph number="5532" type="verses" book="ap1">
_Svāgataṃ vata me āsi,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ._  

</Paragraph>

</Division>
<Division number="60" book="ap1" bookNo="20" page="434">

<Paragraph number="5533" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="5534" book="ap1">
Itthaṃ sudaṃ āyasmā ekāsanadāyako thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="5535" type="center" book="ap1">
Ekāsanadāyakattherassāpadānaṃ tatiyaṃ.
