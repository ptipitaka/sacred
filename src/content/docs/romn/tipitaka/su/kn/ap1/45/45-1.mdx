<Division number="1" book="ap1">

<Paragraph number="5775" type="verses" book="ap1">
_“Kakusandho mahāvīro,_  
_sabbadhammāna pāragū;_  
_Gaṇamhā vūpakaṭṭho so,_  
_agamāsi vanantaraṃ._  

</Paragraph>

</Division>
<Division number="2" book="ap1">

<Paragraph number="5776" type="verses" book="ap1">
_Bījamiñjaṃ gahetvāna,_  
_latāya āvuṇiṃ ahaṃ;_  
_Bhagavā tamhi samaye,_  
_jhāyate pabbatantare._  

</Paragraph>

</Division>
<Division number="3" book="ap1">

<Paragraph number="5777" type="verses" book="ap1">
_Disvānahaṃ devadevaṃ,_  
_vippasannena cetasā;_  
_Dakkhiṇeyyassa vīrassa,_  
_bījamiñjamadāsahaṃ._  

</Paragraph>

</Division>
<Division number="4" book="ap1">

<Paragraph number="5778" type="verses" book="ap1">
_Imasmiṃyeva kappamhi,_  
_yaṃ miñjamadadiṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_bījamiñjassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="5" book="ap1">

<Paragraph number="5779" type="verses" book="ap1">
_Kilesā jhāpitā mayhaṃ,_  
_bhavā sabbe samūhatā;_  
_Nāgova bandhanaṃ chetvā,_  
_viharāmi anāsavo._  

</Paragraph>

</Division>
<Division number="6" book="ap1">

<Paragraph number="5780" type="verses" book="ap1">
_Svāgataṃ vata me āsi,_  
_mama buddhassa santike;_  
_Tisso vijjā anuppattā,_  
_kataṃ buddhassa sāsanaṃ._  

</Paragraph>

</Division>
<Division number="7" book="ap1">

<Paragraph number="5781" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_vimokkhāpi ca aṭṭhime;_  
_Chaḷabhiññā sacchikatā,_  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="5782" book="ap1">
Itthaṃ sudaṃ āyasmā vibhītakamiñjiyo thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="5783" type="center" book="ap1">
Vibhītakamiñjiyattherassāpadānaṃ paṭhamaṃ.
</Paragraph>
