_vedi kārāpitā mayā;_  
_Ārakkho ca mayā dinno,_  
_sugatassa mahesino._

<Division number="20" book="ap1" bookNo="20" page="241">

<Paragraph number="2924" type="verses" book="ap1">
_Tena kammavisesena,_  
_na passiṃ bhayabheravaṃ;_  
_Kuhiñci upapannassa,_  
_tāso mayhaṃ na vijjati._  

</Paragraph>

</Division>
<Division number="21" book="ap1" bookNo="20" page="241">

<Paragraph number="2925" type="verses" book="ap1">
_Catunnavutito kappe,_  
_yaṃ vediṃ kārayiṃ pure;_  
_Duggatiṃ nābhijānāmi,_  
_vedikāya idaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="22" book="ap1" bookNo="20" page="241">

<Paragraph number="2926" type="verses" book="ap1">

_apassenasanāmako;_  
_Sattaratanasampanno,_  
_cakkavattī mahabbalo._  

</Paragraph>

</Division>

_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="2928" book="ap1">
