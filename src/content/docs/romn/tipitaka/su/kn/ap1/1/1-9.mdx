<Division number="596" book="ap1">

<Paragraph number="616" type="verses" book="ap1">
_“Padumuttarasambuddhaṃ,_  
_lokajeṭṭhaṃ vināyakaṃ;_  
_Buddhabhūmimanuppattaṃ,_  
_paṭhamaṃ addasaṃ ahaṃ._  

</Paragraph>

</Division>
<Division number="597" book="ap1">

<Paragraph number="617" type="verses" book="ap1">
_Yāvatā bodhiyā mūle,_  
_yakkhā sabbe samāgatā;_  
_Sambuddhaṃ parivāretvā,_  
_vandanti pañjalīkatā._  

</Paragraph>

</Division>
<Division number="598" book="ap1">

<Paragraph number="618" type="verses" book="ap1">
_Sabbe devā tuṭṭhamanā,_  
_ākāse sañcaranti te;_  
_Buddho ayaṃ anuppatto,_  
_andhakāratamonudo._  

</Paragraph>

</Division>
<Division number="599" book="ap1">

<Paragraph number="619" type="verses" book="ap1">
_Tesaṃ hāsaparetānaṃ,_  
_mahānādo avattatha;_  
_Kilese jhāpayissāma,_  
_sammāsambuddhasāsane._  

</Paragraph>

</Division>
<Division number="600" book="ap1">

<Paragraph number="620" type="verses" book="ap1">
_Devānaṃ giramaññāya,_  
_vācāsabhimudīrihaṃ;_  
_Haṭṭho haṭṭhena cittena,_  
_ādibhikkhamadāsahaṃ._  

</Paragraph>

</Division>
<Division number="601" book="ap1">

<Paragraph number="621" type="verses" book="ap1">
_Mama saṅkappamaññāya,_  
_satthā loke anuttaro;_  
_Devasaṅghe nisīditvā,_  
_imā gāthā abhāsatha._  

</Paragraph>

</Division>
<Division number="602" book="ap1">

<Paragraph number="622" type="verses" book="ap1">
_Sattāhaṃ abhinikkhamma,_  
_bodhiṃ ajjhagamaṃ ahaṃ;_  
_Idaṃ me paṭhamaṃ bhattaṃ,_  
_brahmacārissa yāpanaṃ._  

</Paragraph>

</Division>
<Division number="603" book="ap1">

<Paragraph number="623" type="verses" book="ap1">
_Tusitā hi idhāgantvā,_  
_yo me bhikkhaṃ upānayi;_  
_Tamahaṃ kittayissāmi,_  
_suṇotha mama bhāsato._  

</Paragraph>

</Division>
<Division number="604" book="ap1">

<Paragraph number="624" type="verses" book="ap1">
_Tiṃsakappasahassāni,_  
_devarajjaṃ karissati;_  
_Sabbe deve abhibhotvā,_  
_tidivaṃ āvasissati._  

</Paragraph>

</Division>
<Division number="605" book="ap1">

<Paragraph number="625" type="verses" book="ap1">
_Devalokā cavitvāna,_  
_manussattaṃ gamissati;_  
_Sahassadhā cakkavattī,_  
_tattha rajjaṃ karissati._  

</Paragraph>

</Division>
<Division number="606" book="ap1">

<Paragraph number="626" type="verses" book="ap1">
_Kappasatasahassamhi,_  
_okkākakulasambhavo;_  
_Gotamo nāma gottena,_  
_satthā loke bhavissati._  

</Paragraph>

</Division>
<Division number="607" book="ap1">

<Paragraph number="627" type="verses" book="ap1">
_Tidasā so cavitvāna,_  
_manussattaṃ gamissati;_  
_Agārā pabbajitvāna,_  
_chabbassāni vasissati._  

</Paragraph>

</Division>
<Division number="608" book="ap1">

<Paragraph number="628" type="verses" book="ap1">
_‘Tato sattamake vasse,_  
_buddho saccaṃ kathessati;_  
_Koṇḍañño nāma nāmena,_  
_paṭhamaṃ sacchikāhiti’._  

</Paragraph>

</Division>
<Division number="609" book="ap1">

<Paragraph number="629" type="verses" book="ap1">
_Nikkhantenānupabbajiṃ,_  
_padhānaṃ sukataṃ mayā;_  
_Kilese jhāpanatthāya,_  
_pabbajiṃ anagāriyaṃ._  

</Paragraph>

</Division>
<Division number="610" book="ap1">

<Paragraph number="630" type="verses" book="ap1">
_Abhigantvāna sabbaññū,_  
_buddho loke sadevake;_  
_Isināme migāraññe,_  
_amatabherimāhani._  

</Paragraph>

</Division>
<Division number="611" book="ap1">

<Paragraph number="631" type="verses" book="ap1">
_So dāni patto amataṃ,_  
_santipadamanuttaraṃ;_  
_Sabbāsave pariññāya,_  
_viharāmi anāsavo._  

</Paragraph>

</Division>
<Division number="612" book="ap1">

<Paragraph number="632" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_vimokkhāpi ca aṭṭhime;_  
_Chaḷabhiññā sacchikatā,_  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="633" book="ap1">
Itthaṃ sudaṃ āyasmā aññāsikoṇḍañño thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="634" type="center" book="ap1">
Aññāsikoṇḍaññattherassāpadānaṃ sattamaṃ.
</Paragraph>
