<Paragraph number="7782" type="verses" book="ap1">
_“Ogayha yaṃ pokkharaṇiṃ,_  
_nānākuñjarasevitaṃ;_  
_Uddharāmi bhisaṃ tattha,_  
_ghāsahetu ahaṃ tadā._  

</Paragraph>

</Division>
<Division number="99" book="ap1" bookNo="20" page="437">

<Paragraph number="7783" type="verses" book="ap1">
_Bhagavā tamhi samaye,_  
_padumuttarasavhayo;_  
_Rattambaradharo buddho,_  
_gacchate anilañjase._  

</Paragraph>

</Division>
<Division number="100" book="ap1" bookNo="20" page="437">

<Paragraph number="7784" type="verses" book="ap1">
_Dhunanto paṃsukūlāni,_  
_saddaṃ assosahaṃ tadā;_  
_Uddhaṃ nijjhāyamānohaṃ,_  
_addasaṃ lokanāyakaṃ._  

</Paragraph>

</Division>
<Division number="101" book="ap1" bookNo="20" page="437">

<Paragraph number="7785" type="verses" book="ap1">
_Tattheva ṭhitako santo,_  
_āyāciṃ lokanāyakaṃ;_  
_‘Madhuṃ bhisehi sahitaṃ,_  
_khīraṃ sappiṃ muḷālikaṃ._  

</Paragraph>

</Division>
<Division number="102" book="ap1" bookNo="20" page="437">

<Paragraph number="7786" type="verses" book="ap1">
_Paṭiggaṇhātu me buddho,_  
_anukampāya cakkhumā’;_  
_Tato kāruṇiko satthā,_  
_orohitvā mahāyaso._  

</Paragraph>

</Division>
<Division number="103" book="ap1" bookNo="20" page="437">

<Paragraph number="7787" type="verses" book="ap1">
_Paṭiggaṇhi mama bhikkhaṃ,_  
_anukampāya cakkhumā;_  
_Paṭiggahetvā sambuddho,_  
_akā me anumodanaṃ._  

</Paragraph>

</Division>
<Division number="104" book="ap1" bookNo="20" page="437">

<Paragraph number="7788" type="verses" book="ap1">
_‘Sukhī hotu mahāpuñña,_  
_gati tuyhaṃ samijjhatu;_  
_Iminā bhisadānena,_  
_labhassu vipulaṃ sukhaṃ’._  

</Paragraph>

</Division>
<Division number="105" book="ap1" bookNo="20" page="437">

<Paragraph number="7789" type="verses" book="ap1">
_Idaṃ vatvāna sambuddho,_  
_jalajuttamanāmako;_  
_Bhikkhamādāya sambuddho,_  
_ākāsenāgamā jino._  

</Paragraph>

</Division>
<Division number="106" book="ap1" bookNo="20" page="437">

<Paragraph number="7790" type="verses" book="ap1">
_Tato bhisaṃ gahetvāna,_  
_āgacchiṃ mama assamaṃ;_  
_Bhisaṃ rukkhe laggetvāna,_  
_mama dānaṃ anussariṃ._  

</Paragraph>

</Division>
<Division number="107" book="ap1" bookNo="20" page="438">

<Paragraph number="7791" type="verses" book="ap1">
_Mahāvāto uṭṭhahitvā,_  
_sañcālesi vanaṃ tadā;_  
_Ākāso abhinādittha,_  
_asanī ca phalī tadā._  

</Paragraph>

</Division>
<Division number="108" book="ap1" bookNo="20" page="438">

<Paragraph number="7792" type="verses" book="ap1">
_Tato me asanīpāto,_  
_matthake nipatī tadā;_  
_Sohaṃ nisinnako santo,_  
_tattha kālaṅkato ahaṃ._  

</Paragraph>

</Division>
<Division number="109" book="ap1" bookNo="20" page="438">

<Paragraph number="7793" type="verses" book="ap1">
_Puññakammena saññutto,_  
_tusitaṃ upapajjahaṃ;_  
_Kaḷevaraṃ me patitaṃ,_  
_devaloke ramāmahaṃ._  

</Paragraph>

</Division>
<Division number="110" book="ap1" bookNo="20" page="438">

<Paragraph number="7794" type="verses" book="ap1">
_Chaḷasītisahassāni,_  
_nāriyo samalaṅkatā;_  
_Sāyaṃ pātaṃ upaṭṭhanti,_  
_bhisadānassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="111" book="ap1" bookNo="20" page="438">

<Paragraph number="7795" type="verses" book="ap1">
_Manussayonimāgantvā,_  
_sukhito homahaṃ tadā;_  
_Bhogā me ūnatā natthi,_  
_bhisadānassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="112" book="ap1" bookNo="20" page="438">

<Paragraph number="7796" type="verses" book="ap1">
_Anukampitako tena,_  
_devadevena tādinā;_  
_Sabbāsavaparikkhīṇā,_  
_natthi dāni punabbhavo._  

</Paragraph>

</Division>
<Division number="113" book="ap1" bookNo="20" page="438">

<Paragraph number="7797" type="verses" book="ap1">
_Satasahassito kappe,_  
_yaṃ bhisaṃ adadiṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_bhisadānassidaṃ phalaṃ._  

</Paragraph>

</Division>

_ …pe… _  
_viharāmi anāsavo._  

</Paragraph>

</Division>

_ …pe… _  
_kataṃ buddhassa sāsanaṃ._  

</Paragraph>

</Division>

_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>
