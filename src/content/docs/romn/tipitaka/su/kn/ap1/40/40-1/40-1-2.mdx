_sūciyo nakhachedane;_  
_Heṭṭhāchatte ṭhapāpesiṃ,_  
_kāretvā tadanucchave._  

</Paragraph>

<Paragraph number="4309" type="verses" book="ap1">

_morahatthe ca cāmare;_  
_Parissāvane teladhāre,_  
_kārayiṃ tadanucchave._  

</Paragraph>

</Division>
<Division number="13" book="ap1">

<Paragraph number="4310" type="verses" book="ap1">
_Sūcighare aṃsabaddhe,_  
_athopi kāyabandhane;_  
_Ādhārake ca sukate,_  
_kārayiṃ tadanucchave._  

</Paragraph>

</Division>
<Division number="14" book="ap1">

<Paragraph number="4311" type="verses" book="ap1">
_Paribhogabhājane ca,_  
_athopi lohathālake;_  
_Bhesajje pūrayitvāna,_  
_heṭṭhāchatte ṭhapesahaṃ._  

</Paragraph>

</Division>
<Division number="15" book="ap1">

<Paragraph number="4312" type="verses" book="ap1">
_Vacaṃ usīraṃ laṭṭhimadhuṃ,_  
_pipphalī maricāni ca;_  
_Harītakiṃ siṅgīveraṃ,_  
_sabbaṃ pūresi bhājane._  

</Paragraph>

</Division>
<Division number="16" book="ap1">

<Paragraph number="4313" type="verses" book="ap1">
_Upāhanā pādukāyo,_  
_atho udakapuñchane;_  
_Kattaradaṇḍe sukate,_  
_kārayiṃ tadanucchave._  

</Paragraph>

</Division>
<Division number="17" book="ap1">

<Paragraph number="4314" type="verses" book="ap1">
_Osadhañjananāḷī ca,_  
_salākā dhammakuttarā;_  
_Kuñcikā pañcavaṇṇehi,_  
_sibbite kuñcikāghare._  

</Paragraph>

</Division>
<Division number="18" book="ap1">

<Paragraph number="4315" type="verses" book="ap1">
_Āyoge dhūmanette ca,_  
_athopi dīpadhārake;_  
_Tumbake ca karaṇḍe ca,_  
_kārayiṃ tadanucchave._  

</Paragraph>

</Division>
<Division number="19" book="ap1">

<Paragraph number="4316" type="verses" book="ap1">
_Saṇḍāse pipphale ceva,_  
_athopi malahārake;_  
_Bhesajjathavike ceva,_  
_kārayiṃ tadanucchave._  

</Paragraph>

</Division>
<Division number="20" book="ap1">

<Paragraph number="4317" type="verses" book="ap1">
_Āsandiyo pīṭhake ca,_  
_pallaṅke caturomaye;_  
_Tadanucchave kārayitvā,_  
_heṭṭhāchatte ṭhapesahaṃ._  

</Paragraph>

</Division>
<Division number="21" book="ap1">

<Paragraph number="4318" type="verses" book="ap1">
_Uṇṇābhisī tūlabhisī,_  
_athopi pīṭhikābhisī;_  
_Bimbohane ca sukate,_  
_kārayiṃ tadanucchave._  

</Paragraph>

</Division>
<Division number="22" book="ap1">

<Paragraph number="4319" type="verses" book="ap1">
_Kuruvinde madhusitthe,_  
_telaṃ hatthappatāpakaṃ;_  
_Sipāṭiphalake sucī,_  
_mañcaṃ attharaṇena ca._  

</Paragraph>

</Division>
<Division number="23" book="ap1">

<Paragraph number="4320" type="verses" book="ap1">
_Senāsane pādapuñche,_  
_sayanāsanadaṇḍake;_  
_Dantapoṇe ca āṭalī,_  
_sīsālepanagandhake._  

</Paragraph>

</Division>
<Division number="24" book="ap1">

<Paragraph number="4321" type="verses" book="ap1">
_Araṇī phalapīṭhe ca,_  
_pattapidhānathālake;_  
_Udakassa kaṭacchū ca,_  
_cuṇṇakaṃ rajanambaṇaṃ._  

</Paragraph>

</Division>
<Division number="25" book="ap1">

<Paragraph number="4322" type="verses" book="ap1">
_Sammajjanaṃ udapattaṃ,_  
_tathā vassikasāṭikaṃ;_  
_Nisīdanaṃ kaṇḍucchādi,_  
_atha antaravāsakaṃ._  

</Paragraph>

</Division>
<Division number="26" book="ap1">

<Paragraph number="4323" type="verses" book="ap1">

_natthukaṃ mukhasodhanaṃ;_  
_Biḷaṅgaloṇaṃ pahūtañca,_  
_madhuñca dadhipānakaṃ._  

</Paragraph>

</Division>

_mukhapuñchanasuttakaṃ;_  
_Dātabbaṃ nāma yaṃ atthi,_  
_yañca kappati satthuno._  

</Paragraph>

</Division>
