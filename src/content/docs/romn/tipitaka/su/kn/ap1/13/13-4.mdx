<Paragraph number="2094" type="verses" book="ap1">
_“Nisajja pāsādavare,_  
_vipassiṃ addasaṃ jinaṃ;_  
_Kakudhaṃ vilasantaṃva,_  
_sabbaññuṃ tamanāsakaṃ._  

</Paragraph>

</Division>
<Division number="36" book="ap1">

<Paragraph number="2095" type="verses" book="ap1">
_Pāsādassāvidūre ca,_  
_gacchati lokanāyako;_  
_Pabhā niddhāvate tassa,_  
_yathā ca sataraṃsino._  

</Paragraph>

</Division>
<Division number="37" book="ap1">

<Paragraph number="2096" type="verses" book="ap1">
_Gandhodakañca paggayha,_  
_buddhaseṭṭhaṃ samokiriṃ;_  
_Tena cittappasādena,_  
_tattha kālaṅkato ahaṃ._  

</Paragraph>

</Division>
<Division number="38" book="ap1">

<Paragraph number="2097" type="verses" book="ap1">
_Ekanavutito kappe,_  
_yaṃ gandhodakamākiriṃ;_  
_Duggatiṃ nābhijānāmi,_  
_buddhapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="39" book="ap1">

<Paragraph number="2098" type="verses" book="ap1">
_Ekattiṃse ito kappe,_  
_sugandho nāma khattiyo;_  
_Sattaratanasampanno,_  
_cakkavattī mahabbalo._  

</Paragraph>

</Division>
<Division number="40" book="ap1">

<Paragraph number="2099" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="2100" book="ap1">
Itthaṃ sudaṃ āyasmā gandhodakiyo thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="2101" type="center" book="ap1">
Gandhodakiyattherassāpadānaṃ catutthaṃ.
</Paragraph>
