_deseti amataṃ padaṃ;_

_vasati gharamuttame._

<Division number="35" book="ap1">

<Paragraph number="3552" type="verses" book="ap1">
_Tāya vācāya madhurāya,_  
_saṅgaṇhāti mahājanaṃ;_  
_Ghoso ca vipulo āsi,_  
_āsīso devamānuse._  

</Paragraph>

</Division>
<Division number="36" book="ap1">

<Paragraph number="3553" type="verses" book="ap1">
_Nigghosasaddaṃ sutvāna,_  
_siddhatthassa mahesino;_  
_Sadde cittaṃ pasādetvā,_  
_avandiṃ lokanāyakaṃ._  

</Paragraph>

</Division>
<Division number="37" book="ap1">

<Paragraph number="3554" type="verses" book="ap1">
_Catunnavutito kappe,_  
_yaṃ saññamalabhiṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_buddhasaññāyidaṃ phalaṃ._  

</Paragraph>

</Division>

_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="3556" book="ap1">

</Paragraph>
