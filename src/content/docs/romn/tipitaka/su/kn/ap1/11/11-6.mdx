_“Suvaṇṇavaṇṇaṃ sambuddhaṃ,_  
_Gacchantaṃ anilañjase;_  
_Ghatāsanaṃva jalitaṃ,_  
_Ādittaṃva hutāsanaṃ._

<Division number="30" book="ap1" bookNo="20" page="157">

<Paragraph number="1891" type="verses" book="ap1">
_Pāṇinā udakaṃ gayha,_  
_ākāse ukkhipiṃ ahaṃ;_  
_Sampaṭicchi mahāvīro,_  
_buddho kāruṇiko isi._  

</Paragraph>

</Division>
<Division number="31" book="ap1" bookNo="20" page="157">

<Paragraph number="1892" type="verses" book="ap1">
_Antalikkhe ṭhito satthā,_  
_padumuttaranāmako;_  
_Mama saṅkappamaññāya,_  
_imaṃ gāthaṃ abhāsatha._  

</Paragraph>

</Division>
<Division number="32" book="ap1" bookNo="20" page="157">

<Paragraph number="1893" type="verses" book="ap1">
_‘Iminā dakadānena,_  
_pītiuppādanena ca;_  
_Kappasatasahassampi,_  
_duggatiṃ nupapajjati’._  

</Paragraph>

</Division>
<Division number="33" book="ap1" bookNo="20" page="157">

<Paragraph number="1894" type="verses" book="ap1">
_Tena kammena dvipadinda,_  
_lokajeṭṭha narāsabha;_  
_Pattomhi acalaṃ ṭhānaṃ,_  
_hitvā jayaparājayaṃ._  

</Paragraph>

</Division>
<Division number="34" book="ap1" bookNo="20" page="157">

<Paragraph number="1895" type="verses" book="ap1">
_Sahassarājanāmena,_  
_tayo te cakkavattino;_  
_Pañcasaṭṭhikappasate,_  
_cāturantā janādhipā._  

</Paragraph>

</Division>

_ …pe… _
