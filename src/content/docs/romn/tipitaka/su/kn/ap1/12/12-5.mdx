<Division number="40" book="ap1">

<Paragraph number="1988" type="verses" book="ap1">
_“Vipassī nāma bhagavā,_  
_lokajeṭṭho narāsabho;_  
_Nisinno janakāyassa,_  
_desesi amataṃ padaṃ._  

</Paragraph>

</Division>
<Division number="41" book="ap1">

<Paragraph number="1989" type="verses" book="ap1">
_Tassāhaṃ dhammaṃ sutvāna,_  
_dvipadindassa tādino;_  
_Soṇṇapupphāni cattāri,_  
_buddhassa abhiropayiṃ._  

</Paragraph>

</Division>
<Division number="42" book="ap1">

<Paragraph number="1990" type="verses" book="ap1">
_Suvaṇṇacchadanaṃ āsi,_  
_yāvatā parisā tadā;_  
_Buddhābhā ca suvaṇṇābhā,_  
_āloko vipulo ahu._  

</Paragraph>

</Division>
<Division number="43" book="ap1">

<Paragraph number="1991" type="verses" book="ap1">
_Udaggacitto sumano,_  
_vedajāto katañjalī;_  
_Vittisañjanano tesaṃ,_  
_diṭṭhadhammasukhāvaho._  

</Paragraph>

</Division>
<Division number="44" book="ap1">

<Paragraph number="1992" type="verses" book="ap1">
_Āyācitvāna sambuddhaṃ,_  
_vanditvāna ca subbataṃ;_  
_Pāmojjaṃ janayitvāna,_  
_sakaṃ bhavanupāgamiṃ._  

</Paragraph>

</Division>
<Division number="45" book="ap1">

<Paragraph number="1993" type="verses" book="ap1">
_Bhavane upaviṭṭhohaṃ,_  
_buddhaseṭṭhaṃ anussariṃ;_  
_Tena cittappasādena,_  
_tusitaṃ upapajjahaṃ._  

</Paragraph>

</Division>
<Division number="46" book="ap1">

<Paragraph number="1994" type="verses" book="ap1">
_Ekanavutito kappe,_  
_yaṃ pupphamabhiropayiṃ;_  
_Duggatiṃ nābhijānāmi,_  
_buddhapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="47" book="ap1">

<Paragraph number="1995" type="verses" book="ap1">
_Soḷasāsiṃsu rājāno,_  
_nemisammatanāmakā;_  
_Tetālīse ito kappe,_  
_cakkavattī mahabbalā._  

</Paragraph>

</Division>
<Division number="48" book="ap1">

<Paragraph number="1996" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="1997" book="ap1">
Itthaṃ sudaṃ āyasmā suvaṇṇapupphiyo thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="1998" type="center" book="ap1">
Suvaṇṇapupphiyattherassāpadānaṃ pañcamaṃ.
</Paragraph>
