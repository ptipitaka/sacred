<Paragraph number="2770" type="verses" book="ap1">
_“Atthadassī tu bhagavā,_  
_dvipadindo narāsabho;_  
_Purakkhato sāvakehi,_  
_gaṅgātīramupāgami._  

</Paragraph>

</Division>
<Division number="16" book="ap1">

<Paragraph number="2771" type="verses" book="ap1">
_Samatitti kākapeyyā,_  
_gaṅgā āsi duruttarā;_  
_Uttārayiṃ bhikkhusaṃghaṃ,_  
_buddhañca dvipaduttamaṃ._  

</Paragraph>

</Division>
<Division number="17" book="ap1">

<Paragraph number="2772" type="verses" book="ap1">
_Aṭṭhārase kappasate,_  
_yaṃ kammamakariṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_taraṇāya idaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="18" book="ap1">

<Paragraph number="2773" type="verses" book="ap1">
_Teraseto kappasate,_  
_pañca sabbobhavā ahuṃ;_  
_Sattaratanasampannā,_  
_cakkavattī mahabbalā._  

</Paragraph>

</Division>
<Division number="19" book="ap1">

<Paragraph number="2774" type="verses" book="ap1">
_Pacchime ca bhave asmiṃ,_  
_jātohaṃ brāhmaṇe kule;_  
_Saddhiṃ tīhi sahāyehi,_  
_pabbajiṃ satthu sāsane._  

</Paragraph>

</Division>
<Division number="20" book="ap1">

<Paragraph number="2775" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="2776" book="ap1">
Itthaṃ sudaṃ āyasmā taraṇiyo thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="2777" type="center" book="ap1">
Taraṇiyattherassāpadānaṃ catutthaṃ.
</Paragraph>
