<Division number="1" book="ap1">

<Paragraph number="5879" type="verses" book="ap1">
_“Dhammadassissa munino,_  
_bodhiyā pādaputtame;_  
_Pasannacitto sumano,_  
_jagatiṃ kārayiṃ ahaṃ._  

</Paragraph>

</Division>
<Division number="2" book="ap1">

<Paragraph number="5880" type="verses" book="ap1">
_Darito pabbatato vā,_  
_rukkhato patito ahaṃ;_  
_Cuto patiṭṭhaṃ vindāmi,_  
_jagatiyā idaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="3" book="ap1">

<Paragraph number="5881" type="verses" book="ap1">
_Na me corā vihesanti,_  
_nātimaññanti khattiyā;_  
_Sabbāmittetikkamāmi,_  
_jagatiyā idaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="4" book="ap1">

<Paragraph number="5882" type="verses" book="ap1">
_Yaṃ yaṃ yonupapajjāmi,_  
_devattaṃ atha mānusaṃ;_  
_Sabbattha pūjito homi,_  
_jagatiyā idaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="5" book="ap1">

<Paragraph number="5883" type="verses" book="ap1">
_Aṭṭhārase kappasate,_  
_jagatiṃ kārayiṃ ahaṃ;_  
_Duggatiṃ nābhijānāmi,_  
_jagatidānassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="6" book="ap1">

<Paragraph number="5884" type="verses" book="ap1">
_Kilesā jhāpitā mayhaṃ,_  
_bhavā sabbe samūhatā;_  
_Nāgova bandhanaṃ chetvā,_  
_viharāmi anāsavo._  

</Paragraph>

</Division>
<Division number="7" book="ap1">

<Paragraph number="5885" type="verses" book="ap1">
_Svāgataṃ vata me āsi,_  
_mama buddhassa santike;_  
_Tisso vijjā anuppattā,_  
_kataṃ buddhassa sāsanaṃ._  

</Paragraph>

</Division>
<Division number="8" book="ap1">

<Paragraph number="5886" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_vimokkhāpi ca aṭṭhime;_  
_Chaḷabhiññā sacchikatā,_  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="5887" book="ap1">
Itthaṃ sudaṃ āyasmā jagatidāyako thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="5888" type="center" book="ap1">
Jagatidāyakattherassāpadānaṃ paṭhamaṃ.
</Paragraph>
