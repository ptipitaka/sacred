<Paragraph number="1912" type="verses" book="ap1">
_“Kānanaṃ vanamogayha,_  
_appasaddaṃ nirākulaṃ;_  
_Sīhāsanaṃ mayā dinnaṃ,_  
_atthadassissa tādino._  

</Paragraph>

</Division>
<Division number="48" book="ap1" bookNo="20" page="158">

<Paragraph number="1913" type="verses" book="ap1">
_Mālāhatthaṃ gahetvāna,_  
_katvā ca naṃ padakkhiṇaṃ;_  
_Satthāraṃ payirupāsitvā,_  
_pakkāmiṃ uttarāmukho._  

</Paragraph>

</Division>
<Division number="49" book="ap1" bookNo="20" page="159">

<Paragraph number="1914" type="verses" book="ap1">
_Tena kammena dvipadinda,_  
_lokajeṭṭha narāsabha;_  
_Sannibbāpemi attānaṃ,_  
_bhavā sabbe samūhatā._  

</Paragraph>

</Division>
<Division number="50" book="ap1" bookNo="20" page="159">

<Paragraph number="1915" type="verses" book="ap1">
_Aṭṭhārasakappasate,_  
_yaṃ dānamadadiṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_sīhāsanassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="51" book="ap1" bookNo="20" page="159">

<Paragraph number="1916" type="verses" book="ap1">
_Ito sattakappasate,_  
_sannibbāpaka khattiyo;_  
_Sattaratanasampanno,_  
_cakkavattī mahabbalo._  

</Paragraph>

</Division>
<Division number="52" book="ap1" bookNo="20" page="159">

<Paragraph number="1917" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="1918" book="ap1">
Itthaṃ sudaṃ āyasmā āsanupaṭṭhāhako thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="1919" type="center" book="ap1">
Āsanupaṭṭhāhakattherassāpadānaṃ aṭṭhamaṃ.
