<Division number="42" book="ap1">

<Paragraph number="1646" type="verses" book="ap1">
_“Migaluddo pure āsiṃ,_  
_Araññe kānane ahaṃ;_  
_Tattha addasaṃ sambuddhaṃ,_  
_Bāttiṃsavaralakkhaṇaṃ._  

</Paragraph>

</Division>
<Division number="43" book="ap1">

<Paragraph number="1647" type="verses" book="ap1">
_Tatthāhaṃ añjaliṃ katvā,_  
_pakkāmiṃ pācināmukho;_  
_Avidūre nisinnassa,_  
_niyake paṇṇasanthare._  

</Paragraph>

</Division>
<Division number="44" book="ap1">

<Paragraph number="1648" type="verses" book="ap1">
_Tato me asanīpāto,_  
_matthake nipatī tadā;_  
_Sohaṃ maraṇakālamhi,_  
_akāsiṃ punarañjaliṃ._  

</Paragraph>

</Division>
<Division number="45" book="ap1">

<Paragraph number="1649" type="verses" book="ap1">
_Dvenavute ito kappe,_  
_añjaliṃ akariṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_añjalissa idaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="46" book="ap1">

<Paragraph number="1650" type="verses" book="ap1">
_Catupaṇṇāsakappamhi,_  
_migaketusanāmako;_  
_Sattaratanasampanno,_  
_cakkavattī mahabbalo._  

</Paragraph>

</Division>
<Division number="47" book="ap1">

<Paragraph number="1651" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="1652" book="ap1">
Itthaṃ sudaṃ āyasmā ṭhitañjaliyo thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="1653" type="center" book="ap1">
Ṭhitañjaliyattherassāpadānaṃ navamaṃ.
</Paragraph>
