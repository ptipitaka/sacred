* Ñāṇasaññikattheraapadāna

<Division number="7" book="ap1" bookNo="20" page="154">

<Paragraph number="1860" type="verses" book="ap1">
_“Suvaṇṇavaṇṇaṃ sambuddhaṃ,_  
_Nisabhājāniyaṃ yathā;_  
_Tidhāpabhinnaṃ mātaṅgaṃ,_  
_Kuñjaraṃva mahesinaṃ._  

</Paragraph>

</Division>
<Division number="8" book="ap1" bookNo="20" page="154">

<Paragraph number="1861" type="verses" book="ap1">
_Obhāsentaṃ disā sabbā,_  
_uḷurājaṃva pūritaṃ;_  
_Rathiyaṃ paṭipajjantaṃ,_  
_lokajeṭṭhaṃ apassahaṃ._  

</Paragraph>

</Division>
<Division number="9" book="ap1" bookNo="20" page="155">

<Paragraph number="1862" type="verses" book="ap1">
_Ñāṇe cittaṃ pasādetvā,_  
_paggahetvāna añjaliṃ;_  
_Pasannacitto sumano,_  
_siddhatthamabhivādayiṃ._  

</Paragraph>

</Division>
<Division number="10" book="ap1" bookNo="20" page="155">

<Paragraph number="1863" type="verses" book="ap1">
_Catunnavutito kappe,_  
_yaṃ kammamakariṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_ñāṇasaññāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="11" book="ap1" bookNo="20" page="155">

<Paragraph number="1864" type="verses" book="ap1">
_Tesattatimhito kappe,_  
_soḷasāsuṃ naruttamā;_  
_Sattaratanasampannā,_  
_cakkavattī mahabbalā._  

</Paragraph>

</Division>
<Division number="12" book="ap1" bookNo="20" page="155">

<Paragraph number="1865" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="1866" book="ap1">
Itthaṃ sudaṃ āyasmā ñāṇasaññiko thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="1867" type="center" book="ap1">
Ñāṇasaññikattherassāpadānaṃ dutiyaṃ.
</Paragraph>
