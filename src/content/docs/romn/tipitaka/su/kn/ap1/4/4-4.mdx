<Division number="48" book="ap1">

<Paragraph number="1126" type="verses" book="ap1">
_“Padumuttarabuddhassa,_  
_lokajeṭṭhassa tādino;_  
_Addhānaṃ paṭipannassa,_  
_carato cārikaṃ tadā._  

</Paragraph>

</Division>
<Division number="49" book="ap1">

<Paragraph number="1127" type="verses" book="ap1">
_Suphullaṃ padumaṃ gayha,_  
_uppalaṃ mallikañcahaṃ;_  
_Paramannaṃ gahetvāna,_  
_adāsiṃ satthuno ahaṃ._  

</Paragraph>

</Division>
<Division number="50" book="ap1">

<Paragraph number="1128" type="verses" book="ap1">
_Paribhuñji mahāvīro,_  
_paramannaṃ subhojanaṃ;_  
_Tañca pupphaṃ gahetvāna,_  
_janassa sampadassayi._  

</Paragraph>

</Division>
<Division number="51" book="ap1">

<Paragraph number="1129" type="verses" book="ap1">
_Iṭṭhaṃ kantaṃ piyaṃ loke,_  
_jalajaṃ pupphamuttamaṃ;_  
_Sudukkaraṃ kataṃ tena,_  
_yo me pupphaṃ adāsidaṃ._  

</Paragraph>

</Division>
<Division number="52" book="ap1">

<Paragraph number="1130" type="verses" book="ap1">
_Yo pupphamabhiropesi,_  
_paramannañcadāsi me;_  
_Tamahaṃ kittayissāmi,_  
_suṇātha mama bhāsato._  

</Paragraph>

</Division>
<Division number="53" book="ap1">

<Paragraph number="1131" type="verses" book="ap1">
_‘Dasa aṭṭha cakkhattuṃ so,_  
_devarajjaṃ karissati;_  
_Uppalaṃ padumañcāpi,_  
_mallikañca taduttari._  

</Paragraph>

</Division>
<Division number="54" book="ap1">

<Paragraph number="1132" type="verses" book="ap1">
_Assa puññavipākena,_  
_dibbagandhasamāyutaṃ;_  
_Ākāse chadanaṃ katvā,_  
_dhārayissati tāvade._  

</Paragraph>

</Division>
<Division number="55" book="ap1">

<Paragraph number="1133" type="verses" book="ap1">
_Pañcavīsatikkhattuñca,_  
_cakkavattī bhavissati;_  
_Pathabyā rajjaṃ pañcasataṃ,_  
_vasudhaṃ āvasissati._  

</Paragraph>

</Division>
<Division number="56" book="ap1">

<Paragraph number="1134" type="verses" book="ap1">
_Kappasatasahassamhi,_  
_okkākakulasambhavo;_  
_Gotamo nāma gottena,_  
_satthā loke bhavissati._  

</Paragraph>

</Division>
<Division number="57" book="ap1">

<Paragraph number="1135" type="verses" book="ap1">
_Sakakammābhiraddho so,_  
_sukkamūlena codito;_  
_Sakyānaṃ nandijanano,_  
_ñātibandhu bhavissati._  

</Paragraph>

</Division>
<Division number="58" book="ap1">

<Paragraph number="1136" type="verses" book="ap1">
_So pacchā pabbajitvāna,_  
_sukkamūlena codito;_  
_Sabbāsave pariññāya,_  
_nibbāyissatināsavo._  

</Paragraph>

</Division>
<Division number="59" book="ap1">

<Paragraph number="1137" type="verses" book="ap1">
_Paṭisambhidamanuppattaṃ,_  
_katakiccamanāsavaṃ;_  
_Gotamo lokabandhu taṃ,_  
_etadagge ṭhapessati._  

</Paragraph>

</Division>
<Division number="60" book="ap1">

<Paragraph number="1138" type="verses" book="ap1">
_Padhānapahitatto so,_  
_upasanto nirūpadhi;_  
_Udāyī nāma nāmena,_  
_hessati satthu sāvako’._  

</Paragraph>

</Division>
<Division number="61" book="ap1">

<Paragraph number="1139" type="verses" book="ap1">
_Rāgo doso ca moho ca,_  
_māno makkho ca dhaṃsito;_  
_Sabbāsave pariññāya,_  
_viharāmi anāsavo._  

</Paragraph>

</Division>
<Division number="62" book="ap1">

<Paragraph number="1140" type="verses" book="ap1">
_Tosayiñcāpi sambuddhaṃ,_  
_ātāpī nipako ahaṃ;_  
_Pasādito ca sambuddho,_  
_etadagge ṭhapesi maṃ._  

</Paragraph>

</Division>
<Division number="63" book="ap1">

<Paragraph number="1141" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="1142" book="ap1">
Itthaṃ sudaṃ āyasmā kāḷudāyī thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="1143" type="center" book="ap1">
Kāḷudāyittherassāpadānaṃ catutthaṃ.
</Paragraph>
