<Paragraph number="4694" type="verses" book="ap1">
_“Himavantassāvidūre,_  
_sobhito nāma pabbato;_  
_Assamo sukato mayhaṃ,_  
_sakasissehi māpito._  

</Paragraph>

</Division>
<Division number="387" book="ap1" bookNo="20" page="385">

<Paragraph number="4695" type="verses" book="ap1">
_Maṇḍapā ca bahū tattha,_  
_pupphitā sinduvārakā;_  
_Kapitthā ca bahū tattha,_  
_pupphitā jīvajīvakā._  

</Paragraph>

</Division>
<Division number="388" book="ap1" bookNo="20" page="385">

<Paragraph number="4696" type="verses" book="ap1">
_Nigguṇḍiyo bahū tattha,_  
_badarāmalakāni ca;_  
_Phārusakā alābū ca,_  
_puṇḍarīkā ca pupphitā._  

</Paragraph>

</Division>
<Division number="389" book="ap1" bookNo="20" page="385">

<Paragraph number="4697" type="verses" book="ap1">
_Āḷakā beluvā tattha,_  
_kadalī mātuluṅgakā;_  
_Mahānāmā bahū tattha,_  
_ajjunā ca piyaṅgukā._  

</Paragraph>

</Division>
<Division number="390" book="ap1" bookNo="20" page="385">

<Paragraph number="4698" type="verses" book="ap1">
_Kosambā saḷalā nimbā,_  
_nigrodhā ca kapitthanā;_  
_Ediso assamo mayhaṃ,_  
_sasissohaṃ tahiṃ vasiṃ._  

</Paragraph>

</Division>
<Division number="391" book="ap1" bookNo="20" page="385">

<Paragraph number="4699" type="verses" book="ap1">
_Anomadassī bhagavā,_  
_sayambhū lokanāyako;_  
_Gavesaṃ paṭisallānaṃ,_  
_mamassamamupāgami._  

</Paragraph>

</Division>
<Division number="392" book="ap1" bookNo="20" page="385">

<Paragraph number="4700" type="verses" book="ap1">
_Upetamhi mahāvīre,_  
_anomadassimahāyase;_  
_Khaṇena lokanāthassa,_  
_vātābādho samuṭṭhahi._  

</Paragraph>

</Division>
<Division number="393" book="ap1" bookNo="20" page="385">

<Paragraph number="4701" type="verses" book="ap1">
_Vicaranto araññamhi,_  
_addasaṃ lokanāyakaṃ;_  
_Upagantvāna sambuddhaṃ,_  
_cakkhumantaṃ mahāyasaṃ._  

</Paragraph>

</Division>
<Division number="394" book="ap1" bookNo="20" page="385">

<Paragraph number="4702" type="verses" book="ap1">
_Iriyañcāpi disvāna,_  
_upalakkhesahaṃ tadā;_  
_Asaṃsayañhi buddhassa,_  
_byādhi no udapajjatha._  

</Paragraph>

</Division>
<Division number="395" book="ap1" bookNo="20" page="385">

<Paragraph number="4703" type="verses" book="ap1">
_Khippaṃ assamamāgañchiṃ,_  
_mama sissāna santike;_  
_Bhesajjaṃ kattukāmohaṃ,_  
_sisse āmantayiṃ tadā._  

</Paragraph>

</Division>
<Division number="396" book="ap1" bookNo="20" page="385">

<Paragraph number="4704" type="verses" book="ap1">
_Paṭissuṇitvāna me vākyaṃ,_  
_sissā sabbe sagāravā;_  
_Ekajjhaṃ sannipatiṃsu,_  
_satthugāravatā mama._  

</Paragraph>

</Division>
<Division number="397" book="ap1" bookNo="20" page="385">

<Paragraph number="4705" type="verses" book="ap1">
_Khippaṃ pabbatamāruyha,_  
_sabbosadhamahāsahaṃ;_  
_Pānīyayogaṃ katvāna,_  
_buddhaseṭṭhassadāsahaṃ._  

</Paragraph>

</Division>
<Division number="398" book="ap1" bookNo="20" page="386">

<Paragraph number="4706" type="verses" book="ap1">
_Paribhutte mahāvīre,_  
_sabbaññulokanāyake;_  
_Khippaṃ vāto vūpasami,_  
_sugatassa mahesino._  

</Paragraph>

</Division>
<Division number="399" book="ap1" bookNo="20" page="386">

<Paragraph number="4707" type="verses" book="ap1">
_Passaddhaṃ darathaṃ disvā,_  
_anomadassī mahāyaso;_  
_Sakāsane nisīditvā,_  
_imā gāthā abhāsatha._  

</Paragraph>

</Division>
<Division number="400" book="ap1" bookNo="20" page="386">

<Paragraph number="4708" type="verses" book="ap1">
_‘Yo me pādāsi bhesajjaṃ,_  
_byādhiñca samayī mama;_  
_Tamahaṃ kittayissāmi,_  
_suṇātha mama bhāsato._  

</Paragraph>

</Division>
<Division number="401" book="ap1" bookNo="20" page="386">

<Paragraph number="4709" type="verses" book="ap1">
_Kappasatasahassāni,_  
_devaloke ramissati;_  
_Vādite tūriye tattha,_  
_modissati sadā ayaṃ._  

</Paragraph>

</Division>
<Division number="402" book="ap1" bookNo="20" page="386">

<Paragraph number="4710" type="verses" book="ap1">
_Manussalokamāgantvā,_  
_sukkamūlena codito;_  
_Sahassakkhattuṃ rājā ca,_  
_cakkavattī bhavissati._  

</Paragraph>

</Division>
<Division number="403" book="ap1" bookNo="20" page="386">

<Paragraph number="4711" type="verses" book="ap1">
_Pañcapaññāsakappamhi,_  
_anomo nāma khattiyo;_  
_Cāturanto vijitāvī,_  
_jambumaṇḍassa issaro._  

</Paragraph>

</Division>
<Division number="404" book="ap1" bookNo="20" page="386">

<Paragraph number="4712" type="verses" book="ap1">
_Sattaratanasampanno,_  
_cakkavattī mahabbalo;_  
_Tāvatiṃsepi khobhetvā,_  
_issaraṃ kārayissati._  

</Paragraph>

</Division>
<Division number="405" book="ap1" bookNo="20" page="386">

<Paragraph number="4713" type="verses" book="ap1">
_Devabhūto manusso vā,_  
_appābādho bhavissati;_  
_Pariggahaṃ vivajjetvā,_  
_byādhiṃ loke tarissati._  

</Paragraph>

</Division>
<Division number="406" book="ap1" bookNo="20" page="386">

<Paragraph number="4714" type="verses" book="ap1">
_Apparimeyye ito kappe,_  
_okkākakulasambhavo;_  
_Gotamo nāma gottena,_  
_satthā loke bhavissati._  

</Paragraph>

</Division>
<Division number="407" book="ap1" bookNo="20" page="386">

<Paragraph number="4715" type="verses" book="ap1">
_Tassa dhammesu dāyādo,_  
_oraso dhammanimmito;_  
_Sabbāsave pariññāya,_  
_nibbāyissatināsavo._  

</Paragraph>

</Division>
<Division number="408" book="ap1" bookNo="20" page="386">

<Paragraph number="4716" type="verses" book="ap1">
_Kilese jhāpayitvāna,_  
_taṇhāsotaṃ tarissati;_  
_Bākulo nāma nāmena,_  
_hessati satthu sāvako._  

</Paragraph>

</Division>
<Division number="409" book="ap1" bookNo="20" page="386">

<Paragraph number="4717" type="verses" book="ap1">
_Idaṃ sabbaṃ abhiññāya,_  
_gotamo sakyapuṅgavo;_  
_Bhikkhusaṃghe nisīditvā,_  
_etadagge ṭhapessati’._  

</Paragraph>

</Division>
<Division number="410" book="ap1" bookNo="20" page="386">

<Paragraph number="4718" type="verses" book="ap1">
_Anomadassī bhagavā,_  
_sayambhū lokanāyako;_  
_Vivekānuvilokento,_  
_mamassamamupāgami._  

</Paragraph>

</Division>
<Division number="411" book="ap1" bookNo="20" page="387">

<Paragraph number="4719" type="verses" book="ap1">
_Upāgataṃ mahāvīraṃ,_  
_sabbaññuṃ lokanāyakaṃ;_  
_Sabbosadhena tappesiṃ,_  
_pasanno sehi pāṇibhi._  

</Paragraph>

</Division>
<Division number="412" book="ap1" bookNo="20" page="387">

<Paragraph number="4720" type="verses" book="ap1">
_Tassa me sukataṃ kammaṃ,_  
_sukhette bījasampadā;_  
_Khepetuṃ neva sakkomi,_  
_tadā hi sukataṃ mama._  

</Paragraph>

</Division>
<Division number="413" book="ap1" bookNo="20" page="387">

<Paragraph number="4721" type="verses" book="ap1">
_Lābhā mama suladdhaṃ me,_  
_yohaṃ addakkhi nāyakaṃ;_  
_Tena kammāvasesena,_  
_pattomhi acalaṃ padaṃ._  

</Paragraph>

</Division>
<Division number="414" book="ap1" bookNo="20" page="387">

<Paragraph number="4722" type="verses" book="ap1">
_Sabbametaṃ abhiññāya,_  
_gotamo sakyapuṅgavo;_  
_Bhikkhusaṃghe nisīditvā,_  
_etadagge ṭhapesi maṃ._  

</Paragraph>

</Division>
<Division number="415" book="ap1" bookNo="20" page="387">

<Paragraph number="4723" type="verses" book="ap1">
_Aparimeyye ito kappe,_  
_yaṃ kammamakariṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_bhesajjassa idaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="416" book="ap1" bookNo="20" page="387">

<Paragraph number="4724" type="verses" book="ap1">
_Kilesā jhāpitā mayhaṃ,_  
_bhavā sabbe samūhatā;_  
_Sabbāsavā parikkhīṇā,_  
_natthi dāni punabbhavo._  

</Paragraph>

</Division>
<Division number="417" book="ap1" bookNo="20" page="387">

<Paragraph number="4725" type="verses" book="ap1">
_Svāgataṃ vata me āsi,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ._  

</Paragraph>

</Division>
<Division number="418" book="ap1" bookNo="20" page="387">

<Paragraph number="4726" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="4727" book="ap1">
Itthaṃ sudaṃ āyasmā bākulo thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

Bākulattherassāpadānaṃ chaṭṭhaṃ.
