<Paragraph number="7369" type="verses" book="ap1">
_“Padumuttaro nāma jino,_  
_sabbadhammesu cakkhumā;_  
_Ito satasahassamhi,_  
_kappe uppajji nāyako._  

</Paragraph>

</Division>
<Division number="35" book="ap1" bookNo="20" page="432">

<Paragraph number="7370" type="verses" book="ap1">
_Sīhahanu brahmagiro,_  
_haṃsadundubhinissano;_  
_Nāgavikkantagamano,_  
_candasūrādikappabho._  

</Paragraph>

</Division>
<Division number="36" book="ap1" bookNo="20" page="432">

<Paragraph number="7371" type="verses" book="ap1">
_Mahāmatī mahāvīro,_  
_mahājhāyī mahābalo;_  
_Mahākāruṇiko nātho,_  
_mahātamapanūdano._  

</Paragraph>

</Division>
<Division number="37" book="ap1" bookNo="20" page="432">

<Paragraph number="7372" type="verses" book="ap1">
_Sa kadāci tilokaggo,_  
_veneyyaṃ vinayaṃ bahuṃ;_  
_Dhammaṃ desesi sambuddho,_  
_sattāsayavidū muni._  

</Paragraph>

</Division>
<Division number="38" book="ap1" bookNo="20" page="432">

<Paragraph number="7373" type="verses" book="ap1">
_Jhāyiṃ jhānarataṃ vīraṃ,_  
_upasantaṃ anāvilaṃ;_  
_Vaṇṇayanto parisatiṃ,_  
_tosesi janataṃ jino._  

</Paragraph>

</Division>
<Division number="39" book="ap1" bookNo="20" page="432">

<Paragraph number="7374" type="verses" book="ap1">
_Tadāhaṃ haṃsavatiyaṃ,_  
_brāhmaṇo vedapāragū;_  
_Dhammaṃ sutvāna mudito,_  
_taṃ ṭhānamabhipatthayiṃ._  

</Paragraph>

</Division>
<Division number="40" book="ap1" bookNo="20" page="432">

<Paragraph number="7375" type="verses" book="ap1">
_Tadā jino viyākāsi,_  
_saṃghamajjhe vināyako;_  
_‘Mudito hohi tvaṃ brahme,_  
_lacchase taṃ manorathaṃ._  

</Paragraph>

</Division>
<Division number="41" book="ap1" bookNo="20" page="432">

<Paragraph number="7376" type="verses" book="ap1">
_Satasahassito kappe,_  
_okkākakulasambhavo;_  
_Gotamo nāma gottena,_  
_satthā loke bhavissati._  

</Paragraph>

</Division>
<Division number="42" book="ap1" bookNo="20" page="432">

<Paragraph number="7377" type="verses" book="ap1">
_Tassa dhammesu dāyādo,_  
_oraso dhammanimmito;_  
_Revato nāma nāmena,_  
_hessati satthu sāvako’._  

</Paragraph>

</Division>
<Division number="43" book="ap1" bookNo="20" page="432">

<Paragraph number="7378" type="verses" book="ap1">
_Tena kammena sukatena,_  
_cetanāpaṇidhīhi ca;_  
_Jahitvā mānusaṃ dehaṃ,_  
_tāvatiṃsamagacchahaṃ._  

</Paragraph>

</Division>
<Division number="44" book="ap1" bookNo="20" page="432">

<Paragraph number="7379" type="verses" book="ap1">
_Pacchime ca bhave dāni,_  
_jātohaṃ koliye pure;_  
_Khattiye kulasampanne,_  
_iddhe phīte mahaddhane._  

</Paragraph>

</Division>
<Division number="45" book="ap1" bookNo="20" page="432">

<Paragraph number="7380" type="verses" book="ap1">
_Yadā kapilavatthusmiṃ,_  
_buddho dhammamadesayi;_  
_Tadā pasanno sugate,_  
_pabbajiṃ anagāriyaṃ._  

</Paragraph>

</Division>
<Division number="46" book="ap1" bookNo="20" page="433">

<Paragraph number="7381" type="verses" book="ap1">
_Kaṅkhā me bahulā āsi,_  
_kappākappe tahiṃ tahiṃ;_  
_Sabbaṃ taṃ vinayī buddho,_  
_desetvā dhammamuttamaṃ._  

</Paragraph>

</Division>
<Division number="47" book="ap1" bookNo="20" page="433">

<Paragraph number="7382" type="verses" book="ap1">
_Tatohaṃ tiṇṇasaṃsāro,_  
_sadā jhānasukhe rato;_  
_Viharāmi tadā buddho,_  
_maṃ disvā etadabravi._  

</Paragraph>

</Division>
<Division number="48" book="ap1" bookNo="20" page="433">

<Paragraph number="7383" type="verses" book="ap1">
_‘Yā kāci kaṅkhā idha vā huraṃ vā,_  
_Sakavediyā vā paravediyā vā;_  
_Ye jhāyino tā pajahanti sabbā,_  
_Ātāpino brahmacariyaṃ carantā’._  

</Paragraph>

</Division>
<Division number="49" book="ap1" bookNo="20" page="433">

<Paragraph number="7384" type="verses" book="ap1">
_Satasahasse kataṃ kammaṃ,_  
_phalaṃ dassesi me idha;_  
_Sumutto saravegova,_  
_kilese jhāpayiṃ mama._  

</Paragraph>

</Division>
<Division number="50" book="ap1" bookNo="20" page="433">

<Paragraph number="7385" type="verses" book="ap1">
_Tato jhānarataṃ disvā,_  
_buddho lokantagū muni;_  
_Jhāyīnaṃ bhikkhūnaṃ aggo,_  
_paññāpeti mahāmati._  

</Paragraph>

</Division>
<Division number="51" book="ap1" bookNo="20" page="433">

<Paragraph number="7386" type="verses" book="ap1">
_Kilesā jhāpitā mayhaṃ,_  
_ …pe… _  
_viharāmi anāsavo._  

</Paragraph>

</Division>
<Division number="52" book="ap1" bookNo="20" page="433">

<Paragraph number="7387" type="verses" book="ap1">
_Svāgataṃ vata me āsi,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ._  

</Paragraph>

</Division>
<Division number="53" book="ap1" bookNo="20" page="433">

<Paragraph number="7388" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="7389" book="ap1">
Itthaṃ sudaṃ āyasmā kaṅkhārevato thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="7390" type="center" book="ap1">
Kaṅkhārevatattherassāpadānaṃ dutiyaṃ.
</Paragraph>
