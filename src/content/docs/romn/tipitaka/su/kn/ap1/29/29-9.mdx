<Paragraph number="3340" type="verses" book="ap1">
_“Veṭambhinīti me nāmaṃ,_  
_pitusantaṃ mamaṃ tadā;_  
_Mama hatthaṃ gahetvāna,_  
_upānayi mahāmuniṃ._  

</Paragraph>

</Division>
<Division number="39" book="ap1" bookNo="20" page="274">

<Paragraph number="3341" type="verses" book="ap1">
_Imemaṃ uddisissanti,_  
_buddhā lokagganāyakā;_  
_Tehaṃ upaṭṭhiṃ sakkaccaṃ,_  
_pasanno sehi pāṇibhi._  

</Paragraph>

</Division>
<Division number="40" book="ap1" bookNo="20" page="274">

<Paragraph number="3342" type="verses" book="ap1">
_Ekattiṃse ito kappe,_  
_buddhe upaṭṭhahiṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_upaṭṭhānassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="41" book="ap1" bookNo="20" page="274">

<Paragraph number="3343" type="verses" book="ap1">
_Tevīsamhi ito kappe,_  
_caturo āsu khattiyā;_  
_Samaṇupaṭṭhākā nāma,_  
_cakkavattī mahabbalā._  

</Paragraph>

</Division>
<Division number="42" book="ap1" bookNo="20" page="274">

<Paragraph number="3344" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="3345" book="ap1">
Itthaṃ sudaṃ āyasmā buddhupaṭṭhāyiko thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="3346" type="center" book="ap1">
Buddhupaṭṭhāyikattherassāpadānaṃ navamaṃ.
</Paragraph>
