<Division number="1" book="ap1">

<Paragraph number="5664" type="verses" book="ap1">
_“Imamhi bhaddake kappe,_  
_brahmabandhu mahāyaso;_  
_Kassapo nāma gottena,_  
_uppajji vadataṃ varo._  

</Paragraph>

</Division>
<Division number="2" book="ap1">

<Paragraph number="5665" type="verses" book="ap1">
_Nippapañco nirālambo,_  
_ākāsasamamānaso;_  
_Suññatābahulo tādī,_  
_animittarato vasī._  

</Paragraph>

</Division>
<Division number="3" book="ap1">

<Paragraph number="5666" type="verses" book="ap1">
_Asaṅgacitto nikleso,_  
_asaṃsaṭṭho kule gaṇe;_  
_Mahākāruṇiko vīro,_  
_vinayopāyakovido._  

</Paragraph>

</Division>
<Division number="4" book="ap1">

<Paragraph number="5667" type="verses" book="ap1">
_Uyyutto parakiccesu,_  
_vinayanto sadevake;_  
_Nibbānagamanaṃ maggaṃ,_  
_gatiṃ paṅkavisosanaṃ._  

</Paragraph>

</Division>
<Division number="5" book="ap1">

<Paragraph number="5668" type="verses" book="ap1">
_Amataṃ paramassādaṃ,_  
_jarāmaccunivāraṇaṃ;_  
_Mahāparisamajjhe so,_  
_nisinno lokatārako._  

</Paragraph>

</Division>
<Division number="6" book="ap1">

<Paragraph number="5669" type="verses" book="ap1">
_Karavīkaruto nātho,_  
_brahmaghoso tathāgato;_  
_Uddharanto mahāduggā,_  
_vippanaṭṭhe anāyake._  

</Paragraph>

</Division>
<Division number="7" book="ap1">

<Paragraph number="5670" type="verses" book="ap1">
_Desento virajaṃ dhammaṃ,_  
_diṭṭho me lokanāyako;_  
_Tassa dhammaṃ suṇitvāna,_  
_pabbajiṃ anagāriyaṃ._  

</Paragraph>

</Division>
<Division number="8" book="ap1">

<Paragraph number="5671" type="verses" book="ap1">
_Pabbajitvā tadāpāhaṃ,_  
_cintento jinasāsanaṃ;_  
_Ekakova vane ramme,_  
_vasiṃ saṃsaggapīḷito._  

</Paragraph>

</Division>
<Division number="9" book="ap1">

<Paragraph number="5672" type="verses" book="ap1">
_Sakkāyavūpakāso me,_  
_hetubhūto mamābhavī;_  
_Manaso vūpakāsassa,_  
_saṃsaggabhayadassino._  

</Paragraph>

</Division>
<Division number="10" book="ap1">

<Paragraph number="5673" type="verses" book="ap1">
_Kilesā jhāpitā mayhaṃ,_  
_bhavā sabbe samūhatā;_  
_Nāgova bandhanaṃ chetvā,_  
_viharāmi anāsavo._  

</Paragraph>

</Division>
<Division number="11" book="ap1">

<Paragraph number="5674" type="verses" book="ap1">
_Svāgataṃ vata me āsi,_  
_mama buddhassa santike;_  
_Tisso vijjā anuppattā,_  
_kataṃ buddhassa sāsanaṃ._  

</Paragraph>

</Division>
<Division number="12" book="ap1">

<Paragraph number="5675" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_vimokkhāpi ca aṭṭhime;_  
_Chaḷabhiññā sacchikatā,_  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="5676" book="ap1">
Itthaṃ sudaṃ āyasmā ekavihāriko thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="5677" type="center" book="ap1">
Ekavihārikattherassāpadānaṃ paṭhamaṃ.
</Paragraph>
