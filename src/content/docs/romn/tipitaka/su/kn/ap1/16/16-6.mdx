<Paragraph number="2392" type="verses" book="ap1">
_“Himavantassāvidūre,_  
_kukkuṭo nāma pabbato;_  
_Tamhi pabbatapādamhi,_  
_satta buddhā vasanti te._  

</Paragraph>

</Division>
<Division number="31" book="ap1" bookNo="20" page="197">

<Paragraph number="2393" type="verses" book="ap1">
_Kadambaṃ pupphitaṃ disvā,_  
_dīparājaṃva uggataṃ;_  
_Ubho hatthehi paggayha,_  
_satta buddhe samokiriṃ._  

</Paragraph>

</Division>
<Division number="32" book="ap1" bookNo="20" page="197">

<Paragraph number="2394" type="verses" book="ap1">
_Catunnavutito kappe,_  
_yaṃ pupphamabhiropayiṃ;_  
_Duggatiṃ nābhijānāmi,_  
_buddhapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="33" book="ap1" bookNo="20" page="197">

<Paragraph number="2395" type="verses" book="ap1">
_Dvenavute ito kappe,_  
_sattāsuṃ pupphanāmakā;_  
_Sattaratanasampannā,_  
_cakkavattī mahabbalā._  

</Paragraph>

</Division>
<Division number="34" book="ap1" bookNo="20" page="197">

<Paragraph number="2396" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="2397" book="ap1">
Itthaṃ sudaṃ āyasmā kadambapupphiyo thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="2398" type="center" book="ap1">

</Paragraph>
