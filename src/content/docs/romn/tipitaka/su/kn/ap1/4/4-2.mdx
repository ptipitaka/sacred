<Division number="17" book="ap1">

<Paragraph number="1091" type="verses" book="ap1">
_“Sobhito nāma nāmena,_  
_ahosiṃ brāhmaṇo tadā;_  
_Purakkhato sasissehi,_  
_ārāmaṃ agamāsahaṃ._  

</Paragraph>

</Division>
<Division number="18" book="ap1">

<Paragraph number="1092" type="verses" book="ap1">
_Bhagavā tamhi samaye,_  
_bhikkhusaṃghapurakkhato;_  
_Ārāmadvārā nikkhamma,_  
_aṭṭhāsi purisuttamo._  

</Paragraph>

</Division>
<Division number="19" book="ap1">

<Paragraph number="1093" type="verses" book="ap1">
_Tamaddasāsiṃ sambuddhaṃ,_  
_dantaṃ dantapurakkhataṃ;_  
_Sakaṃ cittaṃ pasādetvā,_  
_santhaviṃ lokanāyakaṃ._  

</Paragraph>

</Division>
<Division number="20" book="ap1">

<Paragraph number="1094" type="verses" book="ap1">
_Ye keci pādapā sabbe,_  
_mahiyā te virūhare;_  
_Buddhimanto tathā sattā,_  
_ruhanti jinasāsane._  

</Paragraph>

</Division>
<Division number="21" book="ap1">

<Paragraph number="1095" type="verses" book="ap1">
_Satthavāhosi sappañño,_  
_mahesi bahuke jane;_  
_Vipathā uddharitvāna,_  
_pathaṃ ācikkhase tuvaṃ._  

</Paragraph>

</Division>
<Division number="22" book="ap1">

<Paragraph number="1096" type="verses" book="ap1">
_Danto dantaparikiṇṇo,_  
_jhāyī jhānaratehi ca;_  
_Ātāpī pahitattehi,_  
_upasantehi tādibhi._  

</Paragraph>

</Division>
<Division number="23" book="ap1">

<Paragraph number="1097" type="verses" book="ap1">
_Alaṅkato parisāhi,_  
_puññañāṇehi sobhati;_  
_Pabhā niddhāvate tuyhaṃ,_  
_sūriyodayane yathā._  

</Paragraph>

</Division>
<Division number="24" book="ap1">

<Paragraph number="1098" type="verses" book="ap1">
_Pasannacittaṃ disvāna,_  
_mahesī padumuttaro;_  
_Bhikkhusaṃghe ṭhito satthā,_  
_imā gāthā abhāsatha._  

</Paragraph>

</Division>
<Division number="25" book="ap1">

<Paragraph number="1099" type="verses" book="ap1">
_‘Yo so hāsaṃ janetvāna,_  
_mamaṃ kittesi brāhmaṇo;_  
_Kappānaṃ satasahassaṃ,_  
_devaloke ramissati._  

</Paragraph>

</Division>
<Division number="26" book="ap1">

<Paragraph number="1100" type="verses" book="ap1">
_Tusitā hi cavitvāna,_  
_sukkamūlena codito;_  
_Gotamassa bhagavato,_  
_sāsane pabbajissati._  

</Paragraph>

</Division>
<Division number="27" book="ap1">

<Paragraph number="1101" type="verses" book="ap1">
_Tena kammena sukatena,_  
_arahattaṃ labhissati;_  
_Sāgato nāma nāmena,_  
_hessati satthu sāvako’._  

</Paragraph>

</Division>
<Division number="28" book="ap1">

<Paragraph number="1102" type="verses" book="ap1">
_Pabbajitvāna kāyena,_  
_pāpakammaṃ vivajjayiṃ;_  
_Vacīduccaritaṃ hitvā,_  
_ājīvaṃ parisodhayiṃ._  

</Paragraph>

</Division>
<Division number="29" book="ap1">

<Paragraph number="1103" type="verses" book="ap1">
_Evaṃ viharamānohaṃ,_  
_tejodhātūsu kovido;_  
_Sabbāsave pariññāya,_  
_viharāmi anāsavo._  

</Paragraph>

</Division>
<Division number="30" book="ap1">

<Paragraph number="1104" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="1105" book="ap1">
Itthaṃ sudaṃ āyasmā sāgato thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="1106" type="center" book="ap1">
Sāgatattherassāpadānaṃ dutiyaṃ.
</Paragraph>
