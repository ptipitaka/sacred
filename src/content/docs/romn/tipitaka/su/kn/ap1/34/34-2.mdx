<Division number="7" book="ap1">

<Paragraph number="3723" type="verses" book="ap1">
_“Suvaṇṇavaṇṇaṃ sambuddhaṃ,_  
_Gacchantaṃ anilañjase;_  
_Ghatāsanaṃva jalitaṃ,_  
_Ādittaṃva hutāsanaṃ._  

</Paragraph>

</Division>
<Division number="8" book="ap1">

<Paragraph number="3724" type="verses" book="ap1">
_Pāṇinā udakaṃ gayha,_  
_ākāse ukkhipiṃ ahaṃ;_  
_Sampaṭicchi mahāvīro,_  
_buddho kāruṇiko isi._  

</Paragraph>

</Division>
<Division number="9" book="ap1">

<Paragraph number="3725" type="verses" book="ap1">
_Antalikkhe ṭhito satthā,_  
_padumuttaranāmako;_  
_Mama saṅkappamaññāya,_  
_imaṃ gāthamabhāsatha._  

</Paragraph>

</Division>
<Division number="10" book="ap1">

<Paragraph number="3726" type="verses" book="ap1">
_‘Iminā dakadānena,_  
_pītiuppādanena ca;_  
_Kappasatasahassampi,_  
_duggatiṃ nupapajjati’._  

</Paragraph>

</Division>
<Division number="11" book="ap1">

<Paragraph number="3727" type="verses" book="ap1">
_Tena kammena dvipadinda,_  
_lokajeṭṭha narāsabha;_  
_Pattomhi acalaṃ ṭhānaṃ,_  
_hitvā jayaparājayaṃ._  

</Paragraph>

</Division>
<Division number="12" book="ap1">

<Paragraph number="3728" type="verses" book="ap1">
_Sahassarājanāmena,_  
_tayo ca cakkavattino;_  
_Pañcasaṭṭhikappasate,_  
_cāturantā janādhipā._  

</Paragraph>

</Division>
<Division number="13" book="ap1">

<Paragraph number="3729" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="3730" book="ap1">
Itthaṃ sudaṃ āyasmā udakapūjako thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="3731" type="center" book="ap1">
Udakapūjakattherassāpadānaṃ dutiyaṃ.
</Paragraph>
