<Division number="68" book="ap1">

<Paragraph number="773" type="verses" book="ap1">
_“Padumuttarassa bhagavato,_  
_Lokajeṭṭhassa tādino;_  
_Sattabhūmamhi pāsāde,_  
_Ādāsaṃ santhariṃ ahaṃ._  

</Paragraph>

</Division>
<Division number="69" book="ap1">

<Paragraph number="774" type="verses" book="ap1">
_Khīṇāsavasahassehi,_  
_parikiṇṇo mahāmuni;_  
_Upāgami gandhakuṭiṃ,_  
_dvipadindo narāsabho._  

</Paragraph>

</Division>
<Division number="70" book="ap1">

<Paragraph number="775" type="verses" book="ap1">
_Virocento gandhakuṭiṃ,_  
_devadevo narāsabho;_  
_Bhikkhusaṃghe ṭhito satthā,_  
_imā gāthā abhāsatha._  

</Paragraph>

</Division>
<Division number="71" book="ap1">

<Paragraph number="776" type="verses" book="ap1">
_‘Yenāyaṃ jotitā seyyā,_  
_ādāsova susanthato;_  
_Tamahaṃ kittayissāmi,_  
_suṇātha mama bhāsato._  

</Paragraph>

</Division>
<Division number="72" book="ap1">

<Paragraph number="777" type="verses" book="ap1">
_Soṇṇamayā rūpimayā,_  
_atho veḷuriyāmayā;_  
_Nibbattissanti pāsādā,_  
_ye keci manaso piyā._  

</Paragraph>

</Division>
<Division number="73" book="ap1">

<Paragraph number="778" type="verses" book="ap1">
_Catusaṭṭhikkhattuṃ devindo,_  
_devarajjaṃ karissati;_  
_Sahassakkhattuṃ cakkavattī,_  
_bhavissati anantarā._  

</Paragraph>

</Division>
<Division number="74" book="ap1">

<Paragraph number="779" type="verses" book="ap1">
_Ekavīsatikappamhi,_  
_vimalo nāma khattiyo;_  
_Cāturanto vijitāvī,_  
_cakkavattī bhavissati._  

</Paragraph>

</Division>
<Division number="75" book="ap1">

<Paragraph number="780" type="verses" book="ap1">
_Nagaraṃ reṇuvatī nāma,_  
_iṭṭhakāhi sumāpitaṃ;_  
_Āyāmato tīṇi sataṃ,_  
_caturassasamāyutaṃ._  

</Paragraph>

</Division>
<Division number="76" book="ap1">

<Paragraph number="781" type="verses" book="ap1">
_Sudassano nāma pāsādo,_  
_vissakammena māpito;_  
_Kūṭāgāravarūpeto,_  
_sattaratanabhūsito._  

</Paragraph>

</Division>
<Division number="77" book="ap1">

<Paragraph number="782" type="verses" book="ap1">
_Dasasaddāvivittaṃ taṃ,_  
_vijjādharasamākulaṃ;_  
_Sudassanaṃva nagaraṃ,_  
_devatānaṃ bhavissati._  

</Paragraph>

</Division>
<Division number="78" book="ap1">

<Paragraph number="783" type="verses" book="ap1">
_Pabhā niggacchate tassa,_  
_uggacchanteva sūriye;_  
_Virocessati taṃ niccaṃ,_  
_samantā aṭṭhayojanaṃ._  

</Paragraph>

</Division>
<Division number="79" book="ap1">

<Paragraph number="784" type="verses" book="ap1">
_Kappasatasahassamhi,_  
_okkākakulasambhavo;_  
_Gotamo nāma gottena,_  
_satthā loke bhavissati._  

</Paragraph>

</Division>
<Division number="80" book="ap1">

<Paragraph number="785" type="verses" book="ap1">
_Tusitā so cavitvāna,_  
_sukkamūlena codito;_  
_Gotamassa bhagavato,_  
_atrajo so bhavissati._  

</Paragraph>

</Division>
<Division number="81" book="ap1">

<Paragraph number="786" type="verses" book="ap1">
_Sacevaseyya agāraṃ,_  
_cakkavattī bhaveyya so;_  
_Aṭṭhānametaṃ yaṃ tādī,_  
_agāre ratimajjhagā._  

</Paragraph>

</Division>
<Division number="82" book="ap1">

<Paragraph number="787" type="verses" book="ap1">
_Nikkhamitvā agāramhā,_  
_pabbajissati subbato;_  
_Rāhulo nāma nāmena,_  
_arahā so bhavissati’._  

</Paragraph>

</Division>
<Division number="83" book="ap1">

<Paragraph number="788" type="verses" book="ap1">
_Kikīva aṇḍaṃ rakkheyya,_  
_cāmarī viya vāladhiṃ;_  
_Nipako sīlasampanno,_  
_evaṃ rakkhiṃ mahāmuni._  

</Paragraph>

</Division>
<Division number="84" book="ap1">

<Paragraph number="789" type="verses" book="ap1">
_Tassāhaṃ dhammamaññāya,_  
_vihāsiṃ sāsane rato;_  
_Sabbāsave pariññāya,_  
_viharāmi anāsavo._  

</Paragraph>

</Division>
<Division number="85" book="ap1">

<Paragraph number="790" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_vimokkhāpi ca aṭṭhime;_  
_Chaḷabhiññā sacchikatā,_  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="791" book="ap1">
Itthaṃ sudaṃ āyasmā rāhulo thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="792" type="center" book="ap1">
Rāhulattherassāpadānaṃ chaṭṭhaṃ.
</Paragraph>
