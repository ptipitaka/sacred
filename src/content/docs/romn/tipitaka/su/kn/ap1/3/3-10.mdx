_ahosiṃ vāṇijo tadā;_

<Division number="185" book="ap1">

<Paragraph number="1062" type="verses" book="ap1">

_vipassissa mahesino;_  
_Ekaṃ khomaṃ mayā dinnaṃ,_  
_kusalatthāya satthuno._  

</Paragraph>

</Division>
<Division number="186" book="ap1">

<Paragraph number="1063" type="verses" book="ap1">
_Ekanavutito kappe,_  
_yaṃ khomamadadiṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_khomadānassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="187" book="ap1">

<Paragraph number="1064" type="verses" book="ap1">
_Sattarase ito kappe,_  
_eko sindhavasandhano;_  
_Sattaratanasampanno,_  
_catudīpamhi issaro._  

</Paragraph>

</Division>
<Division number="188" book="ap1">

<Paragraph number="1065" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_Vimokkhāpi ca aṭṭhime;_  
_Chaḷabhiññā sacchikatā,_  
_Kataṃ buddhassa sāsanaṃ”. (991)_  

</Paragraph>

<Paragraph number="1066" book="ap1">

</Paragraph>

_saraṇo sīlagāhako;_

_pulino uttiyena ca._

<Paragraph number="1070" type="verses" book="ap1">

</Paragraph>
