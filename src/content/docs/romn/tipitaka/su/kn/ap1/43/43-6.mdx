<Paragraph number="5557" type="verses" book="ap1">
_“Sucintitaṃ bhagavantaṃ,_  
_lokajeṭṭhaṃ narāsabhaṃ;_  
_Upaviṭṭhaṃ mahāraññaṃ,_  
_vātābādhena pīḷitaṃ._  

</Paragraph>

</Division>
<Division number="79" book="ap1" bookNo="20" page="435">

<Paragraph number="5558" type="verses" book="ap1">
_Disvā cittaṃ pasādetvā,_  
_ghatamaṇḍamupānayiṃ;_  
_Katattā ācitattā ca,_  
_gaṅgā bhāgīrathī ayaṃ._  

</Paragraph>

</Division>
<Division number="80" book="ap1" bookNo="20" page="435">

<Paragraph number="5559" type="verses" book="ap1">
_Mahāsamuddā cattāro,_  
_ghataṃ sampajjare mama;_  
_Ayañca pathavī ghorā,_  
_appamāṇā asaṅkhiyā._  

</Paragraph>

</Division>
<Division number="81" book="ap1" bookNo="20" page="435">

<Paragraph number="5560" type="verses" book="ap1">
_Mama saṅkappamaññāya,_  
_bhavate madhusakkarā;_  
_Cātuddīpā ime rukkhā,_  
_pādapā dharaṇīruhā._  

</Paragraph>

</Division>
<Division number="82" book="ap1" bookNo="20" page="435">

<Paragraph number="5561" type="verses" book="ap1">
_Mama saṅkappamaññāya,_  
_kapparukkhā bhavanti te;_  
_Paññāsakkhattuṃ devindo,_  
_devarajjamakārayiṃ._  

</Paragraph>

</Division>
<Division number="83" book="ap1" bookNo="20" page="435">

<Paragraph number="5562" type="verses" book="ap1">
_Ekapaññāsakkhattuñca,_  
_cakkavattī ahosahaṃ;_  
_Padesarajjaṃ vipulaṃ,_  
_gaṇanāto asaṅkhiyaṃ._  

</Paragraph>

</Division>
<Division number="84" book="ap1" bookNo="20" page="436">

<Paragraph number="5563" type="verses" book="ap1">
_Catunnavutito kappe,_  
_yaṃ dānamadadiṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_ghatamaṇḍassidaṃ phalaṃ._  

</Paragraph>

</Division>

<Paragraph number="5564" type="verses" book="ap1">

_ …pe… _  
_viharāmi anāsavo._  

</Paragraph>

</Division>

<Paragraph number="5565" type="verses" book="ap1">

_ …pe… _  
_kataṃ buddhassa sāsanaṃ._  

</Paragraph>

</Division>

_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="5567" book="ap1">

</Paragraph>
