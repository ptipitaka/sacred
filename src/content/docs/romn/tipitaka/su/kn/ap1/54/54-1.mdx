<Paragraph number="6948" type="verses" book="ap1">
_“Padumuttaro nāma jino,_  
_anejo ajitaṃ jayo;_  
_Satasahasse kappānaṃ,_  
_ito uppajji nāyako._  

</Paragraph>

</Division>
<Division number="2" book="ap1" bookNo="20" page="429">

<Paragraph number="6949" type="verses" book="ap1">
_Vīro kamalapattakkho,_  
_sasaṅkavimalānano;_  
_Kanakācalasaṅkāso,_  
_ravidittisamappabho._  

</Paragraph>

</Division>
<Division number="3" book="ap1" bookNo="20" page="429">

<Paragraph number="6950" type="verses" book="ap1">
_Sattanettamanohārī,_  
_varalakkhaṇabhūsito;_  
_Sabbavākyapathātīto,_  
_manujāmarasakkato._  

</Paragraph>

</Division>
<Division number="4" book="ap1" bookNo="20" page="429">

<Paragraph number="6951" type="verses" book="ap1">
_Sambuddho bodhayaṃ satte,_  
_vāgīso madhurassaro;_  
_Karuṇānibandhasantāno,_  
_parisāsu visārado._  

</Paragraph>

</Division>
<Division number="5" book="ap1" bookNo="20" page="429">

<Paragraph number="6952" type="verses" book="ap1">
_Deseti madhuraṃ dhammaṃ,_  
_catusaccūpasaṃhitaṃ;_  
_Nimugge mohapaṅkamhi,_  
_samuddharati pāṇine._  

</Paragraph>

</Division>
<Division number="6" book="ap1" bookNo="20" page="429">

<Paragraph number="6953" type="verses" book="ap1">
_Tadā ekacaro hutvā,_  
_tāpaso himavālayo;_  
_Nabhasā mānusaṃ lokaṃ,_  
_gacchanto jinamaddasaṃ._  

</Paragraph>

</Division>
<Division number="7" book="ap1" bookNo="20" page="429">

<Paragraph number="6954" type="verses" book="ap1">
_Upecca santikaṃ tassa,_  
_assosiṃ dhammadesanaṃ;_  
_Vaṇṇayantassa vīrassa,_  
_sāvakassa mahāguṇaṃ._  

</Paragraph>

</Division>
<Division number="8" book="ap1" bookNo="20" page="429">

<Paragraph number="6955" type="verses" book="ap1">
_‘Saṃkhittena mayā vuttaṃ,_  
_vitthārena pakāsayaṃ;_  
_Parisaṃ mañca toseti,_  
_yathā kaccāyano ayaṃ._  

</Paragraph>

</Division>
<Division number="9" book="ap1" bookNo="20" page="429">

<Paragraph number="6956" type="verses" book="ap1">
_Nāhaṃ evamidhekaccaṃ,_  
_aññaṃ passāmi sāvakaṃ;_  
_Tasmātadagge esaggo,_  
_evaṃ dhāretha bhikkhavo’._  

</Paragraph>

</Division>
<Division number="10" book="ap1" bookNo="20" page="429">

<Paragraph number="6957" type="verses" book="ap1">
_Tadāhaṃ vimhito hutvā,_  
_sutvā vākyaṃ manoramaṃ;_  
_Himavantaṃ gamitvāna,_  
_āhitvā pupphasañcayaṃ._  

</Paragraph>

</Division>
<Division number="11" book="ap1" bookNo="20" page="429">

<Paragraph number="6958" type="verses" book="ap1">
_Pūjetvā lokasaraṇaṃ,_  
_taṃ ṭhānamabhipatthayiṃ;_  
_Tadā mamāsayaṃ ñatvā,_  
_byākāsi sa raṇañjaho._  

</Paragraph>

</Division>
<Division number="12" book="ap1" bookNo="20" page="430">

<Paragraph number="6959" type="verses" book="ap1">
_‘Passathetaṃ isivaraṃ,_  
_niddhantakanakattacaṃ;_  
_Uddhaggalomaṃ pīṇaṃsaṃ,_  
_acalaṃ pañjaliṃ ṭhitaṃ._  

</Paragraph>

</Division>
<Division number="13" book="ap1" bookNo="20" page="430">

<Paragraph number="6960" type="verses" book="ap1">
_Hāsaṃ supuṇṇanayanaṃ,_  
_buddhavaṇṇagatāsayaṃ;_  
_Dhammajaṃ uggahadayaṃ,_  
_amatāsittasannibhaṃ’._  

</Paragraph>

</Division>
<Division number="14" book="ap1" bookNo="20" page="430">

<Paragraph number="6961" type="verses" book="ap1">
_Kaccānassa guṇaṃ sutvā,_  
_taṃ ṭhānaṃ patthayaṃ ṭhito;_  
_Anāgatamhi addhāne,_  
_gotamassa mahāmune._  

</Paragraph>

</Division>
<Division number="15" book="ap1" bookNo="20" page="430">

<Paragraph number="6962" type="verses" book="ap1">
_Tassa dhammesu dāyādo,_  
_oraso dhammanimmito;_  
_Kaccāno nāma nāmena,_  
_hessati satthu sāvako._  

</Paragraph>

</Division>
<Division number="16" book="ap1" bookNo="20" page="430">

<Paragraph number="6963" type="verses" book="ap1">
_Bahussuto mahāñāṇī,_  
_adhippāyavidū mune;_  
_Pāpuṇissati taṃ ṭhānaṃ,_  
_yathāyaṃ byākato mayā._  

</Paragraph>

</Division>
<Division number="17" book="ap1" bookNo="20" page="430">

<Paragraph number="6964" type="verses" book="ap1">
_Satasahassito kappe,_  
_yaṃ kammamakariṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_buddhapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="18" book="ap1" bookNo="20" page="430">

<Paragraph number="6965" type="verses" book="ap1">
_Duve bhave saṃsarāmi,_  
_devatte atha mānuse;_  
_Aññaṃ gatiṃ na gacchāmi,_  
_buddhapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="19" book="ap1" bookNo="20" page="430">

<Paragraph number="6966" type="verses" book="ap1">
_Duve kule pajāyāmi,_  
_khattiye atha brāhmaṇe;_  
_Nīce kule na jāyāmi,_  
_buddhapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="20" book="ap1" bookNo="20" page="430">

<Paragraph number="6967" type="verses" book="ap1">
_Pacchime ca bhave dāni,_  
_jāto ujjeniyaṃ pure;_  
_Pajjotassa ca caṇḍassa,_  
_purohitadijādhino._  

</Paragraph>

</Division>
<Division number="21" book="ap1" bookNo="20" page="430">

<Paragraph number="6968" type="verses" book="ap1">
_Putto tiriṭivacchassa,_  
_nipuṇo vedapāragū;_  
_Mātā ca candimā nāma,_  
_kaccānohaṃ varattaco._  

</Paragraph>

</Division>
<Division number="22" book="ap1" bookNo="20" page="430">

<Paragraph number="6969" type="verses" book="ap1">
_Vīmaṃsanatthaṃ buddhassa,_  
_bhūmipālena pesito;_  
_Disvā mokkhapuradvāraṃ,_  
_nāyakaṃ guṇasañcayaṃ._  

</Paragraph>

</Division>
<Division number="23" book="ap1" bookNo="20" page="430">

<Paragraph number="6970" type="verses" book="ap1">
_Sutvā ca vimalaṃ vākyaṃ,_  
_gatipaṅkavisosanaṃ;_  
_Pāpuṇiṃ amataṃ santaṃ,_  
_sesehi saha sattahi._  

</Paragraph>

</Division>
<Division number="24" book="ap1" bookNo="20" page="431">

<Paragraph number="6971" type="verses" book="ap1">
_Adhippāyavidū jāto,_  
_sugatassa mahāmate;_  
_Ṭhapito etadagge ca,_  
_susamiddhamanoratho._  

</Paragraph>

</Division>
<Division number="25" book="ap1" bookNo="20" page="431">

<Paragraph number="6972" type="verses" book="ap1">
_Kilesā jhāpitā mayhaṃ,_  
_bhavā sabbe samūhatā;_  
_Nāgova bandhanaṃ chetvā,_  
_viharāmi anāsavo._  

</Paragraph>

</Division>
<Division number="26" book="ap1" bookNo="20" page="431">

<Paragraph number="6973" type="verses" book="ap1">
_Svāgataṃ vata me āsi,_  
_mama buddhassa santike;_  
_Tisso vijjā anuppattā,_  
_kataṃ buddhassa sāsanaṃ._  

</Paragraph>

</Division>
<Division number="27" book="ap1" bookNo="20" page="431">

<Paragraph number="6974" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_vimokkhāpi ca aṭṭhime;_  
_Chaḷabhiññā sacchikatā,_  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="6975" book="ap1">
Itthaṃ sudaṃ āyasmā mahākaccāyano thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="6976" type="center" book="ap1">
Mahākaccāyanattherassāpadānaṃ paṭhamaṃ.
