<Paragraph number="7391" type="verses" book="ap1">
_“Padumuttaro nāma jino,_  
_sabbadhammesu cakkhumā;_  
_Ito satasahassamhi,_  
_kappe uppajji nāyako._  

</Paragraph>

</Division>
<Division number="55" book="ap1" bookNo="20" page="433">

<Paragraph number="7392" type="verses" book="ap1">
_Sīlaṃ tassa asaṅkheyyaṃ,_  
_samādhi vajirūpamo;_  
_Asaṅkheyyaṃ ñāṇavaraṃ,_  
_vimutti ca anopamā._  

</Paragraph>

</Division>
<Division number="56" book="ap1" bookNo="20" page="433">

<Paragraph number="7393" type="verses" book="ap1">
_Manujāmaranāgānaṃ,_  
_brahmānañca samāgame;_  
_Samaṇabrāhmaṇākiṇṇe,_  
_dhammaṃ desesi nāyako._  

</Paragraph>

</Division>
<Division number="57" book="ap1" bookNo="20" page="433">

<Paragraph number="7394" type="verses" book="ap1">
_Sasāvakaṃ mahālābhiṃ,_  
_puññavantaṃ jutindharaṃ;_  
_Ṭhapesi etadaggamhi,_  
_parisāsu visārado._  

</Paragraph>

</Division>
<Division number="58" book="ap1" bookNo="20" page="433">

<Paragraph number="7395" type="verses" book="ap1">
_Tadāhaṃ khattiyo āsiṃ,_  
_nagare haṃsasavhaye;_  
_Sutvā jinassa taṃ vākyaṃ,_  
_sāvakassa guṇaṃ bahuṃ._  

</Paragraph>

</Division>
<Division number="59" book="ap1" bookNo="20" page="434">

<Paragraph number="7396" type="verses" book="ap1">
_Nimantayitvā sattāhaṃ,_  
_bhojayitvā sasāvakaṃ;_  
_Mahādānaṃ daditvāna,_  
_taṃ ṭhānamabhipatthayiṃ._  

</Paragraph>

</Division>
<Division number="60" book="ap1" bookNo="20" page="434">

<Paragraph number="7397" type="verses" book="ap1">
_Tadā maṃ vinataṃ pāde,_  
_disvāna purisāsabho;_  
_Sarena mahatā vīro,_  
_idaṃ vacanamabravi._  

</Paragraph>

</Division>
<Division number="61" book="ap1" bookNo="20" page="434">

<Paragraph number="7398" type="verses" book="ap1">
_‘Tato jinassa vacanaṃ,_  
_sotukāmā mahājanā;_  
_Devadānavagandhabbā,_  
_brahmāno ca mahiddhikā’._  

</Paragraph>

</Division>
<Division number="62" book="ap1" bookNo="20" page="434">

<Paragraph number="7399" type="verses" book="ap1">
_Samaṇabrāhmaṇā ceva,_  
_namassiṃsu katañjalī;_  
_‘Namo te purisājañña,_  
_namo te purisuttama._  

</Paragraph>

</Division>
<Division number="63" book="ap1" bookNo="20" page="434">

<Paragraph number="7400" type="verses" book="ap1">
_Khattiyena mahādānaṃ,_  
_dinnaṃ sattāhikampi vo;_  
_Sotukāmā phalaṃ tassa,_  
_byākarohi mahāmune’._  

</Paragraph>

</Division>
<Division number="64" book="ap1" bookNo="20" page="434">

<Paragraph number="7401" type="verses" book="ap1">
_Tato avoca bhagavā,_  
_‘suṇātha mama bhāsitaṃ;_  
_Appameyyamhi buddhamhi,_  
_sasaṃghamhi patiṭṭhitā._  

</Paragraph>

</Division>
<Division number="65" book="ap1" bookNo="20" page="434">

<Paragraph number="7402" type="verses" book="ap1">
_Dakkhiṇā tāya ko vattā,_  
_appameyyaphalā hi sā;_  
_Api ce sa mahābhogo,_  
_ṭhānaṃ pattheti uttamaṃ._  

</Paragraph>

</Division>
<Division number="66" book="ap1" bookNo="20" page="434">

<Paragraph number="7403" type="verses" book="ap1">
_Lābhī vipulalābhānaṃ,_  
_yathā bhikkhu sudassano;_  
_Tathāhampi bhaveyyanti,_  
_lacchase taṃ anāgate._  

</Paragraph>

</Division>
<Division number="67" book="ap1" bookNo="20" page="434">

<Paragraph number="7404" type="verses" book="ap1">
_Satasahassito kappe,_  
_okkākakulasambhavo;_  
_Gotamo nāma gottena,_  
_satthā loke bhavissati._  

</Paragraph>

</Division>
<Division number="68" book="ap1" bookNo="20" page="434">

<Paragraph number="7405" type="verses" book="ap1">
_Tassa dhammesu dāyādo,_  
_oraso dhammanimmito;_  
_Sīvali nāma nāmena,_  
_hessati satthu sāvako’._  

</Paragraph>

</Division>
<Division number="69" book="ap1" bookNo="20" page="434">

<Paragraph number="7406" type="verses" book="ap1">
_Tena kammena sukatena,_  
_cetanāpaṇidhīhi ca;_  
_Jahitvā mānusaṃ dehaṃ,_  
_tāvatiṃsūpago ahaṃ._  

</Paragraph>

</Division>
<Division number="70" book="ap1" bookNo="20" page="434">

<Paragraph number="7407" type="verses" book="ap1">
_Ekanavutito kappe,_  
_vipassī lokanāyako;_  
_Uppajji cārudassano,_  
_sabbadhammavipassako._  

</Paragraph>

</Division>
<Division number="71" book="ap1" bookNo="20" page="434">

<Paragraph number="7408" type="verses" book="ap1">
_Tadāhaṃ bandhumatiyaṃ,_  
_kulassaññatarassa ca;_  
_Dayito passito ceva,_  
_āsiṃ kammantavāvaṭo._  

</Paragraph>

</Division>
<Division number="72" book="ap1" bookNo="20" page="435">

<Paragraph number="7409" type="verses" book="ap1">
_Tadā aññataro pūgo,_  
_vipassissa mahesino;_  
_Parivesaṃ akārayi,_  
_mahantamativissutaṃ._  

</Paragraph>

</Division>
<Division number="73" book="ap1" bookNo="20" page="435">

<Paragraph number="7410" type="verses" book="ap1">
_Niṭṭhite ca mahādāne,_  
_daduṃ khajjakasañhitaṃ;_  
_Navaṃ dadhiṃ madhuñceva,_  
_vicinaṃ neva addasuṃ._  

</Paragraph>

</Division>
<Division number="74" book="ap1" bookNo="20" page="435">

<Paragraph number="7411" type="verses" book="ap1">
_Tadāhaṃ taṃ gahetvāna,_  
_navaṃ dadhiṃ madhumpi ca;_  
_Kammassāmigharaṃ gacchiṃ,_  
_tamesantā mamaddasuṃ._  

</Paragraph>

</Division>
<Division number="75" book="ap1" bookNo="20" page="435">

<Paragraph number="7412" type="verses" book="ap1">
_Sahassamapi datvāna,_  
_nālabhiṃsu ca taṃ dvayaṃ;_  
_Tatohaṃ evaṃ cintesiṃ,_  
_‘netaṃ hessati orakaṃ._  

</Paragraph>

</Division>
<Division number="76" book="ap1" bookNo="20" page="435">

<Paragraph number="7413" type="verses" book="ap1">
_Yathā ime janā sabbe,_  
_sakkaronti tathāgataṃ;_  
_Ahampi kāraṃ kassāmi,_  
_sasaṃghe lokanāyake’._  

</Paragraph>

</Division>
<Division number="77" book="ap1" bookNo="20" page="435">

<Paragraph number="7414" type="verses" book="ap1">
_Tadāhamevaṃ cintetvā,_  
_dadhiṃ madhuñca ekato;_  
_Madditvā lokanāthassa,_  
_sasaṃghassa adāsahaṃ._  

</Paragraph>

</Division>
<Division number="78" book="ap1" bookNo="20" page="435">

<Paragraph number="7415" type="verses" book="ap1">
_Tena kammena sukatena,_  
_cetanāpaṇidhīhi ca;_  
_Jahitvā mānusaṃ dehaṃ,_  
_tāvatiṃsamagacchahaṃ._  

</Paragraph>

</Division>
<Division number="79" book="ap1" bookNo="20" page="435">

<Paragraph number="7416" type="verses" book="ap1">
_Punāhaṃ bārāṇasiyaṃ,_  
_rājā hutvā mahāyaso;_  
_Sattukassa tadā duṭṭho,_  
_dvārarodhamakārayiṃ._  

</Paragraph>

</Division>
<Division number="80" book="ap1" bookNo="20" page="435">

<Paragraph number="7417" type="verses" book="ap1">
_Tadā tapassino ruddhā,_  
_ekāhaṃ rakkhitā ahuṃ;_  
_Tato tassa vipākena,_  
_pāpatiṃ nirayaṃ bhusaṃ._  

</Paragraph>

</Division>
<Division number="81" book="ap1" bookNo="20" page="435">

<Paragraph number="7418" type="verses" book="ap1">
_Pacchime ca bhave dāni,_  
_jātohaṃ koliye pure;_  
_Suppavāsā ca me mātā,_  
_mahāli licchavī pitā._  

</Paragraph>

</Division>
<Division number="82" book="ap1" bookNo="20" page="435">

<Paragraph number="7419" type="verses" book="ap1">
_Khattiye puññakammena,_  
_dvārarodhassa vāhasā;_  
_Satta vassāni nivasiṃ,_  
_mātukucchimhi dukkhito._  

</Paragraph>

</Division>
<Division number="83" book="ap1" bookNo="20" page="435">

<Paragraph number="7420" type="verses" book="ap1">
_Sattāhaṃ dvāramūḷhohaṃ,_  
_mahādukkhasamappito;_  
_Mātā me chandadānena,_  
_evaṃ āsi sudukkhitā._  

</Paragraph>

</Division>
<Division number="84" book="ap1" bookNo="20" page="436">

<Paragraph number="7421" type="verses" book="ap1">
_Suvatthitohaṃ nikkhanto,_  
_buddhena anukampito;_  
_Nikkhantadivaseyeva,_  
_pabbajiṃ anagāriyaṃ._  

</Paragraph>

</Division>
<Division number="85" book="ap1" bookNo="20" page="436">

<Paragraph number="7422" type="verses" book="ap1">
_Upajjhā sāriputto me,_  
_moggallāno mahiddhiko;_  
_Kese oropayanto me,_  
_anusāsi mahāmati._  

</Paragraph>

</Division>
<Division number="86" book="ap1" bookNo="20" page="436">

<Paragraph number="7423" type="verses" book="ap1">
_Kesesu chijjamānesu,_  
_arahattamapāpuṇiṃ;_  
_Devā nāgā manussā ca,_  
_paccaye upanenti me._  

</Paragraph>

</Division>
<Division number="87" book="ap1" bookNo="20" page="436">

<Paragraph number="7424" type="verses" book="ap1">
_Padumuttaranāthañca,_  
_vipassiñca vināyakaṃ;_  
_Yaṃ pūjayiṃ pamudito,_  
_paccayehi visesato._  

</Paragraph>

</Division>
<Division number="88" book="ap1" bookNo="20" page="436">

<Paragraph number="7425" type="verses" book="ap1">
_Tato tesaṃ visesena,_  
_kammānaṃ vipuluttamaṃ;_  
_Lābhaṃ labhāmi sabbattha,_  
_vane gāme jale thale._  

</Paragraph>

</Division>
<Division number="89" book="ap1" bookNo="20" page="436">

<Paragraph number="7426" type="verses" book="ap1">
_Revataṃ dassanatthāya,_  
_yadā yāti vināyako;_  
_Tiṃsabhikkhusahassehi,_  
_saha lokagganāyako._  

</Paragraph>

</Division>
<Division number="90" book="ap1" bookNo="20" page="436">

<Paragraph number="7427" type="verses" book="ap1">
_Tadā devopaṇītehi,_  
_mamatthāya mahāmati;_  
_Paccayehi mahāvīro,_  
_sasaṃgho lokanāyako._  

</Paragraph>

</Division>
<Division number="91" book="ap1" bookNo="20" page="436">

<Paragraph number="7428" type="verses" book="ap1">
_Upaṭṭhito mayā buddho,_  
_gantvā revatamaddasa;_  
_Tato jetavanaṃ gantvā,_  
_etadagge ṭhapesi maṃ._  

</Paragraph>

</Division>
<Division number="92" book="ap1" bookNo="20" page="436">

<Paragraph number="7429" type="verses" book="ap1">
_‘Lābhīnaṃ sīvali aggo,_  
_mama sissesu bhikkhavo’;_  
_Sabbalokahito satthā,_  
_kittayī parisāsu maṃ._  

</Paragraph>

</Division>
<Division number="93" book="ap1" bookNo="20" page="436">

<Paragraph number="7430" type="verses" book="ap1">

_ …pe… _  
_viharāmi anāsavo._  

</Paragraph>

</Division>
<Division number="94" book="ap1" bookNo="20" page="436">

<Paragraph number="7431" type="verses" book="ap1">

_ …pe… _  
_kataṃ buddhassa sāsanaṃ._  

</Paragraph>

</Division>
