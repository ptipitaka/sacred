<Division number="19" book="ap1">

<Paragraph number="3739" type="verses" book="ap1">
_“Nagare haṃsavatiyā,_  
_ahosiṃ tiṇahārako;_  
_Tiṇahārena jīvāmi,_  
_tena posemi dārake._  

</Paragraph>

</Division>
<Division number="20" book="ap1">

<Paragraph number="3740" type="verses" book="ap1">
_Padumuttaro nāma jino,_  
_sabbadhammāna pāragū;_  
_Tamandhakāraṃ nāsetvā,_  
_uppajji lokanāyako._  

</Paragraph>

</Division>
<Division number="21" book="ap1">

<Paragraph number="3741" type="verses" book="ap1">
_Sake ghare nisīditvā,_  
_evaṃ cintesahaṃ tadā;_  
_‘Buddho loke samuppanno,_  
_deyyadhammo ca natthi me._  

</Paragraph>

</Division>
<Division number="22" book="ap1">

<Paragraph number="3742" type="verses" book="ap1">
_Idaṃ me sāṭakaṃ ekaṃ,_  
_natthi me koci dāyako;_  
_Dukkho nirayasamphasso,_  
_ropayissāmi dakkhiṇaṃ’._  

</Paragraph>

</Division>
<Division number="23" book="ap1">

<Paragraph number="3743" type="verses" book="ap1">
_Evāhaṃ cintayitvāna,_  
_sakaṃ cittaṃ pasādayiṃ;_  
_Ekaṃ dussaṃ gahetvāna,_  
_buddhaseṭṭhassadāsahaṃ._  

</Paragraph>

</Division>
<Division number="24" book="ap1">

<Paragraph number="3744" type="verses" book="ap1">
_Ekaṃ dussaṃ daditvāna,_  
_ukkuṭṭhiṃ sampavattayiṃ;_  
_Yadi buddho tuvaṃ vīra,_  
_tārehi maṃ mahāmuni._  

</Paragraph>

</Division>
<Division number="25" book="ap1">

<Paragraph number="3745" type="verses" book="ap1">
_Padumuttaro lokavidū,_  
_āhutīnaṃ paṭiggaho;_  
_Mama dānaṃ pakittento,_  
_akā me anumodanaṃ._  

</Paragraph>

</Division>
<Division number="26" book="ap1">

<Paragraph number="3746" type="verses" book="ap1">
_‘Iminā ekadussena,_  
_cetanāpaṇidhīhi ca;_  
_Kappasatasahassāni,_  
_vinipātaṃ na gacchati._  

</Paragraph>

</Division>
<Division number="27" book="ap1">

<Paragraph number="3747" type="verses" book="ap1">
_Chattiṃsakkhattuṃ devindo,_  
_devarajjaṃ karissati;_  
_Tettiṃsakkhattuṃ rājā ca,_  
_cakkavattī bhavissati;_  
_Padesarajjaṃ vipulaṃ,_  
_gaṇanāto asaṅkhiyaṃ._  

</Paragraph>

</Division>
<Division number="28" book="ap1">

<Paragraph number="3748" type="verses" book="ap1">
_Devaloke manusse vā,_  
_saṃsaranto tuvaṃ bhave;_  
_Rūpavā guṇasampanno,_  
_anavakkantadehavā;_  
_Akkhobhaṃ amitaṃ dussaṃ,_  
_labhissati yadicchakaṃ’._  

</Paragraph>

</Division>
<Division number="29" book="ap1">

<Paragraph number="3749" type="verses" book="ap1">
_Idaṃ vatvāna sambuddho,_  
_jalajuttamanāmako;_  
_Nabhaṃ abbhuggamī dhīro,_  
_haṃsarājāva ambare._  

</Paragraph>

</Division>
<Division number="30" book="ap1">

<Paragraph number="3750" type="verses" book="ap1">
_Yaṃ yaṃ yonupapajjāmi,_  
_devattaṃ atha mānusaṃ;_  
_Bhoge me ūnatā natthi,_  
_ekadussassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="31" book="ap1">

<Paragraph number="3751" type="verses" book="ap1">
_Paduddhāre paduddhāre,_  
_dussaṃ nibbattate mamaṃ;_  
_Heṭṭhā dussamhi tiṭṭhāmi,_  
_upari chadanaṃ mama._  

</Paragraph>

</Division>
<Division number="32" book="ap1">

<Paragraph number="3752" type="verses" book="ap1">
_Cakkavāḷamupādāya,_  
_sakānanaṃ sapabbataṃ;_  
_Icchamāno cahaṃ ajja,_  
_dussehi chādayeyyahaṃ._  

</Paragraph>

</Division>
<Division number="33" book="ap1">

<Paragraph number="3753" type="verses" book="ap1">
_Teneva ekadussena,_  
_saṃsaranto bhavābhave;_  
_Suvaṇṇavaṇṇo hutvāna,_  
_saṃsarāmi bhavābhave._  

</Paragraph>

</Division>
<Division number="34" book="ap1">

<Paragraph number="3754" type="verses" book="ap1">
_Vipākaṃ ekadussassa,_  
_najjhagaṃ katthacikkhayaṃ;_  
_Ayaṃ me antimā jāti,_  
_vipaccati idhāpi me._  

</Paragraph>

</Division>
<Division number="35" book="ap1">

<Paragraph number="3755" type="verses" book="ap1">
_Satasahassito kappe,_  
_yaṃ dussamadadiṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_ekadussassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="36" book="ap1">

<Paragraph number="3756" type="verses" book="ap1">
_Kilesā jhāpitā mayhaṃ,_  
_bhavā sabbe samūhatā;_  
_Nāgova bandhanaṃ chetvā,_  
_viharāmi anāsavo._  

</Paragraph>

</Division>
<Division number="37" book="ap1">

<Paragraph number="3757" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="3758" book="ap1">
Itthaṃ sudaṃ āyasmā ekadussadāyako thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="3759" type="center" book="ap1">
Ekadussadāyakattherassāpadānaṃ catutthaṃ.
</Paragraph>
