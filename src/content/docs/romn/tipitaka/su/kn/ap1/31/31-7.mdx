<Paragraph number="3475" type="verses" book="ap1">
_“Siddhatthassa bhagavato,_  
_ārāmo ropito mayā;_  
_Sandacchāyesu rukkhesu,_  
_upāsantesu pakkhisu._  

</Paragraph>

</Division>
<Division number="33" book="ap1" bookNo="20" page="285">

<Paragraph number="3476" type="verses" book="ap1">
_Addasaṃ virajaṃ buddhaṃ,_  
_āhutīnaṃ paṭiggahaṃ;_  
_Ārāmaṃ abhināmesiṃ,_  
_lokajeṭṭhaṃ narāsabhaṃ._  

</Paragraph>

</Division>
<Division number="34" book="ap1" bookNo="20" page="285">

<Paragraph number="3477" type="verses" book="ap1">
_Haṭṭho haṭṭhena cittena,_  
_phalaṃ pupphamadāsahaṃ;_  
_Tato jātappasādova,_  
_taṃ vanaṃ pariṇāmayiṃ._  

</Paragraph>

</Division>
<Division number="35" book="ap1" bookNo="20" page="285">

<Paragraph number="3478" type="verses" book="ap1">
_Buddhassa yamidaṃ dāsiṃ,_  
_vippasannena cetasā;_  
_Bhave nibbattamānamhi,_  
_nibbattati phalaṃ mama._  

</Paragraph>

</Division>
<Division number="36" book="ap1" bookNo="20" page="285">

<Paragraph number="3479" type="verses" book="ap1">
_Catunnavutito kappe,_  
_yaṃ ārāmamadaṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_ārāmassa idaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="37" book="ap1" bookNo="20" page="285">

<Paragraph number="3480" type="verses" book="ap1">
_Sattatiṃse ito kappe,_  
_sattāsuṃ mudusītalā;_  
_Sattaratanasampannā,_  
_cakkavattī mahabbalā._  

</Paragraph>

</Division>
<Division number="38" book="ap1" bookNo="20" page="286">

<Paragraph number="3481" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="3482" book="ap1">
Itthaṃ sudaṃ āyasmā ārāmadāyako thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="3483" type="center" book="ap1">
Ārāmadāyakattherassāpadānaṃ sattamaṃ.
</Paragraph>
