<Division number="93" book="ap1">

<Paragraph number="1315" type="verses" book="ap1">
_“Atthadassissa munino,_  
_lokajeṭṭhassa tādino;_  
_Iṭṭhakāhi cinitvāna,_  
_caṅkamaṃ kārayiṃ ahaṃ._  

</Paragraph>

</Division>
<Division number="94" book="ap1">

<Paragraph number="1316" type="verses" book="ap1">
_Uccato pañcaratanaṃ,_  
_caṅkamaṃ sādhumāpitaṃ;_  
_Āyāmato hatthasataṃ,_  
_bhāvanīyyaṃ manoramaṃ._  

</Paragraph>

</Division>
<Division number="95" book="ap1">

<Paragraph number="1317" type="verses" book="ap1">
_Paṭiggahesi bhagavā,_  
_atthadassī naruttamo;_  
_Hatthena pulinaṃ gayha,_  
_imā gāthā abhāsatha._  

</Paragraph>

</Division>
<Division number="96" book="ap1">

<Paragraph number="1318" type="verses" book="ap1">
_‘Iminā pulinadānena,_  
_caṅkamaṃ sukatena ca;_  
_Sattaratanasampannaṃ,_  
_pulinaṃ anubhossati._  

</Paragraph>

</Division>
<Division number="97" book="ap1">

<Paragraph number="1319" type="verses" book="ap1">
_Tīṇi kappāni devesu,_  
_devarajjaṃ karissati;_  
_Anubhossati sampattiṃ,_  
_accharāhi purakkhato._  

</Paragraph>

</Division>
<Division number="98" book="ap1">

<Paragraph number="1320" type="verses" book="ap1">
_Manussalokamāgantvā,_  
_rājā raṭṭhe bhavissati;_  
_Tikkhattuṃ cakkavattī ca,_  
_mahiyā so bhavissati’._  

</Paragraph>

</Division>
<Division number="99" book="ap1">

<Paragraph number="1321" type="verses" book="ap1">
_Aṭṭhārase kappasate,_  
_yaṃ kammamakariṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_caṅkamassa idaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="100" book="ap1">

<Paragraph number="1322" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="1323" book="ap1">
Itthaṃ sudaṃ āyasmā caṅkamanadāyako thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="1324" type="center" book="ap1">
Caṅkamanadāyakattherassāpadānaṃ aṭṭhamaṃ.
</Paragraph>
