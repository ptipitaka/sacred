<Division number="18" book="ap1">

<Paragraph number="2604" type="verses" book="ap1">
_“Tisso nāmāsi bhagavā,_  
_lokajeṭṭho narāsabho;_  
_Purakkhato sāvakehi,_  
_rathiyaṃ paṭipajjatha._  

</Paragraph>

</Division>
<Division number="19" book="ap1">

<Paragraph number="2605" type="verses" book="ap1">
_Pañca uppalahatthā ca,_  
_cāturā ṭhapitā mayā;_  
_Āhutiṃ dātukāmohaṃ,_  
_paggaṇhiṃ vatasiddhiyā._  

</Paragraph>

</Division>
<Division number="20" book="ap1">

<Paragraph number="2606" type="verses" book="ap1">
_Suvaṇṇavaṇṇaṃ sambuddhaṃ,_  
_Gacchantaṃ antarāpaṇe;_  
_Buddharaṃsīhi phuṭṭhosmi,_  
_Pūjesiṃ dvipaduttamaṃ._  

</Paragraph>

</Division>
<Division number="21" book="ap1">

<Paragraph number="2607" type="verses" book="ap1">
_Dvenavute ito kappe,_  
_yaṃ pupphamabhipūjayiṃ;_  
_Duggatiṃ nābhijānāmi,_  
_buddhapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="22" book="ap1">

<Paragraph number="2608" type="verses" book="ap1">
_Ito terasakappamhi,_  
_pañca susabhasammatā;_  
_Sattaratanasampannā,_  
_cakkavattī mahabbalā._  

</Paragraph>

</Division>
<Division number="23" book="ap1">

<Paragraph number="2609" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="2610" book="ap1">
Itthaṃ sudaṃ āyasmā pañcahatthiyo thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="2611" type="center" book="ap1">
Pañcahatthiyattherassāpadānaṃ catutthaṃ.
</Paragraph>
