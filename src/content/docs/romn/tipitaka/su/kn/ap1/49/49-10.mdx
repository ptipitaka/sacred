_“Ajjhogāhetvā sālavanaṃ,_  
_sukato assamo mama;_  
_Sālapupphehi sañchanno,_  
_vasāmi vipine tadā._

</Division>
<Division number="191" book="ap1">

<Paragraph number="6394" type="verses" book="ap1">
_Piyadassī ca bhagavā,_  
_sayambhū aggapuggalo;_  
_Vivekakāmo sambuddho,_  
_sālavanamupāgami._  

</Paragraph>

</Division>
<Division number="192" book="ap1">

<Paragraph number="6395" type="verses" book="ap1">
_Assamā abhinikkhamma,_  
_pavanaṃ agamāsahaṃ;_  
_Mūlaphalaṃ gavesanto,_  
_āhiṇḍāmi vane tadā._  

</Paragraph>

</Division>
<Division number="193" book="ap1">

<Paragraph number="6396" type="verses" book="ap1">
_Tatthaddasāsiṃ sambuddhaṃ,_  
_piyadassiṃ mahāyasaṃ;_  
_Sunisinnaṃ samāpannaṃ,_  
_virocantaṃ mahāvane._  

</Paragraph>

</Division>
<Division number="194" book="ap1">

<Paragraph number="6397" type="verses" book="ap1">
_Catudaṇḍe ṭhapetvāna,_  
_buddhassa uparī ahaṃ;_  
_Maṇḍapaṃ sukataṃ katvā,_  
_sālapupphehi chādayiṃ._  

</Paragraph>

</Division>
<Division number="195" book="ap1">

<Paragraph number="6398" type="verses" book="ap1">
_Sattāhaṃ dhārayitvāna,_  
_maṇḍapaṃ sālachāditaṃ;_  
_Tattha cittaṃ pasādetvā,_  
_buddhaseṭṭhamavandahaṃ._  

</Paragraph>

</Division>
<Division number="196" book="ap1">

<Paragraph number="6399" type="verses" book="ap1">
_Bhagavā tamhi samaye,_  
_vuṭṭhahitvā samādhito;_  
_Yugamattaṃ pekkhamāno,_  
_nisīdi purisuttamo._  

</Paragraph>

</Division>
<Division number="197" book="ap1">

<Paragraph number="6400" type="verses" book="ap1">
_Sāvako varuṇo nāma,_  
_piyadassissa satthuno;_  
_Vasīsatasahassehi,_  
_upagacchi vināyakaṃ._  

</Paragraph>

</Division>
<Division number="198" book="ap1">

<Paragraph number="6401" type="verses" book="ap1">
_Piyadassī ca bhagavā,_  
_lokajeṭṭho narāsabho;_  
_Bhikkhusaṃghe nisīditvā,_  
_sitaṃ pātukarī jino._  

</Paragraph>

</Division>
<Division number="199" book="ap1">

<Paragraph number="6402" type="verses" book="ap1">
_Anuruddho upaṭṭhāko,_  
_piyadassissa satthuno;_  
_Ekaṃsaṃ cīvaraṃ katvā,_  
_apucchittha mahāmuniṃ._  

</Paragraph>

</Division>
<Division number="200" book="ap1">

<Paragraph number="6403" type="verses" book="ap1">
_‘Ko nu kho bhagavā hetu,_  
_sitakammassa satthuno;_  
_Kāraṇe vijjamānamhi,_  
_satthā pātukare sitaṃ’._  

</Paragraph>

</Division>
<Division number="201" book="ap1">

<Paragraph number="6404" type="verses" book="ap1">
_‘Sattāhaṃ sālacchadanaṃ,_  
_yo me dhāresi māṇavo;_  
_Tassa kammaṃ saritvāna,_  
_sitaṃ pātukariṃ ahaṃ._  

</Paragraph>

</Division>
<Division number="202" book="ap1">

<Paragraph number="6405" type="verses" book="ap1">
_Anokāsaṃ na passāmi,_  
_yattha puññaṃ vipaccati;_  
_Devaloke manusse vā,_  
_okāsova na sammati._  

</Paragraph>

</Division>
<Division number="203" book="ap1">

<Paragraph number="6406" type="verses" book="ap1">
_Devaloke vasantassa,_  
_puññakammasamaṅgino;_  
_Yāvatā parisā tassa,_  
_sālacchannā bhavissati._  

</Paragraph>

</Division>
<Division number="204" book="ap1">

<Paragraph number="6407" type="verses" book="ap1">
_Tattha dibbehi naccehi,_  
_gītehi vāditehi ca;_  
_Ramissati sadā santo,_  
_puññakammasamāhito._  

</Paragraph>

</Division>
<Division number="205" book="ap1">

<Paragraph number="6408" type="verses" book="ap1">
_Yāvatā parisā tassa,_  
_gandhagandhī bhavissati;_  
_Sālassa pupphavasso ca,_  
_pavassissati tāvade._  

</Paragraph>

</Division>
<Division number="206" book="ap1">

<Paragraph number="6409" type="verses" book="ap1">
_Tato cutoyaṃ manujo,_  
_mānusaṃ āgamissati;_  
_Idhāpi sālacchadanaṃ,_  
_sabbakālaṃ dharissati._  

</Paragraph>

</Division>
<Division number="207" book="ap1">

_sammatāḷasamāhitaṃ;_  
_Parivāressanti maṃ niccaṃ,_  
_buddhapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="208" book="ap1">

<Paragraph number="6411" type="verses" book="ap1">

_sālavassaṃ pavassati;_  
_Puññakammena saṃyuttaṃ,_  
_vassate sabbakālikaṃ._  

</Paragraph>

</Division>

_okkākakulasambhavo;_  
_Gotamo nāma gottena,_  
_satthā loke bhavissati._  

</Paragraph>

</Division>
<Division number="210" book="ap1">

<Paragraph number="6413" type="verses" book="ap1">

_oraso dhammanimmito;_  
_Sabbāsave pariññāya,_  
_nibbāyissatināsavo._  

</Paragraph>

</Division>
<Division number="211" book="ap1">

<Paragraph number="6414" type="verses" book="ap1">
_Dhammaṃ abhisamentassa,_  
_sālacchannaṃ bhavissati;_  
_Citake jhāyamānassa,_  
_chadanaṃ tattha hessati’._  

</Paragraph>

</Division>
<Division number="212" book="ap1">

<Paragraph number="6415" type="verses" book="ap1">

_piyadassī mahāmuni;_  
_Parisāya dhammaṃ desesi,_  
_tappento dhammavuṭṭhiyā._  

</Paragraph>

</Division>
<Division number="213" book="ap1">

<Paragraph number="6416" type="verses" book="ap1">

_devarajjamakārayiṃ;_  
_Saṭṭhi ca sattakkhattuñca,_  
_cakkavattī ahosahaṃ._  

</Paragraph>

</Division>
<Division number="214" book="ap1">

_labhāmi vipulaṃ sukhaṃ;_  
_Idhāpi sālacchadanaṃ,_  
_maṇḍapassa idaṃ phalaṃ._  

</Paragraph>

</Division>

_Idhāpi sālacchadanaṃ,_  
_hessati sabbakālikaṃ._  

</Paragraph>

</Division>

_gotamaṃ sakyapuṅgavaṃ;_

_hitvā jayaparājayaṃ._  

</Paragraph>
