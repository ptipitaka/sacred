<Paragraph number="3511" type="verses" book="ap1">
_“Sujāto sālalaṭṭhīva,_  
_sobhañjanamivuggato;_  
_Indalaṭṭhirivākāse,_  
_virocati sadā jino._  

</Paragraph>

</Division>
<Division number="5" book="ap1">

<Paragraph number="3512" type="verses" book="ap1">
_Tassa devātidevassa,_  
_vessabhussa mahesino;_  
_Adāsi bhojanamahaṃ,_  
_vippasannena cetasā._  

</Paragraph>

</Division>
<Division number="6" book="ap1">

<Paragraph number="3513" type="verses" book="ap1">
_Taṃ me buddho anumodi,_  
_sayambhū aparājito;_  
_Bhave nibbattamānamhi,_  
_phalaṃ nibbattatū tava._  

</Paragraph>

</Division>
<Division number="7" book="ap1">

<Paragraph number="3514" type="verses" book="ap1">
_Ekattiṃse ito kappe,_  
_yaṃ dānamadadiṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_bhojanassa idaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="8" book="ap1">

<Paragraph number="3515" type="verses" book="ap1">
_Pañcavīse ito kappe,_  
_eko āsiṃ amittako;_  
_Sattaratanasampanno,_  
_cakkavattī mahabbalo._  

</Paragraph>

</Division>
<Division number="9" book="ap1">

<Paragraph number="3516" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="3517" book="ap1">
Itthaṃ sudaṃ āyasmā bhojanadāyako thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="3518" type="center" book="ap1">
Bhojanadāyakattherassāpadānaṃ dutiyaṃ.
</Paragraph>
