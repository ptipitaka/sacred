<Division number="31" book="ap1">

<Paragraph number="2864" type="verses" book="ap1">
_“Tālavaṇṭaṃ mayā dinnaṃ,_  
_tissassādiccabandhuno;_  
_Gimhanibbāpanatthāya,_  
_pariḷāhopasantiyā._  

</Paragraph>

</Division>
<Division number="32" book="ap1">

<Paragraph number="2865" type="verses" book="ap1">
_Sannibbāpemi rāgaggiṃ,_  
_dosaggiñca taduttariṃ;_  
_Nibbāpemi ca mohaggiṃ,_  
_tālavaṇṭassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="33" book="ap1">

<Paragraph number="2866" type="verses" book="ap1">
_Kilesā jhāpitā mayhaṃ,_  
_bhavā sabbe samūhatā;_  
_Dhāremi antimaṃ dehaṃ,_  
_sammāsambuddhasāsane._  

</Paragraph>

</Division>
<Division number="34" book="ap1">

<Paragraph number="2867" type="verses" book="ap1">
_Dvenavute ito kappe,_  
_yaṃ kammamakariṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_tālavaṇṭassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="35" book="ap1">

<Paragraph number="2868" type="verses" book="ap1">
_Tesaṭṭhimhi ito kappe,_  
_mahānāmasanāmako;_  
_Sattaratanasampanno,_  
_cakkavattī mahabbalo._  

</Paragraph>

</Division>
<Division number="36" book="ap1">

<Paragraph number="2869" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="2870" book="ap1">
Itthaṃ sudaṃ āyasmā tālavaṇṭadāyako thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="2871" type="center" book="ap1">
Tālavaṇṭadāyakattherassāpadānaṃ sattamaṃ.
</Paragraph>
