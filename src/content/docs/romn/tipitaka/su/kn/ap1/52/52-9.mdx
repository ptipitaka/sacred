<Paragraph number="6757" type="verses" book="ap1">
_“Varuṇo nāma nāmena,_  
_brāhmaṇo mantapāragū;_  
_Chaḍḍetvā dasa puttāni,_  
_vanamajjhogahiṃ tadā._  

</Paragraph>

</Division>
<Division number="52" book="ap1" bookNo="20" page="433">

<Paragraph number="6758" type="verses" book="ap1">
_Assamaṃ sukataṃ katvā,_  
_suvibhattaṃ manoramaṃ;_  
_Paṇṇasālaṃ karitvāna,_  
_vasāmi vipine ahaṃ._  

</Paragraph>

</Division>
<Division number="53" book="ap1" bookNo="20" page="433">

<Paragraph number="6759" type="verses" book="ap1">
_Padumuttaro lokavidū,_  
_āhutīnaṃ paṭiggaho;_  
_Mamuddharitukāmo so,_  
_āgacchi mama assamaṃ._  

</Paragraph>

</Division>
<Division number="54" book="ap1" bookNo="20" page="433">

<Paragraph number="6760" type="verses" book="ap1">
_Yāvatā vanasaṇḍamhi,_  
_obhāso vipulo ahu;_  
_Buddhassa ānubhāvena,_  
_pajjalī vipinaṃ tadā._  

</Paragraph>

</Division>
<Division number="55" book="ap1" bookNo="20" page="433">

<Paragraph number="6761" type="verses" book="ap1">
_Disvāna taṃ pāṭihīraṃ,_  
_buddhaseṭṭhassa tādino;_  
_Pattapuṭaṃ gahetvāna,_  
_phalena pūjayiṃ ahaṃ._  

</Paragraph>

</Division>
<Division number="56" book="ap1" bookNo="20" page="433">

<Paragraph number="6762" type="verses" book="ap1">
_Upagantvāna sambuddhaṃ,_  
_sahakhārimadāsahaṃ;_  
_Anukampāya me buddho,_  
_idaṃ vacanamabravi._  

</Paragraph>

</Division>
<Division number="57" book="ap1" bookNo="20" page="433">

<Paragraph number="6763" type="verses" book="ap1">
_‘Khāribhāraṃ gahetvāna,_  
_pacchato ehi me tuvaṃ;_  
_Paribhutte ca saṃghamhi,_  
_puññaṃ tava bhavissati’._  

</Paragraph>

</Division>
<Division number="58" book="ap1" bookNo="20" page="433">

<Paragraph number="6764" type="verses" book="ap1">
_Puṭakantaṃ gahetvāna,_  
_bhikkhusaṃghassadāsahaṃ;_  
_Tattha cittaṃ pasādetvā,_  
_tusitaṃ upapajjahaṃ._  

</Paragraph>

</Division>
<Division number="59" book="ap1" bookNo="20" page="434">

<Paragraph number="6765" type="verses" book="ap1">
_Tattha dibbehi naccehi,_  
_gītehi vāditehi ca;_  
_Puññakammena saṃyuttaṃ,_  
_anubhomi sadā sukhaṃ._  

</Paragraph>

</Division>
<Division number="60" book="ap1" bookNo="20" page="434">

<Paragraph number="6766" type="verses" book="ap1">
_Yaṃ yaṃ yonupapajjāmi,_  
_devattaṃ atha mānusaṃ;_  
_Bhoge me ūnatā natthi,_  
_phaladānassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="61" book="ap1" bookNo="20" page="434">

<Paragraph number="6767" type="verses" book="ap1">
_Yāvatā caturo dīpā,_  
_sasamuddā sapabbatā;_  
_Phalaṃ buddhassa datvāna,_  
_issaraṃ kārayāmahaṃ._  

</Paragraph>

</Division>
<Division number="62" book="ap1" bookNo="20" page="434">

<Paragraph number="6768" type="verses" book="ap1">
_Yāvatā me pakkhigaṇā,_  
_ākāse uppatanti ce;_  
_Tepi maṃ vasamanventi,_  
_phaladānassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="63" book="ap1" bookNo="20" page="434">

<Paragraph number="6769" type="verses" book="ap1">
_Yāvatā vanasaṇḍamhi,_  
_yakkhā bhūtā ca rakkhasā;_  
_Kumbhaṇḍā garuḷā cāpi,_  
_pāricariyaṃ upenti me._  

</Paragraph>

</Division>
<Division number="64" book="ap1" bookNo="20" page="434">

<Paragraph number="6770" type="verses" book="ap1">
_Kummā soṇā madhukārā,_  
_ḍaṃsā ca makasā ubho;_  
_Tepi maṃ vasamanventi,_  
_phaladānassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="65" book="ap1" bookNo="20" page="434">

<Paragraph number="6771" type="verses" book="ap1">
_Supaṇṇā nāma sakuṇā,_  
_pakkhijātā mahabbalā;_  
_Tepi maṃ saraṇaṃ yanti,_  
_phaladānassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="66" book="ap1" bookNo="20" page="434">

<Paragraph number="6772" type="verses" book="ap1">
_Yepi dīghāyukā nāgā,_  
_iddhimanto mahāyasā;_  
_Tepi maṃ vasamanventi,_  
_phaladānassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="67" book="ap1" bookNo="20" page="434">

<Paragraph number="6773" type="verses" book="ap1">
_Sīhā byagghā ca dīpī ca,_  
_acchakokataracchakā;_  
_Tepi maṃ vasamanventi,_  
_phaladānassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="68" book="ap1" bookNo="20" page="434">

<Paragraph number="6774" type="verses" book="ap1">
_Osadhītiṇavāsī ca,_  
_ye ca ākāsavāsino;_  
_Sabbe maṃ saraṇaṃ yanti,_  
_phaladānassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="69" book="ap1" bookNo="20" page="434">

<Paragraph number="6775" type="verses" book="ap1">
_Sududdasaṃ sunipuṇaṃ,_  
_gambhīraṃ suppakāsitaṃ;_  
_Phassayitvā viharāmi,_  
_phaladānassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="70" book="ap1" bookNo="20" page="434">

<Paragraph number="6776" type="verses" book="ap1">
_Vimokkhe aṭṭha phusitvā,_  
_viharāmi anāsavo;_  
_Ātāpī nipako cāhaṃ,_  
_phaladānassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="71" book="ap1" bookNo="20" page="434">

<Paragraph number="6777" type="verses" book="ap1">
_Ye phalaṭṭhā buddhaputtā,_  
_khīṇadosā mahāyasā;_  
_Ahamaññataro tesaṃ,_  
_phaladānassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="72" book="ap1" bookNo="20" page="435">

<Paragraph number="6778" type="verses" book="ap1">
_Abhiññāpāramiṃ gantvā,_  
_sukkamūlena codito;_  
_Sabbāsave pariññāya,_  
_viharāmi anāsavo._  

</Paragraph>

</Division>
<Division number="73" book="ap1" bookNo="20" page="435">

<Paragraph number="6779" type="verses" book="ap1">
_Tevijjā iddhipattā ca,_  
_buddhaputtā mahāyasā;_  
_Dibbasotasamāpannā,_  
_tesaṃ aññataro ahaṃ._  

</Paragraph>

</Division>
<Division number="74" book="ap1" bookNo="20" page="435">

<Paragraph number="6780" type="verses" book="ap1">
_Satasahassito kappe,_  
_yaṃ phalaṃ adadiṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_phaladānassidaṃ phalaṃ._  

</Paragraph>

</Division>

_ …pe… _  
_viharāmi anāsavo._  

</Paragraph>

</Division>

_ …pe… _

</Paragraph>

</Division>

_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>
