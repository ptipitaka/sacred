_sabbadhammesu cakkhumā;_  
_Ito satasahassamhi,_  
_kappe uppajji nāyako._

_gagane viya tārakā;_  
_Evaṃ pāvacanaṃ tassa,_  
_arahantehi cittitaṃ._  

</Paragraph>

_manujehi purakkhato;_  
_Samaṇabrāhmaṇākiṇṇe,_  
_janamajjhe jinuttamo._  

</Paragraph>

</Division>
<Division number="99" book="ap1" bookNo="20" page="437">

_Pabhāhi anurañjanto,_  
_loke lokantagū jino;_  
_Vacanena vibodhento,_  
_veneyyapadumāni so._  

</Paragraph>

</Division>
<Division number="100" book="ap1" bookNo="20" page="437">

<Paragraph number="7439" type="verses" book="ap1">

_catūhi purisuttamo;_  
_Pahīnabhayasārajjo,_  
_khemappatto visārado._  

</Paragraph>

</Division>
<Division number="101" book="ap1" bookNo="20" page="437">

_Āsabhaṃ pavaraṃ ṭhānaṃ,_  
_buddhabhūmiñca kevalaṃ;_  
_Paṭijānāti lokaggo,_  
_natthi sañcodako kvaci._  

</Paragraph>

</Division>
<Division number="102" book="ap1" bookNo="20" page="437">

<Paragraph number="7441" type="verses" book="ap1">
_Sīhanādamasambhītaṃ,_  
_nadato tassa tādino;_  
_Devo naro vā brahmā vā,_  
_paṭivattā na vijjati._  

</Paragraph>

</Division>
<Division number="103" book="ap1" bookNo="20" page="437">

<Paragraph number="7442" type="verses" book="ap1">
_Desento pavaraṃ dhammaṃ,_  
_santārento sadevakaṃ;_  
_Dhammacakkaṃ pavatteti,_  
_parisāsu visārado._  

</Paragraph>

</Division>
<Division number="104" book="ap1" bookNo="20" page="437">

<Paragraph number="7443" type="verses" book="ap1">
_Paṭibhānavataṃ aggaṃ,_  
_sāvakaṃ sādhusammataṃ;_  
_Guṇaṃ bahuṃ pakittetvā,_  
_etadagge ṭhapesi taṃ._  

</Paragraph>

</Division>
<Division number="105" book="ap1" bookNo="20" page="437">

<Paragraph number="7444" type="verses" book="ap1">
_Tadāhaṃ haṃsavatiyaṃ,_  
_brāhmaṇo sādhusammato;_  
_Sabbavedavidū jāto,_  
_vāgīso vādisūdano._  

</Paragraph>

</Division>
<Division number="106" book="ap1" bookNo="20" page="437">

<Paragraph number="7445" type="verses" book="ap1">
_Upecca taṃ mahāvīraṃ,_  
_sutvāhaṃ dhammadesanaṃ;_  
_Pītivaraṃ paṭilabhiṃ,_  
_sāvakassa guṇe rato._  

</Paragraph>

</Division>
<Division number="107" book="ap1" bookNo="20" page="438">

<Paragraph number="7446" type="verses" book="ap1">
_Nimantetvāva sugataṃ,_  
_sasaṃghaṃ lokanandanaṃ;_  
_Sattāhaṃ bhojayitvāhaṃ,_  
_dussehacchādayiṃ tadā._  

</Paragraph>

</Division>
<Division number="108" book="ap1" bookNo="20" page="438">

<Paragraph number="7447" type="verses" book="ap1">
_Nipacca sirasā pāde,_  
_katokāso katañjalī;_  
_Ekamantaṃ ṭhito haṭṭho,_  
_santhaviṃ jinamuttamaṃ._  

</Paragraph>

</Division>
<Division number="109" book="ap1" bookNo="20" page="438">

_namo te isisattama;_  
_Namo te sabbalokagga,_  
_namo te abhayaṃ kara._  

</Paragraph>

</Division>
<Division number="110" book="ap1" bookNo="20" page="438">

_namo te diṭṭhisūdana;_  
_Namo te santisukhada,_  
_namo te saraṇaṃ kara._  

</Paragraph>

</Division>
<Division number="111" book="ap1" bookNo="20" page="438">

_bhītānaṃ abhayappado;_  
_Vissāmabhūmi santānaṃ,_  
_saraṇaṃ saraṇesinaṃ’._  

</Paragraph>

</Division>
<Division number="112" book="ap1" bookNo="20" page="438">

<Paragraph number="7451" type="verses" book="ap1">
_Evamādīhi sambuddhaṃ,_  
_santhavitvā mahāguṇaṃ;_  
_Avocaṃ vādisūdassa,_  
_gatiṃ pappomi bhikkhuno._  

</Paragraph>

</Division>
<Division number="113" book="ap1" bookNo="20" page="438">

<Paragraph number="7452" type="verses" book="ap1">
_Tadā avoca bhagavā,_  
_anantapaṭibhānavā;_  
_‘Yo so buddhaṃ abhojesi,_  
_sattāhaṃ sahasāvakaṃ._  

</Paragraph>

</Division>
<Division number="114" book="ap1" bookNo="20" page="438">

<Paragraph number="7453" type="verses" book="ap1">
_Guṇañca me pakittesi,_  
_pasanno sehi pāṇibhi;_  
_Eso patthayate ṭhānaṃ,_  
_vādisūdassa bhikkhuno._  

</Paragraph>

</Division>
<Division number="115" book="ap1" bookNo="20" page="438">

<Paragraph number="7454" type="verses" book="ap1">
_Anāgatamhi addhāne,_  
_lacchase taṃ manorathaṃ;_  
_Devamānusasampattiṃ,_  
_anubhotvā anappakaṃ._  

</Paragraph>

</Division>
<Division number="116" book="ap1" bookNo="20" page="438">

_okkākakulasambhavo;_  
_Gotamo nāma gottena,_  
_satthā loke bhavissati._  

</Paragraph>

</Division>
<Division number="117" book="ap1" bookNo="20" page="438">

<Paragraph number="7456" type="verses" book="ap1">
_Tassa dhammesu dāyādo,_  
_oraso dhammanimmito;_  
_Vaṅgīso nāma nāmena,_  
_hessati satthu sāvako’._  

</Paragraph>

</Division>
<Division number="118" book="ap1" bookNo="20" page="438">

_Taṃ sutvā mudito hutvā,_  
_yāvajīvaṃ tadā jinaṃ;_  
_Paccayehi upaṭṭhāsiṃ,_  
_mettacitto tathāgataṃ._  

</Paragraph>

</Division>
<Division number="119" book="ap1" bookNo="20" page="439">

<Paragraph number="7458" type="verses" book="ap1">
_Tena kammena sukatena,_  
_cetanāpaṇidhīhi ca;_  
_Jahitvā mānusaṃ dehaṃ,_  
_tusitaṃ agamāsahaṃ._  

</Paragraph>

</Division>
<Division number="120" book="ap1" bookNo="20" page="439">

<Paragraph number="7459" type="verses" book="ap1">
_Pacchime ca bhave dāni,_  
_jāto vippakule ahaṃ;_  
_Paccājāto yadā āsiṃ,_  
_jātiyā sattavassiko._  

</Paragraph>

</Division>
<Division number="121" book="ap1" bookNo="20" page="439">

_Sabbavedavidū jāto,_  
_vādasatthavisārado;_  
_Vādissaro cittakathī,_  
_paravādappamaddano._  

</Paragraph>

</Division>
<Division number="122" book="ap1" bookNo="20" page="439">

_Vaṅge jātoti vaṅgīso,_  
_vacane issaroti vā;_  
_Vaṅgīso iti me nāmaṃ,_  
_abhavī lokasammataṃ._  

</Paragraph>

</Division>
<Division number="123" book="ap1" bookNo="20" page="439">

_ṭhito paṭhamayobbane;_  
_Tadā rājagahe ramme,_  
_sāriputtamahaddasaṃ._  


</Paragraph>
</Division>

</Paragraph>

_pattapāṇiṃ susaṃvutaṃ;_  
_Alolakkhiṃ mitabhāṇiṃ,_  
_yugamattaṃ nidakkhitaṃ._  

</Paragraph>

</Division>

_avocaṃ mamanucchavaṃ;_  
_Kaṇikāraṃva nicitaṃ,_  
_cittaṃ gāthāpadaṃ ahaṃ._  

</Paragraph>

</Division>

_sambuddhaṃ lokanāyakaṃ;_  
_Tadā so paṇḍito vīro,_  
_uttariṃ samavoca me._  

</Paragraph>

</Division>

_katvā duddasamuttamaṃ;_  
_Vicittapaṭibhānehi,_  
_tosito tena tādinā._  

</Paragraph>

</Division>

_‘pabbājehī’ti maṃ bravi;_  
_Tato maṃ sa mahāpañño,_  
_buddhaseṭṭhamupānayi._  

</Paragraph>

</Division>
<Division number="129" book="ap1" bookNo="20" page="439">

_nisīdiṃ satthu santike;_  
_Mamāha vadataṃ seṭṭho,_  
_‘kacci vaṅgīsa jānāsi’._  

</Paragraph>

</Division>

_‘Matasīsaṃ vanacchuddhaṃ,_  
_api bārasavassikaṃ;_  
_Tava vijjāvisesena,_  
_sace sakkosi vācaya’._  

</Paragraph>

</Division>

_tīṇi sīsāni dassayi;_  
_Nirayanaradevesu,_  
_upapanne avācayiṃ’._  

</Paragraph>

</Division>

_pabbajjaṃ samayācisaṃ._  

</Paragraph>

</Division>

_ujjhāyantiha bhikkhavo._  

</Paragraph>

</Division>

_ṭhānaso paṭibhanti taṃ’._  

</Paragraph>

</Division>

_ṭhānaso santhavāhi maṃ’._  

</Paragraph>

</Division>

_gāthāhi isisattamaṃ;_

_jino agge ṭhapesi maṃ._  

</Paragraph>
