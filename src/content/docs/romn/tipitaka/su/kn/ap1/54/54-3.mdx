_“Padumuttaro nāma jino,_  
_sabbadhammāna pāragū;_

_ravīva saradambare._  

</Paragraph>

</Division>
<Division number="67" book="ap1" bookNo="20" page="434">

<Paragraph number="7018" type="verses" book="ap1">

_veneyyapadumāni so;_  
_Kilesapaṅkaṃ soseti,_  
_matiraṃsīhi nāyako._  

</Paragraph>

</Division>
<Division number="68" book="ap1" bookNo="20" page="434">

<Paragraph number="7019" type="verses" book="ap1">
_Titthiyānaṃ yase hanti,_  
_khajjotābhā yathā ravi;_  
_Saccatthābhaṃ pakāseti,_  
_ratanaṃva divākaro._  

</Paragraph>

</Division>
<Division number="69" book="ap1" bookNo="20" page="434">

<Paragraph number="7020" type="verses" book="ap1">
_Guṇānaṃ āyatibhūto,_  
_ratanānaṃva sāgaro;_  
_Pajjunnoriva bhūtāni,_  
_dhammameghena vassati._  

</Paragraph>

</Division>
<Division number="70" book="ap1" bookNo="20" page="434">

<Paragraph number="7021" type="verses" book="ap1">
_Akkhadasso tadā āsiṃ,_  
_nagare haṃsasavhaye;_  
_Upecca dhammamassosiṃ,_  
_jalajuttamanāmino._  

</Paragraph>

</Division>
<Division number="71" book="ap1" bookNo="20" page="434">

<Paragraph number="7022" type="verses" book="ap1">
_Ovādakassa bhikkhūnaṃ,_  
_sāvakassa katāvino;_  
_Guṇaṃ pakāsayantassa,_  
_tappayantassa me manaṃ._  

</Paragraph>

</Division>
<Division number="72" book="ap1" bookNo="20" page="435">

<Paragraph number="7023" type="verses" book="ap1">
_Sutvā patīto sumano,_  
_nimantetvā tathāgataṃ;_  
_Sasissaṃ bhojayitvāna,_  
_taṃ ṭhānamabhipatthayiṃ._  

</Paragraph>

</Division>
<Division number="73" book="ap1" bookNo="20" page="435">

<Paragraph number="7024" type="verses" book="ap1">
_Tadā haṃsasamabhāgo,_  
_haṃsadundubhinissano;_  
_‘Passathetaṃ mahāmattaṃ,_  
_vinicchayavisāradaṃ._  

</Paragraph>

</Division>
<Division number="74" book="ap1" bookNo="20" page="435">

<Paragraph number="7025" type="verses" book="ap1">
_Patitaṃ pādamūle me,_  
_samuggatatanūruhaṃ;_  
_Jīmūtavaṇṇaṃ pīṇaṃsaṃ,_  
_pasannanayanānanaṃ._  

</Paragraph>

</Division>
<Division number="75" book="ap1" bookNo="20" page="435">

<Paragraph number="7026" type="verses" book="ap1">
_Parivārena mahatā,_  
_rājayuttaṃ mahāyasaṃ;_  
_Eso katāvino ṭhānaṃ,_  
_pattheti muditāsayo._  

</Paragraph>

</Division>
<Division number="76" book="ap1" bookNo="20" page="435">

<Paragraph number="7027" type="verses" book="ap1">
_Iminā paṇipātena,_  
_cāgena paṇidhīhi ca;_  
_Kappasatasahassāni,_  
_nupapajjati duggatiṃ._  

</Paragraph>

</Division>
<Division number="77" book="ap1" bookNo="20" page="435">

<Paragraph number="7028" type="verses" book="ap1">
_Devesu devasobhaggaṃ,_  
_manussesu mahantataṃ;_  
_Anubhotvāna sesena,_  
_nibbānaṃ pāpuṇissati._  

</Paragraph>

</Division>
<Division number="78" book="ap1" bookNo="20" page="435">

<Paragraph number="7029" type="verses" book="ap1">
_Satasahassito kappe,_  
_okkākakulasambhavo;_  
_Gotamo nāma gottena,_  
_satthā loke bhavissati._  

</Paragraph>

</Division>
<Division number="79" book="ap1" bookNo="20" page="435">

<Paragraph number="7030" type="verses" book="ap1">
_Tassa dhammesu dāyādo,_  
_oraso dhammanimmito;_  
_Kappino nāma nāmena,_  
_hessati satthu sāvako’._  

</Paragraph>

</Division>
<Division number="80" book="ap1" bookNo="20" page="435">

<Paragraph number="7031" type="verses" book="ap1">
_Tatohaṃ sukataṃ kāraṃ,_  
_katvāna jinasāsane;_  
_Jahitvā mānusaṃ dehaṃ,_  
_tusitaṃ agamāsahaṃ._  

</Paragraph>

</Division>
<Division number="81" book="ap1" bookNo="20" page="435">

<Paragraph number="7032" type="verses" book="ap1">
_Devamānusarajjāni,_  
_sataso anusāsiya;_  
_Bārāṇasiyamāsanne,_  
_jāto keniyajātiyaṃ._  

</Paragraph>

</Division>
<Division number="82" book="ap1" bookNo="20" page="435">

<Paragraph number="7033" type="verses" book="ap1">
_Sahassaparivārena,_  
_sapajāpatiko ahaṃ;_  
_Pañca paccekabuddhānaṃ,_  
_satāni samupaṭṭhahiṃ._  

</Paragraph>

</Division>
<Division number="83" book="ap1" bookNo="20" page="435">

<Paragraph number="7034" type="verses" book="ap1">
_Temāsaṃ bhojayitvāna,_  
_pacchādamha ticīvaraṃ;_  
_Tato cutā mayaṃ sabbe,_  
_ahumha tidasūpagā._  

</Paragraph>

</Division>
<Division number="84" book="ap1" bookNo="20" page="436">

<Paragraph number="7035" type="verses" book="ap1">
_Puno sabbe manussattaṃ,_  
_agamimha tato cutā;_  
_Kukkuṭamhi pure jātā,_  
_himavantassa passato._  

</Paragraph>

</Division>
<Division number="85" book="ap1" bookNo="20" page="436">

<Paragraph number="7036" type="verses" book="ap1">
_Kappino nāmahaṃ āsiṃ,_  
_rājaputto mahāyaso;_  
_Sesāmaccakule jātā,_  
_mameva parivārayuṃ._  

</Paragraph>

</Division>
<Division number="86" book="ap1" bookNo="20" page="436">

<Paragraph number="7037" type="verses" book="ap1">
_Mahārajjasukhaṃ patto,_  
_sabbakāmasamiddhimā;_  
_Vāṇijehi samakkhātaṃ,_  
_buddhuppādamahaṃ suṇiṃ._  

</Paragraph>

</Division>
<Division number="87" book="ap1" bookNo="20" page="436">

<Paragraph number="7038" type="verses" book="ap1">
_‘Buddho loke samuppanno,_  
_asamo ekapuggalo;_  
_So pakāseti saddhammaṃ,_  
_amataṃ sukhamuttamaṃ._  

</Paragraph>

</Division>
<Division number="88" book="ap1" bookNo="20" page="436">

<Paragraph number="7039" type="verses" book="ap1">
_Suyuttā tassa sissā ca,_  
_sumuttā ca anāsavā’;_  
_Sutvā nesaṃ suvacanaṃ,_  
_sakkaritvāna vāṇije._  

</Paragraph>

</Division>
<Division number="89" book="ap1" bookNo="20" page="436">

_Pahāya rajjaṃ sāmacco,_  
_nikkhamiṃ buddhamāmako;_  
_Nadiṃ disvā mahācandaṃ,_  
_pūritaṃ samatittikaṃ._  

</Paragraph>

</Division>
<Division number="90" book="ap1" bookNo="20" page="436">

_duttaraṃ sīghavāhiniṃ;_  
_Guṇaṃ saritvā buddhassa,_  
_sotthinā samatikkamiṃ._  

</Paragraph>

</Division>

_tiṇṇo lokantagū vidū;_  
_Etena saccavajjena,_  
_gamanaṃ me samijjhatu._  

</Paragraph>

</Division>

_Etena saccavajjena,_  
_gamanaṃ me samijjhatu._  

</Paragraph>

</Division>

_puññakkhetto anuttaro;_  
_Etena saccavajjena,_  
_gamanaṃ me samijjhatu’._  

</Paragraph>

</Division>

_maggā apagataṃ jalaṃ;_  
_Tato sukhena uttiṇṇo,_  
_nadītīre manorame._  

</Paragraph>

</Division>

_udentaṃva pabhaṅkaraṃ;_  
_Jalantaṃ hemaselaṃva,_  
_dīparukkhaṃva jotitaṃ._  

</Paragraph>

</Division>
<Division number="96" book="ap1" bookNo="20" page="436">

_sāvakehi purakkhataṃ;_  
_Vāsavaṃ viya vassantaṃ,_  
_desanājaladantaraṃ._  

</Paragraph>

</Division>

_ekamantamupāvisiṃ;_  
_Tato no āsayaṃ ñatvā,_  
_buddho dhammamadesayi._  

</Paragraph>

</Division>

_avocumha mayaṃ jinaṃ;_  
_‘Pabbājehi mahāvīra,_  
_nibbindāmha mayaṃ bhave’._  

</Paragraph>

</Division>

_dukkhantakaraṇāya vo;_  
_Caratha brahmacariyaṃ’,_  
_iccāha munisattamo._  

</Paragraph>

</Division>

_bhikkhuvesadharā mayaṃ;_  
_Ahumha upasampannā,_  
_sotāpannā ca sāsane._  

</Paragraph>

</Division>

_anusāsi vināyako;_  
_Anusiṭṭho jinenāhaṃ,_  
_arahattamapāpuṇiṃ._  

</Paragraph>

</Division>

_Mamānusāsanakarā,_  
_tepi āsuṃ anāsavā._  

</Paragraph>

_kappinoti mahājane._  

</Paragraph>
