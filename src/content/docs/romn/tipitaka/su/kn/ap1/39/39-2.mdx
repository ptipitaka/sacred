_“Nagare bandhumatiyā,_  
_āsiṃ ārāmiko tadā;_  
_Addasaṃ virajaṃ buddhaṃ,_  
_gacchantaṃ anilañjase._

<Division number="8" book="ap1" bookNo="20" page="341">

<Paragraph number="4186" type="verses" book="ap1">
_Labujassa phalaṃ gayha,_  
_buddhaseṭṭhassadāsahaṃ;_  
_Ākāse ṭhitako santo,_  
_paṭigaṇhi mahāyaso._  

</Paragraph>

</Division>
<Division number="9" book="ap1" bookNo="20" page="341">

<Paragraph number="4187" type="verses" book="ap1">
_Vittisañjananaṃ mayhaṃ,_  
_diṭṭhadhammasukhāvahaṃ;_  
_Phalaṃ buddhassa datvāna,_  
_vippasannena cetasā._  

</Paragraph>

</Division>
<Division number="10" book="ap1" bookNo="20" page="342">

<Paragraph number="4188" type="verses" book="ap1">
_Adhigacchiṃ tadā pītiṃ,_  
_vipulañca sukhuttamaṃ;_  
_Uppajjateva ratanaṃ,_  
_nibbattassa tahiṃ tahiṃ._  

</Paragraph>

</Division>

_yaṃ phalamadadiṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_phaladānassidaṃ phalaṃ._  

</Paragraph>

</Division>
