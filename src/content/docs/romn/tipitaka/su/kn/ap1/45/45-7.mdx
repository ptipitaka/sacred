_sīhāsanassidaṃ phalaṃ._  

</Paragraph>

<Division number="44" book="ap1">

<Paragraph number="5830" type="verses" book="ap1">

_lohitaṅgamayā bahū;_  
_Maṇimayā ca pallaṅkā,_  
_nibbattanti mamaṃ sadā._  

</Paragraph>

</Division>
<Division number="45" book="ap1">

<Paragraph number="5831" type="verses" book="ap1">

_jalajuttamanāmino;_  
_Ucce kule pajāyāmi,_  
_aho dhammasudhammatā._  

</Paragraph>

</Division>

<Paragraph number="5832" type="verses" book="ap1">

_sīhāsanamakāsahaṃ;_

_sīhāsanassidaṃ phalaṃ._  

</Paragraph>

</Division>
