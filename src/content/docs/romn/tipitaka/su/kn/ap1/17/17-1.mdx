<Division number="1" book="ap1">

<Paragraph number="2433" type="verses" book="ap1">
_“Padumo nāma nāmena,_  
_dvipadindo narāsabho;_  
_Pavanā abhinikkhamma,_  
_dhammaṃ deseti cakkhumā._  

</Paragraph>

</Division>
<Division number="2" book="ap1">

<Paragraph number="2434" type="verses" book="ap1">
_Yakkhānaṃ samayo āsi,_  
_avidūre mahesino;_  
_Yena kiccena sampattā,_  
_ajjhāpekkhiṃsu tāvade._  

</Paragraph>

</Division>
<Division number="3" book="ap1">

<Paragraph number="2435" type="verses" book="ap1">
_Buddhassa giramaññāya,_  
_amatassa ca desanaṃ;_  
_Pasannacitto sumano,_  
_apphoṭetvā upaṭṭhahiṃ._  

</Paragraph>

</Division>
<Division number="4" book="ap1">

<Paragraph number="2436" type="verses" book="ap1">
_Suciṇṇassa phalaṃ passa,_  
_upaṭṭhānassa satthuno;_  
_Tiṃsakappasahassesu,_  
_duggatiṃ nupapajjahaṃ._  

</Paragraph>

</Division>
<Division number="5" book="ap1">

<Paragraph number="2437" type="verses" book="ap1">
_Ūnatiṃse kappasate,_  
_samalaṅkatanāmako;_  
_Sattaratanasampanno,_  
_cakkavattī mahabbalo._  

</Paragraph>

</Division>
<Division number="6" book="ap1">

<Paragraph number="2438" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_vimokkhāpi ca aṭṭhime;_  
_Chaḷabhiññā sacchikatā,_  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="2439" book="ap1">
Itthaṃ sudaṃ āyasmā supāricariyo thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="2440" type="center" book="ap1">
Supāricariyattherassāpadānaṃ paṭhamaṃ.
</Paragraph>
