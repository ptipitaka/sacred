<Division number="7" book="ap1">

<Paragraph number="3166" type="verses" book="ap1">

_yaṃ telaṃ makkhayiṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_makkhanāya idaṃ phalaṃ._  

</Paragraph>

</Division>

_Sattaratanasampanno,_  
_cakkavattī mahabbalo._  

</Paragraph>

</Division>

</Paragraph>
