<Division number="13" book="ap1">

<Paragraph number="4112" type="verses" book="ap1">
_“Candabhāgānadītīre,_  
_ahosiṃ vānaro tadā;_  
_Addasaṃ virajaṃ buddhaṃ,_  
_nisinnaṃ pabbatantare._  

</Paragraph>

</Division>
<Division number="14" book="ap1">

<Paragraph number="4113" type="verses" book="ap1">
_Obhāsentaṃ disā sabbā,_  
_sālarājaṃva phullitaṃ;_  
_Lakkhaṇabyañjanūpetaṃ,_  
_disvānattamano ahaṃ._  

</Paragraph>

</Division>
<Division number="15" book="ap1">

<Paragraph number="4114" type="verses" book="ap1">
_Udaggacitto sumano,_  
_pītiyā haṭṭhamānaso;_  
_Tīṇi uppalapupphāni,_  
_matthake abhiropayiṃ._  

</Paragraph>

</Division>
<Division number="16" book="ap1">

<Paragraph number="4115" type="verses" book="ap1">
_Pūjayitvāna pupphāni,_  
_phussassāhaṃ mahesino;_  
_Sagāravo bhavitvāna,_  
_pakkāmiṃ uttarāmukho._  

</Paragraph>

</Division>
<Division number="17" book="ap1">

<Paragraph number="4116" type="verses" book="ap1">
_Gacchanto paṭikuṭiko,_  
_vippasannena cetasā;_  
_Selantare patitvāna,_  
_pāpuṇiṃ jīvitakkhayaṃ._  

</Paragraph>

</Division>
<Division number="18" book="ap1">

<Paragraph number="4117" type="verses" book="ap1">
_Tena kammena sukatena,_  
_cetanāpaṇidhīhi ca;_  
_Jahitvā purimaṃ jātiṃ,_  
_tāvatiṃsamagacchahaṃ._  

</Paragraph>

</Division>
<Division number="19" book="ap1">

<Paragraph number="4118" type="verses" book="ap1">
_Satānaṃ tīṇikkhattuñca,_  
_devarajjamakārayiṃ;_  
_Satānaṃ pañcakkhattuñca,_  
_cakkavattī ahosahaṃ._  

</Paragraph>

</Division>
<Division number="20" book="ap1">

<Paragraph number="4119" type="verses" book="ap1">
_Dvenavute ito kappe,_  
_yaṃ pupphamabhipūjayiṃ;_  
_Duggatiṃ nābhijānāmi,_  
_buddhapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="21" book="ap1">

<Paragraph number="4120" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="4121" book="ap1">
Itthaṃ sudaṃ āyasmā tīṇuppalamāliyo thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="4122" type="center" book="ap1">
Tīṇuppalamāliyattherassāpadānaṃ tatiyaṃ.
</Paragraph>
