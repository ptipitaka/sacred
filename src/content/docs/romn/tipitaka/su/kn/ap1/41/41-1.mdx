<Division number="1" book="ap1">

<Paragraph number="4847" type="verses" book="ap1">
_“Pabbhārakūṭaṃ nissāya,_  
_sobhito nāma tāpaso;_  
_Pavattaphalaṃ bhuñjitvā,_  
_vasati pabbatantare._  

</Paragraph>

</Division>
<Division number="2" book="ap1">

<Paragraph number="4848" type="verses" book="ap1">
_Aggiṃ dāruṃ āharitvā,_  
_ujjālesiṃ ahaṃ tadā;_  
_Uttamatthaṃ gavesanto,_  
_brahmalokūpapattiyā._  

</Paragraph>

</Division>
<Division number="3" book="ap1">

<Paragraph number="4849" type="verses" book="ap1">
_Padumuttaro lokavidū,_  
_āhutīnaṃ paṭiggaho;_  
_Mamuddharitukāmo so,_  
_āgacchi mama santike._  

</Paragraph>

</Division>
<Division number="4" book="ap1">

<Paragraph number="4850" type="verses" book="ap1">
_‘Kiṃ karosi mahāpuñña,_  
_dehi me aggidārukaṃ;_  
_Ahamaggiṃ paricare,_  
_tato me suddhi hohiti’._  

</Paragraph>

</Division>
<Division number="5" book="ap1">

<Paragraph number="4851" type="verses" book="ap1">
_‘Subhaddako tvaṃ manuje,_  
_devate tvaṃ pajānasi;_  
_Tuvaṃ aggiṃ paricara,_  
_handa te aggidārukaṃ’._  

</Paragraph>

</Division>
<Division number="6" book="ap1">

<Paragraph number="4852" type="verses" book="ap1">
_Tato kaṭṭhaṃ gahetvāna,_  
_aggiṃ ujjālayī jino;_  
_Na tattha kaṭṭhaṃ pajjhāyi,_  
_pāṭiheraṃ mahesino._  

</Paragraph>

</Division>
<Division number="7" book="ap1">

<Paragraph number="4853" type="verses" book="ap1">
_‘Na te aggi pajjalati,_  
_āhutī te na vijjati;_  
_Niratthakaṃ vataṃ tuyhaṃ,_  
_aggiṃ paricarassu me’._  

</Paragraph>

</Division>
<Division number="8" book="ap1">

<Paragraph number="4854" type="verses" book="ap1">
_‘Kīdiso so mahāvīra,_  
_aggi tava pavuccati;_  
_Mayhampi kathayassetaṃ,_  
_ubho paricarāmase’._  

</Paragraph>

</Division>
<Division number="9" book="ap1">

<Paragraph number="4855" type="verses" book="ap1">
_‘Hetudhammanirodhāya,_  
_kilesasamaṇāya ca;_  
_Issāmacchariyaṃ hitvā,_  
_tayo ete mamāhutī’._  

</Paragraph>

</Division>
<Division number="10" book="ap1">

<Paragraph number="4856" type="verses" book="ap1">
_‘Kīdiso tvaṃ mahāvīra,_  
_kathaṃ gottosi mārisa;_  
_Ācārapaṭipatti te,_  
_bāḷhaṃ kho mama ruccati’._  

</Paragraph>

</Division>
<Division number="11" book="ap1">

<Paragraph number="4857" type="verses" book="ap1">
_‘Khattiyamhi kule jāto,_  
_abhiññāpāramiṃ gato;_  
_Sabbāsavaparikkhīṇo,_  
_natthi dāni punabbhavo’._  

</Paragraph>

</Division>
<Division number="12" book="ap1">

<Paragraph number="4858" type="verses" book="ap1">
_‘Yadi buddhosi sabbaññū,_  
_pabhaṅkara tamonuda;_  
_Namassissāmi taṃ deva,_  
_dukkhassantakaro tuvaṃ’._  

</Paragraph>

</Division>
<Division number="13" book="ap1">

<Paragraph number="4859" type="verses" book="ap1">
_Pattharitvājinacammaṃ,_  
_nisīdanamadāsahaṃ;_  
_Nisīda nātha sabbaññu,_  
_upaṭṭhissāmahaṃ tuvaṃ._  

</Paragraph>

</Division>
<Division number="14" book="ap1">

<Paragraph number="4860" type="verses" book="ap1">
_Nisīdi bhagavā tattha,_  
_ajinamhi suvitthate;_  
_Nimantayitvā sambuddhaṃ,_  
_pabbataṃ agamāsahaṃ._  

</Paragraph>

</Division>
<Division number="15" book="ap1">

<Paragraph number="4861" type="verses" book="ap1">
_Khāribhārañca pūretvā,_  
_tindukaphalamāhariṃ;_  
_Madhunā yojayitvāna,_  
_phalaṃ buddhassadāsahaṃ._  

</Paragraph>

</Division>
<Division number="16" book="ap1">

<Paragraph number="4862" type="verses" book="ap1">
_Mama nijjhāyamānassa,_  
_paribhuñji tadā jino;_  
_Tattha cittaṃ pasādesiṃ,_  
_pekkhanto lokanāyakaṃ._  

</Paragraph>

</Division>
<Division number="17" book="ap1">

<Paragraph number="4863" type="verses" book="ap1">
_Padumuttaro lokavidū,_  
_āhutīnaṃ paṭiggaho;_  
_Mamassame nisīditvā,_  
_imā gāthā abhāsatha._  

</Paragraph>

</Division>
<Division number="18" book="ap1">

<Paragraph number="4864" type="verses" book="ap1">
_‘Yo maṃ phalena tappesi,_  
_pasanno sehi pāṇibhi;_  
_Tamahaṃ kittayissāmi,_  
_suṇātha mama bhāsato._  

</Paragraph>

</Division>
<Division number="19" book="ap1">

<Paragraph number="4865" type="verses" book="ap1">
_Pañcavīsatikkhattuṃ so,_  
_devarajjaṃ karissati;_  
_Sahassakkhattuṃ rājā ca,_  
_cakkavattī bhavissati._  

</Paragraph>

</Division>
<Division number="20" book="ap1">

<Paragraph number="4866" type="verses" book="ap1">
_Tassa saṅkappamaññāya,_  
_pubbakammasamaṅgino;_  
_Annaṃ pānañca vatthañca,_  
_sayanañca mahārahaṃ._  

</Paragraph>

</Division>
<Division number="21" book="ap1">

<Paragraph number="4867" type="verses" book="ap1">
_Puññakammena saṃyuttā,_  
_nibbattissanti tāvade;_  
_Sadā pamudito cāyaṃ,_  
_bhavissati anāmayo._  

</Paragraph>

</Division>
<Division number="22" book="ap1">

<Paragraph number="4868" type="verses" book="ap1">
_Upapajjati yaṃ yoniṃ,_  
_devattaṃ atha mānusaṃ;_  
_Sabbattha sukhito hutvā,_  
_manussattaṃ gamissati._  

</Paragraph>

</Division>
<Division number="23" book="ap1">

<Paragraph number="4869" type="verses" book="ap1">
_Ajjhāyako mantadharo,_  
_tiṇṇaṃ vedāna pāragū;_  
_Sambuddhaṃ upagantvāna,_  
_arahā so bhavissati’._  

</Paragraph>

</Division>
<Division number="24" book="ap1">

<Paragraph number="4870" type="verses" book="ap1">
_Yato sarāmi attānaṃ,_  
_yato pattosmi viññutaṃ;_  
_Bhoge me ūnatā natthi,_  
_phaladānassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="25" book="ap1">

<Paragraph number="4871" type="verses" book="ap1">
_Varadhammamanuppatto,_  
_rāgadose samūhaniṃ;_  
_Sabbāsavaparikkhīṇo,_  
_natthi dāni punabbhavo._  

</Paragraph>

</Division>
<Division number="26" book="ap1">

<Paragraph number="4872" type="verses" book="ap1">
_Kilesā jhāpitā mayhaṃ,_  
_bhavā sabbe samūhatā;_  
_Nāgova bandhanaṃ chetvā,_  
_viharāmi anāsavo._  

</Paragraph>

</Division>
<Division number="27" book="ap1">

<Paragraph number="4873" type="verses" book="ap1">
_Svāgataṃ vata me āsi,_  
_mama buddhassa santike;_  
_Tisso vijjā anuppattā,_  
_kataṃ buddhassa sāsanaṃ._  

</Paragraph>

</Division>
<Division number="28" book="ap1">

<Paragraph number="4874" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_vimokkhāpi ca aṭṭhime;_  
_Chaḷabhiññā sacchikatā,_  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="4875" book="ap1">
Itthaṃ sudaṃ āyasmā tissametteyyo thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="4876" type="center" book="ap1">
Tissametteyyattherassāpadānaṃ paṭhamaṃ.
</Paragraph>
