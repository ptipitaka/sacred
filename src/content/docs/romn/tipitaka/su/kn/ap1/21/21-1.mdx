<Division number="1" book="ap1">

<Paragraph number="2750" type="verses" book="ap1">
_“Kaṇikāraṃ pupphitaṃ disvā,_  
_ocinitvānahaṃ tadā;_  
_Tissassa abhiropesiṃ,_  
_oghatiṇṇassa tādino._  

</Paragraph>

</Division>
<Division number="2" book="ap1">

<Paragraph number="2751" type="verses" book="ap1">
_Dvenavute ito kappe,_  
_yaṃ pupphamabhiropayiṃ;_  
_Duggatiṃ nābhijānāmi,_  
_buddhapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="3" book="ap1">

<Paragraph number="2752" type="verses" book="ap1">
_Pañcattiṃse ito kappe,_  
_aruṇapāṇīti vissuto;_  
_Sattaratanasampanno,_  
_cakkavattī mahabbalo._  

</Paragraph>

</Division>
<Division number="4" book="ap1">

<Paragraph number="2753" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_vimokkhāpi ca aṭṭhime;_  
_Chaḷabhiññā sacchikatā,_  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="2754" book="ap1">
Itthaṃ sudaṃ āyasmā kaṇikārapupphiyo thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="2755" type="center" book="ap1">
Kaṇikārapupphiyattherassāpadānaṃ paṭhamaṃ.
</Paragraph>
