<Division number="1" book="ap1">

<Paragraph number="3223" type="verses" book="ap1">
_“Ekāsanaṃ ahamadaṃ,_  
_pasanno sehi pāṇibhi;_  
_Bibbohanañca pādāsiṃ,_  
_uttamatthassa pattiyā._  

</Paragraph>

</Division>
<Division number="2" book="ap1">

<Paragraph number="3224" type="verses" book="ap1">
_Ekanavutito kappe,_  
_bibbohanamadāsahaṃ;_  
_Duggatiṃ nābhijānāmi,_  
_bibbohanassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="3" book="ap1">

<Paragraph number="3225" type="verses" book="ap1">
_Ito tesaṭṭhime kappe,_  
_asamo nāma khattiyo;_  
_Sattaratanasampanno,_  
_cakkavattī mahabbalo._  

</Paragraph>

</Division>
<Division number="4" book="ap1">

<Paragraph number="3226" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_vimokkhāpi ca aṭṭhime;_  
_Chaḷabhiññā sacchikatā,_  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="3227" book="ap1">
Itthaṃ sudaṃ āyasmā suvaṇṇabibbohaniyo thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="3228" type="center" book="ap1">
Suvaṇṇabibbohaniyattherassāpadānaṃ paṭhamaṃ.
</Paragraph>
