<Division number="180" book="ap1">

<Paragraph number="1055" type="verses" book="ap1">
_“Suvaṇṇavaṇṇaṃ sambuddhaṃ,_  
_Gacchantaṃ antarāpaṇe;_  
_Vipassiṃ satthavāhaggaṃ,_  
_Naravaraṃ vināyakaṃ._  

</Paragraph>

</Division>
<Division number="181" book="ap1">

<Paragraph number="1056" type="verses" book="ap1">
_Adantadamanaṃ tādiṃ,_  
_mahāvādiṃ mahāmatiṃ;_  
_Disvā pasanno sumano,_  
_ekañjalimakāsahaṃ._  

</Paragraph>

</Division>
<Division number="182" book="ap1">

<Paragraph number="1057" type="verses" book="ap1">
_Ekanavutito kappe,_  
_yamañjaliṃ kariṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_añjalissa idaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="183" book="ap1">

<Paragraph number="1058" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_vimokkhāpi ca aṭṭhime;_  
_Chaḷabhiññā sacchikatā,_  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="1059" book="ap1">
Itthaṃ sudaṃ āyasmā ekañjaliko thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="1060" type="center" book="ap1">
Ekañjalikattherassāpadānaṃ navamaṃ.
</Paragraph>
