<Paragraph number="1266" type="verses" book="ap1">
_“Padumuttarasambuddhaṃ,_  
_mettacittaṃ mahāmuniṃ;_  
_Upeti janatā sabbā,_  
_sabbalokagganāyakaṃ._  

</Paragraph>

</Division>
<Division number="55" book="ap1" bookNo="20" page="106">

<Paragraph number="1267" type="verses" book="ap1">
_Sattukañca baddhakañca,_  
_āmisaṃ pānabhojanaṃ;_  
_Dadanti satthuno sabbe,_  
_puññakkhette anuttare._  

</Paragraph>

</Division>
<Division number="56" book="ap1" bookNo="20" page="106">

<Paragraph number="1268" type="verses" book="ap1">
_Ahampi dānaṃ dassāmi,_  
_devadevassa tādino;_  
_Buddhaseṭṭhaṃ nimantetvā,_  
_saṃghampi ca anuttaraṃ._  

</Paragraph>

</Division>
<Division number="57" book="ap1" bookNo="20" page="106">

<Paragraph number="1269" type="verses" book="ap1">
_Uyyojitā mayā cete,_  
_nimantesuṃ tathāgataṃ;_  
_Kevalaṃ bhikkhusaṃghañca,_  
_puññakkhettaṃ anuttaraṃ._  

</Paragraph>

</Division>
<Division number="58" book="ap1" bookNo="20" page="106">

<Paragraph number="1270" type="verses" book="ap1">
_Satasahassapallaṅkaṃ,_  
_sovaṇṇaṃ gonakatthataṃ;_  
_Tūlikāpaṭalikāya,_  
_khomakappāsikehi ca;_  
_Mahārahaṃ paññāpayiṃ,_  
_āsanaṃ buddhayuttakaṃ._  

</Paragraph>

</Division>
<Division number="59" book="ap1" bookNo="20" page="107">

<Paragraph number="1271" type="verses" book="ap1">
_Padumuttaro lokavidū,_  
_devadevo narāsabho;_  
_Bhikkhusaṃghaparibyūḷho,_  
_mama dvāramupāgami._  

</Paragraph>

</Division>
<Division number="60" book="ap1" bookNo="20" page="107">

<Paragraph number="1272" type="verses" book="ap1">
_Paccuggantvāna sambuddhaṃ,_  
_lokanāthaṃ yasassinaṃ;_  
_Pasannacitto sumano,_  
_abhināmayiṃ saṃgharaṃ._  

</Paragraph>

</Division>
<Division number="61" book="ap1" bookNo="20" page="107">

<Paragraph number="1273" type="verses" book="ap1">
_Bhikkhūnaṃ satasahassaṃ,_  
_buddhañca lokanāyakaṃ;_  
_Pasannacitto sumano,_  
_paramannena tappayiṃ._  

</Paragraph>

</Division>
<Division number="62" book="ap1" bookNo="20" page="107">

<Paragraph number="1274" type="verses" book="ap1">
_Padumuttaro lokavidū,_  
_āhutīnaṃ paṭiggaho;_  
_Bhikkhusaṃghe nisīditvā,_  
_imā gāthā abhāsatha._  

</Paragraph>

</Division>
<Division number="63" book="ap1" bookNo="20" page="107">

<Paragraph number="1275" type="verses" book="ap1">
_‘Yenidaṃ āsanaṃ dinnaṃ,_  
_sovaṇṇaṃ gonakatthataṃ;_  
_Tamahaṃ kittayissāmi,_  
_suṇātha mama bhāsato._  

</Paragraph>

</Division>
<Division number="64" book="ap1" bookNo="20" page="107">

<Paragraph number="1276" type="verses" book="ap1">
_Catusattatikkhattuṃ so,_  
_devarajjaṃ karissati;_  
_Anubhossati sampattiṃ,_  
_accharāhi purakkhato._  

</Paragraph>

</Division>
<Division number="65" book="ap1" bookNo="20" page="107">

<Paragraph number="1277" type="verses" book="ap1">
_Padesarajjaṃ sahassaṃ,_  
_vasudhaṃ āvasissati;_  
_Ekapaññāsakkhattuñca,_  
_cakkavattī bhavissati._  

</Paragraph>

</Division>
<Division number="66" book="ap1" bookNo="20" page="107">

<Paragraph number="1278" type="verses" book="ap1">
_Sabbāsu bhavayonīsu,_  
_uccākulī bhavissati;_  
_So ca pacchā pabbajitvā,_  
_sukkamūlena codito;_  
_Bhaddiyo nāma nāmena,_  
_hessati satthu sāvako’._  

</Paragraph>

</Division>
<Division number="67" book="ap1" bookNo="20" page="107">

<Paragraph number="1279" type="verses" book="ap1">
_Vivekamanuyuttomhi,_  
_pantasenanivāsahaṃ;_  
_Phalañcādhigataṃ sabbaṃ,_  
_cattaklesomhi ajjahaṃ._  

</Paragraph>

</Division>
<Division number="68" book="ap1" bookNo="20" page="107">

<Paragraph number="1280" type="verses" book="ap1">
_Mama sabbaṃ abhiññāya,_  
_sabbaññū lokanāyako;_  
_Bhikkhusaṃghe nisīditvā,_  
_etadagge ṭhapesi maṃ._  

</Paragraph>

</Division>
<Division number="69" book="ap1" bookNo="20" page="107">

<Paragraph number="1281" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="1282" book="ap1">
Itthaṃ sudaṃ āyasmā bhaddiyo kāḷigodhāya putto thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="1283" type="center" book="ap1">
Bhaddiyassa kāḷigodhāya puttattherassāpadānaṃ tatiyaṃ.
</Paragraph>
