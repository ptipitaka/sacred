<Division number="26" book="ap1">

<Paragraph number="2083" type="verses" book="ap1">
_“Suvaṇṇavaṇṇo sambuddho,_  
_Bāttiṃsavaralakkhaṇo;_  
_Pavanā abhinikkhanto,_  
_Bhikkhusaṃghapurakkhato._  

</Paragraph>

</Division>
<Division number="27" book="ap1">

<Paragraph number="2084" type="verses" book="ap1">
_Mahaccā kaṃsapātiyā,_  
_vaḍḍhetvā pāyasaṃ ahaṃ;_  
_Āhutiṃ yiṭṭhukāmo so,_  
_upanesiṃ baliṃ ahaṃ._  

</Paragraph>

</Division>
<Division number="28" book="ap1">

<Paragraph number="2085" type="verses" book="ap1">
_Bhagavā tamhi samaye,_  
_lokajeṭṭho narāsabho;_  
_Caṅkamaṃ susamārūḷho,_  
_ambare anilāyane._  

</Paragraph>

</Division>
<Division number="29" book="ap1">

<Paragraph number="2086" type="verses" book="ap1">
_Tañca acchariyaṃ disvā,_  
_abbhutaṃ lomahaṃsanaṃ;_  
_Ṭhapayitvā kaṃsapātiṃ,_  
_vipassiṃ abhivādayiṃ._  

</Paragraph>

</Division>
<Division number="30" book="ap1">

<Paragraph number="2087" type="verses" book="ap1">
_Tuvaṃ devosi sabbaññū,_  
_sadeve sahamānuse;_  
_Anukampaṃ upādāya,_  
_paṭigaṇha mahāmuni._  

</Paragraph>

</Division>
<Division number="31" book="ap1">

<Paragraph number="2088" type="verses" book="ap1">
_Paṭiggahesi bhagavā,_  
_sabbaññū lokanāyako;_  
_Mama saṅkappamaññāya,_  
_satthā loke mahāmuni._  

</Paragraph>

</Division>
<Division number="32" book="ap1">

<Paragraph number="2089" type="verses" book="ap1">
_Ekanavutito kappe,_  
_yaṃ dānamadadiṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_pāyasassa idaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="33" book="ap1">

<Paragraph number="2090" type="verses" book="ap1">
_Ekatālīsito kappe,_  
_buddho nāmāsi khattiyo;_  
_Sattaratanasampanno,_  
_cakkavattī mahabbalo._  

</Paragraph>

</Division>
<Division number="34" book="ap1">

<Paragraph number="2091" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="2092" book="ap1">
Itthaṃ sudaṃ āyasmā pāyasadāyako thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="2093" type="center" book="ap1">
Pāyasadāyakattherassāpadānaṃ tatiyaṃ.
</Paragraph>
