<Division number="24" book="ap1">

<Paragraph number="2464" type="verses" book="ap1">
_Kaṇikāraṃ ocinitvā,_  
_chattaṃ katvānahaṃ tadā;_  
_Pupphacchadanaṃ katvāna,_  
_buddhassa abhiropayiṃ._  

</Paragraph>

</Division>
<Division number="25" book="ap1">

<Paragraph number="2465" type="verses" book="ap1">

_yaṃ pupphamabhiropayiṃ;_  
_Duggatiṃ nābhijānāmi,_  
_buddhapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="26" book="ap1">

<Paragraph number="2466" type="verses" book="ap1">

_soṇṇābhā aṭṭha khattiyā;_  
_Sattaratanasampannā,_  
_cakkavattī mahabbalā._  

</Paragraph>

</Division>
