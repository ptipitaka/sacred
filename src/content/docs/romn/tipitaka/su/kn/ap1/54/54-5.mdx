<Paragraph number="7105" type="verses" book="ap1">
_“Ito satasahassamhi,_  
_kappe uppajji nāyako;_  
_Sabbalokahito vīro,_  
_padumuttaranāmako._  

</Paragraph>

</Division>
<Division number="151" book="ap1">

<Paragraph number="7106" type="verses" book="ap1">
_Tadāhaṃ brāhmaṇo hutvā,_  
_vissuto vedapāragū;_  
_Divāvihāraṃ vicaraṃ,_  
_addasaṃ lokanāyakaṃ._  

</Paragraph>

</Division>
<Division number="152" book="ap1">

<Paragraph number="7107" type="verses" book="ap1">
_Catusaccaṃ pakāsentaṃ,_  
_bodhayantaṃ sadevakaṃ;_  
_Vicittakathikānaggaṃ,_  
_vaṇṇayantaṃ mahājane._  

</Paragraph>

</Division>
<Division number="153" book="ap1">

<Paragraph number="7108" type="verses" book="ap1">
_Tadā muditacittohaṃ,_  
_nimantetvā tathāgataṃ;_  
_Nānārattehi vatthehi,_  
_alaṅkaritvāna maṇḍapaṃ._  

</Paragraph>

</Division>
<Division number="154" book="ap1">

<Paragraph number="7109" type="verses" book="ap1">
_Nānāratanapajjotaṃ,_  
_sasaṃghaṃ bhojayiṃ tahiṃ;_  
_Bhojayitvāna sattāhaṃ,_  
_nānaggarasabhojanaṃ._  

</Paragraph>

</Division>
<Division number="155" book="ap1">

<Paragraph number="7110" type="verses" book="ap1">
_Nānācittehi pupphehi,_  
_pūjayitvā sasāvakaṃ;_  
_Nipacca pādamūlamhi,_  
_taṃ ṭhānaṃ patthayiṃ ahaṃ._  

</Paragraph>

</Division>
<Division number="156" book="ap1">

<Paragraph number="7111" type="verses" book="ap1">
_Tadā munivaro āha,_  
_karuṇekarasāsayo;_  
_‘Passathetaṃ dijavaraṃ,_  
_padumānanalocanaṃ._  

</Paragraph>

</Division>
<Division number="157" book="ap1">

<Paragraph number="7112" type="verses" book="ap1">
_Pītipāmojjabahulaṃ,_  
_samuggatatanūruhaṃ;_  
_Hāsamhitavisālakkhaṃ,_  
_mama sāsanalālasaṃ._  

</Paragraph>

</Division>
<Division number="158" book="ap1">

<Paragraph number="7113" type="verses" book="ap1">
_Patitaṃ pādamūle me,_  
_ekāvatthasumānasaṃ;_  
_Esa pattheti taṃ ṭhānaṃ,_  
_vicittakathikattanaṃ._  

</Paragraph>

</Division>
<Division number="159" book="ap1">

<Paragraph number="7114" type="verses" book="ap1">
_Satasahassito kappe,_  
_okkākakulasambhavo;_  
_Gotamo nāma gottena,_  
_satthā loke bhavissati._  

</Paragraph>

</Division>
<Division number="160" book="ap1">

<Paragraph number="7115" type="verses" book="ap1">
_Tassa dhammesu dāyādo,_  
_oraso dhammanimmito;_  
_Kumārakassapo nāma,_  
_hessati satthu sāvako._  

</Paragraph>

</Division>
<Division number="161" book="ap1">

<Paragraph number="7116" type="verses" book="ap1">
_Vicittapupphadussānaṃ,_  
_ratanānañca vāhasā;_  
_Vicittakathikānaṃ so,_  
_aggataṃ pāpuṇissati’._  

</Paragraph>

</Division>
<Division number="162" book="ap1">

<Paragraph number="7117" type="verses" book="ap1">
_Tena kammena sukatena,_  
_cetanāpaṇidhīhi ca;_  
_Jahitvā mānusaṃ dehaṃ,_  
_tāvatiṃsamagacchahaṃ._  

</Paragraph>

</Division>
<Division number="163" book="ap1">

<Paragraph number="7118" type="verses" book="ap1">
_Paribbhamaṃ bhavābhave,_  
_raṅgamajjhe yathā naṭo;_  
_Sākhamigatrajo hutvā,_  
_migiyā kucchimokkamiṃ._  

</Paragraph>

</Division>
<Division number="164" book="ap1">

<Paragraph number="7119" type="verses" book="ap1">
_Tadā mayi kucchigate,_  
_vajjhavāro upaṭṭhito;_  
_Sākhena cattā me mātā,_  
_nigrodhaṃ saraṇaṃ gatā._  

</Paragraph>

</Division>
<Division number="165" book="ap1">

<Paragraph number="7120" type="verses" book="ap1">
_Tena sā migarājena,_  
_maraṇā parimocitā;_  
_Pariccajitvā sapāṇaṃ,_  
_mamevaṃ ovadī tadā._  

</Paragraph>

</Division>
<Division number="166" book="ap1">

<Paragraph number="7121" type="verses" book="ap1">
_‘Nigrodhameva seveyya,_  
_na sākhamupasaṃvase;_  
_Nigrodhasmiṃ mataṃ seyyo,_  
_yañce sākhamhi jīvitaṃ’._  

</Paragraph>

</Division>
<Division number="167" book="ap1">

<Paragraph number="7122" type="verses" book="ap1">
_Tenānusiṭṭhā migayūthapena,_  
_Ahañca mātā ca tathetare ca;_  
_Āgamma rammaṃ tusitādhivāsaṃ,_  
_Gatā pavāsaṃ sagharaṃ yatheva._  

</Paragraph>

</Division>
<Division number="168" book="ap1">

<Paragraph number="7123" type="verses" book="ap1">
_Puno kassapavīrassa,_  
_atthamentamhi sāsane;_  
_Āruyha selasikharaṃ,_  
_yuñjitvā jinasāsanaṃ._  

</Paragraph>

</Division>
<Division number="169" book="ap1">

<Paragraph number="7124" type="verses" book="ap1">
_Idānāhaṃ rājagahe,_  
_jāto seṭṭhikule ahuṃ;_  
_Āpannasattā me mātā,_  
_pabbaji anagāriyaṃ._  

</Paragraph>

</Division>
<Division number="170" book="ap1">

<Paragraph number="7125" type="verses" book="ap1">
_Sagabbhaṃ taṃ viditvāna,_  
_devadattamupānayuṃ;_  
_So avoca ‘vināsetha,_  
_pāpikaṃ bhikkhuniṃ imaṃ’._  

</Paragraph>

</Division>
<Division number="171" book="ap1">

<Paragraph number="7126" type="verses" book="ap1">
_Idānipi munindena,_  
_jinena anukampitā;_  
_Sukhinī ajanī mayhaṃ,_  
_mātā bhikkhunupassaye._  

</Paragraph>

</Division>
<Division number="172" book="ap1">

<Paragraph number="7127" type="verses" book="ap1">
_Taṃ viditvā mahīpālo,_  
_kosalo maṃ aposayi;_  
_Kumāraparihārena,_  
_nāmenāhañca kassapo._  

</Paragraph>

</Division>
<Division number="173" book="ap1">

<Paragraph number="7128" type="verses" book="ap1">
_Mahākassapamāgamma,_  
_ahaṃ kumārakassapo;_  
_Vammikasadisaṃ kāyaṃ,_  
_sutvā buddhena desitaṃ._  

</Paragraph>

</Division>
<Division number="174" book="ap1">

<Paragraph number="7129" type="verses" book="ap1">
_Tato cittaṃ vimucci me,_  
_anupādāya sabbaso;_  
_Pāyāsiṃ damayitvāhaṃ,_  
_etadaggamapāpuṇiṃ._  

</Paragraph>

</Division>

_ …pe… _

</Paragraph>

_ …pe… _

_ …pe… _

</Paragraph>

</Paragraph>
