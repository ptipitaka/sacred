<Division number="37" book="ap1">

<Paragraph number="1813" type="verses" book="ap1">
_“Vipine kānane disvā,_  
_appasadde nirākule;_  
_Siddhatthaṃ isinaṃ seṭṭhaṃ,_  
_āhutīnaṃ paṭiggahaṃ._  

</Paragraph>

</Division>
<Division number="38" book="ap1">

<Paragraph number="1814" type="verses" book="ap1">
_Nibbutattaṃ mahānāgaṃ,_  
_nisabhājāniyaṃ yathā;_  
_Osadhiṃva virocantaṃ,_  
_devasaṅghanamassitaṃ._  

</Paragraph>

</Division>
<Division number="39" book="ap1">

<Paragraph number="1815" type="verses" book="ap1">
_Vitti mamāhu tāvade,_  
_ñāṇaṃ uppajji tāvade;_  
_Vuṭṭhitassa samādhimhā,_  
_madhuṃ datvāna satthuno._  

</Paragraph>

</Division>
<Division number="40" book="ap1">

<Paragraph number="1816" type="verses" book="ap1">
_Vanditvā satthuno pāde,_  
_pakkāmiṃ pācināmukho;_  
_Catuttiṃsamhi kappamhi,_  
_rājā āsiṃ sudassano._  

</Paragraph>

</Division>
<Division number="41" book="ap1">

<Paragraph number="1817" type="verses" book="ap1">
_Madhu bhisehi savati,_  
_bhojanamhi ca tāvade;_  
_Madhuvassaṃ pavassittha,_  
_pubbakammassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="42" book="ap1">

<Paragraph number="1818" type="verses" book="ap1">
_Catunnavutito kappe,_  
_yaṃ madhuṃ adadiṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_madhudānassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="43" book="ap1">

<Paragraph number="1819" type="verses" book="ap1">
_Catuttiṃse ito kappe,_  
_cattāro te sudassanā;_  
_Sattaratanasampannā,_  
_cakkavattī mahabbalā._  

</Paragraph>

</Division>
<Division number="44" book="ap1">

<Paragraph number="1820" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="1821" book="ap1">
Itthaṃ sudaṃ āyasmā madhupiṇḍiko thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="1822" type="center" book="ap1">
Madhupiṇḍikattherassāpadānaṃ sattamaṃ.
</Paragraph>
