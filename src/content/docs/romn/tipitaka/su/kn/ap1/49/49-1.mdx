<Paragraph number="6185" type="verses" book="ap1">
_“Tisso nāmāsi bhagavā,_  
_sayambhū aggapuggalo;_  
_Paṃsukūlaṃ ṭhapetvāna,_  
_vihāraṃ pāvisī jino._  

</Paragraph>

</Division>
<Division number="2" book="ap1" bookNo="20" page="429">

<Paragraph number="6186" type="verses" book="ap1">
_Vinataṃ dhanumādāya,_  
_bhakkhatthāya cariṃ ahaṃ;_  
_Maṇḍalaggaṃ gahetvāna,_  
_kānanaṃ pāvisiṃ ahaṃ._  

</Paragraph>

</Division>
<Division number="3" book="ap1" bookNo="20" page="429">

<Paragraph number="6187" type="verses" book="ap1">
_Tatthaddasaṃ paṃsukūlaṃ,_  
_dumagge laggitaṃ tadā;_  
_Cāpaṃ tattheva nikkhippa,_  
_sire katvāna añjaliṃ._  

</Paragraph>

</Division>
<Division number="4" book="ap1" bookNo="20" page="429">

<Paragraph number="6188" type="verses" book="ap1">
_Pasannacitto sumano,_  
_vipulāya ca pītiyā;_  
_Buddhaseṭṭhaṃ saritvāna,_  
_paṃsukūlaṃ avandahaṃ._  

</Paragraph>

</Division>
<Division number="5" book="ap1" bookNo="20" page="429">

<Paragraph number="6189" type="verses" book="ap1">
_Dvenavute ito kappe,_  
_paṃsukūlamavandahaṃ;_  
_Duggatiṃ nābhijānāmi,_  
_vandanāya idaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="6" book="ap1" bookNo="20" page="429">

<Paragraph number="6190" type="verses" book="ap1">
_Kilesā jhāpitā mayhaṃ,_  
_bhavā sabbe samūhatā;_  
_Nāgova bandhanaṃ chetvā,_  
_viharāmi anāsavo._  

</Paragraph>

</Division>
<Division number="7" book="ap1" bookNo="20" page="429">

<Paragraph number="6191" type="verses" book="ap1">
_Svāgataṃ vata me āsi,_  
_mama buddhassa santike;_  
_Tisso vijjā anuppattā,_  
_kataṃ buddhassa sāsanaṃ._  

</Paragraph>

</Division>
<Division number="8" book="ap1" bookNo="20" page="429">

<Paragraph number="6192" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_vimokkhāpi ca aṭṭhime;_  
_Chaḷabhiññā sacchikatā,_  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="6193" book="ap1">
Itthaṃ sudaṃ āyasmā paṃsukūlasaññako thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="6194" type="center" book="ap1">
Paṃsukūlasaññakattherassāpadānaṃ paṭhamaṃ.
</Paragraph>
