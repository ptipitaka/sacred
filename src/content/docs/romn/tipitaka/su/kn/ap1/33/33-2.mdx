<Paragraph number="3594" type="verses" book="ap1">
_“Kakudhaṃ vilasantaṃva,_  
_nisabhājāniyaṃ yathā;_  
_Osadhiṃva virocantaṃ,_  
_obhāsantaṃ narāsabhaṃ._  

</Paragraph>

</Division>
<Division number="8" book="ap1">

<Paragraph number="3595" type="verses" book="ap1">
_Añjaliṃ paggahetvāna,_  
_avandiṃ satthuno ahaṃ;_  
_Satthāraṃ parivaṇṇesiṃ,_  
_sakakammena tosayiṃ._  

</Paragraph>

</Division>
<Division number="9" book="ap1">

<Paragraph number="3596" type="verses" book="ap1">
_Susuddhaṃ pulinaṃ gayha,_  
_gatamagge samokiriṃ;_  
_Ucchaṅgena gahetvāna,_  
_vipassissa mahesino._  

</Paragraph>

</Division>
<Division number="10" book="ap1">

<Paragraph number="3597" type="verses" book="ap1">
_Tato upaḍḍhapulinaṃ,_  
_vippasannena cetasā;_  
_Divāvihāre osiñciṃ,_  
_dvipadindassa tādino._  

</Paragraph>

</Division>
<Division number="11" book="ap1">

<Paragraph number="3598" type="verses" book="ap1">
_Ekanavutito kappe,_  
_pulinaṃ yamasiñcahaṃ;_  
_Duggatiṃ nābhijānāmi,_  
_pulinassa idaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="12" book="ap1">

<Paragraph number="3599" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="3600" book="ap1">
Itthaṃ sudaṃ āyasmā pulinapūjako thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="3601" type="center" book="ap1">
Pulinapūjakattherassāpadānaṃ dutiyaṃ.
</Paragraph>
