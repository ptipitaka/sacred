<Paragraph number="6568" type="verses" book="ap1">
_“Sumedho nāma sambuddho,_  
_bāttiṃsavaralakkhaṇo;_  
_Vivekakāmo sambuddho,_  
_himavantamupāgamiṃ._  

</Paragraph>

</Division>
<Division number="2" book="ap1" bookNo="20" page="429">

<Paragraph number="6569" type="verses" book="ap1">
_Ajjhogayha himavantaṃ,_  
_aggo kāruṇiko muni;_  
_Pallaṅkamābhujitvāna,_  
_nisīdi purisuttamo._  

</Paragraph>

</Division>
<Division number="3" book="ap1" bookNo="20" page="429">

<Paragraph number="6570" type="verses" book="ap1">
_Vijjādharo tadā āsiṃ,_  
_antalikkhacaro ahaṃ;_  
_Tisūlaṃ sukataṃ gayha,_  
_gacchāmi ambare tadā._  

</Paragraph>

</Division>
<Division number="4" book="ap1" bookNo="20" page="429">

<Paragraph number="6571" type="verses" book="ap1">
_Pabbatagge yathā aggi,_  
_puṇṇamāyeva candimā;_  
_Vane obhāsate buddho,_  
_sālarājāva phullito._  

</Paragraph>

</Division>
<Division number="5" book="ap1" bookNo="20" page="429">

<Paragraph number="6572" type="verses" book="ap1">
_Vanaggā nikkhamitvāna,_  
_buddharaṃsībhidhāvare;_  
_Naḷaggivaṇṇasaṅkāsā,_  
_disvā cittaṃ pasādayiṃ._  

</Paragraph>

</Division>
<Division number="6" book="ap1" bookNo="20" page="429">

<Paragraph number="6573" type="verses" book="ap1">
_Vicinaṃ addasaṃ pupphaṃ,_  
_kaṇikāraṃ devagandhikaṃ;_  
_Tīṇi pupphāni ādāya,_  
_buddhaseṭṭhamapūjayiṃ._  

</Paragraph>

</Division>
<Division number="7" book="ap1" bookNo="20" page="429">

<Paragraph number="6574" type="verses" book="ap1">
_Buddhassa ānubhāvena,_  
_tīṇi pupphāni me tadā;_  
_Uddhaṃvaṇṭā adhopattā,_  
_chāyaṃ kubbanti satthuno._  

</Paragraph>

</Division>
<Division number="8" book="ap1" bookNo="20" page="429">

<Paragraph number="6575" type="verses" book="ap1">
_Tena kammena sukatena,_  
_cetanāpaṇidhīhi ca;_  
_Jahitvā mānusaṃ dehaṃ,_  
_tāvatiṃsamagacchahaṃ._  

</Paragraph>

</Division>
<Division number="9" book="ap1" bookNo="20" page="429">

<Paragraph number="6576" type="verses" book="ap1">
_Tattha me sukataṃ byamhaṃ,_  
_kaṇikārīti ñāyati;_  
_Saṭṭhiyojanamubbedhaṃ,_  
_tiṃsayojanavitthataṃ._  

</Paragraph>

</Division>
<Division number="10" book="ap1" bookNo="20" page="429">

<Paragraph number="6577" type="verses" book="ap1">
_Sahassakaṇḍaṃ satabheṇḍu,_  
_dhajāluharitāmayaṃ;_  
_Satasahassaniyyūhā,_  
_byamhe pātubhaviṃsu me._  

</Paragraph>

</Division>
<Division number="11" book="ap1" bookNo="20" page="429">

<Paragraph number="6578" type="verses" book="ap1">
_Soṇṇamayā maṇimayā,_  
_lohitaṅgamayāpi ca;_  
_Phalikāpi ca pallaṅkā,_  
_yenicchakā yadicchakā._  

</Paragraph>

</Division>
<Division number="12" book="ap1" bookNo="20" page="430">

<Paragraph number="6579" type="verses" book="ap1">
_Mahārahañca sayanaṃ,_  
_tūlikāvikatīyutaṃ;_  
_Uddhalomikaekantaṃ,_  
_bibbohanasamāyutaṃ._  

</Paragraph>

</Division>
<Division number="13" book="ap1" bookNo="20" page="430">

<Paragraph number="6580" type="verses" book="ap1">
_Bhavanā nikkhamitvāna,_  
_caranto devacārikaṃ;_  
_Yadā icchāmi gamanaṃ,_  
_devasaṅghapurakkhato._  

</Paragraph>

</Division>
<Division number="14" book="ap1" bookNo="20" page="430">

<Paragraph number="6581" type="verses" book="ap1">
_Pupphassa heṭṭhā tiṭṭhāmi,_  
_uparicchadanaṃ mama;_  
_Samantā yojanasataṃ,_  
_kaṇikārehi chāditaṃ._  

</Paragraph>

</Division>
<Division number="15" book="ap1" bookNo="20" page="430">

<Paragraph number="6582" type="verses" book="ap1">
_Saṭṭhituriyasahassāni,_  
_sāyaṃ pātamupaṭṭhahuṃ;_  
_Parivārenti maṃ niccaṃ,_  
_rattindivamatanditā._  

</Paragraph>

</Division>
<Division number="16" book="ap1" bookNo="20" page="430">

<Paragraph number="6583" type="verses" book="ap1">
_Tattha naccehi gītehi,_  
_tāḷehi vāditehi ca;_  
_Ramāmi khiḍḍāratiyā,_  
_modāmi kāmakāmihaṃ._  

</Paragraph>

</Division>
<Division number="17" book="ap1" bookNo="20" page="430">

<Paragraph number="6584" type="verses" book="ap1">
_Tattha bhutvā pivitvā ca,_  
_modāmi tidase tadā;_  
_Nārīgaṇehi sahito,_  
_modāmi byamhamuttame._  

</Paragraph>

</Division>
<Division number="18" book="ap1" bookNo="20" page="430">

<Paragraph number="6585" type="verses" book="ap1">
_Satānaṃ pañcakkhattuñca,_  
_devarajjamakārayiṃ;_  
_Satānaṃ tīṇikkhattuñca,_  
_cakkavattī ahosahaṃ;_  
_Padesarajjaṃ vipulaṃ,_  
_gaṇanāto asaṅkhiyaṃ._  

</Paragraph>

</Division>
<Division number="19" book="ap1" bookNo="20" page="430">

<Paragraph number="6586" type="verses" book="ap1">
_Bhavābhave saṃsaranto,_  
_mahābhogaṃ labhāmahaṃ;_  
_Bhoge me ūnatā natthi,_  
_buddhapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="20" book="ap1" bookNo="20" page="430">

<Paragraph number="6587" type="verses" book="ap1">
_Duve bhave saṃsarāmi,_  
_devatte atha mānuse;_  
_Aññaṃ gatiṃ na jānāmi,_  
_buddhapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="21" book="ap1" bookNo="20" page="430">

<Paragraph number="6588" type="verses" book="ap1">
_Duve kule pajāyāmi,_  
_khattiye cāpi brāhmaṇe;_  
_Nīce kule na jāyāmi,_  
_buddhapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="22" book="ap1" bookNo="20" page="430">

<Paragraph number="6589" type="verses" book="ap1">
_Hatthiyānaṃ assayānaṃ,_  
_sivikaṃ sandamānikaṃ;_  
_Labhāmi sabbamevetaṃ,_  
_buddhapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="23" book="ap1" bookNo="20" page="430">

<Paragraph number="6590" type="verses" book="ap1">
_Dāsīgaṇaṃ dāsagaṇaṃ,_  
_nāriyo samalaṅkatā;_  
_Labhāmi sabbamevetaṃ,_  
_buddhapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="24" book="ap1" bookNo="20" page="431">

<Paragraph number="6591" type="verses" book="ap1">
_Koseyyakambaliyāni,_  
_khomakappāsikāni ca;_  
_Labhāmi sabbamevetaṃ,_  
_buddhapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="25" book="ap1" bookNo="20" page="431">

<Paragraph number="6592" type="verses" book="ap1">
_Navavatthaṃ navaphalaṃ,_  
_navaggarasabhojanaṃ;_  
_Labhāmi sabbamevetaṃ,_  
_buddhapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="26" book="ap1" bookNo="20" page="431">

<Paragraph number="6593" type="verses" book="ap1">
_Imaṃ khāda imaṃ bhuñja,_  
_imamhi sayane saya;_  
_Labhāmi sabbamevetaṃ,_  
_buddhapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="27" book="ap1" bookNo="20" page="431">

<Paragraph number="6594" type="verses" book="ap1">
_Sabbattha pūjito homi,_  
_yaso abbhuggato mama;_  
_Mahāpakkho sadā homi,_  
_abhejjapariso sadā;_  
_Ñātīnaṃ uttamo homi,_  
_buddhapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="28" book="ap1" bookNo="20" page="431">

<Paragraph number="6595" type="verses" book="ap1">
_Sītaṃ uṇhaṃ na jānāmi,_  
_pariḷāho na vijjati;_  
_Atho cetasikaṃ dukkhaṃ,_  
_hadaye me na vijjati._  

</Paragraph>

</Division>
<Division number="29" book="ap1" bookNo="20" page="431">

<Paragraph number="6596" type="verses" book="ap1">
_Suvaṇṇavaṇṇo hutvāna,_  
_saṃsarāmi bhavābhave;_  
_Vevaṇṇiyaṃ na jānāmi,_  
_buddhapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="30" book="ap1" bookNo="20" page="431">

<Paragraph number="6597" type="verses" book="ap1">
_Devalokā cavitvāna,_  
_sukkamūlena codito;_  
_Sāvatthiyaṃ pure jāto,_  
_mahāsāle suaḍḍhake._  

</Paragraph>

</Division>
<Division number="31" book="ap1" bookNo="20" page="431">

<Paragraph number="6598" type="verses" book="ap1">
_Pañca kāmaguṇe hitvā,_  
_pabbajiṃ anagāriyaṃ;_  
_Jātiyā sattavassohaṃ,_  
_arahattamapāpuṇiṃ._  

</Paragraph>

</Division>
<Division number="32" book="ap1" bookNo="20" page="431">

<Paragraph number="6599" type="verses" book="ap1">
_Upasampādayī buddho,_  
_guṇamaññāya cakkhumā;_  
_Taruṇo pūjanīyohaṃ,_  
_buddhapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="33" book="ap1" bookNo="20" page="431">

<Paragraph number="6600" type="verses" book="ap1">
_Dibbacakkhu visuddhaṃ me,_  
_samādhikusalo ahaṃ;_  
_Abhiññāpāramippatto,_  
_buddhapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="34" book="ap1" bookNo="20" page="432">

<Paragraph number="6601" type="verses" book="ap1">
_Paṭisambhidā anuppatto,_  
_iddhipādesu kovido;_  
_Dhammesu pāramippatto,_  
_buddhapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="35" book="ap1" bookNo="20" page="432">

<Paragraph number="6602" type="verses" book="ap1">
_Tiṃsakappasahassamhi,_  
_yaṃ buddhamabhipūjayiṃ;_  
_Duggatiṃ nābhijānāmi,_  
_buddhapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="36" book="ap1" bookNo="20" page="432">

<Paragraph number="6603" type="verses" book="ap1">
_Kilesā jhāpitā mayhaṃ,_  
_bhavā sabbe samūhatā;_  
_Nāgova bandhanaṃ chetvā,_  
_viharāmi anāsavo._  

</Paragraph>

</Division>
<Division number="37" book="ap1" bookNo="20" page="432">

<Paragraph number="6604" type="verses" book="ap1">
_Svāgataṃ vata me āsi,_  
_mama buddhassa santike;_  
_Tisso vijjā anuppattā,_  
_kataṃ buddhassa sāsanaṃ._  

</Paragraph>

</Division>
<Division number="38" book="ap1" bookNo="20" page="432">

<Paragraph number="6605" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_vimokkhāpi ca aṭṭhime;_  
_Chaḷabhiññā sacchikatā,_  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="6606" book="ap1">
Itthaṃ sudaṃ āyasmā tikaṇikārapupphiyo thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="6607" type="center" book="ap1">
Tikaṇikārapupphiyattherassāpadānaṃ paṭhamaṃ.
</Paragraph>
