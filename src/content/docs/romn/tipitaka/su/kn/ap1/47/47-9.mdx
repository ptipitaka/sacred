<Paragraph number="6030" type="verses" book="ap1">
_“Kakudhaṃ vilasantaṃva,_  
_devadevaṃ narāsabhaṃ;_  
_Rathiyaṃ paṭipajjantaṃ,_  
_ko disvā na pasīdati._  

</Paragraph>

</Division>
<Division number="46" book="ap1" bookNo="20" page="433">

<Paragraph number="6031" type="verses" book="ap1">
_Tamandhakāraṃ nāsetvā,_  
_santāretvā bahuṃ janaṃ;_  
_Ñāṇālokena jotantaṃ,_  
_ko disvā na pasīdati._  

</Paragraph>

</Division>
<Division number="47" book="ap1" bookNo="20" page="433">

<Paragraph number="6032" type="verses" book="ap1">
_Vasīsatasahassehi,_  
_nīyantaṃ lokanāyakaṃ;_  
_Uddharantaṃ bahū satte,_  
_ko disvā na pasīdati._  

</Paragraph>

</Division>
<Division number="48" book="ap1" bookNo="20" page="433">

<Paragraph number="6033" type="verses" book="ap1">
_Āhanantaṃ dhammabheriṃ,_  
_maddantaṃ titthiye gaṇe;_  
_Sīhanādaṃ vinadantaṃ,_  
_ko disvā na pasīdati._  

</Paragraph>

</Division>
<Division number="49" book="ap1" bookNo="20" page="433">

<Paragraph number="6034" type="verses" book="ap1">
_Yāvatā brahmalokato,_  
_āgantvāna sabrahmakā;_  
_Pucchanti nipuṇe pañhe,_  
_ko disvā na pasīdati._  

</Paragraph>

</Division>
<Division number="50" book="ap1" bookNo="20" page="433">

<Paragraph number="6035" type="verses" book="ap1">
_Yassañjaliṃ karitvāna,_  
_āyācanti sadevakā;_  
_Tena puññaṃ anubhonti,_  
_ko disvā na pasīdati._  

</Paragraph>

</Division>
<Division number="51" book="ap1" bookNo="20" page="433">

<Paragraph number="6036" type="verses" book="ap1">
_Sabbe janā samāgantvā,_  
_sampavārenti cakkhumaṃ;_  
_Na vikampati ajjhiṭṭho,_  
_ko disvā na pasīdati._  

</Paragraph>

</Division>
<Division number="52" book="ap1" bookNo="20" page="433">

<Paragraph number="6037" type="verses" book="ap1">
_Nagaraṃ pavisato yassa,_  
_ravanti bheriyo bahū;_  
_Vinadanti gajā mattā,_  
_ko disvā na pasīdati._  

</Paragraph>

</Division>
<Division number="53" book="ap1" bookNo="20" page="433">

<Paragraph number="6038" type="verses" book="ap1">
_Vīthiyā gacchato yassa,_  
_sabbābhā jotate sadā;_  
_Abbhunnatā samā honti,_  
_ko disvā na pasīdati._  

</Paragraph>

</Division>
<Division number="54" book="ap1" bookNo="20" page="433">

<Paragraph number="6039" type="verses" book="ap1">
_Byāharantassa buddhassa,_  
_cakkavāḷampi suyyati;_  
_Sabbe satte viññāpeti,_  
_ko disvā na pasīdati._  

</Paragraph>

</Division>
<Division number="55" book="ap1" bookNo="20" page="433">

<Paragraph number="6040" type="verses" book="ap1">
_Satasahassito kappe,_  
_yaṃ buddhamabhikittayiṃ;_  
_Duggatiṃ nābhijānāmi,_  
_kittanāya idaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="56" book="ap1" bookNo="20" page="433">

<Paragraph number="6041" type="verses" book="ap1">
_Kilesā jhāpitā mayhaṃ,_  
_ …pe… _  
_viharāmi anāsavo._  

</Paragraph>

</Division>
<Division number="57" book="ap1" bookNo="20" page="433">

<Paragraph number="6042" type="verses" book="ap1">
_Svāgataṃ vata me āsi,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ._  

</Paragraph>

</Division>
<Division number="58" book="ap1" bookNo="20" page="433">

<Paragraph number="6043" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="6044" book="ap1">
Itthaṃ sudaṃ āyasmā sayampaṭibhāniyo thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="6045" type="center" book="ap1">
Sayampaṭibhāniyattherassāpadānaṃ navamaṃ.
</Paragraph>
