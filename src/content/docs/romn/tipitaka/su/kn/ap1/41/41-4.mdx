<Division number="72" book="ap1">

<Paragraph number="4924" type="verses" book="ap1">
_“Gaṅgā bhāgīrathī nāma,_  
_himavantā pabhāvitā;_  
_Haṃsavatiyā dvārena,_  
_anusandati tāvade._  



</Paragraph>

</Division>
<Division number="73" book="ap1">

<Paragraph number="4925" type="verses" book="ap1">
_Sobhito nāma ārāmo,_  
_gaṅgākūle sumāpito;_  
_Tattha padumuttaro buddho,_  
_vasate lokanāyako._  



</Paragraph>

</Division>
<Division number="74" book="ap1">

<Paragraph number="4926" type="verses" book="ap1">
_Tidasehi yathā indo,_  
_manujehi purakkhato;_  
_Nisīdi tattha bhagavā,_  
_asambhītova kesarī._  



</Paragraph>

</Division>
<Division number="75" book="ap1">

<Paragraph number="4927" type="verses" book="ap1">
_Nagare haṃsavatiyā,_  
_vasāmi brāhmaṇo ahaṃ;_  
_Chaḷaṅgo nāma nāmena,_  
_evaṃnāmo mahāmuni._  



</Paragraph>

</Division>
<Division number="76" book="ap1">

<Paragraph number="4928" type="verses" book="ap1">
_Aṭṭhārasa sissasatā,_  
_parivārenti maṃ tadā;_  
_Tehi sissehi samito,_  
_gaṅgātīraṃ upāgamiṃ._  



</Paragraph>

</Division>
<Division number="77" book="ap1">

<Paragraph number="4929" type="verses" book="ap1">
_Tatthaddasāsiṃ samaṇe,_  
_nikkuhe dhotapāpake;_  
_Bhāgīrathiṃ tarantehaṃ,_  
_evaṃ cintesi tāvade._  



</Paragraph>

</Division>
<Division number="78" book="ap1">

<Paragraph number="4930" type="verses" book="ap1">
_‘Sāyaṃ pātaṃ tarantāme,_  
_buddhaputtā mahāyasā;_  
_Vihesayanti attānaṃ,_  
_tesaṃ attā vihaññati._  



</Paragraph>

</Division>
<Division number="79" book="ap1">

<Paragraph number="4931" type="verses" book="ap1">
_Sadevakassa lokassa,_  
_buddho aggo pavuccati;_  
_Natthi me dakkhiṇe kāraṃ,_  
_gatimaggavisodhanaṃ._  



</Paragraph>

</Division>
<Division number="80" book="ap1">

<Paragraph number="4932" type="verses" book="ap1">
_Yannūna buddhaseṭṭhassa,_  
_setuṃ gaṅgāya kāraye;_  
_Kārāpetvā imaṃ kammaṃ,_  
_santarāmi imaṃ bhavaṃ’._  



</Paragraph>

</Division>
<Division number="81" book="ap1">

<Paragraph number="4933" type="verses" book="ap1">
_Satasahassaṃ datvāna,_  
_setuṃ kārāpayiṃ ahaṃ;_  
_Saddahanto kataṃ kāraṃ,_  
_vipulaṃ me bhavissati._  



</Paragraph>

</Division>
<Division number="82" book="ap1">

<Paragraph number="4934" type="verses" book="ap1">
_Kārāpetvāna taṃ setuṃ,_  
_upesiṃ lokanāyakaṃ;_  
_Sirasi añjaliṃ katvā,_  
_imaṃ vacanamabraviṃ._  



</Paragraph>

</Division>
<Division number="83" book="ap1">

<Paragraph number="4935" type="verses" book="ap1">
_‘Satasahassassa vayaṃ,_  
_datvā kārāpito mayā;_  
_Tavatthāya mahāsetu,_  
_paṭiggaṇha mahāmune’._  



</Paragraph>

</Division>
<Division number="84" book="ap1">

<Paragraph number="4936" type="verses" book="ap1">
_Padumuttaro lokavidū,_  
_āhutīnaṃ paṭiggaho;_  
_Bhikkhusaṃghe nisīditvā,_  
_imā gāthā abhāsatha._  



</Paragraph>

</Division>
<Division number="85" book="ap1">

<Paragraph number="4937" type="verses" book="ap1">
_‘Yo me setuṃ akāresi,_  
_pasanno sehi pāṇibhi;_  
_Tamahaṃ kittayissāmi,_  
_suṇātha mama bhāsato._  


</Paragraph>
</Division>
