<Paragraph number="5737" type="verses" book="ap1">
_“Harītakaṃ āmalakaṃ,_  
_ambajambuvibhītakaṃ;_  
_Kolaṃ bhallātakaṃ billaṃ,_  
_sayameva harāmahaṃ._  

</Paragraph>

</Division>
<Division number="61" book="ap1">

<Paragraph number="5738" type="verses" book="ap1">
_Disvāna pabbhāragataṃ,_  
_jhāyiṃ jhānarataṃ muniṃ;_  
_Ābādhena āpīḷentaṃ,_  
_adutīyaṃ mahāmuniṃ._  

</Paragraph>

</Division>
<Division number="62" book="ap1">

<Paragraph number="5739" type="verses" book="ap1">
_Harītakaṃ gahetvāna,_  
_sayambhussa adāsahaṃ;_  
_Khādamattamhi bhesajje,_  
_byādhi passambhi tāvade._  

</Paragraph>

</Division>
<Division number="63" book="ap1">

<Paragraph number="5740" type="verses" book="ap1">
_Pahīnadaratho buddho,_  
_anumodamakāsi me;_  
_‘Bhesajjadāneniminā,_  
_byādhivūpasamena ca._  

</Paragraph>

</Division>
<Division number="64" book="ap1">

<Paragraph number="5741" type="verses" book="ap1">
_Devabhūto manusso vā,_  
_jāto vā aññajātiyā;_  
_Sabbattha sukhito hotu,_  
_mā ca te byādhimāgamā’._  

</Paragraph>

</Division>
<Division number="65" book="ap1">

<Paragraph number="5742" type="verses" book="ap1">
_Idaṃ vatvāna sambuddho,_  
_sayambhū aparājito;_  
_Nabhaṃ abbhuggamī dhīro,_  
_haṃsarājāva ambare._  

</Paragraph>

</Division>
<Division number="66" book="ap1">

<Paragraph number="5743" type="verses" book="ap1">
_Yato harītakaṃ dinnaṃ,_  
_sayambhussa mahesino;_  
_Imaṃ jātiṃ upādāya,_  
_byādhi me nupapajjatha._  

</Paragraph>

</Division>
<Division number="67" book="ap1">

<Paragraph number="5744" type="verses" book="ap1">
_Ayaṃ pacchimako mayhaṃ,_  
_carimo vattate bhavo;_  
_Tisso vijjā sacchikatā,_  
_kataṃ buddhassa sāsanaṃ._  

</Paragraph>

</Division>
<Division number="68" book="ap1">

<Paragraph number="5745" type="verses" book="ap1">
_Catunnavutito kappe,_  
_bhesajjamadadiṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_bhesajjassa idaṃ phalaṃ._  

</Paragraph>

</Division>

_ …pe… _  
_viharāmi anāsavo._  

</Paragraph>

</Division>

_ …pe… _  
_kataṃ buddhassa sāsanaṃ._  

</Paragraph>

</Division>

_ …pe… _
