<Paragraph number="2626" type="verses" book="ap1">
_“Yadā devo devakāyā,_  
_cavate āyusaṅkhayā;_  
_Tayo saddā niccharanti,_  
_devānaṃ anumodataṃ._  

</Paragraph>

</Division>
<Division number="35" book="ap1" bookNo="20" page="216">

<Paragraph number="2627" type="verses" book="ap1">
_‘Ito bho sugatiṃ gaccha,_  
_manussānaṃ sahabyataṃ;_  
_Manussabhūto saddhamme,_  
_labha saddhaṃ anuttaraṃ._  

</Paragraph>

</Division>
<Division number="36" book="ap1" bookNo="20" page="216">

<Paragraph number="2628" type="verses" book="ap1">
_Sā te saddhā niviṭṭhāssa,_  
_mūlajātā patiṭṭhitā;_  
_Yāvajīvaṃ asaṃhīrā,_  
_saddhamme suppavedite._  

</Paragraph>

</Division>
<Division number="37" book="ap1" bookNo="20" page="217">

<Paragraph number="2629" type="verses" book="ap1">
_Kāyena kusalaṃ katvā,_  
_vācāya kusalaṃ bahuṃ;_  
_Manasā kusalaṃ katvā,_  
_abyāpajjaṃ nirūpadhiṃ._  

</Paragraph>

</Division>
<Division number="38" book="ap1" bookNo="20" page="217">

<Paragraph number="2630" type="verses" book="ap1">
_Tato opadhikaṃ puññaṃ,_  
_katvā dānena taṃ bahuṃ;_  
_Aññepi macce saddhamme,_  
_brahmacariye nivesaya’._  

</Paragraph>

</Division>
<Division number="39" book="ap1" bookNo="20" page="217">

<Paragraph number="2631" type="verses" book="ap1">
_Imāya anukampāya,_  
_devā devaṃ yadā vidū;_  
_Cavantaṃ anumodanti,_  
_ehi deva punappunaṃ._  

</Paragraph>

</Division>
<Division number="40" book="ap1" bookNo="20" page="217">

<Paragraph number="2632" type="verses" book="ap1">
_Saṃvego me tadā āsi,_  
_devasaṅghe samāgate;_  
_Kaṃ su nāma ahaṃ yoniṃ,_  
_gamissāmi ito cuto._  

</Paragraph>

</Division>
<Division number="41" book="ap1" bookNo="20" page="217">

<Paragraph number="2633" type="verses" book="ap1">
_Mama saṃvegamaññāya,_  
_samaṇo bhāvitindriyo;_  
_Mamuddharitukāmo so,_  
_āgacchi mama santikaṃ._  

</Paragraph>

</Division>
<Division number="42" book="ap1" bookNo="20" page="217">

<Paragraph number="2634" type="verses" book="ap1">
_Sumano nāma nāmena,_  
_padumuttarasāvako;_  
_Atthadhammānusāsitvā,_  
_saṃvejesi mamaṃ tadā._  

</Paragraph>

</Division>
<Division number="43" book="ap1" bookNo="20" page="217">

<Paragraph number="2635" type="verses" book="ap1">
_Tassāhaṃ vacanaṃ sutvā,_  
_buddhe cittaṃ pasādayiṃ;_  
_Taṃ dhīraṃ abhivādetvā,_  
_tattha kālaṅkato ahaṃ._  

</Paragraph>

</Division>
<Division number="44" book="ap1" bookNo="20" page="217">

<Paragraph number="2636" type="verses" book="ap1">
_Upapajjiṃ sa tattheva,_  
_sukkamūlena codito;_  
_Kappānaṃ satasahassaṃ,_  
_duggatiṃ nupapajjahaṃ._  

</Paragraph>

</Division>
<Division number="45" book="ap1" bookNo="20" page="217">

<Paragraph number="2637" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="2638" book="ap1">
Itthaṃ sudaṃ āyasmā ekacintiko thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="2639" type="center" book="ap1">
Ekacintikattherassāpadānaṃ sattamaṃ.
</Paragraph>
