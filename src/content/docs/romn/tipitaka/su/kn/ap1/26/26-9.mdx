<Paragraph number="3142" type="verses" book="ap1">
_“Vipinacārī sambuddho,_  
_rukkhamūle vasī tadā;_  
_Paṇṇasālaṃ karitvāna,_  
_adāsiṃ aparājite._  

</Paragraph>

<Division number="34" book="ap1">

<Paragraph number="3143" type="verses" book="ap1">
_Ekanavutito kappe,_  
_yaṃ paṇṇakuṭikaṃ adaṃ;_  
_Duggatiṃ nābhijānāmi,_  
_kuṭidānassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="35" book="ap1">

<Paragraph number="3144" type="verses" book="ap1">
_Aṭṭhavīse ito kappe,_  
_soḷasāsiṃsu rājāno;_  
_Sabbattha abhivassīti,_  
_vuccare cakkavattino._  

</Paragraph>

</Division>
<Division number="36" book="ap1">

<Paragraph number="3145" type="verses" book="ap1">

_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="3146" book="ap1">

</Paragraph>
</Division>

</Paragraph>
