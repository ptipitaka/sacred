<Division number="90" book="ap1">

<Paragraph number="3818" type="verses" book="ap1">
_“Nagare haṃsavatiyā,_  
_ahosiṃ brāhmaṇo tadā;_  
_Ajjhāyako mantadharo,_  
_tiṇṇaṃ vedāna pāragū._  

</Paragraph>

</Division>
<Division number="91" book="ap1">

<Paragraph number="3819" type="verses" book="ap1">
_Purakkhato sasissehi,_  
_jātimā ca susikkhito;_  
_Toyābhisecanatthāya,_  
_nagarā nikkhamiṃ tadā._  

</Paragraph>

</Division>
<Division number="92" book="ap1">

<Paragraph number="3820" type="verses" book="ap1">
_Padumuttaro nāma jino,_  
_sabbadhammāna pāragū;_  
_Khīṇāsavasahassehi,_  
_pāvisī nagaraṃ jino._  

</Paragraph>

</Division>
<Division number="93" book="ap1">

<Paragraph number="3821" type="verses" book="ap1">
_Sucārurūpaṃ disvāna,_  
_āneñjakāritaṃ viya;_  
_Parivutaṃ arahantehi,_  
_disvā cittaṃ pasādayiṃ._  

</Paragraph>

</Division>
<Division number="94" book="ap1">

<Paragraph number="3822" type="verses" book="ap1">
_Sirasmiṃ añjaliṃ katvā,_  
_namassitvāna subbataṃ;_  
_Pasannacitto sumano,_  
_uttarīyamadāsahaṃ._  

</Paragraph>

</Division>
<Division number="95" book="ap1">

<Paragraph number="3823" type="verses" book="ap1">
_Ubho hatthehi paggayha,_  
_sāṭakaṃ ukkhipiṃ ahaṃ;_  
_Yāvatā buddhaparisā,_  
_tāva chādesi sāṭako._  

</Paragraph>

</Division>
<Division number="96" book="ap1">

<Paragraph number="3824" type="verses" book="ap1">
_Piṇḍacāraṃ carantassa,_  
_mahābhikkhugaṇādino;_  
_Chadaṃ karonto aṭṭhāsi,_  
_hāsayanto mamaṃ tadā._  

</Paragraph>

</Division>
<Division number="97" book="ap1">

<Paragraph number="3825" type="verses" book="ap1">
_Gharato nikkhamantassa,_  
_sayambhū aggapuggalo;_  
_Vīthiyaṃva ṭhito satthā,_  
_akā me anumodanaṃ._  

</Paragraph>

</Division>
<Division number="98" book="ap1">

<Paragraph number="3826" type="verses" book="ap1">
_Pasannacitto sumano,_  
_yo me adāsi sāṭakaṃ;_  
_Tamahaṃ kittayissāmi,_  
_suṇotha mama bhāsato._  

</Paragraph>

</Division>
<Division number="99" book="ap1">

<Paragraph number="3827" type="verses" book="ap1">
_‘Tiṃsakappasahassāni,_  
_devaloke ramissati;_  
_Paññāsakkhattuṃ devindo,_  
_devarajjaṃ karissati._  

</Paragraph>

</Division>
<Division number="100" book="ap1">

<Paragraph number="3828" type="verses" book="ap1">
_Devaloke vasantassa,_  
_puññakammasamaṅgino;_  
_Samantā yojanasataṃ,_  
_dussacchannaṃ bhavissati._  

</Paragraph>

</Division>
<Division number="101" book="ap1">

<Paragraph number="3829" type="verses" book="ap1">
_Chattiṃsakkhattuṃ rājā ca,_  
_cakkavattī bhavissati;_  
_Padesarajjaṃ vipulaṃ,_  
_gaṇanāto asaṅkhiyaṃ._  

</Paragraph>

</Division>
<Division number="102" book="ap1">

<Paragraph number="3830" type="verses" book="ap1">
_Bhave saṃsaramānassa,_  
_puññakammasamaṅgino;_  
_Manasā patthitaṃ sabbaṃ,_  
_nibbattissati tāvade._  

</Paragraph>

</Division>
<Division number="103" book="ap1">

<Paragraph number="3831" type="verses" book="ap1">
_Koseyyakambaliyāni,_  
_khomakappāsikāni ca;_  
_Mahagghāni ca dussāni,_  
_paṭilacchatiyaṃ naro._  

</Paragraph>

</Division>
<Division number="104" book="ap1">

<Paragraph number="3832" type="verses" book="ap1">
_Manasā patthitaṃ sabbaṃ,_  
_paṭilacchatiyaṃ naro;_  
_Ekadussassa vipākaṃ,_  
_anubhossati sabbadā._  

</Paragraph>

</Division>
<Division number="105" book="ap1">

<Paragraph number="3833" type="verses" book="ap1">
_So pacchā pabbajitvāna,_  
_sukkamūlena codito;_  
_Gotamassa bhagavato,_  
_dhammaṃ sacchikarissati’._  

</Paragraph>

</Division>
<Division number="106" book="ap1">

<Paragraph number="3834" type="verses" book="ap1">
_Aho me sukataṃ kammaṃ,_  
_sabbaññussa mahesino;_  
_Ekāhaṃ sāṭakaṃ datvā,_  
_pattomhi amataṃ padaṃ._  

</Paragraph>

</Division>
<Division number="107" book="ap1">

<Paragraph number="3835" type="verses" book="ap1">
_Maṇḍape rukkhamūle vā,_  
_vasato suññake ghare;_  
_Dhāreti dussachadanaṃ,_  
_samantā byāmato mama._  

</Paragraph>

</Division>
<Division number="108" book="ap1">

<Paragraph number="3836" type="verses" book="ap1">
_Aviññattaṃ nivāsemi,_  
_cīvaraṃ paccayañcahaṃ;_  
_Lābhī annassa pānassa,_  
_uttareyyassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="109" book="ap1">

<Paragraph number="3837" type="verses" book="ap1">
_Satasahassito kappe,_  
_yaṃ kammamakariṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_vatthadānassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="110" book="ap1">

<Paragraph number="3838" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="3839" book="ap1">
Itthaṃ sudaṃ āyasmā uttareyyadāyako thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="3840" type="center" book="ap1">
Uttareyyadāyakattherassāpadānaṃ aṭṭhamaṃ.
</Paragraph>
