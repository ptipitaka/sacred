<Division number="29" book="ap1">

<Paragraph number="4877" type="verses" book="ap1">
_“Pabbhārakūṭaṃ nissāya,_  
_sayambhū aparājito;_  
_Ābādhiko ca so buddho,_  
_vasati pabbatantare._  

</Paragraph>

</Division>
<Division number="30" book="ap1">

<Paragraph number="4878" type="verses" book="ap1">
_Mama assamasāmantā,_  
_panādo āsi tāvade;_  
_Buddhe nibbāyamānamhi,_  
_āloko udapajjatha._  

</Paragraph>

</Division>
<Division number="31" book="ap1">

<Paragraph number="4879" type="verses" book="ap1">
_Yāvatā vanasaṇḍasmiṃ,_  
_acchakokataracchakā;_  
_Vāḷā ca kesarī sabbe,_  
_abhigajjiṃsu tāvade._  

</Paragraph>

</Division>
<Division number="32" book="ap1">

<Paragraph number="4880" type="verses" book="ap1">
_Uppātaṃ tamahaṃ disvā,_  
_pabbhāraṃ agamāsahaṃ;_  
_Tatthaddassāsiṃ sambuddhaṃ,_  
_nibbutaṃ aparājitaṃ._  

</Paragraph>

</Division>
<Division number="33" book="ap1">

<Paragraph number="4881" type="verses" book="ap1">
_Suphullaṃ sālarājaṃva,_  
_sataraṃsiṃva uggataṃ;_  
_Vītaccikaṃva aṅgāraṃ,_  
_nibbutaṃ aparājitaṃ._  

</Paragraph>

</Division>
<Division number="34" book="ap1">

<Paragraph number="4882" type="verses" book="ap1">
_Tiṇaṃ kaṭṭhañca pūretvā,_  
_citakaṃ tatthakāsahaṃ;_  
_Citakaṃ sukataṃ katvā,_  
_sarīraṃ jhāpayiṃ ahaṃ._  

</Paragraph>

</Division>
<Division number="35" book="ap1">

<Paragraph number="4883" type="verses" book="ap1">
_Sarīraṃ jhāpayitvāna,_  
_gandhatoyaṃ samokiriṃ;_  
_Antalikkhe ṭhito yakkho,_  
_nāmamaggahi tāvade._  

</Paragraph>

</Division>
<Division number="36" book="ap1">

<Paragraph number="4884" type="verses" book="ap1">
_‘Yaṃ pūritaṃ tayā kiccaṃ,_  
_sayambhussa mahesino;_  
_Puṇṇako nāma nāmena,_  
_sadā hohi tuvaṃ mune’._  

</Paragraph>

</Division>
<Division number="37" book="ap1">

<Paragraph number="4885" type="verses" book="ap1">
_Tamhā kāyā cavitvāna,_  
_devalokaṃ agacchahaṃ;_  
_Tattha dibbamayo gandho,_  
_antalikkhā pavassati._  

</Paragraph>

</Division>
<Division number="38" book="ap1">

<Paragraph number="4886" type="verses" book="ap1">
_Tatrāpi nāmadheyyaṃ me,_  
_puṇṇakoti ahū tadā;_  
_Devabhūto manusso vā,_  
_saṅkappaṃ pūrayāmahaṃ._  

</Paragraph>

</Division>
<Division number="39" book="ap1">

<Paragraph number="4887" type="verses" book="ap1">
_Idaṃ pacchimakaṃ mayhaṃ,_  
_carimo vattate bhavo;_  
_Idhāpi puṇṇako nāma,_  
_nāmadheyyaṃ pakāsati._  

</Paragraph>

</Division>
<Division number="40" book="ap1">

<Paragraph number="4888" type="verses" book="ap1">
_Tosayitvāna sambuddhaṃ,_  
_gotamaṃ sakyapuṅgavaṃ;_  
_Sabbāsave pariññāya,_  
_viharāmi anāsavo._  

</Paragraph>

</Division>
<Division number="41" book="ap1">

<Paragraph number="4889" type="verses" book="ap1">
_Ekanavutito kappe,_  
_yaṃ kammamakariṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_tanukiccassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="42" book="ap1">

<Paragraph number="4890" type="verses" book="ap1">
_Kilesā jhāpitā mayhaṃ,_  
_ …pe… _  
_viharāmi anāsavo._  

</Paragraph>

</Division>
<Division number="43" book="ap1">

<Paragraph number="4891" type="verses" book="ap1">
_Svāgataṃ vata me āsi,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ._  

</Paragraph>

</Division>
<Division number="44" book="ap1">

<Paragraph number="4892" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="4893" book="ap1">
Itthaṃ sudaṃ āyasmā puṇṇako thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="4894" type="center" book="ap1">
Puṇṇakattherassāpadānaṃ dutiyaṃ.
</Paragraph>
