<Paragraph number="6268" type="verses" book="ap1">
_“Pañca kāmaguṇe hitvā,_  
_piyarūpe manorame;_  
_Asītikoṭiyo hitvā,_  
_pabbajiṃ anagāriyaṃ._  

</Paragraph>

</Division>
<Division number="76" book="ap1" bookNo="20" page="435">

<Paragraph number="6269" type="verses" book="ap1">
_Pabbajitvāna kāyena,_  
_pāpakammaṃ vivajjayiṃ;_  
_Vacīduccaritaṃ hitvā,_  
_nadīkūle vasāmahaṃ._  

</Paragraph>

</Division>
<Division number="77" book="ap1" bookNo="20" page="435">

<Paragraph number="6270" type="verses" book="ap1">
_Ekakaṃ maṃ viharantaṃ,_  
_buddhaseṭṭho upāgami;_  
_Nāhaṃ jānāmi buddhoti,_  
_akāsiṃ paṭisanthāraṃ._  

</Paragraph>

</Division>
<Division number="78" book="ap1" bookNo="20" page="435">

<Paragraph number="6271" type="verses" book="ap1">
_Karitvā paṭisanthāraṃ,_  
_nāmagottamapucchahaṃ;_  
_‘Devatānusi gandhabbo,_  
_adu sakko purindado._  

</Paragraph>

</Division>
<Division number="79" book="ap1" bookNo="20" page="435">

<Paragraph number="6272" type="verses" book="ap1">
_Ko vā tvaṃ kassa vā putto,_  
_mahābrahmā idhāgato;_  
_Virocesi disā sabbā,_  
_udayaṃ sūriyo yathā._  

</Paragraph>

</Division>
<Division number="80" book="ap1" bookNo="20" page="435">

<Paragraph number="6273" type="verses" book="ap1">
_Sahassārāni cakkāni,_  
_pāde dissanti mārisa;_  
_Ko vā tvaṃ kassa vā putto,_  
_kathaṃ jānemu taṃ mayaṃ;_  
_Nāmagottaṃ pavedehi,_  
_saṃsayaṃ apanehi me._  

</Paragraph>

</Division>
<Division number="81" book="ap1" bookNo="20" page="435">

<Paragraph number="6274" type="verses" book="ap1">
_Namhi devo na gandhabbo,_  
_namhi sakko purindado;_  
_Brahmabhāvo ca me natthi,_  
_etesaṃ uttamo ahaṃ._  

</Paragraph>

</Division>
<Division number="82" book="ap1" bookNo="20" page="435">

<Paragraph number="6275" type="verses" book="ap1">
_Atīto visayaṃ tesaṃ,_  
_dālayiṃ kāmabandhanaṃ;_  
_Sabbe kilese jhāpetvā,_  
_patto sambodhimuttamaṃ’._  

</Paragraph>

</Division>
<Division number="83" book="ap1" bookNo="20" page="435">

<Paragraph number="6276" type="verses" book="ap1">
_Tassa vācaṃ suṇitvāhaṃ,_  
_idaṃ vacanamabraviṃ;_  
_‘Yadi buddhoti sabbaññū,_  
_nisīda tvaṃ mahāmune._  

</Paragraph>

</Division>
<Division number="84" book="ap1" bookNo="20" page="436">

<Paragraph number="6277" type="verses" book="ap1">
_Tamahaṃ pūjayissāmi,_  
_dukkhassantakaro tuvaṃ’;_  
_Pattharitvā jinacammaṃ,_  
_adāsi satthuno ahaṃ._  

</Paragraph>

</Division>
<Division number="85" book="ap1" bookNo="20" page="436">

<Paragraph number="6278" type="verses" book="ap1">
_Nisīdi tattha bhagavā,_  
_sīhova girigabbhare;_  
_Khippaṃ pabbatamāruyha,_  
_ambassa phalamaggahiṃ._  

</Paragraph>

</Division>
<Division number="86" book="ap1" bookNo="20" page="436">

<Paragraph number="6279" type="verses" book="ap1">
_Sālakalyāṇikaṃ pupphaṃ,_  
_Candanañca mahārahaṃ;_  
_Khippaṃ paggayha taṃ sabbaṃ,_  
_Upetvā lokanāyakaṃ._  

</Paragraph>

</Division>
<Division number="87" book="ap1" bookNo="20" page="436">

<Paragraph number="6280" type="verses" book="ap1">
_Phalaṃ buddhassa datvāna,_  
_sālapupphamapūjayiṃ;_  
_Candanaṃ anulimpitvā,_  
_avandiṃ satthuno ahaṃ._  

</Paragraph>

</Division>
<Division number="88" book="ap1" bookNo="20" page="436">

<Paragraph number="6281" type="verses" book="ap1">
_Pasannacitto sumano,_  
_vipulāya ca pītiyā;_  
_Ajinamhi nisīditvā,_  
_sumedho lokanāyako._  

</Paragraph>

</Division>
<Division number="89" book="ap1" bookNo="20" page="436">

<Paragraph number="6282" type="verses" book="ap1">
_Mama kammaṃ pakittesi,_  
_hāsayanto mamaṃ tadā;_  
_‘Iminā phaladānena,_  
_gandhamālehi cūbhayaṃ._  

</Paragraph>

</Division>
<Division number="90" book="ap1" bookNo="20" page="436">

<Paragraph number="6283" type="verses" book="ap1">
_Pañcavīse kappasate,_  
_devaloke ramissati;_  
_Anūnamanasaṅkappo,_  
_vasavattī bhavissati._  

</Paragraph>

</Division>
<Division number="91" book="ap1" bookNo="20" page="436">

<Paragraph number="6284" type="verses" book="ap1">
_Chabbīsatikappasate,_  
_manussattaṃ gamissati;_  
_Bhavissati cakkavattī,_  
_cāturanto mahiddhiko._  

</Paragraph>

</Division>
<Division number="92" book="ap1" bookNo="20" page="436">

<Paragraph number="6285" type="verses" book="ap1">
_Vebhāraṃ nāma nagaraṃ,_  
_vissakammena māpitaṃ;_  
_Hessati sabbasovaṇṇaṃ,_  
_nānāratanabhūsitaṃ._  

</Paragraph>

</Division>
<Division number="93" book="ap1" bookNo="20" page="436">

<Paragraph number="6286" type="verses" book="ap1">
_Eteneva upāyena,_  
_saṃsarissati so bhave;_  
_Sabbattha pūjito hutvā,_  
_devatte atha mānuse._  

</Paragraph>

</Division>
<Division number="94" book="ap1" bookNo="20" page="436">

<Paragraph number="6287" type="verses" book="ap1">
_Pacchime bhave sampatte,_  
_brahmabandhu bhavissati;_  
_Agārā abhinikkhamma,_  
_anagārī bhavissati;_  
_Abhiññāpāragū hutvā,_  
_nibbāyissatināsavo’._  

</Paragraph>

</Division>
<Division number="95" book="ap1" bookNo="20" page="436">

<Paragraph number="6288" type="verses" book="ap1">
_Idaṃ vatvāna sambuddho,_  
_sumedho lokanāyako;_  
_Mama nijjhāyamānassa,_  
_pakkāmi anilañjase._  

</Paragraph>

</Division>
<Division number="96" book="ap1" bookNo="20" page="436">

<Paragraph number="6289" type="verses" book="ap1">
_Tena kammena sukatena,_  
_cetanāpaṇidhīhi ca;_  
_Jahitvā mānusaṃ dehaṃ,_  
_tāvatiṃsamagacchahaṃ._  

</Paragraph>

</Division>
<Division number="97" book="ap1" bookNo="20" page="437">

<Paragraph number="6290" type="verses" book="ap1">
_Tusitato cavitvāna,_  
_nibbattiṃ mātukucchiyaṃ;_  
_Bhoge me ūnatā natthi,_  
_yamhi gabbhe vasāmahaṃ._  

</Paragraph>

</Division>
<Division number="98" book="ap1" bookNo="20" page="437">

<Paragraph number="6291" type="verses" book="ap1">
_Mātukucchigate mayi,_  
_annapānañca bhojanaṃ;_  
_Mātuyā mama chandena,_  
_nibbattati yadicchakaṃ._  

</Paragraph>

</Division>
<Division number="99" book="ap1" bookNo="20" page="437">

<Paragraph number="6292" type="verses" book="ap1">
_Jātiyā pañcavassena,_  
_pabbajiṃ anagāriyaṃ;_  
_Oropitamhi kesamhi,_  
_arahattamapāpuṇiṃ._  

</Paragraph>

</Division>
<Division number="100" book="ap1" bookNo="20" page="437">

<Paragraph number="6293" type="verses" book="ap1">
_Pubbakammaṃ gavesanto,_  
_orena nāddasaṃ ahaṃ;_  
_Tiṃsakappasahassamhi,_  
_mama kammamanussariṃ._  

</Paragraph>

</Division>
<Division number="101" book="ap1" bookNo="20" page="437">

<Paragraph number="6294" type="verses" book="ap1">
_Namo te purisājañña,_  
_namo te purisuttama;_  
_Tava sāsanamāgamma,_  
_pattomhi acalaṃ padaṃ._  

</Paragraph>

</Division>
<Division number="102" book="ap1" bookNo="20" page="437">

<Paragraph number="6295" type="verses" book="ap1">
_Tiṃsakappasahassamhi,_  
_yaṃ buddhamabhipūjayiṃ;_  
_Duggatiṃ nābhijānāmi,_  
_buddhapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="103" book="ap1" bookNo="20" page="437">

_ …pe… _  
_viharāmi anāsavo._  

</Paragraph>

</Division>

_ …pe… _  
_kataṃ buddhassa sāsanaṃ._  

</Paragraph>
