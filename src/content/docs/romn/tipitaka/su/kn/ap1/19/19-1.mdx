<Division number="1" book="ap1">

<Paragraph number="2581" type="verses" book="ap1">
_“Suvaṇṇavaṇṇaṃ sambuddhaṃ,_  
_Sataraṃsiṃva uggataṃ;_  
_Disaṃ anuvilokentaṃ,_  
_Gacchantaṃ anilañjase._  

</Paragraph>

</Division>
<Division number="2" book="ap1">

<Paragraph number="2582" type="verses" book="ap1">
_Kuṭajaṃ pupphitaṃ disvā,_  
_saṃvitthatasamotthataṃ;_  
_Rukkhato ocinitvāna,_  
_phussassa abhiropayiṃ._  

</Paragraph>

</Division>
<Division number="3" book="ap1">

<Paragraph number="2583" type="verses" book="ap1">
_Dvenavute ito kappe,_  
_yaṃ pupphamabhiropayiṃ;_  
_Duggatiṃ nābhijānāmi,_  
_buddhapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="4" book="ap1">

<Paragraph number="2584" type="verses" book="ap1">
_Ito sattarase kappe,_  
_tayo āsuṃ supupphitā;_  
_Sattaratanasampannā,_  
_cakkavattī mahabbalā._  

</Paragraph>

</Division>
<Division number="5" book="ap1">

<Paragraph number="2585" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_vimokkhāpi ca aṭṭhime;_  
_Chaḷabhiññā sacchikatā,_  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="2586" book="ap1">
Itthaṃ sudaṃ āyasmā kuṭajapupphiyo thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="2587" type="center" book="ap1">
Kuṭajapupphiyattherassāpadānaṃ paṭhamaṃ.
</Paragraph>
