<Paragraph number="652" type="verses" book="ap1">
_“Gaṅgā bhāgīrathī nāma,_  
_himavantā pabhāvitā;_  
_Kutitthe nāviko āsiṃ,_  
_orime ca tariṃ ahaṃ._  

</Paragraph>

</Division>
<Division number="629" book="ap1" bookNo="20" page="55">

<Paragraph number="653" type="verses" book="ap1">
_Padumuttaro nāyako,_  
_sambuddho dvipaduttamo;_  
_Vasī satasahassehi,_  
_gaṅgātīramupāgato._  

</Paragraph>

</Division>
<Division number="630" book="ap1" bookNo="20" page="55">

<Paragraph number="654" type="verses" book="ap1">
_Bahū nāvā samānetvā,_  
_vaḍḍhakīhi susaṅkhataṃ;_  
_Nāvāya chadanaṃ katvā,_  
_paṭimāniṃ narāsabhaṃ._  

</Paragraph>

</Division>
<Division number="631" book="ap1" bookNo="20" page="56">

<Paragraph number="655" type="verses" book="ap1">
_Āgantvāna ca sambuddho,_  
_ārūhi tañca nāvakaṃ;_  
_Vārimajjhe ṭhito satthā,_  
_imā gāthā abhāsatha._  

</Paragraph>

</Division>
<Division number="632" book="ap1" bookNo="20" page="56">

<Paragraph number="656" type="verses" book="ap1">
_‘Yo so tāresi sambuddhaṃ,_  
_saṃghañcāpi anāsavaṃ;_  
_Tena cittappasādena,_  
_devaloke ramissati._  

</Paragraph>

</Division>
<Division number="633" book="ap1" bookNo="20" page="56">

<Paragraph number="657" type="verses" book="ap1">
_Nibbattissati te byamhaṃ,_  
_sukataṃ nāvasaṇṭhitaṃ;_  
_Ākāse pupphachadanaṃ,_  
_dhārayissati sabbadā._  

</Paragraph>

</Division>
<Division number="634" book="ap1" bookNo="20" page="56">

<Paragraph number="658" type="verses" book="ap1">
_Aṭṭhapaññāsakappamhi,_  
_tārako nāma khattiyo;_  
_Cāturanto vijitāvī,_  
_cakkavattī bhavissati._  

</Paragraph>

</Division>
<Division number="635" book="ap1" bookNo="20" page="56">

<Paragraph number="659" type="verses" book="ap1">
_Sattapaññāsakappamhi,_  
_cammako nāma khattiyo;_  
_Uggacchantova sūriyo,_  
_jotissati mahabbalo._  

</Paragraph>

</Division>
<Division number="636" book="ap1" bookNo="20" page="56">

<Paragraph number="660" type="verses" book="ap1">
_Kappasatasahassamhi,_  
_okkākakulasambhavo;_  
_Gotamo nāma gottena,_  
_satthā loke bhavissati._  

</Paragraph>

</Division>
<Division number="637" book="ap1" bookNo="20" page="56">

<Paragraph number="661" type="verses" book="ap1">
_Tidasā so cavitvāna,_  
_manussattaṃ gamissati;_  
_Revato nāma nāmena,_  
_brahmabandhu bhavissati._  

</Paragraph>

</Division>
<Division number="638" book="ap1" bookNo="20" page="56">

<Paragraph number="662" type="verses" book="ap1">
_Agārā nikkhamitvāna,_  
_sukkamūlena codito;_  
_Gotamassa bhagavato,_  
_sāsane pabbajissati._  

</Paragraph>

</Division>
<Division number="639" book="ap1" bookNo="20" page="56">

<Paragraph number="663" type="verses" book="ap1">
_So pacchā pabbajitvāna,_  
_yuttayogo vipassako;_  
_Sabbāsave pariññāya,_  
_nibbāyissatināsavo’._  

</Paragraph>

</Division>
<Division number="640" book="ap1" bookNo="20" page="56">

<Paragraph number="664" type="verses" book="ap1">
_Vīriyaṃ me dhuradhorayhaṃ,_  
_Yogakkhemādhivāhanaṃ;_  
_Dhāremi antimaṃ dehaṃ,_  
_Sammāsambuddhasāsane._  

</Paragraph>

</Division>
<Division number="641" book="ap1" bookNo="20" page="56">

<Paragraph number="665" type="verses" book="ap1">
_Satasahasse kataṃ kammaṃ,_  
_phalaṃ dassesi me idha;_  
_Sumutto saravegova,_  
_kilese jhāpayī mama._  

</Paragraph>

</Division>
<Division number="642" book="ap1" bookNo="20" page="56">

<Paragraph number="666" type="verses" book="ap1">
_Tato maṃ vananirataṃ,_  
_disvā lokantagū muni;_  
_Vanavāsibhikkhūnaggaṃ,_  
_paññapesi mahāmati._  

</Paragraph>

</Division>
<Division number="643" book="ap1" bookNo="20" page="56">

<Paragraph number="667" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_vimokkhāpi ca aṭṭhime;_  
_Chaḷabhiññā sacchikatā,_  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="668" book="ap1">
Itthaṃ sudaṃ āyasmā khadiravaniyo revato thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="669" type="center" book="ap1">
Khadiravaniyarevatattherassāpadānaṃ navamaṃ.
</Paragraph>
