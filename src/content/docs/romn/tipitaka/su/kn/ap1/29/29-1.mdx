<Division number="1" book="ap1">

<Paragraph number="3287" type="verses" book="ap1">
_“Paṇṇasāle nisinnomhi,_  
_paṇṇabhojanabhojano;_  
_Upaviṭṭhañca maṃ santaṃ,_  
_upāgacchi mahāisi._  

</Paragraph>

</Division>
<Division number="2" book="ap1">

<Paragraph number="3288" type="verses" book="ap1">
_Siddhattho lokapajjoto,_  
_sabbalokatikicchako;_  
_Tassa paṇṇaṃ mayā dinnaṃ,_  
_nisinnaṃ paṇṇasanthare._  

</Paragraph>

</Division>
<Division number="3" book="ap1">

<Paragraph number="3289" type="verses" book="ap1">
_Catunnavutito kappe,_  
_yaṃ paṇṇamadadiṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_paṇṇadānassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="4" book="ap1">

<Paragraph number="3290" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_vimokkhāpi ca aṭṭhime;_  
_Chaḷabhiññā sacchikatā,_  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="3291" book="ap1">
Itthaṃ sudaṃ āyasmā paṇṇadāyako thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="3292" type="center" book="ap1">
Paṇṇadāyakattherassāpadānaṃ paṭhamaṃ.
</Paragraph>
