<Paragraph number="5889" type="verses" book="ap1">
_“Morahatthaṃ gahetvāna,_  
_upesiṃ lokanāyakaṃ;_  
_Pasannacitto sumano,_  
_morahatthamadāsahaṃ._  

</Paragraph>

</Division>
<Division number="10" book="ap1" bookNo="20" page="429">

<Paragraph number="5890" type="verses" book="ap1">
_Iminā morahatthena,_  
_cetanāpaṇidhīhi ca;_  
_Nibbāyiṃsu tayo aggī,_  
_labhāmi vipulaṃ sukhaṃ._  

</Paragraph>

</Division>
<Division number="11" book="ap1" bookNo="20" page="429">

<Paragraph number="5891" type="verses" book="ap1">
_Aho buddhā aho dhammā,_  
_aho no satthusampadā;_  
_Datvānahaṃ morahatthaṃ,_  
_labhāmi vipulaṃ sukhaṃ._  

</Paragraph>

</Division>
<Division number="12" book="ap1" bookNo="20" page="430">

<Paragraph number="5892" type="verses" book="ap1">
_Tiyaggī nibbutā mayhaṃ,_  
_bhavā sabbe samūhatā;_  
_Sabbāsavā parikkhīṇā,_  
_natthi dāni punabbhavo._  

</Paragraph>

</Division>
<Division number="13" book="ap1" bookNo="20" page="430">

<Paragraph number="5893" type="verses" book="ap1">
_Ekatiṃse ito kappe,_  
_yaṃ dānamadadiṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_morahatthassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="14" book="ap1" bookNo="20" page="430">

<Paragraph number="5894" type="verses" book="ap1">
_Kilesā jhāpitā mayhaṃ,_  
_ …pe… _  
_viharāmi anāsavo._  

</Paragraph>

</Division>
<Division number="15" book="ap1" bookNo="20" page="430">

<Paragraph number="5895" type="verses" book="ap1">
_Svāgataṃ vata me āsi,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ._  

</Paragraph>

</Division>
<Division number="16" book="ap1" bookNo="20" page="430">

<Paragraph number="5896" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="5897" book="ap1">
Itthaṃ sudaṃ āyasmā morahatthiyo thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="5898" type="center" book="ap1">
Morahatthiyattherassāpadānaṃ dutiyaṃ.
</Paragraph>
