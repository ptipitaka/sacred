<Paragraph number="6347" type="verses" book="ap1">
_“Atthadassī tu bhagavā,_  
_sayambhū lokanāyako;_  
_Vinatā nadiyā tīraṃ,_  
_upāgacchi tathāgato._  

</Paragraph>

</Division>
<Division number="149" book="ap1" bookNo="20" page="441">

<Paragraph number="6348" type="verses" book="ap1">
_Udakā abhinikkhamma,_  
_kacchapo vārigocaro;_  
_Buddhaṃ tāretukāmohaṃ,_  
_upesiṃ lokanāyakaṃ._  

</Paragraph>

</Division>
<Division number="150" book="ap1" bookNo="20" page="441">

<Paragraph number="6349" type="verses" book="ap1">
_‘Abhirūhatu maṃ buddho,_  
_atthadassī mahāmuni;_  
_Ahaṃ taṃ tārayissāmi,_  
_dukkhassantakaro tuvaṃ’._  

</Paragraph>

</Division>
<Division number="151" book="ap1" bookNo="20" page="441">

<Paragraph number="6350" type="verses" book="ap1">
_Mama saṅkappamaññāya,_  
_atthadassī mahāyaso;_  
_Abhirūhitvā me piṭṭhiṃ,_  
_aṭṭhāsi lokanāyako._  

</Paragraph>

</Division>
<Division number="152" book="ap1" bookNo="20" page="441">

<Paragraph number="6351" type="verses" book="ap1">
_Yato sarāmi attānaṃ,_  
_yato pattosmi viññutaṃ;_  
_Sukhaṃ me tādisaṃ natthi,_  
_phuṭṭhe pādatale yathā._  

</Paragraph>

</Division>
<Division number="153" book="ap1" bookNo="20" page="441">

<Paragraph number="6352" type="verses" book="ap1">
_Uttaritvāna sambuddho,_  
_atthadassī mahāyaso;_  
_Naditīramhi ṭhatvāna,_  
_imā gāthā abhāsatha._  

</Paragraph>

</Division>
<Division number="154" book="ap1" bookNo="20" page="442">

<Paragraph number="6353" type="verses" book="ap1">
_‘Yāvatā vattate cittaṃ,_  
_gaṅgāsotaṃ tarāmahaṃ;_  
_Ayañca kacchapo rājā,_  
_tāresi mama paññavā._  

</Paragraph>

</Division>
<Division number="155" book="ap1" bookNo="20" page="442">

<Paragraph number="6354" type="verses" book="ap1">
_Iminā buddhataraṇena,_  
_mettacittavatāya ca;_  
_Aṭṭhārase kappasate,_  
_devaloke ramissati._  

</Paragraph>

</Division>
<Division number="156" book="ap1" bookNo="20" page="442">

<Paragraph number="6355" type="verses" book="ap1">
_Devalokā idhāgantvā,_  
_sukkamūlena codito;_  
_Ekāsane nisīditvā,_  
_kaṅkhāsotaṃ tarissati._  

</Paragraph>

</Division>
<Division number="157" book="ap1" bookNo="20" page="442">

<Paragraph number="6356" type="verses" book="ap1">
_Yathāpi bhaddake khette,_  
_bījaṃ appampi ropitaṃ;_  
_Sammādhāre pavecchante,_  
_phalaṃ toseti kassakaṃ’._  

</Paragraph>

</Division>
<Division number="158" book="ap1" bookNo="20" page="442">

<Paragraph number="6357" type="verses" book="ap1">
_Tathevidaṃ buddhakhettaṃ,_  
_sammāsambuddhadesitaṃ;_  
_Sammādhāre pavecchante,_  
_phalaṃ maṃ tosayissati._  

</Paragraph>

</Division>
<Division number="159" book="ap1" bookNo="20" page="442">

<Paragraph number="6358" type="verses" book="ap1">
_Padhānapahitattomhi,_  
_upasanto nirūpadhi;_  
_Sabbāsave pariññāya,_  
_viharāmi anāsavo._  

</Paragraph>

</Division>
<Division number="160" book="ap1" bookNo="20" page="442">

<Paragraph number="6359" type="verses" book="ap1">
_Aṭṭhārase kappasate,_  
_yaṃ kammamakariṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_taraṇāya idaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="161" book="ap1" bookNo="20" page="442">

<Paragraph number="6360" type="verses" book="ap1">
_Kilesā jhāpitā mayhaṃ,_  
_ …pe… _  
_viharāmi anāsavo._  

</Paragraph>

</Division>
<Division number="162" book="ap1" bookNo="20" page="442">

<Paragraph number="6361" type="verses" book="ap1">
_Svāgataṃ vata me āsi,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ._  

</Paragraph>

</Division>
<Division number="163" book="ap1" bookNo="20" page="442">

<Paragraph number="6362" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="6363" book="ap1">
Itthaṃ sudaṃ āyasmā taraṇiyo thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="6364" type="center" book="ap1">
Taraṇiyattherassāpadānaṃ aṭṭhamaṃ.
</Paragraph>
