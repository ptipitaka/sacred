<Division number="20" book="ap1">

<Paragraph number="1516" type="verses" book="ap1">
_“Usabhaṃ pavaraṃ vīraṃ,_  
_Mahesiṃ vijitāvinaṃ;_  
_Suvaṇṇavaṇṇaṃ sambuddhaṃ,_  
_Ko disvā nappasīdati._  

</Paragraph>

</Division>
<Division number="21" book="ap1">

<Paragraph number="1517" type="verses" book="ap1">
_Himavāvāparimeyyo,_  
_sāgarova duruttaro;_  
_Tatheva jhānaṃ buddhassa,_  
_ko disvā nappasīdati._  

</Paragraph>

</Division>
<Division number="22" book="ap1">

<Paragraph number="1518" type="verses" book="ap1">
_Vasudhā yathāppameyyā,_  
_cittā vanavaṭaṃsakā;_  
_Tatheva sīlaṃ buddhassa,_  
_ko disvā nappasīdati._  

</Paragraph>

</Division>
<Division number="23" book="ap1">

<Paragraph number="1519" type="verses" book="ap1">
_Anilañjasāsaṅkhubbho,_  
_yathākāso asaṅkhiyo;_  
_Tatheva ñāṇaṃ buddhassa,_  
_ko disvā nappasīdati._  

</Paragraph>

</Division>
<Division number="24" book="ap1">

<Paragraph number="1520" type="verses" book="ap1">
_Imāhi catugāthāhi,_  
_brāhmaṇo senasavhayo;_  
_Buddhaseṭṭhaṃ thavitvāna,_  
_siddhatthaṃ aparājitaṃ._  

</Paragraph>

</Division>
<Division number="25" book="ap1">

<Paragraph number="1521" type="verses" book="ap1">
_Catunnavutikappāni,_  
_duggatiṃ nupapajjatha;_  
_Sugatiṃ sukhasampattiṃ,_  
_anubhosimanappakaṃ._  

</Paragraph>

</Division>
<Division number="26" book="ap1">

<Paragraph number="1522" type="verses" book="ap1">
_Catunnavutito kappe,_  
_thavitvā lokanāyakaṃ;_  
_Duggatiṃ nābhijānāmi,_  
_thomanāya idaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="27" book="ap1">

<Paragraph number="1523" type="verses" book="ap1">
_Cātuddasamhi kappamhi,_  
_caturo āsumuggatā;_  
_Sattaratanasampannā,_  
_cakkavattī mahabbalā._  

</Paragraph>

</Division>
<Division number="28" book="ap1">

<Paragraph number="1524" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="1525" book="ap1">
Itthaṃ sudaṃ āyasmā parappasādako thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="1526" type="center" book="ap1">
Parappasādakattherassāpadānaṃ catutthaṃ.
</Paragraph>
