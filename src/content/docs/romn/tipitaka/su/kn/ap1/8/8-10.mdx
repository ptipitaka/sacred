<Division number="48" book="ap1">

<Paragraph number="1654" type="verses" book="ap1">
_“Padumuttaro nāma jino,_  
_sabbadhammāna pāragū;_  
_Danto dantaparivuto,_  
_nagarā nikkhamī tadā._  

</Paragraph>

</Division>
<Division number="49" book="ap1">

<Paragraph number="1655" type="verses" book="ap1">
_Nagare haṃsavatiyaṃ,_  
_ahosiṃ māliko tadā;_  
_Yaṃ tattha uttamaṃ tīṇi,_  
_padmapupphāni aggahiṃ._  

</Paragraph>

</Division>
<Division number="50" book="ap1">

<Paragraph number="1656" type="verses" book="ap1">
_‘Addasaṃ virajaṃ buddhaṃ,_  
_paṭimaggantarāpaṇe;_  
_Saha disvāna sambuddhaṃ,_  
_evaṃ cintesahaṃ tadā._  

</Paragraph>

</Division>
<Division number="51" book="ap1">

<Paragraph number="1657" type="verses" book="ap1">
_Kiṃ me imehi pupphehi,_  
_rañño upanitehi me;_  
_Gāmaṃ vā gāmakhettaṃ vā,_  
_sahassaṃ vā labheyyahaṃ._  

</Paragraph>

</Division>
<Division number="52" book="ap1">

<Paragraph number="1658" type="verses" book="ap1">
_Adantadamanaṃ vīraṃ,_  
_sabbasattasukhāvahaṃ;_  
_Lokanāthaṃ pūjayitvā,_  
_lacchāmi amataṃ dhanaṃ’._  

</Paragraph>

</Division>
<Division number="53" book="ap1">

<Paragraph number="1659" type="verses" book="ap1">
_Evāhaṃ cintayitvāna,_  
_sakaṃ cittaṃ pasādayiṃ;_  
_Tīṇi lohitake gayha,_  
_ākāse ukkhipiṃ tadā._  

</Paragraph>

</Division>
<Division number="54" book="ap1">

<Paragraph number="1660" type="verses" book="ap1">
_Mayā ukkhittamattamhi,_  
_ākāse patthariṃsu te;_  
_Dhāriṃsu matthake tattha,_  
_uddhaṃvaṇṭā adhomukhā._  

</Paragraph>

</Division>
<Division number="55" book="ap1">

<Paragraph number="1661" type="verses" book="ap1">
_Ye keci manujā disvā,_  
_ukkuṭṭhiṃ sampavattayuṃ;_  
_Devatā antalikkhamhi,_  
_sādhukāraṃ pavattayuṃ._  

</Paragraph>

</Division>
<Division number="56" book="ap1">

<Paragraph number="1662" type="verses" book="ap1">
_Accheraṃ loke uppannaṃ,_  
_buddhaseṭṭhassa vāhasā;_  
_Sabbe dhammaṃ suṇissāma,_  
_pupphānaṃ vāhasā mayaṃ._  

</Paragraph>

</Division>
<Division number="57" book="ap1">

<Paragraph number="1663" type="verses" book="ap1">
_Padumuttaro lokavidū,_  
_āhutīnaṃ paṭiggaho;_  
_Vīthiyañhi ṭhito santo,_  
_imā gāthā abhāsatha._  

</Paragraph>

</Division>
<Division number="58" book="ap1">

<Paragraph number="1664" type="verses" book="ap1">
_‘Yo so buddhaṃ apūjesi,_  
_rattapadmehi māṇavo;_  
_Tamahaṃ kittayissāmi,_  
_suṇātha mama bhāsato._  

</Paragraph>

</Division>
<Division number="59" book="ap1">

<Paragraph number="1665" type="verses" book="ap1">
_Tiṃsakappasahassāni,_  
_devaloke ramissati;_  
_Tiṃsakappāni devindo,_  
_devarajjaṃ karissati._  

</Paragraph>

</Division>
<Division number="60" book="ap1">

<Paragraph number="1666" type="verses" book="ap1">
_Mahāvitthārikaṃ nāma,_  
_byamhaṃ hessati tāvade;_  
_Tiyojanasatubbiddhaṃ,_  
_diyaḍḍhasatavitthataṃ._  

</Paragraph>

</Division>
<Division number="61" book="ap1">

<Paragraph number="1667" type="verses" book="ap1">
_Cattārisatasahassāni,_  
_niyyūhā ca sumāpitā;_  
_Kūṭāgāravarūpetā,_  
_mahāsayanamaṇḍitā._  

</Paragraph>

</Division>
<Division number="62" book="ap1">

<Paragraph number="1668" type="verses" book="ap1">
_Koṭisatasahassiyo,_  
_parivāressanti accharā;_  
_Kusalā naccagītassa,_  
_vāditepi padakkhiṇā._  

</Paragraph>

</Division>
<Division number="63" book="ap1">

<Paragraph number="1669" type="verses" book="ap1">
_Etādise byamhavare,_  
_nārīgaṇasamākule;_  
_Vassissati pupphavasso,_  
_dibbo lohitako sadā._  

</Paragraph>

</Division>
<Division number="64" book="ap1">

<Paragraph number="1670" type="verses" book="ap1">
_Bhittikhīle nāgadante,_  
_dvārabāhāya toraṇe;_  
_Cakkamattā lohitakā,_  
_olambissanti tāvade._  

</Paragraph>

</Division>
<Division number="65" book="ap1">

<Paragraph number="1671" type="verses" book="ap1">
_Pattena pattasañchanne,_  
_antobyamhavare imaṃ;_  
_Attharitvā pārupitvā,_  
_tuvaṭṭissanti tāvade._  

</Paragraph>

</Division>
<Division number="66" book="ap1">

<Paragraph number="1672" type="verses" book="ap1">
_Bhavanaṃ parivāretvā,_  
_samantā satayojane;_  
_Tepi padmā lohitakā,_  
_dibbagandhaṃ pavāyare._  

</Paragraph>

</Division>
<Division number="67" book="ap1">

<Paragraph number="1673" type="verses" book="ap1">
_Pañcasattatikkhattuñca,_  
_cakkavattī bhavissati;_  
_Padesarajjaṃ vipulaṃ,_  
_gaṇanāto asaṅkhiyaṃ._  

</Paragraph>

</Division>
<Division number="68" book="ap1">

<Paragraph number="1674" type="verses" book="ap1">
_Sampattiyo duve bhutvā,_  
_anīti anupaddavo;_  
_Sampatte pariyosāne,_  
_nibbānaṃ pāpuṇissati’._  

</Paragraph>

</Division>
<Division number="69" book="ap1">

<Paragraph number="1675" type="verses" book="ap1">
_Sudiṭṭho vata me buddho,_  
_vāṇijjaṃ supayojitaṃ;_  
_Padmāni tīṇi pūjetvā,_  
_anubhosiṃ tisampadā._  

</Paragraph>

</Division>
<Division number="70" book="ap1">

<Paragraph number="1676" type="verses" book="ap1">
_Ajja me dhammappattassa,_  
_vippamuttassa sabbaso;_  
_Supupphitaṃ lohitakaṃ,_  
_dhārayissati matthake._  

</Paragraph>

</Division>
<Division number="71" book="ap1">

<Paragraph number="1677" type="verses" book="ap1">
_Mama kammaṃ kathentassa,_  
_padumuttarasatthuno;_  
_Satapāṇasahassānaṃ,_  
_dhammābhisamayo ahu._  

</Paragraph>

</Division>
<Division number="72" book="ap1">

<Paragraph number="1678" type="verses" book="ap1">
_Satasahassito kappe,_  
_yaṃ buddhamabhipūjayiṃ;_  
_Duggatiṃ nābhijānāmi,_  
_tipadumānidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="73" book="ap1">

<Paragraph number="1679" type="verses" book="ap1">
_Kilesā jhāpitā mayhaṃ,_  
_bhavā sabbe samūhatā;_  
_Sabbāsavā parikkhīṇā,_  
_natthi dāni punabbhavo._  

</Paragraph>

</Division>

_ …pe… _  
_kataṃ buddhassa sāsanaṃ”. (1486)_

</Paragraph>
