<Division number="44" book="ap1">

<Paragraph number="2881" type="verses" book="ap1">
_“Nisinno pāsādavare,_  
_nārīgaṇapurakkhato;_  
_Byādhitaṃ samaṇaṃ disvā,_  
_abhināmesahaṃ gharaṃ._  

</Paragraph>

</Division>
<Division number="45" book="ap1">

<Paragraph number="2882" type="verses" book="ap1">
_Upaviṭṭhaṃ mahāvīraṃ,_  
_devadevaṃ narāsabhaṃ;_  
_Sappitelaṃ mayā dinnaṃ,_  
_siddhatthassa mahesino._  

</Paragraph>

</Division>
<Division number="46" book="ap1">

<Paragraph number="2883" type="verses" book="ap1">
_Passaddhadarathaṃ disvā,_  
_vippasannamukhindriyaṃ;_  
_Vanditvā satthuno pāde,_  
_anusaṃsāvayiṃ pure._  

</Paragraph>

</Division>
<Division number="47" book="ap1">

<Paragraph number="2884" type="verses" book="ap1">
_Disvā maṃ suppasannattaṃ,_  
_iddhiyā pāramiṅgato;_  
_Nabhaṃ abbhuggamī dhīro,_  
_haṃsarājāva ambare._  

</Paragraph>

</Division>
<Division number="48" book="ap1">

<Paragraph number="2885" type="verses" book="ap1">
_Catunnavutito kappe,_  
_yaṃ dānamadadiṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_sappitelassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="49" book="ap1">

<Paragraph number="2886" type="verses" book="ap1">
_Ito sattarase kappe,_  
_jutideva sanāmako;_  
_Sattaratanasampanno,_  
_cakkavattī mahabbalo._  

</Paragraph>

</Division>
<Division number="50" book="ap1">

<Paragraph number="2887" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="2888" book="ap1">
Itthaṃ sudaṃ āyasmā sappidāyako thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="2889" type="center" book="ap1">
Sappidāyakattherassāpadānaṃ navamaṃ.
</Paragraph>
