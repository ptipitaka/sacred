<Paragraph number="3760" type="verses" book="ap1">
_“Vipassī nāma sambuddho,_  
_lokajeṭṭho narāsabho;_  
_Khīṇāsavehi sahito,_  
_saṃghārāme vasī tadā._  

</Paragraph>

</Division>
<Division number="39" book="ap1" bookNo="20" page="308">

<Paragraph number="3761" type="verses" book="ap1">
_Ārāmadvārā nikkhamma,_  
_vipassī lokanāyako;_  
_Saha satasahassehi,_  
_aṭṭha khīṇāsavehi so._  

</Paragraph>

</Division>
<Division number="40" book="ap1" bookNo="20" page="308">

<Paragraph number="3762" type="verses" book="ap1">
_Ajinena nivatthohaṃ,_  
_vākacīradharopi ca;_  
_Kusumodakamādāya,_  
_sambuddhaṃ upasaṅkamiṃ._  

</Paragraph>

</Division>
<Division number="41" book="ap1" bookNo="20" page="308">

<Paragraph number="3763" type="verses" book="ap1">
_Sakaṃ cittaṃ pasādetvā,_  
_vedajāto katañjalī;_  
_Kusumodakamādāya,_  
_buddhamabbhukkiriṃ ahaṃ._  

</Paragraph>

</Division>
<Division number="42" book="ap1" bookNo="20" page="308">

<Paragraph number="3764" type="verses" book="ap1">
_Tena kammena sambuddho,_  
_jalajuttamanāmako;_  
_Mama kammaṃ pakittetvā,_  
_agamā yena patthitaṃ._  

</Paragraph>

</Division>
<Division number="43" book="ap1" bookNo="20" page="309">

<Paragraph number="3765" type="verses" book="ap1">
_Phusitā pañcasahassā,_  
_yehi pūjesahaṃ jinaṃ;_  
_Aḍḍhateyyasahassehi,_  
_devarajjaṃ akārayiṃ._  

</Paragraph>

</Division>
<Division number="44" book="ap1" bookNo="20" page="309">

<Paragraph number="3766" type="verses" book="ap1">
_Aḍḍhateyyasahassehi,_  
_cakkavattī ahosahaṃ;_  
_Avasesena kammena,_  
_arahattamapāpuṇiṃ._  

</Paragraph>

</Division>
<Division number="45" book="ap1" bookNo="20" page="309">

<Paragraph number="3767" type="verses" book="ap1">
_Devarājā yadā homi,_  
_manujādhipatī yadā;_  
_Tameva nāmadheyyaṃ me,_  
_phusito nāma homahaṃ._  

</Paragraph>

</Division>
<Division number="46" book="ap1" bookNo="20" page="309">

<Paragraph number="3768" type="verses" book="ap1">
_Devabhūtassa santassa,_  
_athāpi mānusassa vā;_  
_Samantā byāmato mayhaṃ,_  
_phusitaṃva pavassati._  

</Paragraph>

</Division>
<Division number="47" book="ap1" bookNo="20" page="309">

<Paragraph number="3769" type="verses" book="ap1">
_Bhavā ugghāṭitā mayhaṃ,_  
_kilesā jhāpitā mama;_  
_Sabbāsavaparikkhīṇo,_  
_phusitassa idaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="48" book="ap1" bookNo="20" page="309">

<Paragraph number="3770" type="verses" book="ap1">
_Candanasseva me kāyā,_  
_tathā gandho pavāyati;_  
_Sarīrato mama gandho,_  
_aḍḍhakose pavāyati._  

</Paragraph>

</Division>
<Division number="49" book="ap1" bookNo="20" page="309">

<Paragraph number="3771" type="verses" book="ap1">
_Dibbagandhaṃ sampavantaṃ,_  
_puññakammasamaṅginaṃ;_  
_Gandhaṃ ghatvāna jānanti,_  
_phusito āgato idha._  

</Paragraph>

</Division>
<Division number="50" book="ap1" bookNo="20" page="309">

<Paragraph number="3772" type="verses" book="ap1">
_Sākhāpalāsakaṭṭhāni,_  
_tiṇānipi ca sabbaso;_  
_Mama saṅkappamaññāya,_  
_gandho sampajjate khaṇe._  

</Paragraph>

</Division>
<Division number="51" book="ap1" bookNo="20" page="309">

<Paragraph number="3773" type="verses" book="ap1">
_Satasahassito kappe,_  
_candanaṃ abhipūjayiṃ;_  
_Duggatiṃ nābhijānāmi,_  
_phusitassa idaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="52" book="ap1" bookNo="20" page="309">

<Paragraph number="3774" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="3775" book="ap1">
Itthaṃ sudaṃ āyasmā phusitakampiyo thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="3776" type="center" book="ap1">
Phusitakampiyattherassāpadānaṃ pañcamaṃ.
</Paragraph>
