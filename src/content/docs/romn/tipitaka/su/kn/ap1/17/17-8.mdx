<Paragraph number="2484" type="verses" book="ap1">
_“Tivarāyaṃ pure ramme,_  
_rājaputtosahaṃ tadā;_  
_Paṇṇākāraṃ labhitvāna,_  
_upasantassadāsahaṃ._  

</Paragraph>

</Division>
<Division number="39" book="ap1" bookNo="20" page="205">

<Paragraph number="2485" type="verses" book="ap1">
_Adhivāsesi bhagavā,_  
_vatthaṃ hatthena āmasi;_  
_Siddhattho adhivāsetvā,_  
_vehāsaṃ nabhamuggami._  

</Paragraph>

</Division>
<Division number="40" book="ap1" bookNo="20" page="205">

<Paragraph number="2486" type="verses" book="ap1">
_Buddhassa gacchamānassa,_  
_dussā dhāvanti pacchato;_  
_Tattha cittaṃ pasādesiṃ,_  
_buddho no aggapuggalo._  

</Paragraph>

</Division>
<Division number="41" book="ap1" bookNo="20" page="205">

<Paragraph number="2487" type="verses" book="ap1">
_Catunnavutito kappe,_  
_yaṃ dussamadadiṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_dussadānassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="42" book="ap1" bookNo="20" page="205">

<Paragraph number="2488" type="verses" book="ap1">
_Sattasaṭṭhimhito kappe,_  
_cakkavattī tadā ahuṃ;_  
_Parisuddhoti nāmena,_  
_manujindo mahabbalo._  

</Paragraph>

</Division>
<Division number="43" book="ap1" bookNo="20" page="205">

<Paragraph number="2489" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="2490" book="ap1">
Itthaṃ sudaṃ āyasmā dussadāyako thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="2491" type="center" book="ap1">
Dussadāyakattherassāpadānaṃ aṭṭhamaṃ.
</Paragraph>
