<Division number="1" book="ap1">

<Paragraph number="2822" type="verses" book="ap1">
_“Siddhatthassa bhagavato,_  
_dvipadindassa tādino;_  
_Nāgaseṭṭho mayā dinno,_  
_īsādanto urūḷhavā._  

</Paragraph>

</Division>
<Division number="2" book="ap1">

<Paragraph number="2823" type="verses" book="ap1">
_Uttamatthaṃ anubhomi,_  
_santipadamanuttaraṃ;_  
_Nāgadānaṃ mayā dinnaṃ,_  
_sabbalokahitesino._  

</Paragraph>

</Division>
<Division number="3" book="ap1">

<Paragraph number="2824" type="verses" book="ap1">
_Catunnavutito kappe,_  
_yaṃ nāga madadiṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_nāgadānassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="4" book="ap1">

<Paragraph number="2825" type="verses" book="ap1">
_Aṭṭhasattatikappamhi,_  
_soḷasāsiṃsu khattiyā;_  
_Samantapāsādikā nāma,_  
_cakkavattī mahabbalā._  

</Paragraph>

</Division>
<Division number="5" book="ap1">

<Paragraph number="2826" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_vimokkhāpi ca aṭṭhime;_  
_Chaḷabhiññā sacchikatā,_  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="2827" book="ap1">
Itthaṃ sudaṃ āyasmā hatthidāyako thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="2828" type="center" book="ap1">
Hatthidāyakattherassāpadānaṃ paṭhamaṃ.
</Paragraph>
