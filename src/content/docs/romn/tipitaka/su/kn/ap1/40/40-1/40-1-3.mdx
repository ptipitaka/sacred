<Paragraph number="4326" type="verses" book="ap1">
_‘Ekato jātasaṃvaddhā,_  
_ubhinnaṃ ekato manaṃ;_  
_Sādhāraṇā sukhadukkhe,_  
_ubho ca anuvattakā._  

</Paragraph>

</Division>
<Division number="30" book="ap1" bookNo="20" page="353">

<Paragraph number="4327" type="verses" book="ap1">
_Atthi cetasikaṃ dukkhaṃ,_  
_tavādheyyaṃ arindama;_  
_Yadi sakkosi taṃ dukkhaṃ,_  
_vinodeyyāsi khattiya’._  

</Paragraph>

</Division>
<Division number="31" book="ap1" bookNo="20" page="353">

<Paragraph number="4328" type="verses" book="ap1">
_‘Tava dukkhaṃ mama dukkhaṃ,_  
_Ubhinnaṃ ekato mano;_  
_Niṭṭhitanti vijānāhi,_  
_Mamādheyyaṃ sace tuvaṃ’._  

</Paragraph>

</Division>
<Division number="32" book="ap1" bookNo="20" page="353">

<Paragraph number="4329" type="verses" book="ap1">
_‘Jānāhi kho mahārāja,_  
_dukkhaṃ me dubbinodayaṃ;_  
_Pahu samāno gajjasu,_  
_ekaṃ te duccajaṃ varaṃ._  

</Paragraph>

</Division>
<Division number="33" book="ap1" bookNo="20" page="353">

<Paragraph number="4330" type="verses" book="ap1">
_Yāvatā vijite atthi,_  
_yāvatā mama jīvitaṃ;_  
_Etehi yadi te attho,_  
_dassāmi avikampito._  

</Paragraph>

</Division>
<Division number="34" book="ap1" bookNo="20" page="353">

<Paragraph number="4331" type="verses" book="ap1">
_Gajjitaṃ kho tayā deva,_  
_micchā taṃ bahu gajjitaṃ;_  
_Jānissāmi tuvaṃ ajja,_  
_sabbadhamme patiṭṭhitaṃ’._  

</Paragraph>

</Division>
<Division number="35" book="ap1" bookNo="20" page="354">

<Paragraph number="4332" type="verses" book="ap1">
_‘Atibāḷhaṃ nipīḷesi,_  
_dadamānassa me sato;_  
_Kiṃ te me pīḷitenattho,_  
_patthitaṃ te kathehi me’._  

</Paragraph>

</Division>
<Division number="36" book="ap1" bookNo="20" page="354">

<Paragraph number="4333" type="verses" book="ap1">
_‘Icchāmahaṃ mahārāja,_  
_buddhaseṭṭhaṃ anuttaraṃ;_  
_Bhojayissāmi sambuddhaṃ,_  
_vajjaṃ me māhu jīvitaṃ’._  

</Paragraph>

</Division>
<Division number="37" book="ap1" bookNo="20" page="354">

<Paragraph number="4334" type="verses" book="ap1">
_‘Aññaṃ tehaṃ varaṃ dammi,_  
_mā yācittho tathāgataṃ;_  
_Adeyyo kassaci buddho,_  
_maṇi jotiraso yathā’._  

</Paragraph>

</Division>
<Division number="38" book="ap1" bookNo="20" page="354">

<Paragraph number="4335" type="verses" book="ap1">
_‘Nanu te gajjitaṃ deva,_  
_yāva jīvitamattano;_  
_Jīvitaṃ dadamānena,_  
_yuttaṃ dātuṃ tathāgataṃ’._  

</Paragraph>

</Division>
<Division number="39" book="ap1" bookNo="20" page="354">

<Paragraph number="4336" type="verses" book="ap1">
_‘Ṭhapanīyo mahāvīro,_  
_adeyyo kassaci jino;_  
_Na me paṭissuto buddho,_  
_varassu amitaṃ dhanaṃ’._  

</Paragraph>

</Division>
<Division number="40" book="ap1" bookNo="20" page="354">

<Paragraph number="4337" type="verses" book="ap1">
_‘Vinicchayaṃ pāpuṇāma,_  
_pucchissāma vinicchaye;_  
_Yathāsaṇṭhaṃ kathessanti,_  
_paṭipucchāma taṃ tathā’._  

</Paragraph>

</Division>
<Division number="41" book="ap1" bookNo="20" page="354">

<Paragraph number="4338" type="verses" book="ap1">
_Rañño hatthe gahetvāna,_  
_agamāsiṃ vinicchayaṃ;_  
_Purato akkhadassānaṃ,_  
_idaṃ vacanamabraviṃ._  

</Paragraph>

</Division>
<Division number="42" book="ap1" bookNo="20" page="354">

<Paragraph number="4339" type="verses" book="ap1">
_‘Suṇantu me akkhadassā,_  
_rājā varamadāsi me;_  
_Na kiñci ṭhapayitvāna,_  
_jīvitampi pavārayi._  

</Paragraph>

</Division>
<Division number="43" book="ap1" bookNo="20" page="354">

<Paragraph number="4340" type="verses" book="ap1">
_Tassa me varadinnassa,_  
_buddhaseṭṭhaṃ variṃ ahaṃ;_  
_Sudinno hoti me buddho,_  
_chindatha saṃsayaṃ mama’._  

</Paragraph>

</Division>
<Division number="44" book="ap1" bookNo="20" page="354">

<Paragraph number="4341" type="verses" book="ap1">
_‘Sossāma tava vacanaṃ,_  
_bhūmipālassa rājino;_  
_Ubhinnaṃ vacanaṃ sutvā,_  
_chindissāmettha saṃsayaṃ._  

</Paragraph>

</Division>
<Division number="45" book="ap1" bookNo="20" page="354">

<Paragraph number="4342" type="verses" book="ap1">
_Sabbaṃ deva tayā dinnaṃ,_  
_imassa sabbagāhikaṃ;_  
_Na kiñci ṭhapayitvāna,_  
_jīvitampi pavārayi’._  

</Paragraph>

</Division>
<Division number="46" book="ap1" bookNo="20" page="354">

<Paragraph number="4343" type="verses" book="ap1">
_‘Kicchappattova hutvāna,_  
_yācī varamanuttaraṃ;_  
_Imaṃ sudukkhitaṃ ñatvā,_  
_adāsiṃ sabbagāhikaṃ’._  

</Paragraph>

</Division>
<Division number="47" book="ap1" bookNo="20" page="355">

<Paragraph number="4344" type="verses" book="ap1">
_‘Parājayo tuvaṃ deva,_  
_assa deyyo tathāgato;_  
_Ubhinnaṃ saṃsayo chinno,_  
_yathāsaṇṭhamhi tiṭṭhatha’._  

</Paragraph>

</Division>
<Division number="48" book="ap1" bookNo="20" page="355">

<Paragraph number="4345" type="verses" book="ap1">
_Rājā tattheva ṭhatvāna,_  
_akkhadassetadabravi;_  
_‘Sammā mayhampi deyyātha,_  
_puna buddhaṃ labhāmahaṃ’._  

</Paragraph>

</Division>
<Division number="49" book="ap1" bookNo="20" page="355">

<Paragraph number="4346" type="verses" book="ap1">
_‘Pūretvā tava saṅkappaṃ,_  
_bhojayitvā tathāgataṃ;_  
_Puna deyyāsi sambuddhaṃ,_  
_ānandassa yasassino’._
</Paragraph>
</Division>
