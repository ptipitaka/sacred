<Paragraph number="3977" type="verses" book="ap1">
_“Nagare aruṇavatiyā,_  
_āsiṃ yavasiko tadā;_  
_Panthe disvāna sambuddhaṃ,_  
_yavakalāpaṃ santhariṃ._  

</Paragraph>

</Division>
<Division number="7" book="ap1" bookNo="20" page="325">

<Paragraph number="3978" type="verses" book="ap1">
_Anukampako kāruṇiko,_  
_sikhī lokagganāyako;_  
_Mama saṅkappamaññāya,_  
_nisīdi yavasanthare._  

</Paragraph>

</Division>
<Division number="8" book="ap1" bookNo="20" page="325">

<Paragraph number="3979" type="verses" book="ap1">
_Disvā nisinnaṃ vimalaṃ,_  
_mahājhāyiṃ vināyakaṃ;_  
_Pāmojjaṃ janayitvāna,_  
_tattha kālaṅkato ahaṃ._  

</Paragraph>

</Division>
<Division number="9" book="ap1" bookNo="20" page="325">

<Paragraph number="3980" type="verses" book="ap1">
_Ekattiṃse ito kappe,_  
_yaṃ kammamakariṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_yavatthare idaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="10" book="ap1" bookNo="20" page="326">

<Paragraph number="3981" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="3982" book="ap1">
Itthaṃ sudaṃ āyasmā yavakalāpiyo thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="3983" type="center" book="ap1">
Yavakalāpiyattherassāpadānaṃ dutiyaṃ.
</Paragraph>
