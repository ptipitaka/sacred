<Paragraph number="1713" type="verses" book="ap1">
_“Abhibhū nāma so bhikkhu,_  
_sikhino aggasāvako;_  
_Mahānubhāvo tevijjo,_  
_himavantaṃ upāgami._  

</Paragraph>

</Division>
<Division number="23" book="ap1" bookNo="20" page="142">

<Paragraph number="1714" type="verses" book="ap1">
_Ahampi himavantamhi,_  
_ramaṇīyassame isi;_  
_Vasāmi appamaññāsu,_  
_iddhīsu ca tadā vasī._  

</Paragraph>

</Division>
<Division number="24" book="ap1" bookNo="20" page="142">

<Paragraph number="1715" type="verses" book="ap1">
_Pakkhijāto viyākāse,_  
_pabbataṃ adhivatthayiṃ;_  
_Adhopupphaṃ gahetvāna,_  
_āgacchiṃ pabbataṃ ahaṃ._  

</Paragraph>

</Division>
<Division number="25" book="ap1" bookNo="20" page="142">

<Paragraph number="1716" type="verses" book="ap1">
_Satta pupphāni gaṇhitvā,_  
_matthake okiriṃ ahaṃ;_  
_Ālokite ca vīrena,_  
_pakkāmiṃ pācināmukho._  

</Paragraph>

</Division>
<Division number="26" book="ap1" bookNo="20" page="142">

<Paragraph number="1717" type="verses" book="ap1">
_Āvāsaṃ abhisambhosiṃ,_  
_patvāna assamaṃ ahaṃ;_  
_Khāribhāraṃ gahetvāna,_  
_pāyāsiṃ pabbatantaraṃ._  

</Paragraph>

</Division>
<Division number="27" book="ap1" bookNo="20" page="142">

<Paragraph number="1718" type="verses" book="ap1">
_Ajagaro maṃ pīḷesi,_  
_ghorarūpo mahabbalo;_  
_Pubbakammaṃ saritvāna,_  
_tattha kālaṅkato ahaṃ._  

</Paragraph>

</Division>
<Division number="28" book="ap1" bookNo="20" page="142">

<Paragraph number="1719" type="verses" book="ap1">
_Ekattiṃse ito kappe,_  
_yaṃ pupphamabhiropayiṃ;_  
_Duggatiṃ nābhijānāmi,_  
_pupphapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="29" book="ap1" bookNo="20" page="142">

<Paragraph number="1720" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="1721" book="ap1">
Itthaṃ sudaṃ āyasmā adhopupphiyo thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="1722" type="center" book="ap1">
Adhopupphiyattherassāpadānaṃ catutthaṃ.
</Paragraph>
