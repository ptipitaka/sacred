<Paragraph number="999" type="verses" book="ap1">
_“Nagare candavatiyā,_  
_bhatako āsahaṃ tadā;_  
_Parakammāyane yutto,_  
_pabbajjaṃ na labhāmahaṃ._  

</Paragraph>

</Division>
<Division number="135" book="ap1" bookNo="20" page="85">

<Paragraph number="1000" type="verses" book="ap1">
_Mahandhakārapihitā,_  
_tividhaggīhi ḍayhare;_  
_Kena nu kho upāyena,_  
_visaṃyutto bhave ahaṃ._  

</Paragraph>

</Division>
<Division number="136" book="ap1" bookNo="20" page="85">

<Paragraph number="1001" type="verses" book="ap1">
_Deyyadhammo ca me natthi,_  
_varāko bhatako ahaṃ;_  
_Yannūnāhaṃ pañcasīlaṃ,_  
_rakkheyyaṃ paripūrayaṃ._  

</Paragraph>

</Division>
<Division number="137" book="ap1" bookNo="20" page="85">

<Paragraph number="1002" type="verses" book="ap1">
_Anomadassissa munino,_  
_nisabho nāma sāvako;_  
_Tamahaṃ upasaṅkamma,_  
_pañcasikkhāpadaggahiṃ._  

</Paragraph>

</Division>
<Division number="138" book="ap1" bookNo="20" page="85">

<Paragraph number="1003" type="verses" book="ap1">
_Vassasatasahassāni,_  
_āyu vijjati tāvade;_  
_Tāvatā pañcasīlāni,_  
_paripuṇṇāni gopayiṃ._  

</Paragraph>

</Division>
<Division number="139" book="ap1" bookNo="20" page="85">

<Paragraph number="1004" type="verses" book="ap1">
_Maccukāle ca sampatte,_  
_devā assāsayanti maṃ;_  
_‘Ratho sahassayutto te,_  
_mārisāyaṃ upaṭṭhito’._  

</Paragraph>

</Division>
<Division number="140" book="ap1" bookNo="20" page="85">

<Paragraph number="1005" type="verses" book="ap1">
_Vattante carime citte,_  
_mama sīlaṃ anussariṃ;_  
_Tena kammena sukatena,_  
_tāvatiṃsaṃ agacchahaṃ._  

</Paragraph>

</Division>
<Division number="141" book="ap1" bookNo="20" page="85">

<Paragraph number="1006" type="verses" book="ap1">
_Tiṃsakkhattuñca devindo,_  
_devarajjamakārayiṃ;_  
_Dibbasukhaṃ anubhaviṃ,_  
_accharāhi purakkhato._  

</Paragraph>

</Division>
<Division number="142" book="ap1" bookNo="20" page="85">

<Paragraph number="1007" type="verses" book="ap1">
_Pañcasattatikkhattuñca,_  
_cakkavattī ahosahaṃ;_  
_Padesarajjaṃ vipulaṃ,_  
_gaṇanāto asaṅkhiyaṃ._  

</Paragraph>

</Division>
<Division number="143" book="ap1" bookNo="20" page="85">

<Paragraph number="1008" type="verses" book="ap1">
_Devalokā cavitvāna,_  
_sukkamūlena codito;_  
_Pure vesāliyaṃ jāto,_  
_mahāsāle suaḍḍhake._  

</Paragraph>

</Division>
<Division number="144" book="ap1" bookNo="20" page="85">

<Paragraph number="1009" type="verses" book="ap1">
_Vassūpanāyike kāle,_  
_dippante jinasāsane;_  
_Mātā ca me pitā ceva,_  
_pañcasikkhāpadaggahuṃ._  

</Paragraph>

</Division>
<Division number="145" book="ap1" bookNo="20" page="85">

<Paragraph number="1010" type="verses" book="ap1">
_Saha sutvānahaṃ sīlaṃ,_  
_mama sīlaṃ anussariṃ;_  
_Ekāsane nisīditvā,_  
_arahattamapāpuṇiṃ._  

</Paragraph>

</Division>
<Division number="146" book="ap1" bookNo="20" page="85">

<Paragraph number="1011" type="verses" book="ap1">
_<Emphasis>Jātiyā pañcavassena</Emphasis>,_  
_arahattamapāpuṇiṃ;_  
_Upasampādayi buddho,_  
_guṇamaññāya cakkhumā._  

</Paragraph>

</Division>
<Division number="147" book="ap1" bookNo="20" page="85">

<Paragraph number="1012" type="verses" book="ap1">
_Paripuṇṇāni gopetvā,_  
_pañcasikkhāpadānahaṃ;_  
_Aparimeyye ito kappe,_  
_vinipātaṃ na gacchahaṃ._  

</Paragraph>

</Division>
<Division number="148" book="ap1" bookNo="20" page="86">

<Paragraph number="1013" type="verses" book="ap1">
_Svāhaṃ yasamanubhaviṃ,_  
_tesaṃ sīlāna vāhasā;_  
_Kappakoṭimpi kittento,_  
_kittaye ekadesakaṃ._  

</Paragraph>

</Division>
<Division number="149" book="ap1" bookNo="20" page="86">

<Paragraph number="1014" type="verses" book="ap1">
_Pañca sīlāni gopetvā,_  
_tayo hetū labhāmahaṃ;_  
_Dīghāyuko mahābhogo,_  
_tikkhapañño bhavāmahaṃ._  

</Paragraph>

</Division>
<Division number="150" book="ap1" bookNo="20" page="86">

<Paragraph number="1015" type="verses" book="ap1">
_Saṅkittento ca sabbesaṃ,_  
_adhimattañca porisaṃ;_  
_Bhavābhave saṃsaritvā,_  
_ete ṭhāne labhāmahaṃ._  

</Paragraph>

</Division>
<Division number="151" book="ap1" bookNo="20" page="86">

<Paragraph number="1016" type="verses" book="ap1">
_Aparimeyyasīlesu,_  
_vattantā jinasāvakā;_  
_Bhavesu yadi rajjeyyuṃ,_  
_vipāko kīdiso bhave._  

</Paragraph>

</Division>
<Division number="152" book="ap1" bookNo="20" page="86">

<Paragraph number="1017" type="verses" book="ap1">
_Suciṇṇaṃ me pañcasīlaṃ,_  
_bhatakena tapassinā;_  
_Tena sīlenahaṃ ajja,_  
_mocayiṃ sabbabandhanā._  

</Paragraph>

</Division>
<Division number="153" book="ap1" bookNo="20" page="86">

<Paragraph number="1018" type="verses" book="ap1">
_Aparimeyye ito kappe,_  
_pañca sīlāni gopayiṃ;_  
_Duggatiṃ nābhijānāmi,_  
_pañcasīlānidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="154" book="ap1" bookNo="20" page="86">

<Paragraph number="1019" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_vimokkhāpi ca aṭṭhime;_  
_Chaḷabhiññā sacchikatā,_  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="1020" book="ap1">
Itthaṃ sudaṃ āyasmā pañcasīlasamādāniyo thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="1021" type="center" book="ap1">
Pañcasīlasamādāniyattherassāpadānaṃ catutthaṃ.
</Paragraph>
