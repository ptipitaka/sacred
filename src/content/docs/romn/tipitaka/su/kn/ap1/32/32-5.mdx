<Paragraph number="3533" type="verses" book="ap1">
_“Suvaṇṇavaṇṇaṃ sambuddhaṃ,_  
_Pītaraṃsiṃva bhāṇumaṃ;_  
_Avidūrena gacchantaṃ,_  
_Siddhatthaṃ aparājitaṃ._  

</Paragraph>

</Division>
<Division number="21" book="ap1">

<Paragraph number="3534" type="verses" book="ap1">
_Tassa paccuggamitvāna,_  
_pavesetvāna assamaṃ;_  
_Pupphāsanaṃ mayā dinnaṃ,_  
_vippasannena cetasā._  

</Paragraph>

</Division>
<Division number="22" book="ap1">

<Paragraph number="3535" type="verses" book="ap1">
_Añjaliṃ paggahetvāna,_  
_vedajāto tadā ahaṃ;_  
_Buddhe cittaṃ pasādetvā,_  
_taṃ kammaṃ pariṇāmayiṃ._  

</Paragraph>

</Division>
<Division number="23" book="ap1">

<Paragraph number="3536" type="verses" book="ap1">
_Yaṃ me atthi kataṃ puññaṃ,_  
_sayambhumhaparājite;_  
_Sabbena tena kusalena,_  
_vimalo homi sāsane._  

</Paragraph>

</Division>
<Division number="24" book="ap1">

<Paragraph number="3537" type="verses" book="ap1">
_Catunnavutito kappe,_  
_pupphāsanamadaṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_pupphāsanassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="25" book="ap1">

<Paragraph number="3538" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="3539" book="ap1">
Itthaṃ sudaṃ āyasmā pupphāsanadāyako thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="3540" type="center" book="ap1">
Pupphāsanadāyakattherassāpadānaṃ pañcamaṃ.
</Paragraph>
