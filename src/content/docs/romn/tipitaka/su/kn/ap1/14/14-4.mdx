<Division number="26" book="ap1">

<Paragraph number="2213" type="verses" book="ap1">
_“Sunando nāma nāmena,_  
_brāhmaṇo mantapāragū;_  
_Ajjhāyako yācayogo,_  
_vājapeyyaṃ ayājayi._  

</Paragraph>

</Division>
<Division number="27" book="ap1">

<Paragraph number="2214" type="verses" book="ap1">
_Padumuttaro lokavidū,_  
_aggo kāruṇiko isi;_  
_Janataṃ anukampanto,_  
_ambare caṅkamī tadā._  

</Paragraph>

</Division>
<Division number="28" book="ap1">

<Paragraph number="2215" type="verses" book="ap1">
_Caṅkamitvāna sambuddho,_  
_sabbaññū lokanāyako;_  
_Mettāya aphari satte,_  
_appamāṇe nirūpadhi._  

</Paragraph>

</Division>
<Division number="29" book="ap1">

<Paragraph number="2216" type="verses" book="ap1">
_Vaṇṭe chetvāna pupphāni,_  
_brāhmaṇo mantapāragū;_  
_Sabbe sisse samānetvā,_  
_ākāse ukkhipāpayi._  

</Paragraph>

</Division>
<Division number="30" book="ap1">

<Paragraph number="2217" type="verses" book="ap1">
_Yāvatā nagaraṃ āsi,_  
_pupphānaṃ chadanaṃ tadā;_  
_Buddhassa ānubhāvena,_  
_sattāhaṃ na vigacchatha._  

</Paragraph>

</Division>
<Division number="31" book="ap1">

<Paragraph number="2218" type="verses" book="ap1">
_Teneva sukkamūlena,_  
_anubhotvāna sampadā;_  
_Sabbāsave pariññāya,_  
_tiṇṇo loke visattikaṃ._  

</Paragraph>

</Division>
<Division number="32" book="ap1">

<Paragraph number="2219" type="verses" book="ap1">
_Ekārase kappasate,_  
_pañcatiṃsāsu khattiyā;_  
_Ambaraṃsasanāmā te,_  
_cakkavattī mahabbalā._  

</Paragraph>

</Division>
<Division number="33" book="ap1">

<Paragraph number="2220" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="2221" book="ap1">
Itthaṃ sudaṃ āyasmā pupphacchadaniyo thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="2222" type="center" book="ap1">
Pupphacchadaniyattherassāpadānaṃ catutthaṃ.
</Paragraph>
