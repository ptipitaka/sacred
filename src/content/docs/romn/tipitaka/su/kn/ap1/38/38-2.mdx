<Paragraph number="4103" type="verses" book="ap1">
_“Vipassī nāma bhagavā,_  
_sayambhū aggapuggalo;_  
_Purakkhato sasissehi,_  
_pāvisi bandhumaṃ jino._  

</Paragraph>

</Division>
<Division number="7" book="ap1" bookNo="20" page="335">

<Paragraph number="4104" type="verses" book="ap1">
_Tīṇi pāṭalipupphāni,_  
_ucchaṅge ṭhapitāni me;_  
_Sīsaṃ nhāyitukāmova,_  
_nadītitthaṃ agacchahaṃ._  

</Paragraph>

</Division>
<Division number="8" book="ap1" bookNo="20" page="335">

<Paragraph number="4105" type="verses" book="ap1">
_Nikkhamma bandhumatiyā,_  
_addasaṃ lokanāyakaṃ;_  
_Indīvaraṃva jalitaṃ,_  
_ādittaṃva hutāsanaṃ._  

</Paragraph>

</Division>
<Division number="9" book="ap1" bookNo="20" page="335">

<Paragraph number="4106" type="verses" book="ap1">
_Byagghūsabhaṃva pavaraṃ,_  
_abhijātaṃva kesariṃ;_  
_Gacchantaṃ samaṇānaggaṃ,_  
_bhikkhusaṃghapurakkhataṃ._  

</Paragraph>

</Division>
<Division number="10" book="ap1" bookNo="20" page="336">

<Paragraph number="4107" type="verses" book="ap1">
_Tasmiṃ pasanno sugate,_  
_kilesamaladhovane;_  
_Gahetvā tīṇi pupphāni,_  
_buddhaseṭṭhaṃ apūjayiṃ._  

</Paragraph>

</Division>
<Division number="11" book="ap1" bookNo="20" page="336">

<Paragraph number="4108" type="verses" book="ap1">
_Ekanavutito kappe,_  
_yaṃ pupphamabhipūjayiṃ;_  
_Duggatiṃ nābhijānāmi,_  
_buddhapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="12" book="ap1" bookNo="20" page="336">

<Paragraph number="4109" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="4110" book="ap1">
Itthaṃ sudaṃ āyasmā pāṭalipupphiyo thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="4111" type="center" book="ap1">
Pāṭalipupphiyattherassāpadānaṃ dutiyaṃ.
</Paragraph>
