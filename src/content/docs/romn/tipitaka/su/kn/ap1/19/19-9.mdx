<Paragraph number="2647" type="verses" book="ap1">
_“Tāvatiṃsesu devesu,_  
_mahāghoso tadā ahu;_  
_Buddho ca loke nibbāti,_  
_mayañcamha sarāgino._  

</Paragraph>

</Division>
<Division number="52" book="ap1" bookNo="20" page="218">

<Paragraph number="2648" type="verses" book="ap1">
_Tesaṃ saṃvegajātānaṃ,_  
_sokasallasamaṅginaṃ;_  
_Sabalena upatthaddho,_  
_agamaṃ buddhasantikaṃ._  

</Paragraph>

</Division>
<Division number="53" book="ap1" bookNo="20" page="218">

<Paragraph number="2649" type="verses" book="ap1">
_Mandāravaṃ gahetvāna,_  
_saṅgīti abhinimmitaṃ;_  
_Parinibbutakālamhi,_  
_buddhassa abhiropayiṃ._  

</Paragraph>

</Division>
<Division number="54" book="ap1" bookNo="20" page="218">

<Paragraph number="2650" type="verses" book="ap1">
_Sabbe devānumodiṃsu,_  
_accharāyo ca me tadā;_  
_Kappānaṃ satasahassaṃ,_  
_duggatiṃ nupapajjahaṃ._  

</Paragraph>

</Division>
<Division number="55" book="ap1" bookNo="20" page="218">

<Paragraph number="2651" type="verses" book="ap1">
_Saṭṭhikappasahassamhi,_  
_ito soḷasa te janā;_  
_Mahāmallajanā nāma,_  
_cakkavattī mahabbalā._  

</Paragraph>

</Division>
<Division number="56" book="ap1" bookNo="20" page="218">

<Paragraph number="2652" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="2653" book="ap1">
Itthaṃ sudaṃ āyasmā ekacāriyo thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="2654" type="center" book="ap1">
Ekacāriyattherassāpadānaṃ navamaṃ.
</Paragraph>
