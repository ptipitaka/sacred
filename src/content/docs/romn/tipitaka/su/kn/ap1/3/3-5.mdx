<Division number="155" book="ap1">

<Paragraph number="1022" type="verses" book="ap1">
_“Suvaṇṇavaṇṇaṃ sambuddhaṃ,_  
_Gacchantaṃ antarāpaṇe;_  
_Kañcanagghiyasaṃkāsaṃ,_  
_Bāttiṃsavaralakkhaṇaṃ._  

</Paragraph>

</Division>
<Division number="156" book="ap1">

<Paragraph number="1023" type="verses" book="ap1">
_Siddhatthaṃ lokapajjotaṃ,_  
_appameyyaṃ anopamaṃ;_  
_Alatthaṃ paramaṃ pītiṃ,_  
_disvā dantaṃ jutindharaṃ._  

</Paragraph>

</Division>
<Division number="157" book="ap1">

<Paragraph number="1024" type="verses" book="ap1">
_Sambuddhaṃ abhināmetvā,_  
_bhojayiṃ taṃ mahāmuniṃ;_  
_Mahākāruṇiko loke,_  
_anumodi mamaṃ tadā._  

</Paragraph>

</Division>
<Division number="158" book="ap1">

<Paragraph number="1025" type="verses" book="ap1">
_Tasmiṃ mahākāruṇike,_  
_paramassāsakārake;_  
_Buddhe cittaṃ pasādetvā,_  
_kappaṃ saggamhi modahaṃ._  

</Paragraph>

</Division>
<Division number="159" book="ap1">

<Paragraph number="1026" type="verses" book="ap1">
_Catunnavutito kappe,_  
_yaṃ dānamadadiṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_bhikkhādānassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="160" book="ap1">

<Paragraph number="1027" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_vimokkhāpi ca aṭṭhime;_  
_Chaḷabhiññā sacchikatā,_  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="1028" book="ap1">
Itthaṃ sudaṃ āyasmā annasaṃsāvako thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="1029" type="center" book="ap1">
Annasaṃsāvakattherassāpadānaṃ pañcamaṃ.
</Paragraph>
