<Division number="165" book="ap1">

<Paragraph number="7509" type="verses" book="ap1">
_“Padumuttaro nāma jino,_  
_sabbadhammesu cakkhumā;_  
_Ito satasahassamhi,_  
_kappe uppajji nāyako._  

</Paragraph>

</Division>
<Division number="166" book="ap1">

<Paragraph number="7510" type="verses" book="ap1">
_Nāyakānaṃ varo satthā,_  
_guṇāguṇavidū jino;_  
_Kataññū katavedī ca,_  
_titthe yojeti pāṇine._  

</Paragraph>

</Division>
<Division number="167" book="ap1">

<Paragraph number="7511" type="verses" book="ap1">
_Sabbaññutena ñāṇena,_  
_tulayitvā dayāsayo;_  
_Deseti pavaraṃ dhammaṃ,_  
_anantaguṇasañcayo._  

</Paragraph>

</Division>
<Division number="168" book="ap1">

<Paragraph number="7512" type="verses" book="ap1">
_Sa kadāci mahāvīro,_  
_anantajinasaṃsari;_  
_Deseti madhuraṃ dhammaṃ,_  
_catusaccūpasañhitaṃ._  

</Paragraph>

</Division>
<Division number="169" book="ap1">

<Paragraph number="7513" type="verses" book="ap1">
_Sutvāna taṃ dhammavaraṃ,_  
_ādimajjhantasobhanaṃ;_  
_Pāṇasatasahassānaṃ,_  
_dhammābhisamayo ahu._  

</Paragraph>

</Division>
<Division number="170" book="ap1">

<Paragraph number="7514" type="verses" book="ap1">
_Ninnāditā tadā bhūmi,_  
_gajjiṃsu ca payodharā;_  
_Sādhukāraṃ pavattiṃsu,_  
_devabrahmanarāsurā._  

</Paragraph>

</Division>
<Division number="171" book="ap1">

<Paragraph number="7515" type="verses" book="ap1">
_‘Aho kāruṇiko satthā,_  
_aho saddhammadesanā;_  
_Aho bhavasamuddamhi,_  
_nimugge uddharī jino’._  

</Paragraph>

</Division>
<Division number="172" book="ap1">

<Paragraph number="7516" type="verses" book="ap1">
_Evaṃ pavedajātesu,_  
_sanarāmarabrahmasu;_  
_Kulappasādakānaggaṃ,_  
_sāvakaṃ vaṇṇayī jino._  

</Paragraph>

</Division>
<Division number="173" book="ap1">

<Paragraph number="7517" type="verses" book="ap1">
_Tadāhaṃ haṃsavatiyaṃ,_  
_jātomaccakule ahuṃ;_  
_Pāsādiko dassaniyo,_  
_pahūtadhanadhaññavā._  

</Paragraph>

</Division>
<Division number="174" book="ap1">

<Paragraph number="7518" type="verses" book="ap1">
_Haṃsārāmamupeccāhaṃ,_  
_vanditvā taṃ tathāgataṃ;_  
_Suṇitvā madhuraṃ dhammaṃ,_  
_kāraṃ katvā ca tādino._  

</Paragraph>

</Division>
<Division number="175" book="ap1">

<Paragraph number="7519" type="verses" book="ap1">
_Nipacca pādamūlehaṃ,_  
_imaṃ vacanamabraviṃ;_  
_‘Kulappasādakānaggo,_  
_yo tayā santhuto mune._  

</Paragraph>

</Division>
<Division number="176" book="ap1">

<Paragraph number="7520" type="verses" book="ap1">
_Tādiso homahaṃ vīra,_  
_buddhaseṭṭhassa sāsane’;_  
_Tadā mahākāruṇiko,_  
_siñcanto vāmatena maṃ._  

</Paragraph>

</Division>
<Division number="177" book="ap1">

<Paragraph number="7521" type="verses" book="ap1">
_Āha maṃ ‘putta uttiṭṭha,_  
_lacchase taṃ manorathaṃ;_  
_Kathaṃ nāma jine kāraṃ,_  
_katvāna viphalo siyā._  

</Paragraph>

</Division>
<Division number="178" book="ap1">

<Paragraph number="7522" type="verses" book="ap1">
_Satasahassito kappe,_  
_okkākakulasambhavo;_  
_Gotamo nāma gottena,_  
_satthā loke bhavissati._  

</Paragraph>

</Division>
<Division number="179" book="ap1">

<Paragraph number="7523" type="verses" book="ap1">
_Tassa dhammesu dāyādo,_  
_oraso dhammanimmito;_  
_Udāyi nāma nāmena,_  
_hessati satthu sāvako’._  

</Paragraph>

</Division>
<Division number="180" book="ap1">

<Paragraph number="7524" type="verses" book="ap1">
_Taṃ sutvā mudito hutvā,_  
_yāvajīvaṃ tadā jinaṃ;_  
_Mettacitto paricariṃ,_  
_paccayehi vināyakaṃ._  

</Paragraph>

</Division>
<Division number="181" book="ap1">

<Paragraph number="7525" type="verses" book="ap1">
_Tena kammavipākena,_  
_cetanāpaṇidhīhi ca;_  
_Jahitvā mānusaṃ dehaṃ,_  
_tāvatiṃsamagacchahaṃ._  

</Paragraph>

</Division>
<Division number="182" book="ap1">

<Paragraph number="7526" type="verses" book="ap1">
_Pacchime ca bhave dāni,_  
_ramme kapilavatthave;_  
_Jāto mahāmaccakule,_  
_suddhodanamahīpate._  

</Paragraph>

</Division>
<Division number="183" book="ap1">

<Paragraph number="7527" type="verses" book="ap1">
_Tadā ajāyi siddhattho,_  
_ramme lumbinikānane;_  
_Hitāya sabbalokassa,_  
_sukhāya ca narāsabho._  

</Paragraph>

</Division>
<Division number="184" book="ap1">

<Paragraph number="7528" type="verses" book="ap1">
_Tadaheva ahaṃ jāto,_  
_saha teneva vaḍḍhito;_  
_Piyo sahāyo dayito,_  
_viyatto nītikovido._  

</Paragraph>

</Division>
<Division number="185" book="ap1">

<Paragraph number="7529" type="verses" book="ap1">
_Ekūnatiṃso vayasā,_  
_nikkhamitvā agārato;_  
_Chabbassaṃ vītināmetvā,_  
_āsi buddho vināyako._  

</Paragraph>

</Division>
<Division number="186" book="ap1">

<Paragraph number="7530" type="verses" book="ap1">
_Jetvā sasenakaṃ māraṃ,_  
_khepayitvāna āsave;_  
_Bhavaṇṇavaṃ taritvāna,_  
_buddho āsi sadevake._  

</Paragraph>

</Division>
<Division number="187" book="ap1">

<Paragraph number="7531" type="verses" book="ap1">
_Isivhayaṃ gamitvāna,_  
_vinetvā pañcavaggiye;_  
_Tato vinesi bhagavā,_  
_gantvā gantvā tahiṃ tahiṃ._  

</Paragraph>

</Division>
<Division number="188" book="ap1">

<Paragraph number="7532" type="verses" book="ap1">
_Veneyye vinayanto so,_  
_saṅgaṇhanto sadevakaṃ;_  
_Upecca magadhe giriṃ,_  
_viharittha tadā jino._  

</Paragraph>

</Division>
<Division number="189" book="ap1">

<Paragraph number="7533" type="verses" book="ap1">
_Tadā suddhodanenāhaṃ,_  
_bhūmipālena pesito;_  
_Gantvā disvā dasabalaṃ,_  
_pabbajitvārahā ahuṃ._  

</Paragraph>

</Division>
<Division number="190" book="ap1">

<Paragraph number="7534" type="verses" book="ap1">
_Tadā mahesiṃ yācitvā,_  
_pāpayiṃ kapilavhayaṃ;_  
_Tato purāhaṃ gantvāna,_  
_pasādesiṃ mahākulaṃ._  

</Paragraph>

</Division>
<Division number="191" book="ap1">

<Paragraph number="7535" type="verses" book="ap1">
_Jino tasmiṃ guṇe tuṭṭho,_  
_maṃ mahāparisāya so;_  
_Kulappasādakānaggaṃ,_  
_paññāpesi vināyako._  

</Paragraph>

</Division>
<Division number="192" book="ap1">

<Paragraph number="7536" type="verses" book="ap1">
_Kilesā jhāpitā mayhaṃ,_  
_ …pe… _  
_viharāmi anāsavo._  

</Paragraph>

</Division>
<Division number="193" book="ap1">

<Paragraph number="7537" type="verses" book="ap1">
_Svāgataṃ vata me āsi,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ._  

</Paragraph>

</Division>
<Division number="194" book="ap1">

<Paragraph number="7538" type="verses" book="ap1">

_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="7539" book="ap1">

</Paragraph>
</Division>
