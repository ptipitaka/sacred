<Paragraph number="3942" type="verses" book="ap1">
_“Himavantassāvidūre,_  
_lambako nāma pabbato;_  
_Upatisso nāma sambuddho,_  
_abbhokāsamhi caṅkami._  

</Paragraph>

</Division>
<Division number="49" book="ap1" bookNo="20" page="323">

<Paragraph number="3943" type="verses" book="ap1">
_Migaluddo tadā āsiṃ,_  
_araññe kānane ahaṃ;_  
_Disvāna taṃ devadevaṃ,_  
_sayambhuṃ aparājitaṃ._  

</Paragraph>

</Division>
<Division number="50" book="ap1" bookNo="20" page="323">

<Paragraph number="3944" type="verses" book="ap1">
_Vippasannena cittena,_  
_tadā tassa mahesino;_  
_Nisīdanatthaṃ buddhassa,_  
_tiṇamuṭṭhimadāsahaṃ._  

</Paragraph>

</Division>
<Division number="51" book="ap1" bookNo="20" page="323">

<Paragraph number="3945" type="verses" book="ap1">
_Datvāna devadevassa,_  
_bhiyyo cittaṃ pasādayiṃ;_  
_Sambuddhaṃ abhivādetvā,_  
_pakkāmiṃ uttarāmukho._  

</Paragraph>

</Division>
<Division number="52" book="ap1" bookNo="20" page="323">

<Paragraph number="3946" type="verses" book="ap1">
_Aciraṃ gatamattaṃ maṃ,_  
_migarājā apothayi;_  
_Sīhena pothito santo,_  
_tattha kālaṅkato ahaṃ._  

</Paragraph>

</Division>
<Division number="53" book="ap1" bookNo="20" page="323">

<Paragraph number="3947" type="verses" book="ap1">
_Āsanne me kataṃ kammaṃ,_  
_buddhaseṭṭhe anāsave;_  
_Sumutto saravegova,_  
_devalokaṃ agañchahaṃ._  

</Paragraph>

</Division>
<Division number="54" book="ap1" bookNo="20" page="323">

<Paragraph number="3948" type="verses" book="ap1">
_Yūpo tattha subho āsi,_  
_puññakammābhinimmito;_  
_Sahassakaṇḍo satabheṇḍu,_  
_dhajālu haritāmayo._  

</Paragraph>

</Division>
<Division number="55" book="ap1" bookNo="20" page="323">

<Paragraph number="3949" type="verses" book="ap1">
_Pabhā niddhāvate tassa,_  
_sataraṃsīva uggato;_  
_Ākiṇṇo devakaññāhi,_  
_āmodiṃ kāmakāmahaṃ._  

</Paragraph>

</Division>
<Division number="56" book="ap1" bookNo="20" page="323">

<Paragraph number="3950" type="verses" book="ap1">
_Devalokā cavitvāna,_  
_sukkamūlena codito;_  
_Āgantvāna manussattaṃ,_  
_pattomhi āsavakkhayaṃ._  

</Paragraph>

</Division>
<Division number="57" book="ap1" bookNo="20" page="323">

<Paragraph number="3951" type="verses" book="ap1">
_Catunnavutito kappe,_  
_nisīdanamadāsahaṃ;_  
_Duggatiṃ nābhijānāmi,_  
_tiṇamuṭṭhiyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="58" book="ap1" bookNo="20" page="323">

<Paragraph number="3952" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="3953" book="ap1">
Itthaṃ sudaṃ āyasmā tiṇamuṭṭhidāyako thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="3954" type="center" book="ap1">
Tiṇamuṭṭhidāyakattherassāpadānaṃ aṭṭhamaṃ.
</Paragraph>
