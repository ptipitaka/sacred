_sabbalokavidū muni;_

_kappe uppajji cakkhumā._  

</Paragraph>

</Division>
<Division number="109" book="ap1">

<Paragraph number="7062" type="verses" book="ap1">
_Ovādako viññāpako,_  
_tārako sabbapāṇinaṃ;_  
_Desanākusalo buddho,_  
_tāresi janataṃ bahuṃ._  

</Paragraph>

</Division>
<Division number="110" book="ap1">

<Paragraph number="7063" type="verses" book="ap1">
_Anukampako kāruṇiko,_  
_hitesī sabbapāṇinaṃ;_  
_Sampatte titthiye sabbe,_  
_pañcasīle patiṭṭhapi._  

</Paragraph>

</Division>
<Division number="111" book="ap1">

<Paragraph number="7064" type="verses" book="ap1">
_Evaṃ nirākulaṃ āsi,_  
_suññataṃ titthiyehi ca;_  
_Vicittaṃ arahantehi,_  
_vasībhūtehi tādibhi._  

</Paragraph>

</Division>
<Division number="112" book="ap1">

<Paragraph number="7065" type="verses" book="ap1">
_Ratanānaṭṭhapaññāsaṃ,_  
_uggato so mahāmuni;_  
_Kañcanagghiyasaṅkāso,_  
_bāttiṃsavaralakkhaṇo._  

</Paragraph>

</Division>
<Division number="113" book="ap1">

<Paragraph number="7066" type="verses" book="ap1">
_Vassasatasahassāni,_  
_āyu vijjati tāvade;_  
_Tāvatā tiṭṭhamāno so,_  
_tāresi janataṃ bahuṃ._  

</Paragraph>

</Division>
<Division number="114" book="ap1">

<Paragraph number="7067" type="verses" book="ap1">
_Tadāhaṃ haṃsavatiyaṃ,_  
_seṭṭhiputto mahāyaso;_  
_Upetvā lokapajjotaṃ,_  
_assosiṃ dhammadesanaṃ._  

</Paragraph>

</Division>
<Division number="115" book="ap1">

<Paragraph number="7068" type="verses" book="ap1">
_Senāsanāni bhikkhūnaṃ,_  
_paññāpentaṃ sasāvakaṃ;_  
_Kittayantassa vacanaṃ,_  
_suṇitvā mudito ahaṃ._  

</Paragraph>

</Division>
<Division number="116" book="ap1">

<Paragraph number="7069" type="verses" book="ap1">
_Adhikāraṃ sasaṃghassa,_  
_katvā tassa mahesino;_  
_Nipacca sirasā pāde,_  
_taṃ ṭhānamabhipatthayiṃ._  

</Paragraph>

</Division>
<Division number="117" book="ap1">

<Paragraph number="7070" type="verses" book="ap1">
_Tadāha sa mahāvīro,_  
_mama kammaṃ pakittayaṃ;_  
_‘Yo sasaṃghamabhojesi,_  
_sattāhaṃ lokanāyakaṃ._  

</Paragraph>

</Division>
<Division number="118" book="ap1">

<Paragraph number="7071" type="verses" book="ap1">
_Soyaṃ kamalapattakkho,_  
_sīhaṃso kanakattaco;_  
_Mama pādamūle nipati,_  
_patthayaṃ ṭhānamuttamaṃ._  

</Paragraph>

</Division>
<Division number="119" book="ap1">

<Paragraph number="7072" type="verses" book="ap1">
_Satasahassito kappe,_  
_okkākakulasambhavo;_  
_Gotamo nāma gottena,_  
_satthā loke bhavissati._  

</Paragraph>

</Division>
<Division number="120" book="ap1">

<Paragraph number="7073" type="verses" book="ap1">
_Sāvako tassa buddhassa,_  
_dabbo nāmena vissuto;_  
_Senāsanapaññāpako,_  
_aggo hessatiyaṃ tadā’._  

</Paragraph>

</Division>
<Division number="121" book="ap1">

<Paragraph number="7074" type="verses" book="ap1">
_Tena kammena sukatena,_  
_cetanāpaṇidhīhi ca;_  
_Jahitvā mānusaṃ dehaṃ,_  
_tāvatiṃsamagacchahaṃ._  

</Paragraph>

</Division>
<Division number="122" book="ap1">

<Paragraph number="7075" type="verses" book="ap1">
_Satānaṃ tīṇikkhattuñca,_  
_devarajjamakārayiṃ;_  
_Satānaṃ pañcakkhattuñca,_  
_cakkavattī ahosahaṃ._  

</Paragraph>

</Division>
<Division number="123" book="ap1">

<Paragraph number="7076" type="verses" book="ap1">
_Padesarajjaṃ vipulaṃ,_  
_gaṇanāto asaṅkhiyaṃ;_  
_Sabbattha sukhito āsiṃ,_  
_tassa kammassa vāhasā._  

</Paragraph>

</Division>
<Division number="124" book="ap1">

<Paragraph number="7077" type="verses" book="ap1">
_Ekanavutito kappe,_  
_vipassī nāma nāyako;_  
_Uppajji cārudassano,_  
_sabbadhammavipassako._  

</Paragraph>

</Division>
<Division number="125" book="ap1">

<Paragraph number="7078" type="verses" book="ap1">
_Duṭṭhacitto upavadiṃ,_  
_sāvakaṃ tassa tādino;_  
_Sabbāsavaparikkhīṇaṃ,_  
_suddhoti ca vijāniya._  

</Paragraph>

</Division>
<Division number="126" book="ap1">

<Paragraph number="7079" type="verses" book="ap1">
_Tasseva naravīrassa,_  
_sāvakānaṃ mahesinaṃ;_  
_Salākañca gahetvāna,_  
_khīrodanamadāsahaṃ._  

</Paragraph>

</Division>
<Division number="127" book="ap1">

<Paragraph number="7080" type="verses" book="ap1">
_Imamhi bhaddake kappe,_  
_brahmabandhu mahāyaso;_  
_Kassapo nāma gottena,_  
_uppajji vadataṃ varo._  

</Paragraph>

</Division>
<Division number="128" book="ap1">

<Paragraph number="7081" type="verses" book="ap1">
_Sāsanaṃ jotayitvāna,_  
_abhibhuyya kutitthiye;_  
_Vineyye vinayitvāva,_  
_nibbuto so sasāvako._  

</Paragraph>

</Division>
<Division number="129" book="ap1">

<Paragraph number="7082" type="verses" book="ap1">
_Sasisse nibbute nāthe,_  
_atthamentamhi sāsane;_  
_Devā kandiṃsu saṃviggā,_  
_muttakesā rudammukhā._  

</Paragraph>

</Division>
<Division number="130" book="ap1">

<Paragraph number="7083" type="verses" book="ap1">
_‘Nibbāyissati dhammakkho,_  
_na passissāma subbate;_  
_Na suṇissāma saddhammaṃ,_  
_aho no appapuññatā’._  

</Paragraph>

</Division>
<Division number="131" book="ap1">

<Paragraph number="7084" type="verses" book="ap1">
_Tadāyaṃ pathavī sabbā,_  
_acalā sā calācalā;_  
_Sāgaro ca sasokova,_  
_vinadī karuṇaṃ giraṃ._  

</Paragraph>

</Division>
<Division number="132" book="ap1">

<Paragraph number="7085" type="verses" book="ap1">
_Catuddisā dundubhiyo,_  
_nādayiṃsu amānusā;_  
_Samantato asaniyo,_  
_phaliṃsu ca bhayāvahā._  

</Paragraph>

</Division>
<Division number="133" book="ap1">

<Paragraph number="7086" type="verses" book="ap1">
_Ukkā patiṃsu nabhasā,_  
_dhūmaketu ca dissati;_  
_Sadhūmā jālavaṭṭā ca,_  
_raviṃsu karuṇaṃ migā._  

</Paragraph>

</Division>
<Division number="134" book="ap1">

<Paragraph number="7087" type="verses" book="ap1">
_Uppāde dāruṇe disvā,_  
_sāsanatthaṅgasūcake;_  
_Saṃviggā bhikkhavo satta,_  
_cintayimha mayaṃ tadā._  

</Paragraph>

</Division>
<Division number="135" book="ap1">

<Paragraph number="7088" type="verses" book="ap1">

_jīvitena alaṃ mayaṃ;_  
_Pavisitvā mahāraññaṃ,_  
_yuñjāma jinasāsanaṃ’._  

</Paragraph>

</Division>
<Division number="136" book="ap1">

<Paragraph number="7089" type="verses" book="ap1">
_Addasamha tadāraññe,_  
_ubbiddhaṃ selamuttamaṃ;_  
_Nisseṇiyā tamāruyha,_  
_nisseṇiṃ pātayimhase._  

</Paragraph>

</Division>
<Division number="137" book="ap1">

<Paragraph number="7090" type="verses" book="ap1">
_Tadā ovadi no thero,_  
_‘buddhuppādo sudullabho;_  
_Saddhātidullabhā laddhā,_  
_thokaṃ sesañca sāsanaṃ._  

</Paragraph>

</Division>
<Division number="138" book="ap1">

<Paragraph number="7091" type="verses" book="ap1">
_Nipatanti khaṇātītā,_  
_anante dukkhasāgare;_  
_Tasmā payogo kattabbo,_  
_yāva ṭhāti mune mataṃ’._  

</Paragraph>

</Division>
<Division number="139" book="ap1">

<Paragraph number="7092" type="verses" book="ap1">

_anāgāmī tadānugo;_  
_Susīlā itare yuttā,_  
_devalokaṃ agamhase._  

</Paragraph>

</Division>
<Division number="140" book="ap1">

<Paragraph number="7093" type="verses" book="ap1">

_suddhāvāse ca ekako;_  
_Ahañca pakkusāti ca,_  
_sabhiyo bāhiyo tathā._  

</Paragraph>

</Division>
<Division number="141" book="ap1">

_tattha tatthūpagā mayaṃ;_  
_Saṃsārabandhanā muttā,_  
_gotamenānukampitā._  

</Paragraph>

</Division>
<Division number="142" book="ap1">

_jāto gabbheva me sato;_  
_Mātā matā citāruḷhā,_  
_tato nippatito ahaṃ._  

</Paragraph>

</Division>

_vimutto sattavassiko._  

</Paragraph>

</Division>
<Division number="144" book="ap1">

_pañcahaṅgehupāgato;_  
_Khīṇāsavopavādena,_  
_pāpehi bahucodito._  

</Paragraph>

</Division>

_Patvāna paramaṃ santiṃ,_  
_viharāmi anāsavo._  

</Paragraph>

</Paragraph>
