<Division number="1" book="ap1">

<Paragraph number="6798" type="verses" book="ap1">
_“Himavantassāvidūre,_  
_lambako nāma pabbato;_  
_Tattheva tisso sambuddho,_  
_abbhokāsamhi caṅkami._  

</Paragraph>

</Division>
<Division number="2" book="ap1">

<Paragraph number="6799" type="verses" book="ap1">
_Migaluddo pure āsiṃ,_  
_araññe kānane ahaṃ;_  
_Disvāna taṃ devadevaṃ,_  
_tiṇamuṭṭhimadāsahaṃ._  

</Paragraph>

</Division>
<Division number="3" book="ap1">

<Paragraph number="6800" type="verses" book="ap1">
_Nisīdanatthaṃ buddhassa,_  
_datvā cittaṃ pasādayiṃ;_  
_Sambuddhaṃ abhivādetvā,_  
_pakkāmiṃ uttarāmukho._  

</Paragraph>

</Division>
<Division number="4" book="ap1">

<Paragraph number="6801" type="verses" book="ap1">
_Aciraṃ gatamattassa,_  
_migarājā apothayi;_  
_Sīhena pothito santo,_  
_tattha kālaṅkato ahaṃ._  

</Paragraph>

</Division>
<Division number="5" book="ap1">

<Paragraph number="6802" type="verses" book="ap1">
_Āsanne me kataṃ kammaṃ,_  
_buddhaseṭṭhe anāsave;_  
_Sumutto saravegova,_  
_devalokamagañchahaṃ._  

</Paragraph>

</Division>
<Division number="6" book="ap1">

<Paragraph number="6803" type="verses" book="ap1">
_Yūpo tattha subho āsi,_  
_puññakammābhinimmito;_  
_Sahassakaṇḍo satabheṇḍu,_  
_dhajālu haritāmayo._  

</Paragraph>

</Division>
<Division number="7" book="ap1">

<Paragraph number="6804" type="verses" book="ap1">
_Pabhā niddhāvate tassa,_  
_sataraṃsīva uggato;_  
_Ākiṇṇo devakaññāhi,_  
_āmodiṃ kāmakāmihaṃ._  

</Paragraph>

</Division>
<Division number="8" book="ap1">

<Paragraph number="6805" type="verses" book="ap1">
_Devalokā cavitvāna,_  
_sukkamūlena codito;_  
_Āgantvāna manussattaṃ,_  
_pattomhi āsavakkhayaṃ._  

</Paragraph>

</Division>
<Division number="9" book="ap1">

<Paragraph number="6806" type="verses" book="ap1">
_Catunnavutito kappe,_  
_nisīdanamadāsahaṃ;_  
_Duggatiṃ nābhijānāmi,_  
_tiṇamuṭṭhe idaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="10" book="ap1">

<Paragraph number="6807" type="verses" book="ap1">
_Kilesā jhāpitā mayhaṃ,_  
_bhavā sabbe samūhatā;_  
_Nāgova bandhanaṃ chetvā,_  
_viharāmi anāsavo._  

</Paragraph>

</Division>
<Division number="11" book="ap1">

<Paragraph number="6808" type="verses" book="ap1">
_Svāgataṃ vata me āsi,_  
_mama buddhassa santike;_  
_Tisso vijjā anuppattā,_  
_kataṃ buddhassa sāsanaṃ._  

</Paragraph>

</Division>
<Division number="12" book="ap1">

<Paragraph number="6809" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_vimokkhāpi ca aṭṭhime;_  
_Chaḷabhiññā sacchikatā,_  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="6810" book="ap1">
Itthaṃ sudaṃ āyasmā tiṇamuṭṭhidāyako thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="6811" type="center" book="ap1">
Tiṇamuṭṭhidāyakattherassāpadānaṃ paṭhamaṃ.
</Paragraph>
