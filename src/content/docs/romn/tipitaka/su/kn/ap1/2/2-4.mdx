<Division number="35" book="ap1">

<Paragraph number="736" type="verses" book="ap1">
_“Padumuttaro nāma jino,_  
_āhutīnaṃ paṭiggaho;_  
_Gaṇamhā vūpakaṭṭho so,_  
_himavante vasī tadā._  

</Paragraph>

</Division>
<Division number="36" book="ap1">

<Paragraph number="737" type="verses" book="ap1">
_Ahampi himavantamhi,_  
_vasāmi assame tadā;_  
_Acirāgataṃ mahāvīraṃ,_  
_upesiṃ lokanāyakaṃ._  

</Paragraph>

</Division>
<Division number="37" book="ap1">

<Paragraph number="738" type="verses" book="ap1">
_Pupphacchattaṃ gahetvāna,_  
_upagacchiṃ narāsabhaṃ;_  
_Samādhiṃ samāpajjantaṃ,_  
_antarāyamakāsahaṃ._  

</Paragraph>

</Division>
<Division number="38" book="ap1">

<Paragraph number="739" type="verses" book="ap1">
_Ubho hatthehi paggayha,_  
_pupphacchattaṃ adāsahaṃ;_  
_Paṭiggahesi bhagavā,_  
_padumuttaro mahāmuni._  

</Paragraph>

</Division>
<Division number="39" book="ap1">

<Paragraph number="740" type="verses" book="ap1">
_Sabbe devā attamanā,_  
_himavantaṃ upenti te;_  
_Sādhukāraṃ pavattesuṃ,_  
_anumodissati cakkhumā._  

</Paragraph>

</Division>
<Division number="40" book="ap1">

<Paragraph number="741" type="verses" book="ap1">
_Idaṃ vatvāna te devā,_  
_upagacchuṃ naruttamaṃ;_  
_Ākāse dhārayantassa,_  
_padumacchattamuttamaṃ._  

</Paragraph>

</Division>
<Division number="41" book="ap1">

<Paragraph number="742" type="verses" book="ap1">
_Satapattachattaṃ paggayha,_  
_adāsi tāpaso mama;_  
_‘Tamahaṃ kittayissāmi,_  
_suṇātha mama bhāsato._  

</Paragraph>

</Division>
<Division number="42" book="ap1">

<Paragraph number="743" type="verses" book="ap1">
_Pañcavīsatikappāni,_  
_devarajjaṃ karissati;_  
_Catuttiṃsatikkhattuñca,_  
_cakkavattī bhavissati._  

</Paragraph>

</Division>
<Division number="43" book="ap1">

<Paragraph number="744" type="verses" book="ap1">
_Yaṃ yaṃ yoniṃ saṃsarati,_  
_devattaṃ atha mānusaṃ;_  
_Abbhokāse patiṭṭhantaṃ,_  
_padumaṃ dhārayissati._  

</Paragraph>

</Division>
<Division number="44" book="ap1">

<Paragraph number="745" type="verses" book="ap1">
_Kappasatasahassamhi,_  
_okkākakulasambhavo;_  
_Gotamo nāma gottena,_  
_satthā loke bhavissati._  

</Paragraph>

</Division>
<Division number="45" book="ap1">

<Paragraph number="746" type="verses" book="ap1">
_Pakāsite pāvacane,_  
_manussattaṃ labhissati;_  
_Manomayamhi kāyamhi,_  
_uttamo so bhavissati._  

</Paragraph>

</Division>
<Division number="46" book="ap1">

<Paragraph number="747" type="verses" book="ap1">
_Dve bhātaro bhavissanti,_  
_ubhopi panthakavhayā;_  
_Anubhotvā uttamatthaṃ,_  
_jotayissanti sāsanaṃ’._  

</Paragraph>

</Division>
<Division number="47" book="ap1">

<Paragraph number="748" type="verses" book="ap1">
_Sohaṃ aṭṭhārasavasso,_  
_pabbajiṃ anagāriyaṃ;_  
_Visesāhaṃ na vindāmi,_  
_sakyaputtassa sāsane._  

</Paragraph>

</Division>
<Division number="48" book="ap1">

<Paragraph number="749" type="verses" book="ap1">
_Dandhā mayhaṃ gatī āsi,_  
_paribhūto pure ahuṃ;_  
_Bhātā ca maṃ paṇāmesi,_  
_gaccha dāni sakaṃ gharaṃ._  

</Paragraph>

</Division>
<Division number="49" book="ap1">

<Paragraph number="750" type="verses" book="ap1">
_Sohaṃ paṇāmito santo,_  
_saṃghārāmassa koṭṭhake;_  
_Dummano tattha aṭṭhāsiṃ,_  
_sāmaññasmiṃ apekkhavā._  

</Paragraph>

</Division>
<Division number="50" book="ap1">

<Paragraph number="751" type="verses" book="ap1">
_Bhagavā tattha āgacchi,_  
_sīsaṃ mayhaṃ parāmasi;_  
_Bāhāya maṃ gahetvāna,_  
_saṃghārāmaṃ pavesayi._  

</Paragraph>

</Division>
<Division number="51" book="ap1">

<Paragraph number="752" type="verses" book="ap1">
_Anukampāya me satthā,_  
_adāsi pādapuñchaniṃ;_  
_Evaṃ suddhaṃ adhiṭṭhehi,_  
_ekamantamadhiṭṭhahaṃ._  

</Paragraph>

</Division>
<Division number="52" book="ap1">

<Paragraph number="753" type="verses" book="ap1">
_Hatthehi tamahaṃ gayha,_  
_sariṃ kokanadaṃ ahaṃ;_  
_Tattha cittaṃ vimucci me,_  
_arahattaṃ apāpuṇiṃ._  

</Paragraph>

</Division>
<Division number="53" book="ap1">

<Paragraph number="754" type="verses" book="ap1">
_Manomayesu kāyesu,_  
_sabbattha pāramiṃ gato;_  
_Sabbāsave pariññāya,_  
_viharāmi anāsavo._  

</Paragraph>

</Division>
<Division number="54" book="ap1">

<Paragraph number="755" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_vimokkhāpi ca aṭṭhime;_  
_Chaḷabhiññā sacchikatā,_  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="756" book="ap1">
Itthaṃ sudaṃ āyasmā cūḷapanthako thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="757" type="center" book="ap1">
Cūḷapanthakattherassāpadānaṃ catutthaṃ.
</Paragraph>
