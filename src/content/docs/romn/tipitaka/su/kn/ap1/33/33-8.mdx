<Division number="64" book="ap1">

<Paragraph number="3664" type="verses" book="ap1">
_“Vessabhussa bhagavato,_  
_lokajeṭṭhassa tādino;_  
_Sabbe janā samāgamma,_  
_mahāpūjaṃ karonti te._  

</Paragraph>

</Division>
<Division number="65" book="ap1">

<Paragraph number="3665" type="verses" book="ap1">
_Sudhāya piṇḍaṃ katvāna,_  
_āveḷaṃ sumanāyahaṃ;_  
_Sīhāsanassa purato,_  
_abhiropesahaṃ tadā._  

</Paragraph>

</Division>
<Division number="66" book="ap1">

<Paragraph number="3666" type="verses" book="ap1">
_Sabbe janā samāgamma,_  
_pekkhanti pupphamuttamaṃ;_  
_Kenidaṃ pūjitaṃ pupphaṃ,_  
_buddhaseṭṭhassa tādino._  

</Paragraph>

</Division>
<Division number="67" book="ap1">

<Paragraph number="3667" type="verses" book="ap1">
_Tena cittappasādena,_  
_nimmānaṃ upapajjahaṃ;_  
_Anubhomi sakaṃ kammaṃ,_  
_pubbe sukatamattano._  

</Paragraph>

</Division>
<Division number="68" book="ap1">

<Paragraph number="3668" type="verses" book="ap1">
_Yaṃ yaṃ yonupapajjāmi,_  
_devattaṃ atha mānusaṃ;_  
_Sabbesānaṃ piyo homi,_  
_pupphapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="69" book="ap1">

<Paragraph number="3669" type="verses" book="ap1">
_Nābhijānāmi kāyena,_  
_vācāya uda cetasā;_  
_Saṃyatānaṃ tapassīnaṃ,_  
_kataṃ akkositaṃ mayā._  

</Paragraph>

</Division>
<Division number="70" book="ap1">

<Paragraph number="3670" type="verses" book="ap1">
_Tena sucaritenāhaṃ,_  
_cittassa paṇidhīhi ca;_  
_Sabbesaṃ pūjito homi,_  
_anakkosassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="71" book="ap1">

<Paragraph number="3671" type="verses" book="ap1">
_Ito ekādase kappe,_  
_sahassārosi khattiyo;_  
_Sattaratanasampanno,_  
_cakkavattī mahabbalo._  

</Paragraph>

</Division>
<Division number="72" book="ap1">

<Paragraph number="3672" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="3673" book="ap1">
Itthaṃ sudaṃ āyasmā sumanāveḷiyo thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="3674" type="center" book="ap1">
Sumanāveḷiyattherassāpadānaṃ aṭṭhamaṃ.
</Paragraph>
