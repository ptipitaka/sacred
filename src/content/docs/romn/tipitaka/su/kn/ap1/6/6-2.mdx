<Division number="8" book="ap1">

<Paragraph number="1385" type="verses" book="ap1">
_“Ubbiddhaṃ selamāruyha,_  
_nisīdi purisuttamo;_  
_Pabbatassāvidūramhi,_  
_brāhmaṇo mantapāragū._  

</Paragraph>

</Division>
<Division number="9" book="ap1">

<Paragraph number="1386" type="verses" book="ap1">
_Upaviṭṭhaṃ mahāvīraṃ,_  
_devadevaṃ narāsabhaṃ;_  
_Añjaliṃ paggahetvāna,_  
_santhaviṃ lokanāyakaṃ._  

</Paragraph>

</Division>
<Division number="10" book="ap1">

<Paragraph number="1387" type="verses" book="ap1">
_‘Esa buddho mahāvīro,_  
_varadhammappakāsako;_  
_Jalati aggikhandhova,_  
_bhikkhusaṃghapurakkhato._  

</Paragraph>

</Division>
<Division number="11" book="ap1">

<Paragraph number="1388" type="verses" book="ap1">
_Mahāsamuddovakkhubbho,_  
_aṇṇavova duruttaro;_  
_Migarājāvasambhīto,_  
_dhammaṃ deseti cakkhumā’._  

</Paragraph>

</Division>
<Division number="12" book="ap1">

<Paragraph number="1389" type="verses" book="ap1">
_Mama saṅkappamaññāya,_  
_padumuttaranāyako;_  
_Bhikkhusaṃghe ṭhito satthā,_  
_imā gāthā abhāsatha._  

</Paragraph>

</Division>
<Division number="13" book="ap1">

<Paragraph number="1390" type="verses" book="ap1">
_‘Yenāyaṃ añjalī dinno,_  
_buddhaseṭṭho ca thomito;_  
_Tiṃsakappasahassāni,_  
_devarajjaṃ karissati._  

</Paragraph>

</Division>
<Division number="14" book="ap1">

<Paragraph number="1391" type="verses" book="ap1">
_Kappasatasahassamhi,_  
_aṅgīrasasanāmako;_  
_Vivaṭṭacchado sambuddho,_  
_uppajjissati tāvade._  

</Paragraph>

</Division>
<Division number="15" book="ap1">

<Paragraph number="1392" type="verses" book="ap1">
_Tassa dhammesu dāyādo,_  
_oraso dhammanimmito;_  
_Sataraṃsīti nāmena,_  
_arahā so bhavissati’._  

</Paragraph>

</Division>
<Division number="16" book="ap1">

<Paragraph number="1393" type="verses" book="ap1">
_Jātiyā sattavassohaṃ,_  
_pabbajiṃ anagāriyaṃ;_  
_Sataraṃsimhi nāmena,_  
_pabhā niddhāvate mama._  

</Paragraph>

</Division>
<Division number="17" book="ap1">

<Paragraph number="1394" type="verses" book="ap1">
_Maṇḍape rukkhamūle vā,_  
_jhāyī jhānarato ahaṃ;_  
_Dhāremi antimaṃ dehaṃ,_  
_sammāsambuddhasāsane._  

</Paragraph>

</Division>
<Division number="18" book="ap1">

<Paragraph number="1395" type="verses" book="ap1">
_Saṭṭhikappasahassamhi,_  
_caturo rāmanāmakā;_  
_Sattaratanasampannā,_  
_cakkavattī mahabbalā._  

</Paragraph>

</Division>
<Division number="19" book="ap1">

<Paragraph number="1396" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="1397" book="ap1">
Itthaṃ sudaṃ āyasmā sataraṃsi thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="1398" type="center" book="ap1">
Sataraṃsittherassāpadānaṃ dutiyaṃ.
</Paragraph>
