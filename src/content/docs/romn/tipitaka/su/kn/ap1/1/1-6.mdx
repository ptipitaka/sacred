<Paragraph number="434" type="verses" book="ap1">
_“Sumedhaṃ bhagavantāhaṃ,_  
_lokajeṭṭhaṃ narāsabhaṃ;_  
_Vūpakaṭṭhaṃ viharantaṃ,_  
_addasaṃ lokanāyakaṃ._  

</Paragraph>

</Division>
<Division number="422" book="ap1" bookNo="20" page="38">

<Paragraph number="435" type="verses" book="ap1">
_Upagantvāna sambuddhaṃ,_  
_sumedhaṃ lokanāyakaṃ;_  
_Añjaliṃ paggahetvāna,_  
_buddhaseṭṭhamayācahaṃ._  

</Paragraph>

</Division>
<Division number="423" book="ap1" bookNo="20" page="38">

<Paragraph number="436" type="verses" book="ap1">
_Anukampa mahāvīra,_  
_lokajeṭṭha narāsabha;_  
_Padīpaṃ te padassāmi,_  
_rukkhamūlamhi jhāyato._  

</Paragraph>

</Division>
<Division number="424" book="ap1" bookNo="20" page="38">

<Paragraph number="437" type="verses" book="ap1">
_Adhivāsesi so dhīro,_  
_sayambhū vadataṃ varo;_  
_Dumesu vinivijjhitvā,_  
_yantaṃ yojiyahaṃ tadā._  

</Paragraph>

</Division>
<Division number="425" book="ap1" bookNo="20" page="39">

<Paragraph number="438" type="verses" book="ap1">
_Sahassavaṭṭiṃ pādāsiṃ,_  
_buddhassa lokabandhuno;_  
_Sattāhaṃ pajjalitvāna,_  
_dīpā vūpasamiṃsu me._  

</Paragraph>

</Division>
<Division number="426" book="ap1" bookNo="20" page="39">

<Paragraph number="439" type="verses" book="ap1">
_Tena cittappasādena,_  
_cetanāpaṇidhīhi ca;_  
_Jahitvā mānusaṃ dehaṃ,_  
_vimānamupapajjahaṃ._  

</Paragraph>

</Division>
<Division number="427" book="ap1" bookNo="20" page="39">

<Paragraph number="440" type="verses" book="ap1">
_Upapannassa devattaṃ,_  
_byamhaṃ āsi sunimmitaṃ;_  
_Samantato pajjalati,_  
_dīpadānassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="428" book="ap1" bookNo="20" page="39">

<Paragraph number="441" type="verses" book="ap1">
_Samantā yojanasataṃ,_  
_virocesimahaṃ tadā;_  
_Sabbe deve abhibhomi,_  
_dīpadānassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="429" book="ap1" bookNo="20" page="39">

<Paragraph number="442" type="verses" book="ap1">
_Tiṃsakappāni devindo,_  
_devarajjamakārayiṃ;_  
_Na maṃ kecītimaññanti,_  
_dīpadānassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="430" book="ap1" bookNo="20" page="39">

<Paragraph number="443" type="verses" book="ap1">
_Aṭṭhavīsatikkhattuñca,_  
_cakkavattī ahosahaṃ;_  
_Divā rattiñca passāmi,_  
_samantā yojanaṃ tadā._  

</Paragraph>

</Division>
<Division number="431" book="ap1" bookNo="20" page="39">

<Paragraph number="444" type="verses" book="ap1">
_Sahassalokaṃ ñāṇena,_  
_passāmi satthu sāsane;_  
_Dibbacakkhumanuppatto,_  
_dīpadānassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="432" book="ap1" bookNo="20" page="39">

<Paragraph number="445" type="verses" book="ap1">
_Sumedho nāma sambuddho,_  
_tiṃsakappasahassito;_  
_Tassa dīpo mayā dinno,_  
_vippasannena cetasā._  

</Paragraph>

</Division>
<Division number="433" book="ap1" bookNo="20" page="39">

<Paragraph number="446" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_vimokkhāpi ca aṭṭhime;_  
_Chaḷabhiññā sacchikatā,_  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="447" book="ap1">
Itthaṃ sudaṃ āyasmā anuruddho thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="448" type="center" book="ap1">
Anuruddhattherassāpadānaṃ catutthaṃ.
</Paragraph>
