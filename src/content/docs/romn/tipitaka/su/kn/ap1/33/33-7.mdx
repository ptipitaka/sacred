<Division number="39" book="ap1">

<Paragraph number="3636" type="verses" book="ap1">
_“Yadā devo devakāyā,_  
_cavate āyusaṅkhayā;_  
_Tayo saddā niccharanti,_  
_devānaṃ anumodataṃ._  

</Paragraph>

</Division>
<Division number="40" book="ap1">

<Paragraph number="3637" type="verses" book="ap1">
_‘Ito bho sugatiṃ gaccha,_  
_manussānaṃ sahabyataṃ;_  
_Manussabhūto saddhamme,_  
_labha saddhaṃ anuttaraṃ._  

</Paragraph>

</Division>
<Division number="41" book="ap1">

<Paragraph number="3638" type="verses" book="ap1">
_Sā te saddhā niviṭṭhāssa,_  
_mūlajātā patiṭṭhitā;_  
_Yāvajīvaṃ asaṃhīrā,_  
_saddhamme suppavedite._  

</Paragraph>

</Division>
<Division number="42" book="ap1">

<Paragraph number="3639" type="verses" book="ap1">
_Kāyena kusalaṃ katvā,_  
_vācāya kusalaṃ bahuṃ;_  
_Manasā kusalaṃ katvā,_  
_abyāpajjaṃ nirūpadhiṃ._  

</Paragraph>

</Division>
<Division number="43" book="ap1">

<Paragraph number="3640" type="verses" book="ap1">
_Tato opadhikaṃ puññaṃ,_  
_katvā dānena taṃ bahuṃ;_  
_Aññepi macce saddhamme,_  
_brahmacariye nivesaya’._  

</Paragraph>

</Division>
<Division number="44" book="ap1">

<Paragraph number="3641" type="verses" book="ap1">
_Imāya anukampāya,_  
_devādevaṃ yadā vidū;_  
_Cavantaṃ anumodanti,_  
_ehi deva punappunaṃ._  

</Paragraph>

</Division>
<Division number="45" book="ap1">

<Paragraph number="3642" type="verses" book="ap1">
_‘Saṃvego me tadā āsi,_  
_devasaṅghe samāgate;_  
_Kaṃsu nāma ahaṃ yoniṃ,_  
_gamissāmi ito cuto’._  

</Paragraph>

</Division>
<Division number="46" book="ap1">

<Paragraph number="3643" type="verses" book="ap1">
_Mama saṃvegamaññāya,_  
_samaṇo bhāvitindriyo;_  
_Mamuddharitukāmo so,_  
_āgacchi mama santikaṃ._  

</Paragraph>

</Division>
<Division number="47" book="ap1">

<Paragraph number="3644" type="verses" book="ap1">
_Sumano nāma nāmena,_  
_padumuttarasāvako;_  
_Atthadhammānusāsitvā,_  
_saṃvejesi mamaṃ tadā._  


</Paragraph>
</Division>

<Paragraph number="3645" type="center" book="ap1">
Dvādasamaṃ bhāṇavāraṃ.


</Paragraph>

<Division number="48" book="ap1">

<Paragraph number="3646" type="verses" book="ap1">
_Tassāhaṃ vacanaṃ sutvā,_  
_buddhe cittaṃ pasādayiṃ;_  
_Taṃ dhīraṃ abhivādetvā,_  
_tattha kālaṅkato ahaṃ._  

</Paragraph>

</Division>
<Division number="49" book="ap1">

<Paragraph number="3647" type="verses" book="ap1">
_Upapajjiṃ sa tattheva,_  
_sukkamūlena codito;_  
_Vasanto mātukucchimhi,_  
_puna dhāreti mātuyā._  

</Paragraph>

</Division>
<Division number="50" book="ap1">

<Paragraph number="3648" type="verses" book="ap1">
_Tamhā kāyā cavitvāna,_  
_tidase upapajjahaṃ;_  
_Etthantare na passāmi,_  
_domanassamahaṃ tadā._  

</Paragraph>

</Division>
<Division number="51" book="ap1">

<Paragraph number="3649" type="verses" book="ap1">
_Tāvatiṃsā cavitvāna,_  
_mātukucchiṃ samokkamiṃ;_  
_Nikkhamitvāna kucchimhā,_  
_kaṇhasukkaṃ ajānahaṃ._  

</Paragraph>

</Division>
<Division number="52" book="ap1">

<Paragraph number="3650" type="verses" book="ap1">
_Jātiyā sattavassova,_  
_ārāmaṃ pāvisiṃ ahaṃ;_  
_Gotamassa bhagavato,_  
_sakyaputtassa tādino._  

</Paragraph>

</Division>
<Division number="53" book="ap1">

<Paragraph number="3651" type="verses" book="ap1">
_Vitthārike pāvacane,_  
_bāhujaññamhi sāsane;_  
_Addasaṃ sāsanakare,_  
_bhikkhavo tattha satthuno._  

</Paragraph>

</Division>
<Division number="54" book="ap1">

<Paragraph number="3652" type="verses" book="ap1">
_Sāvatthi nāma nagaraṃ,_  
_rājā tatthāsi kosalo;_  
_Rathena nāgayuttena,_  
_upesi bodhimuttamaṃ._  

</Paragraph>

</Division>
<Division number="55" book="ap1">

<Paragraph number="3653" type="verses" book="ap1">
_Tassāhaṃ nāgaṃ disvāna,_  
_pubbakammaṃ anussariṃ;_  
_Añjaliṃ paggahetvāna,_  
_samayaṃ agamāsahaṃ._  

</Paragraph>

</Division>
<Division number="56" book="ap1">

<Paragraph number="3654" type="verses" book="ap1">
_Jātiyā sattavassova,_  
_pabbajiṃ anagāriyaṃ;_  
_Yo so buddhaṃ upaṭṭhāsi,_  
_ānando nāma sāvako._  

</Paragraph>

</Division>
<Division number="57" book="ap1">

<Paragraph number="3655" type="verses" book="ap1">
_Gatimā dhitimā ceva,_  
_satimā ca bahussuto;_  
_Rañño cittaṃ pasādento,_  
_niyyādesi mahājuti._  

</Paragraph>

</Division>
<Division number="58" book="ap1">

<Paragraph number="3656" type="verses" book="ap1">
_Tassāhaṃ dhammaṃ sutvāna,_  
_pubbakammaṃ anussariṃ;_  
_Tattheva ṭhitako santo,_  
_arahattamapāpuṇiṃ._  

</Paragraph>

</Division>
<Division number="59" book="ap1">

<Paragraph number="3657" type="verses" book="ap1">
_Ekaṃsaṃ cīvaraṃ katvā,_  
_sire katvāna añjaliṃ;_  
_Sambuddhaṃ abhivādetvā,_  
_imaṃ vācaṃ udīrayiṃ._  

</Paragraph>

</Division>
<Division number="60" book="ap1">

<Paragraph number="3658" type="verses" book="ap1">
_‘Padumuttarabuddhassa,_  
_dvipadindassa satthuno;_  
_Nigguṇḍipupphaṃ paggayha,_  
_sīhāsane ṭhapesahaṃ._  

</Paragraph>

</Division>
<Division number="61" book="ap1">

<Paragraph number="3659" type="verses" book="ap1">
_Tena kammena dvipadinda,_  
_lokajeṭṭha narāsabha;_  
_Pattomhi acalaṃ ṭhānaṃ,_  
_hitvā jayaparājayaṃ’._  

</Paragraph>

</Division>
<Division number="62" book="ap1">

<Paragraph number="3660" type="verses" book="ap1">
_Pañcavīsasahassamhi,_  
_kappānaṃ manujādhipā;_  
_Abbudanirabbudāni,_  
_aṭṭhaṭṭhāsiṃsu khattiyā._  

</Paragraph>

</Division>
<Division number="63" book="ap1">

<Paragraph number="3661" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="3662" book="ap1">
Itthaṃ sudaṃ āyasmā nigguṇḍipupphiyo thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="3663" type="center" book="ap1">
Nigguṇḍipupphiyattherassāpadānaṃ sattamaṃ.
</Paragraph>
