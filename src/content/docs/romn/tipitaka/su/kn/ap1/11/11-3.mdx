<Paragraph number="1868" type="verses" book="ap1">
_“Tivarāyaṃ nivāsīhaṃ,_  
_ahosiṃ māliko tadā;_  
_Addasaṃ virajaṃ buddhaṃ,_  
_siddhatthaṃ lokapūjitaṃ._  

</Paragraph>

</Division>
<Division number="14" book="ap1">

<Paragraph number="1869" type="verses" book="ap1">
_Pasannacitto sumano,_  
_pupphahatthamadāsahaṃ;_  
_Yattha yatthupapajjāmi,_  
_tassa kammassa vāhasā._  

</Paragraph>

</Division>
<Division number="15" book="ap1">

<Paragraph number="1870" type="verses" book="ap1">
_Anubhomi phalaṃ iṭṭhaṃ,_  
_pubbe sukatamattano;_  
_Parikkhitto sumallehi,_  
_pupphadānassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="16" book="ap1">

<Paragraph number="1871" type="verses" book="ap1">
_Catunnavutito kappe,_  
_yaṃ pupphamabhiropayiṃ;_  
_Duggatiṃ nābhijānāmi,_  
_pupphapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="17" book="ap1">

<Paragraph number="1872" type="verses" book="ap1">
_Catunnavutupādāya,_  
_ṭhapetvā vattamānakaṃ;_  
_Pañcarājasatā tattha,_  
_najjasamasanāmakā._  

</Paragraph>

</Division>
<Division number="18" book="ap1">

<Paragraph number="1873" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="1874" book="ap1">
Itthaṃ sudaṃ āyasmā uppalahatthiyo thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="1875" type="center" book="ap1">
Uppalahatthiyattherassāpadānaṃ tatiyaṃ.
</Paragraph>
