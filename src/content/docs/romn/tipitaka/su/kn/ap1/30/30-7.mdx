<Division number="32" book="ap1">

_buddho antaradhāyatha._  

</Paragraph>

</Division>
<Division number="33" book="ap1">

<Paragraph number="3402" type="verses" book="ap1">

_buddhaseṭṭhamanussariṃ;_  
_Taṃ desaṃ kittayitvāna,_  
_kappaṃ saggamhi modahaṃ._  

</Paragraph>

</Division>
<Division number="34" book="ap1">

<Paragraph number="3403" type="verses" book="ap1">

_yaṃ desamabhikittayiṃ;_  
_Duggatiṃ nābhijānāmi,_  
_kittanāya idaṃ phalaṃ._  

</Paragraph>

</Division>
