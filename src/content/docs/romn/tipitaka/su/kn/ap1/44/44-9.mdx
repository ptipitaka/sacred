<Division number="72" book="ap1">

<Paragraph number="5751" type="verses" book="ap1">
_“Hatthirājā tadā āsiṃ,_  
_īsādanto uruḷhavā;_  
_Vicaranto brahāraññe,_  
_addasaṃ lokanāyakaṃ._  

</Paragraph>

</Division>
<Division number="73" book="ap1">

<Paragraph number="5752" type="verses" book="ap1">
_Ambapiṇḍaṃ gahetvāna,_  
_adāsiṃ satthuno ahaṃ;_  
_Paṭiggaṇhi mahāvīro,_  
_siddhattho lokanāyako._  

</Paragraph>

</Division>
<Division number="74" book="ap1">

<Paragraph number="5753" type="verses" book="ap1">
_Mama nijjhāyamānassa,_  
_paribhuñji tadā jino;_  
_Tattha cittaṃ pasādetvā,_  
_tusitaṃ upapajjahaṃ._  

</Paragraph>

</Division>
<Division number="75" book="ap1">

<Paragraph number="5754" type="verses" book="ap1">
_Tato ahaṃ cavitvāna,_  
_cakkavattī ahosahaṃ;_  
_Eteneva upāyena,_  
_anubhutvāna sampadā._  

</Paragraph>

</Division>
<Division number="76" book="ap1">

<Paragraph number="5755" type="verses" book="ap1">
_Padhānapahitattohaṃ,_  
_upasanto nirūpadhi;_  
_Sabbāsave pariññāya,_  
_viharāmi anāsavo._  

</Paragraph>

</Division>
<Division number="77" book="ap1">

<Paragraph number="5756" type="verses" book="ap1">
_Catunnavutito kappe,_  
_yaṃ phalamadadiṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_phaladānassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="78" book="ap1">

<Paragraph number="5757" type="verses" book="ap1">
_Kilesā jhāpitā mayhaṃ,_  
_ …pe… _  
_viharāmi anāsavo._  

</Paragraph>

</Division>
<Division number="79" book="ap1">

<Paragraph number="5758" type="verses" book="ap1">
_Svāgataṃ vata me āsi,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ._  

</Paragraph>

</Division>
<Division number="80" book="ap1">

<Paragraph number="5759" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="5760" book="ap1">
Itthaṃ sudaṃ āyasmā ambapiṇḍiyo thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="5761" type="center" book="ap1">
Ambapiṇḍiyattherassāpadānaṃ navamaṃ.
</Paragraph>
