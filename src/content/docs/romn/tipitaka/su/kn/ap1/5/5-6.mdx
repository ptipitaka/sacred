_vipassimhaggapuggale;_

<Division number="84" book="ap1">

<Paragraph number="1302" type="verses" book="ap1">
_Āropite ca citake,_  
_vehāsaṃ nabhamuggami;_  
_Ākāse chadanaṃ katvā,_  
_citakamhi adhārayi._  

</Paragraph>

</Division>
<Division number="85" book="ap1">

<Paragraph number="1303" type="verses" book="ap1">
_Ekanavutito kappe,_  
_yaṃ pupphamabhiropayiṃ;_  
_Duggatiṃ nābhijānāmi,_  
_buddhapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="86" book="ap1">

<Paragraph number="1304" type="verses" book="ap1">
_Sattatālīsito kappe,_  
_padumissaranāmako;_  
_Cāturanto vijitāvī,_  
_cakkavattī mahabbalo._  

</Paragraph>

</Division>

_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="1306" book="ap1">

</Paragraph>
