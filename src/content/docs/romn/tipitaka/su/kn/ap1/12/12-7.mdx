<Paragraph number="2009" type="verses" book="ap1">
_“Yadā vipassī lokaggo,_  
_āyusaṅkhāramossaji;_  
_Pathavī sampakampittha,_  
_medanī jalamekhalā._  

</Paragraph>

</Division>
<Division number="58" book="ap1">

<Paragraph number="2010" type="verses" book="ap1">
_Otataṃ vitthataṃ mayhaṃ,_  
_suvicittavaṭaṃsakaṃ;_  
_Bhavanampi pakampittha,_  
_buddhassa āyusaṅkhaye._  

</Paragraph>

</Division>
<Division number="59" book="ap1">

<Paragraph number="2011" type="verses" book="ap1">
_Tāso mayhaṃ samuppanno,_  
_bhavane sampakampite;_  
_Uppādo nu kimatthāya,_  
_āloko vipulo ahu._  

</Paragraph>

</Division>
<Division number="60" book="ap1">

<Paragraph number="2012" type="verses" book="ap1">
_Vessavaṇo idhāgamma,_  
_nibbāpesi mahājanaṃ;_  
_Pāṇabhūte bhayaṃ natthi,_  
_ekaggā hotha saṃvutā._  

</Paragraph>

</Division>
<Division number="61" book="ap1">

<Paragraph number="2013" type="verses" book="ap1">
_Aho buddhā aho dhammā,_  
_aho no satthu sampadā;_  
_Yasmiṃ uppajjamānamhi,_  
_pathavī sampakampati._  

</Paragraph>

</Division>
<Division number="62" book="ap1">

<Paragraph number="2014" type="verses" book="ap1">
_Buddhānubhāvaṃ kittetvā,_  
_kappaṃ saggamhi modahaṃ;_  
_Avasesesu kappesu,_  
_kusalaṃ caritaṃ mayā._  

</Paragraph>

</Division>
<Division number="63" book="ap1">

<Paragraph number="2015" type="verses" book="ap1">
_Ekanavutito kappe,_  
_yaṃ saññamalabhiṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_buddhasaññāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="64" book="ap1">

<Paragraph number="2016" type="verses" book="ap1">

_rājā āsiṃ patāpavā;_  
_Samito nāma nāmena,_  
_cakkavattī mahabbalo._  

</Paragraph>

</Division>
