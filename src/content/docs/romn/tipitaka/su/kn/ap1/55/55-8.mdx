<Paragraph number="7573" type="verses" book="ap1">
_“Imamhi bhaddake kappe,_  
_brahmabandhu mahāyaso;_  
_Kassapo nāma gottena,_  
_uppajji vadataṃ varo._  

</Paragraph>

</Division>
<Division number="226" book="ap1" bookNo="20" page="415">

<Paragraph number="7574" type="verses" book="ap1">
_Tadāhaṃ candano ceva,_  
_pabbajitvāna sāsane;_  
_Āpāṇakoṭikaṃ dhammaṃ,_  
_pūrayitvāna sāsane._  

</Paragraph>

</Division>
<Division number="227" book="ap1" bookNo="20" page="415">

<Paragraph number="7575" type="verses" book="ap1">
_Tato cutā santusitaṃ,_  
_upapannā ubho mayaṃ;_  
_Tattha dibbehi naccehi,_  
_gītehi vāditehi ca._  

</Paragraph>

</Division>
<Division number="228" book="ap1" bookNo="20" page="415">

<Paragraph number="7576" type="verses" book="ap1">
_Rūpādidasahaṅgehi,_  
_abhibhotvāna sesake;_  
_Yāvatāyuṃ vasitvāna,_  
_anubhotvā mahāsukhaṃ._  

</Paragraph>

</Division>
<Division number="229" book="ap1" bookNo="20" page="415">

<Paragraph number="7577" type="verses" book="ap1">
_Tato cavitvā tidasaṃ,_  
_candano upapajjatha;_  
_Ahaṃ kapilavatthusmiṃ,_  
_ajāyiṃ sākiyatrajo._  

</Paragraph>

</Division>
<Division number="230" book="ap1" bookNo="20" page="416">

<Paragraph number="7578" type="verses" book="ap1">
_Yadā udāyittherena,_  
_ajjhiṭṭho lokanāyako;_  
_Anukampiya sakyānaṃ,_  
_upesi kapilavhayaṃ._  

</Paragraph>

</Division>
<Division number="231" book="ap1" bookNo="20" page="416">

<Paragraph number="7579" type="verses" book="ap1">
_Tadātimānino sakyā,_  
_na buddhassa guṇaññuno;_  
_Paṇamanti na sambuddhaṃ,_  
_jātithaddhā anādarā._  

</Paragraph>

</Division>
<Division number="232" book="ap1" bookNo="20" page="416">

<Paragraph number="7580" type="verses" book="ap1">
_Tesaṃ saṅkappamaññāya,_  
_ākāse caṅkamī jino;_  
_Pajjunno viya vassittha,_  
_pajjalittha yathā sikhī._  

</Paragraph>

</Division>
<Division number="233" book="ap1" bookNo="20" page="416">

<Paragraph number="7581" type="verses" book="ap1">
_Dassetvā rūpamatulaṃ,_  
_puna antaradhāyatha;_  
_Ekopi hutvā bahudhā,_  
_ahosi punarekako._  

</Paragraph>

</Division>
<Division number="234" book="ap1" bookNo="20" page="416">

<Paragraph number="7582" type="verses" book="ap1">
_Andhakāraṃ pakāsañca,_  
_dassayitvā anekadhā;_  
_Pāṭiheraṃ karitvāna,_  
_vinayī ñātake muni._  

</Paragraph>

</Division>
<Division number="235" book="ap1" bookNo="20" page="416">

<Paragraph number="7583" type="verses" book="ap1">
_Cātuddīpo mahāmegho,_  
_tāvadeva pavassatha;_  
_Tadā hi jātakaṃ buddho,_  
_vessantaramadesayi._  

</Paragraph>

</Division>
<Division number="236" book="ap1" bookNo="20" page="416">

<Paragraph number="7584" type="verses" book="ap1">
_Tadā te khattiyā sabbe,_  
_nihantvā jātijaṃ madaṃ;_  
_Upesuṃ saraṇaṃ buddhaṃ,_  
_āha suddhodano tadā._  

</Paragraph>

</Division>
<Division number="237" book="ap1" bookNo="20" page="416">

<Paragraph number="7585" type="verses" book="ap1">
_‘Idaṃ tatiyaṃ tava bhūripañña,_  
_Pādāni vandāmi samantacakkhu;_  
_Yadābhijāto pathavī pakampayī,_  
_Yadā ca taṃ najjahi jambuchāyā’._  

</Paragraph>

</Division>
<Division number="238" book="ap1" bookNo="20" page="416">

<Paragraph number="7586" type="verses" book="ap1">
_Tadā buddhānubhāvaṃ taṃ,_  
_disvā vimhitamānaso;_  
_Pabbajitvāna tattheva,_  
_nivasiṃ mātupūjako._  

</Paragraph>

</Division>
<Division number="239" book="ap1" bookNo="20" page="416">

<Paragraph number="7587" type="verses" book="ap1">
_Candano devaputto maṃ,_  
_upagantvānupucchatha;_  
_Bhaddekarattassa tadā,_  
_saṅkhepavitthāraṃ nayaṃ._  

</Paragraph>

</Division>
<Division number="240" book="ap1" bookNo="20" page="416">

<Paragraph number="7588" type="verses" book="ap1">
_Coditohaṃ tadā tena,_  
_upecca naranāyakaṃ;_  
_Bhaddekarattaṃ sutvāna,_  
_saṃviggo vanamāmako._  

</Paragraph>

</Division>
<Division number="241" book="ap1" bookNo="20" page="417">

<Paragraph number="7589" type="verses" book="ap1">
_Tadā mātaramapucchiṃ,_  
_‘vane vacchāmi ekako’;_  
_‘Sukhumālo’ti me mātā,_  
_vārayī taṃ tadā vacaṃ._  

</Paragraph>

</Division>
<Division number="242" book="ap1" bookNo="20" page="417">

<Paragraph number="7590" type="verses" book="ap1">
_‘Kāsaṃ kusaṃ poṭakilaṃ,_  
_usīraṃ muñjapabbajaṃ;_  
_Urasā panudissāmi,_  
_vivekamanubrūhayaṃ’._  

</Paragraph>

</Division>
<Division number="243" book="ap1" bookNo="20" page="417">

<Paragraph number="7591" type="verses" book="ap1">
_Tadā vanaṃ paviṭṭhohaṃ,_  
_saritvā jinasāsanaṃ;_  
_Bhaddekarattaovādaṃ,_  
_arahattamapāpuṇiṃ._  

</Paragraph>

</Division>
<Division number="244" book="ap1" bookNo="20" page="417">

<Paragraph number="7592" type="verses" book="ap1">
_‘Atītaṃ nānvāgameyya,_  
_nappaṭikaṅkhe anāgataṃ;_  
_Yadatītaṃ pahīnaṃ taṃ,_  
_appattañca anāgataṃ._  

</Paragraph>

</Division>
<Division number="245" book="ap1" bookNo="20" page="417">

<Paragraph number="7593" type="verses" book="ap1">
_Paccuppannañca yo dhammaṃ,_  
_tattha tattha vipassati;_  
_Asaṃhīraṃ asaṃkuppaṃ,_  
_taṃ vidvā manubrūhaye._  

</Paragraph>

</Division>
<Division number="246" book="ap1" bookNo="20" page="417">

<Paragraph number="7594" type="verses" book="ap1">
_Ajjeva kiccamātappaṃ,_  
_ko jaññā maraṇaṃ suve;_  
_Na hi no saṅgaraṃ tena,_  
_mahāsenena maccunā._  

</Paragraph>

</Division>
<Division number="247" book="ap1" bookNo="20" page="417">

<Paragraph number="7595" type="verses" book="ap1">
_Evaṃvihāriṃ ātāpiṃ,_  
_ahorattamatanditaṃ;_  
_Taṃ ve bhaddekarattoti,_  
_santo ācikkhate muni’._  

</Paragraph>

</Division>

_ …pe… _  
_viharāmi anāsavo._  

</Paragraph>

</Division>

<Paragraph number="7597" type="verses" book="ap1">

_ …pe… _  
_kataṃ buddhassa sāsanaṃ._  

</Paragraph>

</Division>

_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>
