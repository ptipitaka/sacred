<Paragraph number="1765" type="verses" book="ap1">
_“Pūjārahe pūjayato,_  
_buddhe yadi va sāvake;_  
_Papañcasamatikkante,_  
_tiṇṇasokapariddave._  

</Paragraph>

</Division>
<Division number="2" book="ap1" bookNo="20" page="147">

<Paragraph number="1766" type="verses" book="ap1">
_Te tādise pūjayato,_  
_nibbute akutobhaye;_  
_Na sakkā puññaṃ saṅkhātuṃ,_  
_imettamapi kenaci._  

</Paragraph>

</Division>
<Division number="3" book="ap1" bookNo="20" page="147">

<Paragraph number="1767" type="verses" book="ap1">
_Catunnamapi dīpānaṃ,_  
_issaraṃ yodha kāraye;_  
_Ekissā pūjanāyetaṃ,_  
_kalaṃ nāgghati soḷasiṃ._  

</Paragraph>

</Division>
<Division number="4" book="ap1" bookNo="20" page="147">

<Paragraph number="1768" type="verses" book="ap1">
_Siddhatthassa naraggassa,_  
_cetiye phalitantare;_  
_Sudhāpiṇḍo mayā dinno,_  
_vippasannena cetasā._  

</Paragraph>

</Division>
<Division number="5" book="ap1" bookNo="20" page="147">

<Paragraph number="1769" type="verses" book="ap1">
_Catunnavutito kappe,_  
_yaṃ kammamakariṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_paṭisaṅkhārassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="6" book="ap1" bookNo="20" page="147">

<Paragraph number="1770" type="verses" book="ap1">
_Ito tiṃsatikappamhi,_  
_paṭisaṅkhārasavhayā;_  
_Sattaratanasampannā,_  
_terasa cakkavattino._  

</Paragraph>

</Division>
<Division number="7" book="ap1" bookNo="20" page="147">

<Paragraph number="1771" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_vimokkhāpi ca aṭṭhime;_  
_Chaḷabhiññā sacchikatā,_  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="1772" book="ap1">
Itthaṃ sudaṃ āyasmā sudhāpiṇḍiyo thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="1773" type="center" book="ap1">
Sudhāpiṇḍiyattherassāpadānaṃ paṭhamaṃ.
</Paragraph>
