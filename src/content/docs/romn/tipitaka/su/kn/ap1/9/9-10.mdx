<Division number="52" book="ap1">

<Paragraph number="1755" type="verses" book="ap1">
_Padumapupphaṃ gahetvāna,_  
_ākāse ukkhipiṃ ahaṃ;_  
_Pāpakammaṃ saritvāna,_  
_pabbajiṃ anagāriyaṃ._  

</Paragraph>

</Division>
<Division number="53" book="ap1">

<Paragraph number="1756" type="verses" book="ap1">
_Pabbajitvāna kāyena,_  
_manasā saṃvutena ca;_  
_Vacīduccaritaṃ hitvā,_  
_ājīvaṃ parisodhayiṃ._  

</Paragraph>

</Division>
<Division number="54" book="ap1">

<Paragraph number="1757" type="verses" book="ap1">

_yaṃ pupphamabhiropayiṃ;_  
_Duggatiṃ nābhijānāmi,_  
_buddhapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>

_Aṭṭhārasesu kappesu,_  
_aṭṭhatālīsamāsisuṃ._  

</Paragraph>

</Division>
