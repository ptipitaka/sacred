<Division number="31" book="ap1">

<Paragraph number="1977" type="verses" book="ap1">
_“Varuṇo nāma nāmena,_  
_devarājā ahaṃ tadā;_  
_Upaṭṭhahesiṃ sambuddhaṃ,_  
_sayoggabalavāhano._  

</Paragraph>

</Division>
<Division number="32" book="ap1">

<Paragraph number="1978" type="verses" book="ap1">
_Nibbute lokanāthamhi,_  
_atthadassīnaruttame;_  
_Tūriyaṃ sabbamādāya,_  
_agamaṃ bodhimuttamaṃ._  

</Paragraph>

</Division>
<Division number="33" book="ap1">

<Paragraph number="1979" type="verses" book="ap1">
_Vāditena ca naccena,_  
_sammatāḷasamāhito;_  
_Sammukhā viya sambuddhaṃ,_  
_upaṭṭhiṃ bodhimuttamaṃ._  

</Paragraph>

</Division>
<Division number="34" book="ap1">

<Paragraph number="1980" type="verses" book="ap1">
_Upaṭṭhahitvā taṃ bodhiṃ,_  
_dharaṇīruhapādapaṃ;_  
_Pallaṅkaṃ ābhujitvāna,_  
_tattha kālaṅkato ahaṃ._  

</Paragraph>

</Division>
<Division number="35" book="ap1">

<Paragraph number="1981" type="verses" book="ap1">
_Sakakammābhiraddhohaṃ,_  
_pasanno bodhimuttame;_  
_Tena cittappasādena,_  
_nimmānaṃ upapajjahaṃ._  

</Paragraph>

</Division>
<Division number="36" book="ap1">

<Paragraph number="1982" type="verses" book="ap1">
_Saṭṭhitūriyasahassāni,_  
_parivārenti maṃ sadā;_  
_Manussesu ca devesu,_  
_vattamānaṃ bhavābhave._  

</Paragraph>

</Division>
<Division number="37" book="ap1">

<Paragraph number="1983" type="verses" book="ap1">
_Tividhaggī nibbutā mayhaṃ,_  
_bhavā sabbe samūhatā;_  
_Dhāremi antimaṃ dehaṃ,_  
_sammāsambuddhasāsane._  

</Paragraph>

</Division>
<Division number="38" book="ap1">

<Paragraph number="1984" type="verses" book="ap1">
_Subāhū nāma nāmena,_  
_catuttiṃsāsu khattiyā;_  
_Sattaratanasampannā,_  
_pañcakappasate ito._  

</Paragraph>

</Division>
<Division number="39" book="ap1">

<Paragraph number="1985" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="1986" book="ap1">
Itthaṃ sudaṃ āyasmā ekāsaniyo thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="1987" type="center" book="ap1">
Ekāsaniyattherassāpadānaṃ catutthaṃ.
</Paragraph>
