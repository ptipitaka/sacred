<Paragraph number="793" type="verses" book="ap1">
_“Padumuttaraṃ bhagavantaṃ,_  
_lokajeṭṭhaṃ narāsabhaṃ;_  
_Pabbhāramhi nisīdantaṃ,_  
_upagacchiṃ naruttamaṃ._  

</Paragraph>

</Division>
<Division number="87" book="ap1" bookNo="20" page="68">

<Paragraph number="794" type="verses" book="ap1">
_Kaṇikārapupphaṃ disvā,_  
_vaṇṭe chetvānahaṃ tadā;_  
_Alaṅkaritvā chattamhi,_  
_buddhassa abhiropayiṃ._  

</Paragraph>

</Division>
<Division number="88" book="ap1" bookNo="20" page="68">

<Paragraph number="795" type="verses" book="ap1">
_Piṇḍapātañca pādāsiṃ,_  
_paramannaṃ subhojanaṃ;_  
_Buddhena navame tattha,_  
_samaṇe aṭṭha bhojayiṃ._  

</Paragraph>

</Division>
<Division number="89" book="ap1" bookNo="20" page="68">

<Paragraph number="796" type="verses" book="ap1">
_Anumodi mahāvīro,_  
_sayambhū aggapuggalo;_  
_Iminā chattadānena,_  
_paramannapavecchanā._  

</Paragraph>

</Division>
<Division number="90" book="ap1" bookNo="20" page="68">

<Paragraph number="797" type="verses" book="ap1">
_Tena cittappasādena,_  
_sampattimanubhossasi;_  
_Chattiṃsakkhattuṃ devindo,_  
_devarajjaṃ karissati._  

</Paragraph>

</Division>
<Division number="91" book="ap1" bookNo="20" page="68">

<Paragraph number="798" type="verses" book="ap1">
_Ekavīsatikkhattuñca,_  
_cakkavattī bhavissati;_  
_Padesarajjaṃ vipulaṃ,_  
_gaṇanāto asaṅkhiyaṃ._  

</Paragraph>

</Division>
<Division number="92" book="ap1" bookNo="20" page="68">

<Paragraph number="799" type="verses" book="ap1">
_Satasahassito kappe,_  
_okkākakulasambhavo;_  
_Gotamo nāma gottena,_  
_satthā loke bhavissati._  

</Paragraph>

</Division>
<Division number="93" book="ap1" bookNo="20" page="68">

<Paragraph number="800" type="verses" book="ap1">
_Sāsane dibbamānamhi,_  
_manussattaṃ gamissati;_  
_Tassa dhammesu dāyādo,_  
_oraso dhammanimmito._  

</Paragraph>

</Division>
<Division number="94" book="ap1" bookNo="20" page="68">

<Paragraph number="801" type="verses" book="ap1">
_Upasenoti nāmena,_  
_hessati satthu sāvako;_  
_Samantapāsādikattā,_  
_aggaṭṭhāne ṭhapessati._  

</Paragraph>

</Division>
<Division number="95" book="ap1" bookNo="20" page="68">

<Paragraph number="802" type="verses" book="ap1">
_Carimaṃ vattate mayhaṃ,_  
_bhavā sabbe samūhatā;_  
_Dhāremi antimaṃ dehaṃ,_  
_jetvā māraṃ savāhiniṃ._  

</Paragraph>

</Division>
<Division number="96" book="ap1" bookNo="20" page="68">

<Paragraph number="803" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_vimokkhāpi ca aṭṭhime;_  
_Chaḷabhiññā sacchikatā,_  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="804" book="ap1">
Itthaṃ sudaṃ āyasmā upaseno vaṅgantaputto thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="805" type="center" book="ap1">
Upasenavaṅgantaputtattherassāpadānaṃ sattamaṃ.

  
Tatiyabhāṇavāraṃ.
</Paragraph>
