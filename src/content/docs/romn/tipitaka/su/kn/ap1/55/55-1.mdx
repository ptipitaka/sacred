<Division number="1" book="ap1">

<Paragraph number="7334" type="verses" book="ap1">
_“Padumuttaro nāma jino,_  
_sabbadhammesu cakkhumā;_  
_Ito satasahassamhi,_  
_kappe uppajji nāyako._  

</Paragraph>

</Division>
<Division number="2" book="ap1">

<Paragraph number="7335" type="verses" book="ap1">
_Tadāhaṃ haṃsavatiyaṃ,_  
_seṭṭhiputto mahaddhano;_  
_Jaṅghāvihāraṃ vicaraṃ,_  
_saṃghārāmaṃ agacchahaṃ._  

</Paragraph>

</Division>
<Division number="3" book="ap1">

<Paragraph number="7336" type="verses" book="ap1">
_Tadā so lokapajjoto,_  
_dhammaṃ desesi nāyako;_  
_Mañjussarānaṃ pavaraṃ,_  
_sāvakaṃ abhikittayi._  

</Paragraph>

</Division>
<Division number="4" book="ap1">

<Paragraph number="7337" type="verses" book="ap1">
_Taṃ sutvā mudito hutvā,_  
_kāraṃ katvā mahesino;_  
_Vanditvā satthuno pāde,_  
_taṃ ṭhānamabhipatthayiṃ._  

</Paragraph>

</Division>
<Division number="5" book="ap1">

<Paragraph number="7338" type="verses" book="ap1">
_Tadā buddho viyākāsi,_  
_saṃghamajjhe vināyako;_  
_‘Anāgatamhi addhāne,_  
_lacchase taṃ manorathaṃ._  

</Paragraph>

</Division>
<Division number="6" book="ap1">

<Paragraph number="7339" type="verses" book="ap1">
_Satasahassito kappe,_  
_okkākakulasambhavo;_  
_Gotamo nāma gottena,_  
_satthā loke bhavissati._  

</Paragraph>

</Division>
<Division number="7" book="ap1">

<Paragraph number="7340" type="verses" book="ap1">
_Tassa dhammesu dāyādo,_  
_oraso dhammanimmito;_  
_Bhaddiyo nāma nāmena,_  
_hessati satthu sāvako’._  

</Paragraph>

</Division>
<Division number="8" book="ap1">

<Paragraph number="7341" type="verses" book="ap1">
_Tena kammena sukatena,_  
_cetanāpaṇidhīhi ca;_  
_Jahitvā mānusaṃ dehaṃ,_  
_tāvatiṃsamagacchahaṃ._  

</Paragraph>

</Division>
<Division number="9" book="ap1">

<Paragraph number="7342" type="verses" book="ap1">
_Dvenavute ito kappe,_  
_phusso uppajji nāyako;_  
_Durāsado duppasaho,_  
_sabbalokuttamo jino._  

</Paragraph>

</Division>
<Division number="10" book="ap1">

<Paragraph number="7343" type="verses" book="ap1">
_Caraṇena ca sampanno,_  
_brahā uju patāpavā;_  
_Hitesī sabbasattānaṃ,_  
_bahuṃ mocesi bandhanā._  

</Paragraph>

</Division>
<Division number="11" book="ap1">

<Paragraph number="7344" type="verses" book="ap1">
_Nandārāmavane tassa,_  
_ahosiṃ phussakokilo;_  
_Gandhakuṭisamāsanne,_  
_ambarukkhe vasāmahaṃ._  

</Paragraph>

</Division>
<Division number="12" book="ap1">

<Paragraph number="7345" type="verses" book="ap1">
_Tadā piṇḍāya gacchantaṃ,_  
_dakkhiṇeyyaṃ jinuttamaṃ;_  
_Disvā cittaṃ pasādetvā,_  
_mañjunābhinikūjahaṃ._  

</Paragraph>

</Division>
<Division number="13" book="ap1">

<Paragraph number="7346" type="verses" book="ap1">
_Rājuyyānaṃ tadā gantvā,_  
_supakkaṃ kanakattacaṃ;_  
_Ambapiṇḍaṃ gahetvāna,_  
_sambuddhassopanāmayiṃ._  

</Paragraph>

</Division>
<Division number="14" book="ap1">

<Paragraph number="7347" type="verses" book="ap1">
_Tadā me cittamaññāya,_  
_mahākāruṇiko jino;_  
_Upaṭṭhākassa hatthato,_  
_pattaṃ paggaṇhi nāyako._  

</Paragraph>

</Division>
<Division number="15" book="ap1">

<Paragraph number="7348" type="verses" book="ap1">
_Adāsiṃ haṭṭhacittohaṃ,_  
_ambapiṇḍaṃ mahāmune;_  
_Patte pakkhippa pakkhehi,_  
_pañjaliṃ katvāna mañjunā._  

</Paragraph>

</Division>
<Division number="16" book="ap1">

<Paragraph number="7349" type="verses" book="ap1">
_Sarena rajanīyena,_  
_savanīyena vaggunā;_  
_Vassanto buddhapūjatthaṃ,_  
_nīḷaṃ gantvā nipajjahaṃ._  

</Paragraph>

</Division>
<Division number="17" book="ap1">

<Paragraph number="7350" type="verses" book="ap1">
_Tadā muditacittaṃ maṃ,_  
_buddhapemagatāsayaṃ;_  
_Sakuṇagghi upāgantvā,_  
_ghātayī duṭṭhamānaso._  

</Paragraph>

</Division>
<Division number="18" book="ap1">

<Paragraph number="7351" type="verses" book="ap1">
_Tato cutohaṃ tusite,_  
_anubhotvā mahāsukhaṃ;_  
_Manussayonimāgacchiṃ,_  
_tassa kammassa vāhasā._  

</Paragraph>

</Division>
<Division number="19" book="ap1">

<Paragraph number="7352" type="verses" book="ap1">
_Imamhi bhaddake kappe,_  
_brahmabandhu mahāyaso;_  
_Kassapo nāma gottena,_  
_uppajji vadataṃ varo._  

</Paragraph>

</Division>
<Division number="20" book="ap1">

<Paragraph number="7353" type="verses" book="ap1">
_Sāsanaṃ jotayitvā so,_  
_abhibhuyya kutitthiye;_  
_Vinayitvāna veneyye,_  
_nibbuto so sasāvako._  

</Paragraph>

</Division>
<Division number="21" book="ap1">

<Paragraph number="7354" type="verses" book="ap1">
_Nibbute tamhi lokagge,_  
_pasannā janatā bahū;_  
_Pūjanatthāya buddhassa,_  
_thūpaṃ kubbanti satthuno._  

</Paragraph>

</Division>
<Division number="22" book="ap1">

<Paragraph number="7355" type="verses" book="ap1">
_‘Sattayojanikaṃ thūpaṃ,_  
_sattaratanabhūsitaṃ;_  
_Karissāma mahesissa’,_  
_iccevaṃ mantayanti te._  

</Paragraph>

</Division>
<Division number="23" book="ap1">

<Paragraph number="7356" type="verses" book="ap1">
_Kikino kāsirājassa,_  
_tadā senāya nāyako;_  
_Hutvāhaṃ appamāṇassa,_  
_pamāṇaṃ cetiye vadiṃ._  

</Paragraph>

</Division>
<Division number="24" book="ap1">

<Paragraph number="7357" type="verses" book="ap1">
_Tadā te mama vākyena,_  
_cetiyaṃ yojanuggataṃ;_  
_Akaṃsu naravīrassa,_  
_nānāratanabhūsitaṃ._  

</Paragraph>

</Division>
<Division number="25" book="ap1">

<Paragraph number="7358" type="verses" book="ap1">
_Tena kammena sukatena,_  
_cetanāpaṇidhīhi ca;_  
_Jahitvā mānusaṃ dehaṃ,_  
_tāvatiṃsamagacchahaṃ._  

</Paragraph>

</Division>
<Division number="26" book="ap1">

<Paragraph number="7359" type="verses" book="ap1">
_Pacchime ca bhave dāni,_  
_jāto seṭṭhikule ahaṃ;_  
_Sāvatthiyaṃ puravare,_  
_iddhe phīte mahaddhane._  

</Paragraph>

</Division>
<Division number="27" book="ap1">

<Paragraph number="7360" type="verses" book="ap1">
_Purappavese sugataṃ,_  
_disvā vimhitamānaso;_  
_Pabbajitvāna na ciraṃ,_  
_arahattamapāpuṇiṃ._  

</Paragraph>

</Division>
<Division number="28" book="ap1">

<Paragraph number="7361" type="verses" book="ap1">
_Cetiyassa pamāṇaṃ yaṃ,_  
_akariṃ tena kammunā;_  
_Lakuṇḍakasarīrohaṃ,_  
_jāto paribhavāraho._  

</Paragraph>

</Division>
<Division number="29" book="ap1">

<Paragraph number="7362" type="verses" book="ap1">
_Sarena madhurenāhaṃ,_  
_pūjitvā isisattamaṃ;_  
_Mañjussarānaṃ bhikkhūnaṃ,_  
_aggattamanupāpuṇiṃ._  

</Paragraph>

</Division>
<Division number="30" book="ap1">

<Paragraph number="7363" type="verses" book="ap1">
_Phaladānena buddhassa,_  
_guṇānussaraṇena ca;_  
_Sāmaññaphalasampanno,_  
_viharāmi anāsavo._  

</Paragraph>

</Division>
<Division number="31" book="ap1">

<Paragraph number="7364" type="verses" book="ap1">
_Kilesā jhāpitā mayhaṃ,_  
_bhavā sabbe samūhatā;_  
_Nāgova bandhanaṃ chetvā,_  
_viharāmi anāsavo._  

</Paragraph>

</Division>
<Division number="32" book="ap1">

<Paragraph number="7365" type="verses" book="ap1">
_Svāgataṃ vata me āsi,_  
_buddhaseṭṭhassa santike;_  
_Tisso vijjā anuppattā,_  
_kataṃ buddhassa sāsanaṃ._  

</Paragraph>

</Division>
<Division number="33" book="ap1">

<Paragraph number="7366" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_vimokkhāpi ca aṭṭhime;_  
_Chaḷabhiññā sacchikatā,_  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="7367" book="ap1">
Itthaṃ sudaṃ āyasmā lakuṇḍabhaddiyo thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="7368" type="center" book="ap1">
Lakuṇḍabhaddiyattherassāpadānaṃ paṭhamaṃ.
</Paragraph>
