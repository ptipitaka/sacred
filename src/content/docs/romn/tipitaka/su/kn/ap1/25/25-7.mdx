<Division number="24" book="ap1">

<Paragraph number="3069" type="verses" book="ap1">

_buddhassādiccabandhuno;_  
_Pañcavasse paricariṃ,_  
_tiṭṭhantaṃva naruttamaṃ._  

</Paragraph>

</Division>
<Division number="25" book="ap1">

<Paragraph number="3070" type="verses" book="ap1">

_yaṃ dhātuṃ pūjayiṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_dhātupaṭṭhahane phalaṃ._  

</Paragraph>

</Division>
