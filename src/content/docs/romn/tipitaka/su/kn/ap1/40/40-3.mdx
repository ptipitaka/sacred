<Paragraph number="4606" type="verses" book="ap1">
_“Kaṇikāraṃva jalitaṃ,_  
_dīparukkhaṃva ujjalaṃ;_  
_Osadhiṃva virocantaṃ,_  
_vijjutaṃ gagane yathā._  

</Paragraph>

</Division>
<Division number="305" book="ap1" bookNo="20" page="378">

<Paragraph number="4607" type="verses" book="ap1">
_Asambhītaṃ anuttāsiṃ,_  
_migarājaṃva kesariṃ;_  
_Ñāṇālokaṃ pakāsentaṃ,_  
_maddantaṃ titthiye gaṇe._  

</Paragraph>

</Division>
<Division number="306" book="ap1" bookNo="20" page="378">

<Paragraph number="4608" type="verses" book="ap1">
_Uddharantaṃ imaṃ lokaṃ,_  
_chindantaṃ sabbasaṃsayaṃ;_  
_Gajjantaṃ migarājaṃva,_  
_addasaṃ lokanāyakaṃ._  

</Paragraph>

</Division>
<Division number="307" book="ap1" bookNo="20" page="378">

<Paragraph number="4609" type="verses" book="ap1">
_Jaṭājinadharo āsiṃ,_  
_brahā uju patāpavā;_  
_Vākacīraṃ gahetvāna,_  
_pādamūle apatthariṃ._  

</Paragraph>

</Division>
<Division number="308" book="ap1" bookNo="20" page="378">

<Paragraph number="4610" type="verses" book="ap1">
_Kāḷānusāriyaṃ gayha,_  
_anulimpiṃ tathāgataṃ;_  
_Sambuddhamanulimpetvā,_  
_santhaviṃ lokanāyakaṃ._  

</Paragraph>

</Division>
<Division number="309" book="ap1" bookNo="20" page="378">

<Paragraph number="4611" type="verses" book="ap1">
_‘Samuddharasimaṃ lokaṃ,_  
_oghatiṇṇamahāmuni;_  
_Ñāṇālokena jotesi,_  
_nāvaṭaṃ ñāṇamuttamaṃ._  

</Paragraph>

</Division>
<Division number="310" book="ap1" bookNo="20" page="378">

<Paragraph number="4612" type="verses" book="ap1">
_Dhammacakkaṃ pavattesi,_  
_maddase paratitthiye;_  
_Usabho jitasaṅgāmo,_  
_sampakampesi medaniṃ._  

</Paragraph>

</Division>
<Division number="311" book="ap1" bookNo="20" page="378">

<Paragraph number="4613" type="verses" book="ap1">
_Mahāsamudde ūmiyo,_  
_velantamhi pabhijjare;_  
_Tatheva tava ñāṇamhi,_  
_sabbadiṭṭhī pabhijjare._  

</Paragraph>

</Division>
<Division number="312" book="ap1" bookNo="20" page="378">

<Paragraph number="4614" type="verses" book="ap1">
_Sukhumacchikajālena,_  
_saramhi sampatānite;_  
_Antojālikatā pāṇā,_  
_pīḷitā honti tāvade._  

</Paragraph>

</Division>
<Division number="313" book="ap1" bookNo="20" page="378">

<Paragraph number="4615" type="verses" book="ap1">
_Tatheva titthiyā loke,_  
_puthupāsaṇḍanissitā;_  
_Antoñāṇavare tuyhaṃ,_  
_parivattanti mārisa._  

</Paragraph>

</Division>
<Division number="314" book="ap1" bookNo="20" page="378">

<Paragraph number="4616" type="verses" book="ap1">
_Patiṭṭhā vuyhataṃ oghe,_  
_tvañhi nātho abandhunaṃ;_  
_Bhayaṭṭitānaṃ saraṇaṃ,_  
_muttitthīnaṃ parāyaṇaṃ._  

</Paragraph>

</Division>
<Division number="315" book="ap1" bookNo="20" page="379">

<Paragraph number="4617" type="verses" book="ap1">
_Ekavīro asadiso,_  
_mettākaruṇasañcayo;_  
_Asamo susamo santo,_  
_vasī tādī jitañjayo._  

</Paragraph>

</Division>
<Division number="316" book="ap1" bookNo="20" page="379">

<Paragraph number="4618" type="verses" book="ap1">
_Dhīro vigatasammoho,_  
_anejo akathaṃkathī;_  
_Tusito vantadososi,_  
_nimmalo saṃyato suci._  

</Paragraph>

</Division>
<Division number="317" book="ap1" bookNo="20" page="379">

<Paragraph number="4619" type="verses" book="ap1">
_Saṅgātigo hatamado,_  
_tevijjo tibhavantago;_  
_Sīmātigo dhammagaru,_  
_gatattho hitavabbhuto._  

</Paragraph>

</Division>
<Division number="318" book="ap1" bookNo="20" page="379">

<Paragraph number="4620" type="verses" book="ap1">
_Tārako tvaṃ yathā nāvā,_  
_nidhīvassāsakārako;_  
_Asambhīto yathā sīho,_  
_gajarājāva dappito’._  

</Paragraph>

</Division>
<Division number="319" book="ap1" bookNo="20" page="379">

<Paragraph number="4621" type="verses" book="ap1">
_Thometvā dasagāthāhi,_  
_padumuttaraṃ mahāyasaṃ;_  
_Vanditvā satthuno pāde,_  
_tuṇhī aṭṭhāsahaṃ tadā._  

</Paragraph>

</Division>
<Division number="320" book="ap1" bookNo="20" page="379">

<Paragraph number="4622" type="verses" book="ap1">
_Padumuttaro lokavidū,_  
_āhutīnaṃ paṭiggaho;_  
_Bhikkhusaṃghe ṭhito satthā,_  
_imā gāthā abhāsatha._  

</Paragraph>

</Division>
<Division number="321" book="ap1" bookNo="20" page="379">

<Paragraph number="4623" type="verses" book="ap1">
_‘Yo me sīlañca ñāṇañca,_  
_saddhammañcāpi vaṇṇayi;_  
_Tamahaṃ kittayissāmi,_  
_suṇātha mama bhāsato._  

</Paragraph>

</Division>
<Division number="322" book="ap1" bookNo="20" page="379">

<Paragraph number="4624" type="verses" book="ap1">
_Saṭṭhi kappasahassāni,_  
_devaloke ramissati;_  
_Aññe devebhibhavitvā,_  
_issaraṃ kārayissati._  

</Paragraph>

</Division>
<Division number="323" book="ap1" bookNo="20" page="379">

<Paragraph number="4625" type="verses" book="ap1">
_So pacchā pabbajitvāna,_  
_sukkamūlena codito;_  
_Gotamassa bhagavato,_  
_sāsane pabbajissati._  

</Paragraph>

</Division>
<Division number="324" book="ap1" bookNo="20" page="379">

<Paragraph number="4626" type="verses" book="ap1">
_Pabbajitvāna kāyena,_  
_pāpakammaṃ vivajjiya;_  
_Sabbāsave pariññāya,_  
_nibbāyissatināsavo’._  

</Paragraph>

</Division>
<Division number="325" book="ap1" bookNo="20" page="379">

<Paragraph number="4627" type="verses" book="ap1">
_Yathāpi megho thanayaṃ,_  
_tappeti mediniṃ imaṃ;_  
_Tatheva tvaṃ mahāvīra,_  
_dhammena tappayī mamaṃ._  

</Paragraph>

</Division>
<Division number="326" book="ap1" bookNo="20" page="379">

<Paragraph number="4628" type="verses" book="ap1">
_Sīlaṃ paññañca dhammañca,_  
_thavitvā lokanāyakaṃ;_  
_Pattomhi paramaṃ santiṃ,_  
_nibbānaṃ padamaccutaṃ._  

</Paragraph>

</Division>
<Division number="327" book="ap1" bookNo="20" page="380">

<Paragraph number="4629" type="verses" book="ap1">
_Aho nūna sa bhagavā,_  
_Ciraṃ tiṭṭheyya cakkhumā;_  
_Aññātañca vijāneyyuṃ,_  
_Phuseyyuṃ amataṃ padaṃ._  

</Paragraph>

</Division>
<Division number="328" book="ap1" bookNo="20" page="380">

<Paragraph number="4630" type="verses" book="ap1">
_Ayaṃ me pacchimā jāti,_  
_bhavā sabbe samūhatā;_  
_Sabbāsave pariññāya,_  
_viharāmi anāsavo._  

</Paragraph>

</Division>
<Division number="329" book="ap1" bookNo="20" page="380">

<Paragraph number="4631" type="verses" book="ap1">
_Satasahassito kappe,_  
_yaṃ buddhamabhithomayiṃ;_  
_Duggatiṃ nābhijānāmi,_  
_kittanāya idaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="330" book="ap1" bookNo="20" page="380">

<Paragraph number="4632" type="verses" book="ap1">
_Kilesā jhāpitā mayhaṃ,_  
_bhavā sabbe samūhatā;_  
_Sabbāsavā parikkhīṇā,_  
_natthi dāni punabbhavo._  

</Paragraph>

</Division>
<Division number="331" book="ap1" bookNo="20" page="380">

<Paragraph number="4633" type="verses" book="ap1">
_Svāgataṃ vata me āsi,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ._  

</Paragraph>

</Division>
<Division number="332" book="ap1" bookNo="20" page="380">

<Paragraph number="4634" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="4635" book="ap1">
Itthaṃ sudaṃ āyasmā sabbakittiko thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="4636" type="center" book="ap1">
Sabbakittikattherassāpadānaṃ tatiyaṃ.
