<Division number="45" book="ap1">

<Paragraph number="2493" type="verses" book="ap1">

_puññakamme samādayiṃ;_  
_Māḷaṃ kassāma saṃghassa,_  
_puññakkhettaṃ anuttaraṃ._  

</Paragraph>

</Division>
<Division number="46" book="ap1">

<Paragraph number="2494" type="verses" book="ap1">
_Sādhūti te paṭissutvā,_  
_mama chandavasānugā;_  
_Niṭṭhāpetvā ca taṃ māḷaṃ,_  
_vipassissa adamhase._  

</Paragraph>

</Division>
<Division number="47" book="ap1">

<Paragraph number="2495" type="verses" book="ap1">

_yaṃ māḷamadadiṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_māḷadānassidaṃ phalaṃ._  

</Paragraph>

</Division>

<Paragraph number="2496" type="verses" book="ap1">

_eko āsi janādhipo;_  
_Ādeyyo nāma nāmena,_  
_cakkavattī mahabbalo._  

</Paragraph>

</Division>
