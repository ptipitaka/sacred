_addasaṃ lokanāyakaṃ;_  
_Vaṭaṃsakaṃ gahetvāna,_  
_sovaṇṇaṃ sādhunimmitaṃ._

<Division number="35" book="ap1" bookNo="20" page="242">

<Paragraph number="2945" type="verses" book="ap1">
_Sīghaṃ tato samāruyha,_  
_hatthikkhandhagato ahaṃ;_  
_Buddhassa abhiropesiṃ,_  
_sikhino lokabandhuno._  

</Paragraph>

</Division>
<Division number="36" book="ap1" bookNo="20" page="242">

<Paragraph number="2946" type="verses" book="ap1">
_Ekattiṃse ito kappe,_  
_yaṃ pupphamabhiropayiṃ;_  
_Duggatiṃ nābhijānāmi,_  
_pupphapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="37" book="ap1" bookNo="20" page="242">

<Paragraph number="2947" type="verses" book="ap1">
_Sattavīse ito kappe,_  
_eko āsiṃ janādhipo;_  
_Mahāpatāpanāmena,_  
_cakkavattī mahabbalo._  

</Paragraph>

</Division>

_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

</Paragraph>
