_“Padumuttaro nāma jino,_  
_sabbadhammesu cakkhumā;_  
_Ito satasahassamhi,_  
_kappe uppajji nāyako._  

</Paragraph>

</Division>
<Division number="196" book="ap1">

<Paragraph number="7542" type="verses" book="ap1">
_Saraṇagamane kiñci,_  
_nivesesi tathāgato;_  
_Kiñci sīle nivesesi,_  
_dasakammapathuttame._  

</Paragraph>

</Division>
<Division number="197" book="ap1">

<Paragraph number="7543" type="verses" book="ap1">
_Deti kassaci so vīro,_  
_sāmaññaphalamuttamaṃ;_  
_Samāpattī tathā aṭṭha,_  
_tisso vijjā pavecchati._  

</Paragraph>

</Division>
<Division number="198" book="ap1">

<Paragraph number="7544" type="verses" book="ap1">
_Chaḷabhiññāsu yojesi,_  
_kiñci sattaṃ naruttamo;_  
_Deti kassaci nātho so,_  
_catasso paṭisambhidā._  

</Paragraph>

</Division>
<Division number="199" book="ap1">

<Paragraph number="7545" type="verses" book="ap1">
_Bodhaneyyaṃ pajaṃ disvā,_  
_asaṅkheyyampi yojanaṃ;_  
_Khaṇena upagantvāna,_  
_vineti narasārathi._  

</Paragraph>

</Division>
<Division number="200" book="ap1">

<Paragraph number="7546" type="verses" book="ap1">
_Tadāhaṃ haṃsavatiyaṃ,_  
_ahosiṃ brāhmaṇatrajo;_  
_Pāragū sabbavedānaṃ,_  
_veyyākaraṇasammato._  

</Paragraph>

</Division>
<Division number="201" book="ap1">

<Paragraph number="7547" type="verses" book="ap1">
_Niruttiyā ca kusalo,_  
_nighaṇḍumhi visārado;_  
_Padako keṭubhavidū,_  
_chandovicitikovido._  

</Paragraph>

</Division>
<Division number="202" book="ap1">

<Paragraph number="7548" type="verses" book="ap1">
_Jaṅghāvihāraṃ vicaraṃ,_  
_haṃsārāmamupeccahaṃ;_  
_Addasaṃ varadaṃ seṭṭhaṃ,_  
_mahājanapurakkhataṃ._  

</Paragraph>

</Division>
<Division number="203" book="ap1">

<Paragraph number="7549" type="verses" book="ap1">
_Desentaṃ virajaṃ dhammaṃ,_  
_paccanīkamatī ahaṃ;_  
_Upetvā tassa kalyāṇaṃ,_  
_sutvāna vimalaṃ ahaṃ._  

</Paragraph>

</Division>
<Division number="204" book="ap1">

<Paragraph number="7550" type="verses" book="ap1">
_Byāhataṃ punaruttaṃ vā,_  
_apatthaṃ vā niratthakaṃ;_  
_Nāddasaṃ tassa munino,_  
_tato pabbajito ahaṃ._  

</Paragraph>

</Division>
<Division number="205" book="ap1">

<Paragraph number="7551" type="verses" book="ap1">
_Nacireneva kālena,_  
_sabbasattavisārado;_  
_Nipuṇo buddhavacane,_  
_ahosiṃ guṇisammato._  

</Paragraph>

</Division>
<Division number="206" book="ap1">

<Paragraph number="7552" type="verses" book="ap1">
_Tadā catasso gāthāyo,_  
_ganthayitvā subyañjanā;_  
_Santhavitvā tilokaggaṃ,_  
_desayissaṃ dine dine._  

</Paragraph>

</Division>
<Division number="207" book="ap1">

<Paragraph number="7553" type="verses" book="ap1">
_‘Virattosi mahāvīro,_  
_saṃsāre sabhaye vasaṃ;_  
_Karuṇāya na nibbāyi,_  
_tato kāruṇiko muni._  

</Paragraph>

</Division>
<Division number="208" book="ap1">

<Paragraph number="7554" type="verses" book="ap1">
_Puthujjano vayo santo,_  
_na kilesavaso ahu;_  
_Sampajāno satiyutto,_  
_tasmā eso acintiyo._  

</Paragraph>

</Division>
<Division number="209" book="ap1">

<Paragraph number="7555" type="verses" book="ap1">
_Dubbalāni kilesāni,_  
_yassāsayagatāni me;_  
_Ñāṇaggiparidaḍḍhāni,_  
_na khīyiṃsu tamabbhutaṃ._  

</Paragraph>

</Division>
<Division number="210" book="ap1">

<Paragraph number="7556" type="verses" book="ap1">
_Yo sabbalokassa garu,_  
_loko yassa tathā garu;_  
_Tathāpi lokācariyo,_  
_loko tassānuvattako’._  

</Paragraph>

</Division>
<Division number="211" book="ap1">

<Paragraph number="7557" type="verses" book="ap1">
_Evamādīhi sambuddhaṃ,_  
_kittayaṃ dhammadesanaṃ;_  
_Yāvajīvaṃ karitvāna,_  
_gato saggaṃ tato cuto._  

</Paragraph>

</Division>
<Division number="212" book="ap1">

<Paragraph number="7558" type="verses" book="ap1">
_Satasahassito kappe,_  
_yaṃ buddhamabhikittayiṃ;_  
_Duggatiṃ nābhijānāmi,_  
_kittanāya idaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="213" book="ap1">

<Paragraph number="7559" type="verses" book="ap1">
_Devaloke mahārajjaṃ,_  
_pādesiṃ kañcanagghiyaṃ;_  
_Cakkavattī mahārajjaṃ,_  
_bahusonubhaviṃ ahaṃ._  

</Paragraph>

</Division>
<Division number="214" book="ap1">

<Paragraph number="7560" type="verses" book="ap1">
_Duve bhave pajāyāmi,_  
_devatte atha mānuse;_  
_Aññaṃ gatiṃ na jānāmi,_  
_kittanāya idaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="215" book="ap1">

<Paragraph number="7561" type="verses" book="ap1">
_Duve kule pajāyāmi,_  
_khattiye atha brāhmaṇe;_  
_Nīce kule na jāyāmi,_  
_kittanāya idaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="216" book="ap1">

<Paragraph number="7562" type="verses" book="ap1">
_Pacchime ca bhave dāni,_  
_giribbajapuruttame;_  
_Raññohaṃ bimbisārassa,_  
_putto nāmena cābhayo._  

</Paragraph>

</Division>
<Division number="217" book="ap1">

<Paragraph number="7563" type="verses" book="ap1">
_Pāpamittavasaṃ gantvā,_  
_nigaṇṭhena vimohito;_  
_Pesito nāṭaputtena,_  
_buddhaseṭṭhamupeccahaṃ._  

</Paragraph>

</Division>
<Division number="218" book="ap1">

<Paragraph number="7564" type="verses" book="ap1">
_Pucchitvā nipuṇaṃ pañhaṃ,_  
_sutvā byākaraṇuttamaṃ;_  
_Pabbajitvāna naciraṃ,_  
_arahattamapāpuṇiṃ._  

</Paragraph>

</Division>
<Division number="219" book="ap1">

<Paragraph number="7565" type="verses" book="ap1">
_Kittayitvā jinavaraṃ,_  
_kittito homi sabbadā;_  
_Sugandhadehavadano,_  
_āsiṃ sukhasamappito._  

</Paragraph>

</Division>
<Division number="220" book="ap1">

<Paragraph number="7566" type="verses" book="ap1">
_Tikkhahāsalahupañño,_  
_mahāpañño tathevahaṃ;_  
_Vicittapaṭibhāno ca,_  
_tassa kammassa vāhasā._  

</Paragraph>

</Division>
<Division number="221" book="ap1">

<Paragraph number="7567" type="verses" book="ap1">
_Abhitthavitvā padumuttarāhaṃ,_  
_Pasannacitto asamaṃ sayambhuṃ;_  
_Na gacchi kappāni apāyabhūmiṃ,_  
_Sataṃsahassāni balena tassa._  

</Paragraph>

</Division>

_ …pe… _  
_viharāmi anāsavo._  

</Paragraph>

</Division>

_ …pe… _

</Paragraph>

</Division>

_ …pe… _
