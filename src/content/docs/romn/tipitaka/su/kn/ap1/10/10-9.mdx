_“Vipassissa bhagavato,_  
_mahāpūgagaṇo ahu;_  
_Veyyāvaccakaro āsiṃ,_  
_sabbakiccesu vāvaṭo._  

</Paragraph>

<Division number="52" book="ap1" bookNo="20" page="152">

<Paragraph number="1832" type="verses" book="ap1">
_Deyyadhammo ca me natthi,_  
_sugatassa mahesino;_  
_Avandiṃ satthuno pāde,_  
_vippasannena cetasā._  

</Paragraph>

</Division>
<Division number="53" book="ap1" bookNo="20" page="152">

<Paragraph number="1833" type="verses" book="ap1">
_Ekanavutito kappe,_  
_veyyāvaccaṃ akāsahaṃ;_  
_Duggatiṃ nābhijānāmi,_  
_veyyāvaccassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="54" book="ap1" bookNo="20" page="152">

<Paragraph number="1834" type="verses" book="ap1">
_Ito ca aṭṭhame kappe,_  
_rājā āsiṃ sucintito;_  
_Sattaratanasampanno,_  
_cakkavattī mahabbalo._  

</Paragraph>

</Division>
<Division number="55" book="ap1" bookNo="20" page="152">

<Paragraph number="1835" type="verses" book="ap1">

_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="1836" book="ap1">

</Paragraph>
