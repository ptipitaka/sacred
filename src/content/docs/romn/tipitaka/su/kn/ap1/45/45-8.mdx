<Paragraph number="5838" type="verses" book="ap1">
_“Sumedho nāma sambuddho,_  
_aggo kāruṇiko muni;_  
_Tārayitvā bahū satte,_  
_nibbuto so mahāyaso._  

</Paragraph>

</Division>
<Division number="51" book="ap1">

<Paragraph number="5839" type="verses" book="ap1">
_Sīhāsanassa sāmantā,_  
_sumedhassa mahesino;_  
_Pasannacitto sumano,_  
_pādapīṭhamakārayiṃ._  

</Paragraph>

</Division>
<Division number="52" book="ap1">

<Paragraph number="5840" type="verses" book="ap1">
_Katvāna kusalaṃ kammaṃ,_  
_sukhapākaṃ sukhudrayaṃ;_  
_Puññakammena saṃyutto,_  
_tāvatiṃsamagacchahaṃ._  

</Paragraph>

</Division>
<Division number="53" book="ap1">

<Paragraph number="5841" type="verses" book="ap1">
_Tattha me vasamānassa,_  
_puññakammasamaṅgino;_  
_Padāni uddharantassa,_  
_soṇṇapīṭhā bhavanti me._  

</Paragraph>

</Division>
<Division number="54" book="ap1">

<Paragraph number="5842" type="verses" book="ap1">
_Lābhā tesaṃ suladdhaṃ vo,_  
_ye labhanti upassutiṃ;_  
_Nibbute kāraṃ katvāna,_  
_labhanti vipulaṃ sukhaṃ._  

</Paragraph>

</Division>
<Division number="55" book="ap1">

<Paragraph number="5843" type="verses" book="ap1">
_Mayāpi sukataṃ kammaṃ,_  
_vāṇijjaṃ suppayojitaṃ;_  
_Pādapīṭhaṃ karitvāna,_  
_soṇṇapīṭhaṃ labhāmahaṃ._  

</Paragraph>

</Division>
<Division number="56" book="ap1">

<Paragraph number="5844" type="verses" book="ap1">
_Yaṃ yaṃ disaṃ pakkamāmi,_  
_kenaci kiccayenahaṃ;_  
_Soṇṇapīṭhe akkamāmi,_  
_puññakammassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="57" book="ap1">

<Paragraph number="5845" type="verses" book="ap1">
_Tiṃsakappasahassamhi,_  
_yaṃ kammamakariṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_pādapīṭhassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="58" book="ap1">

<Paragraph number="5846" type="verses" book="ap1">

_ …pe… _  
_viharāmi anāsavo._  

</Paragraph>

</Division>
<Division number="59" book="ap1">

<Paragraph number="5847" type="verses" book="ap1">
_Svāgataṃ vata me āsi,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ._  

</Paragraph>

</Division>
<Division number="60" book="ap1">

<Paragraph number="5848" type="verses" book="ap1">

_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="5849" book="ap1">

</Paragraph>
</Division>
