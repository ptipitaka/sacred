_isīnaṃ tatiyo ahu;_

<Division number="30" book="ap1">

<Paragraph number="1528" type="verses" book="ap1">
_Bhisamuḷālaṃ gaṇhitvā,_  
_agamaṃ buddhasantikaṃ;_  
_Tañca buddhassa pādāsiṃ,_  
_pasanno sehi pāṇibhi._  

</Paragraph>

</Division>
<Division number="31" book="ap1">

<Paragraph number="1529" type="verses" book="ap1">
_Karena ca parāmaṭṭho,_  
_vessabhūvarabuddhinā;_  
_Sukhāhaṃ nābhijānāmi,_  
_samaṃ tena kutottariṃ._  

</Paragraph>

</Division>
<Division number="32" book="ap1">

<Paragraph number="1530" type="verses" book="ap1">
_Carimo vattate mayhaṃ,_  
_bhavā sabbe samūhatā;_  
_Hatthināgena santena,_  
_kusalaṃ ropitaṃ mayā._  

</Paragraph>

</Division>
<Division number="33" book="ap1">

<Paragraph number="1531" type="verses" book="ap1">
_Ekattiṃse ito kappe,_  
_yaṃ kammamakariṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_bhisadānassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="34" book="ap1">

<Paragraph number="1532" type="verses" book="ap1">
_Samodhānā ca rājāno,_  
_soḷasa manujādhipā;_  
_Kappamhi cuddase āsuṃ,_  
_cakkavattī mahabbalā._  

</Paragraph>

</Division>

_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="1534" book="ap1">
