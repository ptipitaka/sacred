<Division number="37" book="ap1">

<Paragraph number="5927" type="verses" book="ap1">
_“Aṅgārajātā pathavī,_  
_kukkuḷānugatā mahī;_  
_Padumuttaro bhagavā,_  
_abbhokāsamhi caṅkami._  

</Paragraph>

</Division>
<Division number="38" book="ap1">

<Paragraph number="5928" type="verses" book="ap1">
_Paṇḍaraṃ chattamādāya,_  
_addhānaṃ paṭipajjahaṃ;_  
_Tattha disvāna sambuddhaṃ,_  
_vitti me upapajjatha._  

</Paragraph>

</Division>
<Division number="39" book="ap1">

<Paragraph number="5929" type="verses" book="ap1">
_Marīciyotthaṭā bhūmi,_  
_aṅgārāva mahī ayaṃ;_  
_Upahanti mahāvātā,_  
_sarīrassāsukhepanā._  

</Paragraph>

</Division>
<Division number="40" book="ap1">

<Paragraph number="5930" type="verses" book="ap1">
_Sītaṃ uṇhaṃ vihanantaṃ,_  
_vātātapanivāraṇaṃ;_  
_Paṭiggaṇha imaṃ chattaṃ,_  
_phassayissāmi nibbutiṃ._  

</Paragraph>

</Division>
<Division number="41" book="ap1">

<Paragraph number="5931" type="verses" book="ap1">
_Anukampako kāruṇiko,_  
_padumuttaro mahāyaso;_  
_Mama saṅkappamaññāya,_  
_paṭiggaṇhi tadā jino._  

</Paragraph>

</Division>
<Division number="42" book="ap1">

<Paragraph number="5932" type="verses" book="ap1">
_Tiṃsa kappāni devindo,_  
_devarajjamakārayiṃ;_  
_Satānaṃ pañcakkhattuñca,_  
_cakkavattī ahosahaṃ._  

</Paragraph>

</Division>
<Division number="43" book="ap1">

<Paragraph number="5933" type="verses" book="ap1">
_Padesarajjaṃ vipulaṃ,_  
_gaṇanāto asaṅkhiyaṃ;_  
_Anubhomi sakaṃ kammaṃ,_  
_pubbe sukatamattano._  

</Paragraph>

</Division>
<Division number="44" book="ap1">

<Paragraph number="5934" type="verses" book="ap1">
_Ayaṃ me pacchimā jāti,_  
_carimo vattate bhavo;_  
_Ajjāpi setacchattaṃ me,_  
_sabbakālaṃ dharīyati._  

</Paragraph>

</Division>
<Division number="45" book="ap1">

<Paragraph number="5935" type="verses" book="ap1">
_Satasahassito kappe,_  
_yaṃ chattamadadiṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_chattadānassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="46" book="ap1">

<Paragraph number="5936" type="verses" book="ap1">
_Kilesā jhāpitā mayhaṃ,_  
_ …pe… _  
_viharāmi anāsavo._  

</Paragraph>

</Division>
<Division number="47" book="ap1">

<Paragraph number="5937" type="verses" book="ap1">
_Svāgataṃ vata me āsi,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ._  

</Paragraph>

</Division>
<Division number="48" book="ap1">

<Paragraph number="5938" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="5939" book="ap1">
Itthaṃ sudaṃ āyasmā ekachattiyo thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="5940" type="center" book="ap1">
Ekachattiyattherassāpadānaṃ sattamaṃ.
</Paragraph>
