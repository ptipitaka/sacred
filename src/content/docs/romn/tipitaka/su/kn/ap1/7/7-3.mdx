<Paragraph number="1507" type="verses" book="ap1">
_“Sindhuyā nadiyā tīre,_  
_cakkavāko ahaṃ tadā;_  
_Suddhasevālabhakkhohaṃ,_  
_pāpesu ca susaññato._  

</Paragraph>

</Division>
<Division number="14" book="ap1">

<Paragraph number="1508" type="verses" book="ap1">
_Addasaṃ virajaṃ buddhaṃ,_  
_gacchantaṃ anilañjase;_  
_Tuṇḍena sālaṃ paggayha,_  
_vipassissābhiropayiṃ._  

</Paragraph>

</Division>
<Division number="15" book="ap1">

<Paragraph number="1509" type="verses" book="ap1">
_Yassa saddhā tathāgate,_  
_acalā supatiṭṭhitā;_  
_Tena cittappasādena,_  
_duggatiṃ so na gacchati._  

</Paragraph>

</Division>
<Division number="16" book="ap1">

<Paragraph number="1510" type="verses" book="ap1">
_Svāgataṃ vata me āsi,_  
_buddhaseṭṭhassa santike;_  
_Vihaṅgamena santena,_  
_subījaṃ ropitaṃ mayā._  

</Paragraph>

</Division>
<Division number="17" book="ap1">

<Paragraph number="1511" type="verses" book="ap1">
_Ekanavutito kappe,_  
_yaṃ pupphamabhiropayiṃ;_  
_Duggatiṃ nābhijānāmi,_  
_buddhapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="18" book="ap1">

<Paragraph number="1512" type="verses" book="ap1">
_Sucārudassanā nāma,_  
_aṭṭhete ekanāmakā;_  
_Kappe sattarase āsuṃ,_  
_cakkavattī mahabbalā._  

</Paragraph>

</Division>
<Division number="19" book="ap1">

<Paragraph number="1513" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="1514" book="ap1">
Itthaṃ sudaṃ āyasmā paccāgamaniyo thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="1515" type="center" book="ap1">
