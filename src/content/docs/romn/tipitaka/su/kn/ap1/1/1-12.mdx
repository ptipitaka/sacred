<Paragraph number="670" type="verses" book="ap1">
_“Ārāmadvārā nikkhamma,_  
_padumuttaro mahāmuni;_  
_Vassento amataṃ vuṭṭhiṃ,_  
_nibbāpesi mahājanaṃ._  

</Paragraph>

</Division>
<Division number="645" book="ap1">

<Paragraph number="671" type="verses" book="ap1">
_Satasahassaṃ te dhīrā,_  
_chaḷabhiññā mahiddhikā;_  
_Parivārenti sambuddhaṃ,_  
_chāyāva anapāyinī._  

</Paragraph>

</Division>
<Division number="646" book="ap1">

<Paragraph number="672" type="verses" book="ap1">
_Hatthikkhandhagato āsiṃ,_  
_setacchattaṃ varuttamaṃ;_  
_Sucārurūpaṃ disvāna,_  
_vitti me udapajjatha._  

</Paragraph>

</Division>
<Division number="647" book="ap1">

<Paragraph number="673" type="verses" book="ap1">
_Oruyha hatthikhandhamhā,_  
_upagacchiṃ narāsabhaṃ;_  
_Ratanāmayachattaṃ me,_  
_buddhaseṭṭhassa dhārayiṃ._  

</Paragraph>

</Division>
<Division number="648" book="ap1">

<Paragraph number="674" type="verses" book="ap1">
_Mama saṅkappamaññāya,_  
_padumuttaro mahāisi;_  
_Taṃ kathaṃ ṭhapayitvāna,_  
_imā gāthā abhāsatha._  

</Paragraph>

</Division>
<Division number="649" book="ap1">

<Paragraph number="675" type="verses" book="ap1">
_‘Yo so chattamadhāresi,_  
_soṇṇālaṅkārabhūsitaṃ;_  
_Tamahaṃ kittayissāmi,_  
_suṇotha mama bhāsato._  

</Paragraph>

</Division>
<Division number="650" book="ap1">

<Paragraph number="676" type="verses" book="ap1">
_Ito gantvā ayaṃ poso,_  
_tusitaṃ āvasissati;_  
_Anubhossati sampattiṃ,_  
_accharāhi purakkhato._  

</Paragraph>

</Division>
<Division number="651" book="ap1">

<Paragraph number="677" type="verses" book="ap1">
_Catuttiṃsatikkhattuñca,_  
_devarajjaṃ karissati;_  
_Balādhipo aṭṭhasataṃ,_  
_vasudhaṃ āvasissati._  

</Paragraph>

</Division>
<Division number="652" book="ap1">

<Paragraph number="678" type="verses" book="ap1">
_Aṭṭhapaññāsakkhattuñca,_  
_cakkavattī bhavissati;_  
_Padesarajjaṃ vipulaṃ,_  
_mahiyā kārayissati._  

</Paragraph>

</Division>
<Division number="653" book="ap1">

<Paragraph number="679" type="verses" book="ap1">
_Kappasatasahassamhi,_  
_okkākakulasambhavo;_  
_Gotamo nāma gottena,_  
_satthā loke bhavissati._  

</Paragraph>

</Division>
<Division number="654" book="ap1">

<Paragraph number="680" type="verses" book="ap1">
_Sakyānaṃ kulaketussa,_  
_ñātibandhu bhavissati;_  
_Ānando nāma nāmena,_  
_upaṭṭhāko mahesino._  

</Paragraph>

</Division>
<Division number="655" book="ap1">

<Paragraph number="681" type="verses" book="ap1">
_Ātāpī nipako cāpi,_  
_bāhusacce sukovido;_  
_Nivātavutti atthaddho,_  
_sabbapāṭhī bhavissati._  

</Paragraph>

</Division>
<Division number="656" book="ap1">

<Paragraph number="682" type="verses" book="ap1">
_Padhānapahitatto so,_  
_upasanto nirūpadhi;_  
_Sabbāsave pariññāya,_  
_nibbāyissatināsavo._  

</Paragraph>

</Division>
<Division number="657" book="ap1">

<Paragraph number="683" type="verses" book="ap1">
_Santi āraññakā nāgā,_  
_kuñjarā saṭṭhihāyanā;_  
_Tidhāpabhinnā mātaṅgā,_  
_īsādantā urūḷhavā._  

</Paragraph>

</Division>
<Division number="658" book="ap1">

<Paragraph number="684" type="verses" book="ap1">
_Anekasatasahassā,_  
_paṇḍitāpi mahiddhikā;_  
_Sabbe te buddhanāgassa,_  
_na hontupaṇidhimhi te’._  

</Paragraph>

</Division>
<Division number="659" book="ap1">

<Paragraph number="685" type="verses" book="ap1">
_Ādiyāme namassāmi,_  
_majjhime atha pacchime;_  
_Pasannacitto sumano,_  
_buddhaseṭṭhaṃ upaṭṭhahiṃ._  

</Paragraph>

</Division>
<Division number="660" book="ap1">

<Paragraph number="686" type="verses" book="ap1">
_Ātāpī nipako cāpi,_  
_sampajāno patissato;_  
_Sotāpattiphalaṃ patto,_  
_sekhabhūmīsu kovido._  

</Paragraph>

</Division>
<Division number="661" book="ap1">

<Paragraph number="687" type="verses" book="ap1">
_Satasahassito kappe,_  
_yaṃ kammamabhinīhariṃ;_  
_Tāhaṃ bhūmimanuppatto,_  
_ṭhitā saddhammamācalā._  

</Paragraph>

</Division>
<Division number="662" book="ap1">

<Paragraph number="688" type="verses" book="ap1">
_Svāgataṃ vata me āsi,_  
_buddhaseṭṭhassa santike;_  
_Tisso vijjā anuppattā,_  
_kataṃ buddhassa sāsanaṃ._  

</Paragraph>

</Division>
<Division number="663" book="ap1">

<Paragraph number="689" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_vimokkhāpi ca aṭṭhime;_  
_Chaḷabhiññā sacchikatā,_  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="690" book="ap1">
Itthaṃ sudaṃ āyasmā ānando thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

_puṇṇatthero upāli ca._
