<Paragraph number="2673" type="verses" book="ap1">
_“Yaṃ dāyavāsiko isi,_  
_tiṇaṃ lāyati satthuno;_  
_Sabbe padakkhiṇāvaṭṭā,_  
_pathabyā nipatiṃsu te._  

</Paragraph>

</Division>
<Division number="7" book="ap1" bookNo="20" page="220">

<Paragraph number="2674" type="verses" book="ap1">
_Tamahaṃ tiṇamādāya,_  
_santhariṃ dharaṇuttame;_  
_Tīṇeva tālapattāni,_  
_āharitvānahaṃ tadā._  

</Paragraph>

</Division>
<Division number="8" book="ap1" bookNo="20" page="220">

<Paragraph number="2675" type="verses" book="ap1">
_Tiṇena chadanaṃ katvā,_  
_siddhatthassa adāsahaṃ;_  
_Sattāhaṃ dhārayuṃ tassa,_  
_devamānusasatthuno._  

</Paragraph>

</Division>
<Division number="9" book="ap1" bookNo="20" page="220">

<Paragraph number="2676" type="verses" book="ap1">
_Catunnavutito kappe,_  
_yaṃ tiṇaṃ adadiṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_tiṇadānassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="10" book="ap1" bookNo="20" page="221">

<Paragraph number="2677" type="verses" book="ap1">
_Pañcasaṭṭhimhito kappe,_  
_cattārosuṃ mahaddhanā;_  
_Sattaratanasampannā,_  
_cakkavattī mahabbalā._  

</Paragraph>

</Division>
<Division number="11" book="ap1" bookNo="20" page="221">

<Paragraph number="2678" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="2679" book="ap1">
Itthaṃ sudaṃ āyasmā tiṇasanthārako thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="2680" type="center" book="ap1">
Tiṇasanthārakattherassāpadānaṃ dutiyaṃ.
</Paragraph>
