<Division number="55" book="ap1">

<Paragraph number="7735" type="verses" book="ap1">
_“Sumedho nāma sambuddho,_  
_bāttiṃsavaralakkhaṇo;_  
_Vivekakāmo bhagavā,_  
_himavantamupāgami._  

</Paragraph>

</Division>
<Division number="56" book="ap1">

<Paragraph number="7736" type="verses" book="ap1">
_Ajjhogāhetvā himavantaṃ,_  
_aggo kāruṇiko muni;_  
_Pallaṅkaṃ ābhujitvāna,_  
_nisīdi purisuttamo._  

</Paragraph>

</Division>
<Division number="57" book="ap1">

<Paragraph number="7737" type="verses" book="ap1">
_Vijjādharo tadā āsiṃ,_  
_antalikkhacaro ahaṃ;_  
_Tisūlaṃ sukataṃ gayha,_  
_gacchāmi ambare tadā._  

</Paragraph>

</Division>
<Division number="58" book="ap1">

<Paragraph number="7738" type="verses" book="ap1">
_Pabbatagge yathā aggi,_  
_puṇṇamāyeva candimā;_  
_Vanaṃ obhāsate buddho,_  
_sālarājāva phullito._  

</Paragraph>

</Division>
<Division number="59" book="ap1">

<Paragraph number="7739" type="verses" book="ap1">
_Vanaggā nikkhamitvāna,_  
_buddharaṃsībhidhāvare;_  
_Naḷaggivaṇṇasaṅkāsā,_  
_disvā cittaṃ pasādayiṃ._  

</Paragraph>

</Division>
<Division number="60" book="ap1">

<Paragraph number="7740" type="verses" book="ap1">
_Vicinaṃ addasaṃ pupphaṃ,_  
_kaṇikāraṃ devagandhikaṃ;_  
_Tīṇi pupphāni ādāya,_  
_buddhaseṭṭhamapūjayiṃ._  

</Paragraph>

</Division>
<Division number="61" book="ap1">

<Paragraph number="7741" type="verses" book="ap1">
_Buddhassa ānubhāvena,_  
_tīṇi pupphāni me tadā;_  
_Uddhaṃ vaṇṭā adhopattā,_  
_chāyaṃ kubbanti satthuno._  

</Paragraph>

</Division>
<Division number="62" book="ap1">

<Paragraph number="7742" type="verses" book="ap1">
_Tena kammena sukatena,_  
_cetanāpaṇidhīhi ca;_  
_Jahitvā mānusaṃ dehaṃ,_  
_tāvatiṃsamagacchahaṃ._  

</Paragraph>

</Division>
<Division number="63" book="ap1">

<Paragraph number="7743" type="verses" book="ap1">
_Tattha me sukataṃ byamhaṃ,_  
_kaṇikārīti ñāyati;_  
_Saṭṭhiyojanamubbedhaṃ,_  
_tiṃsayojanavitthataṃ._  

</Paragraph>

</Division>
<Division number="64" book="ap1">

<Paragraph number="7744" type="verses" book="ap1">
_Sahassakaṇḍaṃ satabheṇḍu,_  
_dhajālu haritāmayaṃ;_  
_Satasahassaniyyūhā,_  
_byamhe pātubhaviṃsu me._  

</Paragraph>

</Division>
<Division number="65" book="ap1">

<Paragraph number="7745" type="verses" book="ap1">
_Soṇṇamayā maṇimayā,_  
_lohitaṅkamayāpi ca;_  
_Phalikāpi ca pallaṅkā,_  
_yenicchakā yadicchakā._  

</Paragraph>

</Division>
<Division number="66" book="ap1">

<Paragraph number="7746" type="verses" book="ap1">
_Mahārahañca sayanaṃ,_  
_tūlikā vikatīyutaṃ;_  
_Uddhalomiñca ekantaṃ,_  
_bimbohanasamāyutaṃ._  

</Paragraph>

</Division>
<Division number="67" book="ap1">

<Paragraph number="7747" type="verses" book="ap1">
_Bhavanā nikkhamitvāna,_  
_caranto devacārikaṃ;_  
_Yathā icchāmi gamanaṃ,_  
_devasaṅghapurakkhato._  

</Paragraph>

</Division>
<Division number="68" book="ap1">

<Paragraph number="7748" type="verses" book="ap1">
_Pupphassa heṭṭhā tiṭṭhāmi,_  
_uparicchadanaṃ mama;_  
_Samantā yojanasataṃ,_  
_kaṇikārehi chāditaṃ._  

</Paragraph>

</Division>
<Division number="69" book="ap1">

<Paragraph number="7749" type="verses" book="ap1">
_Saṭṭhituriyasahassāni,_  
_sāyapātaṃ upaṭṭhahuṃ;_  
_Parivārenti maṃ niccaṃ,_  
_rattindivamatanditā._  

</Paragraph>

</Division>
<Division number="70" book="ap1">

<Paragraph number="7750" type="verses" book="ap1">
_Tattha naccehi gītehi,_  
_tālehi vāditehi ca;_  
_Ramāmi khiḍḍā ratiyā,_  
_modāmi kāmakāmahaṃ._  

</Paragraph>

</Division>
<Division number="71" book="ap1">

<Paragraph number="7751" type="verses" book="ap1">
_Tattha bhutvā pivitvā ca,_  
_modāmi tidase tadā;_  
_Nārīgaṇehi sahito,_  
_modāmi byamhamuttame._  

</Paragraph>

</Division>
<Division number="72" book="ap1">

<Paragraph number="7752" type="verses" book="ap1">
_Satānaṃ pañcakkhattuñca,_  
_devarajjamakārayiṃ;_  
_Satānaṃ tīṇikkhattuñca,_  
_cakkavattī ahosahaṃ;_  
_Padesarajjaṃ vipulaṃ,_  
_gaṇanāto asaṅkhiyaṃ._  

</Paragraph>

</Division>
<Division number="73" book="ap1">

<Paragraph number="7753" type="verses" book="ap1">
_Bhave bhave saṃsaranto,_  
_mahābhogaṃ labhāmahaṃ;_  
_Bhoge me ūnatā natthi,_  
_buddhapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="74" book="ap1">

<Paragraph number="7754" type="verses" book="ap1">
_Duve bhave saṃsarāmi,_  
_devatte atha mānuse;_  
_Aññaṃ gatiṃ na jānāmi,_  
_buddhapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="75" book="ap1">

<Paragraph number="7755" type="verses" book="ap1">
_Duve kule pajāyāmi,_  
_khattiye cāpi brāhmaṇe;_  
_Nīce kule na jānāmi,_  
_buddhapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="76" book="ap1">

<Paragraph number="7756" type="verses" book="ap1">
_Hatthiyānaṃ assayānaṃ,_  
_sivikaṃ sandamānikaṃ;_  
_Labhāmi sabbamevetaṃ,_  
_buddhapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="77" book="ap1">

<Paragraph number="7757" type="verses" book="ap1">
_Dāsīgaṇaṃ dāsagaṇaṃ,_  
_nāriyo samalaṅkatā;_  
_Labhāmi sabbamevetaṃ,_  
_buddhapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="78" book="ap1">

<Paragraph number="7758" type="verses" book="ap1">
_Koseyyakambaliyāni,_  
_khomakappāsikāni ca;_  
_Labhāmi sabbamevetaṃ,_  
_buddhapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="79" book="ap1">

<Paragraph number="7759" type="verses" book="ap1">
_Navavatthaṃ navaphalaṃ,_  
_navaggarasabhojanaṃ;_  
_Labhāmi sabbamevetaṃ,_  
_buddhapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="80" book="ap1">

<Paragraph number="7760" type="verses" book="ap1">
_Imaṃ khāda imaṃ bhuñja,_  
_imamhi sayane saya;_  
_Labhāmi sabbamevetaṃ,_  
_buddhapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="81" book="ap1">

<Paragraph number="7761" type="verses" book="ap1">
_Sabbattha pūjito homi,_  
_yaso accuggato mama;_  
_Mahāpakkho sadā homi,_  
_abhejjapariso sadā;_  
_Ñātīnaṃ uttamo homi,_  
_buddhapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="82" book="ap1">

<Paragraph number="7762" type="verses" book="ap1">
_Sītaṃ uṇhaṃ na jānāmi,_  
_pariḷāho na vijjati;_  
_Atho cetasikaṃ dukkhaṃ,_  
_hadaye me na vijjati._  

</Paragraph>

</Division>
<Division number="83" book="ap1">

<Paragraph number="7763" type="verses" book="ap1">
_Suvaṇṇavaṇṇo hutvāna,_  
_saṃsarāmi bhavābhave;_  
_Vevaṇṇiyaṃ na jānāmi,_  
_buddhapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="84" book="ap1">

<Paragraph number="7764" type="verses" book="ap1">
_Devalokā cavitvāna,_  
_sukkamūlena codito;_  
_Sāvatthiyaṃ pure jāto,_  
_mahāsālesu aḍḍhake._  

</Paragraph>

</Division>
<Division number="85" book="ap1">

<Paragraph number="7765" type="verses" book="ap1">
_Pañca kāmaguṇe hitvā,_  
_pabbajiṃ anagāriyaṃ;_  
_Jātiyā sattavassohaṃ,_  
_arahattamapāpuṇiṃ._  

</Paragraph>

</Division>
<Division number="86" book="ap1">

<Paragraph number="7766" type="verses" book="ap1">
_Upasampadāyī buddho,_  
_guṇamaññāya cakkhumā;_  
_Taruṇo pūjanīyohaṃ,_  
_buddhapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="87" book="ap1">

<Paragraph number="7767" type="verses" book="ap1">
_Dibbacakkhuvisuddhaṃ me,_  
_samādhikusalo ahaṃ;_  
_Abhiññāpāramippatto,_  
_buddhapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="88" book="ap1">

<Paragraph number="7768" type="verses" book="ap1">
_Paṭisambhidā anuppatto,_  
_iddhipādesu kovido;_  
_Dhammesu pāramippatto,_  
_buddhapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="89" book="ap1">

<Paragraph number="7769" type="verses" book="ap1">
_Tiṃsakappasahassamhi,_  
_yaṃ buddhamabhipūjayiṃ;_  
_Duggatiṃ nābhijānāmi,_  
_buddhapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
