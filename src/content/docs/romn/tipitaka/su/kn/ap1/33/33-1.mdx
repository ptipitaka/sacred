<Division number="1" book="ap1">

<Paragraph number="3586" type="verses" book="ap1">
_“Samāhitaṃ samāpannaṃ,_  
_siddhatthamaparājitaṃ;_  
_Samādhinā upaviṭṭhaṃ,_  
_addasāhaṃ naruttamaṃ._  

</Paragraph>

</Division>
<Division number="2" book="ap1">

<Paragraph number="3587" type="verses" book="ap1">
_Umāpupphaṃ gahetvāna,_  
_buddhassa abhiropayiṃ;_  
_Sabbapupphā ekasīsā,_  
_uddhaṃvaṇṭā adhomukhā._  

</Paragraph>

</Division>
<Division number="3" book="ap1">

<Paragraph number="3588" type="verses" book="ap1">
_Sucittā viya tiṭṭhante,_  
_ākāse pupphasantharā;_  
_Tena cittappasādena,_  
_tusitaṃ upapajjahaṃ._  

</Paragraph>

</Division>
<Division number="4" book="ap1">

<Paragraph number="3589" type="verses" book="ap1">
_Catunnavutito kappe,_  
_yaṃ pupphamabhipūjayiṃ;_  
_Duggatiṃ nābhijānāmi,_  
_buddhapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="5" book="ap1">

<Paragraph number="3590" type="verses" book="ap1">
_Pañcapaññāsito kappe,_  
_eko āsiṃ mahīpati;_  
_Samantachadano nāma,_  
_cakkavattī mahabbalo._  

</Paragraph>

</Division>
<Division number="6" book="ap1">

<Paragraph number="3591" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_vimokkhāpi ca aṭṭhime;_  
_Chaḷabhiññā sacchikatā,_  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="3592" book="ap1">
Itthaṃ sudaṃ āyasmā umāpupphiyo thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="3593" type="center" book="ap1">
Umāpupphiyattherassāpadānaṃ paṭhamaṃ.
</Paragraph>
