<Paragraph number="7601" type="verses" book="ap1">
_“Imamhi bhaddake kappe,_  
_brahmabandhu mahāyaso;_  
_Kassapo nāma gottena,_  
_uppajji vadataṃ varo._  

</Paragraph>

</Division>
<Division number="252" book="ap1" bookNo="20" page="417">

<Paragraph number="7602" type="verses" book="ap1">
_Tadāhaṃ pabbajitvāna,_  
_tassa buddhassa sāsane;_  
_Yāvajīvaṃ caritvāna,_  
_brahmacāraṃ tato cuto._  

</Paragraph>

</Division>
<Division number="253" book="ap1" bookNo="20" page="418">

<Paragraph number="7603" type="verses" book="ap1">
_Tena kammena sukatena,_  
_cetanāpaṇidhīhi ca;_  
_Jahitvā mānusaṃ dehaṃ,_  
_tāvatiṃsamagacchahaṃ._  

</Paragraph>

</Division>
<Division number="254" book="ap1" bookNo="20" page="418">

<Paragraph number="7604" type="verses" book="ap1">
_Tato cuto araññamhi,_  
_kapoto āsahaṃ tahiṃ;_  
_Vasate guṇasampanno,_  
_bhikkhu jhānarato sadā._  

</Paragraph>

</Division>
<Division number="255" book="ap1" bookNo="20" page="418">

<Paragraph number="7605" type="verses" book="ap1">
_Mettacitto kāruṇiko,_  
_sadā pamuditānano;_  
_Upekkhako mahāvīro,_  
_appamaññāsu kovido._  

</Paragraph>

</Division>
<Division number="256" book="ap1" bookNo="20" page="418">

<Paragraph number="7606" type="verses" book="ap1">
_Vinīvaraṇasaṅkappe,_  
_sabbasattahitāsaye;_  
_Visaṭṭho nacirenāsiṃ,_  
_tasmiṃ sugatasāvake._  

</Paragraph>

</Division>
<Division number="257" book="ap1" bookNo="20" page="418">

<Paragraph number="7607" type="verses" book="ap1">
_Upecca pādamūlamhi,_  
_nisinnassa tadāssame;_  
_Kadāci sāmisaṃ deti,_  
_dhammaṃ desesi cekadā._  

</Paragraph>

</Division>
<Division number="258" book="ap1" bookNo="20" page="418">

<Paragraph number="7608" type="verses" book="ap1">
_Tadā vipulapemena,_  
_upāsitvā jinatrajaṃ;_  
_Tato cuto gato saggaṃ,_  
_pavāso sagharaṃ yathā._  

</Paragraph>

</Division>
<Division number="259" book="ap1" bookNo="20" page="418">

<Paragraph number="7609" type="verses" book="ap1">
_Saggā cuto manussesu,_  
_nibbatto puññakammunā;_  
_Agāraṃ chaḍḍayitvāna,_  
_pabbajiṃ bahuso ahaṃ._  

</Paragraph>

</Division>
<Division number="260" book="ap1" bookNo="20" page="418">

<Paragraph number="7610" type="verses" book="ap1">
_Samaṇo tāpaso vippo,_  
_paribbajo tathevahaṃ;_  
_Hutvā vasiṃ araññamhi,_  
_anekasataso ahaṃ._  

</Paragraph>

</Division>
<Division number="261" book="ap1" bookNo="20" page="418">

<Paragraph number="7611" type="verses" book="ap1">
_Pacchime ca bhave dāni,_  
_ramme kapilavatthave;_  
_Vacchagotto dijo tassa,_  
_jāyāya ahamokkamiṃ._  

</Paragraph>

</Division>
<Division number="262" book="ap1" bookNo="20" page="418">

<Paragraph number="7612" type="verses" book="ap1">
_Mātu me dohaḷo āsi,_  
_tirokucchigatassa me;_  
_Jāyamānasamīpamhi,_  
_vanavāsāya nicchayo._  

</Paragraph>

</Division>
<Division number="263" book="ap1" bookNo="20" page="418">

<Paragraph number="7613" type="verses" book="ap1">
_Tato me ajanī mātā,_  
_ramaṇīye vanantare;_  
_Gabbhato nikkhamantaṃ maṃ,_  
_kāsāyena paṭiggahuṃ._  

</Paragraph>

</Division>
<Division number="264" book="ap1" bookNo="20" page="418">

<Paragraph number="7614" type="verses" book="ap1">
_Tato kumāro siddhattho,_  
_jāto sakyakuladdhajo;_  
_Tassa mitto piyo āsiṃ,_  
_saṃvisaṭṭho sumāniyo._  

</Paragraph>

</Division>
<Division number="265" book="ap1" bookNo="20" page="418">

<Paragraph number="7615" type="verses" book="ap1">
_Sattasārebhinikkhante,_  
_ohāya vipulaṃ yasaṃ;_  
_Ahampi pabbajitvāna,_  
_himavantamupāgamiṃ._  

</Paragraph>

</Division>
<Division number="266" book="ap1" bookNo="20" page="418">

<Paragraph number="7616" type="verses" book="ap1">
_Vanālayaṃ bhāvanīyaṃ,_  
_kassapaṃ dhutavādikaṃ;_  
_Disvā sutvā jinuppādaṃ,_  
_upesiṃ narasārathiṃ._  

</Paragraph>

</Division>
<Division number="267" book="ap1" bookNo="20" page="419">

<Paragraph number="7617" type="verses" book="ap1">
_So me dhammamadesesi,_  
_sabbatthaṃ sampakāsayaṃ;_  
_Tatohaṃ pabbajitvāna,_  
_vanameva punāgamaṃ._  

</Paragraph>

</Division>
<Division number="268" book="ap1" bookNo="20" page="419">

<Paragraph number="7618" type="verses" book="ap1">
_Tatthāppamatto viharaṃ,_  
_chaḷabhiññā aphassayiṃ;_  
_Aho suladdhalābhomhi,_  
_sumittenānukampito._  

</Paragraph>

</Division>
<Division number="269" book="ap1" bookNo="20" page="419">

<Paragraph number="7619" type="verses" book="ap1">

_ …pe… _  
_viharāmi anāsavo._  

</Paragraph>

</Division>
<Division number="270" book="ap1" bookNo="20" page="419">

<Paragraph number="7620" type="verses" book="ap1">
_Svāgataṃ vata me āsi,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ._  

</Paragraph>

</Division>
<Division number="271" book="ap1" bookNo="20" page="419">

_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>
