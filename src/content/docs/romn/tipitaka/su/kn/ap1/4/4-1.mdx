<Paragraph number="1073" type="verses" book="ap1">
_“Sattāhaṃ paṭisallīnaṃ,_  
_sayambhuṃ aggapuggalaṃ;_  
_Pasannacitto sumano,_  
_buddhaseṭṭhaṃ upaṭṭhahiṃ._  

</Paragraph>

</Division>
<Division number="2" book="ap1" bookNo="20" page="91">

<Paragraph number="1074" type="verses" book="ap1">
_Vuṭṭhitaṃ kālamaññāya,_  
_padumuttaraṃ mahāmuniṃ;_  
_Mahantiṃ kadalīkaṇṇiṃ,_  
_gahetvā upagacchahaṃ._  

</Paragraph>

</Division>
<Division number="3" book="ap1" bookNo="20" page="91">

<Paragraph number="1075" type="verses" book="ap1">
_Paṭiggahetvā bhagavā,_  
_sabbaññū lokanāyako;_  
_Mama cittaṃ pasādento,_  
_paribhuñji mahāmuni._  

</Paragraph>

</Division>
<Division number="4" book="ap1" bookNo="20" page="91">

<Paragraph number="1076" type="verses" book="ap1">
_Paribhuñjitvā sambuddho,_  
_satthavāho anuttaro;_  
_Sakāsane nisīditvā,_  
_imā gāthā abhāsatha._  

</Paragraph>

</Division>
<Division number="5" book="ap1" bookNo="20" page="91">

<Paragraph number="1077" type="verses" book="ap1">
_‘Ye ca santi samitāro,_  
_yakkhā imamhi pabbate;_  
_Araññe bhūtabhabyāni,_  
_suṇantu vacanaṃ mama._  

</Paragraph>

</Division>
<Division number="6" book="ap1" bookNo="20" page="91">

<Paragraph number="1078" type="verses" book="ap1">
_Yo so buddhaṃ upaṭṭhāsi,_  
_migarājaṃva kesariṃ;_  
_Tamahaṃ kittayissāmi,_  
_suṇātha mama bhāsato._  

</Paragraph>

</Division>
<Division number="7" book="ap1" bookNo="20" page="91">

<Paragraph number="1079" type="verses" book="ap1">
_Ekādasañcakkhattuṃ so,_  
_devarājā bhavissati;_  
_Catutiṃsatikkhattuñca,_  
_cakkavattī bhavissati._  

</Paragraph>

</Division>
<Division number="8" book="ap1" bookNo="20" page="91">

<Paragraph number="1080" type="verses" book="ap1">
_Kappasatasahassamhi,_  
_okkākakulasambhavo;_  
_Gotamo nāma gottena,_  
_satthā loke bhavissati._  

</Paragraph>

</Division>
<Division number="9" book="ap1" bookNo="20" page="91">

<Paragraph number="1081" type="verses" book="ap1">
_Akkositvāna samaṇe,_  
_sīlavante anāsave;_  
_Pāpakammavipākena,_  
_nāmadheyyaṃ labhissati._  

</Paragraph>

</Division>
<Division number="10" book="ap1" bookNo="20" page="91">

<Paragraph number="1082" type="verses" book="ap1">
_Tassa dhamme sudāyādo,_  
_oraso dhammanimmito;_  
_Kuṇḍadhānoti nāmena,_  
_sāvako so bhavissati’._  

</Paragraph>

</Division>
<Division number="11" book="ap1" bookNo="20" page="91">

<Paragraph number="1083" type="verses" book="ap1">
_Pavivekamanuyutto,_  
_jhāyī jhānarato ahaṃ;_  
_Tosayitvāna satthāraṃ,_  
_viharāmi anāsavo._  

</Paragraph>

</Division>
<Division number="12" book="ap1" bookNo="20" page="92">

<Paragraph number="1084" type="verses" book="ap1">
_Sāvakehi parivuto,_  
_bhikkhusaṃghapurakkhato;_  
_Bhikkhusaṃghe nisīditvā,_  
_salākaṃ gāhayī jino._  

</Paragraph>

</Division>
<Division number="13" book="ap1" bookNo="20" page="92">

<Paragraph number="1085" type="verses" book="ap1">
_Ekaṃsaṃ cīvaraṃ katvā,_  
_vanditvā lokanāyakaṃ;_  
_Vadataṃ varassa purato,_  
_paṭhamaṃ aggahesahaṃ._  

</Paragraph>

</Division>
<Division number="14" book="ap1" bookNo="20" page="92">

<Paragraph number="1086" type="verses" book="ap1">
_Tena kammena bhagavā,_  
_dasasahassikampako;_  
_Bhikkhusaṃghe nisīditvā,_  
_aggaṭṭhāne ṭhapesi maṃ._  

</Paragraph>

</Division>
<Division number="15" book="ap1" bookNo="20" page="92">

<Paragraph number="1087" type="verses" book="ap1">
_Vīriyaṃ me dhuradhorayhaṃ,_  
_yogakkhemādhivāhanaṃ;_  
_Dhāremi antimaṃ dehaṃ,_  
_sammāsambuddhasāsane._  

</Paragraph>

</Division>
<Division number="16" book="ap1" bookNo="20" page="92">

<Paragraph number="1088" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_vimokkhāpi ca aṭṭhime;_  
_Chaḷabhiññā sacchikatā,_  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="1089" book="ap1">
Itthaṃ sudaṃ āyasmā kuṇḍadhāno thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="1090" type="center" book="ap1">
Kuṇḍadhānattherassāpadānaṃ paṭhamaṃ.
</Paragraph>
