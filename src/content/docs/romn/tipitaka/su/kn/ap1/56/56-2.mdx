<Paragraph number="7701" type="verses" book="ap1">
_“Padumuttarassa bhagavato,_  
_Lokajeṭṭhassa tādino;_  
_Piṇḍacāraṃ carantassa,_  
_Vārato uttamaṃ yasaṃ;_  
_Aggaphalaṃ gahetvāna,_  
_Adāsiṃ satthuno ahaṃ._  

</Paragraph>

</Division>
<Division number="30" book="ap1" bookNo="20" page="431">

<Paragraph number="7702" type="verses" book="ap1">
_Tena kammena devindo,_  
_lokajeṭṭho narāsabho;_  
_Sampattomhi acalaṃ ṭhānaṃ,_  
_hitvā jayaparājayaṃ._  

</Paragraph>

</Division>
<Division number="31" book="ap1" bookNo="20" page="431">

<Paragraph number="7703" type="verses" book="ap1">

_yaṃ phalaṃ adadiṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_aggadānassidaṃ phalaṃ._  

</Paragraph>

</Division>
