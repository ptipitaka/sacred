<Paragraph number="5017" type="verses" book="ap1">
_“Migaluddo pure āsiṃ,_  
_Araññe kānane ahaṃ;_  
_Pasadaṃ migamesanto,_  
_Sayambhuṃ addasaṃ ahaṃ._  

</Paragraph>

</Division>
<Division number="162" book="ap1" bookNo="20" page="410">

<Paragraph number="5018" type="verses" book="ap1">
_Anuruddho nāma sambuddho,_  
_sayambhū aparājito;_  
_Vivekakāmo so dhīro,_  
_vanamajjhogahī tadā._  

</Paragraph>

</Division>
<Division number="163" book="ap1" bookNo="20" page="410">

<Paragraph number="5019" type="verses" book="ap1">
_Catudaṇḍe gahetvāna,_  
_catuṭṭhāne ṭhapesahaṃ;_  
_Maṇḍapaṃ sukataṃ katvā,_  
_padmapupphehi chādayiṃ._  

</Paragraph>

</Division>
<Division number="164" book="ap1" bookNo="20" page="410">

<Paragraph number="5020" type="verses" book="ap1">
_Maṇḍapaṃ chādayitvāna,_  
_sayambhuṃ abhivādayiṃ;_  
_Dhanuṃ tattheva nikkhippa,_  
_pabbajiṃ anagāriyaṃ._  

</Paragraph>

</Division>
<Division number="165" book="ap1" bookNo="20" page="410">

<Paragraph number="5021" type="verses" book="ap1">
_Naciraṃ pabbajitassa,_  
_byādhi me udapajjatha;_  
_Pubbakammaṃ saritvāna,_  
_tattha kālaṅkato ahaṃ._  

</Paragraph>

</Division>
<Division number="166" book="ap1" bookNo="20" page="410">

<Paragraph number="5022" type="verses" book="ap1">
_Pubbakammena saṃyutto,_  
_tusitaṃ agamāsahaṃ;_  
_Tattha soṇṇamayaṃ byamhaṃ,_  
_nibbattati yadicchakaṃ._  

</Paragraph>

</Division>
<Division number="167" book="ap1" bookNo="20" page="410">

<Paragraph number="5023" type="verses" book="ap1">
_Sahassayuttaṃ hayavāhiṃ,_  
_dibbayānamadhiṭṭhito;_  
_Āruhitvāna taṃ yānaṃ,_  
_gacchāmahaṃ yadicchakaṃ._  

</Paragraph>

</Division>
<Division number="168" book="ap1" bookNo="20" page="411">

<Paragraph number="5024" type="verses" book="ap1">
_Tato me niyyamānassa,_  
_devabhūtassa me sato;_  
_Samantā yojanasataṃ,_  
_maṇḍapo me dharīyati._  

</Paragraph>

</Division>
<Division number="169" book="ap1" bookNo="20" page="411">

<Paragraph number="5025" type="verses" book="ap1">
_Sayanehaṃ tuvaṭṭāmi,_  
_acchanne pupphasanthate;_  
_Antalikkhā ca padumā,_  
_vassante niccakālikaṃ._  

</Paragraph>

</Division>
<Division number="170" book="ap1" bookNo="20" page="411">

<Paragraph number="5026" type="verses" book="ap1">
_Marīcike phandamāne,_  
_tappamāne ca ātape;_  
_Na maṃ tāpeti ātāpo,_  
_maṇḍapassa idaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="171" book="ap1" bookNo="20" page="411">

<Paragraph number="5027" type="verses" book="ap1">
_Duggatiṃ samatikkanto,_  
_apāyā pihitā mama;_  
_Maṇḍape rukkhamūle vā,_  
_santāpo me na vijjati._  

</Paragraph>

</Division>
<Division number="172" book="ap1" bookNo="20" page="411">

<Paragraph number="5028" type="verses" book="ap1">
_Mahīsaññaṃ adhiṭṭhāya,_  
_loṇatoyaṃ tarāmahaṃ;_  
_Tassa me sukataṃ kammaṃ,_  
_buddhapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="173" book="ap1" bookNo="20" page="411">

<Paragraph number="5029" type="verses" book="ap1">
_Apathampi pathaṃ katvā,_  
_gacchāmi anilañjase;_  
_Aho me sukataṃ kammaṃ,_  
_buddhapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="174" book="ap1" bookNo="20" page="411">

<Paragraph number="5030" type="verses" book="ap1">
_Pubbenivāsaṃ jānāmi,_  
_dibbacakkhu visodhitaṃ;_  
_Āsavā me parikkhīṇā,_  
_buddhapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="175" book="ap1" bookNo="20" page="411">

<Paragraph number="5031" type="verses" book="ap1">
_Jahitā purimā jāti,_  
_buddhassa oraso ahaṃ;_  
_Dāyādomhi ca saddhamme,_  
_buddhapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="176" book="ap1" bookNo="20" page="411">

<Paragraph number="5032" type="verses" book="ap1">
_Ārādhitomhi sugataṃ,_  
_Gotamaṃ sakyapuṅgavaṃ;_  
_Dhammadhajo dhammadāyādo,_  
_Buddhapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="177" book="ap1" bookNo="20" page="411">

<Paragraph number="5033" type="verses" book="ap1">
_Upaṭṭhitvāna sambuddhaṃ,_  
_gotamaṃ sakyapuṅgavaṃ;_  
_Pāraṅgamaniyaṃ maggaṃ,_  
_apucchiṃ lokanāyakaṃ._  

</Paragraph>

</Division>
<Division number="178" book="ap1" bookNo="20" page="411">

<Paragraph number="5034" type="verses" book="ap1">
_Ajjhiṭṭho kathayī buddho,_  
_gambhīraṃ nipuṇaṃ padaṃ;_  
_Tassāhaṃ dhammaṃ sutvāna,_  
_pattomhi āsavakkhayaṃ._  

</Paragraph>

</Division>
<Division number="179" book="ap1" bookNo="20" page="411">

<Paragraph number="5035" type="verses" book="ap1">
_Aho me sukataṃ kammaṃ,_  
_parimuttomhi jātiyā;_  
_Sabbāsavaparikkhīṇo,_  
_natthi dāni punabbhavo._  

</Paragraph>

</Division>

_ …pe… _  
_viharāmi anāsavo._  

</Paragraph>
