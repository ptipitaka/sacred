<Division number="18" book="ap1">

<Paragraph number="2456" type="verses" book="ap1">
_“Atthadassī tu bhagavā,_  
_lokajeṭṭho narāsabho;_  
_Abbhuggantvāna vehāsaṃ,_  
_gacchate anilañjase._  

</Paragraph>

</Division>
<Division number="19" book="ap1">

<Paragraph number="2457" type="verses" book="ap1">
_Yamhi dese ṭhito satthā,_  
_abbhuggacchi mahāmuni;_  
_Tāhaṃ desaṃ apūjesiṃ,_  
_pasanno sehi pāṇibhi._  

</Paragraph>

</Division>
<Division number="20" book="ap1">

<Paragraph number="2458" type="verses" book="ap1">
_Aṭṭhārase kappasate,_  
_addasaṃ yaṃ mahāmuniṃ;_  
_Duggatiṃ nābhijānāmi,_  
_desapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="21" book="ap1">

<Paragraph number="2459" type="verses" book="ap1">
_Ekādase kappasate,_  
_gosujātasanāmako;_  
_Sattaratanasampanno,_  
_cakkavattī mahabbalo._  

</Paragraph>

</Division>
<Division number="22" book="ap1">

<Paragraph number="2460" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="2461" book="ap1">
Itthaṃ sudaṃ āyasmā desapūjako thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="2462" type="center" book="ap1">
Desapūjakattherassāpadānaṃ catutthaṃ.
</Paragraph>
