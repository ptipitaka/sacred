<Division number="29" book="ap1">

<Paragraph number="3136" type="verses" book="ap1">
_Pātarāse anuppatte,_  
_siddhattho upagacchi maṃ;_  
_Tāhaṃ buddhassa pādāsiṃ,_  
_pasanno sehi pāṇibhi._  

</Paragraph>

</Division>
<Division number="30" book="ap1">

<Paragraph number="3137" type="verses" book="ap1">
_Catunnavutito kappe,_  
_yaṃ paṇṇamadadiṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_paṇṇadānassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="31" book="ap1">

<Paragraph number="3138" type="verses" book="ap1">

_rājā āsiṃ sadatthiyo;_  
_Sattaratanasampanno,_  
_cakkavattī mahabbalo._  

</Paragraph>

</Division>

</Paragraph>
