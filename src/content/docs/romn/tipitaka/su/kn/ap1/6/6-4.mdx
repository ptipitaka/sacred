<Division number="25" book="ap1">

<Paragraph number="1406" type="verses" book="ap1">
_“Padumuttarabuddhassa,_  
_mahābodhimaho ahu;_  
_Vicittaṃ ghaṭamādāya,_  
_gandhodakamadāsahaṃ._  

</Paragraph>

</Division>
<Division number="26" book="ap1">

<Paragraph number="1407" type="verses" book="ap1">
_Nhānakāle ca bodhiyā,_  
_mahāmegho pavassatha;_  
_Ninnādo ca mahā āsi,_  
_asaniyā phalantiyā._  

</Paragraph>

</Division>
<Division number="27" book="ap1">

<Paragraph number="1408" type="verses" book="ap1">
_Tenevāsanivegena,_  
_tattha kālaṅkato ahaṃ;_  
_Devaloke ṭhito santo,_  
_imā gāthā abhāsahaṃ._  

</Paragraph>

</Division>
<Division number="28" book="ap1">

<Paragraph number="1409" type="verses" book="ap1">
_‘Aho buddhā aho dhammā,_  
_aho no satthusampadā;_  
_Kaḷevaraṃ me patitaṃ,_  
_devaloke ramāmahaṃ._  

</Paragraph>

</Division>
<Division number="29" book="ap1">

<Paragraph number="1410" type="verses" book="ap1">
_Ubbiddhaṃ bhavanaṃ mayhaṃ,_  
_Satabhūmaṃ samuggataṃ;_  
_Kaññāsatasahassāni,_  
_Parivārenti maṃ sadā._  

</Paragraph>

</Division>
<Division number="30" book="ap1">

<Paragraph number="1411" type="verses" book="ap1">
_Ābādhā me na vijjanti,_  
_soko mayhaṃ na vijjati;_  
_Pariḷāhaṃ na passāmi,_  
_puññakammassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="31" book="ap1">

<Paragraph number="1412" type="verses" book="ap1">
_Aṭṭhavīse kappasate,_  
_rājā saṃvasito ahuṃ;_  
_Sattaratanasampanno,_  
_cakkavattī mahabbalo’._  

</Paragraph>

</Division>
<Division number="32" book="ap1">

<Paragraph number="1413" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="1414" book="ap1">
Itthaṃ sudaṃ āyasmā gandhodakiyo thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="1415" type="center" book="ap1">
Gandhodakiyattherassāpadānaṃ catutthaṃ.
</Paragraph>
