<Paragraph number="3715" type="verses" book="ap1">
_“Siddhatthassa bhagavato,_  
_gandhadhūpaṃ adāsahaṃ;_  
_Sumanehi paṭicchannaṃ,_  
_buddhānucchavikañca taṃ._  

</Paragraph>

</Division>
<Division number="2" book="ap1" bookNo="20" page="305">

<Paragraph number="3716" type="verses" book="ap1">
_Kañcanagghiyasaṅkāsaṃ,_  
_buddhaṃ lokagganāyakaṃ;_  
_Indīvaraṃva jalitaṃ,_  
_ādittaṃva hutāsanaṃ._  

</Paragraph>

</Division>
<Division number="3" book="ap1" bookNo="20" page="305">

<Paragraph number="3717" type="verses" book="ap1">
_Byagghusabhaṃva pavaraṃ,_  
_abhijātaṃva kesariṃ;_  
_Nisinnaṃ samaṇānaggaṃ,_  
_bhikkhusaṃghapurakkhataṃ._  

</Paragraph>

</Division>
<Division number="4" book="ap1" bookNo="20" page="305">

<Paragraph number="3718" type="verses" book="ap1">
_Disvā cittaṃ pasādetvā,_  
_paggahetvāna añjaliṃ;_  
_Vanditvā satthuno pāde,_  
_pakkāmiṃ uttarāmukho._  

</Paragraph>

</Division>
<Division number="5" book="ap1" bookNo="20" page="305">

<Paragraph number="3719" type="verses" book="ap1">
_Catunnavutito kappe,_  
_yaṃ gandhamadadiṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_gandhapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="6" book="ap1" bookNo="20" page="305">

<Paragraph number="3720" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_vimokkhāpi ca aṭṭhime;_  
_Chaḷabhiññā sacchikatā,_  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="3721" book="ap1">
Itthaṃ sudaṃ āyasmā gandhadhūpiyo thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="3722" type="center" book="ap1">
Gandhadhūpiyattherassāpadānaṃ paṭhamaṃ.
</Paragraph>
