<Paragraph number="1308" type="verses" book="ap1">
_“Siddhatthassa bhagavato,_  
_mettacittassa tādino;_  
_Sayanaggaṃ mayā dinnaṃ,_  
_dussabhaṇḍehi atthataṃ._  

</Paragraph>

</Division>
<Division number="89" book="ap1" bookNo="20" page="110">

<Paragraph number="1309" type="verses" book="ap1">
_Paṭiggahesi bhagavā,_  
_kappiyaṃ sayanāsanaṃ;_  
_Uṭṭhāya sayanā tamhā,_  
_vehāsaṃ uggamī jino._  

</Paragraph>

</Division>
<Division number="90" book="ap1" bookNo="20" page="110">

<Paragraph number="1310" type="verses" book="ap1">
_Catunnavutito kappe,_  
_yaṃ sayanamadāsahaṃ;_  
_Duggatiṃ nābhijānāmi,_  
_sayanassa idaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="91" book="ap1" bookNo="20" page="110">

<Paragraph number="1311" type="verses" book="ap1">
_Ekapaññāsito kappe,_  
_varako devasavhayo;_  
_Sattaratanasampanno,_  
_cakkavattī mahabbalo._  

</Paragraph>

</Division>
<Division number="92" book="ap1" bookNo="20" page="110">

<Paragraph number="1312" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="1313" book="ap1">
Itthaṃ sudaṃ āyasmā sayanadāyako thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="1314" type="center" book="ap1">
Sayanadāyakattherassāpadānaṃ sattamaṃ.
</Paragraph>
