<Paragraph number="3183" type="verses" book="ap1">
_“Himavantassāvidūre,_  
_romaso nāma pabbato;_  
_Tamhi pabbatapādamhi,_  
_samaṇo bhāvitindriyo._  

</Paragraph>

</Division>
<Division number="19" book="ap1">

<Paragraph number="3184" type="verses" book="ap1">
_Biḷāliyo gahetvāna,_  
_samaṇassa adāsahaṃ;_  
_Anumodi mahāvīro,_  
_sayambhū aparājito._  

</Paragraph>

</Division>
<Division number="20" book="ap1">

<Paragraph number="3185" type="verses" book="ap1">
_Biḷālī te mama dinnā,_  
_vippasannena cetasā;_  
_Bhave nibbattamānamhi,_  
_phalaṃ nibbattataṃ tava._  

</Paragraph>

</Division>
<Division number="21" book="ap1">

<Paragraph number="3186" type="verses" book="ap1">
_Catunnavutito kappe,_  
_yaṃ biḷālimadāsahaṃ;_  
_Duggatiṃ nābhijānāmi,_  
_biḷāliyā idaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="22" book="ap1">

<Paragraph number="3187" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="3188" book="ap1">
Itthaṃ sudaṃ āyasmā biḷālidāyako thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="3189" type="center" book="ap1">
Biḷālidāyakattherassāpadānaṃ pañcamaṃ.
