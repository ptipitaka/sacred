<Paragraph number="7671" type="verses" book="ap1">
_“Mahāsamuddaṃ oggayha,_  
_bhavanaṃ me sunimmitaṃ;_  
_Sunimmitā pokkharaṇī,_  
_cakkavākūpakūjitā._  

</Paragraph>

</Division>
<Division number="2" book="ap1" bookNo="20" page="429">

<Paragraph number="7672" type="verses" book="ap1">
_Mandārakehi sañchannā,_  
_padumuppalakehi ca;_  
_Nadī ca sandate tattha,_  
_supatitthā manoramā._  

</Paragraph>

</Division>
<Division number="3" book="ap1" bookNo="20" page="429">

<Paragraph number="7673" type="verses" book="ap1">
_Macchakacchapasañchannā,_  
_nānādijasamotthaṭā;_  
_Mayūrakoñcābhirudā,_  
_kokilādīhi vagguhi._  

</Paragraph>

</Division>
<Division number="4" book="ap1" bookNo="20" page="429">

<Paragraph number="7674" type="verses" book="ap1">
_Pārevatā ravihaṃsā,_  
_cakkavākā nadīcarā;_  
_Tittirā sālikā cettha,_  
_pāvakā jīvaṃjīvakā._  

</Paragraph>

</Division>
<Division number="5" book="ap1" bookNo="20" page="429">

<Paragraph number="7675" type="verses" book="ap1">
_Haṃsākoñcābhinaditā,_  
_kosiyā piṅgalā bahū;_  
_Sattaratanasampannā,_  
_maṇimuttapavāḷikā._  

</Paragraph>

</Division>
<Division number="6" book="ap1" bookNo="20" page="429">

<Paragraph number="7676" type="verses" book="ap1">
_Sabbe soṇṇamayā rukkhā,_  
_nānākhandhasameritā;_  
_Ujjotenti divārattiṃ,_  
_bhavanaṃ sabbakālikaṃ._  

</Paragraph>

</Division>
<Division number="7" book="ap1" bookNo="20" page="429">

<Paragraph number="7677" type="verses" book="ap1">
_Saṭṭhituriyasahassāni,_  
_sāyaṃ pāto pavajjare;_  
_Soḷasitthisahassāni,_  
_parivārenti maṃ sadā._  

</Paragraph>

</Division>
<Division number="8" book="ap1" bookNo="20" page="429">

<Paragraph number="7678" type="verses" book="ap1">
_Abhinikkhamma bhavanā,_  
_sumedhaṃ lokanāyakaṃ;_  
_Pasannacitto sumano,_  
_vandayiṃ taṃ mahāyasaṃ._  

</Paragraph>

</Division>
<Division number="9" book="ap1" bookNo="20" page="429">

<Paragraph number="7679" type="verses" book="ap1">
_Sambuddhaṃ abhivādetvā,_  
_sasaṃghaṃ taṃ nimantayiṃ;_  
_Adhivāsesi so dhīro,_  
_sumedho lokanāyako._  

</Paragraph>

</Division>
<Division number="10" book="ap1" bookNo="20" page="429">

<Paragraph number="7680" type="verses" book="ap1">
_Mama dhammakathaṃ katvā,_  
_uyyojesi mahāmuni;_  
_Sambuddhaṃ abhivādetvā,_  
_bhavanaṃ me upāgamiṃ._  

</Paragraph>

</Division>
<Division number="11" book="ap1" bookNo="20" page="429">

<Paragraph number="7681" type="verses" book="ap1">
_Āmantayiṃ parijanaṃ,_  
_sabbe sannipatuṃ tadā;_  
_‘Pubbaṇhasamayaṃ buddho,_  
_bhavanaṃ āgamissati._  

</Paragraph>

</Division>
<Division number="12" book="ap1" bookNo="20" page="430">

<Paragraph number="7682" type="verses" book="ap1">
_Lābhā amhaṃ suladdhā no,_  
_ye vasāma tavantike;_  
_Mayampi buddhaseṭṭhassa,_  
_pūjayissāma satthuno’._  

</Paragraph>

</Division>
<Division number="13" book="ap1" bookNo="20" page="430">

<Paragraph number="7683" type="verses" book="ap1">
_Annaṃ pānaṃ paṭṭhapetvā,_  
_kālaṃ ārocayiṃ ahaṃ;_  
_Vasīsatasahassehi,_  
_upesi lokanāyako._  

</Paragraph>

</Division>
<Division number="14" book="ap1" bookNo="20" page="430">

<Paragraph number="7684" type="verses" book="ap1">
_Pañcaṅgikehi turiyehi,_  
_paccuggamamakāsahaṃ;_  
_Sabbasoṇṇamaye pīṭhe,_  
_nisīdi purisuttamo._  

</Paragraph>

</Division>
<Division number="15" book="ap1" bookNo="20" page="430">

<Paragraph number="7685" type="verses" book="ap1">
_Uparicchadanaṃ āsi,_  
_sabbasoṇṇamayaṃ tadā;_  
_Bījanīyo pavāyanti,_  
_bhikkhusaṃghaṃ anuttaraṃ._  

</Paragraph>

</Division>
<Division number="16" book="ap1" bookNo="20" page="430">

<Paragraph number="7686" type="verses" book="ap1">
_Pahūtenannapānena,_  
_bhikkhusaṃghaṃ atappayiṃ;_  
_Paccekadussayugale,_  
_bhikkhusaṃghassadāsahaṃ._  

</Paragraph>

</Division>
<Division number="17" book="ap1" bookNo="20" page="430">

<Paragraph number="7687" type="verses" book="ap1">
_Yaṃ vadeti sumedho so,_  
_āhutīnaṃ paṭiggaho;_  
_Bhikkhusaṃghe nisīditvā,_  
_imā gāthā abhāsatha._  

</Paragraph>

</Division>
<Division number="18" book="ap1" bookNo="20" page="430">

<Paragraph number="7688" type="verses" book="ap1">
_‘Yo maṃ annena pānena,_  
_sabbe ime ca tappayi;_  
_Tamahaṃ kittayissāmi,_  
_suṇātha mama bhāsato._  

</Paragraph>

</Division>
<Division number="19" book="ap1" bookNo="20" page="430">

<Paragraph number="7689" type="verses" book="ap1">
_Aṭṭhārase kappasate,_  
_devaloke ramissati;_  
_Sahassakkhattuṃ rājāyaṃ,_  
_cakkavattī bhavissati._  

</Paragraph>

</Division>
<Division number="20" book="ap1" bookNo="20" page="430">

<Paragraph number="7690" type="verses" book="ap1">
_Upagacchati yaṃ yoniṃ,_  
_devattaṃ atha mānusaṃ;_  
_Sabbasoṇṇamayaṃ tassa,_  
_chadanaṃ dhārayissati._  

</Paragraph>

</Division>
<Division number="21" book="ap1" bookNo="20" page="430">

<Paragraph number="7691" type="verses" book="ap1">
_Tiṃsakappasahassamhi,_  
_okkākakulasambhavo;_  
_Gotamo nāma gottena,_  
_satthā loke bhavissati._  

</Paragraph>

</Division>
<Division number="22" book="ap1" bookNo="20" page="430">

<Paragraph number="7692" type="verses" book="ap1">
_Tassa dhammesu dāyādo,_  
_oraso dhammanimmito;_  
_Sabbāsave pariññāya,_  
_nibbāyissatināsavo._  

</Paragraph>

</Division>
<Division number="23" book="ap1" bookNo="20" page="430">

<Paragraph number="7693" type="verses" book="ap1">
_Bhikkhusaṃghe nisīditvā,_  
_sīhanādaṃ nadissati;_  
_Citake chattaṃ dhārenti,_  
_heṭṭhā chattamhi ḍayhatha’._  

</Paragraph>

</Division>
<Division number="24" book="ap1" bookNo="20" page="431">

<Paragraph number="7694" type="verses" book="ap1">
_Sāmaññaṃ me anuppattaṃ,_  
_kilesā jhāpitā mayā;_  
_Maṇḍape rukkhamūle vā,_  
_santāso me na vijjati._  

</Paragraph>

</Division>
<Division number="25" book="ap1" bookNo="20" page="431">

<Paragraph number="7695" type="verses" book="ap1">
_Tiṃsakappasahassamhi,_  
_yaṃ dānamadadiṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_sabbadānassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="26" book="ap1" bookNo="20" page="431">

<Paragraph number="7696" type="verses" book="ap1">
_Kilesā jhāpitā mayhaṃ,_  
_bhavā sabbe samūhatā;_  
_Nāgova bandhanaṃ chetvā,_  
_viharāmi anāsavo._  

</Paragraph>

</Division>
<Division number="27" book="ap1" bookNo="20" page="431">

<Paragraph number="7697" type="verses" book="ap1">
_Svāgataṃ vata me āsi,_  
_mama buddhassa santike;_  
_Tisso vijjā anuppattā,_  
_kataṃ buddhassa sāsanaṃ._  

</Paragraph>

</Division>
<Division number="28" book="ap1" bookNo="20" page="431">

<Paragraph number="7698" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_vimokkhāpi ca aṭṭhime;_  
_Chaḷabhiññā sacchikatā,_  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="7699" book="ap1">
Itthaṃ sudaṃ āyasmā yaso thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="7700" type="center" book="ap1">
Yasattherassāpadānaṃ paṭhamaṃ.
</Paragraph>
