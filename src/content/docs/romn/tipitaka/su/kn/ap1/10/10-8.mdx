<Paragraph number="1823" type="verses" book="ap1">
_“Siddhatthassa bhagavato,_  
_adāsiṃ paṇṇasantharaṃ;_  
_Samantā upahārañca,_  
_kusumaṃ okiriṃ ahaṃ._  

</Paragraph>

</Division>
<Division number="46" book="ap1" bookNo="20" page="151">

<Paragraph number="1824" type="verses" book="ap1">
_Pāsādevaṃ guṇaṃ rammaṃ,_  
_Anubhomi mahārahaṃ;_  
_Mahagghāni ca pupphāni,_  
_Sayanebhisavanti me._  

</Paragraph>

</Division>
<Division number="47" book="ap1" bookNo="20" page="151">

<Paragraph number="1825" type="verses" book="ap1">
_Sayanehaṃ tuvaṭṭāmi,_  
_vicitte pupphasanthate;_  
_Pupphavuṭṭhi ca sayane,_  
_abhivassati tāvade._  

</Paragraph>

</Division>
<Division number="48" book="ap1" bookNo="20" page="152">

<Paragraph number="1826" type="verses" book="ap1">
_Catunnavutito kappe,_  
_adāsiṃ paṇṇasantharaṃ;_  
_Duggatiṃ nābhijānāmi,_  
_santharassa idaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="49" book="ap1" bookNo="20" page="152">

<Paragraph number="1827" type="verses" book="ap1">
_Tiṇasantharakā nāma,_  
_sattete cakkavattino;_  
_Ito te pañcame kappe,_  
_uppajjiṃsu janādhipā._  

</Paragraph>

</Division>
<Division number="50" book="ap1" bookNo="20" page="152">

<Paragraph number="1828" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="1829" book="ap1">
Itthaṃ sudaṃ āyasmā senāsanadāyako thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="1830" type="center" book="ap1">
