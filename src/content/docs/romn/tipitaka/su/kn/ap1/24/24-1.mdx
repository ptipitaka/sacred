<Division number="1" book="ap1">

<Paragraph number="2973" type="verses" book="ap1">
_“Ārāmadvārā nikkhamma,_  
_phalakaṃ santhariṃ ahaṃ;_  
_Udakañca upaṭṭhāsiṃ,_  
_uttamatthassa pattiyā._  

</Paragraph>

</Division>
<Division number="2" book="ap1">

<Paragraph number="2974" type="verses" book="ap1">
_Ekattiṃse ito kappe,_  
_yaṃ kammamakariṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_āsane codake phalaṃ._  

</Paragraph>

</Division>
<Division number="3" book="ap1">

<Paragraph number="2975" type="verses" book="ap1">
_Ito pannarase kappe,_  
_abhisāmasamavhayo;_  
_Sattaratanasampanno,_  
_cakkavattī mahabbalo._  

</Paragraph>

</Division>
<Division number="4" book="ap1">

<Paragraph number="2976" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_vimokkhāpi ca aṭṭhime;_  
_Chaḷabhiññā sacchikatā,_  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="2977" book="ap1">
Itthaṃ sudaṃ āyasmā udakāsanadāyako thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="2978" type="center" book="ap1">
Udakāsanadāyakattherassāpadānaṃ paṭhamaṃ.
</Paragraph>
