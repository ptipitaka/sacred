<Paragraph number="2640" type="verses" book="ap1">
_“Devabhūto ahaṃ santo,_  
_accharāhi purakkhato;_  
_Pubbakammaṃ saritvāna,_  
_buddhaseṭṭhaṃ anussariṃ._  

</Paragraph>

</Division>
<Division number="47" book="ap1">

<Paragraph number="2641" type="verses" book="ap1">
_Tikaṇṇipupphaṃ paggayha,_  
_sakaṃ cittaṃ pasādayiṃ;_  
_Buddhamhi abhiropesiṃ,_  
_vipassimhi narāsabhe._  

</Paragraph>

</Division>
<Division number="48" book="ap1">

<Paragraph number="2642" type="verses" book="ap1">
_Ekanavutito kappe,_  
_yaṃ pupphamabhipūjayiṃ;_  
_Duggatiṃ nābhijānāmi,_  
_buddhapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="49" book="ap1">

<Paragraph number="2643" type="verses" book="ap1">
_Tesattatimhito kappe,_  
_caturāsuṃ ramuttamā;_  
_Sattaratanasampannā,_  
_cakkavattī mahabbalā._  

</Paragraph>

</Division>
<Division number="50" book="ap1">

<Paragraph number="2644" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="2645" book="ap1">
Itthaṃ sudaṃ āyasmā tikaṇṇipupphiyo thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="2646" type="center" book="ap1">

</Paragraph>
