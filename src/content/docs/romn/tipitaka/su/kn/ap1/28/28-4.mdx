_vītarāgassa tādino;_

<Division number="15" book="ap1">

<Paragraph number="3243" type="verses" book="ap1">
_Sabbamohātivattassa,_  
_sabbalokahitesino;_  
_Abbhañjanaṃ mayā dinnaṃ,_  
_dvipadindassa tādino._  

</Paragraph>

</Division>
<Division number="16" book="ap1">

<Paragraph number="3244" type="verses" book="ap1">
_Aparimeyye ito kappe,_  
_abbhañjanamadaṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_abbhañjanassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="17" book="ap1">

<Paragraph number="3245" type="verses" book="ap1">
_Ito pannarase kappe,_  
_cirappo nāma khattiyo;_  
_Sattaratanasampanno,_  
_cakkavattī mahabbalo._  

</Paragraph>

</Division>
<Division number="18" book="ap1">

<Paragraph number="3246" type="verses" book="ap1">

_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="3247" book="ap1">

</Paragraph>
