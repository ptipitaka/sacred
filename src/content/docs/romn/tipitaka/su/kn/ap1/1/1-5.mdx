<Paragraph number="409" type="verses" book="ap1">
_“Padumuttarassa bhagavato,_  
_Lokajeṭṭhassa tādino;_  
_Nibbute lokanāthamhi,_  
_Pūjaṃ kubbanti satthuno._  

</Paragraph>

</Division>
<Division number="399" book="ap1" bookNo="20" page="36">

<Paragraph number="410" type="verses" book="ap1">
_Udaggacittā janatā,_  
_āmoditapamoditā;_  
_Tesu saṃvegajātesu,_  
_pīti me udapajjatha._  

</Paragraph>

</Division>
<Division number="400" book="ap1" bookNo="20" page="36">

<Paragraph number="411" type="verses" book="ap1">
_Ñātimitte samānetvā,_  
_idaṃ vacanamabraviṃ;_  
_Parinibbuto mahāvīro,_  
_handa pūjaṃ karomase._  

</Paragraph>

</Division>
<Division number="401" book="ap1" bookNo="20" page="36">

<Paragraph number="412" type="verses" book="ap1">
_Sādhūti te paṭissutvā,_  
_bhiyyo hāsaṃ janiṃsu me;_  
_Buddhasmiṃ lokanāthamhi,_  
_kāhāma puññasañcayaṃ._  

</Paragraph>

</Division>
<Division number="402" book="ap1" bookNo="20" page="37">

<Paragraph number="413" type="verses" book="ap1">
_Agghiyaṃ sukataṃ katvā,_  
_satahatthasamuggataṃ;_  
_Diyaḍḍhahatthapatthaṭaṃ,_  
_vimānaṃ nabhamuggataṃ._  

</Paragraph>

</Division>
<Division number="403" book="ap1" bookNo="20" page="37">

<Paragraph number="414" type="verses" book="ap1">
_Katvāna hammiyaṃ tattha,_  
_tālapantīhi cittitaṃ;_  
_Sakaṃ cittaṃ pasādetvā,_  
_cetiyaṃ pūjayuttamaṃ._  

</Paragraph>

</Division>
<Division number="404" book="ap1" bookNo="20" page="37">

<Paragraph number="415" type="verses" book="ap1">
_Aggikkhandhova jalito,_  
_kiṃsuko iva phullito;_  
_Indalaṭṭhīva ākāse,_  
_obhāsati catuddisā._  

</Paragraph>

</Division>
<Division number="405" book="ap1" bookNo="20" page="37">

<Paragraph number="416" type="verses" book="ap1">
_Tattha cittaṃ pasādetvā,_  
_katvāna kusalaṃ bahuṃ;_  
_Pubbakammaṃ saritvāna,_  
_tidasaṃ upapajjahaṃ._  

</Paragraph>

</Division>
<Division number="406" book="ap1" bookNo="20" page="37">

<Paragraph number="417" type="verses" book="ap1">
_Sahassayuttaṃ hayavāhiṃ,_  
_Dibbayānamadhiṭṭhito;_  
_Ubbiddhaṃ bhavanaṃ mayhaṃ,_  
_Sattabhūmaṃ samuggataṃ._  

</Paragraph>

</Division>
<Division number="407" book="ap1" bookNo="20" page="37">

<Paragraph number="418" type="verses" book="ap1">
_Kūṭāgārasahassāni,_  
_sabbasoṇṇamayā ahuṃ;_  
_Jalanti sakatejena,_  
_disā sabbā pabhāsayaṃ._  

</Paragraph>

</Division>
<Division number="408" book="ap1" bookNo="20" page="37">

<Paragraph number="419" type="verses" book="ap1">
_Santi aññepi niyyūhā,_  
_lohitaṅgamayā tadā;_  
_Tepi jotanti ābhāya,_  
_samantā caturo disā._  

</Paragraph>

</Division>
<Division number="409" book="ap1" bookNo="20" page="37">

<Paragraph number="420" type="verses" book="ap1">
_Puññakammābhinibbattā,_  
_kūṭāgārā sunimmitā;_  
_Maṇimayāpi jotanti,_  
_disā dasa samantato._  

</Paragraph>

</Division>
<Division number="410" book="ap1" bookNo="20" page="37">

<Paragraph number="421" type="verses" book="ap1">
_Tesaṃ ujjotamānānaṃ,_  
_obhāso vipulo ahu;_  
_Sabbe deve abhibhomi,_  
_puññakammassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="411" book="ap1" bookNo="20" page="37">

<Paragraph number="422" type="verses" book="ap1">
_Saṭṭhikappasahassamhi,_  
_ubbiddho nāma khattiyo;_  
_Cāturanto vijitāvī,_  
_pathaviṃ āvasiṃ ahaṃ._  

</Paragraph>

</Division>
<Division number="412" book="ap1" bookNo="20" page="37">

<Paragraph number="423" type="verses" book="ap1">
_Tatheva bhaddake kappe,_  
_tiṃsakkhattuṃ ahosahaṃ;_  
_Sakakammābhiraddhomhi,_  
_cakkavattī mahabbalo._  

</Paragraph>

</Division>
<Division number="413" book="ap1" bookNo="20" page="37">

<Paragraph number="424" type="verses" book="ap1">
_Sattaratanasampanno,_  
_catudīpamhi issaro;_  
_Tatthāpi bhavanaṃ mayhaṃ,_  
_indalaṭṭhīva uggataṃ._  

</Paragraph>

</Division>
<Division number="414" book="ap1" bookNo="20" page="37">

<Paragraph number="425" type="verses" book="ap1">
_Āyāmato catubbīsaṃ,_  
_vitthārena ca dvādasa;_  
_Rammaṇaṃ nāma nagaraṃ,_  
_daḷhapākāratoraṇaṃ._  

</Paragraph>

</Division>
<Division number="415" book="ap1" bookNo="20" page="38">

<Paragraph number="426" type="verses" book="ap1">
_Āyāmato pañcasataṃ,_  
_vitthārena tadaḍḍhakaṃ;_  
_Ākiṇṇaṃ janakāyehi,_  
_tidasānaṃ puraṃ viya._  

</Paragraph>

</Division>
<Division number="416" book="ap1" bookNo="20" page="38">

<Paragraph number="427" type="verses" book="ap1">
_Yathā sūcighare sūcī,_  
_pakkhittā paṇṇavīsati;_  
_Aññamaññaṃ paghaṭṭenti,_  
_ākiṇṇaṃ hoti laṅkataṃ._  

</Paragraph>

</Division>
<Division number="417" book="ap1" bookNo="20" page="38">

<Paragraph number="428" type="verses" book="ap1">
_Evampi nagaraṃ mayhaṃ,_  
_hatthissarathasaṅkulaṃ;_  
_Manussehi sadākiṇṇaṃ,_  
_rammaṇaṃ nagaruttamaṃ._  

</Paragraph>

</Division>
<Division number="418" book="ap1" bookNo="20" page="38">

<Paragraph number="429" type="verses" book="ap1">
_Tattha bhutvā pivitvā ca,_  
_puna devattanaṃ gato;_  
_Bhave pacchimake mayhaṃ,_  
_ahosi kulasampadā._  

</Paragraph>

</Division>
<Division number="419" book="ap1" bookNo="20" page="38">

<Paragraph number="430" type="verses" book="ap1">
_Brāhmaññakulasambhūto,_  
_mahāratanasañcayo;_  
_Asītikoṭiyo hitvā,_  
_hiraññassāpi pabbajiṃ._  

</Paragraph>

</Division>
<Division number="420" book="ap1" bookNo="20" page="38">

<Paragraph number="431" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_vimokkhāpi ca aṭṭhime;_  
_Chaḷabhiññā sacchikatā,_  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="432" book="ap1">
Itthaṃ sudaṃ āyasmā mahākassapo thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="433" type="center" book="ap1">
Mahākassapattherassāpadānaṃ tatiyaṃ.
</Paragraph>
