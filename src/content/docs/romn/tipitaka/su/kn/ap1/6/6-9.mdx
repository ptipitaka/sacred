<Division number="67" book="ap1">

<Paragraph number="1458" type="verses" book="ap1">
_“Catusaccaṃ pakāsento,_  
_varadhammappavattako;_  
_Vassate amataṃ vuṭṭhiṃ,_  
_nibbāpento mahājanaṃ._  

</Paragraph>

</Division>
<Division number="68" book="ap1">

<Paragraph number="1459" type="verses" book="ap1">
_Sadhajaṃ padumaṃ gayha,_  
_aḍḍhakose ṭhito ahaṃ;_  
_Padumuttaramunissa,_  
_pahaṭṭho ukkhipimambare._  

</Paragraph>

</Division>
<Division number="69" book="ap1">

<Paragraph number="1460" type="verses" book="ap1">
_Āgacchante ca padume,_  
_abbhuto āsi tāvade;_  
_Mama saṅkappamaññāya,_  
_paggaṇhi vadataṃ varo._  

</Paragraph>

</Division>
<Division number="70" book="ap1">

<Paragraph number="1461" type="verses" book="ap1">
_Karaseṭṭhena paggayha,_  
_jalajaṃ pupphamuttamaṃ;_  
_Bhikkhusaṃghe ṭhito satthā,_  
_imā gāthā abhāsatha._  

</Paragraph>

</Division>
<Division number="71" book="ap1">

<Paragraph number="1462" type="verses" book="ap1">
_‘Yenidaṃ padumaṃ khittaṃ,_  
_sabbaññumhi vināyake;_  
_Tamahaṃ kittayissāmi,_  
_suṇātha mama bhāsato._  

</Paragraph>

</Division>
<Division number="72" book="ap1">

<Paragraph number="1463" type="verses" book="ap1">
_Tiṃsakappāni devindo,_  
_devarajjaṃ karissati;_  
_Pathabyā rajjaṃ sattasataṃ,_  
_vasudhaṃ āvasissati._  

</Paragraph>

</Division>
<Division number="73" book="ap1">

<Paragraph number="1464" type="verses" book="ap1">
_Tattha pattaṃ gaṇetvāna,_  
_cakkavattī bhavissati;_  
_Ākāsato pupphavuṭṭhi,_  
_abhivassissatī tadā._  

</Paragraph>

</Division>
<Division number="74" book="ap1">

<Paragraph number="1465" type="verses" book="ap1">
_Kappasatasahassamhi,_  
_okkākakulasambhavo;_  
_Gotamo nāma gottena,_  
_satthā loke bhavissati._  

</Paragraph>

</Division>
<Division number="75" book="ap1">

<Paragraph number="1466" type="verses" book="ap1">
_Tassa dhammesu dāyādo,_  
_oraso dhammanimmito;_  
_Sabbāsave pariññāya,_  
_nibbāyissatināsavo’._  

</Paragraph>

</Division>
<Division number="76" book="ap1">

<Paragraph number="1467" type="verses" book="ap1">
_Nikkhamitvāna kucchimhā,_  
_sampajāno patissato;_  
_Jātiyā pañcavassohaṃ,_  
_arahattaṃ apāpuṇiṃ._  

</Paragraph>

</Division>
<Division number="77" book="ap1">

<Paragraph number="1468" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="1469" book="ap1">
Itthaṃ sudaṃ āyasmā padumo thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="1470" type="center" book="ap1">
Padumattherassāpadānaṃ navamaṃ.
</Paragraph>
