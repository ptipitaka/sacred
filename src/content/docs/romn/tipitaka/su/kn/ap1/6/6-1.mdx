<Paragraph number="1376" type="verses" book="ap1">
_“Padumuttarabuddhassa,_  
_lokajeṭṭhassa tādino;_  
_Bījanikā mayā dinnā,_  
_dvipadindassa tādino._  

</Paragraph>

</Division>
<Division number="2" book="ap1" bookNo="20" page="115">

<Paragraph number="1377" type="verses" book="ap1">
_Sakaṃ cittaṃ pasādetvā,_  
_paggahetvāna añjaliṃ;_  
_Sambuddhamabhivādetvā,_  
_pakkamiṃ uttarāmukho._  

</Paragraph>

</Division>
<Division number="3" book="ap1" bookNo="20" page="115">

<Paragraph number="1378" type="verses" book="ap1">
_Bījaniṃ paggahetvāna,_  
_satthā lokagganāyako;_  
_Bhikkhusaṃghe ṭhito santo,_  
_imā gāthā abhāsatha._  

</Paragraph>

</Division>
<Division number="4" book="ap1" bookNo="20" page="115">

<Paragraph number="1379" type="verses" book="ap1">
_‘Iminā bījanidānena,_  
_cittassa paṇidhīhi ca;_  
_Kappānaṃ satasahassaṃ,_  
_vinipātaṃ na gacchati’._  

</Paragraph>

</Division>
<Division number="5" book="ap1" bookNo="20" page="115">

<Paragraph number="1380" type="verses" book="ap1">
_Āraddhavīriyo pahitatto,_  
_Cetoguṇasamāhito;_  
_Jātiyā sattavassohaṃ,_  
_Arahattaṃ apāpuṇiṃ._  

</Paragraph>

</Division>
<Division number="6" book="ap1" bookNo="20" page="115">

<Paragraph number="1381" type="verses" book="ap1">
_Saṭṭhikappasahassamhi,_  
_bījamānasanāmakā;_  
_Soḷasāsiṃsu rājāno,_  
_cakkavattī mahabbalā._  

</Paragraph>

</Division>
<Division number="7" book="ap1" bookNo="20" page="115">

<Paragraph number="1382" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_vimokkhāpi ca aṭṭhime;_  
_Chaḷabhiññā sacchikatā,_  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="1383" book="ap1">
Itthaṃ sudaṃ āyasmā vidhūpanadāyako thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="1384" type="center" book="ap1">
Vidhūpanadāyakattherassāpadānaṃ paṭhamaṃ.
</Paragraph>
