<Division number="1" book="ap1">

<Paragraph number="3970" type="verses" book="ap1">
_“Migaluddo pure āsiṃ,_  
_araññe kānane ahaṃ;_  
_Tatthaddasāsiṃ sambuddhaṃ,_  
_devasaṅghapurakkhataṃ._  

</Paragraph>

</Division>
<Division number="2" book="ap1">

<Paragraph number="3971" type="verses" book="ap1">
_Catusaccaṃ pakāsentaṃ,_  
_uddharantaṃ mahājanaṃ;_  
_Assosiṃ madhuraṃ vācaṃ,_  
_karavīkarudopamaṃ._  

</Paragraph>

</Division>
<Division number="3" book="ap1">

<Paragraph number="3972" type="verses" book="ap1">
_Brahmasarassa munino,_  
_sikhino lokabandhuno;_  
_Ghose cittaṃ pasādetvā,_  
_pattomhi āsavakkhayaṃ._  

</Paragraph>

</Division>
<Division number="4" book="ap1">

<Paragraph number="3973" type="verses" book="ap1">
_Ekattiṃse ito kappe,_  
_yaṃ kammamakariṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_pasādassa idaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="5" book="ap1">

<Paragraph number="3974" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_vimokkhāpi ca aṭṭhime;_  
_Chaḷabhiññā sacchikatā,_  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="3975" book="ap1">
Itthaṃ sudaṃ āyasmā saddasaññako thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="3976" type="center" book="ap1">
Saddasaññakattherassāpadānaṃ paṭhamaṃ.
</Paragraph>
