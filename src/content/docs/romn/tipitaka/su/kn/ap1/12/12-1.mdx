<Paragraph number="1941" type="verses" book="ap1">
_“Vipassī nāma bhagavā,_  
_lokajeṭṭho narāsabho;_  
_Aṭṭhasaṭṭhisahassehi,_  
_pāvisi bandhumaṃ tadā._  

</Paragraph>

</Division>
<Division number="2" book="ap1" bookNo="20" page="161">

<Paragraph number="1942" type="verses" book="ap1">
_Nagarā abhinikkhamma,_  
_agamaṃ dīpacetiyaṃ;_  
_Addasaṃ virajaṃ buddhaṃ,_  
_āhutīnaṃ paṭiggahaṃ._  

</Paragraph>

</Division>
<Division number="3" book="ap1" bookNo="20" page="161">

<Paragraph number="1943" type="verses" book="ap1">
_Cullāsītisahassāni,_  
_yakkhā mayhaṃ upantike;_  
_Upaṭṭhahanti sakkaccaṃ,_  
_indaṃva tidasā gaṇā._  

</Paragraph>

</Division>
<Division number="4" book="ap1" bookNo="20" page="161">

<Paragraph number="1944" type="verses" book="ap1">
_Bhavanā abhinikkhamma,_  
_dussaṃ paggayhahaṃ tadā;_  
_Sirasā abhivādesiṃ,_  
_tañcādāsiṃ mahesino._  

</Paragraph>

</Division>
<Division number="5" book="ap1" bookNo="20" page="161">

<Paragraph number="1945" type="verses" book="ap1">
_Aho buddhā aho dhammā,_  
_aho no satthu sampadā;_  
_Buddhassa ānubhāvena,_  
_vasudhāyaṃ pakampatha._  

</Paragraph>

</Division>
<Division number="6" book="ap1" bookNo="20" page="161">

<Paragraph number="1946" type="verses" book="ap1">
_Tañca acchariyaṃ disvā,_  
_abbhutaṃ lomahaṃsanaṃ;_  
_Buddhe cittaṃ pasādemi,_  
_dvipadindamhi tādine._  

</Paragraph>

</Division>
<Division number="7" book="ap1" bookNo="20" page="161">

<Paragraph number="1947" type="verses" book="ap1">
_Sohaṃ cittaṃ pasādetvā,_  
_dussaṃ datvāna satthuno;_  
_Saraṇañca upāgacchiṃ,_  
_sāmacco saparijjano._  

</Paragraph>

</Division>
<Division number="8" book="ap1" bookNo="20" page="161">

<Paragraph number="1948" type="verses" book="ap1">
_Ekanavutito kappe,_  
_yaṃ kammamakariṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_buddhapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="9" book="ap1" bookNo="20" page="161">

<Paragraph number="1949" type="verses" book="ap1">
_Ito pannarase kappe,_  
_soḷasāsuṃ suvāhanā;_  
_Sattaratanasampannā,_  
_cakkavattī mahabbalā._  

</Paragraph>

</Division>
<Division number="10" book="ap1" bookNo="20" page="161">

<Paragraph number="1950" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_vimokkhāpi ca aṭṭhime;_  
_Chaḷabhiññā sacchikatā,_  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="1951" book="ap1">
Itthaṃ sudaṃ āyasmā mahāparivārako thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="1952" type="center" book="ap1">
Mahāparivārakattherassāpadānaṃ paṭhamaṃ.
</Paragraph>
