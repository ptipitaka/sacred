<Paragraph number="2588" type="verses" book="ap1">
_“Siddhattho nāma sambuddho,_  
_sayambhū sabbhi vaṇṇito;_  
_Samādhiṃ so samāpanno,_  
_nisīdi pabbatantare._  

</Paragraph>

</Division>
<Division number="7" book="ap1" bookNo="20" page="213">

<Paragraph number="2589" type="verses" book="ap1">
_Jātassare gavesanto,_  
_dakajaṃ pupphamuttamaṃ;_  
_Bandhujīvakapupphāni,_  
_addasaṃ samanantaraṃ._  

</Paragraph>

</Division>
<Division number="8" book="ap1" bookNo="20" page="213">

<Paragraph number="2590" type="verses" book="ap1">
_Ubho hatthehi paggayha,_  
_upāgacchiṃ mahāmuniṃ;_  
_Pasannacitto sumano,_  
_siddhatthassābhiropayiṃ._  

</Paragraph>

</Division>
<Division number="9" book="ap1" bookNo="20" page="213">

<Paragraph number="2591" type="verses" book="ap1">
_Catunnavutito kappe,_  
_yaṃ pupphamabhiropayiṃ;_  
_Duggatiṃ nābhijānāmi,_  
_buddhapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="10" book="ap1" bookNo="20" page="214">

<Paragraph number="2592" type="verses" book="ap1">
_Ito cātuddase kappe,_  
_eko āsiṃ janādhipo;_  
_Samuddakappo nāmena,_  
_cakkavattī mahabbalo._  

</Paragraph>

</Division>

_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="2594" book="ap1">

</Paragraph>
