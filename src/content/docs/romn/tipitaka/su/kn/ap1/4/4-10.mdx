<Paragraph number="1198" type="verses" book="ap1">
_“Siddhatthassa bhagavato,_  
_thūpaseṭṭhassa sammukhā;_  
_Vicittadusse lagetvā,_  
_kapparukkhaṃ ṭhapesahaṃ._  

</Paragraph>

</Division>
<Division number="109" book="ap1" bookNo="20" page="101">

<Paragraph number="1199" type="verses" book="ap1">
_Yaṃ yaṃ yonupapajjāmi,_  
_devattaṃ atha mānusaṃ;_  
_Sobhayanto mama dvāraṃ,_  
_kapparukkho patiṭṭhati._  

</Paragraph>

</Division>
<Division number="110" book="ap1" bookNo="20" page="101">

<Paragraph number="1200" type="verses" book="ap1">
_Ahañca parisā ceva,_  
_ye keci mama vassitā;_  
_Tamhā dussaṃ gahetvāna,_  
_nivāsema mayaṃ sadā._  

</Paragraph>

</Division>
<Division number="111" book="ap1" bookNo="20" page="101">

<Paragraph number="1201" type="verses" book="ap1">
_Catunnavutito kappe,_  
_yaṃ rukkhaṃ ṭhapayiṃ ahaṃ;_  
_Duggatiṃ nābhijānāmi,_  
_kapparukkhassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="112" book="ap1" bookNo="20" page="101">

<Paragraph number="1202" type="verses" book="ap1">
_Ito ca sattame kappe,_  
_suceḷā aṭṭha khattiyā;_  
_Sattaratanasampannā,_  
_cakkavattī mahabbalā._  

</Paragraph>

</Division>
<Division number="113" book="ap1" bookNo="20" page="101">

<Paragraph number="1203" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”. (1103)_  

</Paragraph>

<Paragraph number="1204" book="ap1">
Itthaṃ sudaṃ āyasmā kapparukkhiyo thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="1205" type="center" book="ap1">
Kapparukkhiyattherassāpadānaṃ dasamaṃ.



</Paragraph>

<Paragraph number="1206" type="center" book="ap1">
Kuṇḍadhānavaggo catuttho.



</Paragraph>

<Paragraph number="1207" type="center" book="ap1">
Tassuddānaṃ


</Paragraph>

_Kuṇḍasāgatakaccānā,_  
_udāyī mogharājako;_  
_Adhimutto lasuṇado,_  
_āyāgī dhammacakkiko;_  
_Kapparukkhī ca dasamo,_  
_gāthā dvayadasasataṃ._
</Paragraph>
