<Paragraph number="2328" type="verses" book="ap1">
_“Yānakāro pure āsiṃ,_  
_dārukamme susikkhito;_  
_Candanaṃ phalakaṃ katvā,_  
_adāsiṃ lokabandhuno._  

</Paragraph>

<Division number="38" book="ap1" bookNo="20" page="192">

<Paragraph number="2329" type="verses" book="ap1">
_Pabhāsati idaṃ byamhaṃ,_  
_suvaṇṇassa sunimmitaṃ;_  
_Hatthiyānaṃ assayānaṃ,_  
_dibbayānaṃ upaṭṭhitaṃ._  

</Paragraph>

</Division>
<Division number="39" book="ap1" bookNo="20" page="192">

<Paragraph number="2330" type="verses" book="ap1">
_Pāsādā sivikā ceva,_  
_Nibbattanti yadicchakaṃ;_  
_Akkhobhaṃ ratanaṃ mayhaṃ,_  
_Phalakassa idaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="40" book="ap1" bookNo="20" page="192">

<Paragraph number="2331" type="verses" book="ap1">
_Ekanavutito kappe,_  
_phalakaṃ yamahaṃ dadiṃ;_  
_Duggatiṃ nābhijānāmi,_  
_phalakassa idaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="41" book="ap1" bookNo="20" page="192">

<Paragraph number="2332" type="verses" book="ap1">

_caturo nimmitāvhayā;_  
_Sattaratanasampannā,_  
_cakkavattī mahabbalā._  

</Paragraph>

</Division>

</Paragraph>

</Paragraph>
