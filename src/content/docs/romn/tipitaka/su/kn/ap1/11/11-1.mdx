<Paragraph number="1852" type="verses" book="ap1">
_“Suvaṇṇavaṇṇaṃ sambuddhaṃ,_  
_Āhutīnaṃ paṭiggahaṃ;_  
_Pavarā abhinikkhantaṃ,_  
_Vanā nibbanamāgataṃ._  

</Paragraph>

</Division>
<Division number="2" book="ap1" bookNo="20" page="154">

<Paragraph number="1853" type="verses" book="ap1">
_Kaṭacchubhikkhaṃ pādāsiṃ,_  
_siddhatthassa mahesino;_  
_Paññāya upasantassa,_  
_mahāvīrassa tādino._  

</Paragraph>

</Division>
<Division number="3" book="ap1" bookNo="20" page="154">

<Paragraph number="1854" type="verses" book="ap1">
_Padenānupadāyantaṃ,_  
_nibbāpente mahājanaṃ;_  
_Uḷārā vitti me jātā,_  
_buddhe ādiccabandhune._  

</Paragraph>

</Division>
<Division number="4" book="ap1" bookNo="20" page="154">

<Paragraph number="1855" type="verses" book="ap1">
_Catunnavutito kappe,_  
_yaṃ dānamadadiṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_bhikkhādānassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="5" book="ap1" bookNo="20" page="154">

<Paragraph number="1856" type="verses" book="ap1">
_Sattāsītimhito kappe,_  
_mahāreṇu sanāmakā;_  
_Sattaratanasampannā,_  
_sattete cakkavattino._  

</Paragraph>

</Division>
<Division number="6" book="ap1" bookNo="20" page="154">

<Paragraph number="1857" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_vimokkhāpi ca aṭṭhime;_  
_Chaḷabhiññā sacchikatā,_  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="1858" book="ap1">
Itthaṃ sudaṃ āyasmā bhikkhadāyako thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="1859" type="center" book="ap1">
Bhikkhadāyakattherassāpadānaṃ paṭhamaṃ.
</Paragraph>
