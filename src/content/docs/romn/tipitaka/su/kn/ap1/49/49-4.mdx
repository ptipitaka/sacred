<Paragraph type="center" book="ap1">
* Ñāṇathavikattheraapadāna
</Paragraph>

<Division number="46" book="ap1" bookNo="20" page="433">

<Paragraph number="6236" type="verses" book="ap1">
_“Dakkhiṇe himavantassa,_  
_sukato assamo mama;_  
_Uttamatthaṃ gavesanto,_  
_vasāmi vipine tadā._  

</Paragraph>

</Division>
<Division number="47" book="ap1" bookNo="20" page="433">

<Paragraph number="6237" type="verses" book="ap1">
_Lābhālābhena santuṭṭho,_  
_mūlena ca phalena ca;_  
_Anvesanto ācariyaṃ,_  
_vasāmi ekako ahaṃ._  

</Paragraph>

</Division>
<Division number="48" book="ap1" bookNo="20" page="433">

<Paragraph number="6238" type="verses" book="ap1">
_Sumedho nāma sambuddho,_  
_loke uppajji tāvade;_  
_Catusaccaṃ pakāseti,_  
_uddharanto mahājanaṃ._  

</Paragraph>

</Division>
<Division number="49" book="ap1" bookNo="20" page="433">

<Paragraph number="6239" type="verses" book="ap1">
_Nāhaṃ suṇomi sambuddhaṃ,_  
_napi me koci saṃsati;_  
_Aṭṭhavasse atikkante,_  
_assosiṃ lokanāyakaṃ._  

</Paragraph>

</Division>
<Division number="50" book="ap1" bookNo="20" page="433">

<Paragraph number="6240" type="verses" book="ap1">
_Aggidāruṃ nīharitvā,_  
_sammajjitvāna assamaṃ;_  
_Khāribhāraṃ gahetvāna,_  
_nikkhamiṃ vipinā ahaṃ._  

</Paragraph>

</Division>
<Division number="51" book="ap1" bookNo="20" page="433">

<Paragraph number="6241" type="verses" book="ap1">
_Ekarattiṃ vasantohaṃ,_  
_gāmesu nigamesu ca;_  
_Anupubbena candavatiṃ,_  
_tadāhaṃ upasaṅkamiṃ._  

</Paragraph>

</Division>
<Division number="52" book="ap1" bookNo="20" page="433">

<Paragraph number="6242" type="verses" book="ap1">
_Bhagavā tamhi samaye,_  
_sumedho lokanāyako;_  
_Uddharanto bahū satte,_  
_deseti amataṃ padaṃ._  

</Paragraph>

</Division>
<Division number="53" book="ap1" bookNo="20" page="433">

<Paragraph number="6243" type="verses" book="ap1">
_Janakāyamatikkamma,_  
_vanditvā jinasāgaraṃ;_  
_Ekaṃsaṃ ajinaṃ katvā,_  
_santhaviṃ lokanāyakaṃ._  

</Paragraph>

</Division>
<Division number="54" book="ap1" bookNo="20" page="433">

<Paragraph number="6244" type="verses" book="ap1">
_‘Tuvaṃ satthā ca ketu ca,_  
_dhajo yūpo ca pāṇinaṃ;_  
_Parāyaṇo patiṭṭhā ca,_  
_dīpo ca dvipaduttamo._  


</Paragraph>
</Division>

<Paragraph number="6245" type="center" book="ap1">
Ekavīsatimaṃ bhāṇavāraṃ.


</Paragraph>

<Division number="55" book="ap1" bookNo="20" page="433">

<Paragraph number="6246" type="verses" book="ap1">
_Nepuñño dassane vīro,_  
_tāresi janataṃ tuvaṃ;_  
_Natthañño tārako loke,_  
_tavuttaritaro mune._  

</Paragraph>

</Division>
<Division number="56" book="ap1" bookNo="20" page="433">

<Paragraph number="6247" type="verses" book="ap1">
_Sakkā theve kusaggena,_  
_pametuṃ sāgaruttame;_  
_Na tveva tava sabbaññu,_  
_ñāṇaṃ sakkā pametave._  

</Paragraph>

</Division>
<Division number="57" book="ap1" bookNo="20" page="433">

<Paragraph number="6248" type="verses" book="ap1">
_Tuladaṇḍe ṭhapetvāna,_  
_mahiṃ sakkā dharetave;_  
_Na tveva tava paññāya,_  
_pamāṇamatthi cakkhuma._  

</Paragraph>

</Division>
<Division number="58" book="ap1" bookNo="20" page="433">

<Paragraph number="6249" type="verses" book="ap1">
_Ākāso minituṃ sakkā,_  
_rajjuyā aṅgulena vā;_  
_Na tveva tava sabbaññu,_  
_sīlaṃ sakkā pametave._  

</Paragraph>

</Division>
<Division number="59" book="ap1" bookNo="20" page="434">

<Paragraph number="6250" type="verses" book="ap1">
_Mahāsamudde udakaṃ,_  
_ākāso ca vasundharā;_  
_Parimeyyāni etāni,_  
_appameyyosi cakkhuma’._  

</Paragraph>

</Division>
<Division number="60" book="ap1" bookNo="20" page="434">

<Paragraph number="6251" type="verses" book="ap1">
_Chahi gāthāhi sabbaññuṃ,_  
_kittayitvā mahāyasaṃ;_  
_Añjaliṃ paggahetvāna,_  
_tuṇhī aṭṭhāsahaṃ tadā._  

</Paragraph>

</Division>
<Division number="61" book="ap1" bookNo="20" page="434">

<Paragraph number="6252" type="verses" book="ap1">
_Yaṃ vadanti sumedhoti,_  
_bhūripaññaṃ sumedhasaṃ;_  
_Bhikkhusaṃghe nisīditvā,_  
_imā gāthā abhāsatha._  

</Paragraph>

</Division>
<Division number="62" book="ap1" bookNo="20" page="434">

<Paragraph number="6253" type="verses" book="ap1">
_‘Yo me ñāṇaṃ pakittesi,_  
_vippasannena cetasā;_  
_Tamahaṃ kittayissāmi,_  
_suṇātha mama bhāsato._  

</Paragraph>

</Division>
<Division number="63" book="ap1" bookNo="20" page="434">

<Paragraph number="6254" type="verses" book="ap1">
_Sattasattati kappāni,_  
_devaloke ramissati;_  
_Sahassakkhattuṃ devindo,_  
_devarajjaṃ karissati._  

</Paragraph>

</Division>
<Division number="64" book="ap1" bookNo="20" page="434">

<Paragraph number="6255" type="verses" book="ap1">
_Anekasatakkhattuñca,_  
_cakkavattī bhavissati;_  
_Padesarajjaṃ vipulaṃ,_  
_gaṇanāto asaṅkhiyaṃ._  

</Paragraph>

</Division>
<Division number="65" book="ap1" bookNo="20" page="434">

<Paragraph number="6256" type="verses" book="ap1">
_Devabhūto manusso vā,_  
_puññakammasamāhito;_  
_Anūnamanasaṅkappo,_  
_tikkhapañño bhavissati’._  

</Paragraph>

</Division>
<Division number="66" book="ap1" bookNo="20" page="434">

<Paragraph number="6257" type="verses" book="ap1">
_Tiṃsakappasahassamhi,_  
_okkākakulasambhavo;_  
_Gotamo nāma gottena,_  
_satthā loke bhavissati._  

</Paragraph>

</Division>
<Division number="67" book="ap1" bookNo="20" page="434">

<Paragraph number="6258" type="verses" book="ap1">
_Agārā abhinikkhamma,_  
_pabbajissatikiñcano;_  
_Jātiyā sattavassena,_  
_arahattaṃ phusissati._  

</Paragraph>

</Division>
<Division number="68" book="ap1" bookNo="20" page="434">

<Paragraph number="6259" type="verses" book="ap1">
_Yato sarāmi attānaṃ,_  
_yato pattosmi sāsanaṃ;_  
_Etthantare na jānāmi,_  
_cetanaṃ amanoramaṃ._  

</Paragraph>

</Division>
<Division number="69" book="ap1" bookNo="20" page="434">

<Paragraph number="6260" type="verses" book="ap1">
_Saṃsaritvā bhave sabbe,_  
_sampattānubhaviṃ ahaṃ;_  
_Bhoge me ūnatā natthi,_  
_phalaṃ ñāṇassa thomane._  

</Paragraph>

</Division>
<Division number="70" book="ap1" bookNo="20" page="434">

<Paragraph number="6261" type="verses" book="ap1">
_Tiyaggī nibbutā mayhaṃ,_  
_bhavā sabbe samūhatā;_  
_Sabbāsavā parikkhīṇā,_  
_natthi dāni punabbhavo._  

</Paragraph>

</Division>
<Division number="71" book="ap1" bookNo="20" page="434">

<Paragraph number="6262" type="verses" book="ap1">
_Tiṃsakappasahassamhi,_  
_yaṃ ñāṇamathaviṃ ahaṃ;_  
_Duggatiṃ nābhijānāmi,_  
_phalaṃ ñāṇassa thomane._  

</Paragraph>

</Division>
<Division number="72" book="ap1" bookNo="20" page="435">

<Paragraph number="6263" type="verses" book="ap1">
_Kilesā jhāpitā mayhaṃ,_  
_ …pe… _  
_viharāmi anāsavo._  

</Paragraph>

</Division>
<Division number="73" book="ap1" bookNo="20" page="435">

<Paragraph number="6264" type="verses" book="ap1">
_Svāgataṃ vata me āsi,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ._  

</Paragraph>

</Division>
<Division number="74" book="ap1" bookNo="20" page="435">

<Paragraph number="6265" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="6266" book="ap1">
Itthaṃ sudaṃ āyasmā ñāṇathaviko thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="6267" type="center" book="ap1">
Ñāṇathavikattherassāpadānaṃ catutthaṃ.
