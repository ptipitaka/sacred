<Paragraph number="6887" type="verses" book="ap1">
_“Padumuttaro nāma jino,_  
_sabbadhammāna pāragū;_  
_Vuṭṭhahitvā samādhimhā,_  
_caṅkamī lokanāyako._  

</Paragraph>

</Division>
<Division number="75" book="ap1">

<Paragraph number="6888" type="verses" book="ap1">
_Khāribhāraṃ gahetvāna,_  
_āharanto phalaṃ tadā;_  
_Addasaṃ virajaṃ buddhaṃ,_  
_caṅkamantaṃ mahāmuniṃ._  

</Paragraph>

</Division>
<Division number="76" book="ap1">

<Paragraph number="6889" type="verses" book="ap1">
_Pasannacitto sumano,_  
_sire katvāna añjaliṃ;_  
_Sambuddhaṃ abhivādetvā,_  
_āmaṇḍamadadiṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="77" book="ap1">

<Paragraph number="6890" type="verses" book="ap1">
_Satasahassito kappe,_  
_yaṃ phalaṃ adadiṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_āmaṇḍassa idaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="78" book="ap1">

<Paragraph number="6891" type="verses" book="ap1">
_Kilesā jhāpitā mayhaṃ,_  
_ …pe… _  
_viharāmi anāsavo._  

</Paragraph>

</Division>
<Division number="79" book="ap1">

<Paragraph number="6892" type="verses" book="ap1">
_Svāgataṃ vata me āsi,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ._  

</Paragraph>

</Division>
<Division number="80" book="ap1">

<Paragraph number="6893" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="6894" book="ap1">
Itthaṃ sudaṃ āyasmā āmaṇḍaphaladāyako thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="6895" type="center" book="ap1">
