<Paragraph number="6215" type="verses" book="ap1">
_“Ogayha yaṃ pokkharaṇiṃ,_  
_nānākuñjarasevitaṃ;_  
_Uddharāmi bhisaṃ tattha,_  
_ghāsahetu ahaṃ tadā._  

</Paragraph>

</Division>
<Division number="28" book="ap1" bookNo="20" page="431">

<Paragraph number="6216" type="verses" book="ap1">
_Bhagavā tamhi samaye,_  
_padumuttarasavhayo;_  
_Rattambaradharo buddho,_  
_gacchati anilañjase._  

</Paragraph>

</Division>
<Division number="29" book="ap1" bookNo="20" page="431">

<Paragraph number="6217" type="verses" book="ap1">
_Dhunanto paṃsukūlāni,_  
_saddamassosahaṃ tadā;_  
_Uddhaṃ nijjhāyamānohaṃ,_  
_addasaṃ lokanāyakaṃ._  

</Paragraph>

</Division>
<Division number="30" book="ap1" bookNo="20" page="431">

<Paragraph number="6218" type="verses" book="ap1">
_Tattheva ṭhitako santo,_  
_āyāciṃ lokanāyakaṃ;_  
_‘Madhuṃ bhisehi savati,_  
_khīraṃ sappiṃ muḷālibhi._  

</Paragraph>

</Division>
<Division number="31" book="ap1" bookNo="20" page="431">

<Paragraph number="6219" type="verses" book="ap1">
_Paṭiggaṇhātu me buddho,_  
_anukampāya cakkhumā’;_  
_Tato kāruṇiko satthā,_  
_oruhitvā mahāyaso._  

</Paragraph>

</Division>
<Division number="32" book="ap1" bookNo="20" page="431">

<Paragraph number="6220" type="verses" book="ap1">
_Paṭiggaṇhi mamaṃ bhikkhaṃ,_  
_anukampāya cakkhumā;_  
_Paṭiggahetvā sambuddho,_  
_akā me anumodanaṃ._  

</Paragraph>

</Division>
<Division number="33" book="ap1" bookNo="20" page="431">

<Paragraph number="6221" type="verses" book="ap1">
_‘Sukhī hotu mahāpuñña,_  
_gati tuyhaṃ samijjhatu;_  
_Iminā bhisadānena,_  
_labhassu vipulaṃ sukhaṃ’._  

</Paragraph>

</Division>
<Division number="34" book="ap1" bookNo="20" page="432">

<Paragraph number="6222" type="verses" book="ap1">
_Idaṃ vatvāna sambuddho,_  
_jalajuttamanāmako;_  
_Bhikkhamādāya sambuddho,_  
_ambarenāgamā jino._  

</Paragraph>

</Division>
<Division number="35" book="ap1" bookNo="20" page="432">

<Paragraph number="6223" type="verses" book="ap1">
_Tato bhisaṃ gahetvāna,_  
_āgacchiṃ mama assamaṃ;_  
_Bhisaṃ rukkhe lagetvāna,_  
_mama dānamanussariṃ._  

</Paragraph>

</Division>
<Division number="36" book="ap1" bookNo="20" page="432">

<Paragraph number="6224" type="verses" book="ap1">
_Mahāvāto vuṭṭhahitvā,_  
_sañcālesi vanaṃ tadā;_  
_Ākāso abhinādittha,_  
_asaniyā phalantiyā._  

</Paragraph>

</Division>
<Division number="37" book="ap1" bookNo="20" page="432">

<Paragraph number="6225" type="verses" book="ap1">
_Tato me asanīpāto,_  
_matthake nipatī tadā;_  
_Sohaṃ nisinnako santo,_  
_tattha kālaṅkato ahuṃ._  

</Paragraph>

</Division>
<Division number="38" book="ap1" bookNo="20" page="432">

<Paragraph number="6226" type="verses" book="ap1">
_Puññakammena saṃyutto,_  
_tusitaṃ upapajjahaṃ;_  
_Kaḷevaraṃ me patitaṃ,_  
_devaloke ramiṃ ahaṃ._  

</Paragraph>

</Division>
<Division number="39" book="ap1" bookNo="20" page="432">

<Paragraph number="6227" type="verses" book="ap1">
_Chaḷasītisahassāni,_  
_nāriyo samalaṅkatā;_  
_Sāyaṃ pātaṃ upaṭṭhanti,_  
_bhisadānassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="40" book="ap1" bookNo="20" page="432">

<Paragraph number="6228" type="verses" book="ap1">
_Manussayonimāgantvā,_  
_sukhito homahaṃ sadā;_  
_Bhoge me ūnatā natthi,_  
_bhisadānassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="41" book="ap1" bookNo="20" page="432">

<Paragraph number="6229" type="verses" book="ap1">
_Anukampitako tena,_  
_devadevena tādinā;_  
_Sabbāsavā parikkhīṇā,_  
_natthi dāni punabbhavo._  

</Paragraph>

</Division>
<Division number="42" book="ap1" bookNo="20" page="432">

<Paragraph number="6230" type="verses" book="ap1">
_Satasahassito kappe,_  
_yaṃ bhisaṃ adadiṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_bhisadānassidaṃ phalaṃ._  

</Paragraph>

</Division>
