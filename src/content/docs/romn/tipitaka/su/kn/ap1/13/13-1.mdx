<Division number="1" book="ap1">

<Paragraph number="2054" type="verses" book="ap1">
_“Ajjhāyako mantadharo,_  
_tiṇṇaṃ vedāna pāragū;_  
_Abbhokāse ṭhito santo,_  
_addasaṃ lokanāyakaṃ._  

</Paragraph>

</Division>
<Division number="2" book="ap1">

<Paragraph number="2055" type="verses" book="ap1">
_Sīhaṃ yathā vanacaraṃ,_  
_byaggharājaṃva nittasaṃ;_  
_Tidhāpabhinnamātaṅgaṃ,_  
_kuñjaraṃva mahesinaṃ._  

</Paragraph>

</Division>
<Division number="3" book="ap1">

<Paragraph number="2056" type="verses" book="ap1">
_Sereyyakaṃ gahetvāna,_  
_ākāse ukkhipiṃ ahaṃ;_  
_Buddhassa ānubhāvena,_  
_parivārenti sabbaso._  

</Paragraph>

</Division>
<Division number="4" book="ap1">

<Paragraph number="2057" type="verses" book="ap1">
_Adhiṭṭhahi mahāvīro,_  
_sabbaññū lokanāyako;_  
_Samantā pupphacchadanā,_  
_okiriṃsu narāsabhaṃ._  

</Paragraph>

</Division>
<Division number="5" book="ap1">

<Paragraph number="2058" type="verses" book="ap1">
_Tato sā pupphakañcukā,_  
_antovaṇṭā bahimukhā;_  
_Sattāhaṃ chadanaṃ katvā,_  
_tato antaradhāyatha._  

</Paragraph>

</Division>
<Division number="6" book="ap1">

<Paragraph number="2059" type="verses" book="ap1">
_Tañca acchariyaṃ disvā,_  
_abbhutaṃ lomahaṃsanaṃ;_  
_Buddhe cittaṃ pasādesiṃ,_  
_sugate lokanāyake._  

</Paragraph>

</Division>
<Division number="7" book="ap1">

<Paragraph number="2060" type="verses" book="ap1">
_Tena cittappasādena,_  
_sukkamūlena codito;_  
_Kappānaṃ satasahassaṃ,_  
_duggatiṃ nupapajjahaṃ._  

</Paragraph>

</Division>
<Division number="8" book="ap1">

<Paragraph number="2061" type="verses" book="ap1">
_Pannarasasahassamhi,_  
_kappānaṃ pañcavīsati;_  
_Vītamalā samānā ca,_  
_cakkavattī mahabbalā._  

</Paragraph>

</Division>
<Division number="9" book="ap1">

<Paragraph number="2062" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_vimokkhāpi ca aṭṭhime;_  
_Chaḷabhiññā sacchikatā,_  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="2063" book="ap1">
Itthaṃ sudaṃ āyasmā sereyyako thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="2064" type="center" book="ap1">
Sereyyakattherassāpadānaṃ paṭhamaṃ.
</Paragraph>
