<Paragraph number="1399" type="verses" book="ap1">
_“Padumuttarabuddhassa,_  
_sabbalokānukampino;_  
_Sayanaṃ tassa pādāsiṃ,_  
_vippasannena cetasā._  

</Paragraph>

</Division>
<Division number="21" book="ap1" bookNo="20" page="117">

<Paragraph number="1400" type="verses" book="ap1">
_Tena sayanadānena,_  
_sukhette bījasampadā;_  
_Bhogā nibbattare tassa,_  
_sayanassa idaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="22" book="ap1" bookNo="20" page="117">

<Paragraph number="1401" type="verses" book="ap1">
_Ākāse seyyaṃ kappemi,_  
_dhāremi pathaviṃ imaṃ;_  
_Pāṇesu me issariyaṃ,_  
_sayanassa idaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="23" book="ap1" bookNo="20" page="117">

<Paragraph number="1402" type="verses" book="ap1">
_Pañcakappasahassamhi,_  
_aṭṭha āsuṃ mahātejā;_  
_Catuttiṃse kappasate,_  
_caturo ca mahabbalā._  

</Paragraph>

</Division>
<Division number="24" book="ap1" bookNo="20" page="117">

<Paragraph number="1403" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="1404" book="ap1">
Itthaṃ sudaṃ āyasmā sayanadāyako thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>
