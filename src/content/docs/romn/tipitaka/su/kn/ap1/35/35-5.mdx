<Division number="31" book="ap1">

<Paragraph number="3919" type="verses" book="ap1">
_“Himavantassāvidūre,_  
_hārito nāma pabbato;_  
_Sayambhū nārado nāma,_  
_rukkhamūle vasī tadā._  

</Paragraph>

</Division>
<Division number="32" book="ap1">

<Paragraph number="3920" type="verses" book="ap1">
_Naḷāgāraṃ karitvāna,_  
_tiṇena chādayiṃ ahaṃ;_  
_Caṅkamaṃ sodhayitvāna,_  
_sayambhussa adāsahaṃ._  

</Paragraph>

</Division>
<Division number="33" book="ap1">

<Paragraph number="3921" type="verses" book="ap1">
_Catuddasasu kappesu,_  
_devaloke ramiṃ ahaṃ;_  
_Catusattatikkhattuñca,_  
_devarajjaṃ akārayiṃ._  

</Paragraph>

</Division>
<Division number="34" book="ap1">

<Paragraph number="3922" type="verses" book="ap1">
_Catusattatikkhattuñca,_  
_cakkavattī ahosahaṃ;_  
_Padesarajjaṃ vipulaṃ,_  
_gaṇanāto asaṅkhiyaṃ._  

</Paragraph>

</Division>
<Division number="35" book="ap1">

<Paragraph number="3923" type="verses" book="ap1">
_Ubbiddhaṃ bhavanaṃ mayhaṃ,_  
_Indalaṭṭhīva uggataṃ;_  
_Sahassathambhaṃ atulaṃ,_  
_Vimānaṃ sapabhassaraṃ._  

</Paragraph>

</Division>
<Division number="36" book="ap1">

<Paragraph number="3924" type="verses" book="ap1">
_Dve sampattī anubhotvā,_  
_sukkamūlena codito;_  
_Gotamassa bhagavato,_  
_sāsane pabbajiṃ ahaṃ._  

</Paragraph>

</Division>
<Division number="37" book="ap1">

<Paragraph number="3925" type="verses" book="ap1">
_Padhānapahitattomhi,_  
_upasanto nirūpadhi;_  
_Nāgova bandhanaṃ chetvā,_  
_viharāmi anāsavo._  

</Paragraph>

</Division>
<Division number="38" book="ap1">

<Paragraph number="3926" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="3927" book="ap1">
Itthaṃ sudaṃ āyasmā naḷāgāriko thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="3928" type="center" book="ap1">
Naḷāgārikattherassāpadānaṃ pañcamaṃ.
</Paragraph>
