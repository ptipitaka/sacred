<Division number="37" book="ap1">

<Paragraph number="2872" type="verses" book="ap1">
_“Kusāṭakaṃ gahetvāna,_  
_upajjhāyassahaṃ pure;_  
_Mantañca anusikkhāmi,_  
_ganthādosassa pattiyā._  

</Paragraph>

</Division>
<Division number="38" book="ap1">

<Paragraph number="2873" type="verses" book="ap1">
_Addasaṃ virajaṃ buddhaṃ,_  
_Āhutīnaṃ paṭiggahaṃ;_  
_Usabhaṃ pavaraṃ aggaṃ,_  
_Tissaṃ buddhaṃ gaṇuttamaṃ._  

</Paragraph>

</Division>
<Division number="39" book="ap1">

<Paragraph number="2874" type="verses" book="ap1">
_Kusāṭakaṃ pattharitaṃ,_  
_akkamantaṃ naruttamaṃ;_  
_Samuggataṃ mahāvīraṃ,_  
_lokajeṭṭhaṃ narāsabhaṃ._  

</Paragraph>

</Division>
<Division number="40" book="ap1">

<Paragraph number="2875" type="verses" book="ap1">
_Disvā taṃ lokapajjotaṃ,_  
_vimalaṃ candasannibhaṃ;_  
_Avandiṃ satthuno pāde,_  
_vippasannena cetasā._  

</Paragraph>

</Division>
<Division number="41" book="ap1">

<Paragraph number="2876" type="verses" book="ap1">
_Catunnavutito kappe,_  
_yaṃ adāsiṃ kusāṭakaṃ;_  
_Duggatiṃ nābhijānāmi,_  
_kusāṭakassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="42" book="ap1">

<Paragraph number="2877" type="verses" book="ap1">
_Sattatiṃse ito kappe,_  
_eko āsiṃ janādhipo;_  
_Sunando nāma nāmena,_  
_cakkavattī mahabbalo._  

</Paragraph>

</Division>
<Division number="43" book="ap1">

<Paragraph number="2878" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="2879" book="ap1">
Itthaṃ sudaṃ āyasmā akkantasaññako thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="2880" type="center" book="ap1">
Akkantasaññakattherassāpadānaṃ aṭṭhamaṃ.
</Paragraph>
