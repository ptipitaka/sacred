<Division number="101" book="ap1">

<Paragraph number="1325" type="verses" book="ap1">
_“Padumuttaro lokavidū,_  
_āhutīnaṃ paṭiggaho;_  
_Janataṃ uddharitvāna,_  
_nibbāyati mahāyaso._  

</Paragraph>

</Division>
<Division number="102" book="ap1">

<Paragraph number="1326" type="verses" book="ap1">
_Nibbāyante ca sambuddhe,_  
_dasasahassi kampatha;_  
_Janakāyo mahā āsi,_  
_devā sannipatuṃ tadā._  

</Paragraph>

</Division>
<Division number="103" book="ap1">

<Paragraph number="1327" type="verses" book="ap1">
_Candanaṃ pūrayitvāna,_  
_tagarāmallikāhi ca;_  
_Haṭṭho haṭṭhena cittena,_  
_āropayiṃ naruttamaṃ._  

</Paragraph>

</Division>
<Division number="104" book="ap1">

<Paragraph number="1328" type="verses" book="ap1">
_Mama saṅkappamaññāya,_  
_satthā loke anuttaro;_  
_Nipannakova sambuddho,_  
_imā gāthā abhāsatha._  

</Paragraph>

</Division>
<Division number="105" book="ap1">

<Paragraph number="1329" type="verses" book="ap1">
_‘Yo me pacchimake kāle,_  
_gandhamālena chādayi;_  
_Tamahaṃ kittayissāmi,_  
_suṇātha mama bhāsato._  

</Paragraph>

</Division>
<Division number="106" book="ap1">

<Paragraph number="1330" type="verses" book="ap1">
_Ito cuto ayaṃ poso,_  
_tusitakāyaṃ gamissati;_  
_Tattha rajjaṃ karitvāna,_  
_nimmānaṃ so gamissati._  

</Paragraph>

</Division>
<Division number="107" book="ap1">

<Paragraph number="1331" type="verses" book="ap1">
_Eteneva upāyena,_  
_datvā mālaṃ varuttamaṃ;_  
_Sakakammābhiraddho so,_  
_sampattiṃ anubhossati._  

</Paragraph>

</Division>
<Division number="108" book="ap1">

<Paragraph number="1332" type="verses" book="ap1">
_Punāpi tusite kāye,_  
_nibbattissatiyaṃ naro;_  
_Tamhā kāyā cavitvāna,_  
_manussattaṃ gamissati._  

</Paragraph>

</Division>
<Division number="109" book="ap1">

<Paragraph number="1333" type="verses" book="ap1">
_Sakyaputto mahānāgo,_  
_aggo loke sadevake;_  
_Bodhayitvā bahū satte,_  
_nibbāyissati cakkhumā._  

</Paragraph>

</Division>
<Division number="110" book="ap1">

<Paragraph number="1334" type="verses" book="ap1">
_Tadā sopagato santo,_  
_sukkamūlena codito;_  
_Upasaṅkamma sambuddhaṃ,_  
_pañhaṃ pucchissati tadā._  

</Paragraph>

</Division>
<Division number="111" book="ap1">

<Paragraph number="1335" type="verses" book="ap1">
_Hāsayitvāna sambuddho,_  
_sabbaññū lokanāyako;_  
_Puññakammaṃ pariññāya,_  
_saccāni vivarissati._  

</Paragraph>

</Division>
<Division number="112" book="ap1">

<Paragraph number="1336" type="verses" book="ap1">
_Āraddho ca ayaṃ pañho,_  
_tuṭṭho ekaggamānaso;_  
_Satthāraṃ abhivādetvā,_  
_pabbajjaṃ yācayissati._  

</Paragraph>

</Division>
<Division number="113" book="ap1">

<Paragraph number="1337" type="verses" book="ap1">
_Pasannamānasaṃ disvā,_  
_sakakammena tositaṃ;_  
_Pabbājessati so buddho,_  
_aggamaggassa kovido._  

</Paragraph>

</Division>
<Division number="114" book="ap1">

<Paragraph number="1338" type="verses" book="ap1">
_Vāyamitvānayaṃ poso,_  
_sammāsambuddhasāsane;_  
_Sabbāsave pariññāya,_  
_nibbāyissatināsavo’._  


</Paragraph>
</Division>

<Paragraph number="1339" type="center" book="ap1">
Pañcamabhāṇavāraṃ.


</Paragraph>

<Division number="115" book="ap1">

<Paragraph number="1340" type="verses" book="ap1">
_Pubbakammena saṃyutto,_  
_ekaggo susamāhito;_  
_Buddhassa oraso putto,_  
_dhammajomhi sunimmito._  

</Paragraph>

</Division>
<Division number="116" book="ap1">

<Paragraph number="1341" type="verses" book="ap1">
_Dhammarājaṃ upagamma,_  
_apucchiṃ pañhamuttamaṃ;_  
_Kathayanto ca me pañhaṃ,_  
_dhammasotaṃ upānayi._  

</Paragraph>

</Division>
<Division number="117" book="ap1">

<Paragraph number="1342" type="verses" book="ap1">
_Tassāhaṃ dhammamaññāya,_  
_vihāsiṃ sāsane rato;_  
_Sabbāsave pariññāya,_  
_viharāmi anāsavo._  

</Paragraph>

</Division>
<Division number="118" book="ap1">

<Paragraph number="1343" type="verses" book="ap1">
_Satasahassito kappe,_  
_jalajuttamanāyako;_  
_Nibbāyi anupādāno,_  
_dīpova telasaṅkhayā._  

</Paragraph>

</Division>
<Division number="119" book="ap1">

<Paragraph number="1344" type="verses" book="ap1">
_Sattayojanikaṃ āsi,_  
_thūpañca ratanāmayaṃ;_  
_Dhajaṃ tattha apūjesiṃ,_  
_sabbabhaddaṃ manoramaṃ._  

</Paragraph>

</Division>
<Division number="120" book="ap1">

<Paragraph number="1345" type="verses" book="ap1">
_Kassapassa ca buddhassa,_  
_tisso nāmaggasāvako;_  
_Putto me oraso āsi,_  
_dāyādo jinasāsane._  

</Paragraph>

</Division>
<Division number="121" book="ap1">

<Paragraph number="1346" type="verses" book="ap1">
_Tassa hīnena manasā,_  
_vācaṃ bhāsiṃ abhaddakaṃ;_  
_Tena kammavipākena,_  
_pacchā me āsi bhaddakaṃ._  

</Paragraph>

</Division>
<Division number="122" book="ap1">

<Paragraph number="1347" type="verses" book="ap1">
_Upavattane sālavane,_  
_pacchime sayane muni;_  
_Pabbājesi mahāvīro,_  
_hito kāruṇiko jino._  

</Paragraph>

</Division>
<Division number="123" book="ap1">

<Paragraph number="1348" type="verses" book="ap1">
_Ajjeva dāni pabbajjā,_  
_ajjeva upasampadā;_  
_Ajjeva parinibbānaṃ,_  
_sammukhā dvipaduttame._  

</Paragraph>

</Division>
<Division number="124" book="ap1">

<Paragraph number="1349" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="1350" book="ap1">
Itthaṃ sudaṃ āyasmā subhaddo thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="1351" type="center" book="ap1">
Subhaddattherassāpadānaṃ navamaṃ.
</Paragraph>
