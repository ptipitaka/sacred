<Division number="11" book="ap1">

<Paragraph number="3372" type="verses" book="ap1">
_“Putto mama pabbajito,_  
_kāsāyavasano tadā;_  
_So ca buddhattaṃ sampatto,_  
_nibbuto lokapūjito._  

</Paragraph>

</Division>
<Division number="12" book="ap1">

<Paragraph number="3373" type="verses" book="ap1">
_Vicinanto sakaṃ puttaṃ,_  
_agamaṃ pacchato ahaṃ;_  
_Nibbutassa mahantassa,_  
_citakaṃ agamāsahaṃ._  

</Paragraph>

</Division>
<Division number="13" book="ap1">

<Paragraph number="3374" type="verses" book="ap1">
_Paggayha añjaliṃ tattha,_  
_vanditvā citakaṃ ahaṃ;_  
_Setacchattañca paggayha,_  
_āropesiṃ ahaṃ tadā._  

</Paragraph>

</Division>
<Division number="14" book="ap1">

<Paragraph number="3375" type="verses" book="ap1">
_Catunnavutito kappe,_  
_yaṃ chattamabhiropayiṃ;_  
_Duggatiṃ nābhijānāmi,_  
_chattadānassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="15" book="ap1">

<Paragraph number="3376" type="verses" book="ap1">
_Pañcavīse ito kappe,_  
_satta āsuṃ janādhipā;_  
_Mahārahasanāmā te,_  
_cakkavattī mahabbalā._  

</Paragraph>

</Division>
<Division number="16" book="ap1">

<Paragraph number="3377" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="3378" book="ap1">
Itthaṃ sudaṃ āyasmā chattadāyako thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="3379" type="center" book="ap1">
Chattadāyakattherassāpadānaṃ tatiyaṃ.
</Paragraph>
