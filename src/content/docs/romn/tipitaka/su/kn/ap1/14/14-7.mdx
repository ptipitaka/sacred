<Paragraph number="2240" type="verses" book="ap1">
_“Visālamāḷe āsīno,_  
_addasaṃ lokanāyakaṃ;_  
_Khīṇāsavaṃ balappattaṃ,_  
_bhikkhusaṃghapurakkhataṃ._  

</Paragraph>

</Division>
<Division number="48" book="ap1" bookNo="20" page="185">

<Paragraph number="2241" type="verses" book="ap1">
_Satasahassā tevijjā,_  
_chaḷabhiññā mahiddhikā;_  
_Parivārenti sambuddhaṃ,_  
_ko disvā nappasīdati._  

</Paragraph>

</Division>
<Division number="49" book="ap1" bookNo="20" page="185">

<Paragraph number="2242" type="verses" book="ap1">
_Ñāṇe upanidhā yassa,_  
_na vijjati sadevake;_  
_Anantañāṇaṃ sambuddhaṃ,_  
_ko disvā nappasīdati._  

</Paragraph>

</Division>
<Division number="50" book="ap1" bookNo="20" page="185">

<Paragraph number="2243" type="verses" book="ap1">
_Dhammakāyañca dīpentaṃ,_  
_kevalaṃ ratanākaraṃ;_  
_Vikappetuṃ na sakkonti,_  
_ko disvā nappasīdati._  

</Paragraph>

</Division>
<Division number="51" book="ap1" bookNo="20" page="185">

<Paragraph number="2244" type="verses" book="ap1">
_Imāhi tīhi gāthāhi,_  
_nāradovhayavacchalo;_  
_Padumuttaraṃ thavitvāna,_  
_sambuddhaṃ aparājitaṃ._  

</Paragraph>

</Division>
<Division number="52" book="ap1" bookNo="20" page="185">

<Paragraph number="2245" type="verses" book="ap1">
_Tena cittappasādena,_  
_buddhasanthavanena ca;_  
_Kappānaṃ satasahassaṃ,_  
_duggatiṃ nupapajjahaṃ._  

</Paragraph>

</Division>
<Division number="53" book="ap1" bookNo="20" page="185">

<Paragraph number="2246" type="verses" book="ap1">
_Ito tiṃsakappasate,_  
_sumitto nāma khattiyo;_  
_Sattaratanasampanno,_  
_cakkavattī mahabbalo._  

</Paragraph>

</Division>
<Division number="54" book="ap1" bookNo="20" page="185">

<Paragraph number="2247" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="2248" book="ap1">
Itthaṃ sudaṃ āyasmā atthasandassako thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="2249" type="center" book="ap1">
Atthasandassakattherassāpadānaṃ sattamaṃ.
</Paragraph>
