<Division number="112" book="ap1">

<Paragraph number="823" type="verses" book="ap1">
_“Pabbhāraṃ sodhayantassa,_  
_vipine pabbatuttame;_  
_Siddhattho nāma bhagavā,_  
_āgacchi mama santikaṃ._  

</Paragraph>

</Division>
<Division number="113" book="ap1">

<Paragraph number="824" type="verses" book="ap1">
_Buddhaṃ upagataṃ disvā,_  
_lokajeṭṭhassa tādino;_  
_Santharaṃ paññapetvāna,_  
_pupphāsanamadāsahaṃ._  

</Paragraph>

</Division>
<Division number="114" book="ap1">

<Paragraph number="825" type="verses" book="ap1">
_Pupphāsane nisīditvā,_  
_siddhattho lokanāyako;_  
_Mamañca gatimaññāya,_  
_aniccatamudāhari._  

</Paragraph>

</Division>
<Division number="115" book="ap1">

<Paragraph number="826" type="verses" book="ap1">
_‘Aniccā vata saṅkhārā,_  
_uppādavayadhammino;_  
_Uppajjitvā nirujjhanti,_  
_tesaṃ vūpasamo sukho’._  

</Paragraph>

</Division>
<Division number="116" book="ap1">

<Paragraph number="827" type="verses" book="ap1">
_Idaṃ vatvāna sabbaññū,_  
_lokajeṭṭho narāsabho;_  
_Nabhaṃ abbhuggami vīro,_  
_haṃsarājāva ambare._  

</Paragraph>

</Division>
<Division number="117" book="ap1">

<Paragraph number="828" type="verses" book="ap1">
_Sakaṃ diṭṭhiṃ jahitvāna,_  
_bhāvayāniccasaññahaṃ;_  
_Ekāhaṃ bhāvayitvāna,_  
_tattha kālaṃ kato ahaṃ._  

</Paragraph>

</Division>
<Division number="118" book="ap1">

<Paragraph number="829" type="verses" book="ap1">
_Dve sampattī anubhotvā,_  
_sukkamūlena codito;_  
_Pacchime bhave sampatte,_  
_sapākayonupāgamiṃ._  

</Paragraph>

</Division>
<Division number="119" book="ap1">

<Paragraph number="830" type="verses" book="ap1">
_Agārā abhinikkhamma,_  
_pabbajiṃ anagāriyaṃ;_  
_Jātiyā sattavassohaṃ,_  
_arahattamapāpuṇiṃ._  

</Paragraph>

</Division>
<Division number="120" book="ap1">

<Paragraph number="831" type="verses" book="ap1">
_Āraddhavīriyo pahitatto,_  
_Sīlesu susamāhito;_  
_Tosetvāna mahānāgaṃ,_  
_Alatthaṃ upasampadaṃ._  

</Paragraph>

</Division>
<Division number="121" book="ap1">

<Paragraph number="832" type="verses" book="ap1">
_Catunnavutito kappe,_  
_yaṃ kammamakariṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_pupphadānassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="122" book="ap1">

<Paragraph number="833" type="verses" book="ap1">
_Catunnavutito kappe,_  
_yaṃ saññaṃ bhāvayiṃ tadā;_  
_Taṃ saññaṃ bhāvayantassa,_  
_patto me āsavakkhayo._  

</Paragraph>

</Division>
<Division number="123" book="ap1">

<Paragraph number="834" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_vimokkhāpi ca aṭṭhime;_  
_Chaḷabhiññā sacchikatā,_  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="835" book="ap1">
Itthaṃ sudaṃ āyasmā sopāko thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="836" type="center" book="ap1">
Sopākattherassāpadānaṃ navamaṃ.
</Paragraph>
