<Paragraph number="4659" type="verses" book="ap1">
_“Piyadassī nāma bhagavā,_  
_sayambhū lokanāyako;_  
_Vivekakāmo sambuddho,_  
_samādhikusalo muni._  

</Paragraph>

</Division>
<Division number="354" book="ap1" bookNo="20" page="382">

<Paragraph number="4660" type="verses" book="ap1">
_Vanasaṇḍaṃ samogayha,_  
_piyadassī mahāmuni;_  
_Paṃsukūlaṃ pattharitvā,_  
_nisīdi purisuttamo._  

</Paragraph>

</Division>
<Division number="355" book="ap1" bookNo="20" page="382">

<Paragraph number="4661" type="verses" book="ap1">
_Migaluddo pure āsiṃ,_  
_araññe kānane ahaṃ;_  
_Pasadaṃ migamesanto,_  
_āhiṇḍāmi ahaṃ tadā._  

</Paragraph>

</Division>
<Division number="356" book="ap1" bookNo="20" page="382">

<Paragraph number="4662" type="verses" book="ap1">
_Tatthaddasāsiṃ sambuddhaṃ,_  
_oghatiṇṇamanāsavaṃ;_  
_Pupphitaṃ sālarājaṃva,_  
_sataraṃsiṃva uggataṃ._  

</Paragraph>

</Division>
<Division number="357" book="ap1" bookNo="20" page="382">

<Paragraph number="4663" type="verses" book="ap1">
_Disvānahaṃ devadevaṃ,_  
_piyadassiṃ mahāyasaṃ;_  
_Jātassaraṃ samogayha,_  
_padumaṃ āhariṃ tadā._  

</Paragraph>

</Division>
<Division number="358" book="ap1" bookNo="20" page="382">

<Paragraph number="4664" type="verses" book="ap1">
_Āharitvāna padumaṃ,_  
_satapattaṃ manoramaṃ;_  
_Kūṭāgāraṃ karitvāna,_  
_chādayiṃ padumenahaṃ._  

</Paragraph>

</Division>
<Division number="359" book="ap1" bookNo="20" page="382">

<Paragraph number="4665" type="verses" book="ap1">
_Anukampako kāruṇiko,_  
_piyadassī mahāmuni;_  
_Sattarattindivaṃ buddho,_  
_kūṭāgāre vasī jino._  

</Paragraph>

</Division>
<Division number="360" book="ap1" bookNo="20" page="382">

<Paragraph number="4666" type="verses" book="ap1">
_Purāṇaṃ chaḍḍayitvāna,_  
_navena chādayiṃ ahaṃ;_  
_Añjaliṃ paggahetvāna,_  
_aṭṭhāsiṃ tāvade ahaṃ._  

</Paragraph>

</Division>
<Division number="361" book="ap1" bookNo="20" page="383">

<Paragraph number="4667" type="verses" book="ap1">
_Vuṭṭhahitvā samādhimhā,_  
_piyadassī mahāmuni;_  
_Disaṃ anuvilokento,_  
_nisīdi lokanāyako._  

</Paragraph>

</Division>
<Division number="362" book="ap1" bookNo="20" page="383">

<Paragraph number="4668" type="verses" book="ap1">
_Tadā sudassano nāma,_  
_upaṭṭhāko mahiddhiko;_  
_Cittamaññāya buddhassa,_  
_piyadassissa satthuno._  

</Paragraph>

</Division>
<Division number="363" book="ap1" bookNo="20" page="383">

<Paragraph number="4669" type="verses" book="ap1">
_Asītiyā sahassehi,_  
_bhikkhūhi parivārito;_  
_Vanante sukhamāsīnaṃ,_  
_upesi lokanāyakaṃ._  

</Paragraph>

</Division>
<Division number="364" book="ap1" bookNo="20" page="383">

<Paragraph number="4670" type="verses" book="ap1">
_Yāvatā vanasaṇḍamhi,_  
_adhivatthā ca devatā;_  
_Buddhassa cittamaññāya,_  
_sabbe sannipatuṃ tadā._  

</Paragraph>

</Division>
<Division number="365" book="ap1" bookNo="20" page="383">

<Paragraph number="4671" type="verses" book="ap1">
_Samāgatesu yakkhesu,_  
_kumbhaṇḍe saharakkhase;_  
_Bhikkhusaṃghe ca sampatte,_  
_gāthā pabyāharī jino._  

</Paragraph>

</Division>
<Division number="366" book="ap1" bookNo="20" page="383">

<Paragraph number="4672" type="verses" book="ap1">
_‘Yo maṃ sattāhaṃ pūjesi,_  
_āvāsañca akāsi me;_  
_Tamahaṃ kittayissāmi,_  
_suṇātha mama bhāsato._  

</Paragraph>

</Division>
<Division number="367" book="ap1" bookNo="20" page="383">

<Paragraph number="4673" type="verses" book="ap1">
_Sududdasaṃ sunipuṇaṃ,_  
_gambhīraṃ suppakāsitaṃ;_  
_Ñāṇena kittayissāmi,_  
_suṇātha mama bhāsato._  

</Paragraph>

</Division>
<Division number="368" book="ap1" bookNo="20" page="383">

<Paragraph number="4674" type="verses" book="ap1">
_Catuddasāni kappāni,_  
_devarajjaṃ karissati;_  
_Kūṭāgāraṃ mahantassa,_  
_padmapupphehi chāditaṃ._  

</Paragraph>

</Division>
<Division number="369" book="ap1" bookNo="20" page="383">

<Paragraph number="4675" type="verses" book="ap1">
_Ākāse dhārayissati,_  
_Pupphakammassidaṃ phalaṃ;_  
_Catubbīse kappasate,_  
_Vokiṇṇaṃ saṃsarissati._  

</Paragraph>

</Division>
<Division number="370" book="ap1" bookNo="20" page="383">

<Paragraph number="4676" type="verses" book="ap1">
_Tattha pupphamayaṃ byamhaṃ,_  
_Ākāse dhārayissati;_  
_Yathā padumapattamhi,_  
_Toyaṃ na upalimpati._  

</Paragraph>

</Division>
<Division number="371" book="ap1" bookNo="20" page="383">

<Paragraph number="4677" type="verses" book="ap1">
_Tathevīmassa ñāṇamhi,_  
_kilesā nopalimpare;_  
_Manasā vinivaṭṭetvā,_  
_pañca nīvaraṇe ayaṃ._  

</Paragraph>

</Division>
<Division number="372" book="ap1" bookNo="20" page="383">

<Paragraph number="4678" type="verses" book="ap1">
_Cittaṃ janetvā nekkhamme,_  
_agārā pabbajissati;_  
_Tato pupphamaye byamhe,_  
_dhārente nikkhamissati._  

</Paragraph>

</Division>
<Division number="373" book="ap1" bookNo="20" page="383">

<Paragraph number="4679" type="verses" book="ap1">
_Rukkhamūle vasantassa,_  
_Nipakassa satīmato;_  
_Tattha pupphamayaṃ byamhaṃ,_  
_Matthake dhārayissati._  

</Paragraph>

</Division>
<Division number="374" book="ap1" bookNo="20" page="384">

<Paragraph number="4680" type="verses" book="ap1">
_Cīvaraṃ piṇḍapātañca,_  
_paccayaṃ sayanāsanaṃ;_  
_Datvāna bhikkhusaṃghassa,_  
_nibbāyissatināsavo’._  

</Paragraph>

</Division>
<Division number="375" book="ap1" bookNo="20" page="384">

<Paragraph number="4681" type="verses" book="ap1">
_Kūṭāgārena caratā,_  
_pabbajjaṃ abhinikkhamiṃ;_  
_Rukkhamūle vasantampi,_  
_kūṭāgāraṃ dharīyati._  

</Paragraph>

</Division>
<Division number="376" book="ap1" bookNo="20" page="384">

<Paragraph number="4682" type="verses" book="ap1">
_Cīvare piṇḍapāte ca,_  
_cetanā me na vijjati;_  
_Puññakammena saṃyutto,_  
_labhāmi pariniṭṭhitaṃ._  

</Paragraph>

</Division>
<Division number="377" book="ap1" bookNo="20" page="384">

<Paragraph number="4683" type="verses" book="ap1">
_Gaṇanāto asaṅkheyyā,_  
_kappakoṭī bahū mama;_  
_Rittakā te atikkantā,_  
_pamuttā lokanāyakā._  

</Paragraph>

</Division>
<Division number="378" book="ap1" bookNo="20" page="384">

<Paragraph number="4684" type="verses" book="ap1">
_Aṭṭhārase kappasate,_  
_piyadassī vināyako;_  
_Tamahaṃ payirupāsitvā,_  
_imaṃ yoniṃ upāgato._  

</Paragraph>

</Division>
<Division number="379" book="ap1" bookNo="20" page="384">

<Paragraph number="4685" type="verses" book="ap1">
_Idha passāmi sambuddhaṃ,_  
_anomaṃ nāma cakkhumaṃ;_  
_Tamahaṃ upagantvāna,_  
_pabbajiṃ anagāriyaṃ._  

</Paragraph>

</Division>
<Division number="380" book="ap1" bookNo="20" page="384">

<Paragraph number="4686" type="verses" book="ap1">
_Dukkhassantakaro buddho,_  
_maggaṃ me desayī jino;_  
_Tassa dhammaṃ suṇitvāna,_  
_pattomhi acalaṃ padaṃ._  

</Paragraph>

</Division>
<Division number="381" book="ap1" bookNo="20" page="384">

<Paragraph number="4687" type="verses" book="ap1">
_Tosayitvāna sambuddhaṃ,_  
_gotamaṃ sakyapuṅgavaṃ;_  
_Sabbāsave pariññāya,_  
_viharāmi anāsavo._  

</Paragraph>

</Division>
<Division number="382" book="ap1" bookNo="20" page="384">

<Paragraph number="4688" type="verses" book="ap1">
_Aṭṭhārase kappasate,_  
_yaṃ buddhamabhipūjayiṃ;_  
_Duggatiṃ nābhijānāmi,_  
_buddhapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="383" book="ap1" bookNo="20" page="384">

<Paragraph number="4689" type="verses" book="ap1">
_Kilesā jhāpitā mayhaṃ,_  
_bhavā sabbe samūhatā;_  
_Sabbāsavā parikkhīṇā,_  
_natthi dāni punabbhavo._  

</Paragraph>

</Division>
<Division number="384" book="ap1" bookNo="20" page="384">

_ …pe… _  
_kataṃ buddhassa sāsanaṃ._  

</Paragraph>

</Division>

_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="4692" book="ap1">
Itthaṃ sudaṃ āyasmā padumakūṭāgāriyo thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>
