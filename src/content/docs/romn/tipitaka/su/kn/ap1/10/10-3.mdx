<Paragraph number="1782" type="verses" book="ap1">
_“Tissassāhaṃ bhagavato,_  
_upaḍḍhadussamadāsahaṃ;_  
_Paramakāpaññapattomhi,_  
_duggatena samappito._  

</Paragraph>

</Division>
<Division number="15" book="ap1" bookNo="20" page="148">

<Paragraph number="1783" type="verses" book="ap1">
_Upaḍḍhadussaṃ datvāna,_  
_kappaṃ saggamhi modahaṃ;_  
_Avasesesu kappesu,_  
_kusalaṃ kāritaṃ mayā._  

</Paragraph>

</Division>
<Division number="16" book="ap1" bookNo="20" page="148">

<Paragraph number="1784" type="verses" book="ap1">
_Dvenavute ito kappe,_  
_yaṃ dussamadadiṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_dussadānassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="17" book="ap1" bookNo="20" page="148">

<Paragraph number="1785" type="verses" book="ap1">
_Ekūnapaññāsakappamhi,_  
_rājāno cakkavattino;_  
_Samantacchadanā nāma,_  
_bāttiṃsāsuṃ janādhipā._  

</Paragraph>

</Division>
<Division number="18" book="ap1" bookNo="20" page="148">

<Paragraph number="1786" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="1787" book="ap1">
Itthaṃ sudaṃ āyasmā aḍḍhaceḷako thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

</Paragraph>
