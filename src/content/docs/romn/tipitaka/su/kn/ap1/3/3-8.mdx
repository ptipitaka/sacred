<Paragraph number="1042" type="verses" book="ap1">
_“Candabhāgānadītīre,_  
_susumāro ahaṃ tadā;_  
_Sagocarappasutohaṃ,_  
_nadītitthaṃ agacchahaṃ._  

</Paragraph>

</Division>
<Division number="170" book="ap1" bookNo="20" page="88">

<Paragraph number="1043" type="verses" book="ap1">
_Siddhattho tamhi samaye,_  
_sayambhū aggapuggalo;_  
_Nadiṃ taritukāmo so,_  
_nadītitthaṃ upāgami._  

</Paragraph>

</Division>
<Division number="171" book="ap1" bookNo="20" page="88">

<Paragraph number="1044" type="verses" book="ap1">
_Upāgate ca sambuddhe,_  
_ahampi tatthupāgamiṃ;_  
_Upagantvāna sambuddhaṃ,_  
_imaṃ vācaṃ udīrayiṃ._  

</Paragraph>

</Division>
<Division number="172" book="ap1" bookNo="20" page="88">

<Paragraph number="1045" type="verses" book="ap1">
_‘Abhirūha mahāvīra,_  
_tāressāmi ahaṃ tuvaṃ;_  
_Pettikaṃ visayaṃ mayhaṃ,_  
_anukampa mahāmuni’._  

</Paragraph>

</Division>
<Division number="173" book="ap1" bookNo="20" page="88">

<Paragraph number="1046" type="verses" book="ap1">
_Mama uggajjanaṃ sutvā,_  
_abhirūhi mahāmuni;_  
_Haṭṭho haṭṭhena cittena,_  
_tāresiṃ lokanāyakaṃ._  

</Paragraph>

</Division>
<Division number="174" book="ap1" bookNo="20" page="88">

<Paragraph number="1047" type="verses" book="ap1">
_Nadiyā pārime tīre,_  
_siddhattho lokanāyako;_  
_Assāsesi mamaṃ tattha,_  
_amataṃ pāpuṇissasi._  

</Paragraph>

</Division>
<Division number="175" book="ap1" bookNo="20" page="88">

<Paragraph number="1048" type="verses" book="ap1">
_Tamhā kāyā cavitvāna,_  
_devalokaṃ āgacchahaṃ;_  
_Dibbasukhaṃ anubhaviṃ,_  
_accharāhi purakkhato._  

</Paragraph>

</Division>
<Division number="176" book="ap1" bookNo="20" page="89">

<Paragraph number="1049" type="verses" book="ap1">
_Sattakkhattuñca devindo,_  
_devarajjamakāsahaṃ;_  
_Tīṇikkhattuṃ cakkavattī,_  
_mahiyā issaro ahuṃ._  

</Paragraph>

</Division>
<Division number="177" book="ap1" bookNo="20" page="89">

<Paragraph number="1050" type="verses" book="ap1">
_Vivekamanuyuttohaṃ,_  
_nipako ca susaṃvuto;_  
_Dhāremi antimaṃ dehaṃ,_  
_sammāsambuddhasāsane._  

</Paragraph>

</Division>
<Division number="178" book="ap1" bookNo="20" page="89">

<Paragraph number="1051" type="verses" book="ap1">
_Catunnavutito kappe,_  
_tāresiṃ yaṃ narāsabhaṃ;_  
_Duggatiṃ nābhijānāmi,_  
_taraṇāya idaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="179" book="ap1" bookNo="20" page="89">

<Paragraph number="1052" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_vimokkhāpi ca aṭṭhime;_  
_Chaḷabhiññā sacchikatā,_  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="1053" book="ap1">
Itthaṃ sudaṃ āyasmā uttiyo thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="1054" type="center" book="ap1">
Uttiyattherassāpadānaṃ aṭṭhamaṃ.
</Paragraph>
