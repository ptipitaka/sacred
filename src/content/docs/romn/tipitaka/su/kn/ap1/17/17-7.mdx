_“Candabhāgānadītīre,_  
_anusotaṃ vajāmahaṃ;_  
_Sayambhuṃ addasaṃ tattha,_  
_sālarājaṃva phullitaṃ._

<Division number="34" book="ap1" bookNo="20" page="204">

<Paragraph number="2478" type="verses" book="ap1">
_Pupphaṃ yūthikamādāya,_  
_upagacchiṃ mahāmuniṃ;_  
_Pasannacitto sumano,_  
_buddhassa abhiropayiṃ._  

</Paragraph>

</Division>
<Division number="35" book="ap1" bookNo="20" page="204">

<Paragraph number="2479" type="verses" book="ap1">
_Catunnavutito kappe,_  
_yaṃ pupphamabhiropayiṃ;_  
_Duggatiṃ nābhijānāmi,_  
_buddhapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="36" book="ap1" bookNo="20" page="204">

<Paragraph number="2480" type="verses" book="ap1">
_Sattasaṭṭhimhito kappe,_  
_eko sāmuddharo ahuṃ;_  
_Sattaratanasampanno,_  
_cakkavattī mahabbalo._  

</Paragraph>

</Division>
<Division number="37" book="ap1" bookNo="20" page="204">

<Paragraph number="2481" type="verses" book="ap1">

_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="2482" book="ap1">

</Paragraph>
