<Paragraph number="696" type="verses" book="ap1">
_“Nibbute lokanāthamhi,_  
_siddhatthe dvipaduttame;_  
_Vitthārike pāvacane,_  
_bāhujaññamhi sāsane._  

</Paragraph>

</Division>
<Division number="2" book="ap1" bookNo="20" page="60">

<Paragraph number="697" type="verses" book="ap1">
_Pasannacitto sumano,_  
_sīhāsanamakāsahaṃ;_  
_Sīhāsanaṃ karitvāna,_  
_pādapīṭhamakāsahaṃ._  

</Paragraph>

</Division>
<Division number="3" book="ap1" bookNo="20" page="60">

<Paragraph number="698" type="verses" book="ap1">
_Sīhāsane ca vassante,_  
_gharaṃ tattha akāsahaṃ;_  
_Tena cittappasādena,_  
_tusitaṃ upapajjahaṃ._  

</Paragraph>

</Division>
<Division number="4" book="ap1" bookNo="20" page="60">

<Paragraph number="699" type="verses" book="ap1">
_Āyāmena catubbīsa-_  
_yojanaṃ āsi tāvade;_  
_Vimānaṃ sukataṃ mayhaṃ,_  
_vitthārena catuddasa._  

</Paragraph>

</Division>
<Division number="5" book="ap1" bookNo="20" page="60">

<Paragraph number="700" type="verses" book="ap1">
_Sataṃ kaññāsahassāni,_  
_parivārenti maṃ sadā;_  
_Soṇṇamayañca pallaṅkaṃ,_  
_byamhe āsi sunimmitaṃ._  

</Paragraph>

</Division>
<Division number="6" book="ap1" bookNo="20" page="60">

<Paragraph number="701" type="verses" book="ap1">
_Hatthiyānaṃ assayānaṃ,_  
_dibbayānaṃ upaṭṭhitaṃ;_  
_Pāsādā sivikā ceva,_  
_nibbattanti yadicchakaṃ._  

</Paragraph>

</Division>
<Division number="7" book="ap1" bookNo="20" page="60">

<Paragraph number="702" type="verses" book="ap1">
_Maṇimayā ca pallaṅkā,_  
_aññe sāramayā bahū;_  
_Nibbattanti mamaṃ sabbe,_  
_sīhāsanassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="8" book="ap1" bookNo="20" page="60">

<Paragraph number="703" type="verses" book="ap1">
_Soṇṇamayā rūpimayā,_  
_phalikāveḷuriyāmayā;_  
_Pādukā abhirūhāmi,_  
_pādapīṭhassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="9" book="ap1" bookNo="20" page="60">

<Paragraph number="704" type="verses" book="ap1">
_Catunnavutito kappe,_  
_yaṃ kammamakariṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_puññakammassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="10" book="ap1" bookNo="20" page="60">

<Paragraph number="705" type="verses" book="ap1">
_Tesattatimhito kappe,_  
_indanāmā tayo janā;_  
_Dvesattatimhito kappe,_  
_tayo sumananāmakā._  

</Paragraph>

</Division>
<Division number="11" book="ap1" bookNo="20" page="60">

<Paragraph number="706" type="verses" book="ap1">
_Samasattatito kappe,_  
_tayo varuṇanāmakā;_  
_Sattaratanasampannā,_  
_catudīpamhi issarā._  

</Paragraph>

</Division>
<Division number="12" book="ap1" bookNo="20" page="61">

<Paragraph number="707" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_vimokkhāpi ca aṭṭhime;_  
_Chaḷabhiññā sacchikatā,_  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="708" book="ap1">
Itthaṃ sudaṃ āyasmā sīhāsanadāyako thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="709" type="center" book="ap1">
Sīhāsanadāyakattherassāpadānaṃ paṭhamaṃ.
</Paragraph>
