<Paragraph number="2688" type="verses" book="ap1">
_“Tivarāyaṃ pure ramme,_  
_rājuyyānaṃ ahu tadā;_  
_Uyyānapālo tatthāsiṃ,_  
_rañño baddhacaro ahaṃ._  

</Paragraph>

</Division>
<Division number="18" book="ap1">

<Paragraph number="2689" type="verses" book="ap1">
_Padumo nāma nāmena,_  
_sayambhū sappabho ahu;_  
_Nisinnaṃ puṇḍarīkamhi,_  
_chāyā na jahi taṃ muniṃ._  

</Paragraph>

</Division>
<Division number="19" book="ap1">

<Paragraph number="2690" type="verses" book="ap1">
_Asokaṃ pupphitaṃ disvā,_  
_piṇḍibhāraṃ sudassanaṃ;_  
_Buddhassa abhiropesiṃ,_  
_jalajuttamanāmino._  

</Paragraph>

</Division>
<Division number="20" book="ap1">

<Paragraph number="2691" type="verses" book="ap1">
_Catunnavutito kappe,_  
_yaṃ pupphamabhiropayiṃ;_  
_Duggatiṃ nābhijānāmi,_  
_buddhapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="21" book="ap1">

<Paragraph number="2692" type="verses" book="ap1">
_Sattatiṃsamhito kappe,_  
_soḷasa araṇañjahā;_  
_Sattaratanasampannā,_  
_cakkavattī mahabbalā._  

</Paragraph>

</Division>
<Division number="22" book="ap1">

<Paragraph number="2693" type="verses" book="ap1">

_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="2694" book="ap1">

</Paragraph>
</Division>
