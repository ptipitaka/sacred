<Paragraph number="2681" type="verses" book="ap1">
_“Phussassa kho bhagavato,_  
_thūpo āsi mahāvane;_  
_Kuñjarehi tadā bhinno,_  
_parūḷho pādapo tahiṃ._  

</Paragraph>

</Division>
<Division number="13" book="ap1">

<Paragraph number="2682" type="verses" book="ap1">
_Visamañca samaṃ katvā,_  
_sudhāpiṇḍaṃ adāsahaṃ;_  
_Tilokagaruno tassa,_  
_guṇehi paritosito._  

</Paragraph>

</Division>
<Division number="14" book="ap1">

<Paragraph number="2683" type="verses" book="ap1">
_Dvenavute ito kappe,_  
_yaṃ kammamakariṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_sudhāpiṇḍassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="15" book="ap1">

<Paragraph number="2684" type="verses" book="ap1">
_Sattasattatikappamhi,_  
_jitasenāsuṃ soḷasa;_  
_Sattaratanasampannā,_  
_cakkavattī mahabbalā._  

</Paragraph>

</Division>
<Division number="16" book="ap1">

<Paragraph number="2685" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="2686" book="ap1">
Itthaṃ sudaṃ āyasmā khaṇḍaphulliyo thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>
