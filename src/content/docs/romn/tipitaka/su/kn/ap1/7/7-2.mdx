<Paragraph number="1499" type="verses" book="ap1">
_“Vihārā abhinikkhamma,_  
_abbhuṭṭhāsi ca caṅkame;_  
_Catusaccaṃ pakāsanto,_  
_desento amataṃ padaṃ._  

</Paragraph>

</Division>
<Division number="8" book="ap1">

<Paragraph number="1500" type="verses" book="ap1">
_Sikhissa giramaññāya,_  
_buddhaseṭṭhassa tādino;_  
_Nānāpupphaṃ gahetvāna,_  
_ākāsamhi samokiriṃ._  

</Paragraph>

</Division>
<Division number="9" book="ap1">

<Paragraph number="1501" type="verses" book="ap1">
_Tena kammena dvipadinda,_  
_lokajeṭṭha narāsabha;_  
_Pattomhi acalaṃ ṭhānaṃ,_  
_hitvā jayaparājayaṃ._  

</Paragraph>

</Division>
<Division number="10" book="ap1">

<Paragraph number="1502" type="verses" book="ap1">
_Ekattiṃse ito kappe,_  
_yaṃ pupphamabhiropayiṃ;_  
_Duggatiṃ nābhijānāmi,_  
_pupphapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="11" book="ap1">

<Paragraph number="1503" type="verses" book="ap1">
_Ito vīsatikappamhi,_  
_sumedho nāma khattiyo;_  
_Sattaratanasampanno,_  
_cakkavattī mahabbalo._  

</Paragraph>

</Division>
<Division number="12" book="ap1">

<Paragraph number="1504" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="1505" book="ap1">
Itthaṃ sudaṃ āyasmā avopupphiyo thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="1506" type="center" book="ap1">
Avopupphiyattherassāpadānaṃ dutiyaṃ.
</Paragraph>
