<Division number="69" book="ap1">

<Paragraph number="6646" type="verses" book="ap1">

_adāsiṃ satthuno ahaṃ;_  
_Dakkhiṇeyyassa vīrassa,_  
_pasanno sehi pāṇibhi._  

</Paragraph>

</Division>

_Duggatiṃ nābhijānāmi,_  
_phaladānassidaṃ phalaṃ._  

</Paragraph>

</Division>
