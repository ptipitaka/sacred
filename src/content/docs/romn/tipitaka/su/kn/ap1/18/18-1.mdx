<Paragraph number="2510" type="verses" book="ap1">
_“Pabbate himavantamhi,_  
_mahājātassaro ahu;_  
_Tatthajo rakkhaso āsiṃ,_  
_ghorarūpo mahabbalo._  

</Paragraph>

</Division>
<Division number="2" book="ap1" bookNo="20" page="207">

<Paragraph number="2511" type="verses" book="ap1">
_Kumudaṃ pupphate tattha,_  
_cakkamattāni jāyare;_  
_Ocināmi ca taṃ pupphaṃ,_  
_balino samitiṃ tadā._  

</Paragraph>

</Division>
<Division number="3" book="ap1" bookNo="20" page="207">

<Paragraph number="2512" type="verses" book="ap1">
_Atthadassī tu bhagavā,_  
_dvipadindo narāsabho;_  
_Pupphasaṅkocitaṃ disvā,_  
_āgacchi mama santikaṃ._  

</Paragraph>

</Division>
<Division number="4" book="ap1" bookNo="20" page="207">

<Paragraph number="2513" type="verses" book="ap1">
_Upāgatañca sambuddhaṃ,_  
_devadevaṃ narāsabhaṃ;_  
_Sabbañca pupphaṃ paggayha,_  
_buddhassa abhiropayiṃ._  

</Paragraph>

</Division>
<Division number="5" book="ap1" bookNo="20" page="207">

<Paragraph number="2514" type="verses" book="ap1">
_Yāvatā himavantantā,_  
_parisā sā tadā ahu;_  
_Tāvacchadanasampanno,_  
_agamāsi tathāgato._  

</Paragraph>

</Division>
<Division number="6" book="ap1" bookNo="20" page="207">

<Paragraph number="2515" type="verses" book="ap1">
_Aṭṭhārase kappasate,_  
_yaṃ pupphamabhiropayiṃ;_  
_Duggatiṃ nābhijānāmi,_  
_buddhapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="7" book="ap1" bookNo="20" page="207">

<Paragraph number="2516" type="verses" book="ap1">
_Ito pannarase kappe,_  
_sattāhesuṃ janādhipā;_  
_Sahassarathanāmā te,_  
_cakkavattī mahabbalā._  

</Paragraph>

</Division>
<Division number="8" book="ap1" bookNo="20" page="207">

<Paragraph number="2517" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_vimokkhāpi ca aṭṭhime;_  
_Chaḷabhiññā sacchikatā,_  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="2518" book="ap1">
Itthaṃ sudaṃ āyasmā kumudamāliyo thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="2519" type="center" book="ap1">
Kumudamāliyattherassāpadānaṃ paṭhamaṃ.
</Paragraph>
