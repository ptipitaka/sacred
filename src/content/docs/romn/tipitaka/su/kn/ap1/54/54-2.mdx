<Division number="28" book="ap1">

<Paragraph number="6977" type="verses" book="ap1">
_“Ito satasahassamhi,_  
_kappe uppajji nāyako;_  
_Anomanāmo amito,_  
_nāmena padumuttaro._  

</Paragraph>

</Division>
<Division number="29" book="ap1">

<Paragraph number="6978" type="verses" book="ap1">
_Padumākāravadano,_  
_padumāmalasucchavī;_  
_Lokenānupalittova,_  
_toyena padumaṃ yathā._  

</Paragraph>

</Division>
<Division number="30" book="ap1">

<Paragraph number="6979" type="verses" book="ap1">
_Vīro padumapattakkho,_  
_kanto ca padumaṃ yathā;_  
_Padumuttaragandhova,_  
_tasmā so padumuttaro._  

</Paragraph>

</Division>
<Division number="31" book="ap1">

<Paragraph number="6980" type="verses" book="ap1">
_Lokajeṭṭho ca nimmāno,_  
_andhānaṃ nayanūpamo;_  
_Santaveso guṇanidhi,_  
_karuṇāmatisāgaro._  

</Paragraph>

</Division>
<Division number="32" book="ap1">

<Paragraph number="6981" type="verses" book="ap1">
_Sa kadāci mahāvīro,_  
_brahmāsurasuraccito;_  
_Sadevamanujākiṇṇe,_  
_janamajjhe jinuttamo._  

</Paragraph>

</Division>
<Division number="33" book="ap1">

<Paragraph number="6982" type="verses" book="ap1">
_Vadanena sugandhena,_  
_madhurena rutena ca;_  
_Rañjayaṃ parisaṃ sabbaṃ,_  
_santhavī sāvakaṃ sakaṃ._  

</Paragraph>

</Division>
<Division number="34" book="ap1">

<Paragraph number="6983" type="verses" book="ap1">
_Saddhādhimutto sumati,_  
_mama dassanalālaso;_  
_Natthi etādiso añño,_  
_yathāyaṃ bhikkhu vakkali._  

</Paragraph>

</Division>
<Division number="35" book="ap1">

<Paragraph number="6984" type="verses" book="ap1">
_Tadāhaṃ haṃsavatiyaṃ,_  
_nagare brāhmaṇatrajo;_  
_Hutvā sutvā ca taṃ vākyaṃ,_  
_taṃ ṭhānamabhirocayiṃ._  

</Paragraph>

</Division>
<Division number="36" book="ap1">

<Paragraph number="6985" type="verses" book="ap1">
_Sasāvakaṃ taṃ vimalaṃ,_  
_nimantetvā tathāgataṃ;_  
_Sattāhaṃ bhojayitvāna,_  
_dussehacchādayiṃ tadā._  

</Paragraph>

</Division>
<Division number="37" book="ap1">

<Paragraph number="6986" type="verses" book="ap1">
_Nipacca sirasā tassa,_  
_anantaguṇasāgare;_  
_Nimuggo pītisampuṇṇo,_  
_idaṃ vacanamabraviṃ._  

</Paragraph>

</Division>
<Division number="38" book="ap1">

<Paragraph number="6987" type="verses" book="ap1">
_‘Yo so tayā santhavito,_  
_ito sattamake muni;_  
_Bhikkhu saddhāvataṃ aggo,_  
_tādiso homahaṃ mune’._  

</Paragraph>

</Division>
<Division number="39" book="ap1">

<Paragraph number="6988" type="verses" book="ap1">
_Evaṃ vutte, mahāvīro,_  
_anāvaraṇadassano;_  
_Imaṃ vākyaṃ udīresi,_  
_parisāya mahāmuni._  

</Paragraph>

</Division>
<Division number="40" book="ap1">

<Paragraph number="6989" type="verses" book="ap1">
_‘Passathetaṃ māṇavakaṃ,_  
_pītamaṭṭhanivāsanaṃ;_  
_Hemayaññopacitaṅgaṃ,_  
_jananettamanoharaṃ._  

</Paragraph>

</Division>
<Division number="41" book="ap1">

<Paragraph number="6990" type="verses" book="ap1">
_Eso anāgataddhāne,_  
_gotamassa mahesino;_  
_Aggo saddhādhimuttānaṃ,_  
_sāvakoyaṃ bhavissati._  

</Paragraph>

</Division>
<Division number="42" book="ap1">

<Paragraph number="6991" type="verses" book="ap1">
_Devabhūto manusso vā,_  
_sabbasantāpavajjito;_  
_Sabbabhogaparibyūḷho,_  
_sukhito saṃsarissati._  

</Paragraph>

</Division>
<Division number="43" book="ap1">

<Paragraph number="6992" type="verses" book="ap1">
_Satasahassito kappe,_  
_okkākakulasambhavo;_  
_Gotamo nāma gottena,_  
_satthā loke bhavissati._  

</Paragraph>

</Division>
<Division number="44" book="ap1">

<Paragraph number="6993" type="verses" book="ap1">
_Tassa dhammesu dāyādo,_  
_oraso dhammanimmito;_  
_Vakkali nāma nāmena,_  
_hessati satthu sāvako’._  

</Paragraph>

</Division>
<Division number="45" book="ap1">

<Paragraph number="6994" type="verses" book="ap1">
_Tena kammavisesena,_  
_cetanāpaṇidhīhi ca;_  
_Jahitvā mānusaṃ dehaṃ,_  
_tāvatiṃsamagacchahaṃ._  

</Paragraph>

</Division>
<Division number="46" book="ap1">

<Paragraph number="6995" type="verses" book="ap1">
_Sabbattha sukhito hutvā,_  
_saṃsaranto bhavābhave;_  
_Sāvatthiyaṃ pure jāto,_  
_kule aññatare ahaṃ._  

</Paragraph>

</Division>
<Division number="47" book="ap1">

<Paragraph number="6996" type="verses" book="ap1">
_Nonītasukhumālaṃ maṃ,_  
_jātapallavakomalaṃ;_  
_Mandaṃ uttānasayanaṃ,_  
_pisācībhayatajjitā._  

</Paragraph>

</Division>
<Division number="48" book="ap1">

<Paragraph number="6997" type="verses" book="ap1">
_Pādamūle mahesissa,_  
_sāyesuṃ dīnamānasā;_  
_‘Imaṃ dadāma te nātha,_  
_saraṇaṃ hohi nāyaka’._  

</Paragraph>

</Division>
<Division number="49" book="ap1">

<Paragraph number="6998" type="verses" book="ap1">
_Tadā paṭiggahi so maṃ,_  
_bhītānaṃ saraṇo muni;_  
_Jālinā cakkaṅkitena,_  
_mudukomalapāṇinā._  

</Paragraph>

</Division>
<Division number="50" book="ap1">

<Paragraph number="6999" type="verses" book="ap1">
_Tadā pabhuti tenāhaṃ,_  
_arakkheyyena rakkhito;_  
_Sabbaveravinimutto,_  
_sukhena parivuddhito._  

</Paragraph>

</Division>
<Division number="51" book="ap1">

<Paragraph number="7000" type="verses" book="ap1">
_Sugatena vinā bhūto,_  
_ukkaṇṭhāmi muhuttakaṃ;_  
_Jātiyā sattavassohaṃ,_  
_pabbajiṃ anagāriyaṃ._  

</Paragraph>

</Division>
<Division number="52" book="ap1">

<Paragraph number="7001" type="verses" book="ap1">
_Sabbapāramisambhūtaṃ,_  
_nīlakkhinayanaṃ varaṃ;_  
_Rūpaṃ sabbasubhākiṇṇaṃ,_  
_atitto viharāmahaṃ._  

</Paragraph>

</Division>
<Division number="53" book="ap1">

<Paragraph number="7002" type="verses" book="ap1">
_Buddharūparatiṃ ñatvā,_  
_tadā ovadi maṃ jino;_  
_‘Alaṃ vakkali kiṃ rūpe,_  
_ramase bālanandite._  

</Paragraph>

</Division>
<Division number="54" book="ap1">

<Paragraph number="7003" type="verses" book="ap1">
_Yo hi passati saddhammaṃ,_  
_so maṃ passati paṇḍito;_  
_Apassamāno saddhammaṃ,_  
_maṃ passampi na passati._  

</Paragraph>

</Division>
<Division number="55" book="ap1">

<Paragraph number="7004" type="verses" book="ap1">
_Anantādīnavo kāyo,_  
_visarukkhasamūpamo;_  
_Āvāso sabbarogānaṃ,_  
_puñjo dukkhassa kevalo._  

</Paragraph>

</Division>
<Division number="56" book="ap1">

<Paragraph number="7005" type="verses" book="ap1">
_Nibbindiya tato rūpe,_  
_khandhānaṃ udayabbayaṃ;_  
_Passa upakkilesānaṃ,_  
_sukhenantaṃ gamissasi’._  

</Paragraph>

</Division>
<Division number="57" book="ap1">

<Paragraph number="7006" type="verses" book="ap1">
_Evaṃ tenānusiṭṭhohaṃ,_  
_nāyakena hitesinā;_  
_Gijjhakūṭaṃ samāruyha,_  
_jhāyāmi girikandare._  

</Paragraph>

</Division>
<Division number="58" book="ap1">

<Paragraph number="7007" type="verses" book="ap1">
_Ṭhito pabbatapādamhi,_  
_assāsayi mahāmuni;_  
_Vakkalīti jino vācaṃ,_  
_taṃ sutvā mudito ahaṃ._  

</Paragraph>

</Division>
<Division number="59" book="ap1">

<Paragraph number="7008" type="verses" book="ap1">
_Pakkhandiṃ selapabbhāre,_  
_anekasataporise;_  
_Tadā buddhānubhāvena,_  
_sukheneva mahiṃ gato._  

</Paragraph>

</Division>
<Division number="60" book="ap1">

<Paragraph number="7009" type="verses" book="ap1">
_Punopi dhammaṃ deseti,_  
_khandhānaṃ udayabbayaṃ;_  
_Tamahaṃ dhammamaññāya,_  
_arahattamapāpuṇiṃ._  

</Paragraph>

</Division>
<Division number="61" book="ap1">

<Paragraph number="7010" type="verses" book="ap1">
_Sumahāparisamajjhe,_  
_tadā maṃ caraṇantago;_  
_Aggaṃ saddhādhimuttānaṃ,_  
_paññapesi mahāmati._  

</Paragraph>

</Division>
<Division number="62" book="ap1">

<Paragraph number="7011" type="verses" book="ap1">
_Satasahassito kappe,_  
_yaṃ kammamakariṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_buddhapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="63" book="ap1">

<Paragraph number="7012" type="verses" book="ap1">
_Kilesā jhāpitā mayhaṃ,_  
_ …pe… _  
_viharāmi anāsavo._  

</Paragraph>

</Division>
<Division number="64" book="ap1">

<Paragraph number="7013" type="verses" book="ap1">
_Svāgataṃ vata me āsi,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ._  

</Paragraph>

</Division>
<Division number="65" book="ap1">

<Paragraph number="7014" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="7015" book="ap1">
Itthaṃ sudaṃ āyasmā vakkalitthero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="7016" type="center" book="ap1">
