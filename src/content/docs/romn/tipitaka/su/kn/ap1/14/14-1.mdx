<Division number="1" book="ap1" bookNo="20" page="180">

<Paragraph number="2182" type="verses" book="ap1">
_“Padumuttaro nāma jino,_  
_lokajeṭṭho narāsabho;_  
_Mahato janakāyassa,_  
_deseti amataṃ padaṃ._  

</Paragraph>

</Division>
<Division number="2" book="ap1" bookNo="20" page="180">

<Paragraph number="2183" type="verses" book="ap1">
_Tassāhaṃ vacanaṃ sutvā,_  
_vācāsabhimudīritaṃ;_  
_Añjaliṃ paggahetvāna,_  
_ekaggo āsahaṃ tadā._  

</Paragraph>

</Division>
<Division number="3" book="ap1" bookNo="20" page="180">

<Paragraph number="2184" type="verses" book="ap1">
_‘Yathā samuddo udadhīnamaggo,_  
_Nerū nagānaṃ pavaro siluccayo;_  
_Tatheva ye cittavasena vattare’,_  
_Na buddhañāṇassa kalaṃ upenti te._  

</Paragraph>

</Division>
<Division number="4" book="ap1" bookNo="20" page="180">

<Paragraph number="2185" type="verses" book="ap1">
_Dhammavidhiṃ ṭhapetvāna,_  
_buddho kāruṇiko isi;_  
_Bhikkhusaṃghe nisīditvā,_  
_imā gāthā abhāsatha._  

</Paragraph>

</Division>
<Division number="5" book="ap1" bookNo="20" page="180">

<Paragraph number="2186" type="verses" book="ap1">
_‘Yo so ñāṇaṃ pakittesi,_  
_buddhamhi lokanāyake;_  
_Kappānaṃ satasahassaṃ,_  
_duggatiṃ na gamissati._  

</Paragraph>

</Division>
<Division number="6" book="ap1" bookNo="20" page="180">

<Paragraph number="2187" type="verses" book="ap1">
_Kilese jhāpayitvāna,_  
_ekaggo susamāhito;_  
_Sobhito nāma nāmena,_  
_hessati satthu sāvako’._  

</Paragraph>

</Division>
<Division number="7" book="ap1" bookNo="20" page="180">

<Paragraph number="2188" type="verses" book="ap1">
_Paññāse kappasahasse,_  
_sattevāsuṃ yasuggatā;_  
_Sattaratanasampannā,_  
_cakkavattī mahabbalā._  

</Paragraph>

</Division>
<Division number="8" book="ap1" bookNo="20" page="180">

<Paragraph number="2189" type="verses" book="ap1">
_Kilesā jhāpitā mayhaṃ,_  
_bhavā sabbe samūhatā;_  
_Tisso vijjā anuppattā,_  
_kataṃ buddhassa sāsanaṃ._  

</Paragraph>

</Division>
<Division number="9" book="ap1" bookNo="20" page="180">

<Paragraph number="2190" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_vimokkhāpi ca aṭṭhime;_  
_Chaḷabhiññā sacchikatā,_  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="2191" book="ap1">
Itthaṃ sudaṃ āyasmā sobhito thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="2192" type="center" book="ap1">
Sobhitattherassāpadānaṃ paṭhamaṃ.
</Paragraph>
