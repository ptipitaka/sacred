<Division number="53" book="ap1">

<Paragraph number="1557" type="verses" book="ap1">
_“Anomadassī bhagavā,_  
_nisinno pabbatantare;_  
_Mettāya aphari loke,_  
_appamāṇe nirūpadhi._  

</Paragraph>

</Division>
<Division number="54" book="ap1">

<Paragraph number="1558" type="verses" book="ap1">
_Kapi ahaṃ tadā āsiṃ,_  
_himavante naguttame;_  
_Disvā anomadassiṃ taṃ,_  
_buddhe cittaṃ pasādayiṃ._  

</Paragraph>

</Division>
<Division number="55" book="ap1">

<Paragraph number="1559" type="verses" book="ap1">
_Avidūre himavantassa,_  
_ambāsuṃ phalino tadā;_  
_Tato pakkaṃ gahetvāna,_  
_ambaṃ samadhukaṃ adaṃ._  

</Paragraph>

</Division>
<Division number="56" book="ap1">

<Paragraph number="1560" type="verses" book="ap1">
_Taṃ me buddho viyākāsi,_  
_anomadassī mahāmuni;_  
_Iminā madhudānena,_  
_ambadānena cūbhayaṃ._  

</Paragraph>

</Division>
<Division number="57" book="ap1">

<Paragraph number="1561" type="verses" book="ap1">
_Sattapaññāsakappamhi,_  
_devaloke ramissati;_  
_Avasesesu kappesu,_  
_vokiṇṇaṃ saṃsarissati._  

</Paragraph>

</Division>
<Division number="58" book="ap1">

<Paragraph number="1562" type="verses" book="ap1">
_Khepetvā pāpakaṃ kammaṃ,_  
_paripakkāya buddhiyā;_  
_Vinipātamagantvāna,_  
_kilese jhāpayissati._  

</Paragraph>

</Division>
<Division number="59" book="ap1">

<Paragraph number="1563" type="verses" book="ap1">
_Damena uttamenāhaṃ,_  
_damitomhi mahesinā;_  
_Pattomhi acalaṃ ṭhānaṃ,_  
_hitvā jayaparājayaṃ._  

</Paragraph>

</Division>
<Division number="60" book="ap1">

<Paragraph number="1564" type="verses" book="ap1">
_Sattasattatikappasate,_  
_ambaṭṭhajasanāmakā;_  
_Catuddasa te rājāno,_  
_cakkavattī mahabbalā._  

</Paragraph>

</Division>
<Division number="61" book="ap1">

<Paragraph number="1565" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="1566" book="ap1">
Itthaṃ sudaṃ āyasmā ambadāyako thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="1567" type="center" book="ap1">
Ambadāyakattherassāpadānaṃ aṭṭhamaṃ.
</Paragraph>
