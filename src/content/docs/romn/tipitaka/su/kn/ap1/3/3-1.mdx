<Division number="1" book="ap1">

<Paragraph number="860" type="verses" book="ap1">
_“Himavantassāvidūre,_  
_nisabho nāma pabbato;_  
_Assamo sukato mayhaṃ,_  
_paṇṇasālā sumāpitā._  

</Paragraph>

</Division>
<Division number="2" book="ap1">

<Paragraph number="861" type="verses" book="ap1">
_Kosiyo nāma nāmena,_  
_jaṭilo uggatāpano;_  
_Ekākiyo adutiyo,_  
_vasāmi nisabhe tadā._  

</Paragraph>

</Division>
<Division number="3" book="ap1">

<Paragraph number="862" type="verses" book="ap1">
_Phalaṃ mūlañca paṇṇañca,_  
_na bhuñjāmi ahaṃ tadā;_  
_Pavattaṃva supātāhaṃ,_  
_upajīvāmi tāvade._  

</Paragraph>

</Division>
<Division number="4" book="ap1">

<Paragraph number="863" type="verses" book="ap1">
_Nāhaṃ kopemi ājīvaṃ,_  
_cajamānopi jīvitaṃ;_  
_Ārādhemi sakaṃ cittaṃ,_  
_vivajjemi anesanaṃ._  

</Paragraph>

</Division>
<Division number="5" book="ap1">

<Paragraph number="864" type="verses" book="ap1">
_Rāgūpasaṃhitaṃ cittaṃ,_  
_yadā uppajjate mama;_  
_Sayaṃva paccavekkhāmi,_  
_ekaggo taṃ damemahaṃ._  

</Paragraph>

</Division>
<Division number="6" book="ap1">

<Paragraph number="865" type="verses" book="ap1">
_‘Rajjase rajjanīye ca,_  
_dussanīye ca dussase;_  
_Muyhase mohanīye ca,_  
_nikkhamassu vanā tuvaṃ._  

</Paragraph>

</Division>
<Division number="7" book="ap1">

<Paragraph number="866" type="verses" book="ap1">
_Visuddhānaṃ ayaṃ vāso,_  
_nimmalānaṃ tapassinaṃ;_  
_Mā kho visuddhaṃ dūsesi,_  
_nikkhamassu vanā tuvaṃ._  

</Paragraph>

</Division>
<Division number="8" book="ap1">

<Paragraph number="867" type="verses" book="ap1">
_Agāriko bhavitvāna,_  
_yadā puttaṃ labhissasi;_  
_Ubhopi mā virādhesi,_  
_nikkhamassu vanā tuvaṃ._  

</Paragraph>

</Division>
<Division number="9" book="ap1">

<Paragraph number="868" type="verses" book="ap1">
_Chavālātaṃ yathā kaṭṭhaṃ,_  
_na kvaci kiccakārakaṃ;_  
_Neva gāme araññe vā,_  
_na hi taṃ kaṭṭhasammataṃ._  

</Paragraph>

</Division>
<Division number="10" book="ap1">

<Paragraph number="869" type="verses" book="ap1">
_Chavālātūpamo tvaṃsi,_  
_na gihī nāpi saññato;_  
_Ubhato muttako ajja,_  
_nikkhamassu vanā tuvaṃ._  

</Paragraph>

</Division>
<Division number="11" book="ap1">

<Paragraph number="870" type="verses" book="ap1">
_Siyā nu kho tava etaṃ,_  
_ko pajānāti te idaṃ;_  
_Saddhādhuraṃ vahisi me,_  
_kosajjabahulāya ca._  

</Paragraph>

</Division>
<Division number="12" book="ap1">

<Paragraph number="871" type="verses" book="ap1">
_Jigucchissanti taṃ viññū,_  
_asuciṃ nāgariko yathā;_  
_Ākaḍḍhitvāna isayo,_  
_codayissanti taṃ sadā._  

</Paragraph>

</Division>
<Division number="13" book="ap1">

<Paragraph number="872" type="verses" book="ap1">
_Taṃ viññū pavadissanti,_  
_samatikkantasāsanaṃ;_  
_Saṃvāsaṃ alabhanto hi,_  
_kathaṃ jīvihisi tuvaṃ._  

</Paragraph>

</Division>
<Division number="14" book="ap1">

<Paragraph number="873" type="verses" book="ap1">
_Tidhā pabhinnaṃ mātaṅgaṃ,_  
_kuñjaraṃ saṭṭhihāyanaṃ;_  
_Balī nāgo upagantvā,_  
_yūthā nīharate gajaṃ._  

</Paragraph>

</Division>
<Division number="15" book="ap1">

<Paragraph number="874" type="verses" book="ap1">
_Yūthā vinissaṭo santo,_  
_sukhaṃ sātaṃ na vindati;_  
_Dukkhito vimano hoti,_  
_pajjhāyanto pavedhati._  

</Paragraph>

</Division>
<Division number="16" book="ap1">

<Paragraph number="875" type="verses" book="ap1">
_Tatheva jaṭilā tampi,_  
_nīharissanti dummatiṃ;_  
_Tehi tvaṃ nissaṭo santo,_  
_sukhaṃ sātaṃ na lacchasi._  

</Paragraph>

</Division>
<Division number="17" book="ap1">

<Paragraph number="876" type="verses" book="ap1">
_Divā vā yadi vā rattiṃ,_  
_sokasallasamappito;_  
_Ḍayhasi pariḷāhena,_  
_gajo yūthāva nissaṭo._  

</Paragraph>

</Division>
<Division number="18" book="ap1">

<Paragraph number="877" type="verses" book="ap1">
_Jātarūpaṃ yathā kūṭaṃ,_  
_neva jhāyati katthaci;_  
_Tathā sīlavihīno tvaṃ,_  
_na jhāyissasi katthaci._  

</Paragraph>

</Division>
<Division number="19" book="ap1">

<Paragraph number="878" type="verses" book="ap1">
_Agāraṃ vasamānopi,_  
_kathaṃ jīvihisi tuvaṃ;_  
_Mattikaṃ pettikañcāpi,_  
_natthi te nihitaṃ dhanaṃ._  

</Paragraph>

</Division>
<Division number="20" book="ap1">

<Paragraph number="879" type="verses" book="ap1">
_Sayaṃ kammaṃ karitvāna,_  
_gatte sedaṃ pamocayaṃ;_  
_Evaṃ jīvihisi gehe,_  
_sādhu te taṃ na ruccati._  

</Paragraph>

</Division>
<Division number="21" book="ap1">

<Paragraph number="880" type="verses" book="ap1">
_Evāhaṃ tattha vāremi,_  
_saṃkilesagataṃ manaṃ;_  
_Nānādhammakathaṃ katvā,_  
_pāpā cittaṃ nivārayiṃ’._  

</Paragraph>

</Division>
<Division number="22" book="ap1">

<Paragraph number="881" type="verses" book="ap1">
_Evaṃ me viharantassa,_  
_appamādavihārino;_  
_Tiṃsavassasahassāni,_  
_vipine me atikkamuṃ._  

</Paragraph>

</Division>
<Division number="23" book="ap1">

<Paragraph number="882" type="verses" book="ap1">
_Appamādarataṃ disvā,_  
_uttamatthaṃ gavesakaṃ;_  
_Padumuttarasambuddho,_  
_āgacchi mama santikaṃ._  

</Paragraph>

</Division>
<Division number="24" book="ap1">

<Paragraph number="883" type="verses" book="ap1">
_Timbarūsakavaṇṇābho,_  
_appameyyo anūpamo;_  
_Rūpenāsadiso buddho,_  
_ākāse caṅkamī tadā._  

</Paragraph>

</Division>
<Division number="25" book="ap1">

<Paragraph number="884" type="verses" book="ap1">
_Suphullo sālarājāva,_  
_vijjūvabbhaghanantare;_  
_Ñāṇenāsadiso buddho,_  
_ākāse caṅkamī tadā._  

</Paragraph>

</Division>
<Division number="26" book="ap1">

<Paragraph number="885" type="verses" book="ap1">
_Sīharājāvasambhīto,_  
_gajarājāva dappito;_  
_Lāsito byaggharājāva,_  
_ākāse caṅkamī tadā._  

</Paragraph>

</Division>
<Division number="27" book="ap1">

<Paragraph number="886" type="verses" book="ap1">
_Siṅgīnikkhasavaṇṇābho,_  
_khadiraṅgārasannibho;_  
_Maṇi yathā jotiraso,_  
_ākāse caṅkamī tadā._  

</Paragraph>

</Division>
<Division number="28" book="ap1">

<Paragraph number="887" type="verses" book="ap1">
_Visuddhakelāsanibho,_  
_puṇṇamāyeva candimā;_  
_Majjhanhikeva sūriyo,_  
_ākāse caṅkamī tadā._  

</Paragraph>

</Division>
<Division number="29" book="ap1">

<Paragraph number="888" type="verses" book="ap1">
_Disvā nabhe caṅkamantaṃ,_  
_evaṃ cintesahaṃ tadā;_  
_‘Devo nu kho ayaṃ satto,_  
_udāhu manujo ayaṃ._  

</Paragraph>

</Division>
<Division number="30" book="ap1">

<Paragraph number="889" type="verses" book="ap1">
_Na me suto vā diṭṭho vā,_  
_mahiyā ediso naro;_  
_Api mantapadaṃ atthi,_  
_ayaṃ satthā bhavissati’._  

</Paragraph>

</Division>
<Division number="31" book="ap1">

<Paragraph number="890" type="verses" book="ap1">
_Evāhaṃ cintayitvāna,_  
_sakaṃ cittaṃ pasādayiṃ;_  
_Nānāpupphañca gandhañca,_  
_sannipātesahaṃ tadā._  

</Paragraph>

</Division>
<Division number="32" book="ap1">

<Paragraph number="891" type="verses" book="ap1">
_Pupphāsanaṃ paññapetvā,_  
_sādhucittaṃ manoramaṃ;_  
_Narasārathinaṃ aggaṃ,_  
_idaṃ vacanamabraviṃ._  

</Paragraph>

</Division>
<Division number="33" book="ap1">

<Paragraph number="892" type="verses" book="ap1">
_‘Idaṃ me āsanaṃ vīra,_  
_paññattaṃ tavanucchavaṃ;_  
_Hāsayanto mamaṃ cittaṃ,_  
_nisīda kusumāsane’._  

</Paragraph>

</Division>
<Division number="34" book="ap1">

<Paragraph number="893" type="verses" book="ap1">
_Nisīdi tattha bhagavā,_  
_asambhītova kesarī;_  
_Sattarattindivaṃ buddho,_  
_pavare kusumāsane._  

</Paragraph>

</Division>
<Division number="35" book="ap1">

<Paragraph number="894" type="verses" book="ap1">
_Namassamāno aṭṭhāsiṃ,_  
_sattarattindivaṃ ahaṃ;_  
_Vuṭṭhahitvā samādhimhā,_  
_satthā loke anuttaro;_  
_Mama kammaṃ pakittento,_  
_idaṃ vacanamabravi._  

</Paragraph>

</Division>
<Division number="36" book="ap1">

<Paragraph number="895" type="verses" book="ap1">
_‘Bhāvehi buddhānussatiṃ,_  
_bhāvanānamanuttaraṃ;_  
_Imaṃ satiṃ bhāvayitvā,_  
_pūrayissasi mānasaṃ._  

</Paragraph>

</Division>
<Division number="37" book="ap1">

<Paragraph number="896" type="verses" book="ap1">
_Tiṃsakappasahassāni,_  
_devaloke ramissasi;_  
_Asītikkhattuṃ devindo,_  
_devarajjaṃ karissasi;_  
_Sahassakkhattuṃ cakkavattī,_  
_rājā raṭṭhe bhavissasi._  

</Paragraph>

</Division>
<Division number="38" book="ap1">

<Paragraph number="897" type="verses" book="ap1">
_Padesarajjaṃ vipulaṃ,_  
_gaṇanāto asaṅkhiyaṃ;_  
_Anubhossasi taṃ sabbaṃ,_  
_buddhānussatiyā phalaṃ._  

</Paragraph>

</Division>
<Division number="39" book="ap1">

<Paragraph number="898" type="verses" book="ap1">
_Bhavābhave saṃsaranto,_  
_mahābhogaṃ labhissasi;_  
_Bhoge te ūnatā natthi,_  
_buddhānussatiyā phalaṃ._  

</Paragraph>

</Division>
<Division number="40" book="ap1">

<Paragraph number="899" type="verses" book="ap1">
_Kappasatasahassamhi,_  
_okkākakulasambhavo;_  
_Gotamo nāma gottena,_  
_satthā loke bhavissati._  

</Paragraph>

</Division>
<Division number="41" book="ap1">

<Paragraph number="900" type="verses" book="ap1">
_Asītikoṭiṃ chaḍḍetvā,_  
_dāse kammakare bahū;_  
_Gotamassa bhagavato,_  
_sāsane pabbajissasi._  

</Paragraph>

</Division>
<Division number="42" book="ap1">

<Paragraph number="901" type="verses" book="ap1">
_Ārādhayitvā sambuddhaṃ,_  
_gotamaṃ sakyapuṅgavaṃ;_  
_Subhūti nāma nāmena,_  
_hessati satthu sāvako._  

</Paragraph>

</Division>
<Division number="43" book="ap1">

<Paragraph number="902" type="verses" book="ap1">
_Bhikkhusaṃghe nisīditvā,_  
_dakkhiṇeyyaguṇamhi taṃ;_  
_Tathāraṇavihāre ca,_  
_dvīsu agge ṭhapessati’._  

</Paragraph>

</Division>
<Division number="44" book="ap1">

<Paragraph number="903" type="verses" book="ap1">
_Idaṃ vatvāna sambuddho,_  
_jalajuttamanāmako;_  
_Nabhaṃ abbhuggamī vīro,_  
_haṃsarājāva ambare._  

</Paragraph>

</Division>
<Division number="45" book="ap1">

<Paragraph number="904" type="verses" book="ap1">
_Sāsito lokanāthena,_  
_namassitvā tathāgataṃ;_  
_Sadā bhāvemi mudito,_  
_buddhānussatimuttamaṃ._  

</Paragraph>

</Division>
<Division number="46" book="ap1">

<Paragraph number="905" type="verses" book="ap1">
_Tena kammena sukatena,_  
_cetanāpaṇidhīhi ca;_  
_Jahitvā mānusaṃ dehaṃ,_  
_tāvatiṃsaṃ agacchahaṃ._  

</Paragraph>

</Division>
<Division number="47" book="ap1">

<Paragraph number="906" type="verses" book="ap1">
_Asītikkhattuṃ devindo,_  
_devarajjamakārayiṃ;_  
_Sahassakkhattuṃ rājā ca,_  
_cakkavattī ahosahaṃ._  

</Paragraph>

</Division>
<Division number="48" book="ap1">

<Paragraph number="907" type="verses" book="ap1">
_Padesarajjaṃ vipulaṃ,_  
_gaṇanāto asaṅkhiyaṃ;_  
_Anubhomi susampattiṃ,_  
_buddhānussatiyā phalaṃ._  

</Paragraph>

</Division>
<Division number="49" book="ap1">

<Paragraph number="908" type="verses" book="ap1">
_Bhavābhave saṃsaranto,_  
_mahābhogaṃ labhāmahaṃ;_  
_Bhoge me ūnatā natthi,_  
_buddhānussatiyā phalaṃ._  

</Paragraph>

</Division>
<Division number="50" book="ap1">

<Paragraph number="909" type="verses" book="ap1">
_Satasahassito kappe,_  
_yaṃ kammamakariṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_buddhānussatiyā phalaṃ._  

</Paragraph>

</Division>
<Division number="51" book="ap1">

<Paragraph number="910" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_vimokkhāpi ca aṭṭhime;_  
_Chaḷabhiññā sacchikatā,_  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="911" book="ap1">
Itthaṃ sudaṃ āyasmā subhūti thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="912" type="center" book="ap1">
Subhūtittherassāpadānaṃ paṭhamaṃ.
