<Division number="31" book="ap1">

<Paragraph number="1107" type="verses" book="ap1">
_“Padumuttaranāthassa,_  
_padumaṃ nāma cetiyaṃ;_  
_Silāsanaṃ kārayitvā,_  
_suvaṇṇenābhilepayiṃ._  

</Paragraph>

</Division>
<Division number="32" book="ap1">

<Paragraph number="1108" type="verses" book="ap1">
_Ratanāmayachattañca,_  
_paggayha vāḷabījaniṃ;_  
_Buddhassa abhiropesiṃ,_  
_lokabandhussa tādino._  

</Paragraph>

</Division>
<Division number="33" book="ap1">

<Paragraph number="1109" type="verses" book="ap1">
_Yāvatā devatā bhummā,_  
_sabbe sannipatuṃ tadā;_  
_Ratanāmayachattānaṃ,_  
_vipākaṃ kathayissati._  

</Paragraph>

</Division>
<Division number="34" book="ap1">

<Paragraph number="1110" type="verses" book="ap1">
_Tañca sabbaṃ suṇissāma,_  
_kathayantassa satthuno;_  
_Bhiyyo hāsaṃ janeyyāma,_  
_sammāsambuddhasāsane._  

</Paragraph>

</Division>
<Division number="35" book="ap1">

<Paragraph number="1111" type="verses" book="ap1">
_Hemāsane nisīditvā,_  
_sayambhū aggapuggalo;_  
_Bhikkhusaṃghaparibyūḷho,_  
_imā gāthā abhāsatha._  

</Paragraph>

</Division>
<Division number="36" book="ap1">

<Paragraph number="1112" type="verses" book="ap1">
_‘Yenidaṃ āsanaṃ dinnaṃ,_  
_sovaṇṇaṃ ratanāmayaṃ;_  
_Tamahaṃ kittayissāmi,_  
_suṇātha mama bhāsato._  

</Paragraph>

</Division>
<Division number="37" book="ap1">

<Paragraph number="1113" type="verses" book="ap1">
_Tiṃsa kappāni devindo,_  
_devarajjaṃ karissati;_  
_Samantā yojanasataṃ,_  
_ābhāyābhibhavissati._  

</Paragraph>

</Division>
<Division number="38" book="ap1">

<Paragraph number="1114" type="verses" book="ap1">
_Manussalokamāgantvā,_  
_cakkavattī bhavissati;_  
_Pabhassaroti nāmena,_  
_uggatejo bhavissati._  

</Paragraph>

</Division>
<Division number="39" book="ap1">

<Paragraph number="1115" type="verses" book="ap1">
_Divā vā yadi vā rattiṃ,_  
_sataraṃsīva uggato;_  
_Samantā aṭṭharatanaṃ,_  
_ujjotissati khattiyo._  

</Paragraph>

</Division>
<Division number="40" book="ap1">

<Paragraph number="1116" type="verses" book="ap1">
_Kappasatasahassamhi,_  
_okkākakulasambhavo;_  
_Gotamo nāma gottena,_  
_satthā loke bhavissati._  

</Paragraph>

</Division>
<Division number="41" book="ap1">

<Paragraph number="1117" type="verses" book="ap1">
_Tusitā hi cavitvāna,_  
_sukkamūlena codito;_  
_Kaccāno nāma nāmena,_  
_brahmabandhu bhavissati._  

</Paragraph>

</Division>
<Division number="42" book="ap1">

<Paragraph number="1118" type="verses" book="ap1">
_So pacchā pabbajitvāna,_  
_arahā hessatināsavo;_  
_Gotamo lokapajjoto,_  
_aggaṭṭhāne ṭhapessati._  

</Paragraph>

</Division>
<Division number="43" book="ap1">

<Paragraph number="1119" type="verses" book="ap1">
_Saṃkhittapucchitaṃ pañhaṃ,_  
_Vitthārena kathessati;_  
_Kathayanto ca taṃ pañhaṃ,_  
_Ajjhāsayaṃ pūrayissati’._  

</Paragraph>

</Division>
<Division number="44" book="ap1">

<Paragraph number="1120" type="verses" book="ap1">
_Aḍḍhe kule abhijāto,_  
_brāhmaṇo mantapāragū;_  
_Ohāya dhanadhaññāni,_  
_pabbajiṃ anagāriyaṃ._  

</Paragraph>

</Division>
<Division number="45" book="ap1">

<Paragraph number="1121" type="verses" book="ap1">
_Saṃkhittenapi pucchante,_  
_vitthārena kathemahaṃ;_  
_Ajjhāsayaṃ tesaṃ pūremi,_  
_tosemi dvipaduttamaṃ._  

</Paragraph>

</Division>
<Division number="46" book="ap1">

<Paragraph number="1122" type="verses" book="ap1">
_Tosito me mahāvīro,_  
_sayambhū aggapuggalo;_  
_Bhikkhusaṃghe nisīditvā,_  
_etadagge ṭhapesi maṃ._  

</Paragraph>

</Division>
<Division number="47" book="ap1">

<Paragraph number="1123" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="1124" book="ap1">
Itthaṃ sudaṃ āyasmā mahākaccāno thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="1125" type="center" book="ap1">
Mahākaccānattherassāpadānaṃ tatiyaṃ.
</Paragraph>
