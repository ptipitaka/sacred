<Division number="47" book="ap1">

<Paragraph number="3568" type="verses" book="ap1">
_“Sindhuyā nadiyā tīre,_  
_ahosiṃ kassako tadā;_  
_Parakammāyane yutto,_  
_parabhattaṃ apassito._  

</Paragraph>

</Division>
<Division number="48" book="ap1">

<Paragraph number="3569" type="verses" book="ap1">
_Sindhuṃ anucarantohaṃ,_  
_siddhatthaṃ jinamaddasaṃ;_  
_Samādhinā nisinnaṃva,_  
_satapattaṃva pupphitaṃ._  

</Paragraph>

</Division>
<Division number="49" book="ap1">

<Paragraph number="3570" type="verses" book="ap1">
_Satta kandalipupphāni,_  
_vaṇṭe chetvānahaṃ tadā;_  
_Matthake abhiropesiṃ,_  
_buddhassādiccabandhuno._  

</Paragraph>

</Division>
<Division number="50" book="ap1">

<Paragraph number="3571" type="verses" book="ap1">
_Suvaṇṇavaṇṇaṃ sambuddhaṃ,_  
_Anukūle samāhitaṃ;_  
_Tidhāpabhinnamātaṅgaṃ,_  
_Kuñjaraṃva durāsadaṃ._  

</Paragraph>

</Division>
<Division number="51" book="ap1">

<Paragraph number="3572" type="verses" book="ap1">
_Tamahaṃ upagantvāna,_  
_nipakaṃ bhāvitindriyaṃ;_  
_Añjaliṃ paggahetvāna,_  
_avandiṃ satthuno ahaṃ._  

</Paragraph>

</Division>
<Division number="52" book="ap1">

<Paragraph number="3573" type="verses" book="ap1">
_Catunnavutito kappe,_  
_yaṃ pupphamabhiropayiṃ;_  
_Duggatiṃ nābhijānāmi,_  
_buddhapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="53" book="ap1">

<Paragraph number="3574" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="3575" book="ap1">
Itthaṃ sudaṃ āyasmā kandalipupphiyo thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="3576" type="center" book="ap1">
Kandalipupphiyattherassāpadānaṃ navamaṃ.
</Paragraph>
