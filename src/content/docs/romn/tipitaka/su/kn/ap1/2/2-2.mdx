<Division number="13" book="ap1">

<Paragraph number="710" type="verses" book="ap1">
_“Siddhatthassa bhagavato,_  
_mahāpūgagaṇo ahu;_  
_Saraṇaṃ gatā ca te buddhaṃ,_  
_saddahanti tathāgataṃ._  

</Paragraph>

</Division>
<Division number="14" book="ap1">

<Paragraph number="711" type="verses" book="ap1">
_Sabbe saṅgamma mantetvā,_  
_māḷaṃ kubbanti satthuno;_  
_Ekatthambhaṃ alabhantā,_  
_vicinanti brahāvane._  

</Paragraph>

</Division>
<Division number="15" book="ap1">

<Paragraph number="712" type="verses" book="ap1">
_Tehaṃ araññe disvāna,_  
_upagamma gaṇaṃ tadā;_  
_Añjaliṃ paggahetvāna,_  
_paripucchiṃ gaṇaṃ ahaṃ._  

</Paragraph>

</Division>
<Division number="16" book="ap1">

<Paragraph number="713" type="verses" book="ap1">
_Te me puṭṭhā viyākaṃsu,_  
_sīlavanto upāsakā;_  
_Māḷaṃ mayaṃ kattukāmā,_  
_ekatthambho na labbhati._  

</Paragraph>

</Division>
<Division number="17" book="ap1">

<Paragraph number="714" type="verses" book="ap1">
_Ekatthambhaṃ mamaṃ detha,_  
_ahaṃ dassāmi satthuno;_  
_Āharissāmahaṃ thambhaṃ,_  
_appossukkā bhavantu te._  

</Paragraph>

</Division>
<Division number="18" book="ap1">

<Paragraph number="715" type="verses" book="ap1">
_Te me thambhaṃ pavecchiṃsu,_  
_pasannā tuṭṭhamānasā;_  
_Tato paṭinivattitvā,_  
_agamaṃsu sakaṃ gharaṃ._  

</Paragraph>

</Division>
<Division number="19" book="ap1">

<Paragraph number="716" type="verses" book="ap1">
_Aciraṃ gate pūgagaṇe,_  
_thambhaṃ ahāsahaṃ tadā;_  
_Haṭṭho haṭṭhena cittena,_  
_paṭhamaṃ ussapesahaṃ._  

</Paragraph>

</Division>
<Division number="20" book="ap1">

<Paragraph number="717" type="verses" book="ap1">
_Tena cittappasādena,_  
_Vimānaṃ upapajjahaṃ;_  
_Ubbiddhaṃ bhavanaṃ mayhaṃ,_  
_Sattabhūmaṃ samuggataṃ._  

</Paragraph>

</Division>
<Division number="21" book="ap1">

<Paragraph number="718" type="verses" book="ap1">
_Vajjamānāsu bherīsu,_  
_paricāremahaṃ sadā;_  
_Pañcapaññāsakappamhi,_  
_rājā āsiṃ yasodharo._  

</Paragraph>

</Division>
<Division number="22" book="ap1">

<Paragraph number="719" type="verses" book="ap1">
_Tatthāpi bhavanaṃ mayhaṃ,_  
_sattabhūmaṃ samuggataṃ;_  
_Kūṭāgāravarūpetaṃ,_  
_ekatthambhaṃ manoramaṃ._  

</Paragraph>

</Division>
<Division number="23" book="ap1">

<Paragraph number="720" type="verses" book="ap1">
_Ekavīsatikappamhi,_  
_udeno nāma khattiyo;_  
_Tatrāpi bhavanaṃ mayhaṃ,_  
_sattabhūmaṃ samuggataṃ._  

</Paragraph>

</Division>
<Division number="24" book="ap1">

<Paragraph number="721" type="verses" book="ap1">
_Yaṃ yaṃ yonupapajjāmi,_  
_devattaṃ atha mānusaṃ;_  
_Anubhomi sukhaṃ sabbaṃ,_  
_ekatthambhassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="25" book="ap1">

<Paragraph number="722" type="verses" book="ap1">
_Catunnavutito kappe,_  
_yaṃ thambhamadadaṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_ekatthambhassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="26" book="ap1">

<Paragraph number="723" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_vimokkhāpi ca aṭṭhime;_  
_Chaḷabhiññā sacchikatā,_  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="724" book="ap1">
Itthaṃ sudaṃ āyasmā ekatthambhiko thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="725" type="center" book="ap1">
Ekatthambhikattherassāpadānaṃ dutiyaṃ.
</Paragraph>
