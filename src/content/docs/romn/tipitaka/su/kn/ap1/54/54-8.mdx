_“Padumuttaro nāma jino,_  
_sabbalokavidū muni;_  
_Ito satasahassamhi,_  
_kappe uppajji cakkhumā._  

</Paragraph>

</Division>
<Division number="252" book="ap1">

<Paragraph number="7213" type="verses" book="ap1">
_Ovādako viññāpako,_  
_tārako sabbapāṇinaṃ;_  
_Desanākusalo buddho,_  
_tāresi janataṃ bahuṃ._  

</Paragraph>

</Division>
<Division number="253" book="ap1">

<Paragraph number="7214" type="verses" book="ap1">
_Anukampako kāruṇiko,_  
_hitesī sabbapāṇinaṃ;_  
_Sampatte titthiye sabbe,_  
_pañcasīle patiṭṭhapi._  

</Paragraph>

</Division>
<Division number="254" book="ap1">

<Paragraph number="7215" type="verses" book="ap1">
_Evaṃ nirākulaṃ āsi,_  
_suññataṃ titthiyehi ca;_  
_Vicittaṃ arahantehi,_  
_vasībhūtehi tādibhi._  

</Paragraph>

</Division>
<Division number="255" book="ap1">

<Paragraph number="7216" type="verses" book="ap1">
_Ratanānaṭṭhapaññāsaṃ,_  
_uggato so mahāmuni;_  
_Kañcanagghiyasaṅkāso,_  
_bāttiṃsavaralakkhaṇo._  

</Paragraph>

</Division>
<Division number="256" book="ap1">

<Paragraph number="7217" type="verses" book="ap1">
_Vassasatasahassāni,_  
_āyu vijjati tāvade;_  
_Tāvatā tiṭṭhamāno so,_  
_tāresi janataṃ bahuṃ._  

</Paragraph>

</Division>
<Division number="257" book="ap1">

<Paragraph number="7218" type="verses" book="ap1">
_Tadāhaṃ haṃsavatiyā,_  
_brāhmaṇo sādhusammato;_  
_Upecca lokapajjotaṃ,_  
_assosiṃ dhammadesanaṃ._  

</Paragraph>

</Division>
<Division number="258" book="ap1">

<Paragraph number="7219" type="verses" book="ap1">
_Tadā mahāparisatiṃ,_  
_mahāparisasāvakaṃ;_  
_Ṭhapentaṃ etadaggamhi,_  
_sutvāna mudito ahaṃ._  

</Paragraph>

</Division>
<Division number="259" book="ap1">

<Paragraph number="7220" type="verses" book="ap1">
_Mahatā parivārena,_  
_nimantetvā mahājinaṃ;_  
_Brāhmaṇānaṃ sahassena,_  
_sahadānamadāsahaṃ._  

</Paragraph>

</Division>
<Division number="260" book="ap1">

<Paragraph number="7221" type="verses" book="ap1">
_Mahādānaṃ daditvāna,_  
_abhivādiya nāyakaṃ;_  
_Ekamantaṃ ṭhito haṭṭho,_  
_idaṃ vacanamabraviṃ._  

</Paragraph>

</Division>
<Division number="261" book="ap1">

<Paragraph number="7222" type="verses" book="ap1">
_‘Tayi saddhāya me vīra,_  
_adhikāraguṇena ca;_  
_Parisā mahatī hotu,_  
_nibbattassa tahiṃ tahiṃ’._  

</Paragraph>

</Division>
<Division number="262" book="ap1">

<Paragraph number="7223" type="verses" book="ap1">
_Tadā avoca parisaṃ,_  
_gajagajjitasussaro;_  
_Karavīkaruto satthā,_  
_‘etaṃ passatha brāhmaṇaṃ._  

</Paragraph>

</Division>
<Division number="263" book="ap1">

<Paragraph number="7224" type="verses" book="ap1">
_Hemavaṇṇaṃ mahābāhuṃ,_  
_kamalānanalocanaṃ;_  
_Udaggatanujaṃ haṭṭhaṃ,_  
_saddhavantaṃ guṇe mama._  

</Paragraph>

</Division>
<Division number="264" book="ap1">

<Paragraph number="7225" type="verses" book="ap1">
_Esa patthayate ṭhānaṃ,_  
_sīhaghosassa bhikkhuno;_  
_Anāgatamhi addhāne,_  
_lacchase taṃ manorathaṃ._  

</Paragraph>

</Division>
<Division number="265" book="ap1">

<Paragraph number="7226" type="verses" book="ap1">
_Satasahassito kappe,_  
_okkākakulasambhavo;_  
_Gotamo nāma gottena,_  
_satthā loke bhavissati._  

</Paragraph>

</Division>
<Division number="266" book="ap1">

<Paragraph number="7227" type="verses" book="ap1">
_Tassa dhammesu dāyādo,_  
_oraso dhammanimmito;_  
_Kassapo nāma gottena,_  
_hessati satthu sāvako’._  

</Paragraph>

</Division>
<Division number="267" book="ap1">

<Paragraph number="7228" type="verses" book="ap1">
_Ito dvenavute kappe,_  
_ahu satthā anuttaro;_  
_Anūpamo asadiso,_  
_phusso lokagganāyako._  

</Paragraph>

</Division>
<Division number="268" book="ap1">

<Paragraph number="7229" type="verses" book="ap1">
_So ca sabbaṃ tamaṃ hantvā,_  
_vijaṭetvā mahājaṭaṃ;_  
_Vassate amataṃ vuṭṭhiṃ,_  
_tappayanto sadevakaṃ._  

</Paragraph>

</Division>
<Division number="269" book="ap1">

<Paragraph number="7230" type="verses" book="ap1">
_Tadā hi bārāṇasiyaṃ,_  
_rājā paccā ahumhase;_  
_Bhātaromha tayo sabbe,_  
_saṃvisaṭṭhāva rājino._  

</Paragraph>

</Division>
<Division number="270" book="ap1">

<Paragraph number="7231" type="verses" book="ap1">
_Vīraṅgarūpā balino,_  
_saṅgāme aparājitā;_  
_Tadā kupitapaccanto,_  
_amhe āha mahīpati._  

</Paragraph>

</Division>
<Division number="271" book="ap1">

<Paragraph number="7232" type="verses" book="ap1">
_‘Etha gantvāna paccantaṃ,_  
_sodhetvā aṭṭavībalaṃ;_  
_Khemaṃ vijiritaṃ katvā,_  
_puna dethāti bhāsatha’._  

</Paragraph>

</Division>
<Division number="272" book="ap1">

<Paragraph number="7233" type="verses" book="ap1">
_Tato mayaṃ avocumha,_  
_yadi deyyāsi nāyakaṃ;_  
_Upaṭṭhānāya amhākaṃ,_  
_sādhayissāma vo tato._  

</Paragraph>

</Division>
<Division number="273" book="ap1">

<Paragraph number="7234" type="verses" book="ap1">
_Tato mayaṃ laddhavarā,_  
_bhūmipālena pesitā;_  
_Nikkhittasatthaṃ paccantaṃ,_  
_katvā punarupacca taṃ._  

</Paragraph>

</Division>
<Division number="274" book="ap1">

<Paragraph number="7235" type="verses" book="ap1">
_Yācitvā satthupaṭṭhānaṃ,_  
_rājānaṃ lokanāyakaṃ;_  
_Munivīraṃ labhitvāna,_  
_yāvajīvaṃ yajimha taṃ._  

</Paragraph>

</Division>
<Division number="275" book="ap1">

<Paragraph number="7236" type="verses" book="ap1">
_Mahagghāni ca vatthāni,_  
_paṇītāni rasāni ca;_  
_Senāsanāni rammāni,_  
_bhesajjāni hitāni ca._  

</Paragraph>

</Division>
<Division number="276" book="ap1">

<Paragraph number="7237" type="verses" book="ap1">
_Datvā sasaṃghamunino,_  
_dhammenuppāditāni no;_  
_Sīlavanto kāruṇikā,_  
_bhāvanāyuttamānasā._  

</Paragraph>

</Division>
<Division number="277" book="ap1">

<Paragraph number="7238" type="verses" book="ap1">
_Saddhā paricaritvāna,_  
_mettacittena nāyakaṃ;_  
_Nibbute tamhi lokagge,_  
_pūjaṃ katvā yathābalaṃ._  

</Paragraph>

</Division>
<Division number="278" book="ap1">

<Paragraph number="7239" type="verses" book="ap1">
_Tato cutā santusitaṃ,_  
_gatā tattha mahāsukhaṃ;_  
_Anubhūtā mayaṃ sabbe,_  
_buddhapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="279" book="ap1">

<Paragraph number="7240" type="verses" book="ap1">
_Māyākāro yathā raṅge,_  
_dassesi vikatiṃ bahuṃ;_  
_Tathā bhave bhamantohaṃ,_  
_videhādhipatī ahuṃ._  

</Paragraph>

</Division>
<Division number="280" book="ap1">

<Paragraph number="7241" type="verses" book="ap1">
_Guṇāceḷassa vākyena,_  
_micchādiṭṭhigatāsayo;_  
_Narakaṃ maggamārūḷho,_  
_rucāya mama dhītuyā._  

</Paragraph>

</Division>
<Division number="281" book="ap1">

<Paragraph number="7242" type="verses" book="ap1">
_Ovādaṃ nādiyitvāna,_  
_brahmunā nāradenahaṃ;_  
_Bahudhā saṃsito santo,_  
_diṭṭhiṃ hitvāna pāpikaṃ._  

</Paragraph>

</Division>
<Division number="282" book="ap1">

<Paragraph number="7243" type="verses" book="ap1">
_Pūrayitvā visesena,_  
_dasa kammapathānihaṃ;_  
_Hitvāna dehamagamiṃ,_  
_saggaṃ sabhavanaṃ yathā._  

</Paragraph>

</Division>
<Division number="283" book="ap1">

<Paragraph number="7244" type="verses" book="ap1">
_Pacchime bhave sampatte,_  
_brahmabandhu ahosahaṃ;_  
_Bārāṇasiyaṃ phītāyaṃ,_  
_jāto vippamahākule._  

</Paragraph>

</Division>
<Division number="284" book="ap1">

<Paragraph number="7245" type="verses" book="ap1">
_Maccubyādhijarā bhīto,_  
_ogāhetvā mahāvanaṃ;_  
_Nibbānaṃ padamesanto,_  
_jaṭilesu paribbajiṃ._  

</Paragraph>

</Division>
<Division number="285" book="ap1">

<Paragraph number="7246" type="verses" book="ap1">
_Tadā dve bhātaro mayhaṃ,_  
_pabbajiṃsu mayā saha;_  
_Uruvelāyaṃ māpetvā,_  
_assamaṃ nivasiṃ ahaṃ._  

</Paragraph>

</Division>
<Division number="286" book="ap1">

<Paragraph number="7247" type="verses" book="ap1">
_Kassapo nāma gottena,_  
_uruveḷanivāsiko;_  
_Tato me āsi paññatti,_  
_uruveḷakassapo iti._  

</Paragraph>

</Division>
<Division number="287" book="ap1">

<Paragraph number="7248" type="verses" book="ap1">
_Nadīsakāse bhātā me,_  
_nadīkassapasavhayo;_  
_Āsī sakāsanāmena,_  
_gayāyaṃ gayākassapo._  

</Paragraph>

</Division>
<Division number="288" book="ap1">

<Paragraph number="7249" type="verses" book="ap1">
_Dve satāni kaniṭṭhassa,_  
_tīṇi majjhassa bhātuno;_  
_Mama pañca satānūnā,_  
_sissā sabbe mamānugā._  

</Paragraph>

</Division>
<Division number="289" book="ap1">

<Paragraph number="7250" type="verses" book="ap1">
_Tadā upecca maṃ buddho,_  
_katvāna vividhāni me;_  
_Pāṭihīrāni lokaggo,_  
_vinesi narasārathi._  

</Paragraph>

</Division>
<Division number="290" book="ap1">

<Paragraph number="7251" type="verses" book="ap1">
_Sahassaparivārena,_  
_ahosiṃ ehibhikkhuko;_  
_Teheva saha sabbehi,_  
_arahattamapāpuṇiṃ._  

</Paragraph>

</Division>
<Division number="291" book="ap1">

<Paragraph number="7252" type="verses" book="ap1">
_Te cevaññe ca bahavo,_  
_sissā maṃ parivārayuṃ;_  
_Sāsituñca samatthohaṃ,_  
_tato maṃ isisattamo._  

</Paragraph>

</Division>
<Division number="292" book="ap1">

<Paragraph number="7253" type="verses" book="ap1">

_etadagge ṭhapesi maṃ;_  
_Aho buddhe kataṃ kāraṃ,_  
_saphalaṃ me ajāyatha._  

</Paragraph>

</Division>
