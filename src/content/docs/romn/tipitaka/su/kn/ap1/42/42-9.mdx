<Paragraph number="5435" type="verses" book="ap1">
_“Vanacārī pure āsiṃ,_  
_satataṃ vanakammiko;_  
_Patthodanaṃ gahetvāna,_  
_kammantaṃ agamāsahaṃ._  

</Paragraph>

</Division>
<Division number="171" book="ap1" bookNo="20" page="443">

<Paragraph number="5436" type="verses" book="ap1">
_Tatthaddasāsiṃ sambuddhaṃ,_  
_sayambhuṃ aparājitaṃ;_  
_Vanā piṇḍāya nikkhantaṃ,_  
_disvā cittaṃ pasādayiṃ._  

</Paragraph>

</Division>
<Division number="172" book="ap1" bookNo="20" page="443">

<Paragraph number="5437" type="verses" book="ap1">
_Parakammāyane yutto,_  
_puññañca me na vijjati;_  
_Ayaṃ patthodano atthi,_  
_bhojayissāmahaṃ muniṃ._  

</Paragraph>

</Division>
<Division number="173" book="ap1" bookNo="20" page="443">

<Paragraph number="5438" type="verses" book="ap1">
_Patthodanaṃ gahetvāna,_  
_sayambhussa adāsahaṃ;_  
_Mama nijjhāyamānassa,_  
_paribhuñji tadā muni._  

</Paragraph>

</Division>
<Division number="174" book="ap1" bookNo="20" page="443">

<Paragraph number="5439" type="verses" book="ap1">
_Tena kammena sukatena,_  
_cetanāpaṇidhīhi ca;_  
_Jahitvā mānusaṃ dehaṃ,_  
_tāvatiṃsamagacchahaṃ._  

</Paragraph>

</Division>
<Division number="175" book="ap1" bookNo="20" page="443">

<Paragraph number="5440" type="verses" book="ap1">
_Chattiṃsakkhattuṃ devindo,_  
_devarajjamakārayiṃ;_  
_Tettiṃsakkhattuṃ rājā ca,_  
_cakkavattī ahosahaṃ._  

</Paragraph>

</Division>
<Division number="176" book="ap1" bookNo="20" page="443">

<Paragraph number="5441" type="verses" book="ap1">
_Padesarajjaṃ vipulaṃ,_  
_gaṇanāto asaṅkhiyaṃ;_  
_Sukhito yasavā homi,_  
_patthodanassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="177" book="ap1" bookNo="20" page="443">

<Paragraph number="5442" type="verses" book="ap1">
_Bhavābhave saṃsaranto,_  
_labhāmi amitaṃ dhanaṃ;_  
_Bhoge me ūnatā natthi,_  
_patthodanassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="178" book="ap1" bookNo="20" page="444">

<Paragraph number="5443" type="verses" book="ap1">
_Nadīsotapaṭibhāgā,_  
_bhogā nibbattare mama;_  
_Parimetuṃ na sakkomi,_  
_patthodanassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="179" book="ap1" bookNo="20" page="444">

<Paragraph number="5444" type="verses" book="ap1">
_Imaṃ khāda imaṃ bhuñja,_  
_imamhi sayane saya;_  
_Tenāhaṃ sukhito homi,_  
_patthodanassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="180" book="ap1" bookNo="20" page="444">

<Paragraph number="5445" type="verses" book="ap1">
_Catunnavutito kappe,_  
_yaṃ dānamadadiṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_patthodanassidaṃ phalaṃ._  

</Paragraph>

</Division>

_ …pe… _  
_viharāmi anāsavo._  

</Paragraph>

</Division>

_ …pe… _  
_kataṃ buddhassa sāsanaṃ._  

</Paragraph>

</Division>

_ …pe… _
