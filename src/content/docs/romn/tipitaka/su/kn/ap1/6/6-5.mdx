<Division number="33" book="ap1">

<Paragraph number="1416" type="verses" book="ap1">
_“Padumuttarabuddhassa,_  
_ājānīyamadāsahaṃ;_  
_Niyyādetvāna sambuddhe,_  
_agamāsiṃ sakaṃ gharaṃ._  

</Paragraph>

</Division>
<Division number="34" book="ap1">

<Paragraph number="1417" type="verses" book="ap1">
_Devalo nāma nāmena,_  
_satthuno aggasāvako;_  
_Varadhammassa dāyādo,_  
_āgacchi mama santikaṃ._  

</Paragraph>

</Division>
<Division number="35" book="ap1">

<Paragraph number="1418" type="verses" book="ap1">
_Sapattabhāro bhagavā,_  
_ājāneyyo na kappati;_  
_Tava saṅkappamaññāya,_  
_adhivāsesi cakkhumā._  

</Paragraph>

</Division>
<Division number="36" book="ap1">

<Paragraph number="1419" type="verses" book="ap1">
_Agghāpetvā vātajavaṃ,_  
_sindhavaṃ sīghavāhanaṃ;_  
_Padumuttarabuddhassa,_  
_khamanīyamadāsahaṃ._  

</Paragraph>

</Division>
<Division number="37" book="ap1">

<Paragraph number="1420" type="verses" book="ap1">
_Yaṃ yaṃ yonupapajjāmi,_  
_devattaṃ atha mānusaṃ;_  
_Khamanīyaṃ vātajavaṃ,_  
_cittaṃ nibbattate mama._  

</Paragraph>

</Division>
<Division number="38" book="ap1">

<Paragraph number="1421" type="verses" book="ap1">
_Lābhaṃ tesaṃ suladdhaṃva,_  
_ye labhantupasampadaṃ;_  
_Punapi payirupāseyyaṃ,_  
_buddho loke sace bhave._  

</Paragraph>

</Division>
<Division number="39" book="ap1">

<Paragraph number="1422" type="verses" book="ap1">
_Aṭṭhavīsatikkhattuṃhaṃ,_  
_rājā āsiṃ mahabbalo;_  
_Cāturanto vijitāvī,_  
_jambusaṇḍassa issaro._  

</Paragraph>

</Division>
<Division number="40" book="ap1">

<Paragraph number="1423" type="verses" book="ap1">
_Idaṃ pacchimakaṃ mayhaṃ,_  
_carimo vattate bhavo;_  
_Pattosmi acalaṃ ṭhānaṃ,_  
_hitvā jayaparājayaṃ._  

</Paragraph>

</Division>
<Division number="41" book="ap1">

<Paragraph number="1424" type="verses" book="ap1">
_Catutiṃsasahassamhi,_  
_mahātejosi khattiyo;_  
_Sattaratanasampanno,_  
_cakkavattī mahabbalo._  

</Paragraph>

</Division>
<Division number="42" book="ap1">

<Paragraph number="1425" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="1426" book="ap1">
Itthaṃ sudaṃ āyasmā opavayho thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="1427" type="center" book="ap1">
Opavayhattherassāpadānaṃ pañcamaṃ.
</Paragraph>
