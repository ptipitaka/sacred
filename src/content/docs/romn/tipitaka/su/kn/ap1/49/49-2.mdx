<Division number="9" book="ap1">

<Paragraph number="6195" type="verses" book="ap1">
_“Ajjhāyako mantadharo,_  
_tiṇṇaṃ vedāna pāragū;_  
_Lakkhaṇe itihāse ca,_  
_sanighaṇḍusakeṭubhe._  

</Paragraph>

</Division>
<Division number="10" book="ap1">

<Paragraph number="6196" type="verses" book="ap1">
_Nadīsotapaṭibhāgā,_  
_sissā āyanti me tadā;_  
_Tesāhaṃ mante vācemi,_  
_rattindivamatandito._  

</Paragraph>

</Division>
<Division number="11" book="ap1">

<Paragraph number="6197" type="verses" book="ap1">
_Siddhattho nāma sambuddho,_  
_loke uppajji tāvade;_  
_Tamandhakāraṃ nāsetvā,_  
_ñāṇālokaṃ pavattayi._  

</Paragraph>

</Division>
<Division number="12" book="ap1">

<Paragraph number="6198" type="verses" book="ap1">
_Mama aññataro sisso,_  
_sissānaṃ so kathesi me;_  
_Sutvāna te etamatthaṃ,_  
_ārocesuṃ mamaṃ tadā._  

</Paragraph>

</Division>
<Division number="13" book="ap1">

<Paragraph number="6199" type="verses" book="ap1">
_‘Buddho loke samuppanno,_  
_sabbaññū lokanāyako;_  
_Tassānuvattati jano,_  
_lābho amhaṃ na vijjati._  

</Paragraph>

</Division>
<Division number="14" book="ap1">

<Paragraph number="6200" type="verses" book="ap1">
_Adhiccuppattikā buddhā,_  
_cakkhumanto mahāyasā;_  
_Yannūnāhaṃ buddhaseṭṭhaṃ,_  
_passeyyaṃ lokanāyakaṃ’._  

</Paragraph>

</Division>
<Division number="15" book="ap1">

<Paragraph number="6201" type="verses" book="ap1">
_Ajinaṃ me gahetvāna,_  
_vākacīraṃ kamaṇḍaluṃ;_  
_Assamā abhinikkhamma,_  
_sisse āmantayiṃ ahaṃ._  

</Paragraph>

</Division>
<Division number="16" book="ap1">

<Paragraph number="6202" type="verses" book="ap1">
_Odumbarikapupphaṃva,_  
_candamhi sasakaṃ yathā;_  
_Vāyasānaṃ yathā khīraṃ,_  
_dullabho lokanāyako._  

</Paragraph>

</Division>
<Division number="17" book="ap1">

<Paragraph number="6203" type="verses" book="ap1">
_Buddho lokamhi uppanno,_  
_manussattampi dullabhaṃ;_  
_Ubhosu vijjamānesu,_  
_savanañca sudullabhaṃ._  

</Paragraph>

</Division>
<Division number="18" book="ap1">

<Paragraph number="6204" type="verses" book="ap1">
_Buddho loke samuppanno,_  
_cakkhuṃ lacchāma no bhavaṃ;_  
_Etha sabbe gamissāma,_  
_sammāsambuddhasantikaṃ._  

</Paragraph>

</Division>
<Division number="19" book="ap1">

<Paragraph number="6205" type="verses" book="ap1">
_Kamaṇḍaludharā sabbe,_  
_kharājinanivāsino;_  
_Te jaṭā bhārabharitā,_  
_nikkhamuṃ vipinā tadā._  

</Paragraph>

</Division>
<Division number="20" book="ap1">

<Paragraph number="6206" type="verses" book="ap1">
_Yugamattaṃ pekkhamānā,_  
_uttamatthaṃ gavesino;_  
_Āsattidosarahitā,_  
_asambhītāva kesarī._  

</Paragraph>

</Division>
<Division number="21" book="ap1">

<Paragraph number="6207" type="verses" book="ap1">
_Appakiccā aloluppā,_  
_nipakā santavuttino;_  
_Uñchāya caramānā te,_  
_buddhaseṭṭhamupāgamuṃ._  

</Paragraph>

</Division>
<Division number="22" book="ap1">

<Paragraph number="6208" type="verses" book="ap1">
_Diyaḍḍhayojane sese,_  
_byādhi me upapajjatha;_  
_Buddhaseṭṭhaṃ saritvāna,_  
_tattha kālaṅkato ahaṃ._  

</Paragraph>

</Division>
<Division number="23" book="ap1">

<Paragraph number="6209" type="verses" book="ap1">
_Catunnavutito kappe,_  
_yaṃ saññamalabhiṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_buddhasaññāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="24" book="ap1">

<Paragraph number="6210" type="verses" book="ap1">
_Kilesā jhāpitā mayhaṃ,_  
_ …pe… _  
_viharāmi anāsavo._  

</Paragraph>

</Division>
<Division number="25" book="ap1">

<Paragraph number="6211" type="verses" book="ap1">
_Svāgataṃ vata me āsi,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ._  

</Paragraph>

</Division>
<Division number="26" book="ap1">

<Paragraph number="6212" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="6213" book="ap1">
Itthaṃ sudaṃ āyasmā buddhasaññako thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="6214" type="center" book="ap1">
Buddhasaññakattherassāpadānaṃ dutiyaṃ.
</Paragraph>
