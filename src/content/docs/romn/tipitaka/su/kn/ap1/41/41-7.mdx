<Division number="183" book="ap1">

<Paragraph number="5041" type="verses" book="ap1">
_“Pabbhārakūṭaṃ nissāya,_  
_anomo nāma tāpaso;_  
_Assamaṃ sukataṃ katvā,_  
_paṇṇasāle vasī tadā._  

</Paragraph>

</Division>
<Division number="184" book="ap1">

<Paragraph number="5042" type="verses" book="ap1">
_Siddhaṃ tassa tapo kammaṃ,_  
_siddhipatto sake bale;_  
_Sakasāmaññavikkanto,_  
_ātāpī nipako muni._  

</Paragraph>

</Division>
<Division number="185" book="ap1">

<Paragraph number="5043" type="verses" book="ap1">
_Visārado sasamaye,_  
_paravāde ca kovido;_  
_Paṭṭho bhūmantalikkhamhi,_  
_uppātamhi ca kovido._  

</Paragraph>

</Division>
<Division number="186" book="ap1">

<Paragraph number="5044" type="verses" book="ap1">
_Vītasoko nirārambho,_  
_appāhāro alolupo;_  
_Lābhālābhena santuṭṭho,_  
_jhāyī jhānarato muni._  

</Paragraph>

</Division>
<Division number="187" book="ap1">

<Paragraph number="5045" type="verses" book="ap1">
_Piyadassī nāma sambuddho,_  
_aggo kāruṇiko muni;_  
_Satte tāretukāmo so,_  
_karuṇāya pharī tadā._  

</Paragraph>

</Division>
<Division number="188" book="ap1">

<Paragraph number="5046" type="verses" book="ap1">
_Bodhaneyyaṃ janaṃ disvā,_  
_piyadassī mahāmuni;_  
_Cakkavāḷasahassampi,_  
_gantvā ovadate muni._  

</Paragraph>

</Division>
<Division number="189" book="ap1">

<Paragraph number="5047" type="verses" book="ap1">
_Mamuddharitukāmo so,_  
_mamassamamupāgami;_  
_Na diṭṭho me jino pubbe,_  
_na sutopi ca kassaci._  

</Paragraph>

</Division>
<Division number="190" book="ap1">

<Paragraph number="5048" type="verses" book="ap1">
_Uppātā supinā mayhaṃ,_  
_lakkhaṇā suppakāsitā;_  
_Paṭṭho bhūmantalikkhamhi,_  
_nakkhattapadakovido._  

</Paragraph>

</Division>
<Division number="191" book="ap1">

<Paragraph number="5049" type="verses" book="ap1">
_Sohaṃ buddhassa sutvāna,_  
_tattha cittaṃ pasādayiṃ;_  
_Tiṭṭhanto vā nisinno vā,_  
_sarāmi niccakālikaṃ._  

</Paragraph>

</Division>
<Division number="192" book="ap1">

<Paragraph number="5050" type="verses" book="ap1">
_Mayi evaṃ sarantamhi,_  
_bhagavāpi anussari;_  
_Buddhaṃ anussarantassa,_  
_pīti me hoti tāvade._  

</Paragraph>

</Division>
<Division number="193" book="ap1">

<Paragraph number="5051" type="verses" book="ap1">
_Kālañca punarāgamma,_  
_upesi maṃ mahāmuni;_  
_Sampattepi na jānāmi,_  
_ayaṃ buddho mahāmuni._  

</Paragraph>

</Division>
<Division number="194" book="ap1">

<Paragraph number="5052" type="verses" book="ap1">
_Anukampako kāruṇiko,_  
_piyadassī mahāmuni;_  
_Sañjānāpesi attānaṃ,_  
_‘ahaṃ buddho sadevake’._  

</Paragraph>

</Division>
<Division number="195" book="ap1">

<Paragraph number="5053" type="verses" book="ap1">
_Sañjānitvāna sambuddhaṃ,_  
_piyadassiṃ mahāmuniṃ;_  
_Sakaṃ cittaṃ pasādetvā,_  
_idaṃ vacanamabraviṃ._  

</Paragraph>

</Division>
<Division number="196" book="ap1">

<Paragraph number="5054" type="verses" book="ap1">
_‘Aññe pīṭhe ca pallaṅke,_  
_āsandīsu nisīdare;_  
_Tuvampi sabbadassāvī,_  
_nisīda ratanāsane’._  

</Paragraph>

</Division>
<Division number="197" book="ap1">

<Paragraph number="5055" type="verses" book="ap1">
_Sabbaratanamayaṃ pīṭhaṃ,_  
_nimminitvāna tāvade;_  
_Piyadassissa munino,_  
_adāsiṃ iddhinimmitaṃ._  

</Paragraph>

</Division>
<Division number="198" book="ap1">

<Paragraph number="5056" type="verses" book="ap1">
_Ratane ca nisinnassa,_  
_pīṭhake iddhinimmite;_  
_Kumbhamattaṃ jambuphalaṃ,_  
_adāsiṃ tāvade ahaṃ._  

</Paragraph>

</Division>
<Division number="199" book="ap1">

<Paragraph number="5057" type="verses" book="ap1">
_Mama hāsaṃ janetvāna,_  
_paribhuñji mahāmuni;_  
_Tadā cittaṃ pasādetvā,_  
_satthāraṃ abhivādayiṃ._  

</Paragraph>

</Division>
<Division number="200" book="ap1">

<Paragraph number="5058" type="verses" book="ap1">
_Piyadassī tu bhagavā,_  
_lokajeṭṭho narāsabho;_  
_Ratanāsanamāsīno,_  
_imā gāthā abhāsatha._  

</Paragraph>

</Division>
<Division number="201" book="ap1">

<Paragraph number="5059" type="verses" book="ap1">
_‘Yo me ratanamayaṃ pīṭhaṃ,_  
_amatañca phalaṃ adā;_  
_Tamahaṃ kittayissāmi,_  
_suṇātha mama bhāsato._  

</Paragraph>

</Division>
<Division number="202" book="ap1">

<Paragraph number="5060" type="verses" book="ap1">
_Sattasattati kappāni,_  
_devaloke ramissati;_  
_Pañcasattatikkhattuñca,_  
_cakkavattī bhavissati._  

</Paragraph>

</Division>
<Division number="203" book="ap1">

<Paragraph number="5061" type="verses" book="ap1">
_Dvattiṃsakkhattuṃ devindo,_  
_devarajjaṃ karissati;_  
_Padesarajjaṃ vipulaṃ,_  
_gaṇanāto asaṅkhiyaṃ._  

</Paragraph>

</Division>
<Division number="204" book="ap1">

<Paragraph number="5062" type="verses" book="ap1">
_Soṇṇamayaṃ rūpimayaṃ,_  
_pallaṅkaṃ sukataṃ bahuṃ;_  
_Lohitaṅgamayañceva,_  
_lacchati ratanāmayaṃ._  

</Paragraph>

</Division>
<Division number="205" book="ap1">

<Paragraph number="5063" type="verses" book="ap1">
_Caṅkamantampi manujaṃ,_  
_puññakammasamaṅginaṃ;_  
_Pallaṅkāni anekāni,_  
_parivāressare tadā._  

</Paragraph>

</Division>
<Division number="206" book="ap1">

<Paragraph number="5064" type="verses" book="ap1">
_Kūṭāgārā ca pāsādā,_  
_sayanañca mahārahaṃ;_  
_Imassa cittamaññāya,_  
_nibbattissanti tāvade._  

</Paragraph>

</Division>
<Division number="207" book="ap1">

<Paragraph number="5065" type="verses" book="ap1">
_Saṭṭhi nāgasahassāni,_  
_sabbālaṅkārabhūsitā;_  
_Suvaṇṇakacchā mātaṅgā,_  
_hemakappanavāsasā._  

</Paragraph>

</Division>
<Division number="208" book="ap1">

<Paragraph number="5066" type="verses" book="ap1">
_Ārūḷhā gāmaṇīyehi,_  
_tomaraṅkusapāṇibhi;_  
_Imaṃ paricarissanti,_  
_ratnapīṭhassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="209" book="ap1">

<Paragraph number="5067" type="verses" book="ap1">
_Saṭṭhi assasahassāni,_  
_sabbālaṅkārabhūsitā;_  
_Ājānīyāva jātiyā,_  
_sindhavā sīghavāhino._  

</Paragraph>

</Division>
<Division number="210" book="ap1">

<Paragraph number="5068" type="verses" book="ap1">
_Ārūḷhā gāmaṇīyehi,_  
_illiyācāpadhāribhi;_  
_Tepimaṃ paricarissanti,_  
_ratnapīṭhassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="211" book="ap1">

<Paragraph number="5069" type="verses" book="ap1">
_Saṭṭhi rathasahassāni,_  
_sabbālaṅkārabhūsitā;_  
_Dīpā athopi veyagghā,_  
_sannaddhā ussitaddhajā._  

</Paragraph>

</Division>
<Division number="212" book="ap1">

<Paragraph number="5070" type="verses" book="ap1">
_Ārūḷhā gāmaṇīyehi,_  
_cāpahatthehi vammibhi;_  
_Parivāressantimaṃ niccaṃ,_  
_ratnapīṭhassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="213" book="ap1">

<Paragraph number="5071" type="verses" book="ap1">
_Saṭṭhi dhenusahassāni,_  
_dohaññā puṅgavūsabhe;_  
_Vacchake janayissanti,_  
_ratnapīṭhassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="214" book="ap1">

<Paragraph number="5072" type="verses" book="ap1">
_Soḷasitthisahassāni,_  
_sabbālaṅkārabhūsitā;_  
_Vicittavatthābharaṇā,_  
_āmukkamaṇikuṇḍalā._  

</Paragraph>

</Division>
<Division number="215" book="ap1">

<Paragraph number="5073" type="verses" book="ap1">
_Aḷārapamhā hasulā,_  
_susaññā tanumajjhimā;_  
_Parivāressantimaṃ niccaṃ,_  
_ratnapīṭhassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="216" book="ap1">

<Paragraph number="5074" type="verses" book="ap1">
_Aṭṭhārase kappasate,_  
_gotamo nāma cakkhumā;_  
_Tamandhakāraṃ vidhamitvā,_  
_buddho loke bhavissati._  

</Paragraph>

</Division>
<Division number="217" book="ap1">

<Paragraph number="5075" type="verses" book="ap1">
_Tassa dassanamāgamma,_  
_pabbajissatikiñcano;_  
_Tosayitvāna satthāraṃ,_  
_sāsanebhiramissati._  

</Paragraph>

</Division>
<Division number="218" book="ap1">

<Paragraph number="5076" type="verses" book="ap1">
_Tassa dhammaṃ suṇitvāna,_  
_kilese ghātayissati;_  
_Sabbāsave pariññāya,_  
_nibbāyissatināsavo’._  

</Paragraph>

</Division>
<Division number="219" book="ap1">

<Paragraph number="5077" type="verses" book="ap1">
_Vīriyaṃ me dhuradhorayhaṃ,_  
_yogakkhemādhivāhanaṃ;_  
_Uttamatthaṃ patthayanto,_  
_sāsane viharāmahaṃ._  

</Paragraph>

</Division>
<Division number="220" book="ap1">

<Paragraph number="5078" type="verses" book="ap1">
_Idaṃ pacchimakaṃ mayhaṃ,_  
_carimo vattate bhavo;_  
_Sabbāsavā parikkhīṇā,_  
_natthi dāni punabbhavo._  

</Paragraph>

</Division>
