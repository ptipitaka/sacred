<Paragraph number="1144" type="verses" book="ap1">
_“Atthadassī tu bhagavā,_  
_sayambhū aparājito;_  
_Bhikkhusaṃghaparibyūḷho,_  
_rathiyaṃ paṭipajjatha._  

</Paragraph>

</Division>
<Division number="65" book="ap1" bookNo="20" page="96">

<Paragraph number="1145" type="verses" book="ap1">
_Sissehi samparivuto,_  
_gharamhā abhinikkhamiṃ;_  
_Nikkhamitvānahaṃ tattha,_  
_addasaṃ lokanāyakaṃ._  

</Paragraph>

</Division>
<Division number="66" book="ap1" bookNo="20" page="96">

<Paragraph number="1146" type="verses" book="ap1">
_Abhivādiya sambuddhaṃ,_  
_sire katvāna añjaliṃ;_  
_Sakaṃ cittaṃ pasādetvā,_  
_santhaviṃ lokanāyakaṃ._  

</Paragraph>

</Division>
<Division number="67" book="ap1" bookNo="20" page="96">

<Paragraph number="1147" type="verses" book="ap1">
_Yāvatā rūpino sattā,_  
_arūpī vā asaññino;_  
_Sabbe te tava ñāṇamhi,_  
_anto honti samogadhā._  

</Paragraph>

</Division>
<Division number="68" book="ap1" bookNo="20" page="97">

<Paragraph number="1148" type="verses" book="ap1">
_Sukhumacchikajālena,_  
_udakaṃ yo parikkhipe;_  
_Ye keci udake pāṇā,_  
_antojāle bhavanti te._  

</Paragraph>

</Division>
<Division number="69" book="ap1" bookNo="20" page="97">

<Paragraph number="1149" type="verses" book="ap1">
_Yesañca cetanā atthi,_  
_rūpino ca arūpino;_  
_Sabbe te tava ñāṇamhi,_  
_anto honti samogadhā._  

</Paragraph>

</Division>
<Division number="70" book="ap1" bookNo="20" page="97">

<Paragraph number="1150" type="verses" book="ap1">
_Samuddharasimaṃ lokaṃ,_  
_andhakārasamākulaṃ;_  
_Tava dhammaṃ suṇitvāna,_  
_kaṅkhāsotaṃ taranti te._  

</Paragraph>

</Division>
<Division number="71" book="ap1" bookNo="20" page="97">

<Paragraph number="1151" type="verses" book="ap1">
_Avijjānivute loke,_  
_andhakārena otthaṭe;_  
_Tava ñāṇamhi jotante,_  
_andhakārā padhaṃsitā._  

</Paragraph>

</Division>
<Division number="72" book="ap1" bookNo="20" page="97">

<Paragraph number="1152" type="verses" book="ap1">
_Tuvaṃ cakkhūsi sabbesaṃ,_  
_mahātamapanūdano;_  
_Tava dhammaṃ suṇitvāna,_  
_nibbāyati bahujjano._  

</Paragraph>

</Division>
<Division number="73" book="ap1" bookNo="20" page="97">

<Paragraph number="1153" type="verses" book="ap1">
_Puṭakaṃ pūrayitvāna,_  
_madhukhuddamaneḷakaṃ;_  
_Ubho hatthehi paggayha,_  
_upanesiṃ mahesino._  

</Paragraph>

</Division>
<Division number="74" book="ap1" bookNo="20" page="97">

<Paragraph number="1154" type="verses" book="ap1">
_Paṭiggaṇhi mahāvīro,_  
_sahatthena mahā isi;_  
_Bhuñjitvā tañca sabbaññū,_  
_vehāsaṃ nabhamuggami._  

</Paragraph>

</Division>
<Division number="75" book="ap1" bookNo="20" page="97">

<Paragraph number="1155" type="verses" book="ap1">
_Antalikkhe ṭhito satthā,_  
_atthadassī narāsabho;_  
_Mama cittaṃ pasādento,_  
_imā gāthā abhāsatha._  

</Paragraph>

</Division>
<Division number="76" book="ap1" bookNo="20" page="97">

<Paragraph number="1156" type="verses" book="ap1">
_‘Yenidaṃ thavitaṃ ñāṇaṃ,_  
_buddhaseṭṭho ca thomito;_  
_Tena cittappasādena,_  
_duggatiṃ so na gacchati._  

</Paragraph>

</Division>
<Division number="77" book="ap1" bookNo="20" page="97">

<Paragraph number="1157" type="verses" book="ap1">
_Catuddasañcakkhattuṃ so,_  
_devarajjaṃ karissati;_  
_Pathabyā rajjaṃ aṭṭhasataṃ,_  
_vasudhaṃ āvasissati._  

</Paragraph>

</Division>
<Division number="78" book="ap1" bookNo="20" page="97">

<Paragraph number="1158" type="verses" book="ap1">
_Pañceva satakkhattuñca,_  
_cakkavattī bhavissati;_  
_Padesarajjaṃ asaṅkheyyaṃ,_  
_mahiyā kārayissati._  

</Paragraph>

</Division>
<Division number="79" book="ap1" bookNo="20" page="97">

<Paragraph number="1159" type="verses" book="ap1">
_Ajjhāyako mantadharo,_  
_tiṇṇaṃ vedāna pāragū;_  
_Gotamassa bhagavato,_  
_sāsane pabbajissati._  

</Paragraph>

</Division>
<Division number="80" book="ap1" bookNo="20" page="97">

<Paragraph number="1160" type="verses" book="ap1">
_Gambhīraṃ nipuṇaṃ atthaṃ,_  
_ñāṇena vicinissati;_  
_Mogharājāti nāmena,_  
_hessati satthu sāvako._  

</Paragraph>

</Division>
<Division number="81" book="ap1" bookNo="20" page="98">

<Paragraph number="1161" type="verses" book="ap1">
_Tīhi vijjāhi sampannaṃ,_  
_katakiccamanāsavaṃ;_  
_Gotamo satthavāhaggo,_  
_etadagge ṭhapessati’._  

</Paragraph>

</Division>
<Division number="82" book="ap1" bookNo="20" page="98">

<Paragraph number="1162" type="verses" book="ap1">
_Hitvā mānusakaṃ yogaṃ,_  
_chetvāna bhavabandhanaṃ;_  
_Sabbāsave pariññāya,_  
_viharāmi anāsavo._  

</Paragraph>

</Division>
<Division number="83" book="ap1" bookNo="20" page="98">

<Paragraph number="1163" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="1164" book="ap1">
Itthaṃ sudaṃ āyasmā mogharājo thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="1165" type="center" book="ap1">
Mogharājattherassāpadānaṃ pañcamaṃ.
</Paragraph>
