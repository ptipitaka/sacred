<Paragraph number="1920" type="verses" book="ap1">
_“Himavantassāvidūre,_  
_vasāmi paṇṇasanthare;_  
_Ghāsesu gedhamāpanno,_  
_seyyasīlo cahaṃ tadā._  

</Paragraph>

</Division>
<Division number="54" book="ap1" bookNo="20" page="159">

<Paragraph number="1921" type="verses" book="ap1">
_Khaṇantālu kalambāni,_  
_biḷālitakkalāni ca;_  
_Kolaṃ bhallātakaṃ billaṃ,_  
_āhatvā paṭiyāditaṃ._  

</Paragraph>

</Division>
<Division number="55" book="ap1" bookNo="20" page="159">

<Paragraph number="1922" type="verses" book="ap1">
_Padumuttaro lokavidū,_  
_āhutīnaṃ paṭiggaho;_  
_Mama saṅkappamaññāya,_  
_āgacchi mama santikaṃ._  

</Paragraph>

</Division>
<Division number="56" book="ap1" bookNo="20" page="159">

<Paragraph number="1923" type="verses" book="ap1">
_Upāgataṃ mahānāgaṃ,_  
_devadevaṃ narāsabhaṃ;_  
_Biḷāliṃ paggahetvāna,_  
_pattamhi okiriṃ ahaṃ._  

</Paragraph>

</Division>
<Division number="57" book="ap1" bookNo="20" page="159">

<Paragraph number="1924" type="verses" book="ap1">
_Paribhuñji mahāvīro,_  
_tosayanto mamaṃ tadā;_  
_Paribhuñjitvāna sabbaññū,_  
_imaṃ gāthaṃ abhāsatha._  

</Paragraph>

</Division>
<Division number="58" book="ap1" bookNo="20" page="159">

<Paragraph number="1925" type="verses" book="ap1">
_‘Sakaṃ cittaṃ pasādetvā,_  
_biḷāliṃ me adā tuvaṃ;_  
_Kappānaṃ satasahassaṃ,_  
_duggatiṃ nupapajjasi’._  

</Paragraph>

</Division>
<Division number="59" book="ap1" bookNo="20" page="160">

<Paragraph number="1926" type="verses" book="ap1">
_Carimaṃ vattate mayhaṃ,_  
_bhavā sabbe samūhatā;_  
_Dhāremi antimaṃ dehaṃ,_  
_sammāsambuddhasāsane._  

</Paragraph>

</Division>
<Division number="60" book="ap1" bookNo="20" page="160">

<Paragraph number="1927" type="verses" book="ap1">
_Catupaññāsito kappe,_  
_sumekhaliyasavhayo;_  
_Sattaratanasampanno,_  
_cakkavattī mahabbalo._  

</Paragraph>

</Division>
<Division number="61" book="ap1" bookNo="20" page="160">

<Paragraph number="1928" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="1929" book="ap1">
Itthaṃ sudaṃ āyasmā biḷālidāyako thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="1930" type="center" book="ap1">
Biḷālidāyakattherassāpadānaṃ navamaṃ.
