<Division number="84" book="ap1" bookNo="20" page="177">

<Paragraph number="2151" type="verses" book="ap1">
_“Pabbate himavantamhi,_  
_vasāmi pabbatantare;_  
_Pulinaṃ sobhanaṃ disvā,_  
_buddhaseṭṭhaṃ anussariṃ._  

</Paragraph>

</Division>
<Division number="85" book="ap1" bookNo="20" page="177">

<Paragraph number="2152" type="verses" book="ap1">
_Ñāṇe upanidhā natthi,_  
_saṅkhāraṃ natthi satthuno;_  
_Sabbadhammaṃ abhiññāya,_  
_ñāṇena adhimuccati._  

</Paragraph>

</Division>
<Division number="86" book="ap1" bookNo="20" page="177">

<Paragraph number="2153" type="verses" book="ap1">
_Namo te purisājañña,_  
_namo te purisuttama;_  
_Ñāṇena te samo natthi,_  
_yāvatā ñāṇamuttamaṃ._  

</Paragraph>

</Division>
<Division number="87" book="ap1" bookNo="20" page="177">

<Paragraph number="2154" type="verses" book="ap1">
_Ñāṇe cittaṃ pasādetvā,_  
_kappaṃ saggamhi modahaṃ;_  
_Avasesesu kappesu,_  
_kusalaṃ caritaṃ mayā._  

</Paragraph>

</Division>
<Division number="88" book="ap1" bookNo="20" page="178">

<Paragraph number="2155" type="verses" book="ap1">
_Ekanavutito kappe,_  
_yaṃ saññamalabhiṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_ñāṇasaññāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="89" book="ap1" bookNo="20" page="178">

<Paragraph number="2156" type="verses" book="ap1">
_Ito sattatikappamhi,_  
_eko pulinapupphiyo;_  
_Sattaratanasampanno,_  
_cakkavattī mahabbalo._  

</Paragraph>

</Division>
<Division number="90" book="ap1" bookNo="20" page="178">

<Paragraph number="2157" type="verses" book="ap1">

_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="2158" book="ap1">

</Paragraph>
</Division>
