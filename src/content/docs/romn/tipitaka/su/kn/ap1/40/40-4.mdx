<Paragraph number="4637" type="verses" book="ap1">
_“Sindhuyā nadiyā tīre,_  
_sukato assamo mama;_  
_Tattha vācemahaṃ sisse,_  
_itihāsaṃ salakkhaṇaṃ._  

</Paragraph>

</Division>
<Division number="334" book="ap1">

<Paragraph number="4638" type="verses" book="ap1">
_Dhammakāmā vinītā te,_  
_sotukāmā susāsanaṃ;_  
_Chaḷaṅge pāramippattā,_  
_sindhukūle vasanti te._  

</Paragraph>

</Division>
<Division number="335" book="ap1">

<Paragraph number="4639" type="verses" book="ap1">
_Uppātagamane ceva,_  
_lakkhaṇesu ca kovidā;_  
_Uttamatthaṃ gavesantā,_  
_vasanti vipine tadā._  

</Paragraph>

</Division>
<Division number="336" book="ap1">

<Paragraph number="4640" type="verses" book="ap1">
_Sumedho nāma sambuddho,_  
_loke uppajji tāvade;_  
_Amhākaṃ anukampanto,_  
_upāgacchi vināyako._  

</Paragraph>

</Division>
<Division number="337" book="ap1">

<Paragraph number="4641" type="verses" book="ap1">
_Upāgataṃ mahāvīraṃ,_  
_sumedhaṃ lokanāyakaṃ;_  
_Tiṇasanthārakaṃ katvā,_  
_lokajeṭṭhassadāsahaṃ._  

</Paragraph>

</Division>
<Division number="338" book="ap1">

<Paragraph number="4642" type="verses" book="ap1">
_Vipināto madhuṃ gayha,_  
_buddhaseṭṭhassadāsahaṃ;_  
_Sambuddho paribhuñjitvā,_  
_idaṃ vacanamabravi._  

</Paragraph>

</Division>
<Division number="339" book="ap1">

<Paragraph number="4643" type="verses" book="ap1">
_‘Yo taṃ adāsi madhuṃ me,_  
_pasanno sehi pāṇibhi;_  
_Tamahaṃ kittayissāmi,_  
_suṇātha mama bhāsato._  

</Paragraph>

</Division>
<Division number="340" book="ap1">

<Paragraph number="4644" type="verses" book="ap1">
_Iminā madhudānena,_  
_tiṇasanthārakena ca;_  
_Tiṃsa kappasahassāni,_  
_devaloke ramissati._  

</Paragraph>

</Division>
<Division number="341" book="ap1">

<Paragraph number="4645" type="verses" book="ap1">
_Tiṃsa kappasahassamhi,_  
_okkākakulasambhavo;_  
_Gotamo nāma gottena,_  
_satthā loke bhavissati._  

</Paragraph>

</Division>
<Division number="342" book="ap1">

<Paragraph number="4646" type="verses" book="ap1">
_Tassa dhammesu dāyādo,_  
_oraso dhammanimmito;_  
_Sabbāsave pariññāya,_  
_nibbāyissatināsavo’._  

</Paragraph>

</Division>
<Division number="343" book="ap1">

<Paragraph number="4647" type="verses" book="ap1">
_Devalokā idhāgantvā,_  
_mātukucchiṃ upāgate;_  
_Madhuvassaṃ pavassittha,_  
_chādayaṃ madhunā mahiṃ._  

</Paragraph>

</Division>
<Division number="344" book="ap1">

<Paragraph number="4648" type="verses" book="ap1">
_Mayi nikkhantamattamhi,_  
_kucchiyā ca suduttarā;_  
_Tatrāpi madhuvassaṃ me,_  
_vassate niccakālikaṃ._  

</Paragraph>

</Division>
<Division number="345" book="ap1">

<Paragraph number="4649" type="verses" book="ap1">
_Agārā abhinikkhamma,_  
_pabbajiṃ anagāriyaṃ;_  
_Lābhī annassa pānassa,_  
_madhudānassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="346" book="ap1">

<Paragraph number="4650" type="verses" book="ap1">
_Sabbakāmasamiddhohaṃ,_  
_bhavitvā devamānuse;_  
_Teneva madhudānena,_  
_pattomhi āsavakkhayaṃ._  

</Paragraph>

</Division>
<Division number="347" book="ap1">

<Paragraph number="4651" type="verses" book="ap1">
_Vuṭṭhamhi deve caturaṅgule tiṇe,_  
_Sampupphite dharaṇīruhe sañchanne;_  
_Suññe ghare maṇḍaparukkhamūlake,_  
_Vasāmi niccaṃ sukhito anāsavo._  

</Paragraph>

</Division>
<Division number="348" book="ap1">

<Paragraph number="4652" type="verses" book="ap1">
_Majjhe mahante hīne ca,_  
_bhave sabbe atikkamiṃ;_  
_Ajja me āsavā khīṇā,_  
_natthi dāni punabbhavo._  

</Paragraph>

</Division>
<Division number="349" book="ap1">

<Paragraph number="4653" type="verses" book="ap1">
_Tiṃsakappasahassamhi,_  
_yaṃ dānamadadiṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_madhudānassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="350" book="ap1">

<Paragraph number="4654" type="verses" book="ap1">
_Kilesā jhāpitā mayhaṃ,_  
_bhavā sabbe samūhatā;_  
_Sabbāsavā parikkhīṇā,_  
_natthi dāni punabbhavo._  

</Paragraph>

</Division>
<Division number="351" book="ap1">

_ …pe… _

</Paragraph>

_ …pe… _

</Paragraph>

</Paragraph>
</Division>
