<Paragraph number="1638" type="verses" book="ap1">
_“Suvaṇṇavaṇṇaṃ sambuddhaṃ,_  
_Gacchantaṃ antarāpaṇe;_  
_Kañcanagghiyasaṅkāsaṃ,_  
_Bāttiṃsavaralakkhaṇaṃ._  

</Paragraph>

</Division>
<Division number="37" book="ap1" bookNo="20" page="136">

<Paragraph number="1639" type="verses" book="ap1">
_Seṭṭhiputto tadā āsiṃ,_  
_sukhumālo sukhedhito;_  
_Ucchaṅge pāṭalipupphaṃ,_  
_katvāna abhisaṃhariṃ._  

</Paragraph>

</Division>
<Division number="38" book="ap1" bookNo="20" page="136">

<Paragraph number="1640" type="verses" book="ap1">
_Haṭṭho haṭṭhena cittena,_  
_pupphehi abhipūjayiṃ;_  
_Tissaṃ lokaviduṃ nāthaṃ,_  
_naradevaṃ namassahaṃ._  

</Paragraph>

</Division>
<Division number="39" book="ap1" bookNo="20" page="136">

<Paragraph number="1641" type="verses" book="ap1">
_Dvenavute ito kappe,_  
_yaṃ kammamakariṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_pupphapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="40" book="ap1" bookNo="20" page="136">

<Paragraph number="1642" type="verses" book="ap1">
_Ito tesaṭṭhikappamhi,_  
_abhisammatanāmako;_  
_Sattaratanasampanno,_  
_cakkavattī mahabbalo._  

</Paragraph>

</Division>
<Division number="41" book="ap1" bookNo="20" page="136">

<Paragraph number="1643" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="1644" book="ap1">
Itthaṃ sudaṃ āyasmā pāṭalipupphiyo thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="1645" type="center" book="ap1">
Pāṭalipupphiyattherassāpadānaṃ aṭṭhamaṃ.
</Paragraph>
