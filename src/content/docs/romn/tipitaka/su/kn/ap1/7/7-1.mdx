<Paragraph number="1491" type="verses" book="ap1">
_“Pavanaṃ kānanaṃ disvā,_  
_appasaddamanāvilaṃ;_  
_Isīnaṃ anuciṇṇaṃva,_  
_āhutīnaṃ paṭiggahaṃ._  

</Paragraph>

</Division>
<Division number="2" book="ap1" bookNo="20" page="124">

<Paragraph number="1492" type="verses" book="ap1">
_Thūpaṃ katvāna pulinaṃ,_  
_nānāpupphaṃ samokiriṃ;_  
_Sammukhā viya sambuddhaṃ,_  
_nimmitaṃ abhivandahaṃ._  

</Paragraph>

</Division>
<Division number="3" book="ap1" bookNo="20" page="124">

<Paragraph number="1493" type="verses" book="ap1">
_Sattaratanasampanno,_  
_rājā raṭṭhamhi issaro;_  
_Sakakammābhiraddhohaṃ,_  
_pupphapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="4" book="ap1" bookNo="20" page="124">

<Paragraph number="1494" type="verses" book="ap1">
_Ekanavutito kappe,_  
_yaṃ pupphamabhiropayiṃ;_  
_Duggatiṃ nābhijānāmi,_  
_pupphapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="5" book="ap1" bookNo="20" page="124">

<Paragraph number="1495" type="verses" book="ap1">
_Asītikappenantayaso,_  
_cakkavattī ahosahaṃ;_  
_Sattaratanasampanno,_  
_catudīpamhi issaro._  

</Paragraph>

</Division>
<Division number="6" book="ap1" bookNo="20" page="124">

<Paragraph number="1496" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_vimokkhāpi ca aṭṭhime;_  
_Chaḷabhiññā sacchikatā,_  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="1497" book="ap1">
Itthaṃ sudaṃ āyasmā sakacintaniyo thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="1498" type="center" book="ap1">
Sakacintaniyattherassāpadānaṃ paṭhamaṃ.
</Paragraph>
