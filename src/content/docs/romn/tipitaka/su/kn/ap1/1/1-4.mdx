<Division number="375" book="ap1">

<Paragraph number="384" type="verses" book="ap1">
_“Anomadassī bhagavā,_  
_lokajeṭṭho narāsabho;_  
_Vihāsi himavantamhi,_  
_devasaṅghapurakkhato._  

</Paragraph>

</Division>
<Division number="376" book="ap1">

<Paragraph number="385" type="verses" book="ap1">
_Varuṇo nāma nāmena,_  
_nāgarājā ahaṃ tadā;_  
_Kāmarūpī vikubbāmi,_  
_mahodadhinivāsahaṃ._  

</Paragraph>

</Division>
<Division number="377" book="ap1">

<Paragraph number="386" type="verses" book="ap1">
_Saṅgaṇiyaṃ gaṇaṃ hitvā,_  
_tūriyaṃ paṭṭhapesahaṃ;_  
_Sambuddhaṃ parivāretvā,_  
_vādesuṃ accharā tadā._  

</Paragraph>

</Division>
<Division number="378" book="ap1">

<Paragraph number="387" type="verses" book="ap1">
_Vajjamānesu tūresu,_  
_devā tūrāni vajjayuṃ;_  
_Ubhinnaṃ saddaṃ sutvāna,_  
_buddhopi sampabujjhatha._  

</Paragraph>

</Division>
<Division number="379" book="ap1">

<Paragraph number="388" type="verses" book="ap1">
_Nimantetvāna sambuddhaṃ,_  
_sakaṃ bhavanupāgamiṃ;_  
_Āsanaṃ paññapetvāna,_  
_kālamārocayiṃ ahaṃ._  

</Paragraph>

</Division>
<Division number="380" book="ap1">

<Paragraph number="389" type="verses" book="ap1">
_Khīṇāsavasahassehi,_  
_parivuto lokanāyako;_  
_Obhāsento disā sabbā,_  
_bhavanaṃ me upāgami._  

</Paragraph>

</Division>
<Division number="381" book="ap1">

<Paragraph number="390" type="verses" book="ap1">
_Upaviṭṭhaṃ mahāvīraṃ,_  
_Devadevaṃ narāsabhaṃ;_  
_Sabhikkhusaṃghaṃ tappesiṃ,_  
_Annapānenahaṃ tadā._  

</Paragraph>

</Division>
<Division number="382" book="ap1">

<Paragraph number="391" type="verses" book="ap1">
_Anumodi mahāvīro,_  
_sayambhū aggapuggalo;_  
_Bhikkhusaṃghe nisīditvā,_  
_imā gāthā abhāsatha._  

</Paragraph>

</Division>
<Division number="383" book="ap1">

<Paragraph number="392" type="verses" book="ap1">
_Yo so saṃghaṃ apūjesi,_  
_buddhañca lokanāyakaṃ;_  
_Tena cittappasādena,_  
_devalokaṃ gamissati._  

</Paragraph>

</Division>
<Division number="384" book="ap1">

<Paragraph number="393" type="verses" book="ap1">
_Sattasattatikkhattuñca,_  
_devarajjaṃ karissati;_  
_Pathabyā rajjaṃ aṭṭhasataṃ,_  
_vasudhaṃ āvasissati._  

</Paragraph>

</Division>
<Division number="385" book="ap1">

<Paragraph number="394" type="verses" book="ap1">
_Pañcapaññāsakkhattuñca,_  
_cakkavattī bhavissati;_  
_Bhogā asaṅkhiyā tassa,_  
_uppajjissanti tāvade._  

</Paragraph>

</Division>
<Division number="386" book="ap1">

<Paragraph number="395" type="verses" book="ap1">
_Aparimeyye ito kappe,_  
_okkākakulasambhavo;_  
_Gotamo nāma gottena,_  
_satthā loke bhavissati._  

</Paragraph>

</Division>
<Division number="387" book="ap1">

<Paragraph number="396" type="verses" book="ap1">
_Nirayā so cavitvāna,_  
_manussataṃ gamissati;_  
_Kolito nāma nāmena,_  
_brahmabandhu bhavissati._  

</Paragraph>

</Division>
<Division number="388" book="ap1">

<Paragraph number="397" type="verses" book="ap1">
_So pacchā pabbajitvāna,_  
_kusalamūlena codito;_  
_Gotamassa bhagavato,_  
_dutiyo hessati sāvako._  

</Paragraph>

</Division>
<Division number="389" book="ap1">

<Paragraph number="398" type="verses" book="ap1">
_Āraddhavīriyo pahitatto,_  
_Iddhiyā pāramiṃ gato;_  
_Sabbāsave pariññāya,_  
_Nibbāyissatināsavo._  

</Paragraph>

</Division>
<Division number="390" book="ap1">

<Paragraph number="399" type="verses" book="ap1">
_Pāpamittopanissāya,_  
_kāmarāgavasaṃ gato;_  
_Mātaraṃ pitarañcāpi,_  
_ghātayiṃ duṭṭhamānaso._  

</Paragraph>

</Division>
<Division number="391" book="ap1">

<Paragraph number="400" type="verses" book="ap1">
_Yaṃ yaṃ yonupapajjāmi,_  
_nirayaṃ atha mānusaṃ;_  
_Pāpakammasamaṅgitā,_  
_bhinnasīso marāmahaṃ._  

</Paragraph>

</Division>
<Division number="392" book="ap1">

<Paragraph number="401" type="verses" book="ap1">
_Idaṃ pacchimakaṃ mayhaṃ,_  
_carimo vattate bhavo;_  
_Idhāpi ediso mayhaṃ,_  
_maraṇakāle bhavissati._  

</Paragraph>

</Division>
<Division number="393" book="ap1">

<Paragraph number="402" type="verses" book="ap1">
_Pavivekamanuyutto,_  
_samādhibhāvanārato;_  
_Sabbāsave pariññāya,_  
_viharāmi anāsavo._  

</Paragraph>

</Division>
<Division number="394" book="ap1">

<Paragraph number="403" type="verses" book="ap1">
_Dharaṇimpi sugambhīraṃ,_  
_bahalaṃ duppadhaṃsiyaṃ;_  
_Vāmaṅguṭṭhena khobheyyaṃ,_  
_iddhiyā pāramiṃ gato._  

</Paragraph>

</Division>
<Division number="395" book="ap1">

<Paragraph number="404" type="verses" book="ap1">
_Asmimānaṃ na passāmi,_  
_māno mayhaṃ na vijjati;_  
_Sāmaṇere upādāya,_  
_garucittaṃ karomahaṃ._  

</Paragraph>

</Division>
<Division number="396" book="ap1">

<Paragraph number="405" type="verses" book="ap1">
_Aparimeyye ito kappe,_  
_yaṃ kammamabhinīhariṃ;_  
_Tāhaṃ bhūmimanuppatto,_  
_pattomhi āsavakkhayaṃ._  

</Paragraph>

</Division>
<Division number="397" book="ap1">

<Paragraph number="406" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_vimokkhāpi ca aṭṭhime;_  
_Chaḷabhiññā sacchikatā,_  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="407" book="ap1">
Itthaṃ sudaṃ āyasmā mahāmoggallāno thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="408" type="center" book="ap1">
Mahāmoggallānattherassāpadānaṃ dutiyaṃ.
</Paragraph>
