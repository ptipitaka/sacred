<Paragraph number="5678" type="verses" book="ap1">
_“Vipassino bhagavato,_  
_mahābodhimaho ahu;_  
_Mahājanā samāgamma,_  
_pūjenti bodhimuttamaṃ._  

</Paragraph>

</Division>
<Division number="14" book="ap1" bookNo="20" page="430">

<Paragraph number="5679" type="verses" book="ap1">
_Na hi taṃ orakaṃ maññe,_  
_buddhaseṭṭho bhavissati;_  
_Yassāyaṃ īdisā bodhi,_  
_pūjanīyā ca satthuno._  

</Paragraph>

</Division>
<Division number="15" book="ap1" bookNo="20" page="430">

<Paragraph number="5680" type="verses" book="ap1">
_Tato saṅkhaṃ gahetvāna,_  
_bodhirukkhamupaṭṭhahiṃ;_  
_Dhamanto sabbadivasaṃ,_  
_avandiṃ bodhimuttamaṃ._  

</Paragraph>

</Division>
<Division number="16" book="ap1" bookNo="20" page="430">

<Paragraph number="5681" type="verses" book="ap1">
_Āsannake kataṃ kammaṃ,_  
_devalokaṃ apāpayī;_  
_Kaḷevaraṃ me patitaṃ,_  
_devaloke ramāmahaṃ._  

</Paragraph>

</Division>
<Division number="17" book="ap1" bookNo="20" page="430">

<Paragraph number="5682" type="verses" book="ap1">
_Saṭṭhituriyasahassāni,_  
_tuṭṭhahaṭṭhā pamoditā;_  
_Sadā mayhaṃ upaṭṭhanti,_  
_buddhapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="18" book="ap1" bookNo="20" page="430">

<Paragraph number="5683" type="verses" book="ap1">
_Ekasattatime kappe,_  
_rājā āsiṃ sudassano;_  
_Cāturanto vijitāvī,_  
_jambumaṇḍassa issaro._  

</Paragraph>

</Division>
<Division number="19" book="ap1" bookNo="20" page="430">

<Paragraph number="5684" type="verses" book="ap1">
_Tato aṅgasatā turiyā,_  
_parivārenti maṃ sadā;_  
_Anubhomi sakaṃ kammaṃ,_  
_upaṭṭhānassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="20" book="ap1" bookNo="20" page="430">

<Paragraph number="5685" type="verses" book="ap1">
_Yaṃ yaṃ yonupapajjāmi,_  
_devattaṃ atha mānusaṃ;_  
_Mātukucchigatassāpi,_  
_vajjare bheriyo sadā._  

</Paragraph>

</Division>
<Division number="21" book="ap1" bookNo="20" page="430">

<Paragraph number="5686" type="verses" book="ap1">
_Upaṭṭhitvāna sambuddhaṃ,_  
_anubhutvāna sampadā;_  
_Sivaṃ sukhemaṃ amataṃ,_  
_pattomhi acalaṃ padaṃ._  

</Paragraph>

</Division>
<Division number="22" book="ap1" bookNo="20" page="430">

<Paragraph number="5687" type="verses" book="ap1">
_Ekanavutito kappe,_  
_yaṃ kammamakariṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_buddhapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="23" book="ap1" bookNo="20" page="430">

<Paragraph number="5688" type="verses" book="ap1">
_Kilesā jhāpitā mayhaṃ,_  
_ …pe… _  
_viharāmi anāsavo._  

</Paragraph>

</Division>
<Division number="24" book="ap1" bookNo="20" page="431">

<Paragraph number="5689" type="verses" book="ap1">
_Svāgataṃ vata me āsi,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ._  

</Paragraph>

</Division>
<Division number="25" book="ap1" bookNo="20" page="431">

<Paragraph number="5690" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="5691" book="ap1">
Itthaṃ sudaṃ āyasmā ekasaṅkhiyo thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="5692" type="center" book="ap1">
Ekasaṅkhiyattherassāpadānaṃ dutiyaṃ.
