<Paragraph number="6468" type="verses" book="ap1">
_“Padumuttarabhagavato,_  
_sujāto nāma sāvako;_  
_Paṃsukūlaṃ gavesanto,_  
_saṅkāre carate tadā._  

</Paragraph>

</Division>
<Division number="32" book="ap1" bookNo="20" page="431">

<Paragraph number="6469" type="verses" book="ap1">
_Nagare haṃsavatiyā,_  
_paresaṃ bhatako ahaṃ;_  
_Upaḍḍhadussaṃ datvāna,_  
_sirasā abhivādayiṃ._  

</Paragraph>

</Division>
<Division number="33" book="ap1" bookNo="20" page="431">

<Paragraph number="6470" type="verses" book="ap1">
_Tena kammena sukatena,_  
_cetanāpaṇidhīhi ca;_  
_Jahitvā mānusaṃ dehaṃ,_  
_tāvatiṃsamagacchahaṃ._  

</Paragraph>

</Division>
<Division number="34" book="ap1" bookNo="20" page="432">

<Paragraph number="6471" type="verses" book="ap1">
_Tettiṃsakkhattuṃ devindo,_  
_devarajjamakārayiṃ;_  
_Sattasattatikkhattuñca,_  
_cakkavattī ahosahaṃ._  

</Paragraph>

</Division>
<Division number="35" book="ap1" bookNo="20" page="432">

<Paragraph number="6472" type="verses" book="ap1">
_Padesarajjaṃ vipulaṃ,_  
_gaṇanāto asaṅkhiyaṃ;_  
_Upaḍḍhadussadānena,_  
_modāmi akutobhayo._  

</Paragraph>

</Division>
<Division number="36" book="ap1" bookNo="20" page="432">

<Paragraph number="6473" type="verses" book="ap1">
_Icchamāno cahaṃ ajja,_  
_sakānanaṃ sapabbataṃ;_  
_Khomadussehi chādeyyaṃ,_  
_aḍḍhadussassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="37" book="ap1" bookNo="20" page="432">

<Paragraph number="6474" type="verses" book="ap1">
_Satasahassito kappe,_  
_yaṃ dānamadadiṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_aḍḍhadussassidaṃ phalaṃ._  

</Paragraph>

</Division>

_ …pe… _

</Paragraph>

</Division>

_ …pe… _  
_kataṃ buddhassa sāsanaṃ._  

</Paragraph>

</Division>

_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="6478" book="ap1">

</Paragraph>
