<Paragraph number="1796" type="verses" book="ap1">
_“Siddhatthassa bhagavato,_  
_gandhathūpaṃ akāsahaṃ;_  
_Sumanehi paṭicchannaṃ,_  
_buddhānucchavikaṃ kataṃ._  

</Paragraph>

</Division>
<Division number="25" book="ap1" bookNo="20" page="149">

<Paragraph number="1797" type="verses" book="ap1">
_Kañcanagghiyasaṅkāsaṃ,_  
_buddhaṃ lokagganāyakaṃ;_  
_Indīvaraṃva jalitaṃ,_  
_ādittaṃva hutāsanaṃ._  

</Paragraph>

</Division>
<Division number="26" book="ap1" bookNo="20" page="149">

<Paragraph number="1798" type="verses" book="ap1">
_Byagghūsabhaṃva pavaraṃ,_  
_abhijātaṃva kesariṃ;_  
_Nisinnaṃ samaṇānaggaṃ,_  
_bhikkhusaṃghapurakkhataṃ._  

</Paragraph>

</Division>
<Division number="27" book="ap1" bookNo="20" page="149">

<Paragraph number="1799" type="verses" book="ap1">
_Vanditvā satthuno pāde,_  
_pakkāmiṃ uttarāmukho;_  
_Catunnavutito kappe,_  
_gandhamālaṃ yato adaṃ._  

</Paragraph>

</Division>
<Division number="28" book="ap1" bookNo="20" page="149">

<Paragraph number="1800" type="verses" book="ap1">
_Buddhe katassa kārassa,_  
_phalenāhaṃ visesato;_  
_Duggatiṃ nābhijānāmi,_  
_buddhapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="29" book="ap1" bookNo="20" page="149">

<Paragraph number="1801" type="verses" book="ap1">
_Cattārīsamhi ekūne,_  
_kappe āsiṃsu soḷasa;_  
_Devagandhasanāmā te,_  
_rājāno cakkavattino._  

</Paragraph>

</Division>
<Division number="30" book="ap1" bookNo="20" page="150">

<Paragraph number="1802" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="1803" book="ap1">
Itthaṃ sudaṃ āyasmā gandhamāliyo thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>
