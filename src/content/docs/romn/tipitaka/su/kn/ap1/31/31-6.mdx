<Division number="24" book="ap1">

<Paragraph number="3465" type="verses" book="ap1">
_“‘Namo te buddha vīratthu,_  
_vippamuttosi sabbadhi;_  
_Byasanamhi anuppatto,_  
_tassa me saraṇaṃ bhava’._  

</Paragraph>

</Division>
<Division number="25" book="ap1">

<Paragraph number="3466" type="verses" book="ap1">
_Siddhattho tassa byākāsi,_  
_loke appaṭipuggalo;_  
_‘Mahodadhisamo saṃgho,_  
_appameyyo anuttaro._  

</Paragraph>

</Division>
<Division number="26" book="ap1">

<Paragraph number="3467" type="verses" book="ap1">
_Tattha tvaṃ viraje khette,_  
_anantaphaladāyake;_  
_Saṃghe cittaṃ pasādetvā,_  
_subījaṃ vāpa ropaya’._  

</Paragraph>

</Division>
<Division number="27" book="ap1">

<Paragraph number="3468" type="verses" book="ap1">
_Idaṃ vatvāna sabbaññū,_  
_lokajeṭṭho narāsabho;_  
_Mameva anusāsitvā,_  
_vehāsaṃ nabhamuggami._  

</Paragraph>

</Division>
<Division number="28" book="ap1">

<Paragraph number="3469" type="verses" book="ap1">
_Aciraṃ gatamattamhi,_  
_sabbaññumhi narāsabhe;_  
_Maraṇaṃ samanuppatto,_  
_tusitaṃ upapajjahaṃ._  

</Paragraph>

</Division>
<Division number="29" book="ap1">

<Paragraph number="3470" type="verses" book="ap1">
_Tadāhaṃ viraje khette,_  
_anantaphaladāyake;_  
_Saṃghe cittaṃ pasādetvā,_  
_kappaṃ saggamhi modahaṃ._  

</Paragraph>

</Division>
<Division number="30" book="ap1">

<Paragraph number="3471" type="verses" book="ap1">
_Catunnavutito kappe,_  
_pasādamalabhiṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_pasādassa idaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="31" book="ap1">

<Paragraph number="3472" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="3473" book="ap1">
Itthaṃ sudaṃ āyasmā sampasādako thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="3474" type="center" book="ap1">
Sampasādakattherassāpadānaṃ chaṭṭhaṃ.
</Paragraph>
