<Division number="1" book="ap1">

<Paragraph number="5470" type="verses" book="ap1">
_“Vipassino bhagavato,_  
_pāṭaliṃ bodhimuttamaṃ;_  
_Disvāva taṃ pādapaggaṃ,_  
_tattha cittaṃ pasādayiṃ._  

</Paragraph>

</Division>
<Division number="2" book="ap1">

<Paragraph number="5471" type="verses" book="ap1">
_Sammajjaniṃ gahetvāna,_  
_bodhiṃ sammajji tāvade;_  
_Sammajjitvāna taṃ bodhiṃ,_  
_avandiṃ pāṭaliṃ ahaṃ._  

</Paragraph>

</Division>
<Division number="3" book="ap1">

<Paragraph number="5472" type="verses" book="ap1">
_Tattha cittaṃ pasādetvā,_  
_sire katvāna añjaliṃ;_  
_Namassamāno taṃ bodhiṃ,_  
_gañchiṃ paṭikuṭiṃ ahaṃ._  

</Paragraph>

</Division>
<Division number="4" book="ap1">

<Paragraph number="5473" type="verses" book="ap1">
_Tādimaggena gacchāmi,_  
_saranto bodhimuttamaṃ;_  
_Ajagaro maṃ pīḷesi,_  
_ghorarūpo mahabbalo._  

</Paragraph>

</Division>
<Division number="5" book="ap1">

<Paragraph number="5474" type="verses" book="ap1">
_Āsanne me kataṃ kammaṃ,_  
_phalena tosayī mamaṃ;_  
_Kaḷevaraṃ me gilati,_  
_devaloke ramāmahaṃ._  

</Paragraph>

</Division>
<Division number="6" book="ap1">

<Paragraph number="5475" type="verses" book="ap1">
_Anāvilaṃ mama cittaṃ,_  
_visuddhaṃ paṇḍaraṃ sadā;_  
_Sokasallaṃ na jānāmi,_  
_cittasantāpanaṃ mama._  

</Paragraph>

</Division>
<Division number="7" book="ap1">

<Paragraph number="5476" type="verses" book="ap1">
_Kuṭṭhaṃ gaṇḍo kilāso ca,_  
_apamāro vitacchikā;_  
_Daddu kaṇḍu ca me natthi,_  
_phalaṃ sammajjanāyidaṃ._  

</Paragraph>

</Division>
<Division number="8" book="ap1">

<Paragraph number="5477" type="verses" book="ap1">
_Soko ca paridevo ca,_  
_hadaye me na vijjati;_  
_Abhantaṃ ujukaṃ cittaṃ,_  
_phalaṃ sammajjanāyidaṃ._  

</Paragraph>

</Division>
<Division number="9" book="ap1">

<Paragraph number="5478" type="verses" book="ap1">
_Samādhīsu na majjāmi,_  
_visadaṃ hoti mānasaṃ;_  
_Yaṃ yaṃ samādhimicchāmi,_  
_so so sampajjate mamaṃ._  

</Paragraph>

</Division>
<Division number="10" book="ap1">

<Paragraph number="5479" type="verses" book="ap1">
_Rajanīye na rajjāmi,_  
_atho dussaniyesu ca;_  
_Mohanīye na muyhāmi,_  
_phalaṃ sammajjanāyidaṃ._  

</Paragraph>

</Division>
<Division number="11" book="ap1">

<Paragraph number="5480" type="verses" book="ap1">
_Ekanavutito kappe,_  
_yaṃ kammamakariṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_phalaṃ sammajjanāyidaṃ._  

</Paragraph>

</Division>
<Division number="12" book="ap1">

<Paragraph number="5481" type="verses" book="ap1">
_Kilesā jhāpitā mayhaṃ,_  
_bhavā sabbe samūhatā;_  
_Nāgova bandhanaṃ chetvā,_  
_viharāmi anāsavo._  

</Paragraph>

</Division>
<Division number="13" book="ap1">

<Paragraph number="5482" type="verses" book="ap1">
_Svāgataṃ vata me āsi,_  
_mama buddhassa santike;_  
_Tisso vijjā anuppattā,_  
_kataṃ buddhassa sāsanaṃ._  

</Paragraph>

</Division>
<Division number="14" book="ap1">

<Paragraph number="5483" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_vimokkhāpi ca aṭṭhime;_  
_Chaḷabhiññā sacchikatā,_  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="5484" book="ap1">
Itthaṃ sudaṃ āyasmā sakiṃsammajjako thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="5485" type="center" book="ap1">
Sakiṃsammajjakattherassāpadānaṃ paṭhamaṃ.
</Paragraph>
