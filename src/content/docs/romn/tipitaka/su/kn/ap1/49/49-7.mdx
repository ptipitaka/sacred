<Division number="111" book="ap1">

<Paragraph number="6308" type="verses" book="ap1">
_“Pabbate himavantamhi,_  
_devalo nāma tāpaso;_  
_Tattha me caṅkamo āsi,_  
_amanussehi māpito._  

</Paragraph>

</Division>
<Division number="112" book="ap1">

<Paragraph number="6309" type="verses" book="ap1">
_Jaṭābhārena bharito,_  
_kamaṇḍaludharo sadā;_  
_Uttamatthaṃ gavesanto,_  
_vipinā nikkhamiṃ tadā._  

</Paragraph>

</Division>
<Division number="113" book="ap1">

<Paragraph number="6310" type="verses" book="ap1">
_Cullāsītisahassāni,_  
_sissā mayhaṃ upaṭṭhahuṃ;_  
_Sakakammābhipasutā,_  
_vasanti vipine tadā._  

</Paragraph>

</Division>
<Division number="114" book="ap1">

<Paragraph number="6311" type="verses" book="ap1">
_Assamā abhinikkhamma,_  
_akaṃ pulinacetiyaṃ;_  
_Nānāpupphaṃ samānetvā,_  
_taṃ cetiyamapūjayiṃ._  

</Paragraph>

</Division>
<Division number="115" book="ap1">

<Paragraph number="6312" type="verses" book="ap1">
_Tattha cittaṃ pasādetvā,_  
_assamaṃ pavisāmahaṃ;_  
_Sabbe sissā samāgantvā,_  
_etamatthaṃ pucchiṃsu maṃ._  

</Paragraph>

</Division>
<Division number="116" book="ap1">

<Paragraph number="6313" type="verses" book="ap1">
_‘Pulinena kato thūpo,_  
_yaṃ tvaṃ deva namassasi;_  
_Mayampi ñātumicchāma,_  
_puṭṭho ācikkha no tuvaṃ’._  

</Paragraph>

</Division>
<Division number="117" book="ap1">

<Paragraph number="6314" type="verses" book="ap1">
_‘Niddiṭṭhā nu mantapade,_  
_cakkhumanto mahāyasā;_  
_Te kho ahaṃ namassāmi,_  
_buddhaseṭṭhe mahāyase._  

</Paragraph>

</Division>
<Division number="118" book="ap1">

<Paragraph number="6315" type="verses" book="ap1">
_Kīdisā te mahāvīrā,_  
_sabbaññū lokanāyakā;_  
_Kathaṃvaṇṇā kathaṃsīlā,_  
_kīdisā te mahāyasā._  

</Paragraph>

</Division>
<Division number="119" book="ap1">

<Paragraph number="6316" type="verses" book="ap1">
_Bāttiṃsalakkhaṇā buddhā,_  
_cattālīsadijāpi ca;_  
_Nettā gopakhumā tesaṃ,_  
_jiñjukā phalasannibhā._  

</Paragraph>

</Division>
<Division number="120" book="ap1">

<Paragraph number="6317" type="verses" book="ap1">
_Gacchamānā ca te buddhā,_  
_yugamattañca pekkhare;_  
_Na tesaṃ jāṇu nadati,_  
_sandhisaddo na suyyati._  

</Paragraph>

</Division>
<Division number="121" book="ap1">

<Paragraph number="6318" type="verses" book="ap1">
_Gacchamānā ca sugatā,_  
_uddharantāva gacchare;_  
_Paṭhamaṃ dakkhiṇaṃ pādaṃ,_  
_buddhānaṃ esa dhammatā._  

</Paragraph>

</Division>
<Division number="122" book="ap1">

<Paragraph number="6319" type="verses" book="ap1">
_Asambhītā ca te buddhā,_  
_migarājāva kesarī;_  
_Nevukkaṃsenti attānaṃ,_  
_no ca vambhenti pāṇinaṃ._  

</Paragraph>

</Division>
<Division number="123" book="ap1">

<Paragraph number="6320" type="verses" book="ap1">
_Mānāvamānato muttā,_  
_samā sabbesu pāṇisu;_  
_Anattukkaṃsakā buddhā,_  
_buddhānaṃ esa dhammatā._  

</Paragraph>

</Division>
<Division number="124" book="ap1">

<Paragraph number="6321" type="verses" book="ap1">
_Uppajjantā ca sambuddhā,_  
_ālokaṃ dassayanti te;_  
_Chappakāraṃ pakampenti,_  
_kevalaṃ vasudhaṃ imaṃ._  

</Paragraph>

</Division>
<Division number="125" book="ap1">

<Paragraph number="6322" type="verses" book="ap1">
_Passanti nirayañcete,_  
_nibbāti nirayo tadā;_  
_Pavassati mahāmegho,_  
_buddhānaṃ esa dhammatā._  

</Paragraph>

</Division>
<Division number="126" book="ap1">

<Paragraph number="6323" type="verses" book="ap1">
_Īdisā te mahānāgā,_  
_atulā ca mahāyasā;_  
_Vaṇṇato anatikkantā,_  
_appameyyā tathāgatā’._  

</Paragraph>

</Division>
<Division number="127" book="ap1">

<Paragraph number="6324" type="verses" book="ap1">
_Anumodiṃsu me vākyaṃ,_  
_sabbe sissā sagāravā;_  
_Tathā ca paṭipajjiṃsu,_  
_yathāsatti yathābalaṃ._  

</Paragraph>

</Division>
<Division number="128" book="ap1">

<Paragraph number="6325" type="verses" book="ap1">
_Paṭipūjenti pulinaṃ,_  
_sakakammābhilāsino;_  
_Saddahantā mama vākyaṃ,_  
_buddhasakkatamānasā._  

</Paragraph>

</Division>
<Division number="129" book="ap1">

<Paragraph number="6326" type="verses" book="ap1">
_Tadā cavitvā tusitā,_  
_devaputto mahāyaso;_  
_Uppajji mātukucchimhi,_  
_dasasahassi kampatha._  

</Paragraph>

</Division>
<Division number="130" book="ap1">

<Paragraph number="6327" type="verses" book="ap1">
_Assamassāvidūramhi,_  
_caṅkamamhi ṭhito ahaṃ;_  
_Sabbe sissā samāgantvā,_  
_āgacchuṃ mama santike._  

</Paragraph>

</Division>
<Division number="131" book="ap1">

<Paragraph number="6328" type="verses" book="ap1">
_‘Usabhova mahī nadati,_  
_migarājāva kūjati;_  
_Susumārova saḷati,_  
_kiṃ vipāko bhavissati’._  

</Paragraph>

</Division>
<Division number="132" book="ap1">

<Paragraph number="6329" type="verses" book="ap1">
_‘Yaṃ pakittemi sambuddhaṃ,_  
_sikatāthūpasantike;_  
_So dāni bhagavā satthā,_  
_mātukucchimupāgami’._  

</Paragraph>

</Division>
<Division number="133" book="ap1">

<Paragraph number="6330" type="verses" book="ap1">
_Tesaṃ dhammakathaṃ vatvā,_  
_kittayitvā mahāmuniṃ;_  
_Uyyojetvā sake sisse,_  
_pallaṅkamābhujiṃ ahaṃ._  

</Paragraph>

</Division>
<Division number="134" book="ap1">

<Paragraph number="6331" type="verses" book="ap1">
_Balañca vata me khīṇaṃ,_  
_byādhinā paramena taṃ;_  
_Buddhaseṭṭhaṃ saritvāna,_  
_tattha kālaṅkato ahaṃ._  

</Paragraph>

</Division>
<Division number="135" book="ap1">

<Paragraph number="6332" type="verses" book="ap1">
_Sabbe sissā samāgantvā,_  
_akaṃsu citakaṃ tadā;_  
_Kaḷevarañca me gayha,_  
_citakaṃ abhiropayuṃ._  

</Paragraph>

</Division>
<Division number="136" book="ap1">

<Paragraph number="6333" type="verses" book="ap1">
_Citakaṃ parivāretvā,_  
_sīse katvāna añjaliṃ;_  
_Sokasallaparetā te,_  
_vikkandiṃsu samāgatā._  

</Paragraph>

</Division>
<Division number="137" book="ap1">

<Paragraph number="6334" type="verses" book="ap1">
_Tesaṃ lālappamānānaṃ,_  
_agamaṃ citakaṃ tadā;_  
_‘Ahaṃ ācariyo tumhaṃ,_  
_mā socittha sumedhasā._  

</Paragraph>

</Division>
<Division number="138" book="ap1">

<Paragraph number="6335" type="verses" book="ap1">
_Sadatthe vāyameyyātha,_  
_rattindivamatanditā;_  
_Mā vo pamattā ahuttha,_  
_khaṇo vo paṭipādito’._  

</Paragraph>

</Division>
<Division number="139" book="ap1">

<Paragraph number="6336" type="verses" book="ap1">
_Sake sissenusāsitvā,_  
_devalokaṃ punāgamiṃ;_  
_Aṭṭhārasa ca kappāni,_  
_devaloke ramāmahaṃ._  

</Paragraph>

</Division>
<Division number="140" book="ap1">

<Paragraph number="6337" type="verses" book="ap1">
_Satānaṃ pañcakkhattuñca,_  
_cakkavattī ahosahaṃ;_  
_Anekasatakkhattuñca,_  
_devarajjamakārayiṃ._  

</Paragraph>

</Division>
<Division number="141" book="ap1">

<Paragraph number="6338" type="verses" book="ap1">
_Avasesesu kappesu,_  
_vokiṇṇo saṃsariṃ ahaṃ;_  
_Duggatiṃ nābhijānāmi,_  
_uppādassa idaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="142" book="ap1">

<Paragraph number="6339" type="verses" book="ap1">
_Yathā komudike māse,_  
_bahū pupphanti pādapā;_  
_Tathevāhampi samaye,_  
_pupphitomhi mahesinā._  

</Paragraph>

</Division>
<Division number="143" book="ap1">

<Paragraph number="6340" type="verses" book="ap1">
_Vīriyaṃ me dhuradhorayhaṃ,_  
_yogakkhemādhivāhanaṃ;_  
_Nāgova bandhanaṃ chetvā,_  
_viharāmi anāsavo._  

</Paragraph>

</Division>
<Division number="144" book="ap1">

<Paragraph number="6341" type="verses" book="ap1">
_Satasahassito kappe,_  
_yaṃ buddhamabhikittayiṃ;_  
_Duggatiṃ nābhijānāmi,_  
_kittanāya idaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="145" book="ap1">

<Paragraph number="6342" type="verses" book="ap1">
_Kilesā jhāpitā mayhaṃ,_  
_ …pe… _  
_viharāmi anāsavo._  

</Paragraph>

</Division>
<Division number="146" book="ap1">

<Paragraph number="6343" type="verses" book="ap1">
_Svāgataṃ vata me āsi,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ._  

</Paragraph>

</Division>
<Division number="147" book="ap1">

<Paragraph number="6344" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="6345" book="ap1">
Itthaṃ sudaṃ āyasmā pulinuppādako thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>
