<Division number="180" book="ap1">

<Paragraph number="4477" type="verses" book="ap1">
_Bibbohane jine datvā,_  
_saṃghe gaṇavaruttame;_  
_Chānisaṃse anubhomi,_  
_kammānucchavike mama._  

</Paragraph>

</Division>
<Division number="181" book="ap1">

<Paragraph number="4478" type="verses" book="ap1">
_Uṇṇike padumake ca,_  
_atho lohitacandane;_  
_Bibbohane upādhemi,_  
_uttamaṅgaṃ sadā mama._  

</Paragraph>

</Division>
<Division number="182" book="ap1">

<Paragraph number="4479" type="verses" book="ap1">
_Aṭṭhaṅgike maggavare,_  
_sāmaññe caturo phale;_  
_Tesu ñāṇaṃ uppādetvā,_  
_vihare niccakālikaṃ._  

</Paragraph>

</Division>
<Division number="183" book="ap1">

<Paragraph number="4480" type="verses" book="ap1">
_Dāne dame saṃyame ca,_  
_appamaññāsu rūpisu;_  
_Tesu ñāṇaṃ uppādetvā,_  
_vihare sabbakālikaṃ._  

</Paragraph>

</Division>
<Division number="184" book="ap1">

<Paragraph number="4481" type="verses" book="ap1">
_Vatte guṇe paṭipatti,_  
_ācārakiriyāsu ca;_  
_Tesu ñāṇaṃ uppādetvā,_  
_vihare sabbadā ahaṃ._  

</Paragraph>

</Division>
<Division number="185" book="ap1">

<Paragraph number="4482" type="verses" book="ap1">
_Caṅkame vā padhāne vā,_  
_vīriye bodhipakkhiye;_  
_Tesu ñāṇaṃ uppādetvā,_  
_viharāmi yadicchakaṃ._  

</Paragraph>

</Division>
<Division number="186" book="ap1">

<Paragraph number="4483" type="verses" book="ap1">
_Sīlaṃ samādhi paññā ca,_  
_vimutti ca anuttarā;_  
_Tesu ñāṇaṃ uppādetvā,_  
_viharāmi sukhaṃ ahaṃ._
</Paragraph>
</Division>
