<Division number="78" book="ap1">

<Paragraph number="1294" type="verses" book="ap1">
_Pañca uppalahatthāni,_  
_āveḷatthaṃ ahaṃsu me;_  
_Tena buddhaṃ apūjesiṃ,_  
_pasanno sehi pāṇibhi._  

</Paragraph>

</Division>
<Division number="79" book="ap1">

<Paragraph number="1295" type="verses" book="ap1">
_Āropitā ca te pupphā,_  
_chadanaṃ assu satthuno;_  
_Samādhiṃsu mahānāgaṃ,_  
_sissā ācariyaṃ yathā._  

</Paragraph>

</Division>
<Division number="80" book="ap1">

<Paragraph number="1296" type="verses" book="ap1">
_Tiṃsakappasahassamhi,_  
_yaṃ pupphamabhiropayiṃ;_  
_Duggatiṃ nābhijānāmi,_  
_buddhapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="81" book="ap1">

<Paragraph number="1297" type="verses" book="ap1">

_ahesuṃ pañca khattiyā;_  
_Hatthiyā nāma nāmena,_  
_cakkavattī mahabbalā._  

</Paragraph>

</Division>
