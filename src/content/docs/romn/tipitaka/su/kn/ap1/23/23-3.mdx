<Paragraph number="2914" type="verses" book="ap1">
_“Migaluddo pure āsiṃ,_  
_araññe kānane ahaṃ;_  
_Addasaṃ virajaṃ buddhaṃ,_  
_āhutīnaṃ paṭiggahaṃ._  

</Paragraph>

</Division>
<Division number="13" book="ap1" bookNo="20" page="240">

<Paragraph number="2915" type="verses" book="ap1">
_Maṃsapesi mayā dinnā,_  
_vipassissa mahesino;_  
_Sadevakasmiṃ lokasmiṃ,_  
_issaraṃ kārayāmahaṃ._  

</Paragraph>

</Division>
<Division number="14" book="ap1" bookNo="20" page="240">

<Paragraph number="2916" type="verses" book="ap1">
_Iminā maṃsadānena,_  
_ratanaṃ nibbattate mama;_  
_Duveme ratanā loke,_  
_diṭṭhadhammassa pattiyā._  

</Paragraph>

</Division>
<Division number="15" book="ap1" bookNo="20" page="240">

<Paragraph number="2917" type="verses" book="ap1">
_Tehaṃ sabbe anubhomi,_  
_maṃsadānassa sattiyā;_  
_Gattañca mudukaṃ mayhaṃ,_  
_paññā nipuṇavedanī._  

</Paragraph>

</Division>
<Division number="16" book="ap1" bookNo="20" page="240">

<Paragraph number="2918" type="verses" book="ap1">
_Ekanavutito kappe,_  
_yaṃ maṃsamadadiṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_maṃsadānassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="17" book="ap1" bookNo="20" page="240">

<Paragraph number="2919" type="verses" book="ap1">
_Ito catutthake kappe,_  
_eko āsiṃ janādhipo;_  
_Mahārohitanāmo so,_  
_cakkavattī mahabbalo._  

</Paragraph>

</Division>
<Division number="18" book="ap1" bookNo="20" page="240">

<Paragraph number="2920" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="2921" book="ap1">
Itthaṃ sudaṃ āyasmā dverataniyo thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="2922" type="center" book="ap1">
Dverataniyattherassāpadānaṃ tatiyaṃ.

  
Dasamaṃ bhāṇavāraṃ.
</Paragraph>
