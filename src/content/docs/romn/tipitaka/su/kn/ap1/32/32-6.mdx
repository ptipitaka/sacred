<Paragraph number="3541" type="verses" book="ap1">
_“Cetiyaṃ uttamaṃ nāma,_  
_sikhino lokabandhuno;_  
_Araññe irīṇe vane,_  
_andhāhiṇḍāmahaṃ tadā._  

</Paragraph>

</Division>
<Division number="27" book="ap1" bookNo="20" page="290">

<Paragraph number="3542" type="verses" book="ap1">
_Pavanā nikkhamantena,_  
_diṭṭhaṃ sīhāsanaṃ mayā;_  
_Ekaṃsaṃ añjaliṃ katvā,_  
_santhaviṃ lokanāyakaṃ._  

</Paragraph>

</Division>
<Division number="28" book="ap1" bookNo="20" page="291">

<Paragraph number="3543" type="verses" book="ap1">
_Divasabhāgaṃ thavitvāna,_  
_buddhaṃ lokagganāyakaṃ;_  
_Haṭṭho haṭṭhena cittena,_  
_imaṃ vācaṃ udīrayiṃ._  

</Paragraph>

</Division>
<Division number="29" book="ap1" bookNo="20" page="291">

<Paragraph number="3544" type="verses" book="ap1">
_‘Namo te purisājañña,_  
_namo te purisuttama;_  
_Sabbaññūsi mahāvīra,_  
_lokajeṭṭha narāsabha’._  

</Paragraph>

</Division>
<Division number="30" book="ap1" bookNo="20" page="291">

<Paragraph number="3545" type="verses" book="ap1">
_Abhitthavitvā sikhinaṃ,_  
_nimittakaraṇenahaṃ;_  
_Āsanaṃ abhivādetvā,_  
_pakkāmiṃ uttarāmukho._  

</Paragraph>

</Division>
<Division number="31" book="ap1" bookNo="20" page="291">

<Paragraph number="3546" type="verses" book="ap1">
_Ekattiṃse ito kappe,_  
_yaṃ thaviṃ vadataṃ varaṃ;_  
_Duggatiṃ nābhijānāmi,_  
_thomanāya idaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="32" book="ap1" bookNo="20" page="291">

<Paragraph number="3547" type="verses" book="ap1">
_Sattavīse ito kappe,_  
_atulā satta āsu te;_  
_Sattaratanasampannā,_  
_cakkavattī mahabbalā._  

</Paragraph>

</Division>
<Division number="33" book="ap1" bookNo="20" page="291">

<Paragraph number="3548" type="verses" book="ap1">

_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="3549" book="ap1">

</Paragraph>
</Division>

<Paragraph number="3550" type="center" book="ap1">
Āsanasanthavikattherassāpadānaṃ chaṭṭhaṃ.
