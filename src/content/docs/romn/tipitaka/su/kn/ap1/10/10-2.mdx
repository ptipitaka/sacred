<Division number="8" book="ap1">

<Paragraph number="1774" type="verses" book="ap1">
_“Tissassa lokanāthassa,_  
_suddhapīṭhamadāsahaṃ;_  
_Haṭṭho haṭṭhena cittena,_  
_buddhassādiccabandhuno._  

</Paragraph>

</Division>
<Division number="9" book="ap1">

<Paragraph number="1775" type="verses" book="ap1">
_Aṭṭhārase ito kappe,_  
_rājā āsiṃ mahāruci;_  
_Bhogo ca vipulo āsi,_  
_sayanañca anappakaṃ._  

</Paragraph>

</Division>
<Division number="10" book="ap1">

<Paragraph number="1776" type="verses" book="ap1">
_Pīṭhaṃ buddhassa datvāna,_  
_vippasannena cetasā;_  
_Anubhomi sakaṃ kammaṃ,_  
_pubbe sukatamattano._  

</Paragraph>

</Division>
<Division number="11" book="ap1">

<Paragraph number="1777" type="verses" book="ap1">
_Dvenavute ito kappe,_  
_yaṃ pīṭhamadadiṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_pīṭhadānassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="12" book="ap1">

<Paragraph number="1778" type="verses" book="ap1">
_Aṭṭhatiṃse ito kappe,_  
_tayo te cakkavattino;_  
_Ruci uparuci ceva,_  
_mahāruci tatiyako._  

</Paragraph>

</Division>
<Division number="13" book="ap1">

<Paragraph number="1779" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="1780" book="ap1">
Itthaṃ sudaṃ āyasmā sucintiko thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="1781" type="center" book="ap1">
Sucintikattherassāpadānaṃ dutiyaṃ.
</Paragraph>
