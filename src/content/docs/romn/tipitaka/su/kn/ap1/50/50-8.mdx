<Division number="57" book="ap1">

<Paragraph number="6500" type="verses" book="ap1">
_“Himavantassāvidūre,_  
_yamako nāma pabbato;_  
_Assamo sukato mayhaṃ,_  
_paṇṇasālā sumāpitā._  

</Paragraph>

</Division>
<Division number="58" book="ap1">

<Paragraph number="6501" type="verses" book="ap1">
_Nārado nāma nāmena,_  
_jaṭilo uggatāpano;_  
_Catuddasasahassāni,_  
_sissā paricaranti maṃ._  

</Paragraph>

</Division>
<Division number="59" book="ap1">

<Paragraph number="6502" type="verses" book="ap1">
_Paṭisallīnako santo,_  
_evaṃ cintesahaṃ tadā;_  
_‘Sabbo jano maṃ pūjeti,_  
_nāhaṃ pūjemi kiñcanaṃ._  

</Paragraph>

</Division>
<Division number="60" book="ap1">

<Paragraph number="6503" type="verses" book="ap1">
_Na me ovādako atthi,_  
_vattā koci na vijjati;_  
_Anācariyupajjhāyo,_  
_vane vāsaṃ upemahaṃ._  

</Paragraph>

</Division>
<Division number="61" book="ap1">

<Paragraph number="6504" type="verses" book="ap1">
_Upāsamāno yamahaṃ,_  
_garucittaṃ upaṭṭhahe;_  
_So me ācariyo natthi,_  
_vanavāso niratthako._  

</Paragraph>

</Division>
<Division number="62" book="ap1">

<Paragraph number="6505" type="verses" book="ap1">
_Āyāgaṃ me gavesissaṃ,_  
_garuṃ bhāvaniyaṃ tathā;_  
_Sāvassayo vasissāmi,_  
_na koci garahissati’._  

</Paragraph>

</Division>
<Division number="63" book="ap1">

<Paragraph number="6506" type="verses" book="ap1">
_Uttānakūlā nadikā,_  
_supatitthā manoramā;_  
_Saṃsuddhapulinākiṇṇā,_  
_avidūre mamassamaṃ._  

</Paragraph>

</Division>
<Division number="64" book="ap1">

<Paragraph number="6507" type="verses" book="ap1">
_Nadiṃ amarikaṃ nāma,_  
_upagantvānahaṃ tadā;_  
_Saṃvaḍḍhayitvā pulinaṃ,_  
_akaṃ pulinacetiyaṃ._  

</Paragraph>

</Division>
<Division number="65" book="ap1">

<Paragraph number="6508" type="verses" book="ap1">
_Ye te ahesuṃ sambuddhā,_  
_bhavantakaraṇā munī;_  
_Tesaṃ etādiso thūpo,_  
_taṃ nimittaṃ karomahaṃ._  

</Paragraph>

</Division>
<Division number="66" book="ap1">

<Paragraph number="6509" type="verses" book="ap1">
_Karitvā pulinaṃ thūpaṃ,_  
_sovaṇṇaṃ māpayiṃ ahaṃ;_  
_Soṇṇakiṅkaṇipupphāni,_  
_sahasse tīṇi pūjayiṃ._  

</Paragraph>

</Division>
<Division number="67" book="ap1">

<Paragraph number="6510" type="verses" book="ap1">
_Sāyaṃ pātaṃ namassāmi,_  
_vedajāto katañjalī;_  
_Sammukhā viya sambuddhaṃ,_  
_vandiṃ pulinacetiyaṃ._  

</Paragraph>

</Division>
<Division number="68" book="ap1">

<Paragraph number="6511" type="verses" book="ap1">
_Yadā kilesā jāyanti,_  
_vitakkā gehanissitā;_  
_Sarāmi sukataṃ thūpaṃ,_  
_paccavekkhāmi tāvade._  

</Paragraph>

</Division>
<Division number="69" book="ap1">

<Paragraph number="6512" type="verses" book="ap1">
_Upanissāya viharaṃ,_  
_satthavāhaṃ vināyakaṃ;_  
_Kilese saṃvaseyyāsi,_  
_na yuttaṃ tava mārisa._  

</Paragraph>

</Division>
<Division number="70" book="ap1">

<Paragraph number="6513" type="verses" book="ap1">
_Saha āvajjite thūpe,_  
_gāravaṃ hoti me tadā;_  
_Kuvitakke vinodesiṃ,_  
_nāgo tuttaṭṭito yathā._  

</Paragraph>

</Division>
<Division number="71" book="ap1">

<Paragraph number="6514" type="verses" book="ap1">
_Evaṃ viharamānaṃ maṃ,_  
_maccurājābhimaddatha;_  
_Tattha kālaṅkato santo,_  
_brahmalokamagacchahaṃ._  

</Paragraph>

</Division>
<Division number="72" book="ap1">

<Paragraph number="6515" type="verses" book="ap1">
_Yāvatāyuṃ vasitvāna,_  
_tidive upapajjahaṃ;_  
_Asītikkhattuṃ devindo,_  
_devarajjamakārayiṃ._  

</Paragraph>

</Division>
<Division number="73" book="ap1">

<Paragraph number="6516" type="verses" book="ap1">
_Satānaṃ tīṇikkhattuñca,_  
_cakkavattī ahosahaṃ;_  
_Padesarajjaṃ vipulaṃ,_  
_gaṇanāto asaṅkhiyaṃ._  

</Paragraph>

</Division>
<Division number="74" book="ap1">

<Paragraph number="6517" type="verses" book="ap1">
_Soṇṇakiṅkaṇipupphānaṃ,_  
_vipākaṃ anubhomahaṃ;_  
_Dhātīsatasahassāni,_  
_parivārenti maṃ bhave._  

</Paragraph>

</Division>
<Division number="75" book="ap1">

<Paragraph number="6518" type="verses" book="ap1">
_Thūpassa pariciṇṇattā,_  
_rajojallaṃ na limpati;_  
_Gatte sedā na muccanti,_  
_suppabhāso bhavāmahaṃ._  

</Paragraph>

</Division>
<Division number="76" book="ap1">

<Paragraph number="6519" type="verses" book="ap1">
_Aho me sukato thūpo,_  
_sudiṭṭhā marikā nadī;_  
_Thūpaṃ katvāna pulinaṃ,_  
_pattomhi acalaṃ padaṃ._  

</Paragraph>

</Division>
<Division number="77" book="ap1">

<Paragraph number="6520" type="verses" book="ap1">
_Kusalaṃ kattukāmena,_  
_jantunā sāragāhinā;_  
_Natthi khettaṃ akhettaṃ vā,_  
_paṭipattīva sādhakā._  

</Paragraph>

</Division>
<Division number="78" book="ap1">

<Paragraph number="6521" type="verses" book="ap1">
_Yathāpi balavā poso,_  
_aṇṇavaṃ taritussahe;_  
_Parittaṃ kaṭṭhamādāya,_  
_pakkhandeyya mahāsaraṃ._  

</Paragraph>

</Division>
<Division number="79" book="ap1">

<Paragraph number="6522" type="verses" book="ap1">
_Imāhaṃ kaṭṭhaṃ nissāya,_  
_tarissāmi mahodadhiṃ;_  
_Ussāhena vīriyena,_  
_tareyya udadhiṃ naro._  

</Paragraph>

</Division>
<Division number="80" book="ap1">

<Paragraph number="6523" type="verses" book="ap1">
_Tatheva me kataṃ kammaṃ,_  
_parittaṃ thokakañca yaṃ;_  
_Taṃ kammaṃ upanissāya,_  
_saṃsāraṃ samatikkamiṃ._  

</Paragraph>

</Division>
<Division number="81" book="ap1">

<Paragraph number="6524" type="verses" book="ap1">
_Pacchime bhave sampatte,_  
_sukkamūlena codito;_  
_Sāvatthiyaṃ pure jāto,_  
_mahāsāle suaḍḍhake._  

</Paragraph>

</Division>
<Division number="82" book="ap1">

<Paragraph number="6525" type="verses" book="ap1">
_Saddhā mātā pitā mayhaṃ,_  
_buddhassa saraṇaṃ gatā;_  
_Ubho diṭṭhapadā ete,_  
_anuvattanti sāsanaṃ._  

</Paragraph>

</Division>
<Division number="83" book="ap1">

<Paragraph number="6526" type="verses" book="ap1">
_Bodhipapaṭikaṃ gayha,_  
_soṇṇathūpamakārayuṃ;_  
_Sāyaṃ pātaṃ namassanti,_  
_sakyaputtassa sammukhā._  

</Paragraph>

</Division>
<Division number="84" book="ap1">

<Paragraph number="6527" type="verses" book="ap1">
_Uposathamhi divase,_  
_soṇṇathūpaṃ vinīharuṃ;_  
_Buddhassa vaṇṇaṃ kittentā,_  
_tiyāmaṃ vītināmayuṃ._  

</Paragraph>

</Division>
<Division number="85" book="ap1">

<Paragraph number="6528" type="verses" book="ap1">
_Saha disvānahaṃ thūpaṃ,_  
_sariṃ pulinacetiyaṃ;_  
_Ekāsane nisīditvā,_  
_arahattamapāpuṇiṃ._  


</Paragraph>
</Division>

<Paragraph number="6529" type="center" book="ap1">
Dvāvīsatimaṃ bhāṇavāraṃ.


</Paragraph>

<Division number="86" book="ap1">

<Paragraph number="6530" type="verses" book="ap1">
_Gavesamāno taṃ vīraṃ,_  
_dhammasenāpatiddasaṃ;_  
_Agārā nikkhamitvāna,_  
_pabbajiṃ tassa santike._  

</Paragraph>

</Division>
<Division number="87" book="ap1">

<Paragraph number="6531" type="verses" book="ap1">
_Jātiyā sattavassena,_  
_arahattamapāpuṇiṃ;_  
_Upasampādayī buddho,_  
_guṇamaññāya cakkhumā._  

</Paragraph>

</Division>
<Division number="88" book="ap1">

<Paragraph number="6532" type="verses" book="ap1">
_Dārakeneva santena,_  
_kiriyaṃ niṭṭhitaṃ mayā;_  
_Kataṃ me karaṇīyajja,_  
_sakyaputtassa sāsane._  

</Paragraph>

</Division>
<Division number="89" book="ap1">

<Paragraph number="6533" type="verses" book="ap1">
_Sabbaverabhayātīto,_  
_sabbasaṅgātigo isi;_  
_Sāvako te mahāvīra,_  
_soṇṇathūpassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="90" book="ap1">

<Paragraph number="6534" type="verses" book="ap1">
_Kilesā jhāpitā mayhaṃ,_  
_ …pe… _  
_viharāmi anāsavo._  

</Paragraph>

</Division>
<Division number="91" book="ap1">

<Paragraph number="6535" type="verses" book="ap1">
_Svāgataṃ vata me āsi,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ._  

</Paragraph>

</Division>
<Division number="92" book="ap1">

<Paragraph number="6536" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="6537" book="ap1">
Itthaṃ sudaṃ āyasmā pulinathūpiyo thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="6538" type="center" book="ap1">
Pulinathūpiyattherassāpadānaṃ aṭṭhamaṃ.
