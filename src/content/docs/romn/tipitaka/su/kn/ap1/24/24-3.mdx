_“Aruṇavatiyā nagare,_  
_ahosiṃ pūpiko tadā;_  
_Mama dvārena gacchantaṃ,_  
_sikhinaṃ addasaṃ jinaṃ._  

</Paragraph>

</Division>
<Division number="10" book="ap1">

<Paragraph number="2986" type="verses" book="ap1">
_Buddhassa pattaṃ paggayha,_  
_sālapupphaṃ adāsahaṃ;_  
_Sammaggatassa buddhassa,_  
_vippasannena cetasā._  

</Paragraph>

</Division>
<Division number="11" book="ap1">

<Paragraph number="2987" type="verses" book="ap1">
_Ekattiṃse ito kappe,_  
_Yaṃ pupphamabhidāsahaṃ;_  
_Duggatiṃ nābhijānāmi,_  
_Sālapupphassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="12" book="ap1">

<Paragraph number="2988" type="verses" book="ap1">
_Ito cuddasakappamhi,_  
_ahosiṃ amitañjalo;_  
_Sattaratanasampanno,_  
_cakkavattī mahabbalo._  

</Paragraph>

</Division>

<Paragraph number="2989" type="verses" book="ap1">

_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="2990" book="ap1">

</Paragraph>
