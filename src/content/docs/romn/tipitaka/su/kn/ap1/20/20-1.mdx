<Division number="1" book="ap1">

<Paragraph number="2666" type="verses" book="ap1">
_“Cullāsītisahassāni,_  
_thambhā sovaṇṇayā ahū;_  
_Devalaṭṭhipaṭibhāgaṃ,_  
_vimānaṃ me sunimmitaṃ._  

</Paragraph>

</Division>
<Division number="2" book="ap1">

<Paragraph number="2667" type="verses" book="ap1">
_Tamālapupphaṃ paggayha,_  
_vippasannena cetasā;_  
_Buddhassa abhiropayiṃ,_  
_sikhino lokabandhuno._  

</Paragraph>

</Division>
<Division number="3" book="ap1">

<Paragraph number="2668" type="verses" book="ap1">
_Ekattiṃse ito kappe,_  
_yaṃ kammamakariṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_buddhapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="4" book="ap1">

<Paragraph number="2669" type="verses" book="ap1">
_Ito vīsatime kappe,_  
_candatittoti ekako;_  
_Sattaratanasampanno,_  
_cakkavattī mahabbalo._  

</Paragraph>

</Division>
<Division number="5" book="ap1">

<Paragraph number="2670" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_vimokkhāpi ca aṭṭhime;_  
_Chaḷabhiññā sacchikatā,_  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="2671" book="ap1">
Itthaṃ sudaṃ āyasmā tamālapupphiyo thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="2672" type="center" book="ap1">
Tamālapupphiyattherassāpadānaṃ paṭhamaṃ.
</Paragraph>
