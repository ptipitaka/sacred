<Paragraph number="2829" type="verses" book="ap1">
_“Āraññikassa isino,_  
_cirarattatapassino;_  
_Vuddhassa bhāvitattassa,_  
_adāsiṃ pānadhiṃ ahaṃ._  

</Paragraph>

</Division>
<Division number="7" book="ap1" bookNo="20" page="233">

<Paragraph number="2830" type="verses" book="ap1">
_Tena kammena dvipadinda,_  
_lokajeṭṭha narāsabha;_  
_Dibbayānaṃ anubhomi,_  
_pubbakammassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="8" book="ap1" bookNo="20" page="233">

<Paragraph number="2831" type="verses" book="ap1">
_Catunnavute ito kappe,_  
_yaṃ kammamakariṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_pānadhissa idaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="9" book="ap1" bookNo="20" page="234">

<Paragraph number="2832" type="verses" book="ap1">
_Sattasattatito kappe,_  
_aṭṭha āsiṃsu khattiyā;_  
_Suyānā nāma nāmena,_  
_cakkavattī mahabbalā._  

</Paragraph>

</Division>

<Paragraph number="2833" type="verses" book="ap1">

_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="2834" book="ap1">

</Paragraph>
