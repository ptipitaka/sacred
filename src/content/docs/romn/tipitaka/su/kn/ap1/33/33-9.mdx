<Paragraph number="3675" type="verses" book="ap1">
_“Siddhatthassa bhagavato,_  
_lokajeṭṭhassa tādino;_  
_Saccaṃ pakāsayantassa,_  
_nibbāpentassa pāṇino._  

</Paragraph>

</Division>
<Division number="74" book="ap1" bookNo="20" page="302">

<Paragraph number="3676" type="verses" book="ap1">
_Jalajaṃ āharitvāna,_  
_satapattaṃ manoramaṃ;_  
_Pupphassa chattaṃ katvāna,_  
_buddhassa abhiropayiṃ._  

</Paragraph>

</Division>
<Division number="75" book="ap1" bookNo="20" page="302">

<Paragraph number="3677" type="verses" book="ap1">
_Siddhattho ca lokavidū,_  
_āhutīnaṃ paṭiggaho;_  
_Bhikkhusaṃghe ṭhito satthā,_  
_imaṃ gāthaṃ abhāsatha._  

</Paragraph>

</Division>
<Division number="76" book="ap1" bookNo="20" page="302">

<Paragraph number="3678" type="verses" book="ap1">
_‘Yo me cittaṃ pasādetvā,_  
_pupphacchattaṃ adhārayiṃ;_  
_Tena cittappasādena,_  
_duggatiṃ so na gacchati’._  

</Paragraph>

</Division>
<Division number="77" book="ap1" bookNo="20" page="302">

<Paragraph number="3679" type="verses" book="ap1">
_Idaṃ vatvāna sambuddho,_  
_siddhattho lokanāyako;_  
_Uyyojetvāna parisaṃ,_  
_vehāsaṃ nabhamuggami._  

</Paragraph>

</Division>
<Division number="78" book="ap1" bookNo="20" page="302">

<Paragraph number="3680" type="verses" book="ap1">
_Vuṭṭhite naradevamhi,_  
_setacchattampi vuṭṭhahi;_  
_Purato buddhaseṭṭhassa,_  
_gacchati chattamuttamaṃ._  

</Paragraph>

</Division>
<Division number="79" book="ap1" bookNo="20" page="302">

<Paragraph number="3681" type="verses" book="ap1">
_Catunnavutito kappe,_  
_yaṃ chattaṃ abhiropayiṃ;_  
_Duggatiṃ nābhijānāmi,_  
_pupphacchattassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="80" book="ap1" bookNo="20" page="302">

<Paragraph number="3682" type="verses" book="ap1">
_Catusattatikappamhi,_  
_aṭṭha jalasikhā ahū;_  
_Sattaratanasampannā,_  
_cakkavattī mahabbalā._  

</Paragraph>

</Division>
<Division number="81" book="ap1" bookNo="20" page="302">

<Paragraph number="3683" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="3684" book="ap1">
Itthaṃ sudaṃ āyasmā pupphacchattiyo thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="3685" type="center" book="ap1">
Pupphacchattiyattherassāpadānaṃ navamaṃ.
</Paragraph>
