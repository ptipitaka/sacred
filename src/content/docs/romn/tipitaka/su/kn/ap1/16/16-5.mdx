_“Devaputto ahaṃ santo,_  
_pūjayiṃ sikhināyakaṃ;_  
_Mandāravena pupphena,_  
_buddhassa abhiropayiṃ._

<Division number="26" book="ap1">

<Paragraph number="2386" type="verses" book="ap1">
_Sattāhaṃ chadanaṃ āsi,_  
_dibbaṃ mālaṃ tathāgate;_  
_Sabbe janā samāgantvā,_  
_namassiṃsu tathāgataṃ._  

</Paragraph>

</Division>
<Division number="27" book="ap1">

<Paragraph number="2387" type="verses" book="ap1">
_Ekattiṃse ito kappe,_  
_yaṃ pupphamabhipūjayiṃ;_  
_Duggatiṃ nābhijānāmi,_  
_buddhapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="28" book="ap1">

<Paragraph number="2388" type="verses" book="ap1">
_Ito ca dasame kappe,_  
_rājāhosiṃ jutindharo;_  
_Sattaratanasampanno,_  
_cakkavattī mahabbalo._  

</Paragraph>

</Division>
<Division number="29" book="ap1">

<Paragraph number="2389" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="2390" book="ap1">

</Paragraph>
