<Division number="222" book="ap1">

<Paragraph number="7181" type="verses" book="ap1">
_Ovādako viññāpako,_  
_tārako sabbapāṇinaṃ;_  
_Desanākusalo buddho,_  
_tāresi janataṃ bahuṃ._  

</Paragraph>

</Division>
<Division number="223" book="ap1">

<Paragraph number="7182" type="verses" book="ap1">
_Anukampako kāruṇiko,_  
_hitesī sabbapāṇinaṃ;_  
_Sampatte titthiye sabbe,_  
_pañcasīle patiṭṭhapi._  

</Paragraph>

</Division>
<Division number="224" book="ap1">

<Paragraph number="7183" type="verses" book="ap1">
_Evaṃ nirākulaṃ āsi,_  
_suññataṃ titthiyehi ca;_  
_Vicittaṃ arahantehi,_  
_vasībhūtehi tādibhi._  

</Paragraph>

</Division>
<Division number="225" book="ap1">

<Paragraph number="7184" type="verses" book="ap1">
_Ratanānaṭṭhapaññāsaṃ,_  
_uggato so mahāmuni;_  
_Kañcanagghiyasaṅkāso,_  
_bāttiṃsavaralakkhaṇo._  

</Paragraph>

</Division>
<Division number="226" book="ap1">

<Paragraph number="7185" type="verses" book="ap1">
_Vassasatasahassāni,_  
_āyu vijjati tāvade;_  
_Tāvatā tiṭṭhamāno so,_  
_tāresi janataṃ bahuṃ._  

</Paragraph>

</Division>
<Division number="227" book="ap1">

<Paragraph number="7186" type="verses" book="ap1">
_Tadāhaṃ haṃsavatiyaṃ,_  
_brāhmaṇo vedapāragū;_  
_Upecca sabbalokaggaṃ,_  
_assosiṃ dhammadesanaṃ._  

</Paragraph>

</Division>
<Division number="228" book="ap1">

<Paragraph number="7187" type="verses" book="ap1">
_Tadā so sāvakaṃ vīro,_  
_pabhinnamatigocaraṃ;_  
_Atthe dhamme nirutte ca,_  
_paṭibhāne ca kovidaṃ._  

</Paragraph>

</Division>
<Division number="229" book="ap1">

<Paragraph number="7188" type="verses" book="ap1">
_Ṭhapesi etadaggamhi,_  
_taṃ sutvā mudito ahaṃ;_  
_Sasāvakaṃ jinavaraṃ,_  
_sattāhaṃ bhojayiṃ tadā._  

</Paragraph>

</Division>
<Division number="230" book="ap1">

<Paragraph number="7189" type="verses" book="ap1">
_Dussehacchādayitvāna,_  
_sasissaṃ buddhisāgaraṃ;_  
_Nipacca pādamūlamhi,_  
_taṃ ṭhānaṃ patthayiṃ ahaṃ._  

</Paragraph>

</Division>
<Division number="231" book="ap1">

<Paragraph number="7190" type="verses" book="ap1">
_Tato avoca lokaggo,_  
_‘passathetaṃ dijuttamaṃ;_  
_Vinataṃ pādamūle me,_  
_kamalodarasappabhaṃ._  

</Paragraph>

</Division>
<Division number="232" book="ap1">

<Paragraph number="7191" type="verses" book="ap1">
_Buddhaseṭṭhassa bhikkhussa,_  
_Ṭhānaṃ patthayate ayaṃ;_  
_Tāya saddhāya cāgena,_  
_Saddhammassavanena ca._  

</Paragraph>

</Division>
<Division number="233" book="ap1">

<Paragraph number="7192" type="verses" book="ap1">
_Sabbattha sukhito hutvā,_  
_saṃsaritvā bhavābhave;_  
_Anāgatamhi addhāne,_  
_lacchase taṃ manorathaṃ._  

</Paragraph>

</Division>
<Division number="234" book="ap1">

<Paragraph number="7193" type="verses" book="ap1">
_Satasahassito kappe,_  
_okkākakulasambhavo;_  
_Gotamo nāma gottena,_  
_satthā loke bhavissati._  

</Paragraph>

</Division>
<Division number="235" book="ap1">

<Paragraph number="7194" type="verses" book="ap1">
_Tassa dhammesu dāyādo,_  
_oraso dhammanimmito;_  
_Koṭṭhiko nāma nāmena,_  
_hessati satthu sāvako’._  

</Paragraph>

</Division>
<Division number="236" book="ap1">

<Paragraph number="7195" type="verses" book="ap1">
_Taṃ sutvā mudito hutvā,_  
_yāvajīvaṃ tadā jinaṃ;_  
_Mettacitto paricariṃ,_  
_sato paññāsamāhito._  

</Paragraph>

</Division>
<Division number="237" book="ap1">

<Paragraph number="7196" type="verses" book="ap1">
_Tena kammavipākena,_  
_cetanāpaṇidhīhi ca;_  
_Jahitvā mānusaṃ dehaṃ,_  
_tāvatiṃsamagacchahaṃ._  

</Paragraph>

</Division>
<Division number="238" book="ap1">

<Paragraph number="7197" type="verses" book="ap1">

_devarajjamakārayiṃ;_  
_Satānaṃ pañcakkhattuñca,_  
_cakkavattī ahosahaṃ._  

</Paragraph>

</Division>
<Division number="239" book="ap1">

<Paragraph number="7198" type="verses" book="ap1">
_Padesarajjaṃ vipulaṃ,_  
_gaṇanāto asaṅkhiyaṃ;_  
_Sabbattha sukhito āsiṃ,_  
_tassa kammassa vāhasā._  

</Paragraph>

</Division>
<Division number="240" book="ap1">

<Paragraph number="7199" type="verses" book="ap1">

_devatte atha mānuse;_  
_Aññaṃ gatiṃ na gacchāmi,_  
_suciṇṇassa idaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="241" book="ap1">

<Paragraph number="7200" type="verses" book="ap1">

_khattiye atha brāhmaṇe;_  
_Nīce kule na jāyāmi,_  
_suciṇṇassa idaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="242" book="ap1">

<Paragraph number="7201" type="verses" book="ap1">
_Pacchime bhave sampatte,_  
_brahmabandhu ahosahaṃ;_  
_Sāvatthiyaṃ vippakule,_  
_paccājāto mahaddhane._  

</Paragraph>

</Division>
<Division number="243" book="ap1">

<Paragraph number="7202" type="verses" book="ap1">
_Mātā candavatī nāma,_  
_pitā me assalāyano;_  
_Yadā me pitaraṃ buddho,_  
_vinayī sabbasuddhiyā._  

</Paragraph>

</Division>
<Division number="244" book="ap1">

<Paragraph number="7203" type="verses" book="ap1">

_pabbajiṃ anagāriyaṃ;_  
_Moggallāno ācariyo,_  
_upajjhā sārisambhavo._  

</Paragraph>

</Division>
<Division number="245" book="ap1">

_diṭṭhi chinnā samūlikā;_  
_Nivāsento ca kāsāvaṃ,_  
_arahattamapāpuṇiṃ._  

</Paragraph>

</Division>

_paṭibhāne ca me mati;_  
_Pabhinnā tena lokaggo,_  
_etadagge ṭhapesi maṃ._  

</Paragraph>

</Division>

_aggo sambuddhasāsane._  

</Paragraph>
