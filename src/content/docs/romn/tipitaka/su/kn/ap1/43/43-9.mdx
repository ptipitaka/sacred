<Paragraph number="5626" type="verses" book="ap1">
_“Saddhāya abhinikkhamma,_  
_pabbajiṃ anagāriyaṃ;_  
_Vākacīradharo āsiṃ,_  
_tapokammamapassito._  

</Paragraph>

</Division>
<Division number="141" book="ap1" bookNo="20" page="440">

<Paragraph number="5627" type="verses" book="ap1">
_Atthadassī tu bhagavā,_  
_lokajeṭṭho narāsabho;_  
_Uppajji tamhi samaye,_  
_tārayanto mahājanaṃ._  

</Paragraph>

</Division>
<Division number="142" book="ap1" bookNo="20" page="440">

<Paragraph number="5628" type="verses" book="ap1">
_Balañca vata me khīṇaṃ,_  
_byādhinā paramena taṃ;_  
_Buddhaseṭṭhaṃ saritvāna,_  
_puline thūpamuttamaṃ._  

</Paragraph>

</Division>
<Division number="143" book="ap1" bookNo="20" page="441">

<Paragraph number="5629" type="verses" book="ap1">
_Karitvā haṭṭhacittohaṃ,_  
_sahatthena samokiriṃ;_  
_Soṇṇakiṅkaṇipupphāni,_  
_udaggamanaso ahaṃ._  

</Paragraph>

</Division>
<Division number="144" book="ap1" bookNo="20" page="441">

<Paragraph number="5630" type="verses" book="ap1">
_Sammukhā viya sambuddhaṃ,_  
_thūpaṃ paricariṃ ahaṃ;_  
_Tena cetopasādena,_  
_atthadassissa tādino._  

</Paragraph>

</Division>
<Division number="145" book="ap1" bookNo="20" page="441">

<Paragraph number="5631" type="verses" book="ap1">
_Devalokaṃ gato santo,_  
_labhāmi vipulaṃ sukhaṃ;_  
_Suvaṇṇavaṇṇo tatthāsiṃ,_  
_buddhapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="146" book="ap1" bookNo="20" page="441">

<Paragraph number="5632" type="verses" book="ap1">
_Asītikoṭiyo mayhaṃ,_  
_nāriyo samalaṅkatā;_  
_Sadā mayhaṃ upaṭṭhanti,_  
_buddhapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="147" book="ap1" bookNo="20" page="441">

<Paragraph number="5633" type="verses" book="ap1">
_Saṭṭhituriyasahassāni,_  
_bheriyo paṇavāni ca;_  
_Saṅkhā ca ḍiṇḍimā tattha,_  
_vaggū vajjanti dundubhī._  

</Paragraph>

</Division>
<Division number="148" book="ap1" bookNo="20" page="441">

<Paragraph number="5634" type="verses" book="ap1">
_Cullāsītisahassāni,_  
_hatthino samalaṅkatā;_  
_Tidhāpabhinnamātaṅgā,_  
_kuñjarā saṭṭhihāyanā._  

</Paragraph>

</Division>
<Division number="149" book="ap1" bookNo="20" page="441">

<Paragraph number="5635" type="verses" book="ap1">
_Hemajālābhisañchannā,_  
_upaṭṭhānaṃ karonti me;_  
_Balakāye gaje ceva,_  
_ūnatā me na vijjati._  

</Paragraph>

</Division>
<Division number="150" book="ap1" bookNo="20" page="441">

<Paragraph number="5636" type="verses" book="ap1">
_Soṇṇakiṅkaṇipupphānaṃ,_  
_vipākaṃ anubhomahaṃ;_  
_Aṭṭhapaññāsakkhattuñca,_  
_devarajjamakārayiṃ._  

</Paragraph>

</Division>
<Division number="151" book="ap1" bookNo="20" page="441">

<Paragraph number="5637" type="verses" book="ap1">
_Ekasattatikkhattuñca,_  
_cakkavattī ahosahaṃ;_  
_Pathabyā rajjaṃ ekasataṃ,_  
_mahiyā kārayiṃ ahaṃ._  

</Paragraph>

</Division>
<Division number="152" book="ap1" bookNo="20" page="441">

<Paragraph number="5638" type="verses" book="ap1">
_So dāni amataṃ patto,_  
_asaṅkhataṃ sududdasaṃ;_  
_Saṃyojanaparikkhīṇo,_  
_natthi dāni punabbhavo._  

</Paragraph>

</Division>
<Division number="153" book="ap1" bookNo="20" page="441">

<Paragraph number="5639" type="verses" book="ap1">
_Aṭṭhārase kappasate,_  
_yaṃ pupphamabhiropayiṃ;_  
_Duggatiṃ nābhijānāmi,_  
_buddhapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="154" book="ap1" bookNo="20" page="442">

<Paragraph number="5640" type="verses" book="ap1">
_Kilesā jhāpitā mayhaṃ,_  
_ …pe… _  
_viharāmi anāsavo._  

</Paragraph>

</Division>
<Division number="155" book="ap1" bookNo="20" page="442">

<Paragraph number="5641" type="verses" book="ap1">
_Svāgataṃ vata me āsi,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ._  

</Paragraph>

</Division>
<Division number="156" book="ap1" bookNo="20" page="442">

<Paragraph number="5642" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="5643" book="ap1">
Itthaṃ sudaṃ āyasmā sovaṇṇakiṅkaṇiyo thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="5644" type="center" book="ap1">
Sovaṇṇakiṅkaṇiyattherassāpadānaṃ navamaṃ.
</Paragraph>
