<Paragraph number="4300" type="verses" book="ap1">
_‘Bahū medhigatā bhogā,_  
_phītaṃ antepuraṃ mama;_  
_Rājāpi sannimantesi,_  
_ānando pathavissaro._

<Division number="4" book="ap1">

<Paragraph number="4301" type="verses" book="ap1">
_Ayañca buddho uppanno,_  
_adhiccuppattiko muni;_  
_Saṃvijjanti ca me bhogā,_  
_dānaṃ dassāmi satthuno._  

</Paragraph>

</Division>
<Division number="5" book="ap1">

<Paragraph number="4302" type="verses" book="ap1">
_Padumena rājaputtena,_  
_dinnaṃ dānavaraṃ jine;_  
_Hatthināge ca pallaṅke,_  
_apassenañcanappakaṃ._  

</Paragraph>

</Division>
<Division number="6" book="ap1">

<Paragraph number="4303" type="verses" book="ap1">
_Ahampi dānaṃ dassāmi,_  
_saṃghe gaṇavaruttame;_  
_Adinnapubbamaññesaṃ,_  
_bhavissaṃ ādikammiko._  

</Paragraph>

</Division>
<Division number="7" book="ap1">

<Paragraph number="4304" type="verses" book="ap1">
_Cintetvāhaṃ bahuvidhaṃ,_  
_yāge yassa sukhaṃphalaṃ;_  
_Parikkhāradānamaddakkhiṃ,_  
_mama saṅkappapūraṇaṃ._  

</Paragraph>

</Division>

_Adinnapubbamaññesaṃ,_  
_bhavissaṃ ādikammiko’._
</Paragraph>
</Division>
