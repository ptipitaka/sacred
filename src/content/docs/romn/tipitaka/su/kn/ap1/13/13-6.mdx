_“Nagare dhaññavatiyā,_  
_ahosiṃ brāhmaṇo tadā;_  
_Lakkhaṇe itihāse ca,_  
_sanighaṇḍusakeṭubhe._  

</Paragraph>

<Division number="66" book="ap1" bookNo="20" page="176">

<Paragraph number="2129" type="verses" book="ap1">
_Padako veyyākaraṇo,_  
_nimittakovido ahaṃ;_  
_Mante ca sisse vācesiṃ,_  
_tiṇṇaṃ vedāna pāragū._  

</Paragraph>

</Division>
<Division number="67" book="ap1" bookNo="20" page="176">

<Paragraph number="2130" type="verses" book="ap1">
_Pañca uppalahatthāni,_  
_piṭṭhiyaṃ ṭhapitāni me;_  
_Āhutiṃ yiṭṭhukāmohaṃ,_  
_pitumātusamāgame._  

</Paragraph>

</Division>
<Division number="68" book="ap1" bookNo="20" page="176">

<Paragraph number="2131" type="verses" book="ap1">
_Tadā vipassī bhagavā,_  
_bhikkhusaṃghapurakkhato;_  
_Obhāsento disā sabbā,_  
_āgacchati narāsabho._  

</Paragraph>

</Division>
<Division number="69" book="ap1" bookNo="20" page="176">

<Paragraph number="2132" type="verses" book="ap1">
_Āsanaṃ paññapetvāna,_  
_nimantetvā mahāmuniṃ;_  
_Santharitvāna taṃ pupphaṃ,_  
_abhinesiṃ sakaṃ gharaṃ._  

</Paragraph>

</Division>
<Division number="70" book="ap1" bookNo="20" page="176">

<Paragraph number="2133" type="verses" book="ap1">
_Yaṃ me atthi sake gehe,_  
_āmisaṃ paccupaṭṭhitaṃ;_  
_Tāhaṃ buddhassa pādāsiṃ,_  
_pasanno sehi pāṇibhi._  

</Paragraph>

</Division>
<Division number="71" book="ap1" bookNo="20" page="176">

<Paragraph number="2134" type="verses" book="ap1">
_Bhuttāviṃ kālamaññāya,_  
_pupphahatthamadāsahaṃ;_  
_Anumoditvāna sabbaññū,_  
_pakkāmi uttarāmukho._  

</Paragraph>

</Division>
<Division number="72" book="ap1" bookNo="20" page="176">

<Paragraph number="2135" type="verses" book="ap1">
_Ekanavutito kappe,_  
_yaṃ pupphamadadiṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_pupphadānassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="73" book="ap1" bookNo="20" page="176">

<Paragraph number="2136" type="verses" book="ap1">
_Anantaraṃ ito kappe,_  
_rājāhuṃ varadassano;_  
_Sattaratanasampanno,_  
_cakkavattī mahabbalo._  

</Paragraph>

</Division>

_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

</Paragraph>
