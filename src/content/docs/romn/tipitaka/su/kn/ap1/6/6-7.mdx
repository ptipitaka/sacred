<Division number="50" book="ap1">

<Paragraph number="1437" type="verses" book="ap1">
_“Padumuttarabuddhassa,_  
_sabbabhūtānukampino;_  
_Saddahitvāna saddhamme,_  
_ujudiṭṭhi ahosahaṃ._  

</Paragraph>

</Division>
<Division number="51" book="ap1">

<Paragraph number="1438" type="verses" book="ap1">
_Padīpadānaṃ pādāsiṃ,_  
_parivāretvāna bodhiyaṃ;_  
_Saddahanto padīpāni,_  
_akariṃ tāvade ahaṃ._  

</Paragraph>

</Division>
<Division number="52" book="ap1">

<Paragraph number="1439" type="verses" book="ap1">
_Yaṃ yaṃ yonupapajjāmi,_  
_devattaṃ atha mānusaṃ;_  
_Ākāse ukkaṃ dhārenti,_  
_dīpadānassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="53" book="ap1">

<Paragraph number="1440" type="verses" book="ap1">
_Tirokuṭṭaṃ tiroselaṃ,_  
_samatiggayha pabbataṃ;_  
_Samantā yojanasataṃ,_  
_dassanaṃ anubhomahaṃ._  

</Paragraph>

</Division>
<Division number="54" book="ap1">

<Paragraph number="1441" type="verses" book="ap1">
_Tena kammāvasesena,_  
_pattomhi āsavakkhayaṃ;_  
_Dhāremi antimaṃ dehaṃ,_  
_dvipadindassa sāsane._  

</Paragraph>

</Division>
<Division number="55" book="ap1">

<Paragraph number="1442" type="verses" book="ap1">
_Catuttiṃse kappasate,_  
_satacakkhusanāmakā;_  
_Rājāhesuṃ mahātejā,_  
_cakkavattī mahabbalā._  

</Paragraph>

</Division>
<Division number="56" book="ap1">

<Paragraph number="1443" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="1444" book="ap1">
Itthaṃ sudaṃ āyasmā pañcadīpako thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="1445" type="center" book="ap1">
Pañcadīpakattherassāpadānaṃ sattamaṃ.
</Paragraph>
