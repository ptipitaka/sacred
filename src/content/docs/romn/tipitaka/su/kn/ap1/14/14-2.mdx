<Paragraph number="2193" type="verses" book="ap1">
_“Vinatā nadiyā tīre,_  
_pilakkhu phalito ahu;_  
_Tāhaṃ rukkhaṃ gavesanto,_  
_addasaṃ lokanāyakaṃ._  

</Paragraph>

</Division>
<Division number="11" book="ap1" bookNo="20" page="181">

<Paragraph number="2194" type="verses" book="ap1">
_Ketakaṃ pupphitaṃ disvā,_  
_vaṇṭe chetvānahaṃ tadā;_  
_Buddhassa abhiropesiṃ,_  
_sikhino lokabandhuno._  

</Paragraph>

</Division>
<Division number="12" book="ap1" bookNo="20" page="181">

<Paragraph number="2195" type="verses" book="ap1">
_Yena ñāṇena pattosi,_  
_accutaṃ amataṃ padaṃ;_  
_Taṃ ñāṇaṃ abhipūjemi,_  
_buddhaseṭṭha mahāmuni._  

</Paragraph>

</Division>
<Division number="13" book="ap1" bookNo="20" page="181">

<Paragraph number="2196" type="verses" book="ap1">
_Ñāṇamhi pūjaṃ katvāna,_  
_pilakkhumaddasaṃ ahaṃ;_  
_Paṭiladdhomhi taṃ paññaṃ,_  
_ñāṇapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="14" book="ap1" bookNo="20" page="181">

<Paragraph number="2197" type="verses" book="ap1">
_Ekattiṃse ito kappe,_  
_yaṃ pupphamabhiropayiṃ;_  
_Duggatiṃ nābhijānāmi,_  
_ñāṇapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="15" book="ap1" bookNo="20" page="181">

<Paragraph number="2198" type="verses" book="ap1">
_Ito terasakappamhi,_  
_dvādasāsuṃ phaluggatā;_  
_Sattaratanasampannā,_  
_cakkavattī mahapphalā._  

</Paragraph>

</Division>
<Division number="16" book="ap1" bookNo="20" page="181">

<Paragraph number="2199" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="2200" book="ap1">
Itthaṃ sudaṃ āyasmā sudassano thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>
