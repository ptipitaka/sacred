<Division number="14" book="ap1">

<Paragraph number="3525" type="verses" book="ap1">
_“Nadīkūle vasāmahaṃ,_  
_nesādo nāma brāhmaṇo;_  
_Satapattehi pupphehi,_  
_sammajjitvāna assamaṃ._  

</Paragraph>

</Division>
<Division number="15" book="ap1">

<Paragraph number="3526" type="verses" book="ap1">
_Suvaṇṇavaṇṇaṃ sambuddhaṃ,_  
_Siddhatthaṃ lokanāyakaṃ;_  
_Disvā nabhena gacchantaṃ,_  
_Hāso me udapajjatha._  

</Paragraph>

</Division>
<Division number="16" book="ap1">

<Paragraph number="3527" type="verses" book="ap1">
_Paccuggantvāna sambuddhaṃ,_  
_lokajeṭṭhaṃ narāsabhaṃ;_  
_Assamaṃ atināmetvā,_  
_jalajaggehi okiriṃ._  

</Paragraph>

</Division>
<Division number="17" book="ap1">

<Paragraph number="3528" type="verses" book="ap1">
_Catunnavutito kappe,_  
_yaṃ pupphamabhiropayiṃ;_  
_Duggatiṃ nābhijānāmi,_  
_buddhapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="18" book="ap1">

<Paragraph number="3529" type="verses" book="ap1">
_Ito te sattame kappe,_  
_caturo pādapāvarā;_  
_Sattaratanasampannā,_  
_cakkavattī mahabbalā._  

</Paragraph>

</Division>
<Division number="19" book="ap1">

<Paragraph number="3530" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="3531" book="ap1">
Itthaṃ sudaṃ āyasmā sattapadumiyo thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="3532" type="center" book="ap1">
Sattapadumiyattherassāpadānaṃ catutthaṃ.
</Paragraph>
