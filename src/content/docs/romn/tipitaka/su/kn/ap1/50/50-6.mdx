<Division number="41" book="ap1">

<Paragraph number="6480" type="verses" book="ap1">
_“Sucintitaṃ bhagavantaṃ,_  
_lokajeṭṭhaṃ narāsabhaṃ;_  
_Upaviṭṭhaṃ mahāraññaṃ,_  
_vātābādhena pīḷitaṃ._  

</Paragraph>

</Division>
<Division number="42" book="ap1">

<Paragraph number="6481" type="verses" book="ap1">
_Disvā cittaṃ pasādetvā,_  
_ghatamaṇḍamupānayiṃ;_  
_Katattā ācitattā ca,_  
_gaṅgā bhāgīrathī ayaṃ._  

</Paragraph>

</Division>
<Division number="43" book="ap1">

<Paragraph number="6482" type="verses" book="ap1">
_Mahāsamuddā cattāro,_  
_ghataṃ sampajjare mama;_  
_Ayañca pathavī ghorā,_  
_appamāṇā asaṅkhiyā._  

</Paragraph>

</Division>
<Division number="44" book="ap1">

<Paragraph number="6483" type="verses" book="ap1">
_Mama saṅkappamaññāya,_  
_bhavate madhusakkarā;_  
_Cātuddīpā ime rukkhā,_  
_pādapā dharaṇīruhā._  

</Paragraph>

</Division>
<Division number="45" book="ap1">

<Paragraph number="6484" type="verses" book="ap1">
_Mama saṅkappamaññāya,_  
_kapparukkhā bhavanti te;_  
_Paññāsakkhattuṃ devindo,_  
_devarajjamakārayiṃ._  

</Paragraph>

</Division>
<Division number="46" book="ap1">

<Paragraph number="6485" type="verses" book="ap1">
_Ekapaññāsakkhattuñca,_  
_cakkavattī ahosahaṃ;_  
_Padesarajjaṃ vipulaṃ,_  
_gaṇanāto asaṅkhiyaṃ._  

</Paragraph>

</Division>
<Division number="47" book="ap1">

<Paragraph number="6486" type="verses" book="ap1">
_Catunnavutito kappe,_  
_yaṃ dānamadadiṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_ghatamaṇḍassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="48" book="ap1">

<Paragraph number="6487" type="verses" book="ap1">
_Kilesā jhāpitā mayhaṃ,_  
_ …pe… _  
_viharāmi anāsavo._  

</Paragraph>

</Division>
<Division number="49" book="ap1">

<Paragraph number="6488" type="verses" book="ap1">
_Svāgataṃ vata me āsi,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ._  

</Paragraph>

</Division>
<Division number="50" book="ap1">

<Paragraph number="6489" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="6490" book="ap1">
Itthaṃ sudaṃ āyasmā ghatamaṇḍadāyako thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>
