<Division number="70" book="ap1">

<Paragraph number="4367" type="verses" book="ap1">
_‘Anūnakaṃ dānavaraṃ,_  
_yo me pādāsi māṇavo;_  
_Tamahaṃ kittayissāmi,_  
_suṇātha mama bhāsato._  

</Paragraph>

</Division>
<Division number="71" book="ap1">

<Paragraph number="4368" type="verses" book="ap1">
_Hatthī assā rathā pattī,_  
_senā ca caturaṅginī;_  
_Parivāressantimaṃ niccaṃ,_  
_sabbadānassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="72" book="ap1">

<Paragraph number="4369" type="verses" book="ap1">
_Hatthiyānaṃ assayānaṃ,_  
_sivikā sandamānikā;_  
_Upaṭṭhissantimaṃ niccaṃ,_  
_sabbadānassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="73" book="ap1">

<Paragraph number="4370" type="verses" book="ap1">
_Saṭṭhi rathasahassāni,_  
_sabbālaṅkārabhūsitā;_  
_Parivāressantimaṃ niccaṃ,_  
_sabbadānassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="74" book="ap1">

<Paragraph number="4371" type="verses" book="ap1">
_Saṭṭhi tūriyasahassāni,_  
_bheriyo samalaṅkatā;_  
_Vajjayissantimaṃ niccaṃ,_  
_sabbadānassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="75" book="ap1">

<Paragraph number="4372" type="verses" book="ap1">
_Chaḷāsītisahassāni,_  
_nāriyo samalaṅkatā;_  
_Vicittavatthābharaṇā,_  
_āmukkamaṇikuṇḍalā._  

</Paragraph>

</Division>
<Division number="76" book="ap1">

<Paragraph number="4373" type="verses" book="ap1">
_Aḷārapamhā hasulā,_  
_susaññā tanumajjhimā;_  
_Parivāressantimaṃ niccaṃ,_  
_sabbadānassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="77" book="ap1">

<Paragraph number="4374" type="verses" book="ap1">
_Tiṃsakappasahassāni,_  
_devaloke ramissati;_  
_Sahassakkhattuṃ devindo,_  
_devarajjaṃ karissati._  

</Paragraph>

</Division>
<Division number="78" book="ap1">

<Paragraph number="4375" type="verses" book="ap1">
_Sahassakkhattuṃ rājā ca,_  
_cakkavattī bhavissati;_  
_Padesarajjaṃ vipulaṃ,_  
_gaṇanāto asaṅkhiyaṃ._  

</Paragraph>

</Division>
<Division number="79" book="ap1">

<Paragraph number="4376" type="verses" book="ap1">
_Devaloke vasantassa,_  
_puññakammasamaṅgino;_  
_Devalokapariyantaṃ,_  
_ratanachattaṃ dharissati._  

</Paragraph>

</Division>
<Division number="80" book="ap1">

<Paragraph number="4377" type="verses" book="ap1">
_Icchissati yadā chāyaṃ,_  
_chadanaṃ dussapupphajaṃ;_  
_Imassa cittamaññāya,_  
_nibaddhaṃ chādayissati._  

</Paragraph>

</Division>
<Division number="81" book="ap1">

<Paragraph number="4378" type="verses" book="ap1">
_Devalokā cavitvāna,_  
_sukkamūlena codito;_  
_Puññakammena saṃyutto,_  
_brahmabandhu bhavissati._  

</Paragraph>

</Division>
<Division number="82" book="ap1">

<Paragraph number="4379" type="verses" book="ap1">
_Kappasatasahassamhi,_  
_okkākakulasambhavo;_  
_Gotamo nāma gottena,_  
_satthā loke bhavissati._  

</Paragraph>

</Division>
<Division number="83" book="ap1">

<Paragraph number="4380" type="verses" book="ap1">
_Sabbametaṃ abhiññāya,_  
_gotamo sakyapuṅgavo;_  
_Bhikkhusaṃghe nisīditvā,_  
_etadagge ṭhapessati._  

</Paragraph>

</Division>
<Division number="84" book="ap1">

<Paragraph number="4381" type="verses" book="ap1">
_Pilindavacchanāmena,_  
_hessati satthusāvako;_  
_Devānaṃ asurānañca,_  
_gandhabbānañca sakkato._  

</Paragraph>

</Division>
<Division number="85" book="ap1">

<Paragraph number="4382" type="verses" book="ap1">
_Bhikkhūnaṃ bhikkhunīnañca,_  
_gihīnañca tatheva so;_  
_Piyo hutvāna sabbesaṃ,_  
_viharissatināsavo’._
</Paragraph>
</Division>
