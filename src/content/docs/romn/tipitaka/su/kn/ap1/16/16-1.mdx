<Paragraph number="2353" type="verses" book="ap1">
_“Candaṃva vimalaṃ suddhaṃ,_  
_vippasannamanāvilaṃ;_  
_Nandībhavaparikkhīṇaṃ,_  
_tiṇṇaṃ loke visattikaṃ._  

</Paragraph>

</Division>
<Division number="2" book="ap1" bookNo="20" page="194">

<Paragraph number="2354" type="verses" book="ap1">
_Nibbāpayantaṃ janataṃ,_  
_Tiṇṇaṃ tārayataṃ varaṃ;_  
_Muniṃ vanamhi jhāyantaṃ,_  
_Ekaggaṃ susamāhitaṃ._  

</Paragraph>

</Division>
<Division number="3" book="ap1" bookNo="20" page="194">

<Paragraph number="2355" type="verses" book="ap1">
_Bandhujīvakapupphāni,_  
_lagetvā suttakenahaṃ;_  
_Buddhassa abhiropayiṃ,_  
_sikhino lokabandhuno._  

</Paragraph>

</Division>
<Division number="4" book="ap1" bookNo="20" page="194">

<Paragraph number="2356" type="verses" book="ap1">
_Ekattiṃse ito kappe,_  
_yaṃ kammamakariṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_buddhapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="5" book="ap1" bookNo="20" page="194">

<Paragraph number="2357" type="verses" book="ap1">
_Ito sattamake kappe,_  
_manujindo mahāyaso;_  
_Samantacakkhu nāmāsi,_  
_cakkavattī mahabbalo._  

</Paragraph>

</Division>
<Division number="6" book="ap1" bookNo="20" page="194">

<Paragraph number="2358" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_vimokkhāpi ca aṭṭhime;_  
_Chaḷabhiññā sacchikatā,_  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="2359" book="ap1">
Itthaṃ sudaṃ āyasmā bandhujīvako thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="2360" type="center" book="ap1">
Bandhujīvakattherassāpadānaṃ paṭhamaṃ.
</Paragraph>
