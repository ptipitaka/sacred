_bhikkhusaṃghapurakkhato;_

<Division number="12" book="ap1">

<Paragraph number="2837" type="verses" book="ap1">
_Paramakāruññapattomhi,_  
_samitiṃ agamāsahaṃ;_  
_Sohaṃ nisinnako santo,_  
_dhammamassosi satthuno._  

</Paragraph>

</Division>
<Division number="13" book="ap1">

<Paragraph number="2838" type="verses" book="ap1">
_Tassāhaṃ dhammaṃ sutvāna,_  
_devalokaṃ agacchahaṃ;_  
_Tiṃsakappāni devesu,_  
_avasiṃ tatthahaṃ pure._  

</Paragraph>

</Division>
<Division number="14" book="ap1">

<Paragraph number="2839" type="verses" book="ap1">
_Ekattiṃse ito kappe,_  
_yaṃ saññamalabhiṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_saccasaññāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="15" book="ap1">

<Paragraph number="2840" type="verses" book="ap1">
_Chabbīsamhi ito kappe,_  
_eko āsiṃ janādhipo;_  
_Ekaphusitanāmena,_  
_cakkavattī mahabbalo._  

</Paragraph>

</Division>
<Division number="16" book="ap1">

<Paragraph number="2841" type="verses" book="ap1">

_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="2842" book="ap1">

</Paragraph>
</Division>

<Paragraph number="2843" type="center" book="ap1">
Saccasaññakattherassāpadānaṃ tatiyaṃ.
</Paragraph>
