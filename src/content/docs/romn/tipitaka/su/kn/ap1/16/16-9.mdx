<Paragraph number="2414" type="verses" book="ap1">
_“Kānanaṃ vanamogayha,_  
_vasāmi luddako ahaṃ;_  
_Punnāgaṃ pupphitaṃ disvā,_  
_buddhaseṭṭhaṃ anussariṃ._  

</Paragraph>

</Division>
<Division number="47" book="ap1" bookNo="20" page="199">

<Paragraph number="2415" type="verses" book="ap1">
_Taṃ pupphaṃ ocinitvāna,_  
_sugandhaṃ gandhitaṃ subhaṃ;_  
_Thūpaṃ karitvā puline,_  
_buddhassa abhiropayiṃ._  

</Paragraph>

</Division>
<Division number="48" book="ap1" bookNo="20" page="199">

<Paragraph number="2416" type="verses" book="ap1">
_Dvenavute ito kappe,_  
_yaṃ pupphamabhipūjayiṃ;_  
_Duggatiṃ nābhijānāmi,_  
_buddhapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="49" book="ap1" bookNo="20" page="199">

<Paragraph number="2417" type="verses" book="ap1">
_Ekamhi navute kappe,_  
_eko āsiṃ tamonudo;_  
_Sattaratanasampanno,_  
_cakkavattī mahabbalo._  

</Paragraph>

</Division>
<Division number="50" book="ap1" bookNo="20" page="199">

<Paragraph number="2418" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="2419" book="ap1">
Itthaṃ sudaṃ āyasmā punnāgapupphiyo thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="2420" type="center" book="ap1">
Punnāgapupphiyattherassāpadānaṃ navamaṃ.
</Paragraph>
