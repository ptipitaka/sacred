<Division number="45" book="ap1">

<Paragraph number="1547" type="verses" book="ap1">
_“Pakkhijāto tadā āsiṃ,_  
_supaṇṇo garuḷādhipo;_  
_Addasaṃ virajaṃ buddhaṃ,_  
_gacchantaṃ gandhamādanaṃ._  

</Paragraph>

</Division>
<Division number="46" book="ap1">

<Paragraph number="1548" type="verses" book="ap1">
_Jahitvā garuḷavaṇṇaṃ,_  
_māṇavakaṃ adhārayiṃ;_  
_Ekaṃ vatthaṃ mayā dinnaṃ,_  
_dvipadindassa tādino._  

</Paragraph>

</Division>
<Division number="47" book="ap1">

<Paragraph number="1549" type="verses" book="ap1">
_Tañca dussaṃ paṭiggayha,_  
_buddho lokagganāyako;_  
_Antalikkhe ṭhito satthā,_  
_imā gāthā abhāsatha._  

</Paragraph>

</Division>
<Division number="48" book="ap1">

<Paragraph number="1550" type="verses" book="ap1">
_‘Iminā vatthadānena,_  
_cittassa paṇidhīhi ca;_  
_Pahāya garuḷaṃ yoniṃ,_  
_devaloke ramissati’._  

</Paragraph>

</Division>
<Division number="49" book="ap1">

<Paragraph number="1551" type="verses" book="ap1">
_Atthadassī tu bhagavā,_  
_lokajeṭṭho narāsabho;_  
_Vatthadānaṃ pasaṃsitvā,_  
_pakkāmi uttarāmukho._  

</Paragraph>

</Division>
<Division number="50" book="ap1">

<Paragraph number="1552" type="verses" book="ap1">
_Bhave nibbattamānamhi,_  
_honti me vatthasampadā;_  
_Ākāse chadanaṃ hoti,_  
_vatthadānassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="51" book="ap1">

<Paragraph number="1553" type="verses" book="ap1">
_Aruṇavā satta janā,_  
_cakkavattī mahabbalā;_  
_Chattiṃsatimhi āsiṃsu,_  
_kappamhi manujādhipā._  

</Paragraph>

</Division>
<Division number="52" book="ap1">

<Paragraph number="1554" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="1555" book="ap1">
Itthaṃ sudaṃ āyasmā vatthadāyako thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="1556" type="center" book="ap1">
Vatthadāyakattherassāpadānaṃ sattamaṃ.
</Paragraph>
