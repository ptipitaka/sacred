<Division number="164" book="ap1">

<Paragraph number="6365" type="verses" book="ap1">
_“Yadā dīpaṅkaro buddho,_  
_sumedhaṃ byākarī jino;_  
_‘Aparimeyye ito kappe,_  
_ayaṃ buddho bhavissati._  

</Paragraph>

</Division>
<Division number="165" book="ap1">

<Paragraph number="6366" type="verses" book="ap1">
_Imassa janikā mātā,_  
_māyā nāma bhavissati;_  
_Pitā suddhodano nāma,_  
_ayaṃ hessati gotamo._  

</Paragraph>

</Division>
<Division number="166" book="ap1">

<Paragraph number="6367" type="verses" book="ap1">
_Padhānaṃ padahitvāna,_  
_katvā dukkarakārikaṃ;_  
_Assatthamūle sambuddho,_  
_bujjhissati mahāyaso._  

</Paragraph>

</Division>
<Division number="167" book="ap1">

<Paragraph number="6368" type="verses" book="ap1">
_Upatisso kolito ca,_  
_aggā hessanti sāvakā;_  
_Ānando nāma nāmena,_  
_upaṭṭhissatimaṃ jinaṃ._  

</Paragraph>

</Division>
<Division number="168" book="ap1">

<Paragraph number="6369" type="verses" book="ap1">
_Khemā uppalavaṇṇā ca,_  
_aggā hessanti sāvikā;_  
_Citto āḷavako ceva,_  
_aggā hessantupāsakā._  

</Paragraph>

</Division>
<Division number="169" book="ap1">

<Paragraph number="6370" type="verses" book="ap1">
_Khujjuttarā nandamātā,_  
_aggā hessantupāsikā;_  
_Bodhi imassa vīrassa,_  
_assatthoti pavuccati’._  

</Paragraph>

</Division>
<Division number="170" book="ap1">

<Paragraph number="6371" type="verses" book="ap1">
_Idaṃ sutvāna vacanaṃ,_  
_asamassa mahesino;_  
_Āmoditā naramarū,_  
_namassanti katañjalī._  

</Paragraph>

</Division>
<Division number="171" book="ap1">

<Paragraph number="6372" type="verses" book="ap1">
_Tadāhaṃ māṇavo āsiṃ,_  
_megho nāma susikkhito;_  
_Sutvā byākaraṇaṃ seṭṭhaṃ,_  
_sumedhassa mahāmune._  

</Paragraph>

</Division>
<Division number="172" book="ap1">

<Paragraph number="6373" type="verses" book="ap1">
_Saṃvisaṭṭho bhavitvāna,_  
_sumedhe karuṇāsaye;_  
_Pabbajantañca taṃ vīraṃ,_  
_sahāva anupabbajiṃ._  

</Paragraph>

</Division>
<Division number="173" book="ap1">

<Paragraph number="6374" type="verses" book="ap1">
_Saṃvuto pātimokkhasmiṃ,_  
_indriyesu ca pañcasu;_  
_Suddhājīvo sato vīro,_  
_jinasāsanakārako._  

</Paragraph>

</Division>
<Division number="174" book="ap1">

<Paragraph number="6375" type="verses" book="ap1">
_Evaṃ viharamānohaṃ,_  
_pāpamittena kenaci;_  
_Niyojito anācāre,_  
_sumaggā paridhaṃsito._  

</Paragraph>

</Division>
<Division number="175" book="ap1">

<Paragraph number="6376" type="verses" book="ap1">
_Vitakkavasiko hutvā,_  
_sāsanato apakkamiṃ;_  
_Pacchā tena kumittena,_  
_payutto mātughātanaṃ._  

</Paragraph>

</Division>
<Division number="176" book="ap1">

<Paragraph number="6377" type="verses" book="ap1">
_Akariṃ ānantariyaṃ,_  
_ghātayiṃ duṭṭhamānaso;_  
_Tato cuto mahāvīciṃ,_  
_upapanno sudāruṇaṃ._  

</Paragraph>

</Division>
<Division number="177" book="ap1">

<Paragraph number="6378" type="verses" book="ap1">
_Vinipātagato santo,_  
_saṃsariṃ dukkhito ciraṃ;_  
_Na puno addasaṃ vīraṃ,_  
_sumedhaṃ narapuṅgavaṃ._  

</Paragraph>

</Division>
<Division number="178" book="ap1">

<Paragraph number="6379" type="verses" book="ap1">
_Asmiṃ kappe samuddamhi,_  
_maccho āsiṃ timiṅgalo;_  
_Disvāhaṃ sāgare nāvaṃ,_  
_gocaratthamupāgamiṃ._  

</Paragraph>

</Division>
<Division number="179" book="ap1">

<Paragraph number="6380" type="verses" book="ap1">
_Disvā maṃ vāṇijā bhītā,_  
_buddhaseṭṭhamanussaruṃ;_  
_Gotamoti mahāghosaṃ,_  
_sutvā tehi udīritaṃ._  

</Paragraph>

</Division>
<Division number="180" book="ap1">

<Paragraph number="6381" type="verses" book="ap1">
_Pubbasaññaṃ saritvāna,_  
_tato kālaṅkato ahaṃ;_  
_Sāvatthiyaṃ kule iddhe,_  
_jāto brāhmaṇajātiyaṃ._  

</Paragraph>

</Division>
<Division number="181" book="ap1">

<Paragraph number="6382" type="verses" book="ap1">
_Āsiṃ dhammaruci nāma,_  
_sabbapāpajigucchako;_  
_Disvāhaṃ lokapajjotaṃ,_  
_jātiyā sattavassiko._  

</Paragraph>

</Division>
<Division number="182" book="ap1">

<Paragraph number="6383" type="verses" book="ap1">
_Mahājetavanaṃ gantvā,_  
_pabbajiṃ anagāriyaṃ;_  
_Upemi buddhaṃ tikkhattuṃ,_  
_rattiyā divasassa ca._  

</Paragraph>

</Division>
<Division number="183" book="ap1">

<Paragraph number="6384" type="verses" book="ap1">
_Tadā disvā muni āha,_  
_ciraṃ dhammarucīti maṃ;_  
_Tatohaṃ avacaṃ buddhaṃ,_  
_pubbakammapabhāvitaṃ._  

</Paragraph>

</Division>
<Division number="184" book="ap1">

<Paragraph number="6385" type="verses" book="ap1">
_Suciraṃ satapuññalakkhaṇaṃ,_  
_Patipubbena visuddhapaccayaṃ;_  
_Ahamajjasupekkhanaṃ vata,_  
_Tava passāmi nirupamaṃ viggahaṃ._  

</Paragraph>

</Division>
<Division number="185" book="ap1">

<Paragraph number="6386" type="verses" book="ap1">
_Suciraṃ vihatattamo mayā,_  
_Sucirakkhena nadī visositā;_  
_Suciraṃ amalaṃ visodhitaṃ,_  
_Nayanaṃ ñāṇamayaṃ mahāmune._  

</Paragraph>

</Division>
<Division number="186" book="ap1">

<Paragraph number="6387" type="verses" book="ap1">
_Cirakālasamaṅgito tayā,_  
_Avinaṭṭho punarantaraṃ ciraṃ;_  
_Punarajjasamāgato tayā,_  
_Na hi nassanti katāni gotama._  

</Paragraph>

</Division>
<Division number="187" book="ap1">

<Paragraph number="6388" type="verses" book="ap1">
_Kilesā jhāpitā mayhaṃ,_  
_ …pe… _  
_viharāmi anāsavo._  

</Paragraph>

</Division>
<Division number="188" book="ap1">

<Paragraph number="6389" type="verses" book="ap1">
_Svāgataṃ vata me āsi,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ._  

</Paragraph>

</Division>
<Division number="189" book="ap1">

<Paragraph number="6390" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="6391" book="ap1">
Itthaṃ sudaṃ āyasmā dhammaruciyo thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="6392" type="center" book="ap1">
Dhammaruciyattherassāpadānaṃ navamaṃ.
</Paragraph>
