<Division number="43" book="ap1">

<Paragraph number="1428" type="verses" book="ap1">
_“Padumuttarabuddhassa,_  
_piṇḍapātaṃ adāsahaṃ;_  
_Gantvā kiliṭṭhakaṃ ṭhānaṃ,_  
_mallikāhi parikkhitaṃ._  

</Paragraph>

</Division>
<Division number="44" book="ap1">

<Paragraph number="1429" type="verses" book="ap1">
_Tamhāsanamhi āsīno,_  
_buddho lokagganāyako;_  
_Akittayi piṇḍapātaṃ,_  
_ujubhūto samāhito._  

</Paragraph>

</Division>
<Division number="45" book="ap1">

<Paragraph number="1430" type="verses" book="ap1">
_Yathāpi bhaddake khette,_  
_bījaṃ appampi ropitaṃ;_  
_Sammā dhāraṃ pavecchante,_  
_phalaṃ toseti kassakaṃ._  

</Paragraph>

</Division>
<Division number="46" book="ap1">

<Paragraph number="1431" type="verses" book="ap1">
_Tathevāyaṃ piṇḍapāto,_  
_sukhette ropito tayā;_  
_Bhave nibbattamānamhi,_  
_phalaṃ te tosayissati._  

</Paragraph>

</Division>
<Division number="47" book="ap1">

<Paragraph number="1432" type="verses" book="ap1">
_Idaṃ vatvāna sambuddho,_  
_jalajuttamanāmako;_  
_Piṇḍapātaṃ gahetvāna,_  
_pakkāmi uttarāmukho._  

</Paragraph>

</Division>
<Division number="48" book="ap1">

<Paragraph number="1433" type="verses" book="ap1">
_Saṃvuto pātimokkhasmiṃ,_  
_indriyesu ca pañcasu;_  
_Pavivekamanuyutto,_  
_viharāmi anāsavo._  

</Paragraph>

</Division>
<Division number="49" book="ap1">

<Paragraph number="1434" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="1435" book="ap1">
Itthaṃ sudaṃ āyasmā saparivārāsano thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="1436" type="center" book="ap1">
Saparivārāsanattherassāpadānaṃ chaṭṭhaṃ.
</Paragraph>
