<Paragraph number="5851" type="verses" book="ap1">
_“Padumuttarassa bhagavato,_  
_Bodhiyā pādaputtame;_  
_Vedikaṃ sukataṃ katvā,_  
_Sakaṃ cittaṃ pasādayiṃ._  

</Paragraph>

</Division>
<Division number="62" book="ap1" bookNo="20" page="434">

<Paragraph number="5852" type="verses" book="ap1">
_Atoḷārāni bhaṇḍāni,_  
_katāni akatāni ca;_  
_Antalikkhā pavassanti,_  
_vedikāya idaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="63" book="ap1" bookNo="20" page="434">

<Paragraph number="5853" type="verses" book="ap1">
_Ubhato byūḷhasaṅgāme,_  
_pakkhandanto bhayānake;_  
_Bhayabheravaṃ na passāmi,_  
_vedikāya idaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="64" book="ap1" bookNo="20" page="434">

<Paragraph number="5854" type="verses" book="ap1">
_Mama saṅkappamaññāya,_  
_byamhaṃ nibbattate subhaṃ;_  
_Sayanāni mahagghāni,_  
_vedikāya idaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="65" book="ap1" bookNo="20" page="434">

<Paragraph number="5855" type="verses" book="ap1">
_Satasahassito kappe,_  
_yaṃ vedikamakārayiṃ;_  
_Duggatiṃ nābhijānāmi,_  
_vedikāya idaṃ phalaṃ._  

</Paragraph>

</Division>

_ …pe… _  
_viharāmi anāsavo._  

</Paragraph>

</Division>
<Division number="67" book="ap1" bookNo="20" page="434">

<Paragraph number="5857" type="verses" book="ap1">
_Svāgataṃ vata me āsi,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ._  

</Paragraph>

</Division>
<Division number="68" book="ap1" bookNo="20" page="434">

<Paragraph number="5858" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="5859" book="ap1">

</Paragraph>
