<Paragraph number="3450" type="verses" book="ap1">
_“Vipassino bhagavato,_  
_Mahābodhimaho ahu;_  
_Rukkhaṭṭhasseva sambuddho,_  
_Lokajeṭṭho narāsabho._  

</Paragraph>

</Division>
<Division number="14" book="ap1" bookNo="20" page="283">

<Paragraph number="3451" type="verses" book="ap1">
_Bhagavā tamhi samaye,_  
_bhikkhusaṃghapurakkhato;_  
_Catusaccaṃ pakāseti,_  
_vācāsabhimudīrayaṃ._  

</Paragraph>

</Division>
<Division number="15" book="ap1" bookNo="20" page="283">

<Paragraph number="3452" type="verses" book="ap1">
_Saṃkhittena ca desento,_  
_vitthārena ca desayaṃ;_  
_Vivaṭṭacchado sambuddho,_  
_nibbāpesi mahājanaṃ._  

</Paragraph>

</Division>
<Division number="16" book="ap1" bookNo="20" page="283">

<Paragraph number="3453" type="verses" book="ap1">
_Tassāhaṃ dhammaṃ sutvāna,_  
_lokajeṭṭhassa tādino;_  
_Vanditvā satthuno pāde,_  
_pakkāmiṃ uttarāmukho._  

</Paragraph>

</Division>
<Division number="17" book="ap1" bookNo="20" page="283">

<Paragraph number="3454" type="verses" book="ap1">
_Ekanavutito kappe,_  
_yaṃ dhammamasuṇiṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_dhammasavassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="18" book="ap1" bookNo="20" page="284">

<Paragraph number="3455" type="verses" book="ap1">
_Tettiṃsamhi ito kappe,_  
_eko āsiṃ mahīpati;_  
_Sutavā nāma nāmena,_  
_cakkavattī mahabbalo._  

</Paragraph>

</Division>
<Division number="19" book="ap1" bookNo="20" page="284">

<Paragraph number="3456" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="3457" book="ap1">
Itthaṃ sudaṃ āyasmā dhammasaññako thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="3458" type="center" book="ap1">

</Paragraph>
