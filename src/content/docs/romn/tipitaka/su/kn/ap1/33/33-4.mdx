<Paragraph number="3608" type="verses" book="ap1">
_“Jātiyā sattavassohaṃ,_  
_ahosiṃ mantapāragū;_  
_Kulavattaṃ adhāresiṃ,_  
_yañño ussāhito mayā._  

</Paragraph>

</Division>
<Division number="18" book="ap1" bookNo="20" page="297">

<Paragraph number="3609" type="verses" book="ap1">
_Cullāsītisahassāni,_  
_pasū haññanti me tadā;_  
_Sārathambhupanītāni,_  
_yaññatthāya upaṭṭhitā._  

</Paragraph>

</Division>
<Division number="19" book="ap1" bookNo="20" page="297">

<Paragraph number="3610" type="verses" book="ap1">
_Ukkāmukhapahaṭṭhova,_  
_khadiraṅgārasannibho;_  
_Udayantova sūriyo,_  
_puṇṇamāyeva candimā._  

</Paragraph>

</Division>
<Division number="20" book="ap1" bookNo="20" page="297">

<Paragraph number="3611" type="verses" book="ap1">
_Siddhattho sabbasiddhattho,_  
_tilokamahito hito;_  
_Upagantvāna sambuddho,_  
_idaṃ vacanamabravi._  

</Paragraph>

</Division>
<Division number="21" book="ap1" bookNo="20" page="297">

<Paragraph number="3612" type="verses" book="ap1">
_‘Ahiṃsā sabbapāṇīnaṃ,_  
_kumāra mama ruccati;_  
_Theyyā ca aticārā ca,_  
_majjapānā ca ārati._  

</Paragraph>

</Division>
<Division number="22" book="ap1" bookNo="20" page="297">

<Paragraph number="3613" type="verses" book="ap1">
_Rati ca samacariyāya,_  
_bāhusaccaṃ kataññutā;_  
_Diṭṭhe dhamme parattha ca,_  
_dhammā ete pasaṃsiyā._  

</Paragraph>

</Division>
<Division number="23" book="ap1" bookNo="20" page="297">

<Paragraph number="3614" type="verses" book="ap1">
_Ete dhamme bhāvayitvā,_  
_sabbasattahite rato;_  
_Buddhe cittaṃ pasādetvā,_  
_bhāvehi maggamuttamaṃ’._  

</Paragraph>

</Division>
<Division number="24" book="ap1" bookNo="20" page="297">

<Paragraph number="3615" type="verses" book="ap1">
_Idaṃ vatvāna sabbaññū,_  
_lokajeṭṭho narāsabho;_  
_Mamevaṃ anusāsitvā,_  
_vehāsaṃ uggato gato._  

</Paragraph>

</Division>
<Division number="25" book="ap1" bookNo="20" page="297">

<Paragraph number="3616" type="verses" book="ap1">
_Pubbe cittaṃ visodhetvā,_  
_pacchā cittaṃ pasādayiṃ;_  
_Tena cittappasādena,_  
_tusitaṃ upapajjahaṃ._  

</Paragraph>

</Division>
<Division number="26" book="ap1" bookNo="20" page="297">

<Paragraph number="3617" type="verses" book="ap1">
_Catunnavutito kappe,_  
_yadā cittaṃ pasādayiṃ;_  
_Duggatiṃ nābhijānāmi,_  
_buddhasaññāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="27" book="ap1" bookNo="20" page="297">

<Paragraph number="3618" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="3619" book="ap1">
Itthaṃ sudaṃ āyasmā yaññasāmiko thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

Yaññasāmikattherassāpadānaṃ catutthaṃ.
</Paragraph>
