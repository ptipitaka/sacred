<Paragraph number="2992" type="verses" book="ap1">
_“Tivarāyaṃ pure ramme,_  
_naḷakāro ahaṃ tadā;_  
_Siddhatthe lokapajjote,_  
_pasannā janatā tahiṃ._  

</Paragraph>

</Division>
<Division number="15" book="ap1" bookNo="20" page="246">

<Paragraph number="2993" type="verses" book="ap1">
_Pūjatthaṃ lokanāthassa,_  
_kilañjaṃ pariyesati;_  
_Buddhapūjaṃ karontānaṃ,_  
_kilañjaṃ adadiṃ ahaṃ._  

</Paragraph>

</Division>
<Division number="16" book="ap1" bookNo="20" page="246">

<Paragraph number="2994" type="verses" book="ap1">
_Catunnavutito kappe,_  
_yaṃ kammamakariṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_kilañjassa idaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="17" book="ap1" bookNo="20" page="246">

<Paragraph number="2995" type="verses" book="ap1">
_Sattasattatikappamhi,_  
_rājā āsiṃ jaladdharo;_  
_Sattaratanasampanno,_  
_cakkavattī mahabbalo._  

</Paragraph>

</Division>

_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="2997" book="ap1">

</Paragraph>
