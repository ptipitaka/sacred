<Paragraph number="1536" type="verses" book="ap1">
_“Giriduggacaro āsiṃ,_  
_abhijātova kesarī;_  
_Migasaṅghaṃ vadhitvāna,_  
_jīvāmi pabbatantare._  

</Paragraph>

</Division>
<Division number="37" book="ap1" bookNo="20" page="127">

<Paragraph number="1537" type="verses" book="ap1">
_Atthadassī tu bhagavā,_  
_sabbaññū vadataṃ varo;_  
_Mamuddharitukāmo so,_  
_āgacchi pabbatuttamaṃ._  

</Paragraph>

</Division>
<Division number="38" book="ap1" bookNo="20" page="127">

<Paragraph number="1538" type="verses" book="ap1">
_Pasadañca migaṃ hantvā,_  
_bhakkhituṃ samupāgamiṃ;_  
_Bhagavā tamhi samaye,_  
_bhikkhamāno upāgami._  

</Paragraph>

</Division>
<Division number="39" book="ap1" bookNo="20" page="127">

<Paragraph number="1539" type="verses" book="ap1">
_Varamaṃsāni paggayha,_  
_adāsiṃ tassa satthuno;_  
_Anumodi mahāvīro,_  
_nibbāpento mamaṃ tadā._  

</Paragraph>

</Division>
<Division number="40" book="ap1" bookNo="20" page="127">

<Paragraph number="1540" type="verses" book="ap1">
_Tena cittappasādena,_  
_giriduggaṃ pavisiṃ ahaṃ;_  
_Pītiṃ uppādayitvāna,_  
_tattha kālaṅkato ahaṃ._  

</Paragraph>

</Division>
<Division number="41" book="ap1" bookNo="20" page="128">

<Paragraph number="1541" type="verses" book="ap1">
_Etena maṃsadānena,_  
_cittassa paṇidhīhi ca;_  
_Pannarase kappasate,_  
_devaloke ramiṃ ahaṃ._  

</Paragraph>

</Division>
<Division number="42" book="ap1" bookNo="20" page="128">

<Paragraph number="1542" type="verses" book="ap1">
_Avasesesu kappesu,_  
_kusalaṃ cintitaṃ mayā;_  
_Teneva maṃsadānena,_  
_buddhānussaraṇena ca._  

</Paragraph>

</Division>
<Division number="43" book="ap1" bookNo="20" page="128">

<Paragraph number="1543" type="verses" book="ap1">
_Aṭṭhattiṃsamhi kappamhi,_  
_aṭṭha dīghāyunāmakā;_  
_Saṭṭhimhito kappasate,_  
_duve varuṇanāmakā._  

</Paragraph>

</Division>
<Division number="44" book="ap1" bookNo="20" page="128">

<Paragraph number="1544" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="1545" book="ap1">
Itthaṃ sudaṃ āyasmā sucintito thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="1546" type="center" book="ap1">
Sucintitattherassāpadānaṃ chaṭṭhaṃ.
</Paragraph>
