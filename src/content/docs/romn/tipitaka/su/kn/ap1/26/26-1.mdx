<Division number="1" book="ap1">

<Paragraph number="3094" type="verses" book="ap1">
_“Devaloke ṭhito santo,_  
_vipassissa mahesino;_  
_Dhammaṃ suṇitvā mudito,_  
_imaṃ vācaṃ abhāsahaṃ._  

</Paragraph>

</Division>
<Division number="2" book="ap1">

<Paragraph number="3095" type="verses" book="ap1">
_‘Namo te purisājañña,_  
_namo te purisuttama;_  
_Bahujjanaṃ tārayasi,_  
_desento amataṃ padaṃ’._  

</Paragraph>

</Division>
<Division number="3" book="ap1">

<Paragraph number="3096" type="verses" book="ap1">
_Ekanavutito kappe,_  
_yaṃ vācamabhaṇiṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_thomanāya idaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="4" book="ap1">

<Paragraph number="3097" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_vimokkhāpi ca aṭṭhime;_  
_Chaḷabhiññā sacchikatā,_  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="3098" book="ap1">
Itthaṃ sudaṃ āyasmā thomako thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="3099" type="center" book="ap1">
Thomakattherassāpadānaṃ paṭhamaṃ.
</Paragraph>
