<Paragraph number="1805" type="verses" book="ap1">
_“Migaluddo pure āsiṃ,_  
_araññe kānane ahaṃ;_  
_Pāṭaliṃ haritaṃ disvā,_  
_tīṇi pupphāni okiriṃ._  

</Paragraph>

</Division>
<Division number="32" book="ap1">

<Paragraph number="1806" type="verses" book="ap1">
_Patitapattāni gaṇhitvā,_  
_bahi chaḍḍesahaṃ tadā;_  
_Antosuddhaṃ bahisuddhaṃ,_  
_suvimuttaṃ anāsavaṃ._  

</Paragraph>

</Division>
<Division number="33" book="ap1">

<Paragraph number="1807" type="verses" book="ap1">
_Sammukhā viya sambuddhaṃ,_  
_vipassiṃ lokanāyakaṃ;_  
_Pāṭaliṃ abhivādetvā,_  
_tattha kālaṅkato ahaṃ._  

</Paragraph>

</Division>
<Division number="34" book="ap1">

<Paragraph number="1808" type="verses" book="ap1">
_Ekanavutito kappe,_  
_yaṃ bodhimabhipūjayiṃ;_  
_Duggatiṃ nābhijānāmi,_  
_bodhipūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="35" book="ap1">

<Paragraph number="1809" type="verses" book="ap1">
_Samantapāsādikā nāma,_  
_terasāsiṃsu rājino;_  
_Ito tettiṃsakappamhi,_  
_cakkavattī mahabbalā._  

</Paragraph>

</Division>

</Paragraph>

</Paragraph>
