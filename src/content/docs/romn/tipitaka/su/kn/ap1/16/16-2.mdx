<Division number="7" book="ap1">

<Paragraph number="2361" type="verses" book="ap1">
_“Parakammāyane yutto,_  
_aparādhaṃ akāsahaṃ;_  
_Vanantaṃ abhidhāvissaṃ,_  
_bhayaverasamappito._  

</Paragraph>

</Division>
<Division number="8" book="ap1">

<Paragraph number="2362" type="verses" book="ap1">
_Pupphitaṃ pādapaṃ disvā,_  
_piṇḍibandhaṃ sunimmitaṃ;_  
_Tambapupphaṃ gahetvāna,_  
_bodhiyaṃ okiriṃ ahaṃ._  

</Paragraph>

</Division>
<Division number="9" book="ap1">

<Paragraph number="2363" type="verses" book="ap1">
_Sammajjitvāna taṃ bodhiṃ,_  
_pāṭaliṃ pādaputtamaṃ;_  
_Pallaṅkaṃ ābhujitvāna,_  
_bodhimūle upāvisiṃ._  

</Paragraph>

</Division>
<Division number="10" book="ap1">

<Paragraph number="2364" type="verses" book="ap1">
_Gatamaggaṃ gavesantā,_  
_āgacchuṃ mama santikaṃ;_  
_Te ca disvānahaṃ tattha,_  
_āvajjiṃ bodhimuttamaṃ._  

</Paragraph>

</Division>
<Division number="11" book="ap1">

<Paragraph number="2365" type="verses" book="ap1">
_Vanditvāna ahaṃ bodhiṃ,_  
_vippasannena cetasā;_  
_Anekatāle papatiṃ,_  
_giridugge bhayānake._  

</Paragraph>

</Division>
<Division number="12" book="ap1">

<Paragraph number="2366" type="verses" book="ap1">
_Ekanavutito kappe,_  
_yaṃ pupphamabhiropayiṃ;_  
_Duggatiṃ nābhijānāmi,_  
_bodhipūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="13" book="ap1">

<Paragraph number="2367" type="verses" book="ap1">
_Ito ca tatiye kappe,_  
_rājā susaññato ahaṃ;_  
_Sattaratanasampanno,_  
_cakkavattī mahabbalo._  

</Paragraph>

</Division>
<Division number="14" book="ap1">

<Paragraph number="2368" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="2369" book="ap1">
Itthaṃ sudaṃ āyasmā tambapupphiyo thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="2370" type="center" book="ap1">
Tambapupphiyattherassāpadānaṃ dutiyaṃ.
</Paragraph>
