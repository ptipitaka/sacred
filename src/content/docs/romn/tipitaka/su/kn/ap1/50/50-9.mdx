<Division number="94" book="ap1">

<Paragraph number="6540" type="verses" book="ap1">
_Naḷāgāraṃ karitvāna,_  
_tiṇena chādayiṃ ahaṃ;_  
_Caṅkamaṃ sodhayitvāna,_  
_sayambhussa adāsahaṃ._  

</Paragraph>

</Division>
<Division number="95" book="ap1">

<Paragraph number="6541" type="verses" book="ap1">

_cetanāpaṇidhīhi ca;_  
_Jahitvā mānusaṃ dehaṃ,_  
_tāvatiṃsamagacchahaṃ._  

</Paragraph>

</Division>
<Division number="96" book="ap1">

<Paragraph number="6542" type="verses" book="ap1">
_Tattha me sukataṃ byamhaṃ,_  
_naḷakuṭikanimmitaṃ;_  
_Saṭṭhiyojanamubbedhaṃ,_  
_tiṃsayojanavitthataṃ._  

</Paragraph>

</Division>
<Division number="97" book="ap1">

<Paragraph number="6543" type="verses" book="ap1">

_devaloke ramiṃ ahaṃ;_  
_Ekasattatikkhattuñca,_  
_devarajjamakārayiṃ._  

</Paragraph>

</Division>
<Division number="98" book="ap1">

<Paragraph number="6544" type="verses" book="ap1">

_cakkavattī ahosahaṃ;_  
_Padesarajjaṃ vipulaṃ,_  
_gaṇanāto asaṅkhiyaṃ._  

</Paragraph>

</Division>
<Division number="99" book="ap1">

<Paragraph number="6545" type="verses" book="ap1">

_sabbākāravarūpamaṃ;_  
_Yadicchakāhaṃ vihare,_  
_sakyaputtassa sāsane._  

</Paragraph>

</Division>

_Duggatiṃ nābhijānāmi,_  
_naḷakuṭiyidaṃ phalaṃ._  

</Paragraph>

</Division>
