<Paragraph number="5802" type="verses" book="ap1">
_“Suvaṇṇavaṇṇaṃ sambuddhaṃ,_  
_Dvattiṃsavaralakkhaṇaṃ;_  
_Vipinaggena gacchantaṃ,_  
_Sālarājaṃva phullitaṃ._  

</Paragraph>

</Division>
<Division number="23" book="ap1" bookNo="20" page="430">

<Paragraph number="5803" type="verses" book="ap1">
_Tiṇattharaṃ paññāpetvā,_  
_buddhaseṭṭhaṃ ayācahaṃ;_  
_‘Anukampatu maṃ buddho,_  
_bhikkhaṃ icchāmi dātave’._  

</Paragraph>

</Division>
<Division number="24" book="ap1" bookNo="20" page="431">

<Paragraph number="5804" type="verses" book="ap1">
_Anukampako kāruṇiko,_  
_atthadassī mahāyaso;_  
_Mama saṅkappamaññāya,_  
_orūhi mama assame._  

</Paragraph>

</Division>
<Division number="25" book="ap1" bookNo="20" page="431">

<Paragraph number="5805" type="verses" book="ap1">
_Orohitvāna sambuddho,_  
_nisīdi paṇṇasanthare;_  
_Bhallātakaṃ gahetvāna,_  
_buddhaseṭṭhassadāsahaṃ._  

</Paragraph>

</Division>
<Division number="26" book="ap1" bookNo="20" page="431">

<Paragraph number="5806" type="verses" book="ap1">
_Mama nijjhāyamānassa,_  
_paribhuñji tadā jino;_  
_Tattha cittaṃ pasādetvā,_  
_abhivandiṃ tadā jinaṃ._  

</Paragraph>

</Division>
<Division number="27" book="ap1" bookNo="20" page="431">

<Paragraph number="5807" type="verses" book="ap1">
_Aṭṭhārase kappasate,_  
_yaṃ phalamadadiṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_phaladānassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="28" book="ap1" bookNo="20" page="431">

<Paragraph number="5808" type="verses" book="ap1">
_Kilesā jhāpitā mayhaṃ,_  
_ …pe… _  
_viharāmi anāsavo._  

</Paragraph>

</Division>
<Division number="29" book="ap1" bookNo="20" page="431">

<Paragraph number="5809" type="verses" book="ap1">
_Svāgataṃ vata me āsi,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ._  

</Paragraph>

</Division>
<Division number="30" book="ap1" bookNo="20" page="431">

<Paragraph number="5810" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="5811" book="ap1">
Itthaṃ sudaṃ āyasmā bhallātadāyako thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="5812" type="center" book="ap1">
Bhallātadāyakattherassāpadānaṃ catutthaṃ.
</Paragraph>
