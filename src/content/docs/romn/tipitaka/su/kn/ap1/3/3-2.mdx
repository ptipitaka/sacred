<Division number="52" book="ap1">

<Paragraph number="913" type="verses" book="ap1">
_“Padumuttaro nāma jino,_  
_sabbadhammāna pāragū;_  
_Jalitvā aggikkhandhova,_  
_sambuddho parinibbuto._  

</Paragraph>

</Division>
<Division number="53" book="ap1">

<Paragraph number="914" type="verses" book="ap1">
_Mahājanā samāgamma,_  
_pūjayitvā tathāgataṃ;_  
_Citaṃ katvāna sukataṃ,_  
_sarīraṃ abhiropayuṃ._  

</Paragraph>

</Division>
<Division number="54" book="ap1">

<Paragraph number="915" type="verses" book="ap1">
_Sarīrakiccaṃ katvāna,_  
_dhātuṃ tattha samānayuṃ;_  
_Sadevamānusā sabbe,_  
_buddhathūpaṃ akaṃsu te._  

</Paragraph>

</Division>
<Division number="55" book="ap1">

<Paragraph number="916" type="verses" book="ap1">
_Paṭhamā kañcanamayā,_  
_dutiyāsi maṇīmayā;_  
_Tatiyā rūpiyamayā,_  
_catutthī phalikāmayā._  

</Paragraph>

</Division>
<Division number="56" book="ap1">

<Paragraph number="917" type="verses" book="ap1">
_Tathā pañcamiyā bhūmi,_  
_lohitaṅgamayā ahu;_  
_Chaṭṭhā masāragallassa,_  
_sabbaratanamayūpari._  

</Paragraph>

</Division>
<Division number="57" book="ap1">

<Paragraph number="918" type="verses" book="ap1">
_Jaṅghā maṇimayā āsi,_  
_vedikā ratanamayā;_  
_Sabbasoṇṇamayo thūpo,_  
_uddhaṃ yojanamuggato._  

</Paragraph>

</Division>
<Division number="58" book="ap1">

<Paragraph number="919" type="verses" book="ap1">
_Devā tattha samāgantvā,_  
_ekato mantayuṃ tadā;_  
_Mayampi thūpaṃ kassāma,_  
_lokanāthassa tādino._  

</Paragraph>

</Division>
<Division number="59" book="ap1">

<Paragraph number="920" type="verses" book="ap1">
_Dhātu āveṇikā natthi,_  
_sarīraṃ ekapiṇḍitaṃ;_  
_Imamhi buddhathūpamhi,_  
_kassāma kañcukaṃ mayaṃ._  

</Paragraph>

</Division>
<Division number="60" book="ap1">

<Paragraph number="921" type="verses" book="ap1">
_Devā sattahi ratnehi,_  
_aññaṃ vaḍḍhesu yojanaṃ;_  
_Thūpo dviyojanubbedho,_  
_timiraṃ byapahanti so._  

</Paragraph>

</Division>
<Division number="61" book="ap1">

<Paragraph number="922" type="verses" book="ap1">
_Nāgā tattha samāgantvā,_  
_ekato mantayuṃ tadā;_  
_Manussā ceva devā ca,_  
_buddhathūpaṃ akaṃsu te._  

</Paragraph>

</Division>
<Division number="62" book="ap1">

<Paragraph number="923" type="verses" book="ap1">
_Mā no pamattā assumha,_  
_appamattā sadevakā;_  
_Mayampi thūpaṃ kassāma,_  
_lokanāthassa tādino._  

</Paragraph>

</Division>
<Division number="63" book="ap1">

<Paragraph number="924" type="verses" book="ap1">
_Indanīlaṃ mahānīlaṃ,_  
_atho jotirasaṃ maṇiṃ;_  
_Ekato sannipātetvā,_  
_buddhathūpaṃ achādayuṃ._  

</Paragraph>

</Division>
<Division number="64" book="ap1">

<Paragraph number="925" type="verses" book="ap1">
_Sabbaṃ maṇimayaṃ āsi,_  
_tāvatā buddhacetiyaṃ;_  
_Tiyojanasamubbiddhaṃ,_  
_ālokakaraṇaṃ tadā._  

</Paragraph>

</Division>
<Division number="65" book="ap1">

<Paragraph number="926" type="verses" book="ap1">
_Garuḷā ca samāgantvā,_  
_ekato mantayuṃ tadā;_  
_Manussā devā nāgā ca,_  
_buddhathūpaṃ akaṃsu te._  

</Paragraph>

</Division>
<Division number="66" book="ap1">

<Paragraph number="927" type="verses" book="ap1">
_‘Mā no pamattā assumha,_  
_appamattā sadevakā;_  
_Mayampi thūpaṃ kassāma,_  
_lokanāthassa tādino’._  

</Paragraph>

</Division>
<Division number="67" book="ap1">

<Paragraph number="928" type="verses" book="ap1">
_Sabbaṃ maṇimayaṃ thūpaṃ,_  
_akaruṃ te ca kañcukaṃ;_  
_Yojanaṃ tepi vaḍḍhesuṃ,_  
_āyataṃ buddhacetiyaṃ._  

</Paragraph>

</Division>
<Division number="68" book="ap1">

<Paragraph number="929" type="verses" book="ap1">
_Catuyojanamubbiddho,_  
_buddhathūpo virocati;_  
_Obhāseti disā sabbā,_  
_sataraṃsīva uggato._  

</Paragraph>

</Division>
<Division number="69" book="ap1">

<Paragraph number="930" type="verses" book="ap1">
_Kumbhaṇḍā ca samāgantvā,_  
_ekato mantayuṃ tadā;_  
_Manussā ceva devā ca,_  
_nāgā ca garuḷā tathā;_  
_Paccekaṃ buddhaseṭṭhassa,_  
_akaṃsu thūpamuttamaṃ._  

</Paragraph>

</Division>
<Division number="70" book="ap1">

<Paragraph number="931" type="verses" book="ap1">
_‘Mā no pamattā assumha,_  
_appamattā sadevakā;_  
_Mayampi thūpaṃ kassāma,_  
_lokanāthassa tādino;_  
_Ratanehi chādessāma,_  
_āyataṃ buddhacetiyaṃ’._  

</Paragraph>

</Division>
<Division number="71" book="ap1">

<Paragraph number="932" type="verses" book="ap1">
_Yojanaṃ tepi vaḍḍhesuṃ,_  
_āyataṃ buddhacetiyaṃ;_  
_Pañcayojanamubbiddho,_  
_thūpo obhāsate tadā._  

</Paragraph>

</Division>
<Division number="72" book="ap1">

<Paragraph number="933" type="verses" book="ap1">
_Yakkhā tattha samāgantvā,_  
_ekato mantayuṃ tadā;_  
_Manussā devā nāgā ca,_  
_garuḷā kumbhaaṇḍakā._  

</Paragraph>

</Division>
<Division number="73" book="ap1">

<Paragraph number="934" type="verses" book="ap1">
_Paccekaṃ buddhaseṭṭhassa,_  
_akaṃsu thūpamuttamaṃ;_  
_‘Mā no pamattā assumha,_  
_appamattā sadevakā._  

</Paragraph>

</Division>
<Division number="74" book="ap1">

<Paragraph number="935" type="verses" book="ap1">
_Mayampi thūpaṃ kassāma,_  
_lokanāthassa tādino;_  
_Phalikāhi chādessāma,_  
_āyataṃ buddhacetiyaṃ’._  

</Paragraph>

</Division>
<Division number="75" book="ap1">

<Paragraph number="936" type="verses" book="ap1">
_Yojanaṃ tepi vaḍḍhesuṃ,_  
_āyataṃ buddhacetiyaṃ;_  
_Cha yojanāni ubbiddho,_  
_thūpo obhāsate tadā._  

</Paragraph>

</Division>
<Division number="76" book="ap1">

<Paragraph number="937" type="verses" book="ap1">
_Gandhabbā ca samāgantvā,_  
_ekato mantayuṃ tadā;_  
_‘Manujā devatā nāgā,_  
_garuḷā kumbhayakkhakā._  

</Paragraph>

</Division>
<Division number="77" book="ap1">

<Paragraph number="938" type="verses" book="ap1">
_Sabbekaṃsu buddhathūpaṃ,_  
_mayamettha akārakā;_  
_Mayampi thūpaṃ kassāma,_  
_lokanāthassa tādino’._  

</Paragraph>

</Division>
<Division number="78" book="ap1">

<Paragraph number="939" type="verses" book="ap1">
_Vediyo satta katvāna,_  
_chattamāropayiṃsu te;_  
_Sabbasoṇṇamayaṃ thūpaṃ,_  
_gandhabbā kārayuṃ tadā._  

</Paragraph>

</Division>
<Division number="79" book="ap1">

<Paragraph number="940" type="verses" book="ap1">
_Sattayojanamubbiddho,_  
_thūpo obhāsate tadā;_  
_Rattindivā na ñāyanti,_  
_āloko hoti sabbadā._  

</Paragraph>

</Division>
<Division number="80" book="ap1">

<Paragraph number="941" type="verses" book="ap1">
_Abhibhonti na tassābhā,_  
_candasūrā satārakā;_  
_Samantā yojanasate,_  
_padīpopi na pajjali._  

</Paragraph>

</Division>
<Division number="81" book="ap1">

<Paragraph number="942" type="verses" book="ap1">
_Tena kālena ye keci,_  
_thūpaṃ pūjenti mānusā;_  
_Na te thūpamāruhanti,_  
_ambare ukkhipanti te._  

</Paragraph>

</Division>
<Division number="82" book="ap1">

<Paragraph number="943" type="verses" book="ap1">
_Devehi ṭhapito yakkho,_  
_abhisammatanāmako;_  
_Dhajaṃ vā pupphadāmaṃ vā,_  
_abhiropeti uttari._  

</Paragraph>

</Division>
<Division number="83" book="ap1">

<Paragraph number="944" type="verses" book="ap1">
_Na te passanti taṃ yakkhaṃ,_  
_dāmaṃ passanti gacchato;_  
_Evaṃ passitvā gacchantā,_  
_sabbe gacchanti suggatiṃ._  

</Paragraph>

</Division>
<Division number="84" book="ap1">

<Paragraph number="945" type="verses" book="ap1">
_Viruddhā ye pāvacane,_  
_pasannā ye ca sāsane;_  
_Pāṭiheraṃ daṭṭhukāmā,_  
_thūpaṃ pūjenti mānusā._  

</Paragraph>

</Division>
<Division number="85" book="ap1">

<Paragraph number="946" type="verses" book="ap1">
_Nagare haṃsavatiyā,_  
_ahosiṃ bhatako tadā;_  
_Āmoditaṃ janaṃ disvā,_  
_evaṃ cintesahaṃ tadā._  

</Paragraph>

</Division>
<Division number="86" book="ap1">

<Paragraph number="947" type="verses" book="ap1">
_‘Uḷāro bhagavā heso,_  
_yassa dhātudharedisaṃ;_  
_Imā ca janatā tuṭṭhā,_  
_kāraṃ kubbaṃ na tappare._  

</Paragraph>

</Division>
<Division number="87" book="ap1">

<Paragraph number="948" type="verses" book="ap1">
_Ahampi kāraṃ kassāmi,_  
_lokanāthassa tādino;_  
_Tassa dhammesu dāyādo,_  
_bhavissāmi anāgate’._  

</Paragraph>

</Division>
<Division number="88" book="ap1">

<Paragraph number="949" type="verses" book="ap1">
_Sudhotaṃ rajakenāhaṃ,_  
_uttareyyapaṭaṃ mama;_  
_Veḷagge ālaggetvāna,_  
_dhajaṃ ukkhipimambare._  

</Paragraph>

</Division>
<Division number="89" book="ap1">

<Paragraph number="950" type="verses" book="ap1">
_Abhisammatako gayha,_  
_ambarehāsi me dhajaṃ;_  
_Vāteritaṃ dhajaṃ disvā,_  
_bhiyyo hāsaṃ janesahaṃ._  

</Paragraph>

</Division>
<Division number="90" book="ap1">

<Paragraph number="951" type="verses" book="ap1">
_Tattha cittaṃ pasādetvā,_  
_samaṇaṃ upasaṅkamiṃ;_  
_Taṃ bhikkhuṃ abhivādetvā,_  
_vipākaṃ pucchahaṃ dhaje._  

</Paragraph>

</Division>
<Division number="91" book="ap1">

<Paragraph number="952" type="verses" book="ap1">
_So me kathesi ānanda,_  
_pītisañjananaṃ mama;_  
_‘Tassa dhajassa vipākaṃ,_  
_anubhossasi sabbadā._  

</Paragraph>

</Division>
<Division number="92" book="ap1">

<Paragraph number="953" type="verses" book="ap1">
_Hatthī assā rathā pattī,_  
_senā ca caturaṅginī;_  
_Parivāressanti taṃ niccaṃ,_  
_dhajadānassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="93" book="ap1">

<Paragraph number="954" type="verses" book="ap1">
_Saṭṭhitūriyasahassāni,_  
_bheriyo samalaṅkatā;_  
_Parivāressanti taṃ niccaṃ,_  
_dhajadānassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="94" book="ap1">

<Paragraph number="955" type="verses" book="ap1">
_Chaḷāsītisahassāni,_  
_nāriyo samalaṅkatā;_  
_Vicittavatthābharaṇā,_  
_āmuttamaṇikuṇḍalā._  

</Paragraph>

</Division>
<Division number="95" book="ap1">

<Paragraph number="956" type="verses" book="ap1">
_Aḷārapamhā hasulā,_  
_susaññā tanumajjhimā;_  
_Parivāressanti taṃ niccaṃ,_  
_dhajadānassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="96" book="ap1">

<Paragraph number="957" type="verses" book="ap1">
_Tiṃsakappasahassāni,_  
_devaloke ramissasi;_  
_Asītikkhattuṃ devindo,_  
_devarajjaṃ karissasi._  

</Paragraph>

</Division>
<Division number="97" book="ap1">

<Paragraph number="958" type="verses" book="ap1">
_Sahassakkhattuṃ rājā ca,_  
_cakkavattī bhavissasi;_  
_Padesarajjaṃ vipulaṃ,_  
_gaṇanāto asaṅkhiyaṃ._  

</Paragraph>

</Division>
<Division number="98" book="ap1">

<Paragraph number="959" type="verses" book="ap1">
_Kappasatasahassamhi,_  
_okkākakulasambhavo;_  
_Gotamo nāma gottena,_  
_satthā loke bhavissati._  

</Paragraph>

</Division>
<Division number="99" book="ap1">

<Paragraph number="960" type="verses" book="ap1">
_Devalokā cavitvāna,_  
_sukkamūlena codito;_  
_Puññakammena saṃyutto,_  
_brahmabandhu bhavissasi._  

</Paragraph>

</Division>
<Division number="100" book="ap1">

<Paragraph number="961" type="verses" book="ap1">
_Asītikoṭiṃ chaḍḍetvā,_  
_dāse kammakare bahū;_  
_Gotamassa bhagavato,_  
_sāsane pabbajissasi._  

</Paragraph>

</Division>
<Division number="101" book="ap1">

<Paragraph number="962" type="verses" book="ap1">
_Ārādhayitvā sambuddhaṃ,_  
_gotamaṃ sakyapuṅgavaṃ;_  
_Upavānoti nāmena,_  
_hessasi satthu sāvako’._  

</Paragraph>

</Division>
<Division number="102" book="ap1">

<Paragraph number="963" type="verses" book="ap1">
_Satasahasse kataṃ kammaṃ,_  
_phalaṃ dassesi me idha;_  
_Sumutto saravegova,_  
_kilese jhāpayī mama._  

</Paragraph>

</Division>
<Division number="103" book="ap1">

<Paragraph number="964" type="verses" book="ap1">
_Cakkavattissa santassa,_  
_catudīpissarassa me;_  
_Tiyojanāni samantā,_  
_ussīsanti dhajā sadā._  

</Paragraph>

</Division>
<Division number="104" book="ap1">

<Paragraph number="965" type="verses" book="ap1">
_Satasahassito kappe,_  
_yaṃ kammamakariṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_dhajadānassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="105" book="ap1">

<Paragraph number="966" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_vimokkhāpi ca aṭṭhime;_  
_Chaḷabhiññā sacchikatā,_  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="967" book="ap1">
Itthaṃ sudaṃ āyasmā upavāno thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="968" type="center" book="ap1">
Upavānattherassāpadānaṃ dutiyaṃ.
