<Division number="1" book="ap1">

<Paragraph number="2899" type="verses" book="ap1">
_“Atthadassissa bhagavato,_  
_lokajeṭṭhassa tādino;_  
_Ālambaṇaṃ mayā dinnaṃ,_  
_dvipadindassa tādino._  

</Paragraph>

</Division>
<Division number="2" book="ap1">

<Paragraph number="2900" type="verses" book="ap1">
_Dharaṇiṃ paṭipajjāmi,_  
_vipulaṃ sāgarapparaṃ;_  
_Pāṇesu ca issariyaṃ,_  
_vattemi vasudhāya ca._  

</Paragraph>

</Division>
<Division number="3" book="ap1">

<Paragraph number="2901" type="verses" book="ap1">
_Kilesā jhāpitā mayhaṃ,_  
_bhavā sabbe samūhatā;_  
_Tisso vijjā anuppattā,_  
_kataṃ buddhassa sāsanaṃ._  

</Paragraph>

</Division>
<Division number="4" book="ap1">

<Paragraph number="2902" type="verses" book="ap1">
_Ito dvesaṭṭhikappamhi,_  
_tayo āsiṃsu khattiyā;_  
_Ekāpassitanāmā te,_  
_cakkavattī mahabbalā._  

</Paragraph>

</Division>
<Division number="5" book="ap1">

<Paragraph number="2903" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_vimokkhāpi ca aṭṭhime;_  
_Chaḷabhiññā sacchikatā,_  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="2904" book="ap1">
Itthaṃ sudaṃ āyasmā ālambaṇadāyako thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="2905" type="center" book="ap1">
Ālambaṇadāyakattherassāpadānaṃ paṭhamaṃ.
</Paragraph>
