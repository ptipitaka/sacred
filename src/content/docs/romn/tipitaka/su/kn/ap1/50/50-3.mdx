<Paragraph number="6448" type="verses" book="ap1">
_“Vanakammiko pure āsiṃ,_  
_pitumātumatenahaṃ;_  
_Pasumārena jīvāmi,_  
_kusalaṃ me na vijjati._  

</Paragraph>

</Division>
<Division number="16" book="ap1">

<Paragraph number="6449" type="verses" book="ap1">
_Mama āsayasāmantā,_  
_tisso lokagganāyako;_  
_Padāni tīṇi dassesi,_  
_anukampāya cakkhumā._  

</Paragraph>

</Division>
<Division number="17" book="ap1">

<Paragraph number="6450" type="verses" book="ap1">
_Akkante ca pade disvā,_  
_tissanāmassa satthuno;_  
_Haṭṭho haṭṭhena cittena,_  
_pade cittaṃ pasādayiṃ._  

</Paragraph>

</Division>
<Division number="18" book="ap1">

<Paragraph number="6451" type="verses" book="ap1">
_Koraṇḍaṃ pupphitaṃ disvā,_  
_pādapaṃ dharaṇīruhaṃ;_  
_Sakosakaṃ gahetvāna,_  
_padaseṭṭhamapūjayiṃ._  

</Paragraph>

</Division>
<Division number="19" book="ap1">

<Paragraph number="6452" type="verses" book="ap1">
_Tena kammena sukatena,_  
_cetanāpaṇidhīhi ca;_  
_Jahitvā mānusaṃ dehaṃ,_  
_tāvatiṃsamagacchahaṃ._  

</Paragraph>

</Division>
<Division number="20" book="ap1">

<Paragraph number="6453" type="verses" book="ap1">
_Yaṃ yaṃ yonupapajjāmi,_  
_devattaṃ atha mānusaṃ;_  
_Koraṇḍakachavi homi,_  
_suppabhāso bhavāmahaṃ._  

</Paragraph>

</Division>
<Division number="21" book="ap1">

<Paragraph number="6454" type="verses" book="ap1">
_Dvenavute ito kappe,_  
_yaṃ kammamakariṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_padapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="22" book="ap1">

<Paragraph number="6455" type="verses" book="ap1">
_Kilesā jhāpitā mayhaṃ,_  
_ …pe… _  
_viharāmi anāsavo._  

</Paragraph>

</Division>
<Division number="23" book="ap1">

<Paragraph number="6456" type="verses" book="ap1">
_Svāgataṃ vata me āsi,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ._  

</Paragraph>

</Division>
<Division number="24" book="ap1">

<Paragraph number="6457" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="6458" book="ap1">
Itthaṃ sudaṃ āyasmā koraṇḍapupphiyo thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="6459" type="center" book="ap1">
Koraṇḍapupphiyattherassāpadānaṃ tatiyaṃ.
</Paragraph>
