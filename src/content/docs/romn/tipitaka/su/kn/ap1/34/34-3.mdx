<Paragraph number="3732" type="verses" book="ap1">
_“Kānanaṃ vanamogayha,_  
_vasāmi luddako ahaṃ;_  
_Punnāgaṃ pupphitaṃ disvā,_  
_buddhaseṭṭhaṃ anussariṃ._  

</Paragraph>

</Division>
<Division number="15" book="ap1" bookNo="20" page="306">

<Paragraph number="3733" type="verses" book="ap1">
_Taṃ pupphaṃ ocinitvāna,_  
_sugandhaṃ gandhitaṃ subhaṃ;_  
_Thūpaṃ katvāna puline,_  
_buddhassa abhiropayiṃ._  

</Paragraph>

</Division>
<Division number="16" book="ap1" bookNo="20" page="306">

<Paragraph number="3734" type="verses" book="ap1">
_Dvenavute ito kappe,_  
_yaṃ pupphamabhipūjayiṃ;_  
_Duggatiṃ nābhijānāmi,_  
_buddhapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="17" book="ap1" bookNo="20" page="306">

<Paragraph number="3735" type="verses" book="ap1">
_Ekamhi navute kappe,_  
_eko āsiṃ tamonudo;_  
_Sattaratanasampanno,_  
_cakkavattī mahabbalo._  

</Paragraph>

</Division>
<Division number="18" book="ap1" bookNo="20" page="306">

<Paragraph number="3736" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="3737" book="ap1">
Itthaṃ sudaṃ āyasmā punnāgapupphiyo thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="3738" type="center" book="ap1">
Punnāgapupphiyattherassāpadānaṃ tatiyaṃ.
</Paragraph>
