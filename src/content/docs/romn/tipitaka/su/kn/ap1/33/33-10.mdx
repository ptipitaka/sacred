_“Padumuttaro lokavidū,_  
_āhutīnaṃ paṭiggaho;_  
_Ākāse jalavuṭṭhīva,_  
_vassate dhammavuṭṭhiyā._  

</Paragraph>

</Division>
<Division number="83" book="ap1">

<Paragraph number="3687" type="verses" book="ap1">
_Tamaddasāsiṃ sambuddhaṃ,_  
_desentaṃ amataṃ padaṃ;_  
_Sakaṃ cittaṃ pasādetvā,_  
_agamāsiṃ sakaṃ gharaṃ._  

</Paragraph>

</Division>
<Division number="84" book="ap1">

<Paragraph number="3688" type="verses" book="ap1">
_Chattaṃ alaṅkataṃ gayha,_  
_upagacchiṃ naruttamaṃ;_  
_Haṭṭho haṭṭhena cittena,_  
_ākāse ukkhipiṃ ahaṃ._  

</Paragraph>

</Division>
<Division number="85" book="ap1">

<Paragraph number="3689" type="verses" book="ap1">
_Susaṅgahitayānaṃva,_  
_dantova sāvakuttamo;_  
_Upagantvāna sambuddhaṃ,_  
_matthake sampatiṭṭhahi._  

</Paragraph>

</Division>
<Division number="86" book="ap1">

<Paragraph number="3690" type="verses" book="ap1">
_Anukampako kāruṇiko,_  
_buddho lokagganāyako;_  
_Bhikkhusaṃghe nisīditvā,_  
_imā gāthā abhāsatha._  

</Paragraph>

</Division>
<Division number="87" book="ap1">

<Paragraph number="3691" type="verses" book="ap1">
_‘Yena chattamidaṃ dinnaṃ,_  
_alaṅkataṃ manoramaṃ;_  
_Tena cittappasādena,_  
_duggatiṃ so na gacchati._  

</Paragraph>

</Division>
<Division number="88" book="ap1">

<Paragraph number="3692" type="verses" book="ap1">
_Sattakkhattuñca devesu,_  
_devarajjaṃ karissati;_  
_Bāttiṃsakkhattuñca rājā,_  
_cakkavattī bhavissati._  

</Paragraph>

</Division>
<Division number="89" book="ap1">

<Paragraph number="3693" type="verses" book="ap1">
_Kappasatasahassamhi,_  
_okkākakulasambhavo;_  
_Gotamo nāma gottena,_  
_satthā loke bhavissati._  

</Paragraph>

</Division>
<Division number="90" book="ap1">

<Paragraph number="3694" type="verses" book="ap1">
_Tassa dhammesu dāyādo,_  
_oraso dhammanimmito;_  
_Sabbāsave pariññāya,_  
_nibbāyissatināsavo’._  

</Paragraph>

</Division>
<Division number="91" book="ap1">

<Paragraph number="3695" type="verses" book="ap1">
_Buddhassa giramaññāya,_  
_vācāsabhimudīritaṃ;_  
_Pasannacitto sumano,_  
_bhiyyo hāsaṃ janesahaṃ._  

</Paragraph>

</Division>
<Division number="92" book="ap1">

<Paragraph number="3696" type="verses" book="ap1">
_Jahitvā mānusaṃ yoniṃ,_  
_Dibbaṃ yoniṃ samajjhagaṃ;_  
_Vimānamuttamaṃ mayhaṃ,_  
_Abbhuggataṃ manoramaṃ._  

</Paragraph>

</Division>
<Division number="93" book="ap1">

<Paragraph number="3697" type="verses" book="ap1">
_Vimānā nikkhamantassa,_  
_setacchattaṃ dharīyati;_  
_Tadā saññaṃ paṭilabhiṃ,_  
_pubbakammassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="94" book="ap1">

<Paragraph number="3698" type="verses" book="ap1">
_Devalokā cavitvāna,_  
_manussattañca āgamiṃ;_  
_Chattiṃsakkhattuṃ cakkavattī,_  
_sattakappasatamhito._  

</Paragraph>

</Division>
<Division number="95" book="ap1">

<Paragraph number="3699" type="verses" book="ap1">
_Tamhā kāyā cavitvāna,_  
_āgacchiṃ tidasaṃ puraṃ;_  
_Saṃsaritvānupubbena,_  
_mānusaṃ punarāgamiṃ._  

</Paragraph>

</Division>
<Division number="96" book="ap1">

<Paragraph number="3700" type="verses" book="ap1">
_Okkantaṃ mātukucchiṃ maṃ,_  
_setacchattaṃ adhārayuṃ;_  
_Jātiyā sattavassohaṃ,_  
_pabbajiṃ anagāriyaṃ._  

</Paragraph>

</Division>
<Division number="97" book="ap1">

<Paragraph number="3701" type="verses" book="ap1">
_Sunando nāma nāmena,_  
_brāhmaṇo mantapāragū;_  
_Phalikaṃ chattamādāya,_  
_sāvakaggassa so tadā._  

</Paragraph>

</Division>
<Division number="98" book="ap1">

<Paragraph number="3702" type="verses" book="ap1">
_Anumodi mahāvīro,_  
_sāriputto mahākathī;_  
_Sutvānumodanaṃ tassa,_  
_pubbakammamanussariṃ._  

</Paragraph>

</Division>
<Division number="99" book="ap1">

<Paragraph number="3703" type="verses" book="ap1">
_Añjaliṃ paggahetvāna,_  
_sakaṃ cittaṃ pasādayiṃ;_  
_Saritvā purimaṃ kammaṃ,_  
_arahattamapāpuṇiṃ._  

</Paragraph>

</Division>
<Division number="100" book="ap1">

<Paragraph number="3704" type="verses" book="ap1">
_Uṭṭhāya āsanā tamhā,_  
_sire katvāna añjaliṃ;_  
_Sambuddhaṃ abhivādetvā,_  
_imaṃ vācaṃ udīrayiṃ._  

</Paragraph>

</Division>
<Division number="101" book="ap1">

<Paragraph number="3705" type="verses" book="ap1">
_Satasahassito kappe,_  
_buddho loke anuttaro;_  
_Padumuttaro lokavidū,_  
_āhutīnaṃ paṭiggaho._  

</Paragraph>

</Division>
<Division number="102" book="ap1">

<Paragraph number="3706" type="verses" book="ap1">
_Tassa chattaṃ mayā dinnaṃ,_  
_vicittaṃ samalaṅkataṃ;_  
_Ubho hatthehi paggaṇhi,_  
_sayambhū aggapuggalo._  

</Paragraph>

</Division>
<Division number="103" book="ap1">

<Paragraph number="3707" type="verses" book="ap1">
_Aho buddhā aho dhammā,_  
_aho no satthusampadā;_  
_Ekacchattassa dānena,_  
_duggatiṃ nupapajjahaṃ._  

</Paragraph>

</Division>
<Division number="104" book="ap1">

_bhavā sabbe samūhatā;_  
_Sabbāsave pariññāya,_  
_viharāmi anāsavo._  

</Paragraph>

</Division>
