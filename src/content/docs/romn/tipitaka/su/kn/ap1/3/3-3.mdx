<Paragraph number="969" type="verses" book="ap1">
_“Nagare candavatiyā,_  
_mātuupaṭṭhāko ahuṃ;_  
_Andhā mātā pitā mayhaṃ,_  
_te posemi ahaṃ tadā._  

</Paragraph>

</Division>
<Division number="107" book="ap1" bookNo="20" page="82">

<Paragraph number="970" type="verses" book="ap1">
_Rahogato nisīditvā,_  
_evaṃ cintesahaṃ tadā;_  
_Posento mātāpitaro,_  
_pabbajjaṃ na labhāmahaṃ._  

</Paragraph>

</Division>
<Division number="108" book="ap1" bookNo="20" page="82">

<Paragraph number="971" type="verses" book="ap1">
_Mahandhakārapihitā,_  
_tividhaggīhi ḍayhare;_  
_Etādise bhave jāte,_  
_natthi koci vināyako._  

</Paragraph>

</Division>
<Division number="109" book="ap1" bookNo="20" page="82">

<Paragraph number="972" type="verses" book="ap1">
_Buddho loke samuppanno,_  
_dippati dāni sāsanaṃ;_  
_Sakkā uddharituṃ attā,_  
_puññakāmena jantunā._  

</Paragraph>

</Division>
<Division number="110" book="ap1" bookNo="20" page="82">

<Paragraph number="973" type="verses" book="ap1">
_Uggayha tīṇi saraṇe,_  
_paripuṇṇāni gopayiṃ;_  
_Tena kammena sukatena,_  
_paṭimokkhāmi duggatiṃ._  

</Paragraph>

</Division>
<Division number="111" book="ap1" bookNo="20" page="83">

<Paragraph number="974" type="verses" book="ap1">
_Nisabho nāma samaṇo,_  
_buddhassa aggasāvako;_  
_Tamahaṃ upagantvāna,_  
_saraṇagamanaṃ gahiṃ._  

</Paragraph>

</Division>
<Division number="112" book="ap1" bookNo="20" page="83">

<Paragraph number="975" type="verses" book="ap1">
_Vassasatasahassāni,_  
_āyu vijjati tāvade;_  
_Tāvatā saraṇagamanaṃ,_  
_paripuṇṇaṃ agopayiṃ._  

</Paragraph>

</Division>
<Division number="113" book="ap1" bookNo="20" page="83">

<Paragraph number="976" type="verses" book="ap1">
_Carime vattamānamhi,_  
_saraṇaṃ taṃ anussariṃ;_  
_Tena kammena sukatena,_  
_tāvatiṃsaṃ agacchahaṃ._  

</Paragraph>

</Division>
<Division number="114" book="ap1" bookNo="20" page="83">

<Paragraph number="977" type="verses" book="ap1">
_Devalokagato santo,_  
_puññakammasamāhito;_  
_Yaṃ desaṃ upapajjāmi,_  
_aṭṭha hetū labhāmahaṃ._  

</Paragraph>

</Division>
<Division number="115" book="ap1" bookNo="20" page="83">

<Paragraph number="978" type="verses" book="ap1">
_Disāsu pūjito homi,_  
_tikkhapañño bhavāmahaṃ;_  
_Sabbe devānuvattanti,_  
_amitabhogaṃ labhāmahaṃ._  

</Paragraph>

</Division>
<Division number="116" book="ap1" bookNo="20" page="83">

<Paragraph number="979" type="verses" book="ap1">
_Suvaṇṇavaṇṇo sabbattha,_  
_paṭikanto bhavāmahaṃ;_  
_Mittānaṃ acalo homi,_  
_yaso abbhuggato mamaṃ._  

</Paragraph>

</Division>
<Division number="117" book="ap1" bookNo="20" page="83">

<Paragraph number="980" type="verses" book="ap1">
_Asītikkhattu devindo,_  
_devarajjamakārayiṃ;_  
_Dibbasukhaṃ anubhaviṃ,_  
_accharāhi purakkhato._  

</Paragraph>

</Division>
<Division number="118" book="ap1" bookNo="20" page="83">

<Paragraph number="981" type="verses" book="ap1">
_Pañcasattatikkhattuñca,_  
_cakkavattī ahosahaṃ;_  
_Padesarajjaṃ vipulaṃ,_  
_gaṇanāto asaṅkhiyaṃ._  

</Paragraph>

</Division>
<Division number="119" book="ap1" bookNo="20" page="83">

<Paragraph number="982" type="verses" book="ap1">
_Pacchime bhave sampatte,_  
_puññakammasamāhito;_  
_Pure sāvatthiyaṃ jāto,_  
_mahāsāle suaḍḍhake._  

</Paragraph>

</Division>
<Division number="120" book="ap1" bookNo="20" page="83">

<Paragraph number="983" type="verses" book="ap1">
_Nagarā nikkhamitvāna,_  
_dārakehi purakkhato;_  
_Hasakhiḍḍasamaṅgīhaṃ,_  
_saṃghārāmaṃ upāgamiṃ._  

</Paragraph>

</Division>
<Division number="121" book="ap1" bookNo="20" page="83">

<Paragraph number="984" type="verses" book="ap1">
_Tatthaddasāsiṃ samaṇaṃ,_  
_vippamuttaṃ nirūpadhiṃ;_  
_So me dhammamadesesi,_  
_saraṇañca adāsi me._  

</Paragraph>

</Division>
<Division number="122" book="ap1" bookNo="20" page="83">

<Paragraph number="985" type="verses" book="ap1">
_Sohaṃ sutvāna saraṇaṃ,_  
_saraṇaṃ me anussariṃ;_  
_Ekāsane nisīditvā,_  
_arahattamapāpuṇiṃ._  

</Paragraph>

</Division>
<Division number="123" book="ap1" bookNo="20" page="83">

<Paragraph number="986" type="verses" book="ap1">
_Jātiyā sattame vasse,_  
_arahattamapāpuṇiṃ;_  
_Upasampādayi buddho,_  
_guṇamaññāya cakkhumā._  

</Paragraph>

</Division>
<Division number="124" book="ap1" bookNo="20" page="84">

<Paragraph number="987" type="verses" book="ap1">
_Aparimeyye ito kappe,_  
_saraṇāni agacchahaṃ;_  
_Tato me sukataṃ kammaṃ,_  
_phalaṃ dassesi me idha._  

</Paragraph>

</Division>
<Division number="125" book="ap1" bookNo="20" page="84">

<Paragraph number="988" type="verses" book="ap1">
_Sugopitaṃ me saraṇaṃ,_  
_mānasaṃ suppaṇīhitaṃ;_  
_Anubhotvā yasaṃ sabbaṃ,_  
_pattomhi acalaṃ padaṃ._  

</Paragraph>

</Division>
<Division number="126" book="ap1" bookNo="20" page="84">

<Paragraph number="989" type="verses" book="ap1">
_Yesaṃ sotāvadhānatthi,_  
_suṇotha mama bhāsato;_  
_Ahaṃ vo kathayissāmi,_  
_sāmaṃ diṭṭhaṃ padaṃ mama._  

</Paragraph>

</Division>
<Division number="127" book="ap1" bookNo="20" page="84">

<Paragraph number="990" type="verses" book="ap1">
_‘Buddho loke samuppanno,_  
_vattate jinasāsanaṃ;_  
_Amatā vāditā bherī,_  
_sokasallavinodanā._  

</Paragraph>

</Division>
<Division number="128" book="ap1" bookNo="20" page="84">

<Paragraph number="991" type="verses" book="ap1">
_Yathāsakena thāmena,_  
_puññakkhette anuttare;_  
_Adhikāraṃ kareyyātha,_  
_passayissatha nibbutiṃ._  

</Paragraph>

</Division>
<Division number="129" book="ap1" bookNo="20" page="84">

<Paragraph number="992" type="verses" book="ap1">
_Paggayha tīṇi saraṇe,_  
_pañcasīlāni gopiya;_  
_Buddhe cittaṃ pasādetvā,_  
_dukkhassantaṃ karissatha._  

</Paragraph>

</Division>
<Division number="130" book="ap1" bookNo="20" page="84">

<Paragraph number="993" type="verses" book="ap1">
_Sammā dhammaṃ bhāvetvāna,_  
_Sīlāni parigopiya;_  
_Aciraṃ arahattaṃ vo,_  
_Sabbepi pāpuṇissatha._  

</Paragraph>

</Division>
<Division number="131" book="ap1" bookNo="20" page="84">

<Paragraph number="994" type="verses" book="ap1">
_Tevijjo iddhipattomhi,_  
_cetopariyakovido;_  
_Sāvako te mahāvīra,_  
_saraṇo vandati satthuno’._  

</Paragraph>

</Division>
<Division number="132" book="ap1" bookNo="20" page="84">

<Paragraph number="995" type="verses" book="ap1">
_Aparimeyye ito kappe,_  
_saraṇaṃ buddhassa gacchahaṃ;_  
_Duggatiṃ nābhijānāmi,_  
_saraṇaṃ gamane phalaṃ._  

</Paragraph>

</Division>
<Division number="133" book="ap1" bookNo="20" page="84">

<Paragraph number="996" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_vimokkhāpi ca aṭṭhime;_  
_Chaḷabhiññā sacchikatā,_  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="997" book="ap1">
Itthaṃ sudaṃ āyasmā tisaraṇagamaniyo thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="998" type="center" book="ap1">
Tisaraṇagamaniyattherassāpadānaṃ tatiyaṃ.
</Paragraph>
