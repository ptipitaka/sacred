<Division number="22" book="ap1">

<Paragraph number="1620" type="verses" book="ap1">
_“Himavantassāvidūre,_  
_mahājātassaro ahu;_  
_Satapattehi sañchanno,_  
_nānāsakuṇamālayo._  

</Paragraph>

</Division>
<Division number="23" book="ap1">

<Paragraph number="1621" type="verses" book="ap1">
_Tamhi nhatvā ca pitvā ca,_  
_avidūre vasāmahaṃ;_  
_Addasaṃ samaṇānaggaṃ,_  
_gacchantaṃ anilañjase._  

</Paragraph>

</Division>
<Division number="24" book="ap1">

<Paragraph number="1622" type="verses" book="ap1">
_Mama saṅkappamaññāya,_  
_satthā loke anuttaro;_  
_Abbhato oruhitvāna,_  
_bhūmiyaṃṭhāsi tāvade._  

</Paragraph>

</Division>
<Division number="25" book="ap1">

<Paragraph number="1623" type="verses" book="ap1">
_Visāṇena tiṇaṃ gayha,_  
_nisīdanamadāsahaṃ;_  
_Nisīdi bhagavā tattha,_  
_tisso lokagganāyako._  

</Paragraph>

</Division>
<Division number="26" book="ap1">

<Paragraph number="1624" type="verses" book="ap1">
_Sakaṃ cittaṃ pasādetvā,_  
_avandi lokanāyakaṃ;_  
_Paṭikuṭiko apasakkiṃ,_  
_nijjhāyanto mahāmuniṃ._  

</Paragraph>

</Division>
<Division number="27" book="ap1">

<Paragraph number="1625" type="verses" book="ap1">
_Tena cittappasādena,_  
_nimmānaṃ upapajjahaṃ;_  
_Duggatiṃ nābhijānāmi,_  
_santharassa idaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="28" book="ap1">

<Paragraph number="1626" type="verses" book="ap1">
_Ito dutiyake kappe,_  
_miga sammatakhattiyo;_  
_Sattaratanasampanno,_  
_cakkavattī mahabbalo._  

</Paragraph>

</Division>
<Division number="29" book="ap1">

<Paragraph number="1627" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="1628" book="ap1">
Itthaṃ sudaṃ āyasmā tiṇasantharadāyako thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="1629" type="center" book="ap1">
Tiṇasantharadāyakattherassāpadānaṃ chaṭṭhaṃ.
</Paragraph>
