<Paragraph number="7135" type="verses" book="ap1">
_“Ito satasahassamhi,_  
_kappe uppajji nāyako;_  
_Mahappabho tilokaggo,_  
_nāmena padumuttaro._  

</Paragraph>

</Division>
<Division number="179" book="ap1">

<Paragraph number="7136" type="verses" book="ap1">
_Khippābhiññassa bhikkhussa,_  
_guṇaṃ kittayato mune;_  
_Sutvā udaggacittohaṃ,_  
_kāraṃ katvā mahesino._  

</Paragraph>

</Division>
<Division number="180" book="ap1">

<Paragraph number="7137" type="verses" book="ap1">
_Datvā sattāhikaṃ dānaṃ,_  
_sasissassa mune ahaṃ;_  
_Abhivādiya sambuddhaṃ,_  
_taṃ ṭhānaṃ patthayiṃ tadā._  

</Paragraph>

</Division>
<Division number="181" book="ap1">

<Paragraph number="7138" type="verses" book="ap1">
_Tato maṃ byākari buddho,_  
_‘etaṃ passatha brāhmaṇaṃ;_  
_Patitaṃ pādamūle me,_  
_cariyaṃ paccavekkhaṇaṃ._  

</Paragraph>

</Division>
<Division number="182" book="ap1">

<Paragraph number="7139" type="verses" book="ap1">
_Hemayaññopacitaṅgaṃ,_  
_avadātatanuttacaṃ;_  
_Palambabimbatamboṭṭhaṃ,_  
_setatiṇhasamaṃ dijaṃ._  

</Paragraph>

</Division>
<Division number="183" book="ap1">

<Paragraph number="7140" type="verses" book="ap1">
_Guṇathāmabahutaraṃ,_  
_samuggatatanūruhaṃ;_  
_Guṇoghāyatanībhūtaṃ,_  
_pītisamphullitānanaṃ._  

</Paragraph>

</Division>
<Division number="184" book="ap1">

<Paragraph number="7141" type="verses" book="ap1">
_Eso patthayate ṭhānaṃ,_  
_khippābhiññassa bhikkhuno;_  
_Anāgate mahāvīro,_  
_gotamo nāma hessati._  

</Paragraph>

</Division>
<Division number="185" book="ap1">

<Paragraph number="7142" type="verses" book="ap1">
_Tassa dhammesu dāyādo,_  
_oraso dhammanimmito;_  
_Bāhiyo nāma nāmena,_  
_hessati satthu sāvako’._  

</Paragraph>

</Division>
<Division number="186" book="ap1">

<Paragraph number="7143" type="verses" book="ap1">
_Tadā hi tuṭṭho vuṭṭhāya,_  
_yāvajīvaṃ mahāmune;_  
_Kāraṃ katvā cuto saggaṃ,_  
_agaṃ sabhavanaṃ yathā._  

</Paragraph>

</Division>
<Division number="187" book="ap1">

<Paragraph number="7144" type="verses" book="ap1">
_Devabhūto manusso vā,_  
_sukhito tassa kammuno;_  
_Vāhasā saṃsaritvāna,_  
_sampattimanubhomahaṃ._  

</Paragraph>

</Division>
<Division number="188" book="ap1">

<Paragraph number="7145" type="verses" book="ap1">
_Puna kassapavīrassa,_  
_atthamentamhi sāsane;_  
_Āruyha selasikharaṃ,_  
_yuñjitvā jinasāsanaṃ._  

</Paragraph>

</Division>
<Division number="189" book="ap1">

<Paragraph number="7146" type="verses" book="ap1">
_Visuddhasīlo sappañño,_  
_jinasāsanakārako;_  
_Tato cutā pañca janā,_  
_devalokaṃ agamhase._  

</Paragraph>

</Division>
<Division number="190" book="ap1">

<Paragraph number="7147" type="verses" book="ap1">
_Tatohaṃ bāhiyo jāto,_  
_bhārukacche puruttame;_  
_Tato nāvāya pakkhando,_  
_sāgaraṃ appasiddhiyaṃ._  

</Paragraph>

</Division>
<Division number="191" book="ap1">

<Paragraph number="7148" type="verses" book="ap1">
_Tato nāvā abhijjittha,_  
_gantvāna katipāhakaṃ;_  
_Tadā bhīsanake ghore,_  
_patito makarākare._  

</Paragraph>

</Division>
<Division number="192" book="ap1">

<Paragraph number="7149" type="verses" book="ap1">
_Tadāhaṃ vāyamitvāna,_  
_santaritvā mahodadhiṃ;_  
_Suppārapaṭṭanavaraṃ,_  
_sampatto mandavedhito._  

</Paragraph>

</Division>
<Division number="193" book="ap1">

<Paragraph number="7150" type="verses" book="ap1">
_Dārucīraṃ nivāsetvā,_  
_gāmaṃ piṇḍāya pāvisiṃ;_  
_Tadāha so jano tuṭṭho,_  
_‘arahāyamidhāgato._  

</Paragraph>

</Division>
<Division number="194" book="ap1">

<Paragraph number="7151" type="verses" book="ap1">
_Imaṃ annena pānena,_  
_vatthena sayanena ca;_  
_Bhesajjena ca sakkatvā,_  
_hessāma sukhitā mayaṃ’._  

</Paragraph>

</Division>
<Division number="195" book="ap1">

<Paragraph number="7152" type="verses" book="ap1">
_Paccayānaṃ tadā lābhī,_  
_tehi sakkatapūjito;_  
_Arahāhanti saṅkappaṃ,_  
_uppādesiṃ ayoniso._  

</Paragraph>

</Division>
<Division number="196" book="ap1">

<Paragraph number="7153" type="verses" book="ap1">
_Tato me cittamaññāya,_  
_codayī pubbadevatā;_  
_‘Na tvaṃ upāyamaggaññū,_  
_kuto tvaṃ arahā bhave’._  

</Paragraph>

</Division>
<Division number="197" book="ap1">

<Paragraph number="7154" type="verses" book="ap1">
_Codito tāya saṃviggo,_  
_tadāhaṃ paripucchi taṃ;_  
_‘Ke vā ete kuhiṃ loke,_  
_arahanto naruttamā’._  

</Paragraph>

</Division>
<Division number="198" book="ap1">

<Paragraph number="7155" type="verses" book="ap1">
_‘Sāvatthiyaṃ kosalamandire jino,_  
_Pahūtapañño varabhūrimedhaso;_  
_So sakyaputto arahā anāsavo,_  
_Deseti dhammaṃ arahattapattiyā’._  

</Paragraph>

</Division>
<Division number="199" book="ap1">

<Paragraph number="7156" type="verses" book="ap1">
_Tadassa sutvā vacanaṃ supīṇito,_  
_Nidhiṃva laddhā kapaṇoti vimhito;_  
_Udaggacitto arahattamuttamaṃ,_  
_Sudassanaṃ daṭṭhumanantagocaraṃ._  

</Paragraph>

</Division>
<Division number="200" book="ap1">

<Paragraph number="7157" type="verses" book="ap1">
_Tadā tato nikkhamitvāna ‘satthuno,_  
_Sadā jinaṃ passāmi vimalānanaṃ’;_  
_Upecca rammaṃ vijitavhayaṃ vanaṃ,_  
_Dije apucchiṃ ‘kuhiṃ lokanandano’._  

</Paragraph>

</Division>
<Division number="201" book="ap1">

<Paragraph number="7158" type="verses" book="ap1">
_‘Tato avocuṃ naradevavandito,_  
_Puraṃ paviṭṭho asanesanāya so;_  
_Sasova khippaṃ munidassanussuko,_  
_Upecca vandāhi tamaggapuggalaṃ’._  

</Paragraph>

</Division>
<Division number="202" book="ap1">

<Paragraph number="7159" type="verses" book="ap1">
_Tatohaṃ tuvaṭaṃ gantvā,_  
_sāvatthiṃ puramuttamaṃ;_  
_Vicarantaṃ tamaddakkhiṃ,_  
_piṇḍatthaṃ apihāgidhaṃ._  

</Paragraph>

</Division>
<Division number="203" book="ap1">

<Paragraph number="7160" type="verses" book="ap1">
_Pattapāṇiṃ alolakkhaṃ,_  
_pācayantaṃ pītākaraṃ;_  
_Sirīnilayasaṅkāsaṃ,_  
_ravidittiharānanaṃ._  

</Paragraph>

</Division>
<Division number="204" book="ap1">

<Paragraph number="7161" type="verses" book="ap1">
_Taṃ samecca nipaccāhaṃ,_  
_idaṃ vacanamabraviṃ;_  
_‘Kupathe vippanaṭṭhassa,_  
_saraṇaṃ hohi gotama’._  

</Paragraph>

</Division>
<Division number="205" book="ap1">

<Paragraph number="7162" type="verses" book="ap1">
_‘Pāṇasantāraṇatthāya,_  
_piṇḍāya vicarāmahaṃ;_  
_Na te dhammakathākālo’,_  
_iccāha munisattamo._  

</Paragraph>

</Division>
<Division number="206" book="ap1">

<Paragraph number="7163" type="verses" book="ap1">
_Tadā punappunaṃ buddhaṃ,_  
_āyāciṃ dhammalālaso;_  
_Yo me dhammamadesesi,_  
_gambhīraṃ suññataṃ padaṃ._  

</Paragraph>

</Division>
<Division number="207" book="ap1">

<Paragraph number="7164" type="verses" book="ap1">
_Tassa dhammaṃ suṇitvāna,_  
_pāpuṇiṃ āsavakkhayaṃ;_  
_Parikkhīṇāyuko santo,_  
_aho satthānukampako._  

</Paragraph>

</Division>
<Division number="208" book="ap1">

<Paragraph number="7165" type="verses" book="ap1">

_ …pe… _  
_viharāmi anāsavo._  

</Paragraph>

</Division>
<Division number="209" book="ap1">

_ …pe… _  
_kataṃ buddhassa sāsanaṃ._  

</Paragraph>

</Division>
<Division number="210" book="ap1">

<Paragraph number="7167" type="verses" book="ap1">

_ …pe… _  
_kataṃ buddhassa sāsanaṃ._  

</Paragraph>

</Division>
<Division number="211" book="ap1">

<Paragraph number="7168" type="verses" book="ap1">
_Evaṃ thero viyākāsi,_  
_bāhiyo dārucīriyo;_  
_Saṅkārakūṭe patito,_  
_bhūtāviṭṭhāya gāviyā._  

</Paragraph>

</Division>
<Division number="212" book="ap1">

<Paragraph number="7169" type="verses" book="ap1">
_Attano pubbacariyaṃ,_  
_kittayitvā mahāmati;_  
_Parinibbāyi so thero,_  
_sāvatthiyaṃ puruttame._  

</Paragraph>

</Division>
<Division number="213" book="ap1">

<Paragraph number="7170" type="verses" book="ap1">
_Nagarā nikkhamanto taṃ,_  
_disvāna isisattamo;_  
_Dārucīradharaṃ dhīraṃ,_  
_bāhiyaṃ bāhitāgamaṃ._  

</Paragraph>

</Division>
<Division number="214" book="ap1">

<Paragraph number="7171" type="verses" book="ap1">
_Bhūmiyaṃ patitaṃ dantaṃ,_  
_indaketūva pātitaṃ;_  
_Gatāyuṃ sukkhakilesaṃ,_  
_jinasāsanakārakaṃ._  

</Paragraph>

</Division>
<Division number="215" book="ap1">

<Paragraph number="7172" type="verses" book="ap1">
_Tato āmantayī satthā,_  
_sāvake sāsane rate;_  
_‘Gaṇhatha netvā jhāpetha,_  
_tanuṃ sabrahmacārino._  

</Paragraph>

</Division>
<Division number="216" book="ap1">

<Paragraph number="7173" type="verses" book="ap1">
_Thūpaṃ karotha pūjetha,_  
_nibbuto so mahāmati;_  
_Khippābhiññānamesaggo,_  
_sāvako me vacokaro._  

</Paragraph>

</Division>
<Division number="217" book="ap1">

<Paragraph number="7174" type="verses" book="ap1">
_Sahassamapi ce gāthā,_  
_anatthapadasañhitā;_  
_Ekaṃ gāthāpadaṃ seyyo,_  
_yaṃ sutvā upasammati._  

</Paragraph>

</Division>
<Division number="218" book="ap1">

<Paragraph number="7175" type="verses" book="ap1">
_Yattha āpo ca pathavī,_  
_tejo vāyo na gādhati;_  
_Na tattha sukkā jotanti,_  
_ādicco na pakāsati._  

</Paragraph>

</Division>
<Division number="219" book="ap1">

<Paragraph number="7176" type="verses" book="ap1">
_Na tattha candimā bhāti,_  
_tamo tattha na vijjati;_  
_Yadā ca attanā vedi,_  
_munimonena brāhmaṇo._  

</Paragraph>

</Division>
<Division number="220" book="ap1">

<Paragraph number="7177" type="verses" book="ap1">
_Atha rūpā arūpā ca,_  
_sukhadukkhā vimuccati’;_  
_Iccevaṃ abhaṇī nātho,_  
_tilokasaraṇo muni”._  

</Paragraph>

<Paragraph number="7178" book="ap1">
Itthaṃ sudaṃ āyasmā bāhiyo thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>
