<Paragraph number="4033" type="verses" book="ap1">
_“Tāvatiṃsā idhāgantvā,_  
_maṅgalo nāma māṇavo;_  
_Mandāravaṃ gahetvāna,_  
_vipassissa mahesino._  

</Paragraph>

</Division>
<Division number="2" book="ap1" bookNo="20" page="330">

<Paragraph number="4034" type="verses" book="ap1">
_Samādhinā nisinnassa,_  
_matthake dhārayiṃ ahaṃ;_  
_Sattāhaṃ dhārayitvāna,_  
_devalokaṃ punāgamiṃ._  

</Paragraph>

</Division>
<Division number="3" book="ap1" bookNo="20" page="330">

<Paragraph number="4035" type="verses" book="ap1">
_Ekanavutito kappe,_  
_yaṃ pupphamabhipūjayiṃ;_  
_Duggatiṃ nābhijānāmi,_  
_buddhapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="4" book="ap1" bookNo="20" page="330">

<Paragraph number="4036" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_vimokkhāpi ca aṭṭhime;_  
_Chaḷabhiññā sacchikatā,_  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="4037" book="ap1">
Itthaṃ sudaṃ āyasmā mandāravapupphiyo thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="4038" type="center" book="ap1">
Mandāravapupphiyattherassāpadānaṃ paṭhamaṃ.
</Paragraph>
