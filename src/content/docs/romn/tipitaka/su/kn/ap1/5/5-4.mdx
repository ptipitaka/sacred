<Paragraph number="1284" type="verses" book="ap1">
_“Araññe kuṭikaṃ katvā,_  
_vasāmi pabbatantare;_  
_Lābhālābhena santuṭṭho,_  
_yasena ayasena ca._  

</Paragraph>

</Division>
<Division number="71" book="ap1" bookNo="20" page="108">

<Paragraph number="1285" type="verses" book="ap1">
_Padumuttaro lokavidū,_  
_āhutīnaṃ paṭiggaho;_  
_Vasīsatasahassehi,_  
_āgacchi mama santikaṃ._  

</Paragraph>

</Division>
<Division number="72" book="ap1" bookNo="20" page="108">

<Paragraph number="1286" type="verses" book="ap1">
_Upāgataṃ mahānāgaṃ,_  
_jalajuttamanāmakaṃ;_  
_Tiṇasantharaṃ paññāpetvā,_  
_adāsiṃ satthuno ahaṃ._  

</Paragraph>

</Division>
<Division number="73" book="ap1" bookNo="20" page="108">

<Paragraph number="1287" type="verses" book="ap1">
_Pasannacitto sumano,_  
_āmaṇḍaṃ pānīyañcahaṃ;_  
_Adāsiṃ ujubhūtassa,_  
_vippasannena cetasā._  

</Paragraph>

</Division>
<Division number="74" book="ap1" bookNo="20" page="108">

<Paragraph number="1288" type="verses" book="ap1">
_Satasahassito kappe,_  
_yaṃ dānamadadiṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_āmaṇḍassa idaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="75" book="ap1" bookNo="20" page="108">

<Paragraph number="1289" type="verses" book="ap1">
_Ekatālīsakappamhi,_  
_eko āsiṃ arindamo;_  
_Sattaratanasampanno,_  
_cakkavattī mahabbalo._  

</Paragraph>

</Division>
<Division number="76" book="ap1" bookNo="20" page="108">

<Paragraph number="1290" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="1291" book="ap1">

</Paragraph>
