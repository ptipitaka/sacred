<Division number="39" book="ap1">

<Paragraph number="3558" type="verses" book="ap1">
_“Kesariṃ abhijātaṃva,_  
_aggikkhandhaṃva pabbate;_  
_Obhāsentaṃ disā sabbā,_  
_siddhatthaṃ pabbatantare._  

</Paragraph>

</Division>
<Division number="40" book="ap1">

<Paragraph number="3559" type="verses" book="ap1">
_Sūriyassa ca ālokaṃ,_  
_candālokaṃ tatheva ca;_  
_Buddhālokañca disvāna,_  
_vitti me udapajjatha._  

</Paragraph>

</Division>
<Division number="41" book="ap1">

<Paragraph number="3560" type="verses" book="ap1">
_Tayo āloke disvāna,_  
_sambuddhaṃ sāvakuttamaṃ;_  
_Ekaṃsaṃ ajinaṃ katvā,_  
_santhaviṃ lokanāyakaṃ._  

</Paragraph>

</Division>
<Division number="42" book="ap1">

<Paragraph number="3561" type="verses" book="ap1">
_Tayo hi ālokakarā,_  
_loke lokatamonudā;_  
_Cando ca sūriyo cāpi,_  
_buddho ca lokanāyako._  

</Paragraph>

</Division>
<Division number="43" book="ap1">

<Paragraph number="3562" type="verses" book="ap1">
_Opammaṃ upadassetvā,_  
_kittito me mahāmuni;_  
_Buddhassa vaṇṇaṃ kittetvā,_  
_kappaṃ saggamhi modahaṃ._  

</Paragraph>

</Division>
<Division number="44" book="ap1">

<Paragraph number="3563" type="verses" book="ap1">
_Catunnavutito kappe,_  
_yaṃ buddhamabhikittayiṃ;_  
_Duggatiṃ nābhijānāmi,_  
_kittanāya idaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="45" book="ap1">

<Paragraph number="3564" type="verses" book="ap1">
_Ekasaṭṭhimhito kappe,_  
_eko ñāṇadharo ahu;_  
_Sattaratanasampanno,_  
_cakkavattī mahabbalo._  

</Paragraph>

</Division>
<Division number="46" book="ap1">

<Paragraph number="3565" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="3566" book="ap1">
Itthaṃ sudaṃ āyasmā tiraṃsiyo thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="3567" type="center" book="ap1">
Tiraṃsiyattherassāpadānaṃ aṭṭhamaṃ.
</Paragraph>
