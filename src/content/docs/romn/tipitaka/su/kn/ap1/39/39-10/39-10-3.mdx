<Division number="74" book="ap1">

<Paragraph number="4268" type="verses" book="ap1">
_Tatthāgato isi bhīmo,_  
_pañcābhiñño mahiddhiko;_  
_Tañcāhaṃ āgataṃ disvā,_  
_abbhācikkhiṃ adūsakaṃ._  

</Paragraph>

</Division>
<Division number="75" book="ap1">

<Paragraph number="4269" type="verses" book="ap1">

_kāmabhogī ayaṃ isi;_  
_Mayhampi bhāsamānassa,_  
_anumodiṃsu māṇavā._  

</Paragraph>

</Division>
<Division number="76" book="ap1">

<Paragraph number="4270" type="verses" book="ap1">

_bhikkhamānaṃ kule kule;_  
_Mahājanassa āhaṃsu,_  
_kāmabhogī ayaṃ isi._  

</Paragraph>

</Division>

_Abbhakkhānaṃ labhuṃ sabbe,_  
_sundarikāya kāraṇā._
