_“Bhariyā me kālaṅkatā,_  
_putto sīvathikaṃ gato;_  
_Mātā pitā matā bhātā,_  
_ekacitamhi ḍayhare._  

</Paragraph>

</Division>
<Division number="420" book="ap1" bookNo="20" page="387">

<Paragraph number="4730" type="verses" book="ap1">
_Tena sokena santatto,_  
_kiso paṇḍu ahosahaṃ;_  
_Cittakkhepo ca me āsi,_  
_tena sokena aṭṭito._  

</Paragraph>

</Division>
<Division number="421" book="ap1" bookNo="20" page="387">

<Paragraph number="4731" type="verses" book="ap1">
_Sokasallaparetohaṃ,_  
_vanantamupasaṅkamiṃ;_  
_Pavattaphalaṃ bhuñjitvā,_  
_rukkhamūle vasāmahaṃ._  

</Paragraph>

</Division>
<Division number="422" book="ap1" bookNo="20" page="388">

<Paragraph number="4732" type="verses" book="ap1">
_Sumedho nāma sambuddho,_  
_dukkhassantakaro jino;_  
_Mamuddharitukāmo so,_  
_āgañchi mama santikaṃ._  

</Paragraph>

</Division>
<Division number="423" book="ap1" bookNo="20" page="388">

<Paragraph number="4733" type="verses" book="ap1">
_Padasaddaṃ suṇitvāna,_  
_sumedhassa mahesino;_  
_Paggahetvānahaṃ sīsaṃ,_  
_ullokesiṃ mahāmuniṃ._  

</Paragraph>

</Division>
<Division number="424" book="ap1" bookNo="20" page="388">

<Paragraph number="4734" type="verses" book="ap1">
_Upāgate mahāvīre,_  
_pīti me udapajjatha;_  
_Tadāsimekaggamano,_  
_disvā taṃ lokanāyakaṃ._  

</Paragraph>

</Division>
<Division number="425" book="ap1" bookNo="20" page="388">

<Paragraph number="4735" type="verses" book="ap1">
_Satiṃ paṭilabhitvāna,_  
_paṇṇamuṭṭhimadāsahaṃ;_  
_Nisīdi bhagavā tattha,_  
_anukampāya cakkhumā._  

</Paragraph>

</Division>
<Division number="426" book="ap1" bookNo="20" page="388">

<Paragraph number="4736" type="verses" book="ap1">
_Nisajja tattha bhagavā,_  
_sumedho lokanāyako;_  
_Dhammaṃ me kathayī buddho,_  
_sokasallavinodanaṃ._  

</Paragraph>

</Division>
<Division number="427" book="ap1" bookNo="20" page="388">

<Paragraph number="4737" type="verses" book="ap1">
_‘Anavhitā tato āguṃ,_  
_ananuññātā ito gatā;_  
_Yathāgatā tathā gatā,_  
_tattha kā paridevanā._  

</Paragraph>

</Division>
<Division number="428" book="ap1" bookNo="20" page="388">

<Paragraph number="4738" type="verses" book="ap1">
_Yathāpi pathikā sattā,_  
_vassamānāya vuṭṭhiyā;_  
_Sabhaṇḍā upagacchanti,_  
_vassassāpatanāya te._  

</Paragraph>

</Division>
<Division number="429" book="ap1" bookNo="20" page="388">

<Paragraph number="4739" type="verses" book="ap1">
_Vasse ca te oramite,_  
_sampayanti yadicchakaṃ;_  
_Evaṃ mātā pitā tuyhaṃ,_  
_tattha kā paridevanā._  

</Paragraph>

</Division>
<Division number="430" book="ap1" bookNo="20" page="388">

<Paragraph number="4740" type="verses" book="ap1">
_Āgantukā pāhunakā,_  
_caliteritakampitā;_  
_Evaṃ mātā pitā tuyhaṃ,_  
_tattha kā paridevanā._  

</Paragraph>

</Division>
<Division number="431" book="ap1" bookNo="20" page="388">

<Paragraph number="4741" type="verses" book="ap1">
_Yathāpi urago jiṇṇaṃ,_  
_hitvā gacchati saṃ tacaṃ;_  
_Evaṃ mātā pitā tuyhaṃ,_  
_saṃ tanuṃ idha hīyare’._  

</Paragraph>

</Division>
<Division number="432" book="ap1" bookNo="20" page="388">

<Paragraph number="4742" type="verses" book="ap1">
_Buddhassa giramaññāya,_  
_sokasallaṃ vivajjayiṃ;_  
_Pāmojjaṃ janayitvāna,_  
_buddhaseṭṭhaṃ avandahaṃ._  

</Paragraph>

</Division>
<Division number="433" book="ap1" bookNo="20" page="388">

<Paragraph number="4743" type="verses" book="ap1">
_Vanditvāna mahānāgaṃ,_  
_Pūjayiṃ girimañjariṃ;_  
_Dibbagandhaṃ sampavantaṃ,_  
_Sumedhaṃ lokanāyakaṃ._  

</Paragraph>

</Division>
<Division number="434" book="ap1" bookNo="20" page="388">

<Paragraph number="4744" type="verses" book="ap1">
_Pūjayitvāna sambuddhaṃ,_  
_sire katvāna añjaliṃ;_  
_Anussaraṃ guṇaggāni,_  
_santhaviṃ lokanāyakaṃ._  

</Paragraph>

</Division>
<Division number="435" book="ap1" bookNo="20" page="389">

<Paragraph number="4745" type="verses" book="ap1">
_‘Nittiṇṇosi mahāvīra,_  
_sabbaññu lokanāyaka;_  
_Sabbe satte uddharasi,_  
_ñāṇena tvaṃ mahāmune._  

</Paragraph>

</Division>
<Division number="436" book="ap1" bookNo="20" page="389">

<Paragraph number="4746" type="verses" book="ap1">
_Vimatiṃ dveḷhakaṃ vāpi,_  
_sañchindasi mahāmune;_  
_Paṭipādesi me maggaṃ,_  
_tava ñāṇena cakkhuma._  

</Paragraph>

</Division>
<Division number="437" book="ap1" bookNo="20" page="389">

<Paragraph number="4747" type="verses" book="ap1">
_Arahā vasipattā ca,_  
_chaḷabhiññā mahiddhikā;_  
_Antalikkhacarā dhīrā,_  
_parivārenti tāvade._  

</Paragraph>

</Division>
<Division number="438" book="ap1" bookNo="20" page="389">

<Paragraph number="4748" type="verses" book="ap1">
_Paṭipannā ca sekhā ca,_  
_phalaṭṭhā santi sāvakā;_  
_Sūrodayeva padumā,_  
_pupphanti tava sāvakā._  

</Paragraph>

</Division>
<Division number="439" book="ap1" bookNo="20" page="389">

<Paragraph number="4749" type="verses" book="ap1">
_Mahāsamuddovakkhobho,_  
_atulopi duruttaro;_  
_Evaṃ ñāṇena sampanno,_  
_appameyyosi cakkhuma’._  

</Paragraph>

</Division>
<Division number="440" book="ap1" bookNo="20" page="389">

<Paragraph number="4750" type="verses" book="ap1">
_Vanditvāhaṃ lokajinaṃ,_  
_cakkhumantaṃ mahāyasaṃ;_  
_Puthu disā namassanto,_  
_paṭikuṭiko agañchahaṃ._  

</Paragraph>

</Division>
<Division number="441" book="ap1" bookNo="20" page="389">

<Paragraph number="4751" type="verses" book="ap1">
_Devalokā cavitvāna,_  
_sampajāno patissato;_  
_Okkamiṃ mātuyā kucchiṃ,_  
_sandhāvanto bhavābhave._  

</Paragraph>

</Division>
<Division number="442" book="ap1" bookNo="20" page="389">

<Paragraph number="4752" type="verses" book="ap1">
_Agārā abhinikkhamma,_  
_pabbajiṃ anagāriyaṃ;_  
_Ātāpī nipako jhāyī,_  
_paṭisallānagocaro._  

</Paragraph>

</Division>
<Division number="443" book="ap1" bookNo="20" page="389">

<Paragraph number="4753" type="verses" book="ap1">
_Padhānaṃ padahitvāna,_  
_tosayitvā mahāmuniṃ;_  
_Candovabbhaghanā mutto,_  
_vicarāmi ahaṃ sadā._  

</Paragraph>

</Division>
<Division number="444" book="ap1" bookNo="20" page="389">

<Paragraph number="4754" type="verses" book="ap1">
_Vivekamanuyuttomhi,_  
_upasanto nirūpadhi;_  
_Sabbāsave pariññāya,_  
_viharāmi anāsavo._  

</Paragraph>

</Division>
<Division number="445" book="ap1" bookNo="20" page="389">

<Paragraph number="4755" type="verses" book="ap1">
_Tiṃsakappasahassamhi,_  
_yaṃ buddhamabhipūjayiṃ;_  
_Duggatiṃ nābhijānāmi,_  
_buddhapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="446" book="ap1" bookNo="20" page="389">

<Paragraph number="4756" type="verses" book="ap1">
_Kilesā jhāpitā mayhaṃ,_  
_bhavā sabbe samūhatā;_  
_Sabbāsavā parikkhīṇā,_  
_natthi dāni punabbhavo._  

</Paragraph>

</Division>
