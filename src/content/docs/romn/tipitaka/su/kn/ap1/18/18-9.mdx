_“Nagare bandhumatiyā,_  
_vejjo āsiṃ susikkhito;_  
_Āturānaṃ sadukkhānaṃ,_  
_mahājanasukhāvaho._

<Division number="40" book="ap1" bookNo="20" page="211">

<Paragraph number="2565" type="verses" book="ap1">
_Byādhitaṃ samaṇaṃ disvā,_  
_sīlavantaṃ mahājutiṃ;_  
_Pasannacitto sumano,_  
_bhesajjamadadiṃ tadā._  

</Paragraph>

</Division>
<Division number="41" book="ap1" bookNo="20" page="211">

<Paragraph number="2566" type="verses" book="ap1">
_Arogo āsi teneva,_  
_samaṇo saṃvutindriyo;_  
_Asoko nāma nāmena,_  
_upaṭṭhāko vipassino._  

</Paragraph>

</Division>
<Division number="42" book="ap1" bookNo="20" page="211">

<Paragraph number="2567" type="verses" book="ap1">
_Ekanavutito kappe,_  
_yaṃ osadhamadāsahaṃ;_  
_Duggatiṃ nābhijānāmi,_  
_bhesajjassa idaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="43" book="ap1" bookNo="20" page="211">

<Paragraph number="2568" type="verses" book="ap1">
_Ito ca aṭṭhame kappe,_  
_sabbosadhasanāmako;_  
_Sattaratanasampanno,_  
_cakkavattī mahabbalo._  

</Paragraph>

</Division>

_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="2570" book="ap1">

</Paragraph>
</Division>
