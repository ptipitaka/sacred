<Division number="26" book="ap1">

<Paragraph number="5693" type="verses" book="ap1">
_“Padumuttaro nāma jino,_  
_āhutīnaṃ paṭiggaho;_  
_Vasīsatasahassehi,_  
_nagaraṃ pāvisī tadā._  

</Paragraph>

</Division>
<Division number="27" book="ap1">

<Paragraph number="5694" type="verses" book="ap1">
_Nagaraṃ pavisantassa,_  
_upasantassa tādino;_  
_Ratanāni pajjotiṃsu,_  
_nigghoso āsi tāvade._  

</Paragraph>

</Division>
<Division number="28" book="ap1">

<Paragraph number="5695" type="verses" book="ap1">
_Buddhassa ānubhāvena,_  
_bherī vajjumaghaṭṭitā;_  
_Sayaṃ vīṇā pavajjanti,_  
_buddhassa pavisato puraṃ._  

</Paragraph>

</Division>
<Division number="29" book="ap1">

<Paragraph number="5696" type="verses" book="ap1">
_Buddhaseṭṭhaṃ namassāmi,_  
_padumuttaramahāmuniṃ;_  
_Pāṭihīrañca passitvā,_  
_tattha cittaṃ pasādayiṃ._  

</Paragraph>

</Division>
<Division number="30" book="ap1">

<Paragraph number="5697" type="verses" book="ap1">
_Aho buddhā aho dhammā,_  
_aho no satthusampadā;_  
_Acetanāpi turiyā,_  
_sayameva pavajjare._  

</Paragraph>

</Division>
<Division number="31" book="ap1">

<Paragraph number="5698" type="verses" book="ap1">
_Satasahassito kappe,_  
_yaṃ saññamalabhiṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_buddhasaññāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="32" book="ap1">

<Paragraph number="5699" type="verses" book="ap1">
_Kilesā jhāpitā mayhaṃ,_  
_ …pe… _  
_viharāmi anāsavo._  

</Paragraph>

</Division>
<Division number="33" book="ap1">

<Paragraph number="5700" type="verses" book="ap1">
_Svāgataṃ vata me āsi,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ._  

</Paragraph>

</Division>
<Division number="34" book="ap1">

<Paragraph number="5701" type="verses" book="ap1">

_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="5702" book="ap1">

</Paragraph>
</Division>
