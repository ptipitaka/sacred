_sabbalokavidū muni;_

_kappe uppajji cakkhumā._  

</Paragraph>

<Division number="297" book="ap1">

<Paragraph number="7260" type="verses" book="ap1">
_Ovādako viññāpako,_  
_tārako sabbapāṇinaṃ;_  
_Desanākusalo buddho,_  
_tāresi janataṃ bahuṃ._  

</Paragraph>

</Division>
<Division number="298" book="ap1">

<Paragraph number="7261" type="verses" book="ap1">
_Anukampako kāruṇiko,_  
_hitesī sabbapāṇinaṃ;_  
_Sampatte titthiye sabbe,_  
_pañcasīle patiṭṭhapi._  

</Paragraph>

</Division>
<Division number="299" book="ap1">

<Paragraph number="7262" type="verses" book="ap1">
_Evaṃ nirākulaṃ āsi,_  
_suññataṃ titthiyehi ca;_  
_Vicittaṃ arahantehi,_  
_vasībhūtehi tādibhi._  

</Paragraph>

</Division>
<Division number="300" book="ap1">

<Paragraph number="7263" type="verses" book="ap1">
_Ratanānaṭṭhapaññāsaṃ,_  
_uggato so mahāmuni;_  
_Kañcanagghiyasaṅkāso,_  
_bāttiṃsavaralakkhaṇo._  

</Paragraph>

</Division>
<Division number="301" book="ap1">

<Paragraph number="7264" type="verses" book="ap1">
_Vassasatasahassāni,_  
_āyu vijjati tāvade;_  
_Tāvatā tiṭṭhamāno so,_  
_tāresi janataṃ bahuṃ._  

</Paragraph>

</Division>
<Division number="302" book="ap1">

<Paragraph number="7265" type="verses" book="ap1">
_Tadāhaṃ haṃsavatiyaṃ,_  
_brāhmaṇo mantapāragū;_  
_Upecca taṃ naravaraṃ,_  
_assosiṃ dhammadesanaṃ._  

</Paragraph>

</Division>
<Division number="303" book="ap1">

<Paragraph number="7266" type="verses" book="ap1">
_Paññāpentaṃ mahāvīraṃ,_  
_parisāsu visāradaṃ;_  
_Paṭibhāneyyakaṃ bhikkhuṃ,_  
_etadagge vināyakaṃ._  

</Paragraph>

</Division>
<Division number="304" book="ap1">

<Paragraph number="7267" type="verses" book="ap1">
_Tadāhaṃ kāraṃ katvāna,_  
_sasaṃghe lokanāyake;_  
_Nipacca sirasā pāde,_  
_taṃ ṭhānaṃ abhipatthayiṃ._  

</Paragraph>

</Division>
<Division number="305" book="ap1">

<Paragraph number="7268" type="verses" book="ap1">
_Tato maṃ bhagavā āha,_  
_siṅgīnikkhasamappabho;_  
_Sarena rajanīyena,_  
_kilesamalahārinā._  

</Paragraph>

</Division>
<Division number="306" book="ap1">

<Paragraph number="7269" type="verses" book="ap1">
_‘Sukhī bhavassu dīghāyu,_  
_sijjhatu paṇidhī tava;_  
_Sasaṃghe me kataṃ kāraṃ,_  
_atīva vipulaṃ tayā._  

</Paragraph>

</Division>
<Division number="307" book="ap1">

<Paragraph number="7270" type="verses" book="ap1">
_Satasahassito kappe,_  
_okkākakulasambhavo;_  
_Gotamo nāma gottena,_  
_satthā loke bhavissati._  

</Paragraph>

</Division>
<Division number="308" book="ap1">

<Paragraph number="7271" type="verses" book="ap1">
_Tassa dhammesu dāyādo,_  
_oraso dhammanimmito;_  
_Rādhoti nāmadheyyena,_  
_hessati satthu sāvako._  

</Paragraph>

</Division>
<Division number="309" book="ap1">

<Paragraph number="7272" type="verses" book="ap1">
_Sa te hetuguṇe tuṭṭho,_  
_sakyaputto narāsabho;_  
_Paṭibhāneyyakānaggaṃ,_  
_paññapessati nāyako’._  

</Paragraph>

</Division>
<Division number="310" book="ap1">

<Paragraph number="7273" type="verses" book="ap1">
_Taṃ sutvā mudito hutvā,_  
_yāvajīvaṃ tadā jinaṃ;_  
_Mettacitto paricariṃ,_  
_sato paññāsamāhito._  

</Paragraph>

</Division>
<Division number="311" book="ap1">

<Paragraph number="7274" type="verses" book="ap1">
_Tena kammena sukatena,_  
_cetanāpaṇidhīhi ca;_  
_Jahitvā mānusaṃ dehaṃ,_  
_tāvatiṃsamagacchahaṃ._  

</Paragraph>

</Division>
<Division number="312" book="ap1">

<Paragraph number="7275" type="verses" book="ap1">
_Satānaṃ tīṇikkhattuñca,_  
_devarajjamakārayiṃ;_  
_Satānaṃ pañcakkhattuñca,_  
_cakkavattī ahosahaṃ._  

</Paragraph>

</Division>
<Division number="313" book="ap1">

<Paragraph number="7276" type="verses" book="ap1">
_Padesarajjaṃ vipulaṃ,_  
_gaṇanāto asaṅkhiyaṃ;_  
_Sabbattha sukhito āsiṃ,_  
_tassa kammassa vāhasā._  

</Paragraph>

</Division>
<Division number="314" book="ap1">

<Paragraph number="7277" type="verses" book="ap1">
_Pacchime bhave sampatte,_  
_giribbajapuruttame;_  
_Jāto vippakule niddhe,_  
_vikalacchādanāsane._  

</Paragraph>

</Division>
<Division number="315" book="ap1">

<Paragraph number="7278" type="verses" book="ap1">
_Kaṭacchubhikkhaṃ pādāsiṃ,_  
_sāriputtassa tādino;_  
_Yadā jiṇṇo ca vuddho ca,_  
_tadārāmamupāgamiṃ._  

</Paragraph>

</Division>
<Division number="316" book="ap1">

<Paragraph number="7279" type="verses" book="ap1">
_Pabbajati na maṃ koci,_  
_jiṇṇadubbalathāmakaṃ;_  
_Tena dīno vivaṇṇaṅgo,_  
_soko cāsiṃ tadā ahaṃ._  

</Paragraph>

</Division>
<Division number="317" book="ap1">

<Paragraph number="7280" type="verses" book="ap1">
_Disvā mahākāruṇiko,_  
_mamamāha mahāmuni;_  
_‘Kimatthaṃ puttasokaṭṭo,_  
_brūhi te cittajaṃ rujaṃ’._  

</Paragraph>

</Division>
<Division number="318" book="ap1">

<Paragraph number="7281" type="verses" book="ap1">
_‘Pabbajjaṃ na labhe vīra,_  
_svākkhāte tava sāsane;_  
_Tena sokena dīnosmi,_  
_saraṇaṃ hohi nāyaka’._  

</Paragraph>

</Division>
<Division number="319" book="ap1">

<Paragraph number="7282" type="verses" book="ap1">
_Tadā bhikkhū samānetvā,_  
_apucchi munisattamo;_  
_‘Imassa adhikāraṃ ye,_  
_saranti byāharantu te’._  

</Paragraph>

</Division>
<Division number="320" book="ap1">

<Paragraph number="7283" type="verses" book="ap1">
_Sāriputto tadāvoca,_  
_‘kāramassa sarāmahaṃ;_  
_Kaṭacchubhikkhaṃ dāpesi,_  
_piṇḍāya carato mama’._  

</Paragraph>

</Division>
<Division number="321" book="ap1">

<Paragraph number="7284" type="verses" book="ap1">
_‘Sādhu sādhu kataññūsi,_  
_sāriputta imaṃ tuvaṃ;_  
_Pabbājehi dijaṃ vuḍḍhaṃ,_  
_hessatājāniyo ayaṃ’._  

</Paragraph>

</Division>
<Division number="322" book="ap1">

<Paragraph number="7285" type="verses" book="ap1">
_Tato alatthaṃ pabbajjaṃ,_  
_kammavācopasampadaṃ;_  
_Na cireneva kālena,_  
_pāpuṇiṃ āsavakkhayaṃ._  

</Paragraph>

</Division>

_suṇāmi mudito yato;_  
_Paṭibhāneyyakānaggaṃ,_  
_tato maṃ ṭhapayī jino._  

</Paragraph>

</Division>
