<Division number="1" book="ap1">

<Paragraph number="3881" type="verses" book="ap1">
_“Padumuttaro nāma jino,_  
_sabbadhammāna pāragū;_  
_Bhavābhave vibhāvento,_  
_tāresi janataṃ bahuṃ._  

</Paragraph>

</Division>
<Division number="2" book="ap1">

<Paragraph number="3882" type="verses" book="ap1">
_Haṃsarājā tadā homi,_  
_dijānaṃ pavaro ahaṃ;_  
_Jātassaraṃ samogayha,_  
_kīḷāmi haṃsakīḷitaṃ._  

</Paragraph>

</Division>
<Division number="3" book="ap1">

<Paragraph number="3883" type="verses" book="ap1">
_Padumuttaro lokavidū,_  
_āhutīnaṃ paṭiggaho;_  
_Jātassarassa upari,_  
_āgacchi tāvade jino._  

</Paragraph>

</Division>
<Division number="4" book="ap1">

<Paragraph number="3884" type="verses" book="ap1">
_Disvānahaṃ devadevaṃ,_  
_sayambhuṃ lokanāyakaṃ;_  
_Vaṇṭe chetvāna padumaṃ,_  
_satapattaṃ manoramaṃ._  

</Paragraph>

</Division>
<Division number="5" book="ap1">

<Paragraph number="3885" type="verses" book="ap1">
_Mukhatuṇḍena paggayha,_  
_pasanno lokanāyake;_  
_Ukkhipitvāna gagane,_  
_buddhaseṭṭhaṃ apūjayiṃ._  

</Paragraph>

</Division>
<Division number="6" book="ap1">

<Paragraph number="3886" type="verses" book="ap1">
_Padumuttaro lokavidū,_  
_āhutīnaṃ paṭiggaho;_  
_Antalikkhe ṭhito satthā,_  
_akā me anumodanaṃ._  

</Paragraph>

</Division>
<Division number="7" book="ap1">

<Paragraph number="3887" type="verses" book="ap1">
_‘Iminā ekapadumena,_  
_cetanāpaṇidhīhi ca;_  
_Kappānaṃ satasahassaṃ,_  
_vinipātaṃ na gacchasi’._  

</Paragraph>

</Division>
<Division number="8" book="ap1">

<Paragraph number="3888" type="verses" book="ap1">
_Idaṃ vatvāna sambuddho,_  
_jalajuttamanāmako;_  
_Mama kammaṃ pakittetvā,_  
_agamā yena patthitaṃ._  

</Paragraph>

</Division>
<Division number="9" book="ap1">

<Paragraph number="3889" type="verses" book="ap1">
_Satasahassito kappe,_  
_yaṃ kammamakariṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_buddhapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="10" book="ap1">

<Paragraph number="3890" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_vimokkhāpi ca aṭṭhime;_  
_Chaḷabhiññā sacchikatā,_  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="3891" book="ap1">
Itthaṃ sudaṃ āyasmā ekapadumiyo thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="3892" type="center" book="ap1">
Ekapadumiyattherassāpadānaṃ paṭhamaṃ.
</Paragraph>
