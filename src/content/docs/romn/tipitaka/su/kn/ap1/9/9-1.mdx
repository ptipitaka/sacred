<Division number="1" book="ap1">

<Paragraph number="1686" type="verses" book="ap1">
_“Candabhāgānadītīre,_  
_anusotaṃ vajāmahaṃ;_  
_Nisinnaṃ samaṇaṃ disvā,_  
_vippasannamanāvilaṃ._  

</Paragraph>

</Division>
<Division number="2" book="ap1">

<Paragraph number="1687" type="verses" book="ap1">
_Tattha cittaṃ pasādetvā,_  
_evaṃ cintesahaṃ tadā;_  
_Tārayissati tiṇṇoyaṃ,_  
_dantoyaṃ damayissati._  

</Paragraph>

</Division>
<Division number="3" book="ap1">

<Paragraph number="1688" type="verses" book="ap1">
_Assāsissati assattho,_  
_santo ca samayissati;_  
_Mocayissati mutto ca,_  
_nibbāpessati nibbuto._  

</Paragraph>

</Division>
<Division number="4" book="ap1">

<Paragraph number="1689" type="verses" book="ap1">
_Evāhaṃ cintayitvāna,_  
_siddhatthassa mahesino;_  
_Gahetvā timirapupphaṃ,_  
_matthake okiriṃ ahaṃ._  

</Paragraph>

</Division>
<Division number="5" book="ap1">

<Paragraph number="1690" type="verses" book="ap1">
_Añjaliṃ paggahetvāna,_  
_katvā ca naṃ padakkhiṇaṃ;_  
_Vanditvā satthuno pāde,_  
_pakkāmiṃ aparaṃ disaṃ._  

</Paragraph>

</Division>
<Division number="6" book="ap1">

<Paragraph number="1691" type="verses" book="ap1">
_Aciraṃ gatamattaṃ maṃ,_  
_migarājā viheṭhayi;_  
_Papātamanugacchanto,_  
_tattheva papatiṃ ahaṃ._  

</Paragraph>

</Division>
<Division number="7" book="ap1">

<Paragraph number="1692" type="verses" book="ap1">
_Catunnavutito kappe,_  
_yaṃ pupphamabhiropayiṃ;_  
_Duggatiṃ nābhijānāmi,_  
_pupphapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="8" book="ap1">

<Paragraph number="1693" type="verses" book="ap1">
_Chappaññāsamhi kappamhi,_  
_sattevāsuṃ mahāyasā;_  
_Sattaratanasampannā,_  
_cakkavattī mahabbalā._  

</Paragraph>

</Division>
<Division number="9" book="ap1">

<Paragraph number="1694" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_vimokkhāpi ca aṭṭhime;_  
_Chaḷabhiññā sacchikatā,_  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="1695" book="ap1">
Itthaṃ sudaṃ āyasmā timirapupphiyo thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="1696" type="center" book="ap1">
Timirapupphiyattherassāpadānaṃ paṭhamaṃ.
</Paragraph>
