<Division number="88" book="ap1">

<Paragraph number="5569" type="verses" book="ap1">
_“Padumuttaro nāma jino,_  
_sabbadhammāna pāragū;_  
_Catusaccaṃ pakāsento,_  
_santāresi bahuṃ janaṃ._  

</Paragraph>

</Division>
<Division number="89" book="ap1">

<Paragraph number="5570" type="verses" book="ap1">
_Ahaṃ tena samayena,_  
_jaṭilo uggatāpano;_  
_Dhunanto vākacīrāni,_  
_gacchāmi ambare tadā._  

</Paragraph>

</Division>
<Division number="90" book="ap1">

<Paragraph number="5571" type="verses" book="ap1">
_Buddhaseṭṭhassa upari,_  
_gantuṃ na visahāmahaṃ;_  
_Pakkhīva selamāsajja,_  
_gamanaṃ na labhāmahaṃ._  

</Paragraph>

</Division>
<Division number="91" book="ap1">

<Paragraph number="5572" type="verses" book="ap1">
_Udake vokkamitvāna,_  
_evaṃ gacchāmi ambare;_  
_Na me idaṃ bhūtapubbaṃ,_  
_iriyāpathavikopanaṃ._  

</Paragraph>

</Division>
<Division number="92" book="ap1">

<Paragraph number="5573" type="verses" book="ap1">
_Handa metaṃ gavesissaṃ,_  
_appevatthaṃ labheyyahaṃ;_  
_Orohanto antalikkhā,_  
_saddamassosi satthuno._  

</Paragraph>

</Division>
<Division number="93" book="ap1">

<Paragraph number="5574" type="verses" book="ap1">
_Sarena rajanīyena,_  
_savanīyena vaggunā;_  
_Aniccataṃ kathentassa,_  
_taññeva uggahiṃ tadā;_  
_Aniccasaññamuggayha,_  
_agamāsiṃ mamassamaṃ._  

</Paragraph>

</Division>
<Division number="94" book="ap1">

<Paragraph number="5575" type="verses" book="ap1">
_Yāvatāyuṃ vasitvāna,_  
_tattha kālaṅkato ahaṃ;_  
_Carime vattamānamhi,_  
_saddhammassavanaṃ sariṃ._  

</Paragraph>

</Division>
<Division number="95" book="ap1">

<Paragraph number="5576" type="verses" book="ap1">
_Tena kammena sukatena,_  
_cetanāpaṇidhīhi ca;_  
_Jahitvā mānusaṃ dehaṃ,_  
_tāvatiṃsamagacchahaṃ._  

</Paragraph>

</Division>
<Division number="96" book="ap1">

<Paragraph number="5577" type="verses" book="ap1">
_Tiṃsakappasahassāni,_  
_devaloke ramiṃ ahaṃ;_  
_Ekapaññāsakkhattuñca,_  
_devarajjamakārayiṃ._  

</Paragraph>

</Division>
<Division number="97" book="ap1">

<Paragraph number="5578" type="verses" book="ap1">
_Ekavīsatikkhattuñca,_  
_cakkavattī ahosahaṃ;_  
_Padesarajjaṃ vipulaṃ,_  
_gaṇanāto asaṅkhiyaṃ._  

</Paragraph>

</Division>
<Division number="98" book="ap1">

<Paragraph number="5579" type="verses" book="ap1">
_Anubhomi sakaṃ puññaṃ,_  
_sukhitohaṃ bhavābhave;_  
_Anussarāmi taṃ saññaṃ,_  
_saṃsaranto bhavābhave;_  
_Na koṭiṃ paṭivijjhāmi,_  
_nibbānaṃ accutaṃ padaṃ._  

</Paragraph>

</Division>
<Division number="99" book="ap1">

<Paragraph number="5580" type="verses" book="ap1">
_Pitugehe nisīditvā,_  
_samaṇo bhāvitindriyo;_  
_Kathaṃsa paridīpento,_  
_aniccatamudāhari._  

</Paragraph>

</Division>
<Division number="100" book="ap1">

<Paragraph number="5581" type="verses" book="ap1">
_‘Aniccā vata saṅkhārā,_  
_uppādavayadhammino;_  
_Uppajjitvā nirujjhanti,_  
_tesaṃ vūpasamo sukho’._  

</Paragraph>

</Division>
<Division number="101" book="ap1">

<Paragraph number="5582" type="verses" book="ap1">
_Saha gāthaṃ suṇitvāna,_  
_pubbasaññamanussariṃ;_  
_Ekāsane nisīditvā,_  
_arahattamapāpuṇiṃ._  

</Paragraph>

</Division>
<Division number="102" book="ap1">

<Paragraph number="5583" type="verses" book="ap1">
_<Emphasis>Jātiyā sattavassena</Emphasis>,_  
_arahattamapāpuṇiṃ;_  
_Upasampādayī buddho,_  
_dhammassavanassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="103" book="ap1">

<Paragraph number="5584" type="verses" book="ap1">
_Satasahassito kappe,_  
_yaṃ dhammamasuṇiṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_dhammassavanassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="104" book="ap1">

<Paragraph number="5585" type="verses" book="ap1">
_Kilesā jhāpitā mayhaṃ,_  
_ …pe… _  
_viharāmi anāsavo._  

</Paragraph>

</Division>
<Division number="105" book="ap1">

<Paragraph number="5586" type="verses" book="ap1">
_Svāgataṃ vata me āsi,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ._  

</Paragraph>

</Division>
<Division number="106" book="ap1">

<Paragraph number="5587" type="verses" book="ap1">

_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="5588" book="ap1">
Itthaṃ sudaṃ āyasmā ekadhammassavaniyo thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="5589" type="center" book="ap1">
Ekadhammassavaniyattherassāpadānaṃ sattamaṃ.
</Paragraph>
