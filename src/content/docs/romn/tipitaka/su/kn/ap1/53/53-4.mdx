_“Nagare bandhumatiyā,_  
_rājuyyāne vasāmahaṃ;_  
_Cammavāsī tadā āsiṃ,_  
_kamaṇḍaludharo ahaṃ._

<Division number="25" book="ap1" bookNo="20" page="431">

<Paragraph number="6828" type="verses" book="ap1">

_sayambhuṃ aparājitaṃ;_  
_Padhānaṃ pahitattaṃ taṃ,_  
_jhāyiṃ jhānarataṃ vasiṃ._  

</Paragraph>

</Division>
<Division number="26" book="ap1" bookNo="20" page="431">

<Paragraph number="6829" type="verses" book="ap1">
_Sabbakāmasamiddhiñca,_  
_oghatiṇṇamanāsavaṃ;_  
_Disvā pasanno sumano,_  
_abbhañjanamadāsahaṃ._  

</Paragraph>

</Division>
<Division number="27" book="ap1" bookNo="20" page="431">

<Paragraph number="6830" type="verses" book="ap1">

_yaṃ dānamadadiṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_abbhañjanassidaṃ phalaṃ._  

</Paragraph>

</Division>
