<Paragraph number="6075" type="verses" book="ap1">
_“Padumuttaro nāma jino,_  
_sabbadhammāna pāragū;_  
_Vivekakāmo sambuddho,_  
_gacchate anilañjase._  

</Paragraph>

</Division>
<Division number="9" book="ap1" bookNo="20" page="429">

<Paragraph number="6076" type="verses" book="ap1">
_Avidūre himavantassa,_  
_mahājātassaro ahu;_  
_Tattha me bhavanaṃ āsi,_  
_puññakammena saṃyutaṃ._  

</Paragraph>

</Division>
<Division number="10" book="ap1" bookNo="20" page="429">

<Paragraph number="6077" type="verses" book="ap1">
_Bhavanā abhinikkhamma,_  
_addasaṃ lokanāyakaṃ;_  
_Indīvaraṃva jalitaṃ,_  
_ādittaṃva hutāsanaṃ._  

</Paragraph>

</Division>
<Division number="11" book="ap1" bookNo="20" page="429">

<Paragraph number="6078" type="verses" book="ap1">
_Vicinaṃ naddasaṃ pupphaṃ,_  
_pūjayissanti nāyakaṃ;_  
_Sakaṃ cittaṃ pasādetvā,_  
_avandiṃ satthuno ahaṃ._  

</Paragraph>

</Division>
<Division number="12" book="ap1" bookNo="20" page="430">

<Paragraph number="6079" type="verses" book="ap1">
_Mama sīse maṇiṃ gayha,_  
_pūjayiṃ lokanāyakaṃ;_  
_Imāya maṇipūjāya,_  
_vipāko hotu bhaddako._  

</Paragraph>

</Division>
<Division number="13" book="ap1" bookNo="20" page="430">

<Paragraph number="6080" type="verses" book="ap1">
_Padumuttaro lokavidū,_  
_āhutīnaṃ paṭiggaho;_  
_Antalikkhe ṭhito satthā,_  
_imaṃ gāthaṃ abhāsatha._  

</Paragraph>

</Division>
<Division number="14" book="ap1" bookNo="20" page="430">

<Paragraph number="6081" type="verses" book="ap1">
_‘So te ijjhatu saṅkappo,_  
_labhassu vipulaṃ sukhaṃ;_  
_Imāya maṇipūjāya,_  
_anubhohi mahāyasaṃ’._  

</Paragraph>

</Division>
<Division number="15" book="ap1" bookNo="20" page="430">

<Paragraph number="6082" type="verses" book="ap1">
_Idaṃ vatvāna bhagavā,_  
_jalajuttamanāmako;_  
_Agamāsi buddhaseṭṭho,_  
_yattha cittaṃ paṇīhitaṃ._  

</Paragraph>

</Division>
<Division number="16" book="ap1" bookNo="20" page="430">

<Paragraph number="6083" type="verses" book="ap1">
_Saṭṭhikappāni devindo,_  
_devarajjamakārayiṃ;_  
_Anekasatakkhattuñca,_  
_cakkavattī ahosahaṃ._  

</Paragraph>

</Division>
<Division number="17" book="ap1" bookNo="20" page="430">

<Paragraph number="6084" type="verses" book="ap1">
_Pubbakammaṃ sarantassa,_  
_devabhūtassa me sato;_  
_Maṇi nibbattate mayhaṃ,_  
_ālokakaraṇo mamaṃ._  

</Paragraph>

</Division>
<Division number="18" book="ap1" bookNo="20" page="430">

<Paragraph number="6085" type="verses" book="ap1">
_Chaḷasītisahassāni,_  
_nāriyo me pariggahā;_  
_Vicittavatthābharaṇā,_  
_āmukkamaṇikuṇḍalā._  

</Paragraph>

</Division>
<Division number="19" book="ap1" bookNo="20" page="430">

<Paragraph number="6086" type="verses" book="ap1">
_Aḷārapamhā hasulā,_  
_susaññā tanumajjhimā;_  
_Parivārenti maṃ niccaṃ,_  
_maṇipūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="20" book="ap1" bookNo="20" page="430">

<Paragraph number="6087" type="verses" book="ap1">
_Soṇṇamayā maṇimayā,_  
_lohitaṅgamayā tathā;_  
_Bhaṇḍā me sukatā honti,_  
_yadicchasi piḷandhanā._  

</Paragraph>

</Division>
<Division number="21" book="ap1" bookNo="20" page="430">

<Paragraph number="6088" type="verses" book="ap1">
_Kūṭāgārā gahārammā,_  
_sayanañca mahārahaṃ;_  
_Mama saṅkappamaññāya,_  
_nibbattanti yadicchakaṃ._  

</Paragraph>

</Division>
<Division number="22" book="ap1" bookNo="20" page="430">

<Paragraph number="6089" type="verses" book="ap1">
_Lābhā tesaṃ suladdhañca,_  
_ye labhanti upassutiṃ;_  
_Puññakkhettaṃ manussānaṃ,_  
_osadhaṃ sabbapāṇinaṃ._  

</Paragraph>

</Division>
<Division number="23" book="ap1" bookNo="20" page="430">

<Paragraph number="6090" type="verses" book="ap1">
_Mayhampi sukataṃ kammaṃ,_  
_yohaṃ adakkhi nāyakaṃ;_  
_Vinipātā pamuttomhi,_  
_pattomhi acalaṃ padaṃ._  

</Paragraph>

</Division>
<Division number="24" book="ap1" bookNo="20" page="431">

<Paragraph number="6091" type="verses" book="ap1">
_Yaṃ yaṃ yonupapajjāmi,_  
_devattaṃ atha mānusaṃ;_  
_Divasañceva rattiñca,_  
_āloko hoti me sadā._  

</Paragraph>

</Division>
<Division number="25" book="ap1" bookNo="20" page="431">

<Paragraph number="6092" type="verses" book="ap1">
_Tāyeva maṇipūjāya,_  
_anubhotvāna sampadā;_  
_Ñāṇāloko mayā diṭṭho,_  
_pattomhi acalaṃ padaṃ._  

</Paragraph>

</Division>
<Division number="26" book="ap1" bookNo="20" page="431">

<Paragraph number="6093" type="verses" book="ap1">
_Satasahassito kappe,_  
_yaṃ maṇiṃ abhipūjayiṃ;_  
_Duggatiṃ nābhijānāmi,_  
_maṇipūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="27" book="ap1" bookNo="20" page="431">

<Paragraph number="6094" type="verses" book="ap1">
_Kilesā jhāpitā mayhaṃ,_  
_ …pe… _  
_viharāmi anāsavo._  

</Paragraph>

</Division>
<Division number="28" book="ap1" bookNo="20" page="431">

<Paragraph number="6095" type="verses" book="ap1">
_Svāgataṃ vata me āsi,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ._  

</Paragraph>

</Division>
<Division number="29" book="ap1" bookNo="20" page="431">

<Paragraph number="6096" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="6097" book="ap1">
Itthaṃ sudaṃ āyasmā maṇipūjako thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="6098" type="center" book="ap1">
Maṇipūjakattherassāpadānaṃ dutiyaṃ.
</Paragraph>
