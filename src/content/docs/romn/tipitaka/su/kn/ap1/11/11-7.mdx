<Paragraph number="1899" type="verses" book="ap1">
_“Padumuttarabuddhassa,_  
_lokajeṭṭhassa tādino;_  
_Tiṇatthare nisinnassa,_  
_upasantassa tādino._  

</Paragraph>

</Division>
<Division number="37" book="ap1" bookNo="20" page="157">

<Paragraph number="1900" type="verses" book="ap1">
_Naḷamālaṃ gahetvāna,_  
_bandhitvā bījaniṃ ahaṃ;_  
_Buddhassa upanāmesiṃ,_  
_dvipadindassa tādino._  

</Paragraph>

</Division>
<Division number="38" book="ap1" bookNo="20" page="157">

<Paragraph number="1901" type="verses" book="ap1">
_Paṭiggahetvā sabbaññū,_  
_bījaniṃ lokanāyako;_  
_Mama saṅkappamaññāya,_  
_imaṃ gāthaṃ abhāsatha._  

</Paragraph>

</Division>
<Division number="39" book="ap1" bookNo="20" page="158">

<Paragraph number="1902" type="verses" book="ap1">
_‘Yathā me kāyo nibbāti,_  
_pariḷāho na vijjati;_  
_Tatheva tividhaggīhi,_  
_cittaṃ tava vimuccatu’._  

</Paragraph>

</Division>
<Division number="40" book="ap1" bookNo="20" page="158">

<Paragraph number="1903" type="verses" book="ap1">
_Sabbe devā samāgacchuṃ,_  
_ye keci vananissitā;_  
_Sossāma buddhavacanaṃ,_  
_hāsayantañca dāyakaṃ._  

</Paragraph>

</Division>
<Division number="41" book="ap1" bookNo="20" page="158">

<Paragraph number="1904" type="verses" book="ap1">
_Nisinno bhagavā tattha,_  
_devasaṅghapurakkhato;_  
_Dāyakaṃ sampahaṃsento,_  
_imā gāthā abhāsatha._  

</Paragraph>

</Division>
<Division number="42" book="ap1" bookNo="20" page="158">

<Paragraph number="1905" type="verses" book="ap1">
_‘Iminā bījanidānena,_  
_cittassa paṇidhīhi ca;_  
_Subbato nāma nāmena,_  
_cakkavattī bhavissati._  

</Paragraph>

</Division>
<Division number="43" book="ap1" bookNo="20" page="158">

<Paragraph number="1906" type="verses" book="ap1">
_Tena kammāvasesena,_  
_sukkamūlena codito;_  
_Māluto nāma nāmena,_  
_cakkavattī bhavissati’._  

</Paragraph>

</Division>
<Division number="44" book="ap1" bookNo="20" page="158">

<Paragraph number="1907" type="verses" book="ap1">
_Iminā bījanidānena,_  
_sammānavipulena ca;_  
_Kappasatasahassampi,_  
_duggatiṃ nupapajjati._  

</Paragraph>

</Division>
<Division number="45" book="ap1" bookNo="20" page="158">

<Paragraph number="1908" type="verses" book="ap1">
_Tiṃsakappasahassamhi,_  
_subbatā aṭṭhatiṃsa te;_  
_Ekūnatiṃsasahasse,_  
_aṭṭha mālutanāmakā._  

</Paragraph>

</Division>
<Division number="46" book="ap1" bookNo="20" page="158">

<Paragraph number="1909" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="1910" book="ap1">
Itthaṃ sudaṃ āyasmā naḷamāliyo thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="1911" type="center" book="ap1">
Naḷamāliyattherassāpadānaṃ sattamaṃ.

  
Sattamabhāṇavāraṃ.
</Paragraph>
