<Paragraph number="5383" type="verses" book="ap1">
_“Parinibbute sugate,_  
_siddhatthe lokanāyake;_  
_Sadevamānusā sabbe,_  
_pūjenti dvipaduttamaṃ._  

</Paragraph>

</Division>
<Division number="125" book="ap1" bookNo="20" page="439">

<Paragraph number="5384" type="verses" book="ap1">
_Āropite ca citake,_  
_siddhatthe lokanāyake;_  
_Yathāsakena thāmena,_  
_citaṃ pūjenti satthuno._  

</Paragraph>

</Division>
<Division number="126" book="ap1" bookNo="20" page="439">

<Paragraph number="5385" type="verses" book="ap1">
_Avidūre citakassa,_  
_dīpaṃ ujjālayiṃ ahaṃ;_  
_Yāva udeti sūriyo,_  
_dīpaṃ me tāva ujjali._  

</Paragraph>

</Division>
<Division number="127" book="ap1" bookNo="20" page="439">

<Paragraph number="5386" type="verses" book="ap1">
_Tena kammena sukatena,_  
_cetanāpaṇidhīhi ca;_  
_Jahitvā mānusaṃ dehaṃ,_  
_tāvatiṃsamagacchahaṃ._  

</Paragraph>

</Division>
<Division number="128" book="ap1" bookNo="20" page="439">

<Paragraph number="5387" type="verses" book="ap1">
_Tattha me sukataṃ byamhaṃ,_  
_ekadīpīti ñāyati;_  
_Dīpasatasahassāni,_  
_byamhe pajjalare mama._  

</Paragraph>

</Division>
<Division number="129" book="ap1" bookNo="20" page="439">

<Paragraph number="5388" type="verses" book="ap1">
_Udayantova sūriyo,_  
_deho me jotate sadā;_  
_Sappabhāhi sarīrassa,_  
_āloko hoti me sadā._  

</Paragraph>

</Division>
<Division number="130" book="ap1" bookNo="20" page="439">

<Paragraph number="5389" type="verses" book="ap1">
_Tirokuṭṭaṃ tiroselaṃ,_  
_samatiggayha pabbataṃ;_  
_Samantā yojanasataṃ,_  
_passāmi cakkhunā ahaṃ._  

</Paragraph>

</Division>
<Division number="131" book="ap1" bookNo="20" page="440">

<Paragraph number="5390" type="verses" book="ap1">
_Sattasattatikkhattuñca,_  
_devaloke ramiṃ ahaṃ;_  
_Ekatiṃsatikkhattuñca,_  
_devarajjamakārayiṃ._  

</Paragraph>

</Division>
<Division number="132" book="ap1" bookNo="20" page="440">

<Paragraph number="5391" type="verses" book="ap1">
_Aṭṭhavīsatikkhattuñca,_  
_cakkavattī ahosahaṃ;_  
_Padesarajjaṃ vipulaṃ,_  
_gaṇanāto asaṅkhiyaṃ._  

</Paragraph>

</Division>
<Division number="133" book="ap1" bookNo="20" page="440">

<Paragraph number="5392" type="verses" book="ap1">
_Devalokā cavitvāna,_  
_nibbattiṃ mātukucchiyaṃ;_  
_Mātukucchigatassāpi,_  
_akkhi me na nimīlati._  

</Paragraph>

</Division>
<Division number="134" book="ap1" bookNo="20" page="440">

<Paragraph number="5393" type="verses" book="ap1">
_Jātiyā catuvassohaṃ,_  
_pabbajiṃ anagāriyaṃ;_  
_Aḍḍhamāse asampatte,_  
_arahattamapāpuṇiṃ._  

</Paragraph>

</Division>
<Division number="135" book="ap1" bookNo="20" page="440">

<Paragraph number="5394" type="verses" book="ap1">
_Dibbacakkhuṃ visodhesiṃ,_  
_bhavā sabbe samūhatā;_  
_Sabbe kilesā sañchinnā,_  
_ekadīpassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="136" book="ap1" bookNo="20" page="440">

<Paragraph number="5395" type="verses" book="ap1">
_Tirokuṭṭaṃ tiroselaṃ,_  
_pabbatañcāpi kevalaṃ;_  
_Samatikkamma passāmi,_  
_ekadīpassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="137" book="ap1" bookNo="20" page="440">

<Paragraph number="5396" type="verses" book="ap1">
_Visamā me samā honti,_  
_andhakāro na vijjati;_  
_Nāhaṃ passāmi timiraṃ,_  
_ekadīpassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="138" book="ap1" bookNo="20" page="440">

<Paragraph number="5397" type="verses" book="ap1">
_Catunnavutito kappe,_  
_yaṃ dīpamadadiṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_ekadīpassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="139" book="ap1" bookNo="20" page="440">

<Paragraph number="5398" type="verses" book="ap1">

_ …pe… _  
_viharāmi anāsavo._  

</Paragraph>

</Division>

_ …pe… _  
_kataṃ buddhassa sāsanaṃ._  

</Paragraph>

</Division>
