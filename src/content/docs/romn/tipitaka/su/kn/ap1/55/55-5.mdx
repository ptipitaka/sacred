<Paragraph number="7485" type="verses" book="ap1">
_“Padumuttaro nāma jino,_  
_sabbadhammesu cakkhumā;_  
_Ito satasahassamhi,_  
_kappe uppajji nāyako._  

</Paragraph>

</Division>
<Division number="144" book="ap1" bookNo="20" page="441">

<Paragraph number="7486" type="verses" book="ap1">
_Hitāya sabbasattānaṃ,_  
_sukhāya vadataṃ varo;_  
_Atthāya purisājañño,_  
_paṭipanno sadevake._  

</Paragraph>

</Division>
<Division number="145" book="ap1" bookNo="20" page="441">

<Paragraph number="7487" type="verses" book="ap1">
_Yasaggapatto sirimā,_  
_kittivaṇṇabhato jino;_  
_Pūjito sabbalokassa,_  
_disā sabbāsu vissuto._  

</Paragraph>

</Division>
<Division number="146" book="ap1" bookNo="20" page="441">

<Paragraph number="7488" type="verses" book="ap1">
_Uttiṇṇavicikiccho so,_  
_vītivattakathaṃkatho;_  
_Paripuṇṇamanasaṅkappo,_  
_patto sambodhimuttamaṃ._  

</Paragraph>

</Division>
<Division number="147" book="ap1" bookNo="20" page="441">

<Paragraph number="7489" type="verses" book="ap1">
_Anuppannassa maggassa,_  
_uppādetā naruttamo;_  
_Anakkhātañca akkhāsi,_  
_asañjātañca sañjanī._  

</Paragraph>

</Division>
<Division number="148" book="ap1" bookNo="20" page="441">

<Paragraph number="7490" type="verses" book="ap1">
_Maggaññū maggavidū ca,_  
_maggakkhāyī narāsabho;_  
_Maggassa kusalo satthā,_  
_sārathīnaṃ varuttamo._  

</Paragraph>

</Division>
<Division number="149" book="ap1" bookNo="20" page="441">

<Paragraph number="7491" type="verses" book="ap1">
_Tadā mahākāruṇiko,_  
_dhammaṃ desesi nāyako;_  
_Nimugge kāmapaṅkamhi,_  
_samuddharati pāṇine._  

</Paragraph>

</Division>
<Division number="150" book="ap1" bookNo="20" page="441">

<Paragraph number="7492" type="verses" book="ap1">
_Bhikkhunīnaṃ ovadane,_  
_sāvakaṃ seṭṭhasammataṃ;_  
_Vaṇṇayaṃ etadaggamhi,_  
_paññapesi mahāmuni._  

</Paragraph>

</Division>
<Division number="151" book="ap1" bookNo="20" page="441">

<Paragraph number="7493" type="verses" book="ap1">
_Taṃ sutvāhaṃ pamudito,_  
_nimantetvā tathāgataṃ;_  
_Bhojayitvā sasaṃghaṃ taṃ,_  
_patthayiṃ ṭhānamuttamaṃ._  

</Paragraph>

</Division>
<Division number="152" book="ap1" bookNo="20" page="441">

<Paragraph number="7494" type="verses" book="ap1">
_Tadā pamudito nātho,_  
_maṃ avoca mahāisi;_  
_‘Sukhī bhavassu dīghāvu,_  
_lacchase taṃ manorathaṃ._  

</Paragraph>

</Division>
<Division number="153" book="ap1" bookNo="20" page="441">

<Paragraph number="7495" type="verses" book="ap1">
_Satasahassito kappe,_  
_okkākakulasambhavo;_  
_Gotamo nāma gottena,_  
_satthā loke bhavissati._  

</Paragraph>

</Division>
<Division number="154" book="ap1" bookNo="20" page="442">

<Paragraph number="7496" type="verses" book="ap1">
_Tassa dhammesu dāyādo,_  
_oraso dhammanimmito;_  
_Nandako nāma nāmena,_  
_hessati satthu sāvako’._  

</Paragraph>

</Division>
<Division number="155" book="ap1" bookNo="20" page="442">

<Paragraph number="7497" type="verses" book="ap1">
_Tena kammena sukatena,_  
_cetanāpaṇidhīhi ca;_  
_Jahitvā mānusaṃ dehaṃ,_  
_tāvatiṃsūpago ahaṃ._  

</Paragraph>

</Division>
<Division number="156" book="ap1" bookNo="20" page="442">

<Paragraph number="7498" type="verses" book="ap1">
_Pacchime ca bhave dāni,_  
_jāto seṭṭhikule ahaṃ;_  
_Sāvatthiyaṃ pure vare,_  
_iddhe phīte mahaddhane._  

</Paragraph>

</Division>
<Division number="157" book="ap1" bookNo="20" page="442">

<Paragraph number="7499" type="verses" book="ap1">
_Purappavese sugataṃ,_  
_disvā vimhitamānaso;_  
_Jetārāmapaṭiggāhe,_  
_pabbajiṃ anagāriyaṃ._  

</Paragraph>

</Division>
<Division number="158" book="ap1" bookNo="20" page="442">

<Paragraph number="7500" type="verses" book="ap1">
_Nacireneva kālena,_  
_arahattamapāpuṇiṃ;_  
_Tatohaṃ tiṇṇasaṃsāro,_  
_sāsito sabbadassinā._  

</Paragraph>

</Division>
<Division number="159" book="ap1" bookNo="20" page="442">

<Paragraph number="7501" type="verses" book="ap1">
_Bhikkhunīnaṃ dhammakathaṃ,_  
_paṭipucchākariṃ ahaṃ;_  
_Sāsitā tā mayā sabbā,_  
_abhaviṃsu anāsavā._  

</Paragraph>

</Division>
<Division number="160" book="ap1" bookNo="20" page="442">

<Paragraph number="7502" type="verses" book="ap1">
_Satāni pañcanūnāni,_  
_tadā tuṭṭho mahāhito;_  
_Bhikkhunīnaṃ ovadataṃ,_  
_aggaṭṭhāne ṭhapesi maṃ._  

</Paragraph>

</Division>
<Division number="161" book="ap1" bookNo="20" page="442">

<Paragraph number="7503" type="verses" book="ap1">
_Satasahasse kataṃ kammaṃ,_  
_phalaṃ dassesi me idha;_  
_Sumutto saravegova,_  
_kilese jhāpayiṃ mama._  

</Paragraph>

</Division>
<Division number="162" book="ap1" bookNo="20" page="442">

<Paragraph number="7504" type="verses" book="ap1">
_Kilesā jhāpitā mayhaṃ,_  
_ …pe… _  
_viharāmi anāsavo._  

</Paragraph>

</Division>
<Division number="163" book="ap1" bookNo="20" page="442">

<Paragraph number="7505" type="verses" book="ap1">
_Svāgataṃ vata me āsi,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ._  

</Paragraph>

</Division>
<Division number="164" book="ap1" bookNo="20" page="442">

<Paragraph number="7506" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="7507" book="ap1">
Itthaṃ sudaṃ āyasmā nandako thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

Nandakattherassāpadānaṃ pañcamaṃ.
