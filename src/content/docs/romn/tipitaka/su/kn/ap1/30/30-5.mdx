_gosīsaṃ santhataṃ mayā;_

_pubbakammassidaṃ phalaṃ._

<Division number="22" book="ap1">

<Paragraph number="3387" type="verses" book="ap1">
_Ājāniyā vātajavā,_  
_sindhavā sīghavāhanā;_  
_Anubhomi sabbametaṃ,_  
_gosīsassa idaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="23" book="ap1">

<Paragraph number="3388" type="verses" book="ap1">
_Aho kāraṃ paramakāraṃ,_  
_sukhette sukataṃ mayā;_  
_Saṃghe katassa kārassa,_  
_na aññaṃ kalamagghati._  

</Paragraph>

</Division>
<Division number="24" book="ap1">

<Paragraph number="3389" type="verses" book="ap1">
_Catunnavutito kappe,_  
_yaṃ sīsaṃ santhariṃ ahaṃ;_  
_Duggatiṃ nābhijānāmi,_  
_santharassa idaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="25" book="ap1">

<Paragraph number="3390" type="verses" book="ap1">
_Pañcasattatikappamhi,_  
_suppatiṭṭhitanāmako;_  
_Eko āsiṃ mahātejo,_  
_cakkavattī mahabbalo._  

</Paragraph>

</Division>
