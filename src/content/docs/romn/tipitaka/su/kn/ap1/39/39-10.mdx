<Division number="64" book="ap1">

<Paragraph number="4258" type="verses" book="ap1">
_Anotattasarāsanne,_  
_ramaṇīye silātale;_  
_Nānāratanapajjote,_  
_nānāgandhavanantare._  



</Paragraph>

</Division>
<Division number="65" book="ap1">

<Paragraph number="4259" type="verses" book="ap1">
_Mahatā bhikkhusaṃghena,_  
_pareto lokanāyako;_  
_Āsīno byākarī tattha,_  
_pubbakammāni attano._  



</Paragraph>

</Division>
<Division number="66" book="ap1">

<Paragraph number="4260" type="verses" book="ap1">
_“Suṇātha bhikkhavo mayhaṃ,_  
_yaṃ kammaṃ pakataṃ mayā;_  
_Pilotikassa kammassa,_  
_buddhattepi vipaccati._  


</Paragraph>
</Division>
