<Paragraph number="2405" type="verses" book="ap1">
_“Suvaccho nāma nāmena,_  
_brāhmaṇo mantapāragū;_  
_Purakkhato sasissehi,_  
_vasate pabbatantare._  

</Paragraph>

</Division>
<Division number="40" book="ap1">

<Paragraph number="2406" type="verses" book="ap1">
_Padumuttaro nāma jino,_  
_āhutīnaṃ paṭiggaho;_  
_Mamuddharitukāmo so,_  
_āgacchi mama santikaṃ._  

</Paragraph>

</Division>
<Division number="41" book="ap1">

<Paragraph number="2407" type="verses" book="ap1">
_Vehāsamhi caṅkamati,_  
_dhūpāyati jalate tathā;_  
_Hāsaṃ mamaṃ viditvāna,_  
_pakkāmi pācināmukho._  

</Paragraph>

</Division>
<Division number="42" book="ap1">

<Paragraph number="2408" type="verses" book="ap1">
_Tañca acchariyaṃ disvā,_  
_abbhutaṃ lomahaṃsanaṃ;_  
_Nāgapupphaṃ gahetvāna,_  
_gatamaggamhi okiriṃ._  

</Paragraph>

</Division>
<Division number="43" book="ap1">

<Paragraph number="2409" type="verses" book="ap1">
_Satasahassito kappe,_  
_yaṃ pupphaṃ okiriṃ ahaṃ;_  
_Tena cittappasādena,_  
_duggatiṃ nupapajjahaṃ._  

</Paragraph>

</Division>
<Division number="44" book="ap1">

<Paragraph number="2410" type="verses" book="ap1">
_Ekattiṃse kappasate,_  
_rājā āsi mahāraho;_  
_Sattaratanasampanno,_  
_cakkavattī mahabbalo._  

</Paragraph>

</Division>
<Division number="45" book="ap1">

<Paragraph number="2411" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="2412" book="ap1">
Itthaṃ sudaṃ āyasmā nāgapupphiyo thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="2413" type="center" book="ap1">
Nāgapupphiyattherassāpadānaṃ aṭṭhamaṃ.
