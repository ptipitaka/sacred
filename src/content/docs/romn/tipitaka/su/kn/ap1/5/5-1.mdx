<Paragraph number="1209" type="verses" book="ap1">
_“Khīṇāsavasahassehi,_  
_parivuto lokanāyako;_  
_Vivekamanuyutto so,_  
_gacchate paṭisallituṃ._  

</Paragraph>

</Division>
<Division number="2" book="ap1" bookNo="20" page="102">

<Paragraph number="1210" type="verses" book="ap1">
_Ajinena nivatthohaṃ,_  
_tidaṇḍaparidhārako;_  
_Bhikkhusaṃghaparibyūḷhaṃ,_  
_addasaṃ lokanāyakaṃ._  

</Paragraph>

</Division>
<Division number="3" book="ap1" bookNo="20" page="102">

<Paragraph number="1211" type="verses" book="ap1">
_Ekaṃsaṃ ajinaṃ katvā,_  
_sire katvāna añjaliṃ;_  
_Sambuddhaṃ abhivādetvā,_  
_santhaviṃ lokanāyakaṃ._  

</Paragraph>

</Division>
<Division number="4" book="ap1" bookNo="20" page="102">

<Paragraph number="1212" type="verses" book="ap1">
_Yathāṇḍajā ca saṃsedā,_  
_opapātī jalābujā;_  
_Kākādipakkhino sabbe,_  
_antalikkhacarā sadā._  

</Paragraph>

</Division>
<Division number="5" book="ap1" bookNo="20" page="102">

<Paragraph number="1213" type="verses" book="ap1">
_‘Ye keci pāṇabhūtatthi,_  
_saññino vā asaññino;_  
_Sabbe te tava ñāṇamhi,_  
_anto honti samogadhā._  

</Paragraph>

</Division>
<Division number="6" book="ap1" bookNo="20" page="102">

<Paragraph number="1214" type="verses" book="ap1">
_Gandhā ca pabbateyyā ye,_  
_himavantanaguttame;_  
_Sabbe te tava sīlamhi,_  
_kalāyapi na yujjare._  

</Paragraph>

</Division>
<Division number="7" book="ap1" bookNo="20" page="102">

<Paragraph number="1215" type="verses" book="ap1">
_Mohandhakārapakkhando,_  
_ayaṃ loko sadevako;_  
_Tava ñāṇamhi jotante,_  
_andhakārā vidhaṃsitā._  

</Paragraph>

</Division>
<Division number="8" book="ap1" bookNo="20" page="102">

<Paragraph number="1216" type="verses" book="ap1">
_Yathā atthaṅgate sūriye,_  
_honti sattā tamogatā;_  
_Evaṃ buddhe anuppanne,_  
_hoti loko tamogato._  

</Paragraph>

</Division>
<Division number="9" book="ap1" bookNo="20" page="102">

<Paragraph number="1217" type="verses" book="ap1">
_Yathodayanto ādicco,_  
_vinodeti tamaṃ sadā;_  
_Tatheva tvaṃ buddhaseṭṭha,_  
_viddhaṃsesi tamaṃ sadā._  

</Paragraph>

</Division>
<Division number="10" book="ap1" bookNo="20" page="102">

<Paragraph number="1218" type="verses" book="ap1">
_Padhānapahitattosi,_  
_buddho loke sadevake;_  
_Tava kammābhiraddhena,_  
_tosesi janataṃ bahuṃ’._  

</Paragraph>

</Division>
<Division number="11" book="ap1" bookNo="20" page="102">

<Paragraph number="1219" type="verses" book="ap1">
_Taṃ sabbaṃ anumoditvā,_  
_padumuttaro mahāmuni;_  
_Nabhaṃ abbhuggamī dhīro,_  
_haṃsarājāva ambare._  

</Paragraph>

</Division>
<Division number="12" book="ap1" bookNo="20" page="102">

<Paragraph number="1220" type="verses" book="ap1">
_Abbhuggantvāna sambuddho,_  
_mahesi padumuttaro;_  
_Antalikkhe ṭhito satthā,_  
_imā gāthā abhāsatha._  

</Paragraph>

</Division>
<Division number="13" book="ap1" bookNo="20" page="103">

<Paragraph number="1221" type="verses" book="ap1">
_Yenidaṃ thavitaṃ ñāṇaṃ,_  
_opammehi samāyutaṃ;_  
_Tamahaṃ kittayissāmi,_  
_suṇātha mama bhāsato._  

</Paragraph>

</Division>
<Division number="14" book="ap1" bookNo="20" page="103">

<Paragraph number="1222" type="verses" book="ap1">
_‘Aṭṭhārasañca khattuṃ so,_  
_devarājā bhavissati;_  
_Pathabyā rajjaṃ tisataṃ,_  
_vasudhaṃ āvasissati._  

</Paragraph>

</Division>
<Division number="15" book="ap1" bookNo="20" page="103">

<Paragraph number="1223" type="verses" book="ap1">
_Pañcavīsatikkhattuñca,_  
_cakkavattī bhavissati;_  
_Padesarajjaṃ vipulaṃ,_  
_gaṇanāto asaṅkhiyaṃ._  

</Paragraph>

</Division>
<Division number="16" book="ap1" bookNo="20" page="103">

<Paragraph number="1224" type="verses" book="ap1">
_Kappasatasahassamhi,_  
_okkākakulasambhavo;_  
_Gotamo nāma gottena,_  
_satthā loke bhavissati._  

</Paragraph>

</Division>
<Division number="17" book="ap1" bookNo="20" page="103">

<Paragraph number="1225" type="verses" book="ap1">
_Tusitā hi cavitvāna,_  
_sukkamūlena codito;_  
_Hīnova jātiyā santo,_  
_upāli nāma hessati._  

</Paragraph>

</Division>
<Division number="18" book="ap1" bookNo="20" page="103">

<Paragraph number="1226" type="verses" book="ap1">
_So pacchā pabbajitvāna,_  
_virājetvāna pāpakaṃ;_  
_Sabbāsave pariññāya,_  
_nibbāyissatināsavo._  

</Paragraph>

</Division>
<Division number="19" book="ap1" bookNo="20" page="103">

<Paragraph number="1227" type="verses" book="ap1">
_Tuṭṭho ca gotamo buddho,_  
_sakyaputto mahāyaso;_  
_Vinayādhigataṃ tassa,_  
_etadagge ṭhapessati’._  

</Paragraph>

</Division>
<Division number="20" book="ap1" bookNo="20" page="103">

<Paragraph number="1228" type="verses" book="ap1">
_Saddhāyāhaṃ pabbajito,_  
_katakicco anāsavo;_  
_Sabbāsave pariññāya,_  
_viharāmi anāsavo._  

</Paragraph>

</Division>
<Division number="21" book="ap1" bookNo="20" page="103">

<Paragraph number="1229" type="verses" book="ap1">
_Bhagavā cānukampī maṃ,_  
_vinayehaṃ visārado;_  
_Sakakammābhiraddho ca,_  
_viharāmi anāsavo._  

</Paragraph>

</Division>
<Division number="22" book="ap1" bookNo="20" page="103">

<Paragraph number="1230" type="verses" book="ap1">
_Saṃvuto pātimokkhamhi,_  
_indriyesu ca pañcasu;_  
_Dhāremi vinayaṃ sabbaṃ,_  
_kevalaṃ ratanākaraṃ._  

</Paragraph>

</Division>
<Division number="23" book="ap1" bookNo="20" page="103">

<Paragraph number="1231" type="verses" book="ap1">
_Mamañca guṇamaññāya,_  
_satthā loke anuttaro;_  
_Bhikkhusaṃghe nisīditvā,_  
_etadagge ṭhapesi maṃ._  

</Paragraph>

</Division>
<Division number="24" book="ap1" bookNo="20" page="103">

<Paragraph number="1232" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_vimokkhāpi ca aṭṭhime;_  
_Chaḷabhiññā sacchikatā,_  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="1233" book="ap1">
Itthaṃ sudaṃ āyasmā upāli thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="1234" type="center" book="ap1">
Bhāgineyyupālittherassāpadānaṃ paṭhamaṃ.
</Paragraph>
