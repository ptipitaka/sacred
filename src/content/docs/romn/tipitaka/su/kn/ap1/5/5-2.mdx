<Division number="25" book="ap1">

<Paragraph number="1235" type="verses" book="ap1">
_“Anomadassissa munino,_  
_lokajeṭṭhassa tādino;_  
_Sudhāya lepanaṃ katvā,_  
_caṅkamaṃ kārayiṃ ahaṃ._  

</Paragraph>

</Division>
<Division number="26" book="ap1">

<Paragraph number="1236" type="verses" book="ap1">
_Nānāvaṇṇehi pupphehi,_  
_caṅkamaṃ santhariṃ ahaṃ;_  
_Ākāse vitānaṃ katvā,_  
_bhojayiṃ buddhamuttamaṃ._  

</Paragraph>

</Division>
<Division number="27" book="ap1">

<Paragraph number="1237" type="verses" book="ap1">
_Añjaliṃ paggahetvāna,_  
_abhivādetvāna subbataṃ;_  
_Dīghasālaṃ bhagavato,_  
_niyyādesimahaṃ tadā._  

</Paragraph>

</Division>
<Division number="28" book="ap1">

<Paragraph number="1238" type="verses" book="ap1">
_Mama saṅkappamaññāya,_  
_satthā loke anuttaro;_  
_Paṭiggahesi bhagavā,_  
_anukampāya cakkhumā._  

</Paragraph>

</Division>
<Division number="29" book="ap1">

<Paragraph number="1239" type="verses" book="ap1">
_Paṭiggahetvāna sambuddho,_  
_dakkhiṇeyyo sadevake;_  
_Bhikkhusaṃghe nisīditvā,_  
_imā gāthā abhāsatha._  

</Paragraph>

</Division>
<Division number="30" book="ap1">

<Paragraph number="1240" type="verses" book="ap1">
_‘Yo so haṭṭhena cittena,_  
_dīghasālaṃ adāsi me;_  
_Tamahaṃ kittayissāmi,_  
_suṇātha mama bhāsato._  

</Paragraph>

</Division>
<Division number="31" book="ap1">

<Paragraph number="1241" type="verses" book="ap1">
_Imassa maccukālamhi,_  
_puññakammasamaṅgino;_  
_Sahassayuttassaratho,_  
_upaṭṭhissati tāvade._  

</Paragraph>

</Division>
<Division number="32" book="ap1">

<Paragraph number="1242" type="verses" book="ap1">
_Tena yānenayaṃ poso,_  
_devalokaṃ gamissati;_  
_Anumodissare devā,_  
_sampatte kusalabbhave._  

</Paragraph>

</Division>
<Division number="33" book="ap1">

<Paragraph number="1243" type="verses" book="ap1">
_Mahārahaṃ byamhaṃ seṭṭhaṃ,_  
_ratanamattikalepanaṃ;_  
_Kūṭāgāravarūpetaṃ,_  
_byamhaṃ ajjhāvasissati._  

</Paragraph>

</Division>
<Division number="34" book="ap1">

<Paragraph number="1244" type="verses" book="ap1">
_Tiṃsakappasahassāni,_  
_devaloke ramissati;_  
_Pañcavīsati kappāni,_  
_devarājā bhavissati._  

</Paragraph>

</Division>
<Division number="35" book="ap1">

<Paragraph number="1245" type="verses" book="ap1">
_Sattasattatikkhattuñca,_  
_cakkavattī bhavissati;_  
_Yasodharasanāmā te,_  
_sabbepi ekanāmakā._  

</Paragraph>

</Division>
<Division number="36" book="ap1">

<Paragraph number="1246" type="verses" book="ap1">
_Dve sampattī anubhotvā,_  
_Vaḍḍhetvā puññasañcayaṃ;_  
_Aṭṭhavīsatikappamhi,_  
_Cakkavattī bhavissati._  

</Paragraph>

</Division>
<Division number="37" book="ap1">

<Paragraph number="1247" type="verses" book="ap1">
_Tatrāpi byamhaṃ pavaraṃ,_  
_vissakammena māpitaṃ;_  
_Dasasaddāvivittaṃ taṃ,_  
_puramajjhāvasissati._  

</Paragraph>

</Division>
<Division number="38" book="ap1">

<Paragraph number="1248" type="verses" book="ap1">
_Aparimeyye ito kappe,_  
_bhūmipālo mahiddhiko;_  
_Okkāko nāma nāmena,_  
_rājā raṭṭhe bhavissati._  

</Paragraph>

</Division>
<Division number="39" book="ap1">

<Paragraph number="1249" type="verses" book="ap1">
_Soḷasitthisahassānaṃ,_  
_sabbāsaṃ pavarā ca sā;_  
_Abhijātā khattiyānī,_  
_nava putte janessati._  

</Paragraph>

</Division>
<Division number="40" book="ap1">

<Paragraph number="1250" type="verses" book="ap1">
_Nava putte janetvāna,_  
_khattiyānī marissati;_  
_Taruṇī ca piyā kaññā,_  
_mahesittaṃ karissati._  

</Paragraph>

</Division>
<Division number="41" book="ap1">

<Paragraph number="1251" type="verses" book="ap1">
_Okkākaṃ tosayitvāna,_  
_varaṃ kaññā labhissati;_  
_Varaṃ laddhāna sā kaññā,_  
_putte pabbājayissati._  

</Paragraph>

</Division>
<Division number="42" book="ap1">

<Paragraph number="1252" type="verses" book="ap1">
_Pabbājitā ca te sabbe,_  
_gamissanti naguttamaṃ;_  
_Jātibhedabhayā sabbe,_  
_bhaginīhi vasissare._  

</Paragraph>

</Division>
<Division number="43" book="ap1">

<Paragraph number="1253" type="verses" book="ap1">
_Ekā ca kaññā byādhīhi,_  
_bhavissati parikkhatā;_  
_Mā no jāti pabhijjīti,_  
_nikhaṇissanti khattiyā._  

</Paragraph>

</Division>
<Division number="44" book="ap1">

<Paragraph number="1254" type="verses" book="ap1">
_Khattiyo nīharitvāna,_  
_tāya saddhiṃ vasissati;_  
_Bhavissati tadā bhedo,_  
_okkākakulasambhavo._  

</Paragraph>

</Division>
<Division number="45" book="ap1">

<Paragraph number="1255" type="verses" book="ap1">
_Tesaṃ pajā bhavissanti,_  
_koḷiyā nāma jātiyā;_  
_Tattha mānusakaṃ bhogaṃ,_  
_anubhossatinappakaṃ._  

</Paragraph>

</Division>
<Division number="46" book="ap1">

<Paragraph number="1256" type="verses" book="ap1">
_Tamhā kāyā cavitvāna,_  
_devalokaṃ gamissati;_  
_Tatrāpi pavaraṃ byamhaṃ,_  
_labhissati manoramaṃ._  

</Paragraph>

</Division>
<Division number="47" book="ap1">

<Paragraph number="1257" type="verses" book="ap1">
_Devalokā cavitvāna,_  
_sukkamūlena codito;_  
_Āgantvāna manussattaṃ,_  
_soṇo nāma bhavissati._  

</Paragraph>

</Division>
<Division number="48" book="ap1">

<Paragraph number="1258" type="verses" book="ap1">
_Āraddhavīriyo pahitatto,_  
_Padahaṃ satthu sāsane;_  
_Sabbāsave pariññāya,_  
_Nibbāyissatināsavo._  

</Paragraph>

</Division>
<Division number="49" book="ap1">

<Paragraph number="1259" type="verses" book="ap1">
_Anantadassī bhagavā,_  
_gotamo sakyapuṅgavo;_  
_Visesaññū mahāvīro,_  
_aggaṭṭhāne ṭhapessati’._  

</Paragraph>

</Division>
<Division number="50" book="ap1">

<Paragraph number="1260" type="verses" book="ap1">
_Vuṭṭhamhi deve caturaṅgulamhi,_  
_Tiṇe anileritaaṅgaṇamhi;_  
_Ṭhatvāna yogassa payuttatādino,_  
_Tatottariṃ pāramatā na vijjati._  

</Paragraph>

</Division>
<Division number="51" book="ap1">

<Paragraph number="1261" type="verses" book="ap1">
_Uttame damathe danto,_  
_cittaṃ me supaṇīhitaṃ;_  
_Bhāro me ohito sabbo,_  
_nibbutomhi anāsavo._  

</Paragraph>

</Division>
<Division number="52" book="ap1">

<Paragraph number="1262" type="verses" book="ap1">
_Aṅgīraso mahānāgo,_  
_abhijātova kesarī;_  
_Bhikkhusaṃghe nisīditvā,_  
_etadagge ṭhapesi maṃ._  

</Paragraph>

</Division>
<Division number="53" book="ap1">

<Paragraph number="1263" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="1264" book="ap1">
Itthaṃ sudaṃ āyasmā soṇo koḷiviso thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="1265" type="center" book="ap1">
Soṇakoḷivisattherassāpadānaṃ dutiyaṃ.
</Paragraph>
