<Paragraph number="5361" type="verses" book="ap1">
_“Nagare bandhumatiyā,_  
_sūkariko ahosahaṃ;_  
_Ukkoṭakaṃ randhayitvā,_  
_madhumaṃsamhi okiriṃ._  

</Paragraph>

</Division>
<Division number="107" book="ap1">

<Paragraph number="5362" type="verses" book="ap1">
_Sannipātaṃ ahaṃ gantvā,_  
_ekaṃ pattaṃ gahesahaṃ;_  
_Pūrayitvāna taṃ pattaṃ,_  
_bhikkhusaṃghassadāsahaṃ._  

</Paragraph>

</Division>
<Division number="108" book="ap1">

<Paragraph number="5363" type="verses" book="ap1">
_Yottha therataro bhikkhu,_  
_niyyādesi mamaṃ tadā;_  
_Iminā pattapūrena,_  
_labhassu vipulaṃ sukhaṃ._  

</Paragraph>

</Division>
<Division number="109" book="ap1">

<Paragraph number="5364" type="verses" book="ap1">
_Duve sampattiyo bhutvā,_  
_sukkamūlena codito;_  
_Pacchime vattamānamhi,_  
_kilese jhāpayissati._  

</Paragraph>

</Division>
<Division number="110" book="ap1">

<Paragraph number="5365" type="verses" book="ap1">
_Tattha cittaṃ pasādetvā,_  
_tāvatiṃsamagacchahaṃ;_  
_Tattha bhutvā pivitvā ca,_  
_labhāmi vipulaṃ sukhaṃ._  

</Paragraph>

</Division>
<Division number="111" book="ap1">

<Paragraph number="5366" type="verses" book="ap1">
_Maṇḍape rukkhamūle vā,_  
_pubbakammaṃ anussariṃ;_  
_Annapānābhivasso me,_  
_abhivassati tāvade._  

</Paragraph>

</Division>
<Division number="112" book="ap1">

<Paragraph number="5367" type="verses" book="ap1">
_Idaṃ pacchimakaṃ mayhaṃ,_  
_carimo vattate bhavo;_  
_Idhāpi annapānaṃ me,_  
_vassate sabbakālikaṃ._  

</Paragraph>

</Division>
<Division number="113" book="ap1">

<Paragraph number="5368" type="verses" book="ap1">
_Teneva madhudānena,_  
_sandhāvitvā bhave ahaṃ;_  
_Sabbāsave pariññāya,_  
_viharāmi anāsavo._  

</Paragraph>

</Division>
<Division number="114" book="ap1">

<Paragraph number="5369" type="verses" book="ap1">
_Ekanavutito kappe,_  
_yaṃ dānamadadiṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_madhudānassidaṃ phalaṃ._  

</Paragraph>

</Division>

_ …pe… _  
_viharāmi anāsavo._  

</Paragraph>

</Division>

_ …pe… _

</Paragraph>

_ …pe… _
