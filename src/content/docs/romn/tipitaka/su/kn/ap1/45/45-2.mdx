<Division number="9" book="ap1">

<Paragraph number="5785" type="verses" book="ap1">

_eko adutiyo ahu;_  
_Mamassamaṃ upāgacchi,_  
_jānanto sabbakālikaṃ._  

</Paragraph>

</Division>
<Division number="10" book="ap1">

<Paragraph number="5786" type="verses" book="ap1">
_Sakaṃ cittaṃ pasādetvā,_  
_vanditvāna ca subbataṃ;_  
_Ubho hatthehi paggayha,_  
_kolaṃ buddhassadāsahaṃ._  

</Paragraph>

</Division>
<Division number="11" book="ap1">

<Paragraph number="5787" type="verses" book="ap1">

_yaṃ phalamadadiṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_koladānassidaṃ phalaṃ._  

</Paragraph>

</Division>
