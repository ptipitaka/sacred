<Paragraph number="3301" type="verses" book="ap1">
_“Sīhaṃ yathā vanacaraṃ,_  
_nisabhājāniyaṃ yathā;_  
_Kakudhaṃ vilasantaṃva,_  
_āgacchantaṃ narāsabhaṃ._  

</Paragraph>

</Division>
<Division number="12" book="ap1" bookNo="20" page="271">

<Paragraph number="3302" type="verses" book="ap1">
_Siddhatthaṃ lokapajjotaṃ,_  
_sabbalokatikicchakaṃ;_  
_Akāsiṃ paccuggamanaṃ,_  
_vippasannena cetasā._  

</Paragraph>

</Division>
<Division number="13" book="ap1" bookNo="20" page="271">

<Paragraph number="3303" type="verses" book="ap1">
_Catunnavutito kappe,_  
_paccuggacchiṃ narāsabhaṃ;_  
_Duggatiṃ nābhijānāmi,_  
_paccuggamane idaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="14" book="ap1" bookNo="20" page="271">

<Paragraph number="3304" type="verses" book="ap1">
_Sattatiṃse ito kappe,_  
_eko āsiṃ janādhipo;_  
_Saparivāroti nāmena,_  
_cakkavattī mahabbalo._  

</Paragraph>

</Division>
<Division number="15" book="ap1" bookNo="20" page="271">

<Paragraph number="3305" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="3306" book="ap1">
Itthaṃ sudaṃ āyasmā paccuggamaniyo thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="3307" type="center" book="ap1">
Paccuggamaniyattherassāpadānaṃ tatiyaṃ.
</Paragraph>
