<Paragraph number="758" type="verses" book="ap1">
_“Nibbute lokanāthamhi,_  
_sumedhe aggapuggale;_  
_Pasannacitto sumano,_  
_thūpapūjaṃ akāsahaṃ._  

</Paragraph>

</Division>
<Division number="56" book="ap1" bookNo="20" page="65">

<Paragraph number="759" type="verses" book="ap1">
_Ye ca khīṇāsavā tattha,_  
_chaḷabhiññā mahiddhikā;_  
_Tehaṃ tattha samānetvā,_  
_saṃghabhattaṃ akāsahaṃ._  

</Paragraph>

</Division>
<Division number="57" book="ap1" bookNo="20" page="65">

<Paragraph number="760" type="verses" book="ap1">
_Sumedhassa bhagavato,_  
_upaṭṭhāko tadā ahu;_  
_Sumedho nāma nāmena,_  
_anumodittha so tadā._  

</Paragraph>

</Division>
<Division number="58" book="ap1" bookNo="20" page="65">

<Paragraph number="761" type="verses" book="ap1">
_Tena cittappasādena,_  
_vimānaṃ upapajjahaṃ;_  
_Chaḷāsītisahassāni,_  
_accharāyo ramiṃsu me._  

</Paragraph>

</Division>
<Division number="59" book="ap1" bookNo="20" page="65">

<Paragraph number="762" type="verses" book="ap1">
_Mameva anuvattanti,_  
_sabbakāmehi tā sadā;_  
_Aññe deve abhibhomi,_  
_puññakammassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="60" book="ap1" bookNo="20" page="65">

<Paragraph number="763" type="verses" book="ap1">
_Pañcavīsatikappamhi,_  
_varuṇo nāma khattiyo;_  
_Visuddhabhojano āsiṃ,_  
_cakkavattī ahaṃ tadā._  

</Paragraph>

</Division>
<Division number="61" book="ap1" bookNo="20" page="65">

<Paragraph number="764" type="verses" book="ap1">
_Na te bījaṃ pavapanti,_  
_napi nīyanti naṅgalā;_  
_Akaṭṭhapākimaṃ sāliṃ,_  
_paribhuñjanti mānusā._  

</Paragraph>

</Division>
<Division number="62" book="ap1" bookNo="20" page="65">

<Paragraph number="765" type="verses" book="ap1">
_Tattha rajjaṃ karitvāna,_  
_devattaṃ puna gacchahaṃ;_  
_Tadāpi edisā mayhaṃ,_  
_nibbattā bhogasampadā._  

</Paragraph>

</Division>
<Division number="63" book="ap1" bookNo="20" page="65">

<Paragraph number="766" type="verses" book="ap1">
_Na maṃ mittā amittā vā,_  
_hiṃsanti sabbapāṇino;_  
_Sabbesampi piyo homi,_  
_puññakammassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="64" book="ap1" bookNo="20" page="65">

<Paragraph number="767" type="verses" book="ap1">
_Tiṃsakappasahassamhi,_  
_yaṃ dānamadadiṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_gandhālepassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="65" book="ap1" bookNo="20" page="65">

<Paragraph number="768" type="verses" book="ap1">
_Imasmiṃ bhaddake kappe,_  
_eko āsiṃ janādhipo;_  
_Mahānubhāvo rājāhaṃ,_  
_cakkavattī mahabbalo._  

</Paragraph>

</Division>
<Division number="66" book="ap1" bookNo="20" page="65">

<Paragraph number="769" type="verses" book="ap1">
_Sohaṃ pañcasu sīlesu,_  
_ṭhapetvā janataṃ bahuṃ;_  
_Pāpetvā sugatiṃyeva,_  
_devatānaṃ piyo ahuṃ._  

</Paragraph>

</Division>
<Division number="67" book="ap1" bookNo="20" page="66">

<Paragraph number="770" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_vimokkhāpi ca aṭṭhime;_  
_Chaḷabhiññā sacchikatā,_  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="771" book="ap1">
Itthaṃ sudaṃ āyasmā pilindavaccho thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="772" type="center" book="ap1">
Pilindavacchattherassāpadānaṃ pañcamaṃ.
</Paragraph>
