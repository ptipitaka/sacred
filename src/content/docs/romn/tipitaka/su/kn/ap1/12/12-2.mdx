<Paragraph number="1953" type="verses" book="ap1">
_“Atthadassī jinavaro,_  
_lokajeṭṭho narāsabho;_  
_Vihārā abhinikkhamma,_  
_taḷākaṃ upasaṅkami._  

</Paragraph>

</Division>
<Division number="12" book="ap1" bookNo="20" page="162">

<Paragraph number="1954" type="verses" book="ap1">
_Nhatvā pitvā ca sambuddho,_  
_uttaritvekacīvaro;_  
_Aṭṭhāsi bhagavā tattha,_  
_vilokento disodisaṃ._  

</Paragraph>

</Division>
<Division number="13" book="ap1" bookNo="20" page="162">

<Paragraph number="1955" type="verses" book="ap1">
_Bhavane upaviṭṭhohaṃ,_  
_addasaṃ lokanāyakaṃ;_  
_Haṭṭho haṭṭhena cittena,_  
_apphoṭesiṃ ahaṃ tadā._  

</Paragraph>

</Division>
<Division number="14" book="ap1" bookNo="20" page="162">

<Paragraph number="1956" type="verses" book="ap1">
_Sataraṃsiṃva jotantaṃ,_  
_pabhāsantaṃva kañcanaṃ;_  
_Naccagīte payuttohaṃ,_  
_pañcaṅgatūriyamhi ca._  

</Paragraph>

</Division>
<Division number="15" book="ap1" bookNo="20" page="162">

<Paragraph number="1957" type="verses" book="ap1">
_Yaṃ yaṃ yonupapajjāmi,_  
_devattaṃ atha mānusaṃ;_  
_Sabbe satte abhibhomi,_  
_vipulo hoti me yaso._  

</Paragraph>

</Division>
<Division number="16" book="ap1" bookNo="20" page="162">

<Paragraph number="1958" type="verses" book="ap1">
_Namo te purisājañña,_  
_namo te purisuttama;_  
_Attānaṃ tosayitvāna,_  
_pare tosesi tvaṃ muni._  

</Paragraph>

</Division>
<Division number="17" book="ap1" bookNo="20" page="162">

<Paragraph number="1959" type="verses" book="ap1">
_Pariggahe nisīditvā,_  
_hāsaṃ katvāna subbate;_  
_Upaṭṭhahitvā sambuddhaṃ,_  
_tusitaṃ upapajjahaṃ._  

</Paragraph>

</Division>
<Division number="18" book="ap1" bookNo="20" page="162">

<Paragraph number="1960" type="verses" book="ap1">
_Soḷaseto kappasate,_  
_dvinavaekacintitā;_  
_Sattaratanasampannā,_  
_cakkavattī mahabbalā._  

</Paragraph>

</Division>
<Division number="19" book="ap1" bookNo="20" page="162">

<Paragraph number="1961" type="verses" book="ap1">

_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="1962" book="ap1">

</Paragraph>
</Division>
