<Paragraph number="5084" type="verses" book="ap1">
_“Rājāsi vijayo nāma,_  
_ketumatīpuruttame;_  
_Sūro vikkamasampanno,_  
_puramajjhāvasī tadā._  

</Paragraph>

</Division>
<Division number="225" book="ap1" bookNo="20" page="415">

<Paragraph number="5085" type="verses" book="ap1">
_Tassa rañño pamattassa,_  
_aṭaviyo samuṭṭhahuṃ;_  
_Otārā tuṇḍikā ceva,_  
_raṭṭhaṃ viddhaṃsayuṃ tadā._  

</Paragraph>

</Division>
<Division number="226" book="ap1" bookNo="20" page="415">

<Paragraph number="5086" type="verses" book="ap1">
_Paccante kupite khippaṃ,_  
_sannipātesirindamo;_  
_Bhaṭe ceva balatthe ca,_  
_ariṃ niggāhayi tadā._  

</Paragraph>

</Division>
<Division number="227" book="ap1" bookNo="20" page="415">

<Paragraph number="5087" type="verses" book="ap1">
_Hatthārohā anīkaṭṭhā,_  
_sūrā ca cammayodhino;_  
_Dhanuggahā ca uggā ca,_  
_sabbe sannipatuṃ tadā._  

</Paragraph>

</Division>
<Division number="228" book="ap1" bookNo="20" page="415">

<Paragraph number="5088" type="verses" book="ap1">
_Āḷārikā ca kappakā,_  
_nhāpakā mālakārakā;_  
_Sūrā vijitasaṅgāmā,_  
_sabbe sannipatuṃ tadā._  

</Paragraph>

</Division>
<Division number="229" book="ap1" bookNo="20" page="415">

<Paragraph number="5089" type="verses" book="ap1">
_Khaggahatthā ca purisā,_  
_cāpahatthā ca vammino;_  
_Luddā vijitasaṅgāmā,_  
_sabbe sannipatuṃ tadā._  

</Paragraph>

</Division>
<Division number="230" book="ap1" bookNo="20" page="416">

<Paragraph number="5090" type="verses" book="ap1">
_Tidhāpabhinnā mātaṅgā,_  
_kuñjarā saṭṭhihāyanā;_  
_Suvaṇṇakacchālaṅkārā,_  
_sabbe sannipatuṃ tadā._  

</Paragraph>

</Division>
<Division number="231" book="ap1" bookNo="20" page="416">

<Paragraph number="5091" type="verses" book="ap1">
_Khamā sītassa uṇhassa,_  
_ukkāruharaṇassa ca;_  
_Yodhājīvā katakammā,_  
_sabbe sannipatuṃ tadā._  

</Paragraph>

</Division>
<Division number="232" book="ap1" bookNo="20" page="416">

<Paragraph number="5092" type="verses" book="ap1">
_Saṅkhasaddaṃ bherisaddaṃ,_  
_atho utujasaddakaṃ;_  
_Etehi te hāsayantā,_  
_sabbe sannipatuṃ tadā._  

</Paragraph>

</Division>
<Division number="233" book="ap1" bookNo="20" page="416">

<Paragraph number="5093" type="verses" book="ap1">
_Tisūlakontimantehi ca,_  
_kavacatomarehi ca;_  
_Koṭṭentānaṃ nipātentā,_  
_sabbe sannipatuṃ tadā._  

</Paragraph>

</Division>
<Division number="234" book="ap1" bookNo="20" page="416">

<Paragraph number="5094" type="verses" book="ap1">
_Kimevātinisāmetvā,_  
_sarājā ajitaṃ jino;_  
_Saṭṭhi pāṇasahassāni,_  
_sūle uttāsayiṃ tadā._  

</Paragraph>

</Division>
<Division number="235" book="ap1" bookNo="20" page="416">

<Paragraph number="5095" type="verses" book="ap1">
_Saddaṃ mānusakākaṃsu,_  
_‘aho rājā adhammiko;_  
_Niraye paccamānassa,_  
_kadā anto bhavissati’._  

</Paragraph>

</Division>
<Division number="236" book="ap1" bookNo="20" page="416">

<Paragraph number="5096" type="verses" book="ap1">
_Sayanehaṃ tuvaṭṭento,_  
_passāmi niraye tadā;_  
_Na supāmi divārattiṃ,_  
_sūlena tajjayanti maṃ._  

</Paragraph>

</Division>
<Division number="237" book="ap1" bookNo="20" page="416">

<Paragraph number="5097" type="verses" book="ap1">
_Kiṃ pamādena rajjena,_  
_vāhanena balena ca;_  
_Na te pahonti dhāretuṃ,_  
_tāpayanti mamaṃ sadā._  

</Paragraph>

</Division>
<Division number="238" book="ap1" bookNo="20" page="416">

<Paragraph number="5098" type="verses" book="ap1">
_Kiṃ me puttehi dārehi,_  
_rajjena sakalena ca;_  
_Yannūna pabbajeyyāhaṃ,_  
_gatimaggaṃ visodhaye._  

</Paragraph>

</Division>
<Division number="239" book="ap1" bookNo="20" page="416">

<Paragraph number="5099" type="verses" book="ap1">
_Saṭṭhi nāgasahassāni,_  
_sabbālaṅkārabhūsite;_  
_Suvaṇṇakacche mātaṅge,_  
_hemakappanavāsase._  

</Paragraph>

</Division>
<Division number="240" book="ap1" bookNo="20" page="416">

<Paragraph number="5100" type="verses" book="ap1">
_Ārūḷhe gāmaṇīyehi,_  
_tomaraṅkusapāṇibhi;_  
_Saṅgāmāvacare ṭhāne,_  
_anapekkho vihāyahaṃ;_  
_Sakakammena santatto,_  
_nikkhamiṃ anagāriyaṃ._  

</Paragraph>

</Division>
<Division number="241" book="ap1" bookNo="20" page="417">

<Paragraph number="5101" type="verses" book="ap1">
_Saṭṭhi assasahassāni,_  
_sabbālaṅkārabhūsite;_  
_Ājānīyeva jātiyā,_  
_sindhave sīghavāhane._  

</Paragraph>

</Division>
<Division number="242" book="ap1" bookNo="20" page="417">

<Paragraph number="5102" type="verses" book="ap1">
_Ārūḷhe gāmaṇīyehi,_  
_cāpahatthehi vammibhi;_  
_Pahāretvāna te sabbe,_  
_nikkhamiṃ anagāriyaṃ._  

</Paragraph>

</Division>
<Division number="243" book="ap1" bookNo="20" page="417">

<Paragraph number="5103" type="verses" book="ap1">
_Saṭṭhi rathasahassāni,_  
_sabbālaṅkārabhūsite;_  
_Dīpe athopi veyagghe,_  
_sannaddhe ussitaddhaje;_  
_Te sabbe parihāretvā,_  
_pabbajiṃ anagāriyaṃ._  

</Paragraph>

</Division>
<Division number="244" book="ap1" bookNo="20" page="417">

<Paragraph number="5104" type="verses" book="ap1">
_Saṭṭhi dhenusahassāni,_  
_sabbā kaṃsūpadhāraṇā;_  
_Tāyopi chaḍḍayitvāna,_  
_pabbajiṃ anagāriyaṃ._  

</Paragraph>

</Division>
<Division number="245" book="ap1" bookNo="20" page="417">

<Paragraph number="5105" type="verses" book="ap1">
_Saṭṭhi itthisahassāni,_  
_sabbālaṅkārabhūsitā;_  
_Vicittavatthābharaṇā,_  
_āmukkamaṇikuṇḍalā._  

</Paragraph>

</Division>
<Division number="246" book="ap1" bookNo="20" page="417">

<Paragraph number="5106" type="verses" book="ap1">
_Aḷārapamhā hasulā,_  
_susaññā tanumajjhimā;_  
_Tā hitvā kandamānāyo,_  
_pabbajiṃ anagāriyaṃ._  

</Paragraph>

</Division>
<Division number="247" book="ap1" bookNo="20" page="417">

<Paragraph number="5107" type="verses" book="ap1">
_Saṭṭhi gāmasahassāni,_  
_paripuṇṇāni sabbaso;_  
_Chaḍḍayitvāna taṃ rajjaṃ,_  
_pabbajiṃ anagāriyaṃ._  

</Paragraph>

</Division>
<Division number="248" book="ap1" bookNo="20" page="417">

<Paragraph number="5108" type="verses" book="ap1">
_Nagarā nikkhamitvāna,_  
_himavantamupāgamiṃ;_  
_Bhāgīrathīnadītīre,_  
_assamaṃ māpayiṃ ahaṃ._  

</Paragraph>

</Division>
<Division number="249" book="ap1" bookNo="20" page="417">

<Paragraph number="5109" type="verses" book="ap1">
_Paṇṇasālaṃ karitvāna,_  
_Agyāgāraṃ akāsahaṃ;_  
_Āraddhavīriyo pahitatto,_  
_Vasāmi assame ahaṃ._  

</Paragraph>

</Division>
<Division number="250" book="ap1" bookNo="20" page="417">

<Paragraph number="5110" type="verses" book="ap1">
_Maṇḍape rukkhamūle vā,_  
_suññāgāre ca jhāyato;_  
_Na tu vijjati tāso me,_  
_na passe bhayabheravaṃ._  

</Paragraph>

</Division>
<Division number="251" book="ap1" bookNo="20" page="417">

<Paragraph number="5111" type="verses" book="ap1">
_Sumedho nāma sambuddho,_  
_aggo kāruṇiko muni;_  
_Ñāṇālokena jotanto,_  
_loke uppajji tāvade._  

</Paragraph>

</Division>
<Division number="252" book="ap1" bookNo="20" page="417">

<Paragraph number="5112" type="verses" book="ap1">
_Mama assamasāmantā,_  
_yakkho āsi mahiddhiko;_  
_Buddhaseṭṭhamhi uppanne,_  
_ārocesi mamaṃ tadā._  

</Paragraph>

</Division>
<Division number="253" book="ap1" bookNo="20" page="418">

<Paragraph number="5113" type="verses" book="ap1">
_‘Buddho loke samuppanno,_  
_sumedho nāma cakkhumā;_  
_Tāreti janataṃ sabbaṃ,_  
_tampi so tārayissati’._  

</Paragraph>

</Division>
<Division number="254" book="ap1" bookNo="20" page="418">

<Paragraph number="5114" type="verses" book="ap1">
_Yakkhassa vacanaṃ sutvā,_  
_saṃviggo āsi tāvade;_  
_Buddho buddhoti cintento,_  
_assamaṃ paṭisāmayiṃ._  

</Paragraph>

</Division>
<Division number="255" book="ap1" bookNo="20" page="418">

<Paragraph number="5115" type="verses" book="ap1">
_Aggidāruñca chaḍḍetvā,_  
_saṃsāmetvāna santhataṃ;_  
_Assamaṃ abhivanditvā,_  
_nikkhamiṃ vipinā ahaṃ._  

</Paragraph>

</Division>
<Division number="256" book="ap1" bookNo="20" page="418">

<Paragraph number="5116" type="verses" book="ap1">
_Tato candanamādāya,_  
_gāmā gāmaṃ purā puraṃ;_  
_Devadevaṃ gavesanto,_  
_upagacchiṃ vināyakaṃ._  

</Paragraph>

</Division>
<Division number="257" book="ap1" bookNo="20" page="418">

<Paragraph number="5117" type="verses" book="ap1">
_Bhagavā tamhi samaye,_  
_sumedho lokanāyako;_  
_Catusaccaṃ pakāsento,_  
_bodheti janataṃ bahuṃ._  

</Paragraph>

</Division>
<Division number="258" book="ap1" bookNo="20" page="418">

<Paragraph number="5118" type="verses" book="ap1">
_Añjaliṃ paggahetvāna,_  
_sīse katvāna candanaṃ;_  
_Sambuddhaṃ abhivādetvā,_  
_imā gāthā abhāsahaṃ._  

</Paragraph>

</Division>
<Division number="259" book="ap1" bookNo="20" page="418">

<Paragraph number="5119" type="verses" book="ap1">
_‘Vassike pupphamānamhi,_  
_santike upavāyati;_  
_Tvaṃ vīra guṇagandhena,_  
_disā sabbā pavāyasi._  

</Paragraph>

</Division>
<Division number="260" book="ap1" bookNo="20" page="418">

<Paragraph number="5120" type="verses" book="ap1">
_Campake nāgavanike,_  
_atimuttakaketake;_  
_Sālesu pupphamānesu,_  
_anuvātaṃ pavāyati._  

</Paragraph>

</Division>
<Division number="261" book="ap1" bookNo="20" page="418">

<Paragraph number="5121" type="verses" book="ap1">
_Tava gandhaṃ suṇitvāna,_  
_himavantā idhāgamiṃ;_  
_Pūjemi taṃ mahāvīra,_  
_lokajeṭṭha mahāyasa’._  

</Paragraph>

</Division>
<Division number="262" book="ap1" bookNo="20" page="418">

<Paragraph number="5122" type="verses" book="ap1">
_Varacandanenānulimpiṃ,_  
_sumedhaṃ lokanāyakaṃ;_  
_Sakaṃ cittaṃ pasādetvā,_  
_tuṇhī aṭṭhāsi tāvade._  

</Paragraph>

</Division>
<Division number="263" book="ap1" bookNo="20" page="418">

<Paragraph number="5123" type="verses" book="ap1">

_lokajeṭṭho narāsabho;_  
_Bhikkhusaṃghe nisīditvā,_  
_imā gāthā abhāsatha._  

</Paragraph>

</Division>
<Division number="264" book="ap1" bookNo="20" page="418">

_candanañca apūjayi;_  
_Tamahaṃ kittayissāmi,_  
_suṇātha mama bhāsato._  

</Paragraph>

</Division>

_brahmā uju patāpavā;_  
_Pañcavīsatikappāni,_  
_sappabhāso bhavissati._  

</Paragraph>

</Division>

_devaloke ramissati;_  
_Sahassakkhattuṃ rājā ca,_  
_cakkavattī bhavissati._  

</Paragraph>

</Division>

_devarajjaṃ karissati;_  
_Padesarajjaṃ vipulaṃ,_  
_gaṇanāto asaṅkhiyaṃ._  

</Paragraph>

</Division>

</Paragraph>

_bāvarī nāma brāhmaṇo._  

</Paragraph>

</Division>

</Paragraph>
