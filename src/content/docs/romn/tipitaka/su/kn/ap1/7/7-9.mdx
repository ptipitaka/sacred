<Paragraph number="1568" type="verses" book="ap1">
_“Sumano nāma nāmena,_  
_mālākāro ahaṃ tadā;_  
_Addasaṃ virajaṃ buddhaṃ,_  
_lokāhutipaṭiggahaṃ._  

</Paragraph>

</Division>
<Division number="63" book="ap1" bookNo="20" page="130">

<Paragraph number="1569" type="verses" book="ap1">
_Ubho hatthehi paggayha,_  
_sumanaṃ pupphamuttamaṃ;_  
_Buddhassa abhiropesiṃ,_  
_sikhino lokabandhuno._  

</Paragraph>

</Division>
<Division number="64" book="ap1" bookNo="20" page="130">

<Paragraph number="1570" type="verses" book="ap1">
_Imāya pupphapūjāya,_  
_cetanāpaṇidhīhi ca;_  
_Duggatiṃ nābhijānāmi,_  
_buddhapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="65" book="ap1" bookNo="20" page="130">

<Paragraph number="1571" type="verses" book="ap1">
_Ekattiṃse ito kappe,_  
_yaṃ pupphamabhiropayiṃ;_  
_Duggatiṃ nābhijānāmi,_  
_buddhapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="66" book="ap1" bookNo="20" page="130">

<Paragraph number="1572" type="verses" book="ap1">
_Chabbīsatimhi kappamhi,_  
_cattārosuṃ mahāyasā;_  
_Sattaratanasampannā,_  
_rājāno cakkavattino._  

</Paragraph>

</Division>
<Division number="67" book="ap1" bookNo="20" page="130">

<Paragraph number="1573" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="1574" book="ap1">
Itthaṃ sudaṃ āyasmā sumano thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="1575" type="center" book="ap1">
Sumanattherassāpadānaṃ navamaṃ.
</Paragraph>
