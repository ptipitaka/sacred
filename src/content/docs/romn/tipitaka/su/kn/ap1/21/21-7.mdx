<Paragraph number="2791" type="verses" book="ap1">
_“Kaṇikāraṃva jotantaṃ,_  
_nisinnaṃ pabbatantare;_  
_Obhāsentaṃ disā sabbā,_  
_siddhatthaṃ narasārathiṃ._  

</Paragraph>

</Division>
<Division number="31" book="ap1" bookNo="20" page="230">

<Paragraph number="2792" type="verses" book="ap1">
_Dhanuṃ advejjhaṃ katvāna,_  
_usuṃ sannayhahaṃ tadā;_  
_Pupphaṃ savaṇṭaṃ chetvāna,_  
_buddhassa abhiropayiṃ._  

</Paragraph>

</Division>
<Division number="32" book="ap1" bookNo="20" page="230">

<Paragraph number="2793" type="verses" book="ap1">
_Catunnavutito kappe,_  
_yaṃ pupphamabhiropayiṃ;_  
_Duggatiṃ nābhijānāmi,_  
_buddhapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="33" book="ap1" bookNo="20" page="230">

<Paragraph number="2794" type="verses" book="ap1">
_Ekapaññāsito kappe,_  
_eko āsiṃ jutindharo;_  
_Sattaratanasampanno,_  
_cakkavattī mahabbalo._  

</Paragraph>

</Division>
<Division number="34" book="ap1" bookNo="20" page="230">

<Paragraph number="2795" type="verses" book="ap1">

_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="2796" book="ap1">

</Paragraph>
</Division>
