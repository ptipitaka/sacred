<Paragraph number="2441" type="verses" book="ap1">
_“Siddhattho nāma bhagavā,_  
_lokajeṭṭho narāsabho;_  
_Purakkhato sāvakehi,_  
_nagaraṃ paṭipajjatha._  

</Paragraph>

</Division>
<Division number="8" book="ap1" bookNo="20" page="201">

<Paragraph number="2442" type="verses" book="ap1">
_Rañño antepure āsiṃ,_  
_gopako abhisammato;_  
_Pāsāde upaviṭṭhohaṃ,_  
_addasaṃ lokanāyakaṃ._  

</Paragraph>

</Division>
<Division number="9" book="ap1" bookNo="20" page="201">

<Paragraph number="2443" type="verses" book="ap1">
_Kaṇaveraṃ gahetvāna,_  
_bhikkhusaṃghe samokiriṃ;_  
_Buddhassa visuṃ katvāna,_  
_tato bhiyyo samokiriṃ._  

</Paragraph>

</Division>
<Division number="10" book="ap1" bookNo="20" page="202">

<Paragraph number="2444" type="verses" book="ap1">
_Catunnavutito kappe,_  
_Yaṃ pupphamabhipūjayiṃ;_  
_Duggatiṃ nābhijānāmi,_  
_Buddhapūjāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="11" book="ap1" bookNo="20" page="202">

<Paragraph number="2445" type="verses" book="ap1">
_Sattāsītimhito kappe,_  
_caturāsuṃ mahiddhikā;_  
_Sattaratanasampannā,_  
_cakkavattī mahabbalā._  

</Paragraph>

</Division>
<Division number="12" book="ap1" bookNo="20" page="202">

<Paragraph number="2446" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="2447" book="ap1">
Itthaṃ sudaṃ āyasmā kaṇaverapupphiyo thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="2448" type="center" book="ap1">
Kaṇaverapupphiyattherassāpadānaṃ dutiyaṃ.
</Paragraph>
