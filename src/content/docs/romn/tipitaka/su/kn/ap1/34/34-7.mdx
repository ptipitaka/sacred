<Division number="74" book="ap1">

<Paragraph number="3800" type="verses" book="ap1">
_“Nagare bandhumatiyā,_  
_ahosiṃ parakammiko;_  
_Parakammāyane yutto,_  
_parabhattaṃ apassito._  

</Paragraph>

</Division>
<Division number="75" book="ap1">

<Paragraph number="3801" type="verses" book="ap1">
_Rahogato nisīditvā,_  
_evaṃ cintesahaṃ tadā;_  
_Buddho loke samuppanno,_  
_adhikāro ca natthi me._  

</Paragraph>

</Division>
<Division number="76" book="ap1">

<Paragraph number="3802" type="verses" book="ap1">
_Kālo me gatiṃ sodhetuṃ,_  
_khaṇo me paṭipādito;_  
_Dukkho nirayasamphasso,_  
_apuññānañhi pāṇinaṃ._  

</Paragraph>

</Division>
<Division number="77" book="ap1">

<Paragraph number="3803" type="verses" book="ap1">
_Evāhaṃ cintayitvāna,_  
_kammasāmiṃ upāgamiṃ;_  
_Ekāhaṃ kammaṃ yācitvā,_  
_vipinaṃ pāvisiṃ ahaṃ._  

</Paragraph>

</Division>
<Division number="78" book="ap1">

<Paragraph number="3804" type="verses" book="ap1">
_Tiṇakaṭṭhañca valliñca,_  
_āharitvānahaṃ tadā;_  
_Tidaṇḍake ṭhapetvāna,_  
_akaṃ tiṇakuṭiṃ ahaṃ._  

</Paragraph>

</Division>
<Division number="79" book="ap1">

<Paragraph number="3805" type="verses" book="ap1">
_Saṃghassatthāya kuṭikaṃ,_  
_niyyādetvāna taṃ ahaṃ;_  
_Tadaheyeva āgantvā,_  
_kammasāmiṃ upāgamiṃ._  

</Paragraph>

</Division>
<Division number="80" book="ap1">

<Paragraph number="3806" type="verses" book="ap1">
_Tena kammena sukatena,_  
_tāvatiṃsamagacchahaṃ;_  
_Tattha me sukataṃ byamhaṃ,_  
_kuṭikāya sunimmitaṃ._  

</Paragraph>

</Division>
<Division number="81" book="ap1">

<Paragraph number="3807" type="verses" book="ap1">
_Sahassakaṇḍaṃ satabheṇḍu,_  
_dhajālu haritāmayaṃ;_  
_Satasahassaniyyūhā,_  
_byamhe pātubhaviṃsu me._  

</Paragraph>

</Division>
<Division number="82" book="ap1">

<Paragraph number="3808" type="verses" book="ap1">
_Yaṃ yaṃ yonupapajjāmi,_  
_devattaṃ atha mānusaṃ;_  
_Mama saṅkappamaññāya,_  
_pāsādo upatiṭṭhati._  

</Paragraph>

</Division>
<Division number="83" book="ap1">

<Paragraph number="3809" type="verses" book="ap1">
_Bhayaṃ vā chambhitattaṃ vā,_  
_lomahaṃso na vijjati;_  
_Tāsaṃ mama na jānāmi,_  
_tiṇakuṭikāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="84" book="ap1">

<Paragraph number="3810" type="verses" book="ap1">
_Sīhabyagghā ca dīpī ca,_  
_acchakokataracchakā;_  
_Sabbe maṃ parivajjenti,_  
_tiṇakuṭikāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="85" book="ap1">

<Paragraph number="3811" type="verses" book="ap1">
_Sarīsapā ca bhūtā ca,_  
_ahī kumbhaṇḍarakkhasā;_  
_Tepi maṃ parivajjenti,_  
_tiṇakuṭikāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="86" book="ap1">

<Paragraph number="3812" type="verses" book="ap1">
_Na pāpasupinassāpi,_  
_sarāmi dassanaṃ mama;_  
_Upaṭṭhitā sati mayhaṃ,_  
_tiṇakuṭikāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="87" book="ap1">

<Paragraph number="3813" type="verses" book="ap1">
_Tāyeva tiṇakuṭikāya,_  
_anubhotvāna sampadā;_  
_Gotamassa bhagavato,_  
_dhammaṃ sacchikariṃ ahaṃ._  

</Paragraph>

</Division>
<Division number="88" book="ap1">

<Paragraph number="3814" type="verses" book="ap1">
_Ekanavutito kappe,_  
_yaṃ kammamakariṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_tiṇakuṭikāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="89" book="ap1">

<Paragraph number="3815" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="3816" book="ap1">
Itthaṃ sudaṃ āyasmā tiṇakuṭidāyako thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="3817" type="center" book="ap1">
Tiṇakuṭidāyakattherassāpadānaṃ sattamaṃ.
</Paragraph>
