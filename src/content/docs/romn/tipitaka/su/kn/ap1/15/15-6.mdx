<Paragraph number="2314" type="verses" book="ap1">
_Disvā taṃ sukataṃ kammaṃ,_  
_anomadassī naruttamo;_  
_Bhikkhusaṃghe ṭhito satthā,_  
_imaṃ gāthaṃ abhāsatha._  

</Paragraph>

</Division>

<Paragraph number="2315" type="verses" book="ap1">
_‘Iminā sudhakammena,_  
_cetanāpaṇidhīhi ca;_  
_Sampattiṃ anubhotvāna,_  
_dukkhassantaṃ karissati’._  

</Paragraph>

</Division>
<Division number="29" book="ap1" bookNo="20" page="191">

<Paragraph number="2316" type="verses" book="ap1">
_Pasannamukhavaṇṇomhi,_  
_ekaggo susamāhito;_  
_Dhāremi antimaṃ dehaṃ,_  
_sammāsambuddhasāsane._  

</Paragraph>

</Division>
<Division number="30" book="ap1" bookNo="20" page="191">

<Paragraph number="2317" type="verses" book="ap1">

_paripuṇṇe anūnake;_  
_Rājā sabbaghano nāma,_  
_cakkavattī mahabbalo._  

</Paragraph>

</Division>
