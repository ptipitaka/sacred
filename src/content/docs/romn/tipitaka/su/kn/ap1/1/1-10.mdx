<Division number="613" book="ap1">

<Paragraph number="635" type="verses" book="ap1">
_“Padumuttaro nāma jino,_  
_sayambhū aggapuggalo;_  
_Purato himavantassa,_  
_cittakūṭe vasī tadā._  

</Paragraph>

</Division>
<Division number="614" book="ap1">

<Paragraph number="636" type="verses" book="ap1">
_Abhītarūpo tatthāsiṃ,_  
_migarājā catukkamo;_  
_Tassa saddaṃ suṇitvāna,_  
_vikkhambhanti bahujjanā._  

</Paragraph>

</Division>
<Division number="615" book="ap1">

<Paragraph number="637" type="verses" book="ap1">
_Suphullaṃ padumaṃ gayha,_  
_upagacchiṃ narāsabhaṃ;_  
_Vuṭṭhitassa samādhimhā,_  
_buddhassa abhiropayiṃ._  

</Paragraph>

</Division>
<Division number="616" book="ap1">

<Paragraph number="638" type="verses" book="ap1">
_Catuddisaṃ namassitvā,_  
_buddhaseṭṭhaṃ naruttamaṃ;_  
_Sakaṃ cittaṃ pasādetvā,_  
_sīhanādaṃ nadiṃ ahaṃ._  

</Paragraph>

</Division>
<Division number="617" book="ap1">

<Paragraph number="639" type="verses" book="ap1">
_Padumuttaro lokavidū,_  
_āhutīnaṃ paṭiggaho;_  
_Sakāsane nisīditvā,_  
_imā gāthā abhāsatha._  

</Paragraph>

</Division>
<Division number="618" book="ap1">

<Paragraph number="640" type="verses" book="ap1">
_Buddhassa giramaññāya,_  
_Sabbe devā samāgatā;_  
_‘Āgato vadataṃ seṭṭho,_  
_Dhammaṃ sossāma taṃ mayaṃ’._  

</Paragraph>

</Division>
<Division number="619" book="ap1">

<Paragraph number="641" type="verses" book="ap1">
_Tesaṃ hāsaparetānaṃ,_  
_purato lokanāyako;_  
_Mama saddaṃ pakittesi,_  
_dīghadassī mahāmuni._  

</Paragraph>

</Division>
<Division number="620" book="ap1">

<Paragraph number="642" type="verses" book="ap1">
_Yenidaṃ padumaṃ dinnaṃ,_  
_sīhanādo ca nādito;_  
_Tamahaṃ kittayissāmi,_  
_suṇātha mama bhāsato._  

</Paragraph>

</Division>
<Division number="621" book="ap1">

<Paragraph number="643" type="verses" book="ap1">
_Ito aṭṭhamake kappe,_  
_cakkavattī bhavissati;_  
_Sattaratanasampanno,_  
_catudīpamhi issaro._  

</Paragraph>

</Division>
<Division number="622" book="ap1">

<Paragraph number="644" type="verses" book="ap1">
_Kārayissati issariyaṃ,_  
_mahiyā catusaṭṭhiyā;_  
_Padumo nāma nāmena,_  
_cakkavattī mahabbalo._  

</Paragraph>

</Division>
<Division number="623" book="ap1">

<Paragraph number="645" type="verses" book="ap1">
_Kappasatasahassamhi,_  
_okkākakulasambhavo;_  
_Gotamo nāma gottena,_  
_satthā loke bhavissati._  

</Paragraph>

</Division>
<Division number="624" book="ap1">

<Paragraph number="646" type="verses" book="ap1">
_Pakāsite pāvacane,_  
_brahmabandhu bhavissati;_  
_Brahmaññā abhinikkhamma,_  
_pabbajissati tāvade._  

</Paragraph>

</Division>
<Division number="625" book="ap1">

<Paragraph number="647" type="verses" book="ap1">
_Padhānapahitatto so,_  
_upasanto nirūpadhi;_  
_Sabbāsave pariññāya,_  
_nibbāyissatināsavo._  

</Paragraph>

</Division>
<Division number="626" book="ap1">

<Paragraph number="648" type="verses" book="ap1">
_Vijane pantaseyyamhi,_  
_vāḷamigasamākule;_  
_Sabbāsave pariññāya,_  
_nibbāyissatināsavo._  

</Paragraph>

</Division>
<Division number="627" book="ap1">

<Paragraph number="649" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_vimokkhāpi ca aṭṭhime;_  
_Chaḷabhiññā sacchikatā,_  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="650" book="ap1">
Itthaṃ sudaṃ āyasmā piṇḍolabhāradvājo thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="651" type="center" book="ap1">
Piṇḍolabhāradvājattherassāpadānaṃ aṭṭhamaṃ.
</Paragraph>
