_“Himavantassāvidūre,_  
_asoko nāma pabbato;_  
_Tatthāsi assamo mayhaṃ,_  
_vissakammena māpito._  

</Paragraph>

</Division>
<Division number="46" book="ap1" bookNo="20" page="400">

<Paragraph number="4896" type="verses" book="ap1">
_Sumedho nāma sambuddho,_  
_aggo kāruṇiko muni;_  
_Nivāsayitvā pubbaṇhe,_  
_piṇḍāya me upāgami._  

</Paragraph>

</Division>
<Division number="47" book="ap1" bookNo="20" page="401">

<Paragraph number="4897" type="verses" book="ap1">
_Upāgataṃ mahāvīraṃ,_  
_sumedhaṃ lokanāyakaṃ;_  
_Paggayha sugatapattaṃ,_  
_sappitelaṃ apūrayiṃ._  

</Paragraph>

</Division>
<Division number="48" book="ap1" bookNo="20" page="401">

<Paragraph number="4898" type="verses" book="ap1">
_Datvānahaṃ buddhaseṭṭhe,_  
_sumedhe lokanāyake;_  
_Añjaliṃ paggahetvāna,_  
_bhiyyo hāsaṃ janesahaṃ._  

</Paragraph>

</Division>
<Division number="49" book="ap1" bookNo="20" page="401">

<Paragraph number="4899" type="verses" book="ap1">
_Iminā sappidānena,_  
_cetanāpaṇidhīhi ca;_  
_Devabhūto manusso vā,_  
_labhāmi vipulaṃ sukhaṃ._  

</Paragraph>

</Division>
<Division number="50" book="ap1" bookNo="20" page="401">

<Paragraph number="4900" type="verses" book="ap1">
_Vinipātaṃ vivajjetvā,_  
_saṃsarāmi bhavābhave;_  
_Tattha cittaṃ paṇidhitvā,_  
_labhāmi acalaṃ padaṃ._  

</Paragraph>

</Division>
<Division number="51" book="ap1" bookNo="20" page="401">

<Paragraph number="4901" type="verses" book="ap1">
_‘Lābhā tuyhaṃ suladdhaṃ te,_  
_yaṃ maṃ addakkhi brāhmaṇa;_  
_Mama dassanamāgamma,_  
_arahattaṃ bhavissati._  

</Paragraph>

</Division>
<Division number="52" book="ap1" bookNo="20" page="401">

<Paragraph number="4902" type="verses" book="ap1">
_Vissattho hohi mā bhāyi,_  
_adhigantvā mahāyasaṃ;_  
_Mamañhi sappiṃ datvāna,_  
_parimokkhasi jātiyā’._  

</Paragraph>

</Division>
<Division number="53" book="ap1" bookNo="20" page="401">

<Paragraph number="4903" type="verses" book="ap1">
_Iminā sappidānena,_  
_cetanāpaṇidhīhi ca;_  
_Devabhūto manusso vā,_  
_labhase vipulaṃ sukhaṃ._  

</Paragraph>

</Division>
<Division number="54" book="ap1" bookNo="20" page="401">

<Paragraph number="4904" type="verses" book="ap1">
_Iminā sappidānena,_  
_mettacittavatāya ca;_  
_Aṭṭhārase kappasate,_  
_devaloke ramissasi._  

</Paragraph>

</Division>
<Division number="55" book="ap1" bookNo="20" page="401">

<Paragraph number="4905" type="verses" book="ap1">
_Aṭṭhatiṃsatikkhattuñca,_  
_devarājā bhavissasi;_  
_Padesarajjaṃ vipulaṃ,_  
_gaṇanāto asaṅkhiyaṃ._  

</Paragraph>

</Division>
<Division number="56" book="ap1" bookNo="20" page="401">

<Paragraph number="4906" type="verses" book="ap1">
_Ekapaññāsakkhattuñca,_  
_cakkavattī bhavissasi;_  
_Cāturanto vijitāvī,_  
_jambumaṇḍassa issaro._  

</Paragraph>

</Division>
<Division number="57" book="ap1" bookNo="20" page="401">

<Paragraph number="4907" type="verses" book="ap1">
_Mahāsamuddovakkhobho,_  
_duddharo pathavī yathā;_  
_Evamevaṃ ca te bhogā,_  
_appameyyā bhavissare._  

</Paragraph>

</Division>
<Division number="58" book="ap1" bookNo="20" page="401">

<Paragraph number="4908" type="verses" book="ap1">
_Saṭṭhikoṭī hiraññassa,_  
_cajitvā pabbajiṃ ahaṃ;_  
_Kiṃ kusalaṃ gavesanto,_  
_bāvariṃ upasaṅkamiṃ._  

</Paragraph>

</Division>
<Division number="59" book="ap1" bookNo="20" page="402">

<Paragraph number="4909" type="verses" book="ap1">
_Tattha mante adhīyāmi,_  
_chaḷaṅgaṃ nāma lakkhaṇaṃ;_  
_Tamandhakāraṃ vidhamaṃ,_  
_uppajji tvaṃ mahāmuni._  

</Paragraph>

</Division>
<Division number="60" book="ap1" bookNo="20" page="402">

<Paragraph number="4910" type="verses" book="ap1">
_Tava dassanakāmohaṃ,_  
_āgatomhi mahāmuni;_  
_Tava dhammaṃ suṇitvāna,_  
_pattomhi acalaṃ padaṃ._  

</Paragraph>

</Division>
<Division number="61" book="ap1" bookNo="20" page="402">

<Paragraph number="4911" type="verses" book="ap1">
_Tiṃsakappasahassamhi,_  
_sappiṃ buddhassadāsahaṃ;_  
_Etthantare nābhijāne,_  
_sappiṃ viññāpitaṃ mayā._  

</Paragraph>

</Division>
<Division number="62" book="ap1" bookNo="20" page="402">

<Paragraph number="4912" type="verses" book="ap1">
_Mama saṅkappamaññāya,_  
_uppajjati yadicchakaṃ;_  
_Cittamaññāya nibbattaṃ,_  
_sabbe santappayāmahaṃ._  

</Paragraph>

</Division>
<Division number="63" book="ap1" bookNo="20" page="402">

<Paragraph number="4913" type="verses" book="ap1">
_Aho buddhā aho dhammā,_  
_aho no satthu sampadā;_  
_Thokañhi sappiṃ datvāna,_  
_appameyyaṃ labhāmahaṃ._  

</Paragraph>

</Division>
<Division number="64" book="ap1" bookNo="20" page="402">

<Paragraph number="4914" type="verses" book="ap1">
_Mahāsamudde udakaṃ,_  
_yāvatā nerupassato;_  
_Mama sappiṃ upādāya,_  
_kalabhāgaṃ na hessati._  

</Paragraph>

</Division>
<Division number="65" book="ap1" bookNo="20" page="402">

<Paragraph number="4915" type="verses" book="ap1">
_Yāvatā cakkavāḷassa,_  
_kariyantassa rāsito;_  
_Mama nibbattavatthānaṃ,_  
_okāso so na sammati._  

</Paragraph>

</Division>
<Division number="66" book="ap1" bookNo="20" page="402">

<Paragraph number="4916" type="verses" book="ap1">
_Pabbatarājā himavā,_  
_pavaropi siluccayo;_  
_Mamānulittagandhassa,_  
_upanidhiṃ na hessati._  

</Paragraph>

</Division>
<Division number="67" book="ap1" bookNo="20" page="402">

<Paragraph number="4917" type="verses" book="ap1">
_Vatthaṃ gandhañca sappiñca,_  
_aññañca diṭṭhadhammikaṃ;_  
_Asaṅkhatañca nibbānaṃ,_  
_sappidānassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="68" book="ap1" bookNo="20" page="402">

<Paragraph number="4918" type="verses" book="ap1">
_Satipaṭṭhānasayano,_  
_samādhijhānagocaro;_  
_Bojjhaṅgabhojano ajja,_  
_sappidānassidaṃ phalaṃ._  

</Paragraph>

</Division>
