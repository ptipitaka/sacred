<Paragraph number="2223" type="verses" book="ap1">
_“Himavantassāvidūre,_  
_vasabho nāma pabbato;_  
_Tasmiṃ pabbatapādamhi,_  
_assamo āsi māpito._  

</Paragraph>

</Division>
<Division number="35" book="ap1">

<Paragraph number="2224" type="verses" book="ap1">
_Tīṇi sissasahassāni,_  
_vācesiṃ brāhmaṇo tadā;_  
_Saṃharitvāna te sisse,_  
_ekamantaṃ upāvisiṃ._  

</Paragraph>

</Division>
<Division number="36" book="ap1">

<Paragraph number="2225" type="verses" book="ap1">
_Ekamantaṃ nisīditvā,_  
_brāhmaṇo mantapāragū;_  
_Buddhavedaṃ gavesanto,_  
_ñāṇe cittaṃ pasādayiṃ._  

</Paragraph>

</Division>
<Division number="37" book="ap1">

<Paragraph number="2226" type="verses" book="ap1">
_Tattha cittaṃ pasādetvā,_  
_nisīdiṃ paṇṇasanthare;_  
_Pallaṅkaṃ ābhujitvāna,_  
_tattha kālaṅkato ahaṃ._  

</Paragraph>

</Division>
<Division number="38" book="ap1">

<Paragraph number="2227" type="verses" book="ap1">
_Ekattiṃse ito kappe,_  
_yaṃ saññamalabhiṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_ñāṇasaññāyidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="39" book="ap1">

<Paragraph number="2228" type="verses" book="ap1">
_Sattavīsatikappamhi,_  
_rājā siridharo ahu;_  
_Sattaratanasampanno,_  
_cakkavattī mahabbalo._  

</Paragraph>

</Division>
<Division number="40" book="ap1">

<Paragraph number="2229" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="2230" book="ap1">
Itthaṃ sudaṃ āyasmā rahosaññako thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>
