<Paragraph number="1630" type="verses" book="ap1">
_“Tiṃsakappasahassamhi,_  
_sambuddho lokanāyako;_  
_Sumedho nāma nāmena,_  
_bāttiṃsavaralakkhaṇo._  

</Paragraph>

</Division>
<Division number="31" book="ap1" bookNo="20" page="135">

<Paragraph number="1631" type="verses" book="ap1">
_Tassa kañcanavaṇṇassa,_  
_dvipadindassa tādino;_  
_Pañca sūcī mayā dinnā,_  
_sibbanatthāya cīvaraṃ._  

</Paragraph>

</Division>
<Division number="32" book="ap1" bookNo="20" page="135">

<Paragraph number="1632" type="verses" book="ap1">
_Teneva sūcidānena,_  
_nipuṇatthavipassakaṃ;_  
_Tikkhaṃ lahuñca phāsuñca,_  
_ñāṇaṃ me udapajjatha._  

</Paragraph>

</Division>
<Division number="33" book="ap1" bookNo="20" page="135">

<Paragraph number="1633" type="verses" book="ap1">
_Kilesā jhāpitā mayhaṃ,_  
_bhavā sabbe samūhatā;_  
_Dhāremi antimaṃ dehaṃ,_  
_sammāsambuddhasāsane._  

</Paragraph>

</Division>
<Division number="34" book="ap1" bookNo="20" page="135">

<Paragraph number="1634" type="verses" book="ap1">
_Dvipadādhipatī nāma,_  
_rājāno caturo ahuṃ;_  
_Sattaratanasampannā,_  
_cakkavattī mahabbalā._  

</Paragraph>

</Division>
<Division number="35" book="ap1" bookNo="20" page="135">

<Paragraph number="1635" type="verses" book="ap1">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="1636" book="ap1">
Itthaṃ sudaṃ āyasmā sūcidāyako thero imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="1637" type="center" book="ap1">
Sūcidāyakattherassāpadānaṃ sattamaṃ.
</Paragraph>
