<Paragraph number="231" type="verses" book="vv">
_“Nīlā pītā ca kāḷā ca,_  
_mañjiṭṭhā atha lohitā;_  
_Uccāvacānaṃ vaṇṇānaṃ,_  
_kiñjakkhaparivāritā._  

</Paragraph>

</Division>
<Division number="207" book="vv" bookNo="19" page="26">

<Paragraph number="232" type="verses" book="vv">
_Mandāravānaṃ pupphānaṃ,_  
_mālaṃ dhāresi muddhani;_  
_Nayime aññesu kāyesu,_  
_rukkhā santi sumedhase._  

</Paragraph>

</Division>
<Division number="208" book="vv" bookNo="19" page="27">

<Paragraph number="233" type="verses" book="vv">
_Kena kāyaṃ upapannā,_  
_tāvatiṃsaṃ yasassinī;_  
_Devate pucchitācikkha,_  
_kissa kammassidaṃ phalan”ti._  

</Paragraph>

</Division>
<Division number="209" book="vv" bookNo="19" page="27">

<Paragraph number="234" type="verses" book="vv">
_“Bhadditthikāti maṃ aññaṃsu,_  
_Kimilāyaṃ upāsikā;_  
_Saddhā sīlena sampannā,_  
_Saṃvibhāgaratā sadā._  

</Paragraph>

</Division>
<Division number="210" book="vv" bookNo="19" page="27">

<Paragraph number="235" type="verses" book="vv">
_Acchādanañca bhattañca,_  
_senāsanaṃ padīpiyaṃ;_  
_Adāsiṃ ujubhūtesu,_  
_vippasannena cetasā._  

</Paragraph>

</Division>
<Division number="211" book="vv" bookNo="19" page="27">

<Paragraph number="236" type="verses" book="vv">
_Cātuddasiṃ pañcadasiṃ,_  
_yā ca pakkhassa aṭṭhamī;_  
_Pāṭihāriyapakkhañca,_  
_aṭṭhaṅgasusamāgataṃ._  

</Paragraph>

</Division>
<Division number="212" book="vv" bookNo="19" page="27">

<Paragraph number="237" type="verses" book="vv">
_Uposathaṃ upavasissaṃ,_  
_sadā sīlesu saṃvutā;_  
_Saññamā saṃvibhāgā ca,_  
_vimānaṃ āvasāmahaṃ._  

</Paragraph>

</Division>
<Division number="213" book="vv" bookNo="19" page="27">

<Paragraph number="238" type="verses" book="vv">
_Pāṇātipātā viratā,_  
_musāvādā ca saññatā;_  
_Theyyā ca aticārā ca,_  
_majjapānā ca ārakā._  

</Paragraph>

</Division>
<Division number="214" book="vv" bookNo="19" page="27">

<Paragraph number="239" type="verses" book="vv">
_Pañcasikkhāpade ratā,_  
_Ariyasaccāna kovidā;_  
_Upāsikā cakkhumato,_  
_Appamādavihārinī;_  
_Katāvāsā katakusalā tato cutā,_  
_Sayaṃpabhā anuvicarāmi nandanaṃ._  

</Paragraph>

</Division>
<Division number="215" book="vv" bookNo="19" page="27">

<Paragraph number="240" type="verses" book="vv">
_Bhikkhū cāhaṃ paramahitānukampake,_  
_Abhojayiṃ tapassiyugaṃ mahāmuniṃ;_  
_Katāvāsā katakusalā tato cutā,_  
_Sayaṃpabhā anuvicarāmi nandanaṃ._  

</Paragraph>

</Division>
<Division number="216" book="vv" bookNo="19" page="27">

<Paragraph number="241" type="verses" book="vv">
_Aṭṭhaṅgikaṃ aparimitaṃ sukhāvahaṃ,_  
_Uposathaṃ satatamupāvasiṃ ahaṃ;_  
_Katāvāsā katakusalā tato cutā,_  
_Sayaṃpabhā anuvicarāmi nandanan”ti._  


</Paragraph>
</Division>

<Paragraph number="242" type="center" book="vv">
Bhadditthivimānaṃ pañcamaṃ.
</Paragraph>
