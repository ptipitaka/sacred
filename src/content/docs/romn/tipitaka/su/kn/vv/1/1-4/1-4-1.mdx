<Paragraph number="535" type="verses" book="vv">
_“Mañjiṭṭhake vimānasmiṃ,_  
_soṇṇavālukasanthate;_  
_Pañcaṅgikena tūriyena,_  
_ramasi suppavādite._  

</Paragraph>

</Division>
<Division number="690" book="vv" bookNo="19" page="56">

<Paragraph number="536" type="verses" book="vv">
_Tamhā vimānā oruyha,_  
_nimmitā ratanāmayā;_  
_Ogāhasi sālavanaṃ,_  
_pupphitaṃ sabbakālikaṃ._  

</Paragraph>

</Division>
<Division number="691" book="vv" bookNo="19" page="56">

<Paragraph number="537" type="verses" book="vv">
_Yassa yasseva sālassa,_  
_mūle tiṭṭhasi devate;_  
_So so muñcati pupphāni,_  
_onamitvā dumuttamo._  

</Paragraph>

</Division>
<Division number="692" book="vv" bookNo="19" page="56">

<Paragraph number="538" type="verses" book="vv">
_Vāteritaṃ sālavanaṃ,_  
_ādhutaṃ dijasevitaṃ;_  
_Vāti gandho disā sabbā,_  
_rukkho mañjūsako yathā._  

</Paragraph>

</Division>
<Division number="693" book="vv" bookNo="19" page="56">

<Paragraph number="539" type="verses" book="vv">
_Ghāyase taṃ sucigandhaṃ,_  
_rūpaṃ passasi amānusaṃ;_  
_Devate pucchitācikkha,_  
_kissa kammassidaṃ phalan”ti._  

</Paragraph>

</Division>
<Division number="694" book="vv" bookNo="19" page="56">

<Paragraph number="540" type="verses" book="vv">
_“Ahaṃ manussesu manussabhūtā,_  
_Dāsī ayirakule ahuṃ;_  
_Buddhaṃ nisinnaṃ disvāna,_  
_Sālapupphehi okiriṃ._  

</Paragraph>

</Division>
<Division number="695" book="vv" bookNo="19" page="56">

<Paragraph number="541" type="verses" book="vv">
_Vaṭaṃsakañca sukataṃ,_  
_sālapupphamayaṃ ahaṃ;_  
_Buddhassa upanāmesiṃ,_  
_pasannā sehi pāṇibhi._  

</Paragraph>

</Division>
<Division number="696" book="vv" bookNo="19" page="57">

<Paragraph number="542" type="verses" book="vv">
_Tāhaṃ kammaṃ karitvāna,_  
_kusalaṃ buddhavaṇṇitaṃ;_  
_Apetasokā sukhitā,_  
_sampamodāmanāmayā”ti._  


</Paragraph>
</Division>

<Paragraph number="543" type="center" book="vv">
Mañjiṭṭhakavimānaṃ paṭhamaṃ.
</Paragraph>
