<Division number="195" book="vv">

<Paragraph number="219" type="verses" book="vv">
_“Caṇḍāli vanda pādāni,_  
_gotamassa yasassino;_  
_Tameva anukampāya,_  
_aṭṭhāsi isisattamo._  

</Paragraph>

</Division>
<Division number="196" book="vv">

<Paragraph number="220" type="verses" book="vv">
_Abhippasādehi manaṃ,_  
_arahantamhi tādini;_  
_Khippaṃ pañjalikā vanda,_  
_parittaṃ tava jīvitan”ti._  

</Paragraph>

</Division>
<Division number="197" book="vv">

<Paragraph number="221" type="verses" book="vv">
_Coditā bhāvitattena,_  
_sarīrantimadhārinā;_  
_Caṇḍālī vandi pādāni,_  
_gotamassa yasassino._  

</Paragraph>

</Division>
<Division number="198" book="vv">

<Paragraph number="222" type="verses" book="vv">
_Tamenaṃ avadhī gāvī,_  
_caṇḍāliṃ pañjaliṃ ṭhitaṃ;_  
_Namassamānaṃ sambuddhaṃ,_  
_andhakāre pabhaṅkaranti._  

</Paragraph>

</Division>
<Division number="199" book="vv">

<Paragraph number="223" type="verses" book="vv">
_“Khīṇāsavaṃ vigatarajaṃ anejaṃ,_  
_Ekaṃ araññamhi raho nisinnaṃ;_  
_Deviddhipattā upasaṅkamitvā,_  
_Vandāmi taṃ vīra mahānubhāvan”ti._  

</Paragraph>

</Division>
<Division number="200" book="vv">

<Paragraph number="224" type="verses" book="vv">
_“Suvaṇṇavaṇṇā jalitā mahāyasā,_  
_Vimānamoruyha anekacittā;_  
_Parivāritā accharāsaṅgaṇena,_  
_Kā tvaṃ subhe devate vandase maman”ti._  

</Paragraph>

</Division>
<Division number="201" book="vv">

<Paragraph number="225" type="verses" book="vv">
_“Ahaṃ bhaddante caṇḍālī,_  
_tayā vīrena pesitā;_  
_Vandiṃ arahato pāde,_  
_gotamassa yasassino._  

</Paragraph>

</Division>
<Division number="202" book="vv">

<Paragraph number="226" type="verses" book="vv">
_Sāhaṃ vanditvā pādāni,_  
_cutā caṇḍālayoniyā;_  
_Vimānaṃ sabbato bhaddaṃ,_  
_upapannamhi nandane._  

</Paragraph>

</Division>
<Division number="203" book="vv">

<Paragraph number="227" type="verses" book="vv">
_Accharānaṃ satasahassaṃ,_  
_purakkhatvāna tiṭṭhati;_  
_Tāsāhaṃ pavarā seṭṭhā,_  
_vaṇṇena yasasāyunā._  

</Paragraph>

</Division>
<Division number="204" book="vv">

<Paragraph number="228" type="verses" book="vv">
_Pahūtakatakalyāṇā,_  
_Sampajānā paṭissatā;_  
_Muniṃ kāruṇikaṃ loke,_  
_Taṃ bhante vanditumāgatā”ti._  

</Paragraph>

</Division>
<Division number="205" book="vv">

<Paragraph number="229" type="verses" book="vv">
_Idaṃ vatvāna caṇḍālī,_  
_kataññū katavedinī;_  
_Vanditvā arahato pāde,_  
_tatthevantaradhāyathāti._  


</Paragraph>
</Division>

<Paragraph number="230" type="center" book="vv">
Caṇḍālivimānaṃ catutthaṃ.
</Paragraph>
