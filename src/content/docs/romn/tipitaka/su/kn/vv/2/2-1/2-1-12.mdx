<Paragraph number="818" type="verses" book="vv">
_“Ko nu dibbena yānena,_  
_sabbasetena hatthinā;_  
_Tūriyatāḷitanigghoso,_  
_antalikkhe mahīyati._  

</Paragraph>

</Division>
<Division number="977" book="vv" bookNo="19" page="86">

<Paragraph number="819" type="verses" book="vv">
_Devatā nusi gandhabbo,_  
_adu sakko purindado;_  
_Ajānantā taṃ pucchāma,_  
_kathaṃ jānemu taṃ mayan”ti._  

</Paragraph>

</Division>
<Division number="978" book="vv" bookNo="19" page="87">

<Paragraph number="820" type="verses" book="vv">
_“Namhi devo na gandhabbo,_  
_nāpi sakko purindado;_  
_Sudhammā nāma ye devā,_  
_tesaṃ aññataro ahan”ti._  

</Paragraph>

</Division>
<Division number="979" book="vv" bookNo="19" page="87">

<Paragraph number="821" type="verses" book="vv">
_“Pucchāma devaṃ sudhammaṃ,_  
_Puthuṃ katvāna añjaliṃ;_  
_Kiṃ katvā mānuse kammaṃ,_  
_Sudhammaṃ upapajjatī”ti._  

</Paragraph>

</Division>
<Division number="980" book="vv" bookNo="19" page="87">

<Paragraph number="822" type="verses" book="vv">
_“Ucchāgāraṃ tiṇāgāraṃ,_  
_vatthāgārañca yo dade;_  
_Tiṇṇaṃ aññataraṃ datvā,_  
_sudhammaṃ upapajjatī”ti._  


</Paragraph>
</Division>

<Paragraph number="823" type="center" book="vv">
Tatiyanāgavimānaṃ dvādasamaṃ.
</Paragraph>
