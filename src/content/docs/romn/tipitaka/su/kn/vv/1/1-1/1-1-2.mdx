<Paragraph number="10" type="verses" book="vv">
_“Pīṭhaṃ te veḷuriyamayaṃ uḷāraṃ,_  
_Manojavaṃ gacchati yenakāmaṃ;_  
_Alaṅkate malyadhare suvatthe,_  
_Obhāsasi vijjurivabbhakūṭaṃ._  

</Paragraph>

</Division>
<Division number="9" book="vv">

<Paragraph number="11" type="verses" book="vv">
_Kena tetādiso vaṇṇo,_  
_kena te idha mijjhati;_  
_Uppajjanti ca te bhogā,_  
_ye keci manaso piyā._  

</Paragraph>

</Division>
<Division number="10" book="vv">

<Paragraph number="12" type="verses" book="vv">
_Pucchāmi taṃ devi mahānubhāve,_  
_Manussabhūtā kimakāsi puññaṃ;_  
_Kenāsi evaṃ jalitānubhāvā,_  
_Vaṇṇo ca te sabbadisā pabhāsatī”ti._  

</Paragraph>

</Division>
<Division number="11" book="vv">

<Paragraph number="13" type="verses" book="vv">
_Sā devatā attamanā,_  
_moggallānena pucchitā;_  
_Pañhaṃ puṭṭhā viyākāsi,_  
_yassa kammassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="12" book="vv">

<Paragraph number="14" type="verses" book="vv">
_“Ahaṃ manussesu manussabhūtā,_  
_Abbhāgatānāsanakaṃ adāsiṃ;_  
_Abhivādayiṃ añjalikaṃ akāsiṃ,_  
_Yathānubhāvañca adāsi dānaṃ._  

</Paragraph>

</Division>
<Division number="13" book="vv">

<Paragraph number="15" type="verses" book="vv">
_Tena metādiso vaṇṇo,_  
_tena me idha mijjhati;_  
_Uppajjanti ca me bhogā,_  
_ye keci manaso piyā._  

</Paragraph>

</Division>
<Division number="14" book="vv">

<Paragraph number="16" type="verses" book="vv">
_Akkhāmi te bhikkhu mahānubhāva,_  
_Manussabhūtā yamakāsi puññaṃ;_  
_Tenamhi evaṃ jalitānubhāvā,_  
_Vaṇṇo ca me sabbadisā pabhāsatī”ti._  


</Paragraph>
</Division>

<Paragraph number="17" type="center" book="vv">
Dutiyapīṭhavimānaṃ dutiyaṃ.
</Paragraph>
