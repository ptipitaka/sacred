<Paragraph number="36" type="verses" book="vv">
_“Kuñjaro te varāroho,_  
_Nānāratanakappano;_  
_Ruciro thāmavā javasampanno,_  
_Ākāsamhi samīhati._  

</Paragraph>

</Division>
<Division number="32" book="vv" bookNo="19" page="5">

<Paragraph number="37" type="verses" book="vv">
_Padumi padmapattakkhi,_  
_padmuppalajutindharo;_  
_Padmacuṇṇābhikiṇṇaṅgo,_  
_soṇṇapokkharamāladhā._  

</Paragraph>

</Division>
<Division number="33" book="vv" bookNo="19" page="5">

<Paragraph number="38" type="verses" book="vv">
_Padumānusaṭaṃ maggaṃ,_  
_padmapattavibhūsitaṃ;_  
_Ṭhitaṃ vaggu manugghātī,_  
_mitaṃ gacchati vāraṇo._  

</Paragraph>

</Division>
<Division number="34" book="vv" bookNo="19" page="5">

<Paragraph number="39" type="verses" book="vv">
_Tassa pakkamamānassa,_  
_soṇṇakaṃsā ratissarā;_  
_Tesaṃ suyyati nigghoso,_  
_tūriye pañcaṅgike yathā._  

</Paragraph>

</Division>
<Division number="35" book="vv" bookNo="19" page="5">

<Paragraph number="40" type="verses" book="vv">
_Tassa nāgassa khandhamhi,_  
_sucivatthā alaṅkatā;_  
_Mahantaṃ accharāsaṅghaṃ,_  
_vaṇṇena atirocasi._  

</Paragraph>

</Division>
<Division number="36" book="vv" bookNo="19" page="5">

<Paragraph number="41" type="verses" book="vv">
_Dānassa te idaṃ phalaṃ,_  
_atho sīlassa vā pana;_  
_Atho añjalikammassa,_  
_taṃ me akkhāhi pucchitā”ti._  

</Paragraph>

</Division>
<Division number="37" book="vv" bookNo="19" page="5">

<Paragraph number="42" type="verses" book="vv">
_Sā devatā attamanā,_  
_moggallānena pucchitā;_  
_Pañhaṃ puṭṭhā viyākāsi,_  
_yassa kammassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="38" book="vv" bookNo="19" page="6">

<Paragraph number="43" type="verses" book="vv">
_“Disvāna guṇasampannaṃ,_  
_jhāyiṃ jhānarataṃ sataṃ;_  
_Adāsiṃ pupphābhikiṇṇaṃ,_  
_āsanaṃ dussasanthataṃ._  

</Paragraph>

</Division>
<Division number="39" book="vv" bookNo="19" page="6">

<Paragraph number="44" type="verses" book="vv">
_Upaḍḍhaṃ padmamālāhaṃ,_  
_āsanassa samantato;_  
_Abbhokirissaṃ pattehi,_  
_pasannā sehi pāṇibhi._  

</Paragraph>

</Division>
<Division number="40" book="vv" bookNo="19" page="6">

<Paragraph number="45" type="verses" book="vv">
_Tassa kammakusalassa,_  
_idaṃ me īdisaṃ phalaṃ;_  
_Sakkāro garukāro ca,_  
_devānaṃ apacitā ahaṃ._  

</Paragraph>

</Division>
<Division number="41" book="vv" bookNo="19" page="6">

<Paragraph number="46" type="verses" book="vv">
_Yo ve sammāvimuttānaṃ,_  
_santānaṃ brahmacārinaṃ;_  
_Pasanno āsanaṃ dajjā,_  
_evaṃ nande yathā ahaṃ._  

</Paragraph>

</Division>
<Division number="42" book="vv" bookNo="19" page="6">

<Paragraph number="47" type="verses" book="vv">
_Tasmā hi attakāmena,_  
_mahattamabhikaṅkhatā;_  
_Āsanaṃ dātabbaṃ hoti,_  
_sarīrantimadhārinan”ti._  


</Paragraph>
</Division>

<Paragraph number="48" type="center" book="vv">
Kuñjaravimānaṃ pañcamaṃ.
</Paragraph>
