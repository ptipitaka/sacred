<Division number="1207" book="vv">

<Paragraph number="1059" type="verses" book="vv">
_“Alaṅkato maṭṭhakuṇḍalī,_  
_Māladhārī haricandanussado;_  
_Bāhā paggayha kandasi,_  
_Vanamajjhe kiṃ dukkhito tuvan”ti._  

</Paragraph>

</Division>
<Division number="1208" book="vv">

<Paragraph number="1060" type="verses" book="vv">
_“Sovaṇṇamayo pabhassaro,_  
_Uppanno rathapañjaro mama;_  
_Tassa cakkayugaṃ na vindāmi,_  
_Tena dukkhena jahāmi jīvitan”ti._  

</Paragraph>

</Division>
<Division number="1209" book="vv">

<Paragraph number="1061" type="verses" book="vv">
_“Sovaṇṇamayaṃ maṇimayaṃ,_  
_Lohitakamayaṃ atha rūpiyamayaṃ;_  
_Ācikkha me bhaddamāṇava,_  
_Cakkayugaṃ paṭipādayāmi te”ti._  

</Paragraph>

</Division>
<Division number="1210" book="vv">

<Paragraph number="1062" type="verses" book="vv">
_So māṇavo tassa pāvadi,_  
_“Candasūriyā ubhayettha dissare;_  
_Sovaṇṇamayo ratho mama,_  
_Tena cakkayugena sobhatī”ti._  

</Paragraph>

</Division>
<Division number="1211" book="vv">

<Paragraph number="1063" type="verses" book="vv">
_“Bālo kho tvaṃ asi māṇava,_  
_Yo tvaṃ patthayase apatthiyaṃ;_  
_Maññāmi tuvaṃ marissasi,_  
_Na hi tvaṃ lacchasi candasūriye”ti._  

</Paragraph>

</Division>
<Division number="1212" book="vv">

<Paragraph number="1064" type="verses" book="vv">
_“Gamanāgamanampi dissati,_  
_Vaṇṇadhātu ubhayattha vīthiyā;_  
_Peto kālakato na dissati,_  
_Ko nidha kandataṃ bālyataro”ti._  

</Paragraph>

</Division>
<Division number="1213" book="vv">

<Paragraph number="1065" type="verses" book="vv">
_“Saccaṃ kho vadesi māṇava,_  
_Ahameva kandataṃ bālyataro;_  
_Candaṃ viya dārako rudaṃ,_  
_Petaṃ kālakatābhipatthayinti._  

</Paragraph>

</Division>
<Division number="1214" book="vv">

<Paragraph number="1066" type="verses" book="vv">
_Ādittaṃ vata maṃ santaṃ,_  
_ghatasittaṃva pāvakaṃ;_  
_Vārinā viya osiñcaṃ,_  
_sabbaṃ nibbāpaye daraṃ._  

</Paragraph>

</Division>
<Division number="1215" book="vv">

<Paragraph number="1067" type="verses" book="vv">
_Abbahī vata me sallaṃ,_  
_sokaṃ hadayanissitaṃ;_  
_Yo me sokaparetassa,_  
_puttasokaṃ apānudi._  

</Paragraph>

</Division>
<Division number="1216" book="vv">

<Paragraph number="1068" type="verses" book="vv">
_Svāhaṃ abbūḷhasallosmi,_  
_sītibhūtosmi nibbuto;_  
_Na socāmi na rodāmi,_  
_tava sutvāna māṇavā”ti._  

</Paragraph>

</Division>
<Division number="1217" book="vv">

<Paragraph number="1069" type="verses" book="vv">
_“Devatā nusi gandhabbo,_  
_adu sakko purindado;_  
_Ko vā tvaṃ kassa vā putto,_  
_kathaṃ jānemu taṃ mayan”ti._  

</Paragraph>

</Division>
<Division number="1218" book="vv">

<Paragraph number="1070" type="verses" book="vv">
_“Yañca kandasi yañca rodasi,_  
_Puttaṃ āḷāhane sayaṃ dahitvā;_  
_Svāhaṃ kusalaṃ karitvā kammaṃ,_  
_Tidasānaṃ sahabyataṃ gato”ti._  

</Paragraph>

</Division>
<Division number="1219" book="vv">

<Paragraph number="1071" type="verses" book="vv">
_“Appaṃ vā bahuṃ vā nāddasāma,_  
_Dānaṃ dadantassa sake agāre;_  
_Uposathakammaṃ vā tādisaṃ,_  
_Kena kammena gatosi devalokan”ti._  

</Paragraph>

</Division>
<Division number="1220" book="vv">

<Paragraph number="1072" type="verses" book="vv">
_“Ābādhikohaṃ dukkhito gilāno,_  
_Āturarūpomhi sake nivesane;_  
_Buddhaṃ vigatarajaṃ vitiṇṇakaṅkhaṃ,_  
_Addakkhiṃ sugataṃ anomapaññaṃ._  

</Paragraph>

</Division>
<Division number="1221" book="vv">

<Paragraph number="1073" type="verses" book="vv">
_Svāhaṃ muditamano pasannacitto,_  
_Añjaliṃ akariṃ tathāgatassa;_  
_Tāhaṃ kusalaṃ karitvāna kammaṃ,_  
_Tidasānaṃ sahabyataṃ gato”ti._  

</Paragraph>

</Division>
<Division number="1222" book="vv">

<Paragraph number="1074" type="verses" book="vv">
_“Acchariyaṃ vata abbhutaṃ vata,_  
_Añjalikammassa ayamīdiso vipāko;_  
_Ahampi muditamano pasannacitto,_  
_Ajjeva buddhaṃ saraṇaṃ vajāmī”ti._  

</Paragraph>

</Division>
<Division number="1223" book="vv">

<Paragraph number="1075" type="verses" book="vv">
_“Ajjeva buddhaṃ saraṇaṃ vajāhi,_  
_Dhammañca saṃghañca pasannacitto;_  
_Tatheva sikkhāya padāni pañca,_  
_Akhaṇḍaphullāni samādiyassu._  

</Paragraph>

</Division>
<Division number="1224" book="vv">

<Paragraph number="1076" type="verses" book="vv">
_Pāṇātipātā viramassu khippaṃ,_  
_Loke adinnaṃ parivajjayassu;_  
_Amajjapo mā ca musā bhaṇāhi,_  
_Sakena dārena ca hohi tuṭṭho”ti._  

</Paragraph>

</Division>
<Division number="1225" book="vv">

<Paragraph number="1077" type="verses" book="vv">
_“Atthakāmosi me yakkha,_  
_hitakāmosi devate;_  
_Karomi tuyhaṃ vacanaṃ,_  
_tvaṃsi ācariyo mamāti._  

</Paragraph>

</Division>
<Division number="1226" book="vv">

<Paragraph number="1078" type="verses" book="vv">
_Upemi saraṇaṃ buddhaṃ,_  
_dhammañcāpi anuttaraṃ;_  
_Saṃghañca naradevassa,_  
_gacchāmi saraṇaṃ ahaṃ._  

</Paragraph>

</Division>
<Division number="1227" book="vv">

<Paragraph number="1079" type="verses" book="vv">
_Pāṇātipātā viramāmi khippaṃ,_  
_Loke adinnaṃ parivajjayāmi;_  
_Amajjapo no ca musā bhaṇāmi,_  
_Sakena dārena ca homi tuṭṭho”ti._  


</Paragraph>
</Division>

<Paragraph number="1080" type="center" book="vv">
Maṭṭhakuṇḍalīvimānaṃ navamaṃ.
</Paragraph>
