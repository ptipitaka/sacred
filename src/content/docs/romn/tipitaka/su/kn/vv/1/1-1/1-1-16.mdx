<Paragraph number="153" type="verses" book="vv">
_“Yuttā ca te paramaalaṅkatā hayā,_  
_Adhomukhā aghasigamā balī javā;_  
_Abhinimmitā pañcarathāsatā ca te,_  
_Anventi taṃ sārathicoditā hayā._  

</Paragraph>

</Division>
<Division number="138" book="vv" bookNo="19" page="18">

<Paragraph number="154" type="verses" book="vv">
_Sā tiṭṭhasi rathavare alaṅkatā,_  
_Obhāsayaṃ jalamiva joti pāvako;_  
_Pucchāmi taṃ varatanu anomadassane,_  
_Kasmā nu kāyā anadhivaraṃ upāgami”._  

</Paragraph>

</Division>
<Division number="139" book="vv" bookNo="19" page="18">

<Paragraph number="155" type="verses" book="vv">
_“Kāmaggapattānaṃ yamāhunuttaraṃ,_  
_Nimmāya nimmāya ramanti devatā;_  
_Tasmā kāyā accharā kāmavaṇṇinī,_  
_Idhāgatā anadhivaraṃ namassituṃ”._  

</Paragraph>

</Division>
<Division number="140" book="vv" bookNo="19" page="18">

<Paragraph number="156" type="verses" book="vv">
_“Kiṃ tvaṃ pure sucaritamācarīdha,_  
_Kenacchasi tvaṃ amitayasā sukhedhitā;_  
_Iddhī ca te anadhivarā vihaṅgamā,_  
_Vaṇṇo ca te dasa disā virocati._  

</Paragraph>

</Division>
<Division number="141" book="vv" bookNo="19" page="18">

<Paragraph number="157" type="verses" book="vv">
_Devehi tvaṃ parivutā sakkatā casi,_  
_Kuto cutā sugatigatāsi devate;_  
_Kassa vā tvaṃ vacanakarānusāsaniṃ,_  
_Ācikkha me tvaṃ yadi buddhasāvikā”ti._  

</Paragraph>

</Division>
<Division number="142" book="vv" bookNo="19" page="19">

<Paragraph number="158" type="verses" book="vv">
_“Nagantare nagaravare sumāpite,_  
_Paricārikā rājavarassa sirimato;_  
_Nacce gīte paramasusikkhitā ahuṃ,_  
_Sirimāti maṃ rājagahe avediṃsu._  

</Paragraph>

</Division>
<Division number="143" book="vv" bookNo="19" page="19">

<Paragraph number="159" type="verses" book="vv">
_Buddho ca me isinisabho vināyako,_  
_Adesayī samudayadukkhaniccataṃ;_  
_Asaṅkhataṃ dukkhanirodhasassataṃ,_  
_Maggañcimaṃ akuṭilamañjasaṃ sivaṃ._  

</Paragraph>

</Division>
<Division number="144" book="vv" bookNo="19" page="19">

<Paragraph number="160" type="verses" book="vv">
_Sutvānahaṃ amatapadaṃ asaṅkhataṃ,_  
_Tathāgatassanadhivarassa sāsanaṃ;_  
_Sīlesvahaṃ paramasusaṃvutā ahuṃ,_  
_Dhamme ṭhitā naravarabuddhadesite._  

</Paragraph>

</Division>
<Division number="145" book="vv" bookNo="19" page="19">

<Paragraph number="161" type="verses" book="vv">
_Ñatvānahaṃ virajapadaṃ asaṅkhataṃ,_  
_Tathāgatenanadhivarena desitaṃ;_  
_Tatthevahaṃ samathasamādhimāphusiṃ,_  
_Sāyeva me paramaniyāmatā ahu._  

</Paragraph>

</Division>
<Division number="146" book="vv" bookNo="19" page="19">

<Paragraph number="162" type="verses" book="vv">
_Laddhānahaṃ amatavaraṃ visesanaṃ,_  
_Ekaṃsikā abhisamaye visesiya;_  
_Asaṃsayā bahujanapūjitā ahaṃ,_  
_Khiḍḍāratiṃ paccanubhomanappakaṃ._  

</Paragraph>

</Division>
<Division number="147" book="vv" bookNo="19" page="19">

<Paragraph number="163" type="verses" book="vv">
_Evaṃ ahaṃ amatadasamhi devatā,_  
_Tathāgatassanadhivarassa sāvikā;_  
_Dhammaddasā paṭhamaphale patiṭṭhitā,_  
_Sotāpannā na ca pana matthi duggati._  

</Paragraph>

</Division>
<Division number="148" book="vv" bookNo="19" page="19">

<Paragraph number="164" type="verses" book="vv">
_Sā vandituṃ anadhivaraṃ upāgamiṃ,_  
_Pāsādike kusalarate ca bhikkhavo;_  
_Namassituṃ samaṇasamāgamaṃ sivaṃ,_  
_Sagāravā sirimato dhammarājino._  

</Paragraph>

</Division>
<Division number="149" book="vv" bookNo="19" page="20">

<Paragraph number="165" type="verses" book="vv">
_Disvā muniṃ muditamanamhi pīṇitā,_  
_Tathāgataṃ naravaradammasārathiṃ;_  
_Taṇhacchidaṃ kusalarataṃ vināyakaṃ,_  
_Vandāmahaṃ paramahitānukampakan”ti._  


</Paragraph>
</Division>

<Paragraph number="166" type="center" book="vv">
Sirimāvimānaṃ soḷasamaṃ.
</Paragraph>
