_yā tvaṃ tiṭṭhasi devate;_  
_Hatthapāde ca viggayha,_  
_naccasi suppavādite._  

</Paragraph>

<Division number="827" book="vv" bookNo="19" page="69">

<Paragraph number="665" type="verses" book="vv">
_Tassā te naccamānāya,_  
_aṅgamaṅgehi sabbaso;_  
_Dibbā saddā niccharanti,_  
_savanīyā manoramā._  

</Paragraph>

</Division>
<Division number="828" book="vv" bookNo="19" page="69">

<Paragraph number="666" type="verses" book="vv">
_Tassā te naccamānāya,_  
_aṅgamaṅgehi sabbaso;_  
_Dibbā gandhā pavāyanti,_  
_sucigandhā manoramā._  

</Paragraph>

</Division>
<Division number="829" book="vv" bookNo="19" page="69">

<Paragraph number="667" type="verses" book="vv">
_Vivattamānā kāyena,_  
_yā veṇīsu piḷandhanā;_  
_Tesaṃ suyyati nigghoso,_  
_tūriye pañcaṅgike yathā._  

</Paragraph>

</Division>
<Division number="830" book="vv" bookNo="19" page="69">

<Paragraph number="668" type="verses" book="vv">
_Vaṭaṃsakā vātadhutā,_  
_vātena sampakampitā;_  
_Tesaṃ suyyati nigghoso,_  
_tūriye pañcaṅgike yathā._  

</Paragraph>

</Division>
<Division number="831" book="vv" bookNo="19" page="69">

<Paragraph number="669" type="verses" book="vv">
_Yāpi te sirasmiṃ mālā,_  
_sucigandhā manoramā;_  
_Vāti gandho disā sabbā,_  
_rukkho mañjūsako yathā._  

</Paragraph>

</Division>
<Division number="832" book="vv" bookNo="19" page="69">

<Paragraph number="670" type="verses" book="vv">
_Ghāyase taṃ sucigandhaṃ,_  
_rūpaṃ passasi amānusaṃ;_  
_Devate pucchitācikkha,_  
_kissa kammassidaṃ phalan”ti._  

</Paragraph>

</Division>
<Division number="833" book="vv" bookNo="19" page="69">

<Paragraph number="671" type="verses" book="vv">
_“Dāsī ahaṃ pure āsiṃ,_  
_gayāyaṃ brāhmaṇassahaṃ;_  
_Appapuññā alakkhikā,_  
_rajjumālāti maṃ viduṃ._  

</Paragraph>

</Division>
<Division number="834" book="vv" bookNo="19" page="69">

<Paragraph number="672" type="verses" book="vv">
_Akkosānaṃ vadhānañca,_  
_tajjanāya ca uggatā;_  
_Kuṭaṃ gahetvā nikkhamma,_  
_agañchiṃ udahāriyā._  

</Paragraph>

</Division>
<Division number="835" book="vv" bookNo="19" page="70">

<Paragraph number="673" type="verses" book="vv">
_Vipathe kuṭaṃ nikkhipitvā,_  
_vanasaṇḍaṃ upāgamiṃ;_  
_Idhevāhaṃ marissāmi,_  
_ko attho jīvitena me._  

</Paragraph>

</Division>
<Division number="836" book="vv" bookNo="19" page="70">

<Paragraph number="674" type="verses" book="vv">
_Daḷhaṃ pāsaṃ karitvāna,_  
_āsumbhitvāna pādape;_  
_Tato disā vilokesiṃ,_  
_‘ko nu kho vanamassito’._  

</Paragraph>

</Division>
<Division number="837" book="vv" bookNo="19" page="70">

<Paragraph number="675" type="verses" book="vv">
_Tatthaddasāsiṃ sambuddhaṃ,_  
_sabbalokahitaṃ muniṃ;_  
_Nisinnaṃ rukkhamūlasmiṃ,_  
_jhāyantaṃ akutobhayaṃ._  

</Paragraph>

</Division>
<Division number="838" book="vv" bookNo="19" page="70">

<Paragraph number="676" type="verses" book="vv">
_Tassā me ahu saṃvego,_  
_abbhuto lomahaṃsano;_  
_Ko nu kho vanamassito,_  
_manusso udāhu devatā._  

</Paragraph>

</Division>
<Division number="839" book="vv" bookNo="19" page="70">

<Paragraph number="677" type="verses" book="vv">
_Pāsādikaṃ pasādanīyaṃ,_  
_vanā nibbanamāgataṃ;_  
_Disvā mano me pasīdi,_  
_nāyaṃ yādisakīdiso._  

</Paragraph>

</Division>
<Division number="840" book="vv" bookNo="19" page="70">

<Paragraph number="678" type="verses" book="vv">
_Guttindriyo jhānarato,_  
_abahiggatamānaso;_  
_Hito sabbassa lokassa,_  
_buddho ayaṃ bhavissati._  

</Paragraph>

</Division>
<Division number="841" book="vv" bookNo="19" page="70">

<Paragraph number="679" type="verses" book="vv">
_Bhayabheravo durāsado,_  
_sīhova guhamassito;_  
_Dullabhāyaṃ dassanāya,_  
_pupphaṃ odumbaraṃ yathā._  

</Paragraph>

</Division>
<Division number="842" book="vv" bookNo="19" page="70">

<Paragraph number="680" type="verses" book="vv">
_So maṃ mudūhi vācāhi,_  
_ālapitvā tathāgato;_  
_Rajjumāleti maṃvoca,_  
_saraṇaṃ gaccha tathāgataṃ._  

</Paragraph>

</Division>
<Division number="843" book="vv" bookNo="19" page="70">

<Paragraph number="681" type="verses" book="vv">
_Tāhaṃ giraṃ suṇitvāna,_  
_nelaṃ atthavatiṃ suciṃ;_  
_Saṇhaṃ muduñca vagguñca,_  
_sabbasokāpanūdanaṃ._  

</Paragraph>

</Division>
<Division number="844" book="vv" bookNo="19" page="70">

<Paragraph number="682" type="verses" book="vv">
_Kallacittañca maṃ ñatvā,_  
_pasannaṃ suddhamānasaṃ;_  
_Hito sabbassa lokassa,_  
_anusāsi tathāgato._  

</Paragraph>

</Division>
<Division number="845" book="vv" bookNo="19" page="70">

<Paragraph number="683" type="verses" book="vv">
_Idaṃ dukkhanti maṃvoca,_  
_Ayaṃ dukkhassa sambhavo;_  
_Dukkha nirodho maggo ca,_  
_Añjaso amatogadho._  

</Paragraph>

</Division>
<Division number="846" book="vv" bookNo="19" page="71">

<Paragraph number="684" type="verses" book="vv">
_Anukampakassa kusalassa,_  
_Ovādamhi ahaṃ ṭhitā;_  
_Ajjhagā amataṃ santiṃ,_  
_Nibbānaṃ padamaccutaṃ._  

</Paragraph>

</Division>
<Division number="847" book="vv" bookNo="19" page="71">

<Paragraph number="685" type="verses" book="vv">
_Sāhaṃ avaṭṭhitāpemā,_  
_dassane avikampinī;_  
_Mūlajātāya saddhāya,_  
_dhītā buddhassa orasā._  

</Paragraph>

</Division>
<Division number="848" book="vv" bookNo="19" page="71">

<Paragraph number="686" type="verses" book="vv">
_Sāhaṃ ramāmi kīḷāmi,_  
_modāmi akutobhayā;_  
_Dibbamālaṃ dhārayāmi,_  
_pivāmi madhumaddavaṃ._  

</Paragraph>

</Division>
<Division number="849" book="vv" bookNo="19" page="71">

<Paragraph number="687" type="verses" book="vv">
_Saṭṭhitūriyasahassāni,_  
_paṭibodhaṃ karonti me;_  
_Āḷambo gaggaro bhīmo,_  
_sādhuvādī ca saṃsayo._  

</Paragraph>

</Division>
<Division number="850" book="vv" bookNo="19" page="71">

<Paragraph number="688" type="verses" book="vv">
_Pokkharo ca suphasso ca,_  
_vīṇāmokkhā ca nāriyo;_  
_Nandā ceva sunandā ca,_  
_soṇadinnā sucimhitā._  

</Paragraph>

</Division>
<Division number="851" book="vv" bookNo="19" page="71">

<Paragraph number="689" type="verses" book="vv">
_Alambusā missakesī ca,_  
_puṇḍarīkātidāruṇī;_  
_Eṇīphassā suphassā ca,_  
_subhaddā muduvādinī._  

</Paragraph>

</Division>
<Division number="852" book="vv" bookNo="19" page="71">

<Paragraph number="690" type="verses" book="vv">
_Etā caññā ca seyyāse,_  
_accharānaṃ pabodhikā;_  
_Tā maṃ kālenupāgantvā,_  
_abhibhāsanti devatā._  

</Paragraph>

</Division>
<Division number="853" book="vv" bookNo="19" page="71">

<Paragraph number="691" type="verses" book="vv">
_Handa naccāma gāyāma,_  
_handa taṃ ramayāmase;_  
_Nayidaṃ akatapuññānaṃ,_  
_katapuññānamevidaṃ._  

</Paragraph>

</Division>
<Division number="854" book="vv" bookNo="19" page="71">

<Paragraph number="692" type="verses" book="vv">
_Asokaṃ nandanaṃ rammaṃ,_  
_tidasānaṃ mahāvanaṃ;_  
_Sukhaṃ akatapuññānaṃ,_  
_idha natthi parattha ca._  

</Paragraph>

</Division>
<Division number="855" book="vv" bookNo="19" page="71">

<Paragraph number="693" type="verses" book="vv">
_Sukhañca katapuññānaṃ,_  
_idha ceva parattha ca;_  
_Tesaṃ sahabyakāmānaṃ,_  
_kattabbaṃ kusalaṃ bahuṃ;_  
_Katapuññā hi modanti,_  
_sagge bhogasamaṅgino._  

</Paragraph>

</Division>
<Division number="856" book="vv" bookNo="19" page="71">

<Paragraph number="694" type="verses" book="vv">
_Bahūnaṃ vata atthāya,_  
_uppajjanti tathāgatā;_  
_Dakkhiṇeyyā manussānaṃ,_  
_puññakhettānamākarā;_  
_Yattha kāraṃ karitvāna,_  
_sagge modanti dāyakā”ti._  


</Paragraph>
</Division>

_Vaggo tena pavuccatīti._  


</Paragraph>
