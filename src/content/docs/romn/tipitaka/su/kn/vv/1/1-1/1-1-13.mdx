<Paragraph number="121" type="verses" book="vv">
_“Abhikkantena vaṇṇena,_  
_yā tvaṃ tiṭṭhasi devate;_  
_Obhāsentī disā sabbā,_  
_osadhī viya tārakā._  

</Paragraph>

</Division>
<Division number="109" book="vv" bookNo="19" page="15">

<Paragraph number="122" type="verses" book="vv">
_Kena tetādiso vaṇṇo,_  
_kena te idha mijjhati;_  
_Uppajjanti ca te bhogā,_  
_ye keci manaso piyā._  

</Paragraph>

</Division>
<Division number="110" book="vv" bookNo="19" page="15">

<Paragraph number="123" type="verses" book="vv">
_Pucchāmi taṃ devi mahānubhāve,_  
_Manussabhūtā kimakāsi puññaṃ;_  
_Kenāsi evaṃ jalitānubhāvā,_  
_Vaṇṇo ca te sabbadisā pabhāsatī”ti._  

</Paragraph>

</Division>
<Division number="111" book="vv" bookNo="19" page="15">

<Paragraph number="124" type="verses" book="vv">
_Sā devatā attamanā,_  
_moggallānena pucchitā;_  
_Pañhaṃ puṭṭhā viyākāsi,_  
_yassa kammassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="112" book="vv" bookNo="19" page="15">

<Paragraph number="125" type="verses" book="vv">
_“Ahaṃ manussesu manussabhūtā,_  
_Suṇisā ahosiṃ sasurassa gehe._  

</Paragraph>

</Division>
<Division number="113" book="vv" bookNo="19" page="15">

<Paragraph number="126" type="verses" book="vv">
_Addasaṃ virajaṃ bhikkhuṃ,_  
_vippasannamanāvilaṃ;_  
_Tassa adāsahaṃ pūvaṃ,_  
_pasannā sehi pāṇibhi;_  
_Bhāgaḍḍhabhāgaṃ datvāna,_  
_modāmi nandane vane._  

</Paragraph>

</Division>
<Division number="114" book="vv" bookNo="19" page="15">

<Paragraph number="127" type="verses" book="vv">
_Tena metādiso vaṇṇo,_  
_tena me idha mijjhati;_  
_Uppajjanti ca me bhogā,_  
_ye keci manaso piyā._  

</Paragraph>

</Division>
<Division number="115" book="vv" bookNo="19" page="15">

<Paragraph number="128" type="verses" book="vv">
_Akkhāmi te bhikkhu mahānubhāva,_  
_Manussabhūtā yamakāsi puññaṃ;_  
_Tenamhi evaṃ jalitānubhāvā,_  
_Vaṇṇo ca me sabbadisā pabhāsatī”ti._  


</Paragraph>
</Division>

<Paragraph number="129" type="center" book="vv">
Paṭhamasuṇisāvimānaṃ terasamaṃ.
</Paragraph>
