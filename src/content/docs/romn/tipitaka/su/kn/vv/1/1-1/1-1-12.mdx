<Division number="101" book="vv">

<Paragraph number="113" type="verses" book="vv">
_“Veḷuriyathambhaṃ ruciraṃ pabhassaraṃ,_  
_Vimānamāruyha anekacittaṃ;_  
_Tatthacchasi devi mahānubhāve,_  
_Uccāvacā iddhi vikubbamānā;_  
_Imā ca te accharāyo samantato,_  
_Naccanti gāyanti pamodayanti ca._  

</Paragraph>

</Division>
<Division number="102" book="vv">

<Paragraph number="114" type="verses" book="vv">
_Deviddhipattāsi mahānubhāve,_  
_Manussabhūtā kimakāsi puññaṃ;_  
_Kenāsi evaṃ jalitānubhāvā,_  
_Vaṇṇo ca te sabbadisā pabhāsatī”ti._  

</Paragraph>

</Division>
<Division number="103" book="vv">

<Paragraph number="115" type="verses" book="vv">
_Sā devatā attamanā,_  
_moggallānena pucchitā;_  
_Pañhaṃ puṭṭhā viyākāsi,_  
_yassa kammassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="104" book="vv">

<Paragraph number="116" type="verses" book="vv">
_“Ahaṃ manussesu manussabhūtā,_  
_Upāsikā cakkhumato ahosiṃ;_  
_Pāṇātipātā viratā ahosiṃ,_  
_Loke adinnaṃ parivajjayissaṃ._  

</Paragraph>

</Division>
<Division number="105" book="vv">

<Paragraph number="117" type="verses" book="vv">
_Amajjapā no ca musā abhāṇiṃ,_  
_Sakena sāminā ahosiṃ tuṭṭhā;_  
_Annañca pānañca pasannacittā,_  
_Sakkacca dānaṃ vipulaṃ adāsiṃ._  

</Paragraph>

</Division>
<Division number="106" book="vv">

<Paragraph number="118" type="verses" book="vv">
_Tena metādiso vaṇṇo,_  
_tena me idha mijjhati;_  
_Uppajjanti ca me bhogā,_  
_ye keci manaso piyā._  

</Paragraph>

</Division>
<Division number="107" book="vv">

<Paragraph number="119" type="verses" book="vv">
_Akkhāmi te bhikkhu mahānubhāva,_  
_Manussabhūtā yamakāsi puññaṃ;_  
_Tenamhi evaṃ jalitānubhāvā,_  
_Vaṇṇo ca me sabbadisā pabhāsatī”ti._  


</Paragraph>
</Division>

<Paragraph number="120" type="center" book="vv">
Dutiyapatibbatāvimānaṃ dvādasamaṃ.
</Paragraph>
