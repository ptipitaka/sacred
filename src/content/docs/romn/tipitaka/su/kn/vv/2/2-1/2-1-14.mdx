<Paragraph number="859" type="verses" book="vv">
_“Sahassayuttaṃ hayavāhanaṃ subhaṃ,_  
_Āruyhimaṃ sandanaṃ nekacittaṃ;_  
_Uyyānabhūmiṃ abhito anukkamaṃ,_  
_Purindado bhūtapatīva vāsavo._  

</Paragraph>

</Division>
<Division number="1016" book="vv" bookNo="19" page="90">

<Paragraph number="860" type="verses" book="vv">
_Sovaṇṇamayā te rathakubbarā ubho,_  
_Phalehi aṃsehi atīva saṅgatā;_  
_Sujātagumbā naravīraniṭṭhitā,_  
_Virocatī pannaraseva cando._  

</Paragraph>

</Division>
<Division number="1017" book="vv" bookNo="19" page="91">

<Paragraph number="861" type="verses" book="vv">
_Suvaṇṇajālāvatato ratho ayaṃ,_  
_Bahūhi nānāratanehi cittito;_  
_Sunandighoso ca subhassaro ca,_  
_Virocatī cāmarahatthabāhubhi._  

</Paragraph>

</Division>
<Division number="1018" book="vv" bookNo="19" page="91">

<Paragraph number="862" type="verses" book="vv">
_Imā ca nābhyo manasābhinimmitā,_  
_Rathassa pādantaramajjhabhūsitā;_  
_Imā ca nābhyo satarājicittitā,_  
_Sateratā vijjurivappabhāsare._  

</Paragraph>

</Division>
<Division number="1019" book="vv" bookNo="19" page="91">

<Paragraph number="863" type="verses" book="vv">
_Anekacittāvatato ratho ayaṃ,_  
_Puthū ca nemī ca sahassaraṃsiko;_  
_Tesaṃ saro suyyati vaggurūpo,_  
_Pañcaṅgikaṃ tūriyamivappavāditaṃ._  

</Paragraph>

</Division>
<Division number="1020" book="vv" bookNo="19" page="91">

<Paragraph number="864" type="verses" book="vv">
_Sirasmiṃ cittaṃ maṇicandakappitaṃ,_  
_Sadā visuddhaṃ ruciraṃ pabhassaraṃ;_  
_Suvaṇṇarājīhi atīva saṅgataṃ,_  
_Veḷuriyarājīva atīva sobhati._  

</Paragraph>

</Division>
<Division number="1021" book="vv" bookNo="19" page="91">

<Paragraph number="865" type="verses" book="vv">
_Ime ca vāḷī maṇicandakappitā,_  
_Ārohakambū sujavā brahūpamā;_  
_Brahā mahantā balino mahājavā,_  
_Mano tavaññāya tatheva siṃsare._  

</Paragraph>

</Division>
<Division number="1022" book="vv" bookNo="19" page="91">

<Paragraph number="866" type="verses" book="vv">
_Ime ca sabbe sahitā catukkamā,_  
_Mano tavaññāya tatheva siṃsare;_  
_Samaṃ vahantī mudukā anuddhatā,_  
_Āmodamānā turagānamuttama._  

</Paragraph>

</Division>
<Division number="1023" book="vv" bookNo="19" page="91">

<Paragraph number="867" type="verses" book="vv">
_Dhunanti vagganti patanti cambare,_  
_Abbhuddhunantā sukate piḷandhane;_  
_Tesaṃ saro suyyati vaggurūpo,_  
_Pañcaṅgikaṃ tūriyamivappavāditaṃ._  

</Paragraph>

</Division>
<Division number="1024" book="vv" bookNo="19" page="92">

<Paragraph number="868" type="verses" book="vv">
_Rathassa ghoso apiḷandhanāna ca,_  
_Khurassa nādo abhihiṃsanāya ca;_  
_Ghoso suvaggū samitassa suyyati,_  
_Gandhabbatūriyāni vicitrasaṃvane._  

</Paragraph>

</Division>
<Division number="1025" book="vv" bookNo="19" page="92">

<Paragraph number="869" type="verses" book="vv">
_Rathe ṭhitā tā migamandalocanā,_  
_Āḷārapamhā hasitā piyaṃvadā;_  
_Veḷuriyajālāvatatā tanucchavā,_  
_Sadeva gandhabbasūraggapūjitā._  

</Paragraph>

</Division>
<Division number="1026" book="vv" bookNo="19" page="92">

<Paragraph number="870" type="verses" book="vv">
_Tā rattarattambarapītavāsasā,_  
_Visālanettā abhirattalocanā;_  
_Kule sujātā sutanū sucimhitā,_  
_Rathe ṭhitā pañjalikā upaṭṭhitā._  

</Paragraph>

</Division>
<Division number="1027" book="vv" bookNo="19" page="92">

<Paragraph number="871" type="verses" book="vv">
_Tā kambukeyūradharā suvāsasā,_  
_Sumajjhimā ūruthanūpapannā;_  
_Vaṭṭaṅguliyo sumukhā sudassanā,_  
_Rathe ṭhitā pañjalikā upaṭṭhitā._  

</Paragraph>

</Division>
<Division number="1028" book="vv" bookNo="19" page="92">

<Paragraph number="872" type="verses" book="vv">
_Aññā suveṇī susu missakesiyo,_  
_Samaṃ vibhattāhi pabhassarāhi ca;_  
_Anubbatā tā tava mānase ratā,_  
_Rathe ṭhitā pañjalikā upaṭṭhitā._  

</Paragraph>

</Division>
<Division number="1029" book="vv" bookNo="19" page="92">

<Paragraph number="873" type="verses" book="vv">
_Āveḷiniyo padumuppalacchadā,_  
_Alaṅkatā candanasāravāsitā;_  
_Anubbatā tā tava mānase ratā,_  
_Rathe ṭhitā pañjalikā upaṭṭhitā._  

</Paragraph>

</Division>
<Division number="1030" book="vv" bookNo="19" page="92">

<Paragraph number="874" type="verses" book="vv">
_Tā māliniyo padumuppalacchadā,_  
_Alaṅkatā candanasāravāsitā;_  
_Anubbatā tā tava mānase ratā,_  
_Rathe ṭhitā pañjalikā upaṭṭhitā._  

</Paragraph>

</Division>
<Division number="1031" book="vv" bookNo="19" page="93">

<Paragraph number="875" type="verses" book="vv">
_Kaṇṭhesu te yāni piḷandhanāni,_  
_Hatthesu pādesu tatheva sīse;_  
_Obhāsayantī dasa sabbaso disā,_  
_Abbhuddayaṃ sāradikova bhāṇumā._  

</Paragraph>

</Division>
<Division number="1032" book="vv" bookNo="19" page="93">

<Paragraph number="876" type="verses" book="vv">
_Vātassa vegena ca sampakampitā,_  
_Bhujesu mālā apiḷandhanāni ca;_  
_Muñcanti ghosaṃ ruciraṃ suciṃ subhaṃ,_  
_Sabbehi viññūhi sutabbarūpaṃ._  

</Paragraph>

</Division>
<Division number="1033" book="vv" bookNo="19" page="93">

<Paragraph number="877" type="verses" book="vv">
_Uyyānabhūmyā ca duvaddhato ṭhitā,_  
_Rathā ca nāgā tūriyāni ca saro;_  
_Tameva devinda pamodayanti,_  
_Vīṇā yathā pokkharapattabāhubhi._  

</Paragraph>

</Division>
<Division number="1034" book="vv" bookNo="19" page="93">

<Paragraph number="878" type="verses" book="vv">
_Imāsu vīṇāsu bahūsu vaggūsu,_  
_Manuññarūpāsu hadayeritaṃ pītiṃ;_  
_Pavajjamānāsu atīva accharā,_  
_Bhamanti kaññā padumesu sikkhitā._  

</Paragraph>

</Division>
<Division number="1035" book="vv" bookNo="19" page="93">

<Paragraph number="879" type="verses" book="vv">
_Yadā ca gītāni ca vāditāni ca,_  
_Naccāni cimāni samenti ekato;_  
_Athettha naccanti athettha accharā,_  
_Obhāsayantī ubhato varitthiyo._  

</Paragraph>

</Division>
<Division number="1036" book="vv" bookNo="19" page="93">

<Paragraph number="880" type="verses" book="vv">
_So modasi tūriyagaṇappabodhano,_  
_Mahīyamāno vajirāvudhoriva;_  
_Imāsu vīṇāsu bahūsu vaggūsu,_  
_Manuññarūpāsu hadayeritaṃ pītiṃ._  

</Paragraph>

</Division>
<Division number="1037" book="vv" bookNo="19" page="93">

<Paragraph number="881" type="verses" book="vv">
_Kiṃ tvaṃ pure kammamakāsi attanā,_  
_Manussabhūto purimāya jātiyā;_  
_Uposathaṃ kaṃ vā tuvaṃ upāvasi,_  
_Kaṃ dhammacariyaṃ vatamābhirocayi._  

</Paragraph>

</Division>
<Division number="1038" book="vv" bookNo="19" page="94">

<Paragraph number="882" type="verses" book="vv">
_Nayīdamappassa katassa kammuno,_  
_Pubbe suciṇṇassa uposathassa vā;_  
_Iddhānubhāvo vipulo ayaṃ tava,_  
_Yaṃ devasaṅghaṃ abhirocase bhusaṃ._  

</Paragraph>

</Division>
<Division number="1039" book="vv" bookNo="19" page="94">

<Paragraph number="883" type="verses" book="vv">
_Dānassa te idaṃ phalaṃ,_  
_atho sīlassa vā pana;_  
_Atho añjalikammassa,_  
_taṃ me akkhāhi pucchito”ti._  

</Paragraph>

</Division>
<Division number="1040" book="vv" bookNo="19" page="94">

<Paragraph number="884" type="verses" book="vv">
_So devaputto attamano,_  
_moggallānena pucchito;_  
_Pañhaṃ puṭṭho viyākāsi,_  
_yassa kammassidaṃ phalanti._  

</Paragraph>

</Division>
<Division number="1041" book="vv" bookNo="19" page="94">

<Paragraph number="885" type="verses" book="vv">
_“Jitindriyaṃ buddhamanomanikkamaṃ,_  
_Naruttamaṃ kassapamaggapuggalaṃ;_  
_Avāpurantaṃ amatassa dvāraṃ,_  
_Devātidevaṃ satapuññalakkhaṇaṃ._  

</Paragraph>

</Division>
<Division number="1042" book="vv" bookNo="19" page="94">

<Paragraph number="886" type="verses" book="vv">
_Tamaddasaṃ kuñjaramoghatiṇṇaṃ,_  
_Suvaṇṇasiṅgīnadabimbasādisaṃ;_  
_Disvāna taṃ khippamahuṃ sucīmano,_  
_Tameva disvāna subhāsitaddhajaṃ._  

</Paragraph>

</Division>
<Division number="1043" book="vv" bookNo="19" page="94">

<Paragraph number="887" type="verses" book="vv">
_Tamannapānaṃ athavāpi cīvaraṃ,_  
_Suciṃ paṇītaṃ rasasā upetaṃ;_  
_Pupphābhikkiṇamhi sake nivesane,_  
_Patiṭṭhapesiṃ sa asaṅgamānaso._  

</Paragraph>

</Division>
<Division number="1044" book="vv" bookNo="19" page="94">

<Paragraph number="888" type="verses" book="vv">
_Tamannapānena ca cīvarena ca,_  
_Khajjena bhojjena ca sāyanena ca;_  
_Santappayitvā dvipadānamuttamaṃ,_  
_So saggaso devapure ramāmahaṃ._  

</Paragraph>

</Division>
<Division number="1045" book="vv" bookNo="19" page="94">

<Paragraph number="889" type="verses" book="vv">
_Etenupāyena imaṃ niraggaḷaṃ,_  
_Yaññaṃ yajitvā tividhaṃ visuddhaṃ;_  
_Pahāyahaṃ mānusakaṃ samussayaṃ,_  
_Indūpamo devapure ramāmahaṃ._  

</Paragraph>

</Division>
<Division number="1046" book="vv" bookNo="19" page="95">

<Paragraph number="890" type="verses" book="vv">
_Āyuñca vaṇṇañca sukhaṃ balañca,_  
_Paṇītarūpaṃ abhikaṅkhatā muni;_  
_Annañca pānañca bahuṃ susaṅkhataṃ,_  
_Patiṭṭhapetabbamasaṅgamānase._  

</Paragraph>

</Division>
<Division number="1047" book="vv" bookNo="19" page="95">

<Paragraph number="891" type="verses" book="vv">
_Nayimasmiṃ loke parasmiṃ vā pana,_  
_Buddhena seṭṭho va samo va vijjati;_  
_Āhuneyyānaṃ paramāhutiṃ gato,_  
_Puññatthikānaṃ vipulapphalesinan”ti._  


</Paragraph>
</Division>

<Paragraph number="892" type="center" book="vv">
Mahārathavimānaṃ cuddasamaṃ.

  
Mahārathavaggo pañcamo.



</Paragraph>

Tassuddānaṃ

_Dve karaṇīyā dve sūci,_  
_Tayo nāgā ca dve rathā;_  
_Purisānaṃ paṭhamo vaggo pavuccatīti._
