<Paragraph number="27" type="verses" book="vv">
_“Pīṭhaṃ te veḷuriyamayaṃ uḷāraṃ,_  
_Manojavaṃ gacchati yenakāmaṃ;_  
_Alaṅkate malyadhare suvatthe,_  
_Obhāsasi vijjurivabbhakūṭaṃ._  

</Paragraph>

</Division>
<Division number="24" book="vv" bookNo="19" page="4">

<Paragraph number="28" type="verses" book="vv">
_Kena tetādiso vaṇṇo,_  
_kena te idha mijjhati;_  
_Uppajjanti ca te bhogā,_  
_ye keci manaso piyā._  

</Paragraph>

</Division>
<Division number="25" book="vv" bookNo="19" page="4">

<Paragraph number="29" type="verses" book="vv">
_Pucchāmi taṃ devi mahānubhāve,_  
_Manussabhūtā kimakāsi puññaṃ;_  
_Kenāsi evaṃ jalitānubhāvā,_  
_Vaṇṇo ca te sabbadisā pabhāsatī”ti._  

</Paragraph>

</Division>
<Division number="26" book="vv" bookNo="19" page="4">

<Paragraph number="30" type="verses" book="vv">
_Sā devatā attamanā,_  
_moggallānena pucchitā;_  
_Pañhaṃ puṭṭhā viyākāsi,_  
_yassa kammassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="27" book="vv" bookNo="19" page="4">

<Paragraph number="31" type="verses" book="vv">
_“Appassa kammassa phalaṃ mamedaṃ,_  
_Yenamhi evaṃ jalitānubhāvā;_  
_Ahaṃ manussesu manussabhūtā,_  
_Purimāya jātiyā manussaloke._  

</Paragraph>

</Division>
<Division number="28" book="vv" bookNo="19" page="5">

<Paragraph number="32" type="verses" book="vv">
_Addasaṃ virajaṃ bhikkhuṃ,_  
_vippasannamanāvilaṃ;_  
_Tassa adāsahaṃ pīṭhaṃ,_  
_pasannā sehi pāṇibhi._  

</Paragraph>

</Division>
<Division number="29" book="vv" bookNo="19" page="5">

<Paragraph number="33" type="verses" book="vv">
_Tena metādiso vaṇṇo,_  
_tena me idha mijjhati;_  
_Uppajjanti ca me bhogā,_  
_ye keci manaso piyā._  

</Paragraph>

</Division>
<Division number="30" book="vv" bookNo="19" page="5">

<Paragraph number="34" type="verses" book="vv">
_Akkhāmi te bhikkhu mahānubhāva,_  
_Manussabhūtā yamakāsi puññaṃ;_  
_Tenamhi evaṃ jalitānubhāvā,_  
_Vaṇṇo ca me sabbadisā pabhāsatī”ti._  


</Paragraph>
</Division>

<Paragraph number="35" type="center" book="vv">
Catutthapīṭhavimānaṃ catutthaṃ.
</Paragraph>
