<Division number="1134" book="vv">

<Paragraph number="982" type="verses" book="vv">
_“Sovaṇṇamaye pabbatasmiṃ,_  
_vimānaṃ sabbatopabhaṃ;_  
_Hemajālapaṭicchannaṃ,_  
_kiṅkiṇijālakappitaṃ._  

</Paragraph>

</Division>
<Division number="1135" book="vv">

<Paragraph number="983" type="verses" book="vv">
_Aṭṭhaṃsā sukatā thambhā,_  
_sabbe veḷuriyāmayā;_  
_Ekamekāya aṃsiyā,_  
_ratanā satta nimmitā._  

</Paragraph>

</Division>
<Division number="1136" book="vv">

<Paragraph number="984" type="verses" book="vv">
_Veḷuriyasuvaṇṇassa,_  
_phalikā rūpiyassa ca;_  
_Masāragallamuttāhi,_  
_lohitaṅgamaṇīhi ca._  

</Paragraph>

</Division>
<Division number="1137" book="vv">

<Paragraph number="985" type="verses" book="vv">
_Citrā manoramā bhūmi,_  
_na tatthuddhaṃsatī rajo;_  
_Gopānasīgaṇā pītā,_  
_kūṭaṃ dhārenti nimmitā._  

</Paragraph>

</Division>
<Division number="1138" book="vv">

<Paragraph number="986" type="verses" book="vv">
_Sopānāni ca cattāri,_  
_nimmitā caturo disā;_  
_Nānāratanagabbhehi,_  
_ādiccova virocati._  

</Paragraph>

</Division>
<Division number="1139" book="vv">

<Paragraph number="987" type="verses" book="vv">
_Vediyā catasso tattha,_  
_vibhattā bhāgaso mitā;_  
_Daddallamānā ābhanti,_  
_samantā caturo disā._  

</Paragraph>

</Division>
<Division number="1140" book="vv">

<Paragraph number="988" type="verses" book="vv">
_Tasmiṃ vimāne pavare,_  
_devaputto mahappabho;_  
_Atirocasi vaṇṇena,_  
_udayantova bhāṇumā._  

</Paragraph>

</Division>
<Division number="1141" book="vv">

<Paragraph number="989" type="verses" book="vv">
_Dānassa te idaṃ phalaṃ,_  
_atho sīlassa vā pana;_  
_Atho añjalikammassa,_  
_taṃ me akkhāhi pucchito”._  

</Paragraph>

</Division>
<Division number="1142" book="vv">

<Paragraph number="990" type="verses" book="vv">
_So devaputto attamano,_  
_ …pe… _  
_yassa kammassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="1143" book="vv">

<Paragraph number="991" type="verses" book="vv">
_“Ahaṃ andhakavindasmiṃ,_  
_buddhassādiccabandhuno;_  
_Vihāraṃ satthu kāresiṃ,_  
_pasanno sehi pāṇibhi._  

</Paragraph>

</Division>
<Division number="1144" book="vv">

<Paragraph number="992" type="verses" book="vv">
_Tattha gandhañca mālañca,_  
_paccayañca vilepanaṃ;_  
_Vihāraṃ satthu adāsiṃ,_  
_vippasannena cetasā;_  
_Tena mayhaṃ idaṃ laddhaṃ,_  
_vasaṃ vattemi nandane._  

</Paragraph>

</Division>
<Division number="1145" book="vv">

<Paragraph number="993" type="verses" book="vv">
_Nandane ca vane ramme,_  
_nānādijagaṇāyute;_  
_Ramāmi naccagītehi,_  
_accharāhi purakkhato”ti._  


</Paragraph>
</Division>

<Paragraph number="994" type="center" book="vv">
Suvaṇṇavimānaṃ catutthaṃ.
</Paragraph>
