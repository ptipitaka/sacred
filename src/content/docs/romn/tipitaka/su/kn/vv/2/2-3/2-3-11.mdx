<Division number="1282" book="vv">

<Paragraph number="1136" type="verses" book="vv">
_“Uccamidaṃ maṇithūṇaṃ vimānaṃ,_  
_Samantato dvādasa yojanāni;_  
_Kūṭāgārā sattasatā uḷārā,_  
_Veḷuriyathambhā rucakatthatā subhā._  

</Paragraph>

</Division>
<Division number="1283" book="vv">

<Paragraph number="1137" type="verses" book="vv">
_Tatthacchasi pivasi khādasi ca,_  
_Dibbā ca vīṇā pavadanti vagguṃ;_  
_Dibbā rasā kāmaguṇettha pañca,_  
_Nāriyo ca naccanti suvaṇṇachannā._  

</Paragraph>

</Division>
<Division number="1284" book="vv">

<Paragraph number="1138" type="verses" book="vv">
_Kena tetādiso vaṇṇo,_  
_kena te idha mijjhati;_  
_Uppajjanti ca te bhogā,_  
_ye keci manaso piyā._  

</Paragraph>

</Division>
<Division number="1285" book="vv">

<Paragraph number="1139" type="verses" book="vv">
_Pucchāmi taṃ deva mahānubhāva,_  
_Manussabhūto kimakāsi puññaṃ;_  
_Kenāsi evaṃ jalitānubhāvo,_  
_Vaṇṇo ca te sabbadisā pabhāsatī”ti._  

</Paragraph>

</Division>
<Division number="1286" book="vv">

<Paragraph number="1140" type="verses" book="vv">
_So devaputto attamano,_  
_moggallānena pucchito;_  
_Pañhaṃ puṭṭho viyākāsi,_  
_yassa kammassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="1287" book="vv">

<Paragraph number="1141" type="verses" book="vv">
_“Dunnikkhittaṃ mālaṃ sunikkhipitvā,_  
_Patiṭṭhapetvā sugatassa thūpe;_  
_Mahiddhiko camhi mahānubhāvo,_  
_Dibbehi kāmehi samaṅgibhūto._  

</Paragraph>

</Division>
<Division number="1288" book="vv">

<Paragraph number="1142" type="verses" book="vv">
_Tena metādiso vaṇṇo,_  
_tena me idha mijjhati;_  
_Uppajjanti ca me bhogā,_  
_ye keci manaso piyā._  

</Paragraph>

</Division>
<Division number="1289" book="vv">

<Paragraph number="1143" type="verses" book="vv">
_Akkhāmi te bhikkhu mahānubhāva,_  
_Manussabhūto yamahaṃ akāsiṃ;_  
_Tenamhi evaṃ jalitānubhāvo,_  
_Vaṇṇo ca me sabbadisā pabhāsatī”ti._  


</Paragraph>
</Division>

<Paragraph number="1144" type="center" book="vv">
Sunikkhittavimānaṃ ekādasamaṃ.

  
Sunikkhittavaggo sattamo.



</Paragraph>

<Paragraph number="1145" type="center" book="vv">
Tassuddānaṃ


</Paragraph>

<Paragraph number="1146" type="verses" book="vv">
_Dve daliddā vanavihārā,_  
_Bhatako gopālakaṇḍako;_  
_Anekavaṇṇamaṭṭhakuṇḍalī,_  
_Serīsako sunikkhittaṃ;_  
_Purisānaṃ tatiyo vaggo pavuccatīti._  


</Paragraph>

<Paragraph number="1147" type="center" book="vv">
Bhāṇavāraṃ catutthaṃ.



</Paragraph>

<Paragraph number="1148" type="center" book="vv">
Vimānavatthupāḷi niṭṭhitā.
</Paragraph>
