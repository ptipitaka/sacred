<Paragraph number="1050" type="verses" book="vv">
_“Anekavaṇṇaṃ darasokanāsanaṃ,_  
_Vimānamāruyha anekacittaṃ;_  
_Parivārito accharāsaṅgaṇena,_  
_Sunimmito bhūtapatīva modasi._  

</Paragraph>

</Division>
<Division number="1200" book="vv">

<Paragraph number="1051" type="verses" book="vv">
_Samassamo natthi kuto panuttaro,_  
_Yasena puññena ca iddhiyā ca;_  
_Sabbe ca devā tidasagaṇā samecca,_  
_Taṃ taṃ namassanti sasiṃva devā;_  
_Imā ca te accharāyo samantato,_  
_Naccanti gāyanti pamodayanti._  

</Paragraph>

</Division>
<Division number="1201" book="vv">

<Paragraph number="1052" type="verses" book="vv">
_Deviddhipattosi mahānubhāvo,_  
_Manussabhūto kimakāsi puññaṃ;_  
_Kenāsi evaṃ jalitānubhāvo,_  
_Vaṇṇo ca te sabbadisā pabhāsatī”ti._  

</Paragraph>

</Division>
<Division number="1202" book="vv">

<Paragraph number="1053" type="verses" book="vv">
_So devaputto attamano,_  
_ …pe… _  
_yassa kammassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="1203" book="vv">

<Paragraph number="1054" type="verses" book="vv">
_“Ahaṃ bhadante ahuvāsi pubbe,_  
_Sumedhanāmassa jinassa sāvako;_  
_Puthujjano ananubodhohamasmi,_  
_So satta vassāni paribbajissahaṃ._  

</Paragraph>

</Division>
<Division number="1204" book="vv">

<Paragraph number="1055" type="verses" book="vv">
_Sohaṃ sumedhassa jinassa satthuno,_  
_Parinibbutassoghatiṇṇassa tādino;_  
_Ratanuccayaṃ hemajālena channaṃ,_  
_Vanditvā thūpasmiṃ manaṃ pasādayiṃ._  

</Paragraph>

</Division>
<Division number="1205" book="vv">

<Paragraph number="1056" type="verses" book="vv">
_Na māsi dānaṃ na ca matthi dātuṃ,_  
_Pare ca kho tattha samādapesiṃ;_  
_Pūjetha naṃ pūjanīyassa dhātuṃ,_  
_Evaṃ kira saggamito gamissatha._  

</Paragraph>

</Division>
<Division number="1206" book="vv">

<Paragraph number="1057" type="verses" book="vv">
_Tadeva kammaṃ kusalaṃ kataṃ mayā,_  
_Sukhañca dibbaṃ anubhomi attanā;_  
_Modāmahaṃ tidasagaṇassa majjhe,_  
_Na tassa puññassa khayampi ajjhagan”ti._  


</Paragraph>
</Division>

<Paragraph number="1058" type="center" book="vv">
Anekavaṇṇavimānaṃ aṭṭhamaṃ.
</Paragraph>
