<Division number="85" book="vv">

<Paragraph number="95" type="verses" book="vv">
_“Abhikkantena vaṇṇena,_  
_yā tvaṃ tiṭṭhasi devate;_  
_Obhāsentī disā sabbā,_  
_osadhī viya tārakā._  

</Paragraph>

</Division>
<Division number="86" book="vv">

<Paragraph number="96" type="verses" book="vv">
_Kena tetādiso vaṇṇo,_  
_kena te idha mijjhati;_  
_Uppajjanti ca te bhogā,_  
_ye keci manaso piyā._  

</Paragraph>

</Division>
<Division number="87" book="vv">

<Paragraph number="97" type="verses" book="vv">
_Pucchāmi taṃ devi mahānubhāve,_  
_Manussabhūtā kimakāsi puññaṃ;_  
_Kenāsi evaṃ jalitānubhāvā,_  
_Vaṇṇo ca te sabbadisā pabhāsatī”ti._  

</Paragraph>

</Division>
<Division number="88" book="vv">

<Paragraph number="98" type="verses" book="vv">
_Sā devatā attamanā,_  
_moggallānena pucchitā;_  
_Pañhaṃ puṭṭhā viyākāsi,_  
_yassa kammassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="89" book="vv">

<Paragraph number="99" type="verses" book="vv">
_“Ahaṃ manussesu manussabhūtā,_  
_Purimāya jātiyā manussaloke._  

</Paragraph>

</Division>
<Division number="90" book="vv">

<Paragraph number="100" type="verses" book="vv">
_Addasaṃ virajaṃ buddhaṃ,_  
_vippasannamanāvilaṃ;_  
_Āsajja dānaṃ adāsiṃ,_  
_akāmā tiladakkhiṇaṃ;_  
_Dakkhiṇeyyassa buddhassa,_  
_pasannā sehi pāṇibhi._  

</Paragraph>

</Division>
<Division number="91" book="vv">

<Paragraph number="101" type="verses" book="vv">
_Tena metādiso vaṇṇo,_  
_tena me idha mijjhati;_  
_Uppajjanti ca me bhogā,_  
_ye keci manaso piyā._  

</Paragraph>

</Division>
<Division number="92" book="vv">

<Paragraph number="102" type="verses" book="vv">
_Akkhāmi te bhikkhu mahānubhāva,_  
_Manussabhūtā yamakāsi puññaṃ;_  
_Tenamhi evaṃ jalitānubhāvā,_  
_Vaṇṇo ca me sabbadisā pabhāsatī”ti._  


</Paragraph>
</Division>

<Paragraph number="103" type="center" book="vv">
Tiladakkhiṇavimānaṃ dasamaṃ.
</Paragraph>
