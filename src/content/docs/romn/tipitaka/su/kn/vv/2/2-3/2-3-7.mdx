_nakkhattaparivārito;_  
_Samantā anupariyāti,_  
_tārakādhipatī sasī._

_Tathūpamaṃ idaṃ byamhaṃ,_  
_dibbaṃ devapuramhi ca;_  
_Atirocati vaṇṇena,_  
_udayantova raṃsimā._  

</Paragraph>

</Division>
<Division number="1173" book="vv" bookNo="19" page="109">

<Paragraph number="1023" type="verses" book="vv">
_Veḷuriyasuvaṇṇassa,_  
_phalikā rūpiyassa ca;_  
_Masāragallamuttāhi,_  
_lohitaṅgamaṇīhi ca._  

</Paragraph>

</Division>
<Division number="1174" book="vv" bookNo="19" page="109">

<Paragraph number="1024" type="verses" book="vv">
_Citrā manoramā bhūmi,_  
_veḷūriyassa santhatā;_  
_Kūṭāgārā subhā rammā,_  
_pāsādo te sumāpito._  

</Paragraph>

</Division>
<Division number="1175" book="vv" bookNo="19" page="110">

<Paragraph number="1025" type="verses" book="vv">
_Rammā ca te pokkharaṇī,_  
_puthulomanisevitā;_  
_Acchodakā vippasannā,_  
_sovaṇṇavālukasanthatā._  

</Paragraph>

</Division>
<Division number="1176" book="vv" bookNo="19" page="110">

<Paragraph number="1026" type="verses" book="vv">
_Nānāpadumasañchannā,_  
_puṇḍarīkasamotatā;_  
_Surabhiṃ sampavāyanti,_  
_manuññā māluteritā._  

</Paragraph>

</Division>
<Division number="1177" book="vv" bookNo="19" page="110">

<Paragraph number="1027" type="verses" book="vv">
_Tassā te ubhato passe,_  
_vanagumbā sumāpitā;_  
_Upetā puppharukkhehi,_  
_phalarukkhehi cūbhayaṃ._  

</Paragraph>

</Division>
<Division number="1178" book="vv" bookNo="19" page="110">

<Paragraph number="1028" type="verses" book="vv">
_Sovaṇṇapāde pallaṅke,_  
_muduke gonakatthate;_  
_Nisinnaṃ devarājaṃva,_  
_upatiṭṭhanti accharā._  

</Paragraph>

</Division>
<Division number="1179" book="vv" bookNo="19" page="110">

<Paragraph number="1029" type="verses" book="vv">
_Sabbābharaṇasañchannā,_  
_nānāmālāvibhūsitā;_  
_Ramenti taṃ mahiddhikaṃ,_  
_vasavattīva modasi._  

</Paragraph>

</Division>
<Division number="1180" book="vv" bookNo="19" page="110">

<Paragraph number="1030" type="verses" book="vv">
_Bherisaṅkhamudiṅgāhi,_  
_vīṇāhi paṇavehi ca;_  
_Ramasi ratisampanno,_  
_naccagīte suvādite._  

</Paragraph>

</Division>
<Division number="1181" book="vv" bookNo="19" page="110">

<Paragraph number="1031" type="verses" book="vv">
_Dibbā te vividhā rūpā,_  
_dibbā saddā atho rasā;_  
_Gandhā ca te adhippetā,_  
_phoṭṭhabbā ca manoramā._  

</Paragraph>

</Division>
<Division number="1182" book="vv" bookNo="19" page="110">

<Paragraph number="1032" type="verses" book="vv">
_Tasmiṃ vimāne pavare,_  
_devaputta mahappabho;_  
_Atirocasi vaṇṇena,_  
_udayantova bhāṇumā._  

</Paragraph>

</Division>
<Division number="1183" book="vv" bookNo="19" page="110">

<Paragraph number="1033" type="verses" book="vv">
_Dānassa te idaṃ phalaṃ,_  
_atho sīlassa vā pana;_  
_Atho añjalikammassa,_  
_taṃ me akkhāhi pucchito”._  

</Paragraph>

</Division>
<Division number="1184" book="vv" bookNo="19" page="110">

<Paragraph number="1034" type="verses" book="vv">
_So devaputto attamano,_  
_ …pe… _  
_yassa kammassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="1185" book="vv" bookNo="19" page="110">

<Paragraph number="1035" type="verses" book="vv">
_“Ahaṃ kapilavatthusmiṃ,_  
_sākiyānaṃ puruttame;_  
_Suddhodanassa puttassa,_  
_kaṇḍako sahajo ahaṃ._  

</Paragraph>

</Division>
<Division number="1186" book="vv" bookNo="19" page="110">

<Paragraph number="1036" type="verses" book="vv">
_Yadā so aḍḍharattāyaṃ,_  
_bodhāya mabhinikkhami;_  
_So maṃ mudūhi pāṇīhi,_  
_jāli tambanakhehi ca._  

</Paragraph>

</Division>
<Division number="1187" book="vv" bookNo="19" page="110">

<Paragraph number="1037" type="verses" book="vv">
_Satthiṃ ākoṭayitvāna,_  
_vaha sammāti cabravi;_  
_Ahaṃ lokaṃ tārayissaṃ,_  
_patto sambodhimuttamaṃ._  

</Paragraph>

</Division>
<Division number="1188" book="vv" bookNo="19" page="111">

<Paragraph number="1038" type="verses" book="vv">
_Taṃ me giraṃ suṇantassa,_  
_hāso me vipulo ahu;_  
_Udaggacitto sumano,_  
_abhisīsiṃ tadā ahaṃ._  

</Paragraph>

</Division>
<Division number="1189" book="vv" bookNo="19" page="111">

<Paragraph number="1039" type="verses" book="vv">
_Abhirūḷhañca maṃ ñatvā,_  
_sakyaputtaṃ mahāyasaṃ;_  
_Udaggacitto mudito,_  
_vahissaṃ purisuttamaṃ._  

</Paragraph>

</Division>
<Division number="1190" book="vv" bookNo="19" page="111">

<Paragraph number="1040" type="verses" book="vv">
_Paresaṃ vijitaṃ gantvā,_  
_uggatasmiṃ divākare;_  
_Mamaṃ channañca ohāya,_  
_anapekkho so apakkami._  

</Paragraph>

</Division>
<Division number="1191" book="vv" bookNo="19" page="111">

<Paragraph number="1041" type="verses" book="vv">
_Tassa tambanakhe pāde,_  
_jivhāya parilehisaṃ;_  
_Gacchantañca mahāvīraṃ,_  
_rudamāno udikkhisaṃ._  

</Paragraph>

</Division>
<Division number="1192" book="vv" bookNo="19" page="111">

<Paragraph number="1042" type="verses" book="vv">
_Adassanenahaṃ tassa,_  
_sakyaputtassa sirīmato;_  
_Alatthaṃ garukābādhaṃ,_  
_khippaṃ me maraṇaṃ ahu._  

</Paragraph>

</Division>
<Division number="1193" book="vv" bookNo="19" page="111">

<Paragraph number="1043" type="verses" book="vv">
_Tasseva ānubhāvena,_  
_vimānaṃ āvasāmidaṃ;_  
_Sabbakāmaguṇopetaṃ,_  
_dibbaṃ devapuramhi ca._  

</Paragraph>

</Division>
<Division number="1194" book="vv" bookNo="19" page="111">

<Paragraph number="1044" type="verses" book="vv">
_Yañca me ahuvā hāso,_  
_saddaṃ sutvāna bodhiyā;_  
_Teneva kusalamūlena,_  
_phusissaṃ āsavakkhayaṃ._  

</Paragraph>

</Division>
<Division number="1195" book="vv" bookNo="19" page="111">

<Paragraph number="1045" type="verses" book="vv">
_Sace hi bhante gaccheyyāsi,_  
_satthu buddhassa santike;_  
_Mamāpi naṃ vacanena,_  
_sirasā vajjāsi vandanaṃ._  

</Paragraph>

</Division>
<Division number="1196" book="vv" bookNo="19" page="111">

<Paragraph number="1046" type="verses" book="vv">
_Ahampi daṭṭhuṃ gacchissaṃ,_  
_jinaṃ appaṭipuggalaṃ;_  
_Dullabhaṃ dassanaṃ hoti,_  
_lokanāthāna tādinan”ti._  

</Paragraph>

</Division>
<Division number="1197" book="vv" bookNo="19" page="111">

<Paragraph number="1047" type="verses" book="vv">
_So kataññū katavedī,_  
_satthāraṃ upasaṅkami;_  
_Sutvā giraṃ cakkhumato,_  
_dhammacakkhuṃ visodhayi._  

</Paragraph>

</Division>
<Division number="1198" book="vv" bookNo="19" page="111">

<Paragraph number="1048" type="verses" book="vv">
_Visodhetvā diṭṭhigataṃ,_  
_vicikicchaṃ vatāni ca;_  
_Vanditvā satthuno pāde,_  
_tatthevantaradhāyathāti._  


</Paragraph>
</Division>
