<Paragraph number="632" type="verses" book="vv">
_“Pītavatthe pītadhaje,_  
_pītālaṅkārabhūsite;_  
_Pītacandanalittaṅge,_  
_pītauppalamālinī._  

</Paragraph>

</Division>
<Division number="796" book="vv" bookNo="19" page="65">

<Paragraph number="633" type="verses" book="vv">
_Pītapāsādasayane,_  
_pītāsane pītabhājane;_  
_Pītachatte pītarathe,_  
_pītasse pītabījane._  

</Paragraph>

</Division>
<Division number="797" book="vv" bookNo="19" page="66">

<Paragraph number="634" type="verses" book="vv">
_Kiṃ kammamakarī bhadde,_  
_pubbe mānusake bhave;_  
_Devate pucchitācikkha,_  
_kissa kammassidaṃ phalan”ti._  

</Paragraph>

</Division>
<Division number="798" book="vv" bookNo="19" page="66">

<Paragraph number="635" type="verses" book="vv">
_“Kosātakī nāma latatthi bhante,_  
_Tittikā anabhicchitā;_  
_Tassā cattāri pupphāni,_  
_Thūpaṃ abhihariṃ ahaṃ._  

</Paragraph>

</Division>
<Division number="799" book="vv" bookNo="19" page="66">

<Paragraph number="636" type="verses" book="vv">
_Satthu sarīramuddissa,_  
_vippasannena cetasā;_  
_Nāssa maggaṃ avekkhissaṃ,_  
_na taggamanasā satī._  

</Paragraph>

</Division>
<Division number="800" book="vv" bookNo="19" page="66">

<Paragraph number="637" type="verses" book="vv">
_Tato maṃ avadhī gāvī,_  
_thūpaṃ apattamānasaṃ;_  
_Tañcāhaṃ abhisañceyyaṃ,_  
_bhiyyo nūna ito siyā._  

</Paragraph>

</Division>
<Division number="801" book="vv" bookNo="19" page="66">

<Paragraph number="638" type="verses" book="vv">
_Tena kammena devinda,_  
_maghavā devakuñjara;_  
_Pahāya mānusaṃ dehaṃ,_  
_tava sahabyamāgatā”ti._  

</Paragraph>

</Division>
<Division number="802" book="vv" bookNo="19" page="66">

<Paragraph number="639" type="verses" book="vv">
_Idaṃ sutvā tidasādhipati,_  
_Maghavā devakuñjaro;_  
_Tāvatiṃse pasādento,_  
_Mātaliṃ etadabravi._  

</Paragraph>

</Division>
<Division number="803" book="vv" bookNo="19" page="66">

<Paragraph number="640" type="verses" book="vv">
_“Passa mātali accheraṃ,_  
_cittaṃ kammaphalaṃ idaṃ;_  
_Appakampi kataṃ deyyaṃ,_  
_puññaṃ hoti mahapphalaṃ._  

</Paragraph>

</Division>
<Division number="804" book="vv" bookNo="19" page="66">

<Paragraph number="641" type="verses" book="vv">
_Natthi citte pasannamhi,_  
_appakā nāma dakkhiṇā;_  
_Tathāgate vā sambuddhe,_  
_atha vā tassa sāvake._  

</Paragraph>

</Division>
<Division number="805" book="vv" bookNo="19" page="66">

<Paragraph number="642" type="verses" book="vv">
_Ehi mātali amhepi,_  
_bhiyyo bhiyyo mahemase;_  
_Tathāgatassa dhātuyo,_  
_sukho puññāna muccayo._  

</Paragraph>

</Division>
<Division number="806" book="vv" bookNo="19" page="66">

<Paragraph number="643" type="verses" book="vv">
_Tiṭṭhante nibbute cāpi,_  
_same citte samaṃ phalaṃ;_  
_Cetopaṇidhihetu hi,_  
_sattā gacchanti suggatiṃ._  

</Paragraph>

</Division>
<Division number="807" book="vv" bookNo="19" page="66">

<Paragraph number="644" type="verses" book="vv">
_Bahūnaṃ vata atthāya,_  
_uppajjanti tathāgatā;_  
_Yattha kāraṃ karitvāna,_  
_saggaṃ gacchanti dāyakā”ti._  


</Paragraph>
</Division>

<Paragraph number="645" type="center" book="vv">
Pītavimānaṃ navamaṃ.
</Paragraph>
