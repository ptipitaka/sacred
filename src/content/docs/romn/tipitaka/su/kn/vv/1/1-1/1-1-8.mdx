<Paragraph number="71" type="verses" book="vv">
_“Suvaṇṇacchadanaṃ nāvaṃ,_  
_Nāri āruyha tiṭṭhasi;_  
_Ogāhasi pokkharaṇiṃ,_  
_Padmaṃ chindasi pāṇinā._  

</Paragraph>

</Division>
<Division number="64" book="vv" bookNo="19" page="9">

<Paragraph number="72" type="verses" book="vv">
_Kūṭāgārā nivesā te,_  
_vibhattā bhāgaso mitā;_  
_Daddallamānā ābhanti,_  
_samantā caturo disā._  

</Paragraph>

</Division>
<Division number="65" book="vv" bookNo="19" page="9">

<Paragraph number="73" type="verses" book="vv">
_Kena tetādiso vaṇṇo,_  
_kena te idha mijjhati;_  
_Uppajjanti ca te bhogā,_  
_ye keci manaso piyā._  

</Paragraph>

</Division>
<Division number="66" book="vv" bookNo="19" page="9">

<Paragraph number="74" type="verses" book="vv">
_Pucchāmi taṃ devi mahānubhāve,_  
_Manussabhūtā kimakāsi puññaṃ;_  
_Kenāsi evaṃ jalitānubhāvā,_  
_Vaṇṇo ca te sabbadisā pabhāsatī”ti._  

</Paragraph>

</Division>
<Division number="67" book="vv" bookNo="19" page="9">

<Paragraph number="75" type="verses" book="vv">
_Sā devatā attamanā,_  
_sambuddheneva pucchitā;_  
_Pañhaṃ puṭṭhā viyākāsi,_  
_yassa kammassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="68" book="vv" bookNo="19" page="9">

<Paragraph number="76" type="verses" book="vv">
_“Ahaṃ manussesu manussabhūtā,_  
_Purimāya jātiyā manussaloke;_  
_Disvāna bhikkhū tasite kilante,_  
_Uṭṭhāya pātuṃ udakaṃ adāsiṃ._  

</Paragraph>

</Division>
<Division number="69" book="vv" bookNo="19" page="9">

<Paragraph number="77" type="verses" book="vv">
_Yo ve kilantāna pipāsitānaṃ,_  
_Uṭṭhāya pātuṃ udakaṃ dadāti;_  
_Sītodakā tassa bhavanti najjo,_  
_Pahūtamalyā bahupuṇḍarīkā._  

</Paragraph>

</Division>
<Division number="70" book="vv" bookNo="19" page="10">

<Paragraph number="78" type="verses" book="vv">
_Taṃ āpagā anupariyanti sabbadā,_  
_Sītodakā vālukasanthatā nadī;_  
_Ambā ca sālā tilakā ca jambuyo,_  
_Uddālakā pāṭaliyo ca phullā._  

</Paragraph>

</Division>
<Division number="71" book="vv" bookNo="19" page="10">

<Paragraph number="79" type="verses" book="vv">
_Taṃ bhūmibhāgehi upetarūpaṃ,_  
_Vimānaseṭṭhaṃ bhusa sobhamānaṃ;_  
_Tassīdha kammassa ayaṃ vipāko,_  
_Etādisaṃ puññakatā labhanti._  

</Paragraph>

</Division>
<Division number="72" book="vv" bookNo="19" page="10">

<Paragraph number="80" type="verses" book="vv">
_Kūṭāgārā nivesā me,_  
_vibhattā bhāgaso mitā;_  
_Daddallamānā ābhanti,_  
_samantā caturo disā._  

</Paragraph>

</Division>
<Division number="73" book="vv" bookNo="19" page="10">

<Paragraph number="81" type="verses" book="vv">
_Tena metādiso vaṇṇo,_  
_tena me idha mijjhati;_  
_Uppajjanti ca me bhogā,_  
_ye keci manaso piyā._  

</Paragraph>

</Division>
<Division number="74" book="vv" bookNo="19" page="10">

<Paragraph number="82" type="verses" book="vv">
_Akkhāmi te buddha mahānubhāva,_  
_Manussabhūtā yamakāsi puññaṃ;_  
_Tenamhi evaṃ jalitānubhāvā,_  
_Vaṇṇo ca me sabbadisā pabhāsati;_  
_Etassa kammassa phalaṃ mamedaṃ,_  
_Atthāya buddho udakaṃ apāyī”ti._  


</Paragraph>
</Division>

<Paragraph number="83" type="center" book="vv">
Tatiyanāvāvimānaṃ aṭṭhamaṃ.
