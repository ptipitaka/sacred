_ramme cittalatāvane;_  
_Samantā anupariyāsi,_  
_nārīgaṇapurakkhatā;_  
_Obhāsentī disā sabbā,_  
_osadhī viya tārakā._  

</Paragraph>

</Division>
<Division number="158" book="vv">

<Paragraph number="179" type="verses" book="vv">
_Kena tetādiso vaṇṇo,_  
_kena te idha mijjhati;_  
_Uppajjanti ca te bhogā,_  
_ye keci manaso piyā._  

</Paragraph>

</Division>
<Division number="159" book="vv">

<Paragraph number="180" type="verses" book="vv">
_Pucchāmi taṃ devi mahānubhāve,_  
_Manussabhūtā kimakāsi puññaṃ;_  
_Kenāsi evaṃ jalitānubhāvā,_  
_Vaṇṇo ca te sabbadisā pabhāsatī”ti._  

</Paragraph>

</Division>
<Division number="160" book="vv">

<Paragraph number="181" type="verses" book="vv">
_Sā devatā attamanā,_  
_moggallānena pucchitā;_  
_Pañhaṃ puṭṭhā viyākāsi,_  
_yassa kammassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="161" book="vv">

<Paragraph number="182" type="verses" book="vv">
_“Ahaṃ manussesu manussabhūtā,_  
_Dāsī ahosiṃ parapessiyā kule._  

</Paragraph>

</Division>
<Division number="162" book="vv">

<Paragraph number="183" type="verses" book="vv">
_Upāsikā cakkhumato,_  
_gotamassa yasassino;_  
_Tassā me nikkamo āsi,_  
_sāsane tassa tādino._  

</Paragraph>

</Division>
<Division number="163" book="vv">

<Paragraph number="184" type="verses" book="vv">
_Kāmaṃ bhijjatuyaṃ kāyo,_  
_neva atthettha saṇṭhanaṃ;_  
_Sikkhāpadānaṃ pañcannaṃ,_  
_maggo sovatthiko sivo._  

</Paragraph>

</Division>
<Division number="164" book="vv">

<Paragraph number="185" type="verses" book="vv">
_Akaṇṭako agahano,_  
_uju sabbhi pavedito;_  
_Nikkamassa phalaṃ passa,_  
_yathidaṃ pāpuṇitthikā._  

</Paragraph>

</Division>
<Division number="165" book="vv">

<Paragraph number="186" type="verses" book="vv">
_Āmantanikā raññomhi,_  
_sakkassa vasavattino;_  
_Saṭṭhi tūriyasahassāni,_  
_paṭibodhaṃ karonti me._  

</Paragraph>

</Division>
<Division number="166" book="vv">

<Paragraph number="187" type="verses" book="vv">
_Ālambo gaggaro bhīmo,_  
_sādhuvādī ca saṃsayo;_  
_Pokkharo ca suphasso ca,_  
_viṇāmokkhā ca nāriyo._  

</Paragraph>

</Division>
<Division number="167" book="vv">

<Paragraph number="188" type="verses" book="vv">
_Nandā ceva sunandā ca,_  
_soṇadinnā sucimhitā;_  
_Alambusā missakesī ca,_  
_puṇḍarīkāti dāruṇī._  

</Paragraph>

</Division>
<Division number="168" book="vv">

<Paragraph number="189" type="verses" book="vv">
_Eṇīphassā suphassā ca,_  
_subhaddā muduvādinī;_  
_Etā caññā ca seyyāse,_  
_accharānaṃ pabodhikā._  

</Paragraph>

</Division>
<Division number="169" book="vv">

<Paragraph number="190" type="verses" book="vv">
_Tā maṃ kālenupāgantvā,_  
_abhibhāsanti devatā;_  
_Handa naccāma gāyāma,_  
_handa taṃ ramayāmase._  

</Paragraph>

</Division>
<Division number="170" book="vv">

<Paragraph number="191" type="verses" book="vv">
_Nayidaṃ akatapuññānaṃ,_  
_katapuññānamevidaṃ;_  
_Asokaṃ nandanaṃ rammaṃ,_  
_tidasānaṃ mahāvanaṃ._  

</Paragraph>

</Division>
<Division number="171" book="vv">

<Paragraph number="192" type="verses" book="vv">
_Sukhaṃ akatapuññānaṃ,_  
_idha natthi parattha ca;_  
_Sukhañca katapuññānaṃ,_  
_idha ceva parattha ca._  

</Paragraph>

</Division>
<Division number="172" book="vv">

<Paragraph number="193" type="verses" book="vv">
_Tesaṃ sahabyakāmānaṃ,_  
_kattabbaṃ kusalaṃ bahuṃ;_  
_Katapuññā hi modanti,_  
_sagge bhogasamaṅgino”ti._  


</Paragraph>
</Division>
