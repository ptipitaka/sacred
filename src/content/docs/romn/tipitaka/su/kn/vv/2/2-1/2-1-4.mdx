<Paragraph number="756" type="verses" book="vv">
_“Uccamidaṃ maṇithūṇaṃ vimānaṃ,_  
_Samantato dvādasa yojanāni;_  
_Kūṭāgārā sattasatā uḷārā,_  
_Veḷuriyathambhā rucakatthatā subhā._  

</Paragraph>

</Division>
<Division number="911" book="vv" bookNo="19" page="80">

<Paragraph number="757" type="verses" book="vv">
_Tatthacchasi pivasi khādasi ca,_  
_Dibbā ca vīṇā pavadanti vagguṃ;_  
_Dibbā rasā kāmaguṇettha pañca,_  
_Nāriyo ca naccanti suvaṇṇachannā._  

</Paragraph>

</Division>
<Division number="912" book="vv" bookNo="19" page="80">

<Paragraph number="758" type="verses" book="vv">
_Kena tetādiso vaṇṇo,_  
_kena te idha mijjhati;_  
_Uppajjanti ca te bhogā,_  
_ye keci manaso piyā._  

</Paragraph>

</Division>
<Division number="913" book="vv" bookNo="19" page="80">

<Paragraph number="759" type="verses" book="vv">
_Pucchāmi taṃ deva mahānubhāva,_  
_Manussabhūto kimakāsi puññaṃ;_  
_Kenāsi evaṃ jalitānubhāvo,_  
_Vaṇṇo ca te sabbadisā pabhāsatī”ti._  

</Paragraph>

</Division>
<Division number="914" book="vv" bookNo="19" page="80">

<Paragraph number="760" type="verses" book="vv">
_So devaputto attamano,_  
_moggallānena pucchito;_  
_Pañhaṃ puṭṭho viyākāsi,_  
_yassa kammassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="915" book="vv" bookNo="19" page="80">

<Paragraph number="761" type="verses" book="vv">
_“Satisamuppādakaro,_  
_dvāre kakkaṭako ṭhito;_  
_Niṭṭhito jātarūpassa,_  
_sobhati dasapādako._  

</Paragraph>

</Division>
<Division number="916" book="vv" bookNo="19" page="81">

<Paragraph number="762" type="verses" book="vv">
_Tena metādiso vaṇṇo,_  
_tena me idha mijjhati;_  
_Uppajjanti ca me bhogā,_  
_ye keci manaso piyā._  

</Paragraph>

</Division>
<Division number="917" book="vv" bookNo="19" page="81">

<Paragraph number="763" type="verses" book="vv">
_Akkhāmi te bhikkhu mahānubhāva,_  
_Manussabhūto yamakāsi puññaṃ;_  
_Tenamhi evaṃ jalitānubhāvo,_  
_Vaṇṇo ca me sabbadisā pabhāsatī”ti._  


</Paragraph>
</Division>

<Paragraph number="764" type="center" book="vv">
Kakkaṭakarasadāyakavimānaṃ catutthaṃ.


</Paragraph>

<Paragraph number="765" type="center" book="vv">
(Anantaraṃ pañcavimānaṃ yathā kakkaṭakarasadāyakavimānaṃ tathā vitthāretabbaṃ.)
</Paragraph>
