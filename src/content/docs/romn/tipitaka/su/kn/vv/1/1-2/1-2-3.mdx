_tuṇhībhūtassa tiṭṭhato;_  
_Daliddā kapaṇā nārī,_  
_parāgāraṃ apassitā._

<Division number="188" book="vv" bookNo="19" page="24">

<Paragraph number="211" type="verses" book="vv">
_Yā me adāsi ācāmaṃ,_  
_pasannā sehi pāṇibhi;_  
_Sā hitvā mānusaṃ dehaṃ,_  
_vippamuttā ito cutā._  

</Paragraph>

</Division>
<Division number="189" book="vv" bookNo="19" page="24">

<Paragraph number="212" type="verses" book="vv">
_Nimmānaratino nāma,_  
_santi devā mahiddhikā;_  
_Tattha sā sukhitā nārī,_  
_modatācāmadāyikā”ti._  

</Paragraph>

</Division>
<Division number="190" book="vv" bookNo="19" page="25">

<Paragraph number="213" type="verses" book="vv">
_“Aho dānaṃ varākiyā,_  
_kassape suppatiṭṭhitaṃ;_  
_Parābhatena dānena,_  
_ijjhittha vata dakkhiṇā._  

</Paragraph>

</Division>
<Division number="191" book="vv" bookNo="19" page="25">

<Paragraph number="214" type="verses" book="vv">
_Yā mahesittaṃ kāreyya,_  
_cakkavattissa rājino;_  
_Nārī sabbaṅgakalyāṇī,_  
_bhattu cānomadassikā;_  
_Etassācāmadānassa,_  
_kalaṃ nāgghati soḷasiṃ._  

</Paragraph>

</Division>
<Division number="192" book="vv" bookNo="19" page="25">

<Paragraph number="215" type="verses" book="vv">
_Sataṃ nikkhā sataṃ assā,_  
_sataṃ assatarīrathā;_  
_Sataṃ kaññāsahassāni,_  
_āmuttamaṇikuṇḍalā;_  
_Etassācāmadānassa,_  
_kalaṃ nāgghanti soḷasiṃ._  

</Paragraph>

</Division>
<Division number="193" book="vv" bookNo="19" page="25">

<Paragraph number="216" type="verses" book="vv">
_Sataṃ hemavatā nāgā,_  
_īsādantā urūḷhavā;_  
_Suvaṇṇakacchā mātaṅgā,_  
_hemakappanavāsasā;_  
_Etassācāmadānassa,_  
_kalaṃ nāgghati soḷasiṃ._  

</Paragraph>

</Division>
<Division number="194" book="vv" bookNo="19" page="25">

<Paragraph number="217" type="verses" book="vv">
_Catunnamapi dīpānaṃ,_  
_issaraṃ yodha kāraye;_  
_Etassācāmadānassa,_  
_kalaṃ nāgghati soḷasin”ti._  


</Paragraph>
</Division>
