<Paragraph number="731" type="verses" book="vv">
_“Yo vadataṃ pavaro manujesu,_  
_Sakyamunī bhagavā katakicco;_  
_Pāragato balaviriyasamaṅgī,_  
_Taṃ sugataṃ saraṇatthamupehi._  

</Paragraph>

</Division>
<Division number="887" book="vv" bookNo="19" page="76">

<Paragraph number="732" type="verses" book="vv">
_Rāgavirāgamanejamasokaṃ,_  
_Dhammamasaṅkhatamappaṭikūlaṃ;_  
_Madhuramimaṃ paguṇaṃ suvibhattaṃ,_  
_Dhammamimaṃ saraṇatthamupehi._  

</Paragraph>

</Division>
<Division number="888" book="vv" bookNo="19" page="76">

<Paragraph number="733" type="verses" book="vv">
_Yattha ca dinna mahapphalamāhu,_  
_Catūsu sucīsu purisayugesu;_  
_Aṭṭha ca puggaladhammadasā te,_  
_Saṃghamimaṃ saraṇatthamupehi._  

</Paragraph>

</Division>
<Division number="889" book="vv" bookNo="19" page="77">

<Paragraph number="734" type="verses" book="vv">
_Na tathā tapati nabhe sūriyo,_  
_Cando ca na bhāsati na phusso;_  
_Yathā atulamidaṃ mahappabhāsaṃ,_  
_Ko nu tvaṃ tidivā mahiṃ upāgā._  

</Paragraph>

</Division>
<Division number="890" book="vv" bookNo="19" page="77">

<Paragraph number="735" type="verses" book="vv">
_Chindati raṃsī pabhaṅkarassa,_  
_Sādhikavīsatiyojanāni ābhā;_  
_Rattimapi yathā divaṃ karoti,_  
_Parisuddhaṃ vimalaṃ subhaṃ vimānaṃ._  

</Paragraph>

</Division>
<Division number="891" book="vv" bookNo="19" page="77">

<Paragraph number="736" type="verses" book="vv">
_Bahupadumavicitrapuṇḍarīkaṃ,_  
_Vokiṇṇaṃ kusumehi nekacittaṃ;_  
_Arajavirajahemajālachannaṃ,_  
_Ākāse tapati yathāpi sūriyo._  

</Paragraph>

</Division>
<Division number="892" book="vv" bookNo="19" page="77">

<Paragraph number="737" type="verses" book="vv">
_Rattambarapītavāsasāhi,_  
_Agarupiyaṅgucandanussadāhi;_  
_Kañcanatanusannibhattacāhi,_  
_Paripūraṃ gaganaṃva tārakāhi._  

</Paragraph>

</Division>
<Division number="893" book="vv" bookNo="19" page="77">

<Paragraph number="738" type="verses" book="vv">
_Naranāriyo bahuketthanekavaṇṇā,_  
_Kusumavibhūsitābharaṇettha sumanā;_  
_Anilapamuñcitā pavanti surabhiṃ,_  
_Tapaniyavitatā suvaṇṇachannā._  

</Paragraph>

</Division>
<Division number="894" book="vv" bookNo="19" page="77">

<Paragraph number="739" type="verses" book="vv">
_Kissa saṃyamassa ayaṃ vipāko,_  
_Kenāsi kammaphalenidhūpapanno;_  
_Yathā ca te adhigatamidaṃ vimānaṃ,_  
_Tadanupadaṃ avacāsi iṅgha puṭṭho”ti._  

</Paragraph>

</Division>
<Division number="895" book="vv" bookNo="19" page="77">

<Paragraph number="740" type="verses" book="vv">
_“Sayamidha pathe samecca māṇavena,_  
_Satthānusāsi anukampamāno;_  
_Tava ratanavarassa dhammaṃ sutvā,_  
_Karissāmīti ca bravittha chatto._  

</Paragraph>

</Division>
<Division number="896" book="vv" bookNo="19" page="78">

<Paragraph number="741" type="verses" book="vv">
_Jinavarapavaraṃ upehi saraṇaṃ,_  
_Dhammañcāpi tatheva bhikkhusaṃghaṃ;_  
_Noti paṭhamaṃ avocahaṃ bhante,_  
_Pacchā te vacanaṃ tathevakāsiṃ._  

</Paragraph>

</Division>
<Division number="897" book="vv" bookNo="19" page="78">

<Paragraph number="742" type="verses" book="vv">
_Mā ca pāṇavadhaṃ vividhaṃ carassu asuciṃ,_  
_Na hi pāṇesu asaññataṃ avaṇṇayiṃsu sappaññā;_  
_Noti paṭhamaṃ avocahaṃ bhante,_  
_Pacchā te vacanaṃ tathevakāsiṃ._  

</Paragraph>

</Division>
<Division number="898" book="vv" bookNo="19" page="78">

<Paragraph number="743" type="verses" book="vv">
_Mā ca parajanassa rakkhitampi,_  
_Ādātabbamamaññitho adinnaṃ;_  
_Noti paṭhamaṃ avocahaṃ bhante,_  
_Pacchā te vacanaṃ tathevakāsiṃ._  

</Paragraph>

</Division>
<Division number="899" book="vv" bookNo="19" page="78">

<Paragraph number="744" type="verses" book="vv">
_Mā ca parajanassa rakkhitāyo,_  
_Parabhariyā agamā anariyametaṃ;_  
_Noti paṭhamaṃ avocahaṃ bhante,_  
_Pacchā te vacanaṃ tathevakāsiṃ._  

</Paragraph>

</Division>
<Division number="900" book="vv" bookNo="19" page="78">

<Paragraph number="745" type="verses" book="vv">
_Mā ca vitathaṃ aññathā abhāṇi,_  
_Na hi musāvādaṃ avaṇṇayiṃsu sappaññā;_  
_Noti paṭhamaṃ avocahaṃ bhante,_  
_Pacchā te vacanaṃ tathevakāsiṃ._  

</Paragraph>

</Division>
<Division number="901" book="vv" bookNo="19" page="78">

<Paragraph number="746" type="verses" book="vv">
_Yena ca purisassa apeti saññā,_  
_Taṃ majjaṃ parivajjayassu sabbaṃ;_  
_Noti paṭhamaṃ avocahaṃ bhante,_  
_Pacchā te vacanaṃ tathevakāsiṃ._  

</Paragraph>

</Division>
<Division number="902" book="vv" bookNo="19" page="78">

<Paragraph number="747" type="verses" book="vv">
_Svāhaṃ idha pañca sikkhā karitvā,_  
_Paṭipajjitvā tathāgatassa dhamme;_  
_Dvepathamagamāsiṃ coramajjhe,_  
_Te maṃ tattha vadhiṃsu bhogahetu._  

</Paragraph>

</Division>
<Division number="903" book="vv" bookNo="19" page="79">

<Paragraph number="748" type="verses" book="vv">
_Ettakamidaṃ anussarāmi kusalaṃ,_  
_Tato paraṃ na me vijjati aññaṃ;_  
_Tena sucaritena kammunāhaṃ,_  
_Uppanno tidivesu kāmakāmī._  

</Paragraph>

</Division>
<Division number="904" book="vv" bookNo="19" page="79">

<Paragraph number="749" type="verses" book="vv">
_Passa khaṇamuhuttasaññamassa,_  
_Anudhammappaṭipattiyā vipākaṃ;_  
_Jalamiva yasasā samekkhamānā,_  
_Bahukā maṃ pihayanti hīnakammā._  

</Paragraph>

</Division>
<Division number="905" book="vv" bookNo="19" page="79">

<Paragraph number="750" type="verses" book="vv">
_Passa katipayāya desanāya,_  
_Sugatiñcamhi gato sukhañca patto;_  
_Ye ca te satataṃ suṇanti dhammaṃ,_  
_Maññe te amataṃ phusanti khemaṃ._  

</Paragraph>

</Division>
<Division number="906" book="vv" bookNo="19" page="79">

<Paragraph number="751" type="verses" book="vv">
_Appampi kataṃ mahāvipākaṃ,_  
_Vipulaṃ hoti tathāgatassa dhamme;_  
_Passa katapuññatāya chatto,_  
_Obhāseti pathaviṃ yathāpi sūriyo._  

</Paragraph>

</Division>
<Division number="907" book="vv" bookNo="19" page="79">

<Paragraph number="752" type="verses" book="vv">
_Kimidaṃ kusalaṃ kimācarema,_  
_Icceke hi samecca mantayanti;_  
_Te mayaṃ punareva laddha mānusattaṃ,_  
_Paṭipannā viharemu sīlavanto._  

</Paragraph>

</Division>
<Division number="908" book="vv" bookNo="19" page="79">

<Paragraph number="753" type="verses" book="vv">
_Bahukāro anukampako ca satthā,_  
_Iti me sati agamā divā divassa;_  
_Svāhaṃ upagatomhi saccanāmaṃ,_  
_Anukampassu punapi suṇemu dhammaṃ._  

</Paragraph>

</Division>
<Division number="909" book="vv" bookNo="19" page="80">

<Paragraph number="754" type="verses" book="vv">
_Ye cidha pajahanti kāmarāgaṃ,_  
_Bhavarāgānusayañca pahāya mohaṃ;_  
_Na ca te punamupenti gabbhaseyyaṃ,_  
_Parinibbānagatā hi sītibhūtā”ti._  


</Paragraph>
</Division>

<Paragraph number="755" type="center" book="vv">
Chattamāṇavakavimānaṃ tatiyaṃ.
</Paragraph>
