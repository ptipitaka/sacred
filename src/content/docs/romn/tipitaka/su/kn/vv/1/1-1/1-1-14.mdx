<Division number="116" book="vv">

<Paragraph number="130" type="verses" book="vv">
_“Abhikkantena vaṇṇena,_  
_yā tvaṃ tiṭṭhasi devate;_  
_Obhāsentī disā sabbā,_  
_osadhī viya tārakā._  

</Paragraph>

</Division>
<Division number="117" book="vv">

<Paragraph number="131" type="verses" book="vv">
_Kena tetādiso vaṇṇo,_  
_kena te idha mijjhati;_  
_Uppajjanti ca te bhogā,_  
_ye keci manaso piyā._  

</Paragraph>

</Division>
<Division number="118" book="vv">

<Paragraph number="132" type="verses" book="vv">
_Pucchāmi taṃ devi mahānubhāve,_  
_Manussabhūtā kimakāsi puññaṃ;_  
_Kenāsi evaṃ jalitānubhāvā,_  
_Vaṇṇo ca te sabbadisā pabhāsatī”ti._  

</Paragraph>

</Division>
<Division number="119" book="vv">

<Paragraph number="133" type="verses" book="vv">
_Sā devatā attamanā,_  
_moggallānena pucchitā;_  
_Pañhaṃ puṭṭhā viyākāsi,_  
_yassa kammassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="120" book="vv">

<Paragraph number="134" type="verses" book="vv">
_“Ahaṃ manussesu manussabhūtā,_  
_Suṇisā ahosiṃ sasurassa gehe._  

</Paragraph>

</Division>
<Division number="121" book="vv">

<Paragraph number="135" type="verses" book="vv">
_Addasaṃ virajaṃ bhikkhuṃ,_  
_vippasannamanāvilaṃ;_  
_Tassa adāsahaṃ bhāgaṃ,_  
_pasannā sehi pāṇibhi;_  
_Kummāsapiṇḍaṃ datvāna,_  
_modāmi nandane vane._  

</Paragraph>

</Division>
<Division number="122" book="vv">

<Paragraph number="136" type="verses" book="vv">
_Tena metādiso vaṇṇo,_  
_tena me idha mijjhati;_  
_Uppajjanti ca me bhogā,_  
_ye keci manaso piyā._  

</Paragraph>

</Division>
<Division number="123" book="vv">

<Paragraph number="137" type="verses" book="vv">
_Akkhāmi te bhikkhu mahānubhāva,_  
_Manussabhūtā yamakāsi puññaṃ;_  
_Tenamhi evaṃ jalitānubhāvā,_  
_Vaṇṇo ca me sabbadisā pabhāsatī”ti._  


</Paragraph>
</Division>

<Paragraph number="138" type="center" book="vv">
Dutiyasuṇisāvimānaṃ cuddasamaṃ.
</Paragraph>
