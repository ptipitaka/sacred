<Division number="75" book="vv">

<Paragraph number="84" type="verses" book="vv">
_“Abhikkantena vaṇṇena,_  
_yā tvaṃ tiṭṭhasi devate;_  
_Obhāsentī disā sabbā,_  
_osadhī viya tārakā._  

</Paragraph>

</Division>
<Division number="76" book="vv">

<Paragraph number="85" type="verses" book="vv">
_Kena tetādiso vaṇṇo,_  
_kena te idha mijjhati;_  
_Uppajjanti ca te bhogā,_  
_ye keci manaso piyā._  

</Paragraph>

</Division>
<Division number="77" book="vv">

<Paragraph number="86" type="verses" book="vv">
_Kena tvaṃ vimalobhāsā,_  
_atirocasi devatā;_  
_Kena te sabbagattehi,_  
_sabbā obhāsate disā._  

</Paragraph>

</Division>
<Division number="78" book="vv">

<Paragraph number="87" type="verses" book="vv">
_Pucchāmi taṃ devi mahānubhāve,_  
_Manussabhūtā kimakāsi puññaṃ;_  
_Kenāsi evaṃ jalitānubhāvā,_  
_Vaṇṇo ca te sabbadisā pabhāsatī”ti._  

</Paragraph>

</Division>
<Division number="79" book="vv">

<Paragraph number="88" type="verses" book="vv">
_Sā devatā attamanā,_  
_moggallānena pucchitā;_  
_Pañhaṃ puṭṭhā viyākāsi,_  
_yassa kammassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="80" book="vv">

<Paragraph number="89" type="verses" book="vv">
_“Ahaṃ manussesu manussabhūtā,_  
_Purimāya jātiyā manussaloke;_  
_Tamandhakāramhi timīsikāyaṃ,_  
_Padīpakālamhi adāsi dīpaṃ._  

</Paragraph>

</Division>
<Division number="81" book="vv">

<Paragraph number="90" type="verses" book="vv">
_Yo andhakāramhi timīsikāyaṃ,_  
_Padīpakālamhi dadāti dīpaṃ;_  
_Upapajjati jotirasaṃ vimānaṃ,_  
_Pahūtamalyaṃ bahupuṇḍarīkaṃ._  

</Paragraph>

</Division>
<Division number="82" book="vv">

<Paragraph number="91" type="verses" book="vv">
_Tena metādiso vaṇṇo,_  
_tena me idha mijjhati;_  
_Uppajjanti ca me bhogā,_  
_ye keci manaso piyā._  

</Paragraph>

</Division>
<Division number="83" book="vv">

<Paragraph number="92" type="verses" book="vv">
_Tenāhaṃ vimalobhāsā,_  
_atirocāmi devatā;_  
_Tena me sabbagattehi,_  
_sabbā obhāsate disā._  

</Paragraph>

</Division>
<Division number="84" book="vv">

<Paragraph number="93" type="verses" book="vv">
_Akkhāmi te bhikkhu mahānubhāva,_  
_Manussabhūtā yamakāsi puññaṃ;_  
_Tenamhi evaṃ jalitānubhāvā,_  
_Vaṇṇo ca me sabbadisā pabhāsatī”ti._  


</Paragraph>
</Division>

<Paragraph number="94" type="center" book="vv">
Dīpavimānaṃ navamaṃ.
</Paragraph>
