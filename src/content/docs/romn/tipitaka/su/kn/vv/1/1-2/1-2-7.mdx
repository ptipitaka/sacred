_ …pe… _

</Paragraph>

<Division number="233" book="vv" bookNo="19" page="29">

<Paragraph number="258" type="verses" book="vv">
_“Uposathāti maṃ aññaṃsu,_  
_sāketāyaṃ upāsikā;_  
_Saddhā sīlena sampannā,_  
_saṃvibhāgaratā sadā._  

</Paragraph>

</Division>
<Division number="234" book="vv" bookNo="19" page="29">

<Paragraph number="259" type="verses" book="vv">
_Acchādanañca bhattañca,_  
_senāsanaṃ padīpiyaṃ;_  
_Adāsiṃ ujubhūtesu,_  
_vippasannena cetasā._  

</Paragraph>

</Division>
<Division number="235" book="vv" bookNo="19" page="29">

<Paragraph number="260" type="verses" book="vv">
_Cātuddasiṃ pañcadasiṃ,_  
_yā ca pakkhassa aṭṭhamī;_  
_Pāṭihāriyapakkhañca,_  
_aṭṭhaṅgasusamāgataṃ._  

</Paragraph>

</Division>
<Division number="236" book="vv" bookNo="19" page="29">

<Paragraph number="261" type="verses" book="vv">
_Uposathaṃ upavasissaṃ,_  
_sadā sīlesu saṃvutā;_  
_Saññamā saṃvibhāgā ca,_  
_vimānaṃ āvasāmahaṃ._  

</Paragraph>

</Division>
<Division number="237" book="vv" bookNo="19" page="29">

<Paragraph number="262" type="verses" book="vv">
_Pāṇātipātā viratā,_  
_musāvādā ca saññatā;_  
_Theyyā ca aticārā ca,_  
_majjapānā ca ārakā._  

</Paragraph>

</Division>
<Division number="238" book="vv" bookNo="19" page="29">

<Paragraph number="263" type="verses" book="vv">
_Pañcasikkhāpade ratā,_  
_ariyasaccāna kovidā;_  
_Upāsikā cakkhumato,_  
_gotamassa yasassino._  


(239–240.)
</Paragraph>

<Paragraph number="264" type="verses" book="vv">
_Tena metādiso vaṇṇo,_  
_ …pe… _  
_Vaṇṇo ca me sabbadisā pabhāsatī”ti._  

</Paragraph>

</Division>
<Division number="241" book="vv" bookNo="19" page="29">

<Paragraph number="265" type="verses" book="vv">
_“Abhikkhaṇaṃ nandanaṃ sutvā,_  
_Chando me udapajjatha;_  
_Tattha cittaṃ paṇidhāya,_  
_Upapannāmhi nandanaṃ._  

</Paragraph>

</Division>
<Division number="242" book="vv" bookNo="19" page="29">

<Paragraph number="266" type="verses" book="vv">
_Nākāsiṃ satthu vacanaṃ,_  
_buddhassādiccabandhuno;_  
_Hīne cittaṃ paṇidhāya,_  
_sāmhi pacchānutāpinī”ti._  

</Paragraph>

</Division>
<Division number="243" book="vv" bookNo="19" page="29">

<Paragraph number="267" type="verses" book="vv">
_“Kīva ciraṃ vimānamhi,_  
_idha vacchasuposathe;_  
_Devate pucchitācikkha,_  
_yadi jānāsi āyuno”ti._  

</Paragraph>

</Division>
<Division number="244" book="vv" bookNo="19" page="30">

<Paragraph number="268" type="verses" book="vv">
_“Saṭṭhi vassasahassāni,_  
_tisso ca vassakoṭiyo;_  
_Idha ṭhatvā mahāmuni,_  
_ito cutā gamissāmi;_  
_Manussānaṃ sahabyatan”ti._  

</Paragraph>

</Division>
<Division number="245" book="vv" bookNo="19" page="30">

<Paragraph number="269" type="verses" book="vv">
_“Mā tvaṃ uposathe bhāyi,_  
_sambuddhenāsi byākatā;_  
_Sotāpannā visesayi,_  
_pahīnā tava duggatī”ti._  


</Paragraph>
</Division>

Uposathāvimānaṃ sattamaṃ.
