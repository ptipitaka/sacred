<Paragraph number="49" type="verses" book="vv">
_“Suvaṇṇacchadanaṃ nāvaṃ,_  
_Nāri āruyha tiṭṭhasi;_  
_Ogāhasi pokkharaṇiṃ,_  
_Padmaṃ chindasi pāṇinā._  

</Paragraph>

</Division>
<Division number="44" book="vv" bookNo="19" page="6">

<Paragraph number="50" type="verses" book="vv">
_Kena tetādiso vaṇṇo,_  
_kena te idha mijjhati;_  
_Uppajjanti ca te bhogā,_  
_ye keci manaso piyā._  

</Paragraph>

</Division>
<Division number="45" book="vv" bookNo="19" page="6">

<Paragraph number="51" type="verses" book="vv">
_Pucchāmi taṃ devi mahānubhāve,_  
_Manussabhūtā kimakāsi puññaṃ;_  
_Kenāsi evaṃ jalitānubhāvā,_  
_Vaṇṇo ca te sabbadisā pabhāsatī”ti._  

</Paragraph>

</Division>
<Division number="46" book="vv" bookNo="19" page="6">

<Paragraph number="52" type="verses" book="vv">
_Sā devatā attamanā,_  
_moggallānena pucchitā;_  
_Pañhaṃ puṭṭhā viyākāsi,_  
_yassa kammassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="47" book="vv" bookNo="19" page="7">

<Paragraph number="53" type="verses" book="vv">
_“Ahaṃ manussesu manussabhūtā,_  
_Purimāya jātiyā manussaloke;_  
_Disvāna bhikkhū tasite kilante,_  
_Uṭṭhāya pātuṃ udakaṃ adāsiṃ._  

</Paragraph>

</Division>
<Division number="48" book="vv" bookNo="19" page="7">

<Paragraph number="54" type="verses" book="vv">
_Yo ve kilantāna pipāsitānaṃ,_  
_Uṭṭhāya pātuṃ udakaṃ dadāti;_  
_Sītodakā tassa bhavanti najjo,_  
_Pahūtamalyā bahupuṇḍarīkā._  

</Paragraph>

</Division>
<Division number="49" book="vv" bookNo="19" page="7">

<Paragraph number="55" type="verses" book="vv">
_Taṃ āpagā anupariyanti sabbadā,_  
_Sītodakā vālukasanthatā nadī;_  
_Ambā ca sālā tilakā ca jambuyo,_  
_Uddālakā pāṭaliyo ca phullā._  

</Paragraph>

</Division>
<Division number="50" book="vv" bookNo="19" page="7">

<Paragraph number="56" type="verses" book="vv">
_Taṃ bhūmibhāgehi upetarūpaṃ,_  
_Vimānaseṭṭhaṃ bhusa sobhamānaṃ;_  
_Tassīdha kammassa ayaṃ vipāko,_  
_Etādisaṃ puññakatā labhanti._  

</Paragraph>

</Division>
<Division number="51" book="vv" bookNo="19" page="7">

<Paragraph number="57" type="verses" book="vv">
_Tena metādiso vaṇṇo,_  
_tena me idha mijjhati;_  
_Uppajjanti ca me bhogā,_  
_ye keci manaso piyā._  

</Paragraph>

</Division>
<Division number="52" book="vv" bookNo="19" page="7">

<Paragraph number="58" type="verses" book="vv">
_Akkhāmi te bhikkhu mahānubhāva,_  
_Manussabhūtā yamakāsi puññaṃ;_  
_Tenamhi evaṃ jalitānubhāvā,_  
_Vaṇṇo ca me sabbadisā pabhāsatī”ti._  


</Paragraph>
</Division>

<Paragraph number="59" type="center" book="vv">
Paṭhamanāvāvimānaṃ chaṭṭhaṃ.
</Paragraph>
