<Division number="287" book="vv">

<Paragraph number="309" type="verses" book="vv">

_tava pūjāya devate;_  
_Sovaṇṇāni vimānāni,_  
_tavimāni sudassane._  

</Paragraph>

</Division>
<Division number="288" book="vv">

<Paragraph number="310" type="verses" book="vv">
_Tuvaṃsi issarā tesaṃ,_  
_sabbakāmasamiddhinī;_  
_Abhijātā mahantāsi,_  
_devakāye pamodasi;_  
_Devate pucchitācikkha,_  
_kissa kammassidaṃ phalan”ti._  

</Paragraph>

</Division>
<Division number="289" book="vv">

<Paragraph number="311" type="verses" book="vv">
_“Ahaṃ manussesu manussabhūtā,_  
_Purimāya jātiyā manussaloke;_  
_Dussīlakule suṇisā ahosiṃ,_  
_Assaddhesu kadariyesu ahaṃ._  

</Paragraph>

</Division>
<Division number="290" book="vv">

<Paragraph number="312" type="verses" book="vv">
_Saddhā sīlena sampannā,_  
_saṃvibhāgaratā sadā;_  
_Piṇḍāya caramānassa,_  
_apūvaṃ te adāsahaṃ._  

</Paragraph>

</Division>
<Division number="291" book="vv">

<Paragraph number="313" type="verses" book="vv">

_samaṇo āgato idha;_  
_Tassa adāsahaṃ pūvaṃ,_  
_pasannā sehi pāṇibhi._  

</Paragraph>

</Division>
<Division number="292" book="vv">

<Paragraph number="314" type="verses" book="vv">
_Itissā sassu paribhāsi,_  
_avinītāsi tvaṃ vadhu;_  
_Na maṃ sampucchituṃ icchi,_  
_samaṇassa dadāmahaṃ._  

</Paragraph>

</Division>
<Division number="293" book="vv">

<Paragraph number="315" type="verses" book="vv">

_pahāsi musalena maṃ;_  
_Kūṭaṅgacchi avadhi maṃ,_  
_nāsakkhiṃ jīvituṃ ciraṃ._  

</Paragraph>

</Division>

_upapannā sahabyataṃ._  

</Paragraph>

</Division>
