<Paragraph number="824" type="verses" book="vv">
_“Daḷhadhammā nisārassa,_  
_dhanuṃ olubbha tiṭṭhasi;_  
_Khattiyo nusi rājañño,_  
_adu luddo vanecaro”ti._  

</Paragraph>

</Division>
<Division number="982" book="vv">

<Paragraph number="825" type="verses" book="vv">
_“Assakādhipatissāhaṃ,_  
_bhante putto vanecaro;_  
_Nāmaṃ me bhikkhu te brūmi,_  
_sujāto iti maṃ vidū._  

</Paragraph>

</Division>
<Division number="983" book="vv">

<Paragraph number="826" type="verses" book="vv">
_Mige gavesamānohaṃ,_  
_Ogāhanto brahāvanaṃ;_  
_Migaṃ tañceva nāddakkhiṃ,_  
_Tañca disvā ṭhito ahan”ti._  

</Paragraph>

</Division>
<Division number="984" book="vv">

<Paragraph number="827" type="verses" book="vv">
_“Svāgataṃ te mahāpuñña,_  
_atho te adurāgataṃ;_  
_Etto udakamādāya,_  
_pāde pakkhālayassu te._  

</Paragraph>

</Division>
<Division number="985" book="vv">

<Paragraph number="828" type="verses" book="vv">
_Idampi pānīyaṃ sītaṃ,_  
_ābhataṃ girigabbharā;_  
_Rājaputta tato pitvā,_  
_santhatasmiṃ upāvisā”ti._  

</Paragraph>

</Division>
<Division number="986" book="vv">

<Paragraph number="829" type="verses" book="vv">
_“Kalyāṇī vata te vācā,_  
_Savanīyā mahāmuni;_  
_Nelā atthavatī vaggu,_  
_Mantvā atthañca bhāsasi._  

</Paragraph>

</Division>
<Division number="987" book="vv">

<Paragraph number="830" type="verses" book="vv">
_Kā te rati vane viharato,_  
_Isinisabha vadehi puṭṭho;_  
_Tava vacanapathaṃ nisāmayitvā,_  
_Atthadhammapadaṃ samācaremase”ti._  

</Paragraph>

</Division>
<Division number="988" book="vv">

<Paragraph number="831" type="verses" book="vv">
_“Ahiṃsā sabbapāṇīnaṃ,_  
_kumāramhāka ruccati;_  
_Theyyā ca aticārā ca,_  
_majjapānā ca ārati._  

</Paragraph>

</Division>
<Division number="989" book="vv">

<Paragraph number="832" type="verses" book="vv">
_Ārati samacariyā ca,_  
_bāhusaccaṃ kataññutā;_  
_Diṭṭheva dhamme pāsaṃsā,_  
_dhammā ete pasaṃsiyāti._  

</Paragraph>

</Division>
<Division number="990" book="vv">

<Paragraph number="833" type="verses" book="vv">
_Santike maraṇaṃ tuyhaṃ,_  
_oraṃ māsehi pañcahi;_  
_Rājaputta vijānāhi,_  
_attānaṃ parimocayā”ti._  

</Paragraph>

</Division>
<Division number="991" book="vv">

<Paragraph number="834" type="verses" book="vv">
_“Katamaṃ svāhaṃ janapadaṃ gantvā,_  
_Kiṃ kammaṃ kiñca porisaṃ;_  
_Kāya vā pana vijjāya,_  
_Bhaveyyaṃ ajarāmaro”ti._  

</Paragraph>

</Division>
<Division number="992" book="vv">

<Paragraph number="835" type="verses" book="vv">
_“Na vijjate so padeso,_  
_kammaṃ vijjā ca porisaṃ;_  
_Yattha gantvā bhave macco,_  
_rājaputtājarāmaro._  

</Paragraph>

</Division>
<Division number="993" book="vv">

<Paragraph number="836" type="verses" book="vv">
_Mahaddhanā mahābhogā,_  
_raṭṭhavantopi khattiyā;_  
_Pahūtadhanadhaññāse,_  
_tepi no ajarāmarā._  

</Paragraph>

</Division>
<Division number="994" book="vv">

<Paragraph number="837" type="verses" book="vv">
_Yadi te sutā andhakaveṇḍuputtā,_  
_Sūrā vīrā vikkantappahārino;_  
_Tepi āyukkhayaṃ pattā,_  
_Viddhastā sassatīsamā._  

</Paragraph>

</Division>
<Division number="995" book="vv">

<Paragraph number="838" type="verses" book="vv">
_Khattiyā brāhmaṇā vessā,_  
_suddā caṇḍālapukkusā;_  
_Ete caññe ca jātiyā,_  
_tepi no ajarāmarā._  

</Paragraph>

</Division>
<Division number="996" book="vv">

<Paragraph number="839" type="verses" book="vv">
_Ye mantaṃ parivattenti,_  
_chaḷaṅgaṃ brahmacintitaṃ;_  
_Ete caññe ca vijjāya,_  
_tepi no ajarāmarā._  

</Paragraph>

</Division>
<Division number="997" book="vv">

<Paragraph number="840" type="verses" book="vv">
_Isayo cāpi ye santā,_  
_saññatattā tapassino;_  
_Sarīraṃ tepi kālena,_  
_vijahanti tapassino._  

</Paragraph>

</Division>
<Division number="998" book="vv">

<Paragraph number="841" type="verses" book="vv">
_Bhāvitattāpi arahanto,_  
_katakiccā anāsavā;_  
_Nikkhipanti imaṃ dehaṃ,_  
_puññapāpaparikkhayā”ti._  

</Paragraph>

</Division>
<Division number="999" book="vv">

<Paragraph number="842" type="verses" book="vv">
_“Subhāsitā atthavatī,_  
_gāthāyo te mahāmuni;_  
_Nijjhattomhi subhaṭṭhena,_  
_tvañca me saraṇaṃ bhavā”ti._  

</Paragraph>

</Division>
<Division number="1000" book="vv">

<Paragraph number="843" type="verses" book="vv">
_“Mā maṃ tvaṃ saraṇaṃ gaccha,_  
_tameva saraṇaṃ vaja;_  
_Sakyaputtaṃ mahāvīraṃ,_  
_yamahaṃ saraṇaṃ gato”ti._  

</Paragraph>

</Division>
<Division number="1001" book="vv">

<Paragraph number="844" type="verses" book="vv">
_“Katarasmiṃ so janapade,_  
_satthā tumhāka mārisa;_  
_Ahampi daṭṭhuṃ gacchissaṃ,_  
_jinaṃ appaṭipuggalan”ti._  

</Paragraph>

</Division>
<Division number="1002" book="vv">

<Paragraph number="845" type="verses" book="vv">
_“Puratthimasmiṃ janapade,_  
_okkākakulasambhavo;_  
_Tatthāsi purisājañño,_  
_so ca kho parinibbuto”ti._  

</Paragraph>

</Division>
<Division number="1003" book="vv">

<Paragraph number="846" type="verses" book="vv">
_“Sace hi buddho tiṭṭheyya,_  
_Satthā tumhāka mārisa;_  
_Yojanāni sahassāni,_  
_Gaccheyyaṃ payirupāsituṃ._  

</Paragraph>

</Division>
<Division number="1004" book="vv">

<Paragraph number="847" type="verses" book="vv">
_Yato ca kho parinibbuto,_  
_satthā tumhāka mārisa;_  
_Nibbutampi mahāvīraṃ,_  
_gacchāmi saraṇaṃ ahaṃ._  

</Paragraph>

</Division>
<Division number="1005" book="vv">

<Paragraph number="848" type="verses" book="vv">
_Upemi saraṇaṃ buddhaṃ,_  
_dhammañcāpi anuttaraṃ;_  
_Saṃghañca naradevassa,_  
_gacchāmi saraṇaṃ ahaṃ._  

</Paragraph>

</Division>
<Division number="1006" book="vv">

<Paragraph number="849" type="verses" book="vv">
_Pāṇātipātā viramāmi khippaṃ,_  
_Loke adinnaṃ parivajjayāmi;_  
_Amajjapo no ca musā bhaṇāmi,_  
_Sakena dārena ca homi tuṭṭho”ti._  

</Paragraph>

</Division>
<Division number="1007" book="vv">

<Paragraph number="850" type="verses" book="vv">
_“Sahassaraṃsīva yathā mahappabho,_  
_Disaṃ yathā bhāti nabhe anukkamaṃ;_  
_Tathāpakāro tavāyaṃ mahāratho,_  
_Samantato yojanasattamāyato._  

</Paragraph>

</Division>
<Division number="1008" book="vv">

<Paragraph number="851" type="verses" book="vv">
_Suvaṇṇapaṭṭehi samantamotthaṭo,_  
_Urassa muttāhi maṇīhi cittito;_  
_Lekhā suvaṇṇassa ca rūpiyassa ca,_  
_Sobhenti veḷuriyamayā sunimmitā._  

</Paragraph>

</Division>
<Division number="1009" book="vv">

<Paragraph number="852" type="verses" book="vv">
_Sīsañcidaṃ veḷuriyassa nimmitaṃ,_  
_Yugañcidaṃ lohitakāya cittitaṃ;_  
_Yuttā suvaṇṇassa ca rūpiyassa ca,_  
_Sobhanti assā ca ime manojavā._  

</Paragraph>

</Division>
<Division number="1010" book="vv">

<Paragraph number="853" type="verses" book="vv">
_So tiṭṭhasi hemarathe adhiṭṭhito,_  
_Devānamindova sahassavāhano;_  
_Pucchāmi tāhaṃ yasavanta kovidaṃ,_  
_Kathaṃ tayā laddho ayaṃ uḷāro”ti._  

</Paragraph>

</Division>
<Division number="1011" book="vv">

<Paragraph number="854" type="verses" book="vv">
_“Sujāto nāmahaṃ bhante,_  
_rājaputto pure ahuṃ;_  
_Tvañca maṃ anukampāya,_  
_saññamasmiṃ nivesayi._  

</Paragraph>

</Division>
<Division number="1012" book="vv">

<Paragraph number="855" type="verses" book="vv">
_Khīṇāyukañca maṃ ñatvā,_  
_sarīraṃ pādāsi satthuno;_  
_Imaṃ sujāta pūjehi,_  
_taṃ te atthāya hehiti._  

</Paragraph>

</Division>
<Division number="1013" book="vv">

<Paragraph number="856" type="verses" book="vv">
_Tāhaṃ gandhehi mālehi,_  
_pūjayitvā samuyyuto;_  
_Pahāya mānusaṃ dehaṃ,_  
_upapannomhi nandanaṃ._  

</Paragraph>

</Division>
<Division number="1014" book="vv">

<Paragraph number="857" type="verses" book="vv">
_Nandane ca vane ramme,_  
_nānādijagaṇāyute;_  
_Ramāmi naccagītehi,_  
_accharāhi purakkhato”ti._  


</Paragraph>
</Division>

<Paragraph number="858" type="center" book="vv">
Cūḷarathavimānaṃ terasamaṃ.
</Paragraph>
