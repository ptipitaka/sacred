<Paragraph number="60" type="verses" book="vv">
_“Suvaṇṇacchadanaṃ nāvaṃ,_  
_Nāri āruyha tiṭṭhasi;_  
_Ogāhasi pokkharaṇiṃ,_  
_Padmaṃ chindasi pāṇinā._  

</Paragraph>

</Division>
<Division number="54" book="vv" bookNo="19" page="8">

<Paragraph number="61" type="verses" book="vv">
_Kena tetādiso vaṇṇo,_  
_kena te idha mijjhati;_  
_Uppajjanti ca te bhogā,_  
_ye keci manaso piyā._  

</Paragraph>

</Division>
<Division number="55" book="vv" bookNo="19" page="8">

<Paragraph number="62" type="verses" book="vv">
_Pucchāmi taṃ devi mahānubhāve,_  
_Manussabhūtā kimakāsi puññaṃ;_  
_Kenāsi evaṃ jalitānubhāvā,_  
_Vaṇṇo ca te sabbadisā pabhāsatī”ti._  

</Paragraph>

</Division>
<Division number="56" book="vv" bookNo="19" page="8">

<Paragraph number="63" type="verses" book="vv">
_Sā devatā attamanā,_  
_moggallānena pucchitā;_  
_Pañhaṃ puṭṭhā viyākāsi,_  
_yassa kammassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="57" book="vv" bookNo="19" page="8">

<Paragraph number="64" type="verses" book="vv">
_“Ahaṃ manussesu manussabhūtā,_  
_Purimāya jātiyā manussaloke;_  
_Disvāna bhikkhuṃ tasitaṃ kilantaṃ,_  
_Uṭṭhāya pātuṃ udakaṃ adāsiṃ._  

</Paragraph>

</Division>
<Division number="58" book="vv" bookNo="19" page="8">

<Paragraph number="65" type="verses" book="vv">
_Yo ve kilantassa pipāsitassa,_  
_Uṭṭhāya pātuṃ udakaṃ dadāti;_  
_Sītodakā tassa bhavanti najjo,_  
_Pahūtamalyā bahupuṇḍarīkā._  

</Paragraph>

</Division>
<Division number="59" book="vv" bookNo="19" page="8">

<Paragraph number="66" type="verses" book="vv">
_Taṃ āpagā anupariyanti sabbadā,_  
_Sītodakā vālukasanthatā nadī;_  
_Ambā ca sālā tilakā ca jambuyo,_  
_Uddālakā pāṭaliyo ca phullā._  

</Paragraph>

</Division>
<Division number="60" book="vv" bookNo="19" page="8">

<Paragraph number="67" type="verses" book="vv">
_Taṃ bhūmibhāgehi upetarūpaṃ,_  
_Vimānaseṭṭhaṃ bhusa sobhamānaṃ;_  
_Tassīdha kammassa ayaṃ vipāko,_  
_Etādisaṃ puññakatā labhanti._  

</Paragraph>

</Division>
<Division number="61" book="vv" bookNo="19" page="8">

<Paragraph number="68" type="verses" book="vv">
_Tena metādiso vaṇṇo,_  
_tena me idha mijjhati;_  
_Uppajjanti ca me bhogā,_  
_ye keci manaso piyā._  

</Paragraph>

</Division>
<Division number="62" book="vv" bookNo="19" page="9">

<Paragraph number="69" type="verses" book="vv">
_Akkhāmi te bhikkhu mahānubhāva,_  
_Manussabhūtā yamakāsi puññaṃ;_  
_Tenamhi evaṃ jalitānubhāvā,_  
_Vaṇṇo ca me sabbadisā pabhāsatī”ti._  


</Paragraph>
</Division>

<Paragraph number="70" type="center" book="vv">
Dutiyanāvāvimānaṃ sattamaṃ.
</Paragraph>
