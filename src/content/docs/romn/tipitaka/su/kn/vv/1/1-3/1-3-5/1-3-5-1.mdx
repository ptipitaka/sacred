<Division number="327" book="vv">

<Paragraph number="353" type="verses" book="vv">
_“Sattatantiṃ sumadhuraṃ,_  
_Rāmaṇeyyaṃ avācayiṃ;_  
_So maṃ raṅgamhi avheti,_  
_‘Saraṇaṃ me hohi kosiyā’”ti._  

</Paragraph>

</Division>
<Division number="328" book="vv">

<Paragraph number="354" type="verses" book="vv">
_“Ahaṃ te saraṇaṃ homi,_  
_Ahamācariyapūjako;_  
_Na taṃ jayissati sisso,_  
_Sissamācariya jessasī”ti._  

</Paragraph>

</Division>
<Division number="329" book="vv">

<Paragraph number="355" type="verses" book="vv">
_“Abhikkantena vaṇṇena,_  
_yā tvaṃ tiṭṭhasi devate;_  
_Obhāsentī disā sabbā,_  
_osadhī viya tārakā._  

</Paragraph>

</Division>
<Division number="330" book="vv">

<Paragraph number="356" type="verses" book="vv">
_Kena tetādiso vaṇṇo,_  
_kena te idha mijjhati;_  
_Uppajjanti ca te bhogā,_  
_ye keci manaso piyā._  

</Paragraph>

</Division>
<Division number="331" book="vv">

<Paragraph number="357" type="verses" book="vv">
_Pucchāmi taṃ devi mahānubhāve,_  
_Manussabhūtā kimakāsi puññaṃ;_  
_Kenāsi evaṃ jalitānubhāvā,_  
_Vaṇṇo ca te sabbadisā pabhāsatī”ti._  

</Paragraph>

</Division>
<Division number="332" book="vv">

<Paragraph number="358" type="verses" book="vv">
_Sā devatā attamanā,_  
_moggallānena pucchitā;_  
_Pañhaṃ puṭṭhā viyākāsi,_  
_yassa kammassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="333" book="vv">

<Paragraph number="359" type="verses" book="vv">
_“Vatthuttamadāyikā nārī,_  
_Pavarā hoti naresu nārīsu;_  
_Evaṃ piyarūpadāyikā manāpaṃ,_  
_Dibbaṃ sā labhate upecca ṭhānaṃ._  

</Paragraph>

</Division>
<Division number="334" book="vv">

<Paragraph number="360" type="verses" book="vv">
_Tassā me passa vimānaṃ,_  
_Accharā kāmavaṇṇinīhamasmi;_  
_Accharāsahassassāhaṃ,_  
_Pavarā passa puññānaṃ vipākaṃ._  

</Paragraph>

</Division>
<Division number="335" book="vv">

<Paragraph number="361" type="verses" book="vv">
_Tena metādiso vaṇṇo,_  
_tena me idha mijjhati;_  
_Uppajjanti ca me bhogā,_  
_ye keci manaso piyā._  

</Paragraph>

</Division>
<Division number="336" book="vv">

<Paragraph number="362" type="verses" book="vv">
_Akkhāmi te bhikkhu mahānubhāva,_  
_Manussabhūtā yamakāsi puññaṃ;_  
_Tenamhi evaṃ jalitānubhāvā,_  
_Vaṇṇo ca me sabbadisā pabhāsatī”ti._  

</Paragraph>

<Paragraph number="363" book="vv">
(Anantaraṃ caturavimānaṃ yathā vatthadāyikāvimānaṃ tathā vitthāretabbaṃ.)
</Paragraph>
</Division>
