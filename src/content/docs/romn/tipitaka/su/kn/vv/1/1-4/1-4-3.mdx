<Division number="705" book="vv">

<Paragraph number="553" type="verses" book="vv">
_“Alaṅkatā maṇikañcanācitaṃ,_  
_Sovaṇṇajālacitaṃ mahantaṃ;_  
_Abhiruyha gajavaraṃ sukappitaṃ,_  
_Idhāgamā vehāyasaṃ antalikkhe._  

</Paragraph>

</Division>
<Division number="706" book="vv">

<Paragraph number="554" type="verses" book="vv">
_Nāgassa dantesu duvesu nimmitā,_  
_Acchodakā paduminiyo suphullā;_  
_Padumesu ca tūriyagaṇā pabhijjare,_  
_Imā ca naccanti manoharāyo._  

</Paragraph>

</Division>
<Division number="707" book="vv">

<Paragraph number="555" type="verses" book="vv">
_Deviddhipattāsi mahānubhāve,_  
_Manussabhūtā kimakāsi puññaṃ;_  
_Kenāsi evaṃ jalitānubhāvā,_  
_Vaṇṇo ca te sabbadisā pabhāsatī”ti._  

</Paragraph>

</Division>
<Division number="708" book="vv">

<Paragraph number="556" type="verses" book="vv">
_“Bārāṇasiyaṃ upasaṅkamitvā,_  
_Buddhassahaṃ vatthayugaṃ adāsiṃ;_  
_Pādāni vanditvā chamā nisīdiṃ,_  
_Vittā cahaṃ añjalikaṃ akāsiṃ._  

</Paragraph>

</Division>
<Division number="709" book="vv">

<Paragraph number="557" type="verses" book="vv">
_Buddho ca me kañcanasannibhattaco,_  
_Adesayi samudayadukkhaniccataṃ;_  
_Asaṅkhataṃ dukkhanirodhasassataṃ,_  
_Maggaṃ adesayi yato vijānisaṃ._  

</Paragraph>

</Division>
<Division number="710" book="vv">

<Paragraph number="558" type="verses" book="vv">
_Appāyukī kālakatā tato cutā,_  
_Upapannā tidasagaṇaṃ yasassinī;_  
_Sakkassahaṃ aññatarā pajāpati,_  
_Yasuttarā nāma disāsu vissutā”ti._  


</Paragraph>
</Division>

<Paragraph number="559" type="center" book="vv">
Nāgavimānaṃ tatiyaṃ.
</Paragraph>
