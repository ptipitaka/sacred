_“Pāricchattake koviḷāre,_  
_ramaṇīye manorame;_  
_Dibbamālaṃ ganthamānā,_  
_gāyantī sampamodasi._  

</Paragraph>

<Division number="681" book="vv" bookNo="19" page="55">

<Paragraph number="523" type="verses" book="vv">
_Tassā te naccamānāya,_  
_aṅgamaṅgehi sabbaso;_  
_Dibbā saddā niccharanti,_  
_savanīyā manoramā._  

</Paragraph>

</Division>
<Division number="682" book="vv" bookNo="19" page="55">

<Paragraph number="524" type="verses" book="vv">
_Tassā te naccamānāya,_  
_aṅgamaṅgehi sabbaso;_  
_Dibbā gandhā pavāyanti,_  
_sucigandhā manoramā._  

</Paragraph>

</Division>
<Division number="683" book="vv" bookNo="19" page="55">

<Paragraph number="525" type="verses" book="vv">
_Vivattamānā kāyena,_  
_yā veṇīsu piḷandhanā;_  
_Tesaṃ suyyati nigghoso,_  
_tūriye pañcaṅgike yathā._  

</Paragraph>

</Division>
<Division number="684" book="vv" bookNo="19" page="55">

<Paragraph number="526" type="verses" book="vv">
_Vaṭaṃsakā vātadhutā,_  
_vātena sampakampitā;_  
_Tesaṃ suyyati nigghoso,_  
_tūriye pañcaṅgike yathā._  

</Paragraph>

</Division>
<Division number="685" book="vv" bookNo="19" page="55">

<Paragraph number="527" type="verses" book="vv">
_Yāpi te sirasmiṃ mālā,_  
_sucigandhā manoramā;_  
_Vāti gandho disā sabbā,_  
_rukkho mañjūsako yathā._  

</Paragraph>

</Division>
<Division number="686" book="vv" bookNo="19" page="55">

<Paragraph number="528" type="verses" book="vv">
_Ghāyase taṃ sucigandhaṃ,_  
_rūpaṃ passasi amānusaṃ;_  
_Devate pucchitācikkha,_  
_kissa kammassidaṃ phalan”ti._  

</Paragraph>

</Division>
<Division number="687" book="vv" bookNo="19" page="55">

<Paragraph number="529" type="verses" book="vv">
_“Pabhassaraṃ accimantaṃ,_  
_vaṇṇagandhena saṃyutaṃ;_  
_Asokapupphamālāhaṃ,_  
_buddhassa upanāmayiṃ._  

</Paragraph>

</Division>
<Division number="688" book="vv" bookNo="19" page="55">

<Paragraph number="530" type="verses" book="vv">
_Tāhaṃ kammaṃ karitvāna,_  
_kusalaṃ buddhavaṇṇitaṃ;_  
_Apetasokā sukhitā,_  
_sampamodāmanāmayā”ti._  


</Paragraph>
</Division>

<Paragraph number="531" type="center" book="vv">
Pāricchattakavimānaṃ dasamaṃ.



</Paragraph>

<Paragraph number="532" type="center" book="vv">
Tassuddānaṃ

_latā ca guttilena ca;_  
_Daddallapesamallikā,_  
_visālakkhi pāricchattako;_  
_Vaggo tena pavuccatīti._  


</Paragraph>
