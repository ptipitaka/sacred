<Division number="646" book="vv">

<Paragraph number="485" type="verses" book="vv">
_“Phalikarajatahemajālachannaṃ,_  
_Vividhacitratalamaddasaṃ surammaṃ;_  
_Byamhaṃ sunimmitaṃ toraṇūpapannaṃ,_  
_Rucakupakiṇṇamidaṃ subhaṃ vimānaṃ._  

</Paragraph>

</Division>
<Division number="647" book="vv">

<Paragraph number="486" type="verses" book="vv">
_Bhāti ca dasa disā nabheva suriyo,_  
_Sarade tamonudo sahassaraṃsī;_  
_Tathā tapati midaṃ tava vimānaṃ,_  
_Jalamiva dhūmasikho nise nabhagge._  

</Paragraph>

</Division>
<Division number="648" book="vv">

<Paragraph number="487" type="verses" book="vv">
_Musatīva nayanaṃ sateratāva,_  
_Ākāse ṭhapitamidaṃ manuññaṃ;_  
_Vīṇāmurajasammatāḷaghuṭṭhaṃ,_  
_Iddhaṃ indapuraṃ yathā tavedaṃ._  

</Paragraph>

</Division>
<Division number="649" book="vv">

<Paragraph number="488" type="verses" book="vv">
_Padumakumuduppalakuvalayaṃ,_  
_Yodhika bandhukanojakā ca santi;_  
_Sālakusumitapupphitā asokā,_  
_Vividhadumaggasugandhasevitamidaṃ._  

</Paragraph>

</Division>
<Division number="650" book="vv">

<Paragraph number="489" type="verses" book="vv">
_Saḷalalabujabhujaka saṃyuttā,_  
_Kusakasuphullitalatāvalambinīhi;_  
_Maṇijālasadisā yasassinī,_  
_Rammā pokkharaṇī upaṭṭhitā te._  

</Paragraph>

</Division>
<Division number="651" book="vv">

<Paragraph number="490" type="verses" book="vv">
_Udakaruhā ca yetthi pupphajātā,_  
_Thalajā ye ca santi rukkhajātā;_  
_Mānusakāmānussakā ca dibbā,_  
_Sabbe tuyhaṃ nivesanamhi jātā._  

</Paragraph>

</Division>
<Division number="652" book="vv">

<Paragraph number="491" type="verses" book="vv">
_Kissa saṃyamadamassayaṃ vipāko,_  
_Kenāsi kammaphalenidhūpapannā;_  
_Yathā ca te adhigatamidaṃ vimānaṃ,_  
_Tadanupadaṃ avacāsiḷārapamhe”ti._  

</Paragraph>

</Division>
<Division number="653" book="vv">

<Paragraph number="492" type="verses" book="vv">
_“Yathā ca me adhigatamidaṃ vimānaṃ,_  
_Koñcamayūracakorasaṅghacaritaṃ;_  
_Dibyapilavahaṃsarājaciṇṇaṃ,_  
_Dijakāraṇḍavakokilābhinaditaṃ._  

</Paragraph>

</Division>
<Division number="654" book="vv">

<Paragraph number="493" type="verses" book="vv">
_Nānāsantānakapuppharukkhavividhā,_  
_Pāṭalijambuasokarukkhavantaṃ;_  
_Yathā ca me adhigatamidaṃ vimānaṃ,_  
_Taṃ te pavedayāmi suṇohi bhante._  

</Paragraph>

</Division>
<Division number="655" book="vv">

<Paragraph number="494" type="verses" book="vv">
_Magadhavarapuratthimena,_  
_Nāḷakagāmo nāma atthi bhante;_  
_Tattha ahosiṃ pure suṇisā,_  
_Pesavatīti tattha jāniṃsu mamaṃ._  

</Paragraph>

</Division>
<Division number="656" book="vv">

<Paragraph number="495" type="verses" book="vv">
_Sāhamapacitatthadhammakusalaṃ,_  
_Devamanussapūjitaṃ mahantaṃ;_  
_Upatissaṃ nibbutamappameyyaṃ,_  
_Muditamanā kusumehi abbhukiriṃ._  

</Paragraph>

</Division>
<Division number="657" book="vv">

<Paragraph number="496" type="verses" book="vv">
_Paramagatigatañca pūjayitvā,_  
_Antimadehadharaṃ isiṃ uḷāraṃ;_  
_Pahāya mānusakaṃ samussayaṃ,_  
_Tidasagatā idha māvasāmi ṭhānan”ti._  


</Paragraph>
</Division>

<Paragraph number="497" type="center" book="vv">
Pesavatīvimānaṃ sattamaṃ.
</Paragraph>
