<Division number="307" book="vv">

<Paragraph number="331" type="verses" book="vv">
_“Pallaṅkaseṭṭhe maṇisoṇṇacitte,_  
_Pupphābhikiṇṇe sayane uḷāre;_  
_Tatthacchasi devi mahānubhāve,_  
_Uccāvacā iddhi vikubbamānā._  

</Paragraph>

</Division>
<Division number="308" book="vv">

<Paragraph number="332" type="verses" book="vv">
_Imā ca te accharāyo samantato,_  
_Naccanti gāyanti pamodayanti;_  
_Deviddhipattāsi mahānubhāve,_  
_Manussabhūtā kimakāsi puññaṃ;_  
_Kenāsi evaṃ jalitānubhāvā,_  
_Vaṇṇo ca te sabbadisā pabhāsatī”ti._  

</Paragraph>

</Division>
<Division number="309" book="vv">

<Paragraph number="333" type="verses" book="vv">
_“Ahaṃ manussesu manussabhūtā,_  
_Aḍḍhe kule suṇisā ahosiṃ;_  
_Akkodhanā bhattuvasānuvattinī,_  
_Uposathe appamattā ahosiṃ._  

</Paragraph>

</Division>
<Division number="310" book="vv">

<Paragraph number="334" type="verses" book="vv">
_Manussabhūtā daharā apāpikā,_  
_Pasannacittā patimābhirādhayiṃ;_  
_Divā ca ratto ca manāpacārinī,_  
_Ahaṃ pure sīlavatī ahosiṃ._  

</Paragraph>

</Division>
<Division number="311" book="vv">

<Paragraph number="335" type="verses" book="vv">
_Pāṇātipātā viratā acorikā,_  
_Saṃsuddhakāyā sucibrahmacārinī;_  
_Amajjapā no ca musā abhāṇiṃ,_  
_Sikkhāpadesu paripūrakārinī._  

</Paragraph>

</Division>
<Division number="312" book="vv">

<Paragraph number="336" type="verses" book="vv">
_Cātuddasiṃ pañcadasiṃ,_  
_yā ca pakkhassa aṭṭhamī;_  
_Pāṭihāriyapakkhañca,_  
_pasannamānasā ahaṃ._  

</Paragraph>

</Division>
<Division number="313" book="vv">

<Paragraph number="337" type="verses" book="vv">
_Aṭṭhaṅgupetaṃ anudhammacārinī,_  
_Uposathaṃ pītimanā upāvasiṃ;_  
_Imañca ariyaṃ aṭṭhaṅgavarehupetaṃ,_  
_Samādiyitvā kusalaṃ sukhudrayaṃ;_  
_Patimhi kalyāṇī vasānuvattinī,_  
_Ahosiṃ pubbe sugatassa sāvikā._  

</Paragraph>

</Division>
<Division number="314" book="vv">

<Paragraph number="338" type="verses" book="vv">
_Etādisaṃ kusalaṃ jīvaloke,_  
_Kammaṃ karitvāna visesabhāginī;_  
_Kāyassa bhedā abhisamparāyaṃ,_  
_Deviddhipattā sugatimhi āgatā._  

</Paragraph>

</Division>
<Division number="315" book="vv">

<Paragraph number="339" type="verses" book="vv">
_Vimānapāsādavare manorame,_  
_Parivāritā accharāsaṅgaṇena;_  
_Sayaṃpabhā devagaṇā ramenti maṃ,_  
_Dīghāyukiṃ devavimānamāgatan”ti._  


</Paragraph>
</Division>

<Paragraph number="340" type="center" book="vv">
Pallaṅkavimānaṃ tatiyaṃ.
</Paragraph>
