<Paragraph number="498" type="verses" book="vv">
_“Pītavatthe pītadhaje,_  
_pītālaṅkārabhūsite;_  
_Pītantarāhi vaggūhi,_  
_apiḷandhāva sobhasi._  

</Paragraph>

</Division>
<Division number="659" book="vv">

<Paragraph number="499" type="verses" book="vv">
_Kā kambukāyūradhare,_  
_kañcanāveḷabhūsite;_  
_Hemajālakasañchanne,_  
_nānāratanamālinī._  

</Paragraph>

</Division>
<Division number="660" book="vv">

<Paragraph number="500" type="verses" book="vv">
_Sovaṇṇamayā lohitaṅgamayā ca,_  
_Muttāmayā veḷuriyamayā ca;_  
_Masāragallā sahalohitaṅgā,_  
_Pārevatakkhīhi maṇīhi cittatā._  

</Paragraph>

</Division>
<Division number="661" book="vv">

<Paragraph number="501" type="verses" book="vv">
_Koci koci ettha mayūrasussaro,_  
_Haṃsassa rañño karavīkasussaro;_  
_Tesaṃ saro suyyati vaggurūpo,_  
_Pañcaṅgikaṃ tūriyamivappavāditaṃ._  

</Paragraph>

</Division>
<Division number="662" book="vv">

<Paragraph number="502" type="verses" book="vv">
_Ratho ca te subho vaggu,_  
_nānāratanacittito;_  
_Nānāvaṇṇāhi dhātūhi,_  
_suvibhattova sobhati._  

</Paragraph>

</Division>
<Division number="663" book="vv">

<Paragraph number="503" type="verses" book="vv">
_Tasmiṃ rathe kañcanabimbavaṇṇe,_  
_Yā tvaṃ ṭhitā bhāsasimaṃ padesaṃ;_  
_Devate pucchitācikkha,_  
_Kissa kammassidaṃ phalan”ti._  

</Paragraph>

</Division>
<Division number="664" book="vv">

<Paragraph number="504" type="verses" book="vv">
_“Sovaṇṇajālaṃ maṇisoṇṇacittitaṃ,_  
_Muttācitaṃ hemajālena channaṃ;_  
_Parinibbute gotame appameyye,_  
_Pasannacittā ahamābhiropayiṃ._  

</Paragraph>

</Division>
<Division number="665" book="vv">

<Paragraph number="505" type="verses" book="vv">
_Tāhaṃ kammaṃ karitvāna,_  
_kusalaṃ buddhavaṇṇitaṃ;_  
_Apetasokā sukhitā,_  
_sampamodāmanāmayā”ti._  


</Paragraph>
</Division>

Mallikāvimānaṃ aṭṭhamaṃ.
