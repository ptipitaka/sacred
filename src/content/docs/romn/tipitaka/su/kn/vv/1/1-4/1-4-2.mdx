<Division number="697" book="vv">

<Paragraph number="544" type="verses" book="vv">
_“Pabhassaravaravaṇṇanibhe,_  
_Surattavatthavasane;_  
_Mahiddhike candanaruciragatte,_  
_Kā tvaṃ subhe devate vandase mamaṃ._  

</Paragraph>

</Division>
<Division number="698" book="vv">

<Paragraph number="545" type="verses" book="vv">
_Pallaṅko ca te mahaggho,_  
_Nānāratanacittito ruciro;_  
_Yattha tvaṃ nisinnā virocasi,_  
_Devarājāriva nandane vane._  

</Paragraph>

</Division>
<Division number="699" book="vv">

<Paragraph number="546" type="verses" book="vv">
_Kiṃ tvaṃ pure sucaritamācarī bhadde,_  
_Kissa kammassa vipākaṃ;_  
_Anubhosi devalokasmiṃ,_  
_Devate pucchitācikkha;_  
_Kissa kammassidaṃ phalan”ti._  

</Paragraph>

</Division>
<Division number="700" book="vv">

<Paragraph number="547" type="verses" book="vv">
_“Piṇḍāya te carantassa,_  
_Mālaṃ phāṇitañca adadaṃ bhante;_  
_Tassa kammassidaṃ vipākaṃ,_  
_Anubhomi devalokasmiṃ._  

</Paragraph>

</Division>
<Division number="701" book="vv">

<Paragraph number="548" type="verses" book="vv">
_Hoti ca me anutāpo,_  
_Aparaddhaṃ dukkhitañca me bhante;_  
_Sāhaṃ dhammaṃ nāssosiṃ,_  
_Sudesitaṃ dhammarājena._  

</Paragraph>

</Division>
<Division number="702" book="vv">

<Paragraph number="549" type="verses" book="vv">
_Taṃ taṃ vadāmi bhaddante,_  
_‘Yassa me anukampiyo koci;_  
_Dhammesu taṃ samādapetha’,_  
_Sudesitaṃ dhammarājena._  

</Paragraph>

</Division>
<Division number="703" book="vv">

<Paragraph number="550" type="verses" book="vv">
_Yesaṃ atthi saddhā buddhe,_  
_dhamme ca saṃgharatane;_  
_Te maṃ ativirocanti,_  
_āyunā yasasā siriyā._  

</Paragraph>

</Division>
<Division number="704" book="vv">

<Paragraph number="551" type="verses" book="vv">
_Patāpena vaṇṇena uttaritarā,_  
_Aññe mahiddhikatarā mayā devā”ti._  


</Paragraph>
</Division>

<Paragraph number="552" type="center" book="vv">
Pabhassaravimānaṃ dutiyaṃ.
</Paragraph>
