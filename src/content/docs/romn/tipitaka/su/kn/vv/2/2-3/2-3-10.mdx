<Paragraph number="1081" type="verses" book="vv">
_“Suṇotha yakkhassa ca vāṇijāna ca,_  
_Samāgamo yattha tadā ahosi;_  
_Yathā kathaṃ itaritarena cāpi,_  
_Subhāsitaṃ tañca suṇātha sabbe._  

</Paragraph>

</Division>
<Division number="1229" book="vv">

<Paragraph number="1082" type="verses" book="vv">
_Yo so ahu rājā pāyāsi nāma,_  
_Bhummānaṃ sahabyagato yasassī;_  
_So modamānova sake vimāne,_  
_Amānuso mānuse ajjhabhāsī”ti._  

</Paragraph>

</Division>
<Division number="1230" book="vv">

<Paragraph number="1083" type="verses" book="vv">
_“Vaṅke araññe amanussaṭṭhāne,_  
_Kantāre appodake appabhakkhe;_  
_Suduggame vaṇṇupathassa majjhe,_  
_Vaṅkaṃ bhayā naṭṭhamanā manussā._  

</Paragraph>

</Division>
<Division number="1231" book="vv">

<Paragraph number="1084" type="verses" book="vv">
_Nayidha phalā mūlamayā ca santi,_  
_Upādānaṃ natthi kutodha bhakkho;_  
_Aññatra paṃsūhi ca vālukāhi ca,_  
_Tatāhi uṇhāhi ca dāruṇāhi ca._  

</Paragraph>

</Division>
<Division number="1232" book="vv">

<Paragraph number="1085" type="verses" book="vv">
_Ujjaṅgalaṃ tattamivaṃ kapālaṃ,_  
_Anāyasaṃ paralokena tulyaṃ;_  
_Luddānamāvāsamidaṃ purāṇaṃ,_  
_Bhūmippadeso abhisattarūpo._  

</Paragraph>

</Division>
<Division number="1233" book="vv">

<Paragraph number="1086" type="verses" book="vv">
_Atha tumhe kena vaṇṇena,_  
_Kimāsamānā imaṃ padesaṃ hi;_  
_Anupaviṭṭhā sahasā samecca,_  
_Lobhā bhayā atha vā sampamūḷhā”ti._  

</Paragraph>

</Division>
<Division number="1234" book="vv">

<Paragraph number="1087" type="verses" book="vv">
_“Magadhesu aṅgesu ca satthavāhā,_  
_Āropayitvā paṇiyaṃ puthuttaṃ;_  
_Te yāmase sindhusovīrabhūmiṃ,_  
_Dhanatthikā uddayaṃ patthayānā._  

</Paragraph>

</Division>
<Division number="1235" book="vv">

<Paragraph number="1088" type="verses" book="vv">
_Divā pipāsaṃ nadhivāsayantā,_  
_Yoggānukampañca samekkhamānā;_  
_Etena vegena āyāma sabbe,_  
_Rattiṃ maggaṃ paṭipannā vikāle._  

</Paragraph>

</Division>
<Division number="1236" book="vv">

<Paragraph number="1089" type="verses" book="vv">
_Te duppayātā aparaddhamaggā,_  
_Andhākulā vippanaṭṭhā araññe;_  
_Suduggame vaṇṇupathassa majjhe,_  
_Disaṃ na jānāma pamūḷhacittā._  

</Paragraph>

</Division>
<Division number="1237" book="vv">

<Paragraph number="1090" type="verses" book="vv">
_Idañca disvāna adiṭṭhapubbaṃ,_  
_Vimānaseṭṭhañca tavañca yakkha;_  
_Tatuttariṃ jīvitamāsamānā,_  
_Disvā patītā sumanā udaggā”ti._  

</Paragraph>

</Division>
<Division number="1238" book="vv">

<Paragraph number="1091" type="verses" book="vv">
_“Pāraṃ samuddassa imañca vaṇṇuṃ,_  
_Vettācaraṃ saṅkupathañca maggaṃ;_  
_Nadiyo pana pabbatānañca duggā,_  
_Puthuddisā gacchatha bhogahetu._  

</Paragraph>

</Division>
<Division number="1239" book="vv">

<Paragraph number="1092" type="verses" book="vv">
_Pakkhandiyāna vijitaṃ paresaṃ,_  
_Verajjake mānuse pekkhamānā;_  
_Yaṃ vo sutaṃ vā atha vāpi diṭṭhaṃ,_  
_Accherakaṃ taṃ vo suṇoma tātā”ti._  

</Paragraph>

</Division>
<Division number="1240" book="vv">

<Paragraph number="1093" type="verses" book="vv">
_“Itopi accherataraṃ kumāra,_  
_Na no sutaṃ vā atha vāpi diṭṭhaṃ;_  
_Atītamānussakameva sabbaṃ,_  
_Disvā na tappāma anomavaṇṇaṃ._  

</Paragraph>

</Division>
<Division number="1241" book="vv">

<Paragraph number="1094" type="verses" book="vv">
_Vehāyasaṃ pokkharañño savanti,_  
_Pahūtamalyā bahupuṇḍarīkā;_  
_Dumā cime niccaphalūpapannā,_  
_Atīva gandhā surabhiṃ pavāyanti._  

</Paragraph>

</Division>
<Division number="1242" book="vv">

<Paragraph number="1095" type="verses" book="vv">
_Veḷūriyathambhā satamussitāse,_  
_Silāpavāḷassa ca āyataṃsā;_  
_Masāragallā sahalohitaṅgā,_  
_Thambhā ime jotirasāmayāse._  

</Paragraph>

</Division>
<Division number="1243" book="vv">

<Paragraph number="1096" type="verses" book="vv">
_Sahassathambhaṃ atulānubhāvaṃ,_  
_Tesūpari sādhumidaṃ vimānaṃ;_  
_Ratanantaraṃ kañcanavedimissaṃ,_  
_Tapanīyapaṭṭehi ca sādhuchannaṃ._  

</Paragraph>

</Division>
<Division number="1244" book="vv">

<Paragraph number="1097" type="verses" book="vv">
_Jambonaduttattamidaṃ sumaṭṭho,_  
_Pāsādasopānaphalūpapanno;_  
_Daḷho ca vaggu ca susaṅgato ca,_  
_Atīva nijjhānakhamo manuñño._  

</Paragraph>

</Division>
<Division number="1245" book="vv">

<Paragraph number="1098" type="verses" book="vv">
_Ratanantarasmiṃ bahuannapānaṃ,_  
_Parivārito accharāsaṅgaṇena;_  
_Murajaālambaratūriyaghuṭṭho,_  
_Abhivanditosi thutivandanāya._  

</Paragraph>

</Division>
<Division number="1246" book="vv">

<Paragraph number="1099" type="verses" book="vv">
_So modasi nārigaṇappabodhano,_  
_Vimānapāsādavare manorame;_  
_Acintiyo sabbaguṇūpapanno,_  
_Rājā yathā vessavaṇo naḷinyā._  

</Paragraph>

</Division>
<Division number="1247" book="vv">

<Paragraph number="1100" type="verses" book="vv">
_Devo nu āsi udavāsi yakkho,_  
_Udāhu devindo manussabhūto;_  
_Pucchanti taṃ vāṇijā satthavāhā,_  
_Ācikkha ko nāma tuvaṃsi yakkho”ti._  

</Paragraph>

</Division>
<Division number="1248" book="vv">

<Paragraph number="1101" type="verses" book="vv">
_“Serīsako nāma ahamhi yakkho,_  
_Kantāriyo vaṇṇupathamhi gutto;_  
_Imaṃ padesaṃ abhipālayāmi,_  
_Vacanakaro vessavaṇassa rañño”ti._  

</Paragraph>

</Division>
<Division number="1249" book="vv">

<Paragraph number="1102" type="verses" book="vv">
_“Adhiccaladdhaṃ pariṇāmajaṃ te,_  
_Sayaṃkataṃ udāhu devehi dinnaṃ;_  
_Pucchanti taṃ vāṇijā satthavāhā,_  
_Kathaṃ tayā laddhamidaṃ manuññan”ti._  

</Paragraph>

</Division>
<Division number="1250" book="vv">

<Paragraph number="1103" type="verses" book="vv">
_“Nādhiccaladdhaṃ na pariṇāmajaṃ me,_  
_Na sayaṃkataṃ na hi devehi dinnaṃ;_  
_Sakehi kammehi apāpakehi,_  
_Puññehi me laddhamidaṃ manuññan”ti._  

</Paragraph>

</Division>
<Division number="1251" book="vv">

<Paragraph number="1104" type="verses" book="vv">
_“Kiṃ te vataṃ kiṃ pana brahmacariyaṃ,_  
_Kissa suciṇṇassa ayaṃ vipāko;_  
_Pucchanti taṃ vāṇijā satthavāhā,_  
_Kathaṃ tayā laddhamidaṃ vimānan”ti._  

</Paragraph>

</Division>
<Division number="1252" book="vv">

<Paragraph number="1105" type="verses" book="vv">
_“Mamaṃ pāyāsīti ahu samaññā,_  
_Rajjaṃ yadā kārayiṃ kosalānaṃ;_  
_Natthikadiṭṭhi kadariyo pāpadhammo,_  
_Ucchedavādī ca tadā ahosiṃ._  

</Paragraph>

</Division>
<Division number="1253" book="vv">

<Paragraph number="1106" type="verses" book="vv">
_Samaṇo ca kho āsi kumārakassapo,_  
_Bahussuto cittakathī uḷāro;_  
_So me tadā dhammakathaṃ abhāsi,_  
_Diṭṭhivisūkāni vinodayī me._  

</Paragraph>

</Division>
<Division number="1254" book="vv">

<Paragraph number="1107" type="verses" book="vv">
_Tāhaṃ tassa dhammakathaṃ suṇitvā,_  
_Upāsakattaṃ paṭivedayissaṃ;_  
_Pāṇātipātā virato ahosiṃ,_  
_Loke adinnaṃ parivajjayissaṃ;_  
_Amajjapo no ca musā abhāṇiṃ,_  
_Sakena dārena ca ahosi tuṭṭho._  

</Paragraph>

</Division>
<Division number="1255" book="vv">

<Paragraph number="1108" type="verses" book="vv">
_Taṃ me vataṃ taṃ pana brahmacariyaṃ,_  
_Tassa suciṇṇassa ayaṃ vipāko;_  
_Teheva kammehi apāpakehi,_  
_Puññehi me laddhamidaṃ vimānan”ti._  

</Paragraph>

</Division>
<Division number="1256" book="vv">

<Paragraph number="1109" type="verses" book="vv">
_“Saccaṃ kirāhaṃsu narā sapaññā,_  
_Anaññathā vacanaṃ paṇḍitānaṃ;_  
_Yahiṃ yahiṃ gacchati puññakammo,_  
_Tahiṃ tahiṃ modati kāmakāmī._  

</Paragraph>

</Division>
<Division number="1257" book="vv">

<Paragraph number="1110" type="verses" book="vv">
_Yahiṃ yahiṃ sokapariddavo ca,_  
_Vadho ca bandho ca parikkileso;_  
_Tahiṃ tahiṃ gacchati pāpakammo,_  
_Na muccati duggatiyā kadācīti._  

</Paragraph>

</Division>
<Division number="1258" book="vv">

<Paragraph number="1111" type="verses" book="vv">
_Sammūḷharūpova jano ahosi,_  
_Asmiṃ muhutte kalalīkatova;_  
_Janassimassa tuyhañca kumāra,_  
_Appaccayo kena nu kho ahosī”ti._  

</Paragraph>

</Division>
<Division number="1259" book="vv">

<Paragraph number="1112" type="verses" book="vv">
_“Ime ca sirīsavanā tātā,_  
_Dibbā gandhā surabhī sampavanti;_  
_Te sampavāyanti imaṃ vimānaṃ,_  
_Divā ca ratto ca tamaṃ nihantvā._  

</Paragraph>

</Division>
<Division number="1260" book="vv">

<Paragraph number="1113" type="verses" book="vv">
_Imesañca kho vassasataccayena,_  
_Sipāṭikā phalati ekamekā;_  
_Mānussakaṃ vassasataṃ atītaṃ,_  
_Yadagge kāyamhi idhūpapanno._  

</Paragraph>

</Division>
<Division number="1261" book="vv">

<Paragraph number="1114" type="verses" book="vv">
_Disvānahaṃ vassasatāni pañca,_  
_Asmiṃ vimāne ṭhatvāna tātā;_  
_Āyukkhayā puññakkhayā cavissaṃ,_  
_Teneva sokena pamucchitosmī”ti._  

</Paragraph>

</Division>
<Division number="1262" book="vv">

<Paragraph number="1115" type="verses" book="vv">
_“Kathaṃ nu soceyya tathāvidho so,_  
_Laddhā vimānaṃ atulaṃ cirāya;_  
_Ye cāpi kho ittaramupapannā,_  
_Te nūna soceyyuṃ parittapuññā”ti._  

</Paragraph>

</Division>
<Division number="1263" book="vv">

<Paragraph number="1116" type="verses" book="vv">
_“Anucchaviṃ ovadiyañca me taṃ,_  
_Yaṃ maṃ tumhe peyyavācaṃ vadetha;_  
_Tumhe ca kho tātā mayānuguttā,_  
_Yenicchakaṃ tena paletha sotthin”ti._  

</Paragraph>

</Division>
<Division number="1264" book="vv">

<Paragraph number="1117" type="verses" book="vv">
_“Gantvā mayaṃ sindhusovīrabhūmiṃ,_  
_Dhanatthikā uddayaṃ patthayānā;_  
_Yathāpayogā paripuṇṇacāgā,_  
_Kāhāma serīsamahaṃ uḷāran”ti._  

</Paragraph>

</Division>
<Division number="1265" book="vv">

<Paragraph number="1118" type="verses" book="vv">
_“Mā ceva serīsamahaṃ akattha,_  
_Sabbañca vo bhavissati yaṃ vadetha;_  
_Pāpāni kammāni vivajjayātha,_  
_Dhammānuyogañca adhiṭṭhahātha._  

</Paragraph>

</Division>
<Division number="1266" book="vv">

<Paragraph number="1119" type="verses" book="vv">
_Upāsako atthi imamhi saṃghe,_  
_Bahussuto sīlavatūpapanno;_  
_Saddho ca cāgī ca supesalo ca,_  
_Vicakkhaṇo santusito mutīmā._  

</Paragraph>

</Division>
<Division number="1267" book="vv">

<Paragraph number="1120" type="verses" book="vv">
_Sañjānamāno na musā bhaṇeyya,_  
_Parūpaghātāya na cetayeyya;_  
_Vebhūtikaṃ pesuṇaṃ no kareyya,_  
_Saṇhañca vācaṃ sakhilaṃ bhaṇeyya._  

</Paragraph>

</Division>
<Division number="1268" book="vv">

<Paragraph number="1121" type="verses" book="vv">
_Sagāravo sappatisso vinīto,_  
_Apāpako adhisīle visuddho;_  
_So mātaraṃ pitarañcāpi jantu,_  
_Dhammena poseti ariyavutti._  

</Paragraph>

</Division>
<Division number="1269" book="vv">

<Paragraph number="1122" type="verses" book="vv">
_Maññe so mātāpitūnaṃ kāraṇā,_  
_Bhogāni pariyesati na attahetu;_  
_Mātāpitūnañca yo accayena,_  
_Nekkhammapoṇo carissati brahmacariyaṃ._  

</Paragraph>

</Division>
<Division number="1270" book="vv">

<Paragraph number="1123" type="verses" book="vv">
_Ujū avaṅko asaṭho amāyo,_  
_Na lesakappena ca vohareyya;_  
_So tādiso sukatakammakārī,_  
_Dhamme ṭhito kinti labhetha dukkhaṃ._  

</Paragraph>

</Division>
<Division number="1271" book="vv">

<Paragraph number="1124" type="verses" book="vv">
_Taṃ kāraṇā pātukatomhi attanā,_  
_Tasmā dhammaṃ passatha vāṇijāse;_  
_Aññatra teniha bhasmī bhavetha,_  
_Andhākulā vippanaṭṭhā araññe;_  
_Taṃ khippamānena lahuṃ parena,_  
_Sukho have sappurisena saṅgamo”ti._  

</Paragraph>

</Division>
<Division number="1272" book="vv">

<Paragraph number="1125" type="verses" book="vv">
_“Kiṃ nāma so kiñca karoti kammaṃ,_  
_Kiṃ nāmadheyyaṃ kiṃ pana tassa gottaṃ;_  
_Mayampi naṃ daṭṭhukāmamha yakkha,_  
_Yassānukampāya idhāgatosi;_  
_Lābhā hi tassa yassa tuvaṃ pihesī”ti._  

</Paragraph>

</Division>
<Division number="1273" book="vv">

<Paragraph number="1126" type="verses" book="vv">
_“Yo kappako sambhavanāmadheyyo,_  
_Upāsako kocchaphalūpajīvī;_  
_Jānātha naṃ tumhākaṃ pesiyo so,_  
_Mā kho naṃ hīḷittha supesalo so”ti._  

</Paragraph>

</Division>
<Division number="1274" book="vv">

<Paragraph number="1127" type="verses" book="vv">
_“Jānāmase yaṃ tvaṃ pavadesi yakkha,_  
_Na kho naṃ jānāma sa edisoti;_  
_Mayampi naṃ pūjayissāma yakkha,_  
_Sutvāna tuyhaṃ vacanaṃ uḷāran”ti._  

</Paragraph>

</Division>
<Division number="1275" book="vv">

<Paragraph number="1128" type="verses" book="vv">
_“Ye keci imasmiṃ satthe manussā,_  
_Daharā mahantā athavāpi majjhimā;_  
_Sabbeva te ālambantu vimānaṃ,_  
_Passantu puññānaṃ phalaṃ kadariyā”ti._  

</Paragraph>

</Division>
<Division number="1276" book="vv">

<Paragraph number="1129" type="verses" book="vv">
_Te tattha sabbeva “ahaṃ pure”ti,_  
_Taṃ kappakaṃ tattha purakkhatvā;_  
_Sabbeva te ālambiṃsu vimānaṃ,_  
_Masakkasāraṃ viya vāsavassa._  

</Paragraph>

</Division>
<Division number="1277" book="vv">

<Paragraph number="1130" type="verses" book="vv">
_Te tattha sabbeva “ahaṃ pure”ti,_  
_Upāsakattaṃ paṭivedayiṃsu;_  
_Pāṇātipātā viratā ahesuṃ,_  
_Loke adinnaṃ parivajjayiṃsu;_  
_Amajjapā no ca musā bhaṇiṃsu,_  
_Sakena dārena ca ahesuṃ tuṭṭhā._  

</Paragraph>

</Division>
<Division number="1278" book="vv">

<Paragraph number="1131" type="verses" book="vv">
_Te tattha sabbeva “ahaṃ pure”ti,_  
_Upāsakattaṃ paṭivedayitvā;_  
_Pakkāmi sattho anumodamāno,_  
_Yakkhiddhiyā anumato punappunaṃ._  

</Paragraph>

</Division>
<Division number="1279" book="vv">

<Paragraph number="1132" type="verses" book="vv">
_Gantvāna te sindhusovīrabhūmiṃ,_  
_Dhanatthikā uddayaṃ patthayānā;_  
_Yathāpayogā paripuṇṇalābhā,_  
_Paccāgamuṃ pāṭaliputtamakkhataṃ._  

</Paragraph>

</Division>
<Division number="1280" book="vv">

<Paragraph number="1133" type="verses" book="vv">
_Gantvāna te saṃgharaṃ sotthivanto,_  
_Puttehi dārehi samaṅgibhūtā;_  
_Ānandī vittā sumanā patītā,_  
_Akaṃsu serīsamahaṃ uḷāraṃ;_  
_Serīsakaṃ te pariveṇaṃ māpayiṃsu._  

</Paragraph>

</Division>
<Division number="1281" book="vv">

<Paragraph number="1134" type="verses" book="vv">
_Etādisā sappurisāna sevanā,_  
_Mahatthikā dhammaguṇāna sevanā;_  
_Ekassa atthāya upāsakassa,_  
_Sabbeva sattā sukhitā ahesunti._  


</Paragraph>
</Division>

<Paragraph number="1135" type="center" book="vv">
Serīsakavimānaṃ dasamaṃ.
