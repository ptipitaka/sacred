<Paragraph number="700" type="verses" book="vv">
_“Ko me vandati pādāni,_  
_iddhiyā yasasā jalaṃ;_  
_Abhikkantena vaṇṇena,_  
_sabbā obhāsayaṃ disā”ti._  

</Paragraph>

</Division>
<Division number="858" book="vv">

<Paragraph number="701" type="verses" book="vv">
_“Maṇḍūkohaṃ pure āsiṃ,_  
_udake vārigocaro;_  
_Tava dhammaṃ suṇantassa,_  
_avadhī vacchapālako._  

</Paragraph>

</Division>
<Division number="859" book="vv">

<Paragraph number="702" type="verses" book="vv">
_Muhuttaṃ cittapasādassa,_  
_iddhiṃ passa yasañca me;_  
_Ānubhāvañca me passa,_  
_vaṇṇaṃ passa jutiñca me._  

</Paragraph>

</Division>
<Division number="860" book="vv">

<Paragraph number="703" type="verses" book="vv">
_Ye ca te dīghamaddhānaṃ,_  
_dhammaṃ assosuṃ gotama;_  
_Pattā te acalaṭṭhānaṃ,_  
_yattha gantvā na socare”ti._  


</Paragraph>
</Division>

<Paragraph number="704" type="center" book="vv">
Maṇḍūkadevaputtavimānaṃ paṭhamaṃ.
</Paragraph>
