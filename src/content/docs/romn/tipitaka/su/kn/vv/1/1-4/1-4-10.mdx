<Paragraph number="646" type="verses" book="vv">
_“Obhāsayitvā pathaviṃ sadevakaṃ,_  
_Atirocasi candimasūriyā viya;_  
_Siriyā ca vaṇṇena yasena tejasā,_  
_Brahmāva deve tidase sahindake._  

</Paragraph>

</Division>
<Division number="809" book="vv" bookNo="19" page="67">

<Paragraph number="647" type="verses" book="vv">
_Pucchāmi taṃ uppalamāladhārinī,_  
_Āveḷinī kañcanasannibhattace;_  
_Alaṅkate uttamavatthadhārinī,_  
_Kā tvaṃ subhe devate vandase mamaṃ._  

</Paragraph>

</Division>
<Division number="810" book="vv" bookNo="19" page="67">

<Paragraph number="648" type="verses" book="vv">
_Kiṃ tvaṃ pure kammamakāsi attanā,_  
_Manussabhūtā purimāya jātiyā;_  
_Dānaṃ suciṇṇaṃ atha sīlasaññamaṃ,_  
_Kenupapannā sugatiṃ yasassinī;_  
_Devate pucchitācikkha,_  
_Kissa kammassidaṃ phalan”ti._  

</Paragraph>

</Division>
<Division number="811" book="vv" bookNo="19" page="67">

<Paragraph number="649" type="verses" book="vv">
_“Idāni bhante imameva gāmaṃ,_  
_Piṇḍāya amhāka gharaṃ upāgami;_  
_Tato te ucchussa adāsi khaṇḍikaṃ,_  
_Pasannacittā atulāya pītiyā._  

</Paragraph>

</Division>
<Division number="812" book="vv" bookNo="19" page="67">

<Paragraph number="650" type="verses" book="vv">
_Sassu ca pacchā anuyuñjate mamaṃ,_  
_Kahaṃ nu ucchuṃ vadhuke avākiri;_  
_Na chaḍḍitaṃ no pana khāditaṃ mayā,_  
_Santassa bhikkhussa sayaṃ adāsahaṃ._  

</Paragraph>

</Division>
<Division number="813" book="vv" bookNo="19" page="67">

<Paragraph number="651" type="verses" book="vv">
_Tuyhaṃ nvidaṃ issariyaṃ atho mama,_  
_Itissā sassu paribhāsate mamaṃ;_  
_Leḍḍuṃ gahetvā pahāraṃ adāsi me,_  
_Tato cutā kālakatāmhi devatā._  

</Paragraph>

</Division>
<Division number="814" book="vv" bookNo="19" page="67">

<Paragraph number="652" type="verses" book="vv">
_Tadeva kammaṃ kusalaṃ kataṃ mayā,_  
_Sukhañca kammaṃ anubhomi attanā;_  
_Devehi saddhiṃ paricārayāmahaṃ,_  
_Modāmahaṃ kāmaguṇehi pañcahi._  

</Paragraph>

</Division>
<Division number="815" book="vv" bookNo="19" page="68">

<Paragraph number="653" type="verses" book="vv">
_Tadeva kammaṃ kusalaṃ kataṃ mayā,_  
_Sukhañca kammaṃ anubhomi attanā;_  
_Devindaguttā tidasehi rakkhitā,_  
_Samappitā kāmaguṇehi pañcahi._  

</Paragraph>

</Division>
<Division number="816" book="vv" bookNo="19" page="68">

<Paragraph number="654" type="verses" book="vv">
_Etādisaṃ puññaphalaṃ anappakaṃ,_  
_Mahāvipākā mama ucchudakkhiṇā;_  
_Devehi saddhiṃ paricārayāmahaṃ,_  
_Modāmahaṃ kāmaguṇehi pañcahi._  

</Paragraph>

</Division>
<Division number="817" book="vv" bookNo="19" page="68">

<Paragraph number="655" type="verses" book="vv">
_Etādisaṃ puññaphalaṃ anappakaṃ,_  
_Mahājutikā mama ucchudakkhiṇā;_  
_Devindaguttā tidasehi rakkhitā,_  
_Sahassanettoriva nandane vane._  

</Paragraph>

</Division>
<Division number="818" book="vv" bookNo="19" page="68">

<Paragraph number="656" type="verses" book="vv">
_Tuvañca bhante anukampakaṃ viduṃ,_  
_Upecca vandiṃ kusalañca pucchisaṃ;_  
_Tato te ucchussa adāsiṃ khaṇḍikaṃ,_  
_Pasannacittā atulāya pītiyā”ti._  


</Paragraph>
</Division>

<Paragraph number="657" type="center" book="vv">
Ucchuvimānaṃ dasamaṃ.
</Paragraph>
