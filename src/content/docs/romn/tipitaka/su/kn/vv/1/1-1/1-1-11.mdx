<Division number="93" book="vv">

<Paragraph number="104" type="verses" book="vv">
_“Koñcā mayūrā diviyā ca haṃsā,_  
_Vaggussarā kokilā sampatanti;_  
_Pupphābhikiṇṇaṃ rammamidaṃ vimānaṃ,_  
_Anekacittaṃ naranārisevitaṃ._  

</Paragraph>

</Division>
<Division number="94" book="vv">

<Paragraph number="105" type="verses" book="vv">
_Tatthacchasi devi mahānubhāve,_  
_Iddhī vikubbanti anekarūpā;_  
_Imā ca te accharāyo samantato,_  
_Naccanti gāyanti pamodayanti ca._  

</Paragraph>

</Division>
<Division number="95" book="vv">

<Paragraph number="106" type="verses" book="vv">
_Deviddhipattāsi mahānubhāve,_  
_Manussabhūtā kimakāsi puññaṃ;_  
_Kenāsi evaṃ jalitānubhāvā,_  
_Vaṇṇo ca te sabbadisā pabhāsatī”ti._  

</Paragraph>

</Division>
<Division number="96" book="vv">

<Paragraph number="107" type="verses" book="vv">
_Sā devatā attamanā,_  
_moggallānena pucchitā;_  
_Pañhaṃ puṭṭhā viyākāsi,_  
_yassa kammassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="97" book="vv">

<Paragraph number="108" type="verses" book="vv">
_“Ahaṃ manussesu manussabhūtā,_  
_Patibbatānaññamanā ahosiṃ;_  
_Mātāva puttaṃ anurakkhamānā,_  
_Kuddhāpihaṃ nappharusaṃ avocaṃ._  

</Paragraph>

</Division>
<Division number="98" book="vv">

<Paragraph number="109" type="verses" book="vv">
_Sacce ṭhitā mosavajjaṃ pahāya,_  
_Dāne ratā saṅgahitattabhāvā;_  
_Annañca pānañca pasannacittā,_  
_Sakkacca dānaṃ vipulaṃ adāsiṃ._  

</Paragraph>

</Division>
<Division number="99" book="vv">

<Paragraph number="110" type="verses" book="vv">
_Tena metādiso vaṇṇo,_  
_tena me idha mijjhati;_  
_Uppajjanti ca me bhogā,_  
_ye keci manaso piyā._  

</Paragraph>

</Division>
<Division number="100" book="vv">

<Paragraph number="111" type="verses" book="vv">
_Akkhāmi te bhikkhu mahānubhāva,_  
_Manussabhūtā yamakāsi puññaṃ;_  
_Tenamhi evaṃ jalitānubhāvā,_  
_Vaṇṇo ca me sabbadisā pabhāsatī”ti._  


</Paragraph>
</Division>

<Paragraph number="112" type="center" book="vv">
Paṭhamapatibbatāvimānaṃ ekādasamaṃ.
</Paragraph>
