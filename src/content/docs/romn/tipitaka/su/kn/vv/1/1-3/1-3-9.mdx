<Paragraph number="507" type="verses" book="vv">
_“Kā nāma tvaṃ visālakkhi,_  
_ramme cittalatāvane;_  
_Samantā anupariyāsi,_  
_nārīgaṇapurakkhatā._  

</Paragraph>

</Division>
<Division number="667" book="vv" bookNo="19" page="54">

<Paragraph number="508" type="verses" book="vv">
_Yadā devā tāvatiṃsā,_  
_pavisanti imaṃ vanaṃ;_  
_Sayoggā sarathā sabbe,_  
_citrā honti idhāgatā._  

</Paragraph>

</Division>
<Division number="668" book="vv" bookNo="19" page="54">

<Paragraph number="509" type="verses" book="vv">
_Tuyhañca idha pattāya,_  
_uyyāne vicarantiyā;_  
_Kāye na dissatī cittaṃ,_  
_kena rūpaṃ tavedisaṃ;_  
_Devate pucchitācikkha,_  
_kissa kammassidaṃ phalan”ti._  

</Paragraph>

</Division>
<Division number="669" book="vv" bookNo="19" page="54">

<Paragraph number="510" type="verses" book="vv">
_“Yena kammena devinda,_  
_rūpaṃ mayhaṃ gatī ca me;_  
_Iddhi ca ānubhāvo ca,_  
_taṃ suṇohi purindada._  

</Paragraph>

</Division>
<Division number="670" book="vv" bookNo="19" page="54">

<Paragraph number="511" type="verses" book="vv">
_Ahaṃ rājagahe ramme,_  
_sunandā nāmupāsikā;_  
_Saddhā sīlena sampannā,_  
_saṃvibhāgaratā sadā._  

</Paragraph>

</Division>
<Division number="671" book="vv" bookNo="19" page="54">

<Paragraph number="512" type="verses" book="vv">
_Acchādanañca bhattañca,_  
_senāsanaṃ padīpiyaṃ;_  
_Adāsiṃ ujubhūtesu,_  
_vippasannena cetasā._  

</Paragraph>

</Division>
<Division number="672" book="vv" bookNo="19" page="54">

<Paragraph number="513" type="verses" book="vv">
_Cātuddasiṃ pañcadasiṃ,_  
_yā ca pakkhassa aṭṭhamī;_  
_Pāṭihāriyapakkhañca,_  
_aṭṭhaṅgasusamāgataṃ._  

</Paragraph>

</Division>
<Division number="673" book="vv" bookNo="19" page="54">

<Paragraph number="514" type="verses" book="vv">
_Uposathaṃ upavasissaṃ,_  
_sadā sīlesu saṃvutā;_  
_Saññamā saṃvibhāgā ca,_  
_vimānaṃ āvasāmahaṃ._  

</Paragraph>

</Division>
<Division number="674" book="vv" bookNo="19" page="54">

<Paragraph number="515" type="verses" book="vv">
_Pāṇātipātā viratā,_  
_musāvādā ca saññatā;_  
_Theyyā ca aticārā ca,_  
_majjapānā ca ārakā._  

</Paragraph>

</Division>
<Division number="675" book="vv" bookNo="19" page="54">

<Paragraph number="516" type="verses" book="vv">
_Pañcasikkhāpade ratā,_  
_ariyasaccāna kovidā;_  
_Upāsikā cakkhumato,_  
_gotamassa yasassino._  

</Paragraph>

</Division>
<Division number="676" book="vv" bookNo="19" page="54">

<Paragraph number="517" type="verses" book="vv">
_Tassā me ñātikulā dāsī,_  
_sadā mālābhihārati;_  
_Tāhaṃ bhagavato thūpe,_  
_sabbamevābhiropayiṃ._  

</Paragraph>

</Division>
<Division number="677" book="vv" bookNo="19" page="54">

<Paragraph number="518" type="verses" book="vv">
_Uposathe cahaṃ gantvā,_  
_mālāgandhavilepanaṃ;_  
_Thūpasmiṃ abhiropesiṃ,_  
_pasannā sehi pāṇibhi._  

</Paragraph>

</Division>
<Division number="678" book="vv" bookNo="19" page="55">

<Paragraph number="519" type="verses" book="vv">
_Tena kammena devinda,_  
_rūpaṃ mayhaṃ gatī ca me;_  
_Iddhī ca ānubhāvo ca,_  
_yaṃ mālaṃ abhiropayiṃ._  

</Paragraph>

</Division>
<Division number="679" book="vv" bookNo="19" page="55">

<Paragraph number="520" type="verses" book="vv">
_Yañca sīlavatī āsiṃ,_  
_na taṃ tāva vipaccati;_  
_Āsā ca pana me devinda,_  
_sakadāgāminī siyan”ti._  


</Paragraph>
</Division>

<Paragraph number="521" type="center" book="vv">
Visālakkhivimānaṃ navamaṃ.
</Paragraph>
