<Paragraph number="18" type="verses" book="vv">
_“Pīṭhaṃ te sovaṇṇamayaṃ uḷāraṃ,_  
_Manojavaṃ gacchati yenakāmaṃ;_  
_Alaṅkate malyadhare suvatthe,_  
_Obhāsasi vijjurivabbhakūṭaṃ._  

</Paragraph>

</Division>
<Division number="16" book="vv" bookNo="19" page="3">

<Paragraph number="19" type="verses" book="vv">
_Kena tetādiso vaṇṇo,_  
_kena te idha mijjhati;_  
_Uppajjanti ca te bhogā,_  
_ye keci manaso piyā._  

</Paragraph>

</Division>
<Division number="17" book="vv" bookNo="19" page="3">

<Paragraph number="20" type="verses" book="vv">
_Pucchāmi taṃ devi mahānubhāve,_  
_Manussabhūtā kimakāsi puññaṃ;_  
_Kenāsi evaṃ jalitānubhāvā,_  
_Vaṇṇo ca te sabbadisā pabhāsatī”ti._  

</Paragraph>

</Division>
<Division number="18" book="vv" bookNo="19" page="3">

<Paragraph number="21" type="verses" book="vv">
_Sā devatā attamanā,_  
_moggallānena pucchitā;_  
_Pañhaṃ puṭṭhā viyākāsi,_  
_yassa kammassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="19" book="vv" bookNo="19" page="3">

<Paragraph number="22" type="verses" book="vv">
_“Appassa kammassa phalaṃ mamedaṃ,_  
_Yenamhi evaṃ jalitānubhāvā;_  
_Ahaṃ manussesu manussabhūtā,_  
_Purimāya jātiyā manussaloke._  

</Paragraph>

</Division>
<Division number="20" book="vv" bookNo="19" page="4">

<Paragraph number="23" type="verses" book="vv">
_Addasaṃ virajaṃ bhikkhuṃ,_  
_vippasannamanāvilaṃ;_  
_Tassa adāsahaṃ pīṭhaṃ,_  
_pasannā sehi pāṇibhi._  

</Paragraph>

</Division>
<Division number="21" book="vv" bookNo="19" page="4">

<Paragraph number="24" type="verses" book="vv">
_Tena metādiso vaṇṇo,_  
_tena me idha mijjhati;_  
_Uppajjanti ca me bhogā,_  
_ye keci manaso piyā._  

</Paragraph>

</Division>
<Division number="22" book="vv" bookNo="19" page="4">

<Paragraph number="25" type="verses" book="vv">
_Akkhāmi te bhikkhu mahānubhāva,_  
_Manussabhūtā yamakāsi puññaṃ;_  
_Tenamhi evaṃ jalitānubhāvā,_  
_Vaṇṇo ca me sabbadisā pabhāsatī”ti._  


</Paragraph>
</Division>

<Paragraph number="26" type="center" book="vv">
Tatiyapīṭhavimānaṃ tatiyaṃ.
</Paragraph>
