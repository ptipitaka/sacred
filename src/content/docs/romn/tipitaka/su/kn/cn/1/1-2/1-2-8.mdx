<Paragraph number="118" type="verses" book="cn">
_“Ye me pubbe viyākaṃsu, __(iccāyasmā hemako)___  
_Huraṃ gotamasāsanā;_  
_Iccāsi iti bhavissati,_  
_Sabbaṃ taṃ itihītihaṃ;_  
_Sabbaṃ taṃ takkavaḍḍhanaṃ,_  
_Nāhaṃ tattha abhiramiṃ. (1)_  

</Paragraph>

</Division>
<Division number="110" book="cn">

<Paragraph number="119" type="verses" book="cn">
_Tvañca me dhammamakkhāhi,_  
_taṇhānigghātanaṃ muni;_  
_Yaṃ viditvā sato caraṃ,_  
_tare loke visattikaṃ”. (2)_  

</Paragraph>

</Division>
<Division number="111" book="cn">

<Paragraph number="120" type="verses" book="cn">
_“Idha diṭṭhasutamutaviññātesu,_  
_Piyarūpesu hemaka;_  
_Chandarāgavinodanaṃ,_  
_Nibbānapadamaccutaṃ. (3)_  

</Paragraph>

</Division>
<Division number="112" book="cn">

<Paragraph number="121" type="verses" book="cn">
_Etadaññāya ye satā,_  
_Diṭṭhadhammābhinibbutā;_  
_Upasantā ca te sadā,_  
_Tiṇṇā loke visattikan”ti. (4)_  


</Paragraph>
</Division>
