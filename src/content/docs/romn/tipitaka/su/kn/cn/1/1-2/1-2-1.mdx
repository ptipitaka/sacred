<Paragraph number="59" type="verses" book="cn">
_“Kenassu nivuto loko, __(iccāyasmā ajito)___  
_Kenassu nappakāsati;_  
_Kissābhilepanaṃ brūsi,_  
_Kiṃsu tassa mahabbhayaṃ”. (1)_  

</Paragraph>

</Division>
<Division number="58" book="cn">

<Paragraph number="60" type="verses" book="cn">
_“Avijjāya nivuto loko, __(ajitāti bhagavā)___  
_Vevicchā pamādā nappakāsati;_  
_Jappābhilepanaṃ brūmi,_  
_Dukkhamassa mahabbhayaṃ”. (2)_  

</Paragraph>

</Division>
<Division number="59" book="cn">

<Paragraph number="61" type="verses" book="cn">
_“Savanti sabbadhi sotā, __(iccāyasmā ajito)___  
_Sotānaṃ kiṃ nivāraṇaṃ;_  
_Sotānaṃ saṃvaraṃ brūhi,_  
_Kena sotā pidhiyyare”. (3)_  

</Paragraph>

</Division>
<Division number="60" book="cn">

<Paragraph number="62" type="verses" book="cn">
_“Yāni sotāni lokasmiṃ, __(ajitāti bhagavā)___  
_Sati tesaṃ nivāraṇaṃ;_  
_Sotānaṃ saṃvaraṃ brūmi,_  
_Paññāyete pidhiyyare”. (4)_  

</Paragraph>

</Division>
<Division number="61" book="cn">

<Paragraph number="63" type="verses" book="cn">
_“Paññā ceva sati cāpi, __(iccāyasmā ajito)___  
_Nāmarūpañca mārisa;_  
_Etaṃ me puṭṭho pabrūhi,_  
_Katthetaṃ uparujjhati”. (5)_  

</Paragraph>

</Division>
<Division number="62" book="cn">

<Paragraph number="64" type="verses" book="cn">
_“Yametaṃ pañhaṃ apucchi,_  
_ajita taṃ vadāmi te;_  
_Yattha nāmañca rūpañca,_  
_asesaṃ uparujjhati;_  
_Viññāṇassa nirodhena,_  
_etthetaṃ uparujjhati”. (6)_  

</Paragraph>

</Division>
<Division number="63" book="cn">

<Paragraph number="65" type="verses" book="cn">
_“Ye ca saṅkhātadhammāse,_  
_ye ca sekhā puthū idha;_  
_Tesaṃ me nipako iriyaṃ,_  
_puṭṭho pabrūhi mārisa”. (7)_  

</Paragraph>

</Division>
<Division number="64" book="cn">

<Paragraph number="66" type="verses" book="cn">
_“Kāmesu nābhigijjheyya,_  
_Manasānāvilo siyā;_  
_Kusalo sabbadhammānaṃ,_  
_Sato bhikkhu paribbaje”ti. (8)_  


</Paragraph>
</Division>

<Paragraph number="67" type="center" book="cn">
Ajitamāṇavapucchā paṭhamā.
</Paragraph>
