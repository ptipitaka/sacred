<Paragraph number="175" type="verses" book="cn">
_“Pārāyanamanugāyissaṃ, __(iccāyasmā piṅgiyo)___  
_Yathāddakkhi tathākkhāsi;_  
_Vimalo bhūrimedhaso,_  
_Nikkāmo nibbano nāgo;_  
_Kissa hetu musā bhaṇe. (1)_  

</Paragraph>

</Division>
<Division number="157" book="cn">

<Paragraph number="176" type="verses" book="cn">
_Pahīnamalamohassa,_  
_mānamakkhappahāyino;_  
_Handāhaṃ kittayissāmi,_  
_giraṃ vaṇṇūpasañhitaṃ. (2)_  

</Paragraph>

</Division>
<Division number="158" book="cn">

<Paragraph number="177" type="verses" book="cn">
_Tamonudo buddho samantacakkhu,_  
_Lokantagū sabbabhavātivatto;_  
_Anāsavo sabbadukkhappahīno,_  
_Saccavhayo brahme upāsito me. (3)_  

</Paragraph>

</Division>
<Division number="159" book="cn">

<Paragraph number="178" type="verses" book="cn">
_Dijo yathā kubbanakaṃ pahāya,_  
_Bahupphalaṃ kānanamāvaseyya;_  
_Evampahaṃ appadasse pahāya,_  
_Mahodadhiṃ haṃsoriva ajjhapatto. (4)_  

</Paragraph>

</Division>
<Division number="160" book="cn">

<Paragraph number="179" type="verses" book="cn">
_Yeme pubbe viyākaṃsu,_  
_Huraṃ gotamasāsanā;_  
_Iccāsi iti bhavissati,_  
_Sabbaṃ taṃ itihītihaṃ;_  
_Sabbaṃ taṃ takkavaḍḍhanaṃ. (5)_  

</Paragraph>

</Division>
<Division number="161" book="cn">

<Paragraph number="180" type="verses" book="cn">
_Eko tamonudāsino,_  
_jutimā so pabhaṅkaro;_  
_Gotamo bhūripaññāṇo,_  
_gotamo bhūrimedhaso. (6)_  

</Paragraph>

</Division>
<Division number="162" book="cn">

<Paragraph number="181" type="verses" book="cn">
_Yo me dhammamadesesi,_  
_Sandiṭṭhikamakālikaṃ;_  
_Taṇhakkhayamanītikaṃ,_  
_Yassa natthi upamā kvaci”. (7)_  

</Paragraph>

</Division>
<Division number="163" book="cn">

<Paragraph number="182" type="verses" book="cn">
_“Kiṃ nu tamhā vippavasasi,_  
_muhuttamapi piṅgiya;_  
_Gotamā bhūripaññāṇā,_  
_gotamā bhūrimedhasā. (8)_  

</Paragraph>

</Division>
<Division number="164" book="cn">

<Paragraph number="183" type="verses" book="cn">

_Sandiṭṭhikamakālikaṃ;_  
_Taṇhakkhayamanītikaṃ,_  
_Yassa natthi upamā kvaci”. (9)_  

</Paragraph>

</Division>
<Division number="165" book="cn">

<Paragraph number="184" type="verses" book="cn">
_“Nāhaṃ tamhā vippavasāmi,_  
_Muhuttamapi brāhmaṇa;_  
_Gotamā bhūripaññāṇā,_  
_Gotamā bhūrimedhasā. (10)_  

</Paragraph>

</Division>
<Division number="166" book="cn">

<Paragraph number="185" type="verses" book="cn">
_Yo me dhammamadesesi,_  
_Sandiṭṭhikamakālikaṃ;_  
_Taṇhakkhayamanītikaṃ,_  
_Yassa natthi upamā kvaci. (11)_  

</Paragraph>

</Division>
<Division number="167" book="cn">

<Paragraph number="186" type="verses" book="cn">
_Passāmi naṃ manasā cakkhunāva,_  
_Rattindivaṃ brāhmaṇa appamatto;_  
_Namassamāno vivasemi rattiṃ,_  
_Teneva maññāmi avippavāsaṃ. (12)_  

</Paragraph>

</Division>
<Division number="168" book="cn">

<Paragraph number="187" type="verses" book="cn">
_Saddhā ca pīti ca mano sati ca,_  
_Nāpentime gotamasāsanamhā;_  
_Yaṃ yaṃ disaṃ vajati bhūripañño,_  
_Sa tena teneva natohamasmi. (13)_  

</Paragraph>

</Division>
<Division number="169" book="cn">

<Paragraph number="188" type="verses" book="cn">
_Jiṇṇassa me dubbalathāmakassa,_  
_Teneva kāyo na paleti tattha;_  
_Saṅkappayantāya vajāmi niccaṃ,_  
_Mano hi me brāhmaṇa tena yutto. (14)_  

</Paragraph>

</Division>
<Division number="170" book="cn">

<Paragraph number="189" type="verses" book="cn">
_Paṅke sayāno pariphandamāno,_  
_Dīpā dīpaṃ upallaviṃ;_  
_Athaddasāsiṃ sambuddhaṃ,_  
_Oghatiṇṇamanāsavaṃ”. (15)_  

</Paragraph>

</Division>
<Division number="171" book="cn">

<Paragraph number="190" type="verses" book="cn">
_“Yathā ahū vakkali muttasaddho,_  
_Bhadrāvudho āḷavigotamo ca;_  
_Evameva tvampi pamuñcassu saddhaṃ,_  
_Gamissasi tvaṃ piṅgiya maccudheyyassa pāraṃ”. (16)_  

</Paragraph>

</Division>
<Division number="172" book="cn">

<Paragraph number="191" type="verses" book="cn">
_“Esa bhiyyo pasīdāmi,_  
_Sutvāna munino vaco;_  
_Vivaṭṭacchado sambuddho,_  
_Akhilo paṭibhānavā. (17)_  

</Paragraph>

</Division>
<Division number="173" book="cn">

<Paragraph number="192" type="verses" book="cn">
_Adhideve abhiññāya,_  
_Sabbaṃ vedi paroparaṃ;_  
_Pañhānantakaro satthā,_  
_Kaṅkhīnaṃ paṭijānataṃ. (18)_  

</Paragraph>

</Division>
<Division number="174" book="cn">

<Paragraph number="193" type="verses" book="cn">

_Yassa natthi upamā kvaci;_  
_Addhā gamissāmi na mettha kaṅkhā,_  
_Evaṃ maṃ dhārehi adhimuttacittan”ti. (19)_  


</Paragraph>
</Division>
