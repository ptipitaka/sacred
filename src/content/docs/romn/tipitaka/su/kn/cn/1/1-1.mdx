<Paragraph number="2" type="verses" book="cn">
_Kosalānaṃ purā rammā,_  
_agamā dakkhiṇāpathaṃ;_  
_Ākiñcaññaṃ patthayāno,_  
_brāhmaṇo mantapāragū._  

</Paragraph>

</Division>
<Division number="2" book="cn" bookNo="25" page="1">

<Paragraph number="3" type="verses" book="cn">
_So assakassa visaye,_  
_maḷakassa samāsane;_  
_Vasi godhāvarīkūle,_  
_uñchena ca phalena ca._  

</Paragraph>

</Division>
<Division number="3" book="cn" bookNo="25" page="1">

<Paragraph number="4" type="verses" book="cn">
_Tasseva upanissāya,_  
_gāmo ca vipulo ahu;_  
_Tato jātena āyena,_  
_mahāyaññamakappayi._  

</Paragraph>

</Division>
<Division number="4" book="cn" bookNo="25" page="1">

<Paragraph number="5" type="verses" book="cn">
_Mahāyaññaṃ yajitvāna,_  
_puna pāvisi assamaṃ;_  
_Tasmiṃ paṭipaviṭṭhamhi,_  
_añño āgañchi brāhmaṇo._  

</Paragraph>

</Division>
<Division number="5" book="cn" bookNo="25" page="1">

<Paragraph number="6" type="verses" book="cn">
_Ugghaṭṭapādo tasito,_  
_paṅkadanto rajassiro;_  
_So ca naṃ upasaṅkamma,_  
_satāni pañca yācati._  

</Paragraph>

</Division>
<Division number="6" book="cn" bookNo="25" page="1">

<Paragraph number="7" type="verses" book="cn">
_Tamenaṃ bāvarī disvā,_  
_āsanena nimantayi;_  
_Sukhañca kusalaṃ pucchi,_  
_idaṃ vacanamabravi._  

</Paragraph>

</Division>
<Division number="7" book="cn" bookNo="25" page="1">

<Paragraph number="8" type="verses" book="cn">
_“Yaṃ kho mama deyyadhammaṃ,_  
_Sabbaṃ visajjitaṃ mayā;_  
_Anujānāhi me brahme,_  
_Natthi pañcasatāni me”._  

</Paragraph>

</Division>
<Division number="8" book="cn" bookNo="25" page="2">

<Paragraph number="9" type="verses" book="cn">
_“Sace me yācamānassa,_  
_bhavaṃ nānupadassati;_  
_Sattame divase tuyhaṃ,_  
_muddhā phalatu sattadhā”._  

</Paragraph>

</Division>
<Division number="9" book="cn" bookNo="25" page="2">

<Paragraph number="10" type="verses" book="cn">
_Abhisaṅkharitvā kuhako,_  
_bheravaṃ so akittayi;_  
_Tassa taṃ vacanaṃ sutvā,_  
_bāvarī dukkhito ahu._  

</Paragraph>

</Division>
<Division number="10" book="cn" bookNo="25" page="2">

<Paragraph number="11" type="verses" book="cn">
_Ussussati anāhāro,_  
_sokasallasamappito;_  
_Athopi evaṃ cittassa,_  
_jhāne na ramatī mano._  

</Paragraph>

</Division>
<Division number="11" book="cn" bookNo="25" page="2">

<Paragraph number="12" type="verses" book="cn">
_Utrastaṃ dukkhitaṃ disvā,_  
_devatā atthakāminī;_  
_Bāvariṃ upasaṅkamma,_  
_idaṃ vacanamabravi._  

</Paragraph>

</Division>
<Division number="12" book="cn" bookNo="25" page="2">

<Paragraph number="13" type="verses" book="cn">
_“Na so muddhaṃ pajānāti,_  
_kuhako so dhanatthiko;_  
_Muddhani muddhapāte vā,_  
_ñāṇaṃ tassa na vijjati”._  

</Paragraph>

</Division>
<Division number="13" book="cn" bookNo="25" page="2">

<Paragraph number="14" type="verses" book="cn">
_“Bhotī carahi jānāti,_  
_Taṃ me akkhāhi pucchitā;_  
_Muddhaṃ muddhādhipātañca,_  
_Taṃ suṇoma vaco tava”._  

</Paragraph>

</Division>
<Division number="14" book="cn" bookNo="25" page="2">

<Paragraph number="15" type="verses" book="cn">
_“Ahampetaṃ na jānāmi,_  
_ñāṇaṃ mettha na vijjati;_  
_Muddhani muddhādhipāte ca,_  
_jinānaṃ hettha dassanaṃ”._  

</Paragraph>

</Division>
<Division number="15" book="cn" bookNo="25" page="2">

<Paragraph number="16" type="verses" book="cn">
_“Atha ko carahi jānāti,_  
_asmiṃ pathavimaṇḍale;_  
_Muddhaṃ muddhādhipātañca,_  
_taṃ me akkhāhi devate”._  

</Paragraph>

</Division>
<Division number="16" book="cn" bookNo="25" page="2">

<Paragraph number="17" type="verses" book="cn">
_“Purā kapilavatthumhā,_  
_nikkhanto lokanāyako;_  
_Apacco okkākarājassa,_  
_sakyaputto pabhaṅkaro._  

</Paragraph>

</Division>
<Division number="17" book="cn" bookNo="25" page="2">

<Paragraph number="18" type="verses" book="cn">
_So hi brāhmaṇa sambuddho,_  
_sabbadhammāna pāragū;_  
_Sabbābhiññābalappatto,_  
_sabbadhammesu cakkhumā;_  
_Sabbakammakkhayaṃ patto,_  
_vimutto upadhikkhaye._  

</Paragraph>

</Division>
<Division number="18" book="cn" bookNo="25" page="2">

<Paragraph number="19" type="verses" book="cn">
_Buddho so bhagavā loke,_  
_dhammaṃ deseti cakkhumā;_  
_Taṃ tvaṃ gantvāna pucchassu,_  
_so te taṃ byākarissati”._  

</Paragraph>

</Division>
<Division number="19" book="cn" bookNo="25" page="2">

<Paragraph number="20" type="verses" book="cn">
_Sambuddhoti vaco sutvā,_  
_udaggo bāvarī ahu;_  
_Sokassa tanuko āsi,_  
_pītiñca vipulaṃ labhi._  

</Paragraph>

</Division>
<Division number="20" book="cn" bookNo="25" page="3">

<Paragraph number="21" type="verses" book="cn">
_So bāvarī attamano udaggo,_  
_Taṃ devataṃ pucchati vedajāto;_  
_“Katamamhi gāme nigamamhi vā pana,_  
_Katamamhi vā janapade lokanātho;_  
_Yattha gantvāna passemu,_  
_Sambuddhaṃ dvipaduttamaṃ”._  

</Paragraph>

</Division>
<Division number="21" book="cn" bookNo="25" page="3">

<Paragraph number="22" type="verses" book="cn">
_“Sāvatthiyaṃ kosalamandire jino,_  
_Pahūtapañño varabhūrimedhaso;_  
_So sakyaputto vidhuro anāsavo,_  
_Muddhādhipātassa vidū narāsabho”._  

</Paragraph>

</Division>
<Division number="22" book="cn" bookNo="25" page="3">

<Paragraph number="23" type="verses" book="cn">
_Tato āmantayī sisse,_  
_brāhmaṇe mantapāragū;_  
_“Etha māṇavā akkhissaṃ,_  
_suṇātha vacanaṃ mama._  

</Paragraph>

</Division>
<Division number="23" book="cn" bookNo="25" page="3">

<Paragraph number="24" type="verses" book="cn">
_Yasseso dullabho loke,_  
_pātubhāvo abhiṇhaso;_  
_Svājja lokamhi uppanno,_  
_sambuddho iti vissuto;_  
_Khippaṃ gantvāna sāvatthiṃ,_  
_passavho dvipaduttamaṃ”._  

</Paragraph>

</Division>
<Division number="24" book="cn" bookNo="25" page="3">

<Paragraph number="25" type="verses" book="cn">
_“Kathaṃ carahi jānemu,_  
_disvā buddhoti brāhmaṇa;_  
_Ajānataṃ no pabrūhi,_  
_yathā jānemu taṃ mayaṃ”._  

</Paragraph>

</Division>
<Division number="25" book="cn" bookNo="25" page="3">

<Paragraph number="26" type="verses" book="cn">
_“Āgatāni hi mantesu,_  
_mahāpurisalakkhaṇā;_  
_Dvattiṃsāni ca byākkhātā,_  
_samattā anupubbaso._  

</Paragraph>

</Division>
<Division number="26" book="cn" bookNo="25" page="3">

<Paragraph number="27" type="verses" book="cn">
_Yassete honti gattesu,_  
_mahāpurisalakkhaṇā;_  
_Dveyeva tassa gatiyo,_  
_tatiyā hi na vijjati._  

</Paragraph>

</Division>
<Division number="27" book="cn" bookNo="25" page="3">

<Paragraph number="28" type="verses" book="cn">
_Sace agāraṃ āvasati,_  
_vijeyya pathaviṃ imaṃ;_  
_Adaṇḍena asatthena,_  
_dhammena anusāsati._  

</Paragraph>

</Division>
<Division number="28" book="cn" bookNo="25" page="3">

<Paragraph number="29" type="verses" book="cn">
_Sace ca so pabbajati,_  
_Agārā anagāriyaṃ;_  
_Vivaṭṭacchado sambuddho,_  
_Arahā bhavati anuttaro._  

</Paragraph>

</Division>
<Division number="29" book="cn" bookNo="25" page="3">

<Paragraph number="30" type="verses" book="cn">
_Jātiṃ gottañca lakkhaṇaṃ,_  
_mante sisse punāpare;_  
_Muddhaṃ muddhādhipātañca,_  
_manasāyeva pucchatha._  

</Paragraph>

</Division>
<Division number="30" book="cn" bookNo="25" page="3">

<Paragraph number="31" type="verses" book="cn">
_Anāvaraṇadassāvī,_  
_yadi buddho bhavissati;_  
_Manasā pucchite pañhe,_  
_vācāya visajjessati”._  

</Paragraph>

</Division>
<Division number="31" book="cn" bookNo="25" page="4">

<Paragraph number="32" type="verses" book="cn">
_Bāvarissa vaco sutvā,_  
_sissā soḷasa brāhmaṇā;_  
_Ajito tissametteyyo,_  
_puṇṇako atha mettagū._  

</Paragraph>

</Division>
<Division number="32" book="cn" bookNo="25" page="4">

<Paragraph number="33" type="verses" book="cn">
_Dhotako upasīvo ca,_  
_nando ca atha hemako;_  
_Todeyyakappā dubhayo,_  
_jatukaṇṇī ca paṇḍito._  

</Paragraph>

</Division>
<Division number="33" book="cn" bookNo="25" page="4">

<Paragraph number="34" type="verses" book="cn">
_Bhadrāvudho udayo ca,_  
_posālo cāpi brāhmaṇo;_  
_Mogharājā ca medhāvī,_  
_piṅgiyo ca mahāisi._  

</Paragraph>

</Division>
<Division number="34" book="cn" bookNo="25" page="4">

<Paragraph number="35" type="verses" book="cn">
_Paccekagaṇino sabbe,_  
_sabbalokassa vissutā;_  
_Jhāyī jhānaratā dhīrā,_  
_pubbavāsanavāsitā._  

</Paragraph>

</Division>
<Division number="35" book="cn" bookNo="25" page="4">

<Paragraph number="36" type="verses" book="cn">
_Bāvariṃ abhivādetvā,_  
_katvā ca naṃ padakkhiṇaṃ;_  
_Jaṭājinadharā sabbe,_  
_pakkāmuṃ uttarāmukhā._  

</Paragraph>

</Division>
<Division number="36" book="cn" bookNo="25" page="4">

<Paragraph number="37" type="verses" book="cn">
_Maḷakassa patiṭṭhānaṃ,_  
_puramāhissatiṃ tadā;_  
_Ujjeniñcāpi gonaddhaṃ,_  
_vedisaṃ vanasavhayaṃ._  

</Paragraph>

</Division>
<Division number="37" book="cn" bookNo="25" page="4">

<Paragraph number="38" type="verses" book="cn">
_Kosambiñcāpi sāketaṃ,_  
_sāvatthiñca puruttamaṃ;_  
_Setabyaṃ kapilavatthuṃ,_  
_kusinārañca mandiraṃ._  

</Paragraph>

</Division>
<Division number="38" book="cn" bookNo="25" page="4">

<Paragraph number="39" type="verses" book="cn">
_Pāvañca bhoganagaraṃ,_  
_vesāliṃ māgadhaṃ puraṃ;_  
_Pāsāṇakaṃ cetiyañca,_  
_ramaṇīyaṃ manoramaṃ._  

</Paragraph>

</Division>
<Division number="39" book="cn" bookNo="25" page="4">

<Paragraph number="40" type="verses" book="cn">
_Tasitovudakaṃ sītaṃ,_  
_mahālābhaṃva vāṇijo;_  
_Chāyaṃ ghammābhitattova,_  
_turitā pabbatamāruhuṃ._  

</Paragraph>

</Division>
<Division number="40" book="cn" bookNo="25" page="4">

<Paragraph number="41" type="verses" book="cn">
_Bhagavā tamhi samaye,_  
_bhikkhusaṃghapurakkhato;_  
_Bhikkhūnaṃ dhammaṃ deseti,_  
_sīhova nadatī vane._  

</Paragraph>

</Division>
<Division number="41" book="cn" bookNo="25" page="4">

<Paragraph number="42" type="verses" book="cn">
_Ajito addasa buddhaṃ,_  
_Pītaraṃsiṃva bhāṇumaṃ;_  
_Candaṃ yathā pannarase,_  
_Paripūraṃ upāgataṃ._  

</Paragraph>

</Division>
<Division number="42" book="cn" bookNo="25" page="4">

<Paragraph number="43" type="verses" book="cn">
_Athassa gatte disvāna,_  
_paripūrañca byañjanaṃ;_  
_Ekamantaṃ ṭhito haṭṭho,_  
_manopañhe apucchatha._  

</Paragraph>

</Division>
<Division number="43" book="cn" bookNo="25" page="4">

<Paragraph number="44" type="verses" book="cn">
_“Ādissa jammanaṃ brūhi,_  
_gottaṃ brūhi salakkhaṇaṃ;_  
_Mantesu pāramiṃ brūhi,_  
_kati vāceti brāhmaṇo”._  

</Paragraph>

</Division>
<Division number="44" book="cn" bookNo="25" page="5">

<Paragraph number="45" type="verses" book="cn">
_“Vīsaṃ vassasataṃ āyu,_  
_so ca gottena bāvarī;_  
_Tīṇissa lakkhaṇā gatte,_  
_tiṇṇaṃ vedāna pāragū._  

</Paragraph>

</Division>
<Division number="45" book="cn" bookNo="25" page="5">

<Paragraph number="46" type="verses" book="cn">
_Lakkhaṇe itihāse ca,_  
_sanighaṇḍusakeṭubhe;_  
_Pañcasatāni vāceti,_  
_sadhamme pāramiṃ gato”._  

</Paragraph>

</Division>
<Division number="46" book="cn" bookNo="25" page="5">

<Paragraph number="47" type="verses" book="cn">
_“Lakkhaṇānaṃ pavicayaṃ,_  
_bāvarissa naruttama;_  
_Taṇhacchida pakāsehi,_  
_mā no kaṅkhāyitaṃ ahu”._  

</Paragraph>

</Division>
<Division number="47" book="cn" bookNo="25" page="5">

<Paragraph number="48" type="verses" book="cn">
_“Mukhaṃ jivhāya chādeti,_  
_uṇṇassa bhamukantare;_  
_Kosohitaṃ vatthaguyhaṃ,_  
_evaṃ jānāhi māṇava”._  

</Paragraph>

</Division>
<Division number="48" book="cn" bookNo="25" page="5">

<Paragraph number="49" type="verses" book="cn">
_Pucchañhi kiñci asuṇanto,_  
_sutvā pañhe viyākate;_  
_Vicinteti jano sabbo,_  
_vedajāto katañjalī._  

</Paragraph>

</Division>
<Division number="49" book="cn" bookNo="25" page="5">

<Paragraph number="50" type="verses" book="cn">
_“Ko nu devo vā brahmā vā,_  
_indo vāpi sujampati;_  
_Manasā pucchite pañhe,_  
_kametaṃ paṭibhāsati”._  

</Paragraph>

</Division>
<Division number="50" book="cn" bookNo="25" page="5">

<Paragraph number="51" type="verses" book="cn">
_“Muddhaṃ muddhādhipātañca,_  
_bāvarī paripucchati;_  
_Taṃ byākarohi bhagavā,_  
_kaṅkhaṃ vinaya no ise”._  

</Paragraph>

</Division>
<Division number="51" book="cn" bookNo="25" page="5">

<Paragraph number="52" type="verses" book="cn">
_“Avijjā muddhāti jānāhi,_  
_vijjā muddhādhipātinī;_  
_Saddhāsatisamādhīhi,_  
_chandavīriyena saṃyutā”._  

</Paragraph>

</Division>
<Division number="52" book="cn" bookNo="25" page="5">

<Paragraph number="53" type="verses" book="cn">
_Tato vedena mahatā,_  
_santhambhetvāna māṇavo;_  
_Ekaṃsaṃ ajinaṃ katvā,_  
_pādesu sirasā pati._  

</Paragraph>

</Division>
<Division number="53" book="cn" bookNo="25" page="5">

<Paragraph number="54" type="verses" book="cn">
_“Bāvarī brāhmaṇo bhoto,_  
_saha sissehi mārisa;_  
_Udaggacitto sumano,_  
_pāde vandati cakkhuma”._  

</Paragraph>

</Division>
<Division number="54" book="cn" bookNo="25" page="5">

<Paragraph number="55" type="verses" book="cn">
_“Sukhito bāvarī hotu,_  
_saha sissehi brāhmaṇo;_  
_Tvañcāpi sukhito hohi,_  
_ciraṃ jīvāhi māṇava._  

</Paragraph>

</Division>
<Division number="55" book="cn" bookNo="25" page="5">

<Paragraph number="56" type="verses" book="cn">
_Bāvarissa ca tuyhaṃ vā,_  
_sabbesaṃ sabbasaṃsayaṃ;_  
_Katāvakāsā pucchavho,_  
_yaṃ kiñci manasicchatha”._  

</Paragraph>

</Division>
<Division number="56" book="cn" bookNo="25" page="5">

<Paragraph number="57" type="verses" book="cn">
_Sambuddhena katokāso,_  
_nisīditvāna pañjalī;_  
_Ajito paṭhamaṃ pañhaṃ,_  
_tattha pucchi tathāgataṃ._  


</Paragraph>
</Division>

<Paragraph number="58" type="center" book="cn">
Vatthugāthā niṭṭhitā.
</Paragraph>
