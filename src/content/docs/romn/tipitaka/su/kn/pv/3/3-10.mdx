<Division number="507" book="pv">

<Paragraph number="550" type="verses" book="pv">
_“Antalikkhasmiṃ tiṭṭhanto,_  
_Duggandho pūti vāyasi;_  
_Mukhañca te kimayo pūtigandhaṃ,_  
_Khādanti kiṃ kammamakāsi pubbe._  

</Paragraph>

</Division>
<Division number="508" book="pv">

<Paragraph number="551" type="verses" book="pv">
_Tato satthaṃ gahetvāna,_  
_okkantanti punappunaṃ;_  
_Khārena paripphositvā,_  
_okkantanti punappunaṃ._  

</Paragraph>

</Division>
<Division number="509" book="pv">

<Paragraph number="552" type="verses" book="pv">
_Kiṃ nu kāyena vācāya,_  
_manasā dukkaṭaṃ kataṃ;_  
_Kissa kammavipākena,_  
_idaṃ dukkhaṃ nigacchasī”ti._  

</Paragraph>

</Division>
<Division number="510" book="pv">

<Paragraph number="553" type="verses" book="pv">
_“Ahaṃ rājagahe ramme,_  
_ramaṇīye giribbaje;_  
_Issaro dhanadhaññassa,_  
_supahūtassa mārisa._  

</Paragraph>

</Division>
<Division number="511" book="pv">

<Paragraph number="554" type="verses" book="pv">
_Tassāyaṃ me bhariyā ca,_  
_dhītā ca suṇisā ca me;_  
_Tā mālaṃ uppalañcāpi,_  
_paccagghañca vilepanaṃ;_  
_Thūpaṃ harantiyo vāresiṃ,_  
_taṃ pāpaṃ pakataṃ mayā._  

</Paragraph>

</Division>
<Division number="512" book="pv">

<Paragraph number="555" type="verses" book="pv">
_Chaḷāsītisahassāni,_  
_mayaṃ paccattavedanā;_  
_Thūpapūjaṃ vivaṇṇetvā,_  
_paccāma niraye bhusaṃ._  

</Paragraph>

</Division>
<Division number="513" book="pv">

<Paragraph number="556" type="verses" book="pv">
_Ye ca kho thūpapūjāya,_  
_vattante arahato mahe;_  
_Ādīnavaṃ pakāsenti,_  
_vivecayetha ne tato._  

</Paragraph>

</Division>
<Division number="514" book="pv">

<Paragraph number="557" type="verses" book="pv">
_Imā ca passa āyantiyo,_  
_Māladhārī alaṅkatā;_  
_Mālāvipākaṃnubhontiyo,_  
_Samiddhā ca tā yasassiniyo._  

</Paragraph>

</Division>
<Division number="515" book="pv">

<Paragraph number="558" type="verses" book="pv">
_Tañca disvāna accheraṃ,_  
_abbhutaṃ lomahaṃsanaṃ;_  
_Namo karonti sappaññā,_  
_vandanti taṃ mahāmuniṃ._  

</Paragraph>

</Division>
<Division number="516" book="pv">

<Paragraph number="559" type="verses" book="pv">
_Sohaṃ nūna ito gantvā,_  
_yoniṃ laddhāna mānusiṃ;_  
_Thūpapūjaṃ karissāmi,_  
_appamatto punappunan”ti._  


</Paragraph>
</Division>

<Paragraph number="560" type="center" book="pv">
Dhātuvivaṇṇapetavatthu dasamaṃ.



</Paragraph>
