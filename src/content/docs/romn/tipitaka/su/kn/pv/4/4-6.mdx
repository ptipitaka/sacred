<Paragraph number="799" type="verses" book="pv">
_“Sāvatthi nāma nagaraṃ,_  
_himavantassa passato;_  
_Tattha āsuṃ dve kumārā,_  
_rājaputtāti me sutaṃ._  

</Paragraph>

</Division>
<Division number="747" book="pv" bookNo="19" page="211">

<Paragraph number="800" type="verses" book="pv">
_Sammattā rajanīyesu,_  
_kāmassādābhinandino;_  
_Paccuppannasukhe giddhā,_  
_na te passiṃsunāgataṃ._  

</Paragraph>

</Division>
<Division number="748" book="pv" bookNo="19" page="211">

<Paragraph number="801" type="verses" book="pv">
_Te cutā ca manussattā,_  
_paralokaṃ ito gatā;_  
_Tedha ghosentyadissantā,_  
_pubbe dukkaṭamattano._  

</Paragraph>

</Division>
<Division number="749" book="pv" bookNo="19" page="211">

<Paragraph number="802" type="verses" book="pv">
_Bahūsu vata santesu,_  
_deyyadhamme upaṭṭhite;_  
_Nāsakkhimhā ca attānaṃ,_  
_parittaṃ kātuṃ sukhāvahaṃ._  

</Paragraph>

</Division>
<Division number="750" book="pv" bookNo="19" page="211">

<Paragraph number="803" type="verses" book="pv">
_Kiṃ tato pāpakaṃ assa,_  
_yaṃ no rājakulā cutā;_  
_Upapannā pettivisayaṃ,_  
_khuppipāsasamappitā._  

</Paragraph>

</Division>
<Division number="751" book="pv" bookNo="19" page="211">

<Paragraph number="804" type="verses" book="pv">
_Sāmino idha hutvāna,_  
_honti asāmino tahiṃ;_  
_Bhamanti khuppipāsāya,_  
_manussā unnatonatā._  

</Paragraph>

</Division>
<Division number="752" book="pv" bookNo="19" page="211">

<Paragraph number="805" type="verses" book="pv">
_Etamādīnavaṃ ñatvā,_  
_issaramadasambhavaṃ;_  
_Pahāya issaramadaṃ,_  
_bhave saggagato naro;_  
_Kāyassa bhedā sappañño,_  
_saggaṃ so upapajjatī”ti._  


</Paragraph>
</Division>

<Paragraph number="806" type="center" book="pv">
Kumārapetavatthu chaṭṭhaṃ.
</Paragraph>
