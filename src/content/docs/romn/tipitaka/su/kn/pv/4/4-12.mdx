<Division number="796" book="pv">

<Paragraph number="855" type="verses" book="pv">
_“Ayañca te pokkharaṇī surammā,_  
_Samā sutitthā ca mahodakā ca;_  
_Supupphitā bhamaragaṇānukiṇṇā,_  
_Kathaṃ tayā laddhā ayaṃ manuññā._  

</Paragraph>

</Division>
<Division number="797" book="pv">

<Paragraph number="856" type="verses" book="pv">
_Idañca te ambavanaṃ surammaṃ,_  
_Sabbotukaṃ dhārayate phalāni;_  
_Supupphitaṃ bhamaragaṇānukiṇṇaṃ,_  
_Kathaṃ tayā laddhamidaṃ vimānaṃ”._  

</Paragraph>

</Division>
<Division number="798" book="pv">

<Paragraph number="857" type="verses" book="pv">
_“Ambapakkaṃ dakaṃ yāgu,_  
_sītacchāyā manoramā;_  
_Dhītāya dinnadānena,_  
_tena me idha labbhati”._  

</Paragraph>

</Division>
<Division number="799" book="pv">

<Paragraph number="858" type="verses" book="pv">
_“Sandiṭṭhikaṃ kammaṃ evaṃ passatha,_  
_Dānassa damassa saṃyamassa vipākaṃ;_  
_Dāsī ahaṃ ayyakulesu hutvā,_  
_Suṇisā homi agārassa issarā”ti._  


</Paragraph>
</Division>

<Paragraph number="859" type="center" book="pv">
Ambavanapetavatthu dvādasamaṃ.
</Paragraph>
