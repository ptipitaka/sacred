<Paragraph number="228" type="verses" book="pv">
_“Uṭṭhehi kaṇha kiṃ sesi,_  
_Ko attho supanena te;_  
_Yo ca tuyhaṃ sako bhātā,_  
_Hadayaṃ cakkhu ca dakkhiṇaṃ;_  
_Tassa vātā balīyanti,_  
_Sasaṃ jappati kesavā”ti._  

</Paragraph>

</Division>
<Division number="208" book="pv" bookNo="19" page="149">

<Paragraph number="229" type="verses" book="pv">
_“Tassa taṃ vacanaṃ sutvā,_  
_rohiṇeyyassa kesavo;_  
_Taramānarūpo vuṭṭhāsi,_  
_bhātusokena aṭṭito._  

</Paragraph>

</Division>
<Division number="209" book="pv" bookNo="19" page="149">

<Paragraph number="230" type="verses" book="pv">
_Kiṃ nu ummattarūpova,_  
_kevalaṃ dvārakaṃ imaṃ;_  
_Saso sasoti lapasi,_  
_kīdisaṃ sasamicchasi._  

</Paragraph>

</Division>
<Division number="210" book="pv" bookNo="19" page="149">

<Paragraph number="231" type="verses" book="pv">
_Sovaṇṇamayaṃ maṇimayaṃ,_  
_Lohamayaṃ atha rūpiyamayaṃ;_  
_Saṅkhasilāpavāḷamayaṃ,_  
_Kārayissāmi te sasaṃ._  

</Paragraph>

</Division>
<Division number="211" book="pv" bookNo="19" page="149">

<Paragraph number="232" type="verses" book="pv">
_Santi aññepi sasakā,_  
_araññavanagocarā;_  
_Tepi te ānayissāmi,_  
_kīdisaṃ sasamicchasī”ti._  

</Paragraph>

</Division>
<Division number="212" book="pv" bookNo="19" page="149">

<Paragraph number="233" type="verses" book="pv">
_“Nāhamete sase icche,_  
_ye sasā pathavissitā;_  
_Candato sasamicchāmi,_  
_taṃ me ohara kesavā”ti._  

</Paragraph>

</Division>
<Division number="213" book="pv" bookNo="19" page="149">

<Paragraph number="234" type="verses" book="pv">
_“So nūna madhuraṃ ñāti,_  
_jīvitaṃ vijahissasi;_  
_Apatthiyaṃ patthayasi,_  
_candato sasamicchasī”ti._  

</Paragraph>

</Division>
<Division number="214" book="pv" bookNo="19" page="149">

<Paragraph number="235" type="verses" book="pv">
_“Evañce kaṇha jānāsi,_  
_yathaññamanusāsasi;_  
_Kasmā pure mataṃ puttaṃ,_  
_ajjāpi manusocasi._  

</Paragraph>

</Division>
<Division number="215" book="pv" bookNo="19" page="149">

<Paragraph number="236" type="verses" book="pv">
_Na yaṃ labbhā manussena,_  
_amanussena vā pana;_  
_Jāto me mā mari putto,_  
_kuto labbhā alabbhiyaṃ._  

</Paragraph>

</Division>
<Division number="216" book="pv" bookNo="19" page="150">

<Paragraph number="237" type="verses" book="pv">
_Na mantā mūlabhesajjā,_  
_osadhehi dhanena vā;_  
_Sakkā ānayituṃ kaṇha,_  
_yaṃ petamanusocasi._  

</Paragraph>

</Division>
<Division number="217" book="pv" bookNo="19" page="150">

<Paragraph number="238" type="verses" book="pv">
_Mahaddhanā mahābhogā,_  
_raṭṭhavantopi khattiyā;_  
_Pahūtadhanadhaññāse,_  
_tepi no ajarāmarā._  

</Paragraph>

</Division>
<Division number="218" book="pv" bookNo="19" page="150">

<Paragraph number="239" type="verses" book="pv">
_Khattiyā brāhmaṇā vessā,_  
_suddā caṇḍālapukkusā;_  
_Ete caññe ca jātiyā,_  
_tepi no ajarāmarā._  

</Paragraph>

</Division>
<Division number="219" book="pv" bookNo="19" page="150">

<Paragraph number="240" type="verses" book="pv">
_Ye mantaṃ parivattenti,_  
_chaḷaṅgaṃ brahmacintitaṃ;_  
_Ete caññe ca vijjāya,_  
_tepi no ajarāmarā._  

</Paragraph>

</Division>
<Division number="220" book="pv" bookNo="19" page="150">

<Paragraph number="241" type="verses" book="pv">
_Isayo vāpi ye santā,_  
_saññatattā tapassino;_  
_Sarīraṃ tepi kālena,_  
_vijahanti tapassino._  

</Paragraph>

</Division>
<Division number="221" book="pv" bookNo="19" page="150">

<Paragraph number="242" type="verses" book="pv">
_Bhāvitattā arahanto,_  
_katakiccā anāsavā;_  
_Nikkhipanti imaṃ dehaṃ,_  
_puññapāpaparikkhayā”ti._  

</Paragraph>

</Division>
<Division number="222" book="pv" bookNo="19" page="150">

<Paragraph number="243" type="verses" book="pv">
_“Ādittaṃ vata maṃ santaṃ,_  
_ghatasittaṃva pāvakaṃ;_  
_Vārinā viya osiñcaṃ,_  
_sabbaṃ nibbāpaye daraṃ._  

</Paragraph>

</Division>
<Division number="223" book="pv" bookNo="19" page="150">

<Paragraph number="244" type="verses" book="pv">
_Abbahī vata me sallaṃ,_  
_sokaṃ hadayanissitaṃ;_  
_Yo me sokaparetassa,_  
_puttasokaṃ apānudi._  

</Paragraph>

</Division>
<Division number="224" book="pv" bookNo="19" page="150">

<Paragraph number="245" type="verses" book="pv">
_Svāhaṃ abbūḷhasallosmi,_  
_sītibhūtosmi nibbuto;_  
_Na socāmi na rodāmi,_  
_tava sutvāna bhātika”._  

</Paragraph>

</Division>
<Division number="225" book="pv" bookNo="19" page="150">

<Paragraph number="246" type="verses" book="pv">
_Evaṃ karonti sappaññā,_  
_ye honti anukampakā;_  
_Nivattayanti sokamhā,_  
_ghaṭo jeṭṭhaṃva bhātaraṃ._  

</Paragraph>

</Division>
<Division number="226" book="pv" bookNo="19" page="150">

<Paragraph number="247" type="verses" book="pv">
_Yassa etādisā honti,_  
_amaccā paricārakā;_  
_Subhāsitena anventi,_  
_ghaṭo jeṭṭhaṃva bhātaranti._  


</Paragraph>
</Division>

<Paragraph number="248" type="center" book="pv">
Kaṇhapetavatthu chaṭṭhaṃ.
</Paragraph>
