<Division number="658" book="pv">

<Paragraph number="707" type="verses" book="pv">
_Rājā piṅgalako nāma,_  
_Suraṭṭhānaṃ adhipati ahu;_  
_Moriyānaṃ upaṭṭhānaṃ gantvā,_  
_Suraṭṭhaṃ punarāgamā._  

</Paragraph>

</Division>
<Division number="659" book="pv">

<Paragraph number="708" type="verses" book="pv">
_Uṇhe majjhanhike kāle,_  
_Rājā paṅkaṃ upāgami;_  
_Addasa maggaṃ ramaṇīyaṃ,_  
_Petānaṃ taṃ vaṇṇupathaṃ._  

</Paragraph>

</Division>
<Division number="660" book="pv">

<Paragraph number="709" book="pv">
Sārathiṃ āmantayī rājā—
</Paragraph>

<Paragraph number="710" type="verses" book="pv">
_“Ayaṃ maggo ramaṇīyo,_  
_khemo sovatthiko sivo;_  
_Iminā sārathi yāma,_  
_suraṭṭhānaṃ santike ito”._  

</Paragraph>

</Division>
<Division number="661" book="pv">

<Paragraph number="711" type="verses" book="pv">
_Tena pāyāsi soraṭṭho,_  
_senāya caturaṅginiyā;_  
_Ubbiggarūpo puriso,_  
_soraṭṭhaṃ etadabravi._  

</Paragraph>

</Division>
<Division number="662" book="pv">

<Paragraph number="712" type="verses" book="pv">
_“Kummaggaṃ paṭipannamhā,_  
_bhiṃsanaṃ lomahaṃsanaṃ;_  
_Purato dissati maggo,_  
_pacchato ca na dissati._  

</Paragraph>

</Division>
<Division number="663" book="pv">

<Paragraph number="713" type="verses" book="pv">
_Kummaggaṃ paṭipannamhā,_  
_yamapurisāna santike;_  
_Amānuso vāyati gandho,_  
_ghoso suyyati dāruṇo”._  

</Paragraph>

</Division>
<Division number="664" book="pv">

<Paragraph number="714" type="verses" book="pv">
_Saṃviggo rājā soraṭṭho,_  
_sārathiṃ etadabravi;_  
_“Kummaggaṃ paṭipannamhā,_  
_bhiṃsanaṃ lomahaṃsanaṃ;_  
_Purato dissati maggo,_  
_pacchato ca na dissati._  

</Paragraph>

</Division>
<Division number="665" book="pv">

<Paragraph number="715" type="verses" book="pv">
_Kummaggaṃ paṭipannamhā,_  
_yamapurisāna santike;_  
_Amānuso vāyati gandho,_  
_ghoso suyyati dāruṇo”._  

</Paragraph>

</Division>
<Division number="666" book="pv">

<Paragraph number="716" type="verses" book="pv">
_Hatthikkhandhaṃ samāruyha,_  
_Olokento catuddisaṃ;_  
_Addasa nigrodhaṃ ramaṇīyaṃ,_  
_Pādapaṃ chāyāsampannaṃ;_  
_Nīlabbhavaṇṇasadisaṃ,_  
_Meghavaṇṇasirīnibhaṃ._  

</Paragraph>

</Division>
<Division number="667" book="pv">

<Paragraph number="717" type="verses" book="pv">
_Sārathiṃ āmantayī rājā,_  
_“kiṃ eso dissati brahā;_  
_Nīlabbhavaṇṇasadiso,_  
_meghavaṇṇasirīnibho”._  

</Paragraph>

</Division>
<Division number="668" book="pv">

<Paragraph number="718" type="verses" book="pv">
_“Nigrodho so mahārāja,_  
_pādapo chāyāsampanno;_  
_Nīlabbhavaṇṇasadiso,_  
_meghavaṇṇasirīnibho”._  

</Paragraph>

</Division>
<Division number="669" book="pv">

<Paragraph number="719" type="verses" book="pv">
_Tena pāyāsi soraṭṭho,_  
_yena so dissate brahā;_  
_Nīlabbhavaṇṇasadiso,_  
_meghavaṇṇasirīnibho._  

</Paragraph>

</Division>
<Division number="670" book="pv">

<Paragraph number="720" type="verses" book="pv">
_Hatthikkhandhato oruyha,_  
_rājā rukkhaṃ upāgami;_  
_Nisīdi rukkhamūlasmiṃ,_  
_sāmacco saparijjano;_  
_Pūraṃ pānīyasarakaṃ,_  
_pūve vitte ca addasa._  

</Paragraph>

</Division>
<Division number="671" book="pv">

<Paragraph number="721" type="verses" book="pv">
_Puriso ca devavaṇṇī,_  
_sabbābharaṇabhūsito;_  
_Upasaṅkamitvā rājānaṃ,_  
_soraṭṭhaṃ etadabravi._  

</Paragraph>

</Division>
<Division number="672" book="pv">

<Paragraph number="722" type="verses" book="pv">
_“Svāgataṃ te mahārāja,_  
_atho te adurāgataṃ;_  
_Pivatu devo pānīyaṃ,_  
_pūve khāda arindama”._  

</Paragraph>

</Division>
<Division number="673" book="pv">

<Paragraph number="723" type="verses" book="pv">
_Pivitvā rājā pānīyaṃ,_  
_sāmacco saparijjano;_  
_Pūve khāditvā pitvā ca,_  
_soraṭṭho etadabravi._  

</Paragraph>

</Division>
<Division number="674" book="pv">

<Paragraph number="724" type="verses" book="pv">
_“Devatā nusi gandhabbo,_  
_adu sakko purindado;_  
_Ajānantā taṃ pucchāma,_  
_kathaṃ jānemu taṃ mayan”ti._  

</Paragraph>

</Division>
<Division number="675" book="pv">

<Paragraph number="725" type="verses" book="pv">
_“Nāmhi devo na gandhabbo,_  
_nāpi sakko purindado;_  
_Peto ahaṃ mahārāja,_  
_suraṭṭhā idha māgato”ti._  

</Paragraph>

</Division>
<Division number="676" book="pv">

<Paragraph number="726" type="verses" book="pv">
_“Kiṃsīlo kiṃsamācāro,_  
_suraṭṭhasmiṃ pure tuvaṃ;_  
_Kena te brahmacariyena,_  
_ānubhāvo ayaṃ tavā”ti._  

</Paragraph>

</Division>
<Division number="677" book="pv">

<Paragraph number="727" type="verses" book="pv">
_“Taṃ suṇohi mahārāja,_  
_arindama raṭṭhavaḍḍhana;_  
_Amaccā pārisajjā ca,_  
_brāhmaṇo ca purohito._  

</Paragraph>

</Division>
<Division number="678" book="pv">

<Paragraph number="728" type="verses" book="pv">
_Suraṭṭhasmiṃ ahaṃ deva,_  
_puriso pāpacetaso;_  
_Micchādiṭṭhi ca dussīlo,_  
_kadariyo paribhāsako._  

</Paragraph>

</Division>
<Division number="679" book="pv">

<Paragraph number="729" type="verses" book="pv">
_Dadantānaṃ karontānaṃ,_  
_vārayissaṃ bahujjanaṃ;_  
_Aññesaṃ dadamānānaṃ,_  
_antarāyakaro ahaṃ._  

</Paragraph>

</Division>
<Division number="680" book="pv">

<Paragraph number="730" type="verses" book="pv">
_Vipāko natthi dānassa,_  
_saṃyamassa kuto phalaṃ;_  
_Natthi ācariyo nāma,_  
_adantaṃ ko damessati._  

</Paragraph>

</Division>
<Division number="681" book="pv">

<Paragraph number="731" type="verses" book="pv">
_Samatulyāni bhūtāni,_  
_kuto jeṭṭhāpacāyiko;_  
_Natthi balaṃ vīriyaṃ vā,_  
_kuto uṭṭhānaporisaṃ._  

</Paragraph>

</Division>
<Division number="682" book="pv">

<Paragraph number="732" type="verses" book="pv">
_Natthi dānaphalaṃ nāma,_  
_na visodheti verinaṃ;_  
_Laddheyyaṃ labhate macco,_  
_niyatipariṇāmajaṃ._  

</Paragraph>

</Division>
<Division number="683" book="pv">

<Paragraph number="733" type="verses" book="pv">
_Natthi mātā pitā bhātā,_  
_loko natthi ito paraṃ;_  
_Natthi dinnaṃ natthi hutaṃ,_  
_sunihitaṃ na vijjati._  

</Paragraph>

</Division>
<Division number="684" book="pv">

<Paragraph number="734" type="verses" book="pv">
_Yopi haneyya purisaṃ,_  
_parassa chindate siraṃ;_  
_Na koci kañci hanati,_  
_sattannaṃ vivaramantare._  

</Paragraph>

</Division>
<Division number="685" book="pv">

<Paragraph number="735" type="verses" book="pv">
_Acchejjābhejjo hi jīvo,_  
_Aṭṭhaṃso guḷaparimaṇḍalo;_  
_Yojanānaṃ sataṃ pañca,_  
_Ko jīvaṃ chettumarahati._  

</Paragraph>

</Division>
<Division number="686" book="pv">

<Paragraph number="736" type="verses" book="pv">
_Yathā suttaguḷe khitte,_  
_nibbeṭhentaṃ palāyati;_  
_Evamevaṃ ca so jīvo,_  
_nibbeṭhento palāyati._  

</Paragraph>

</Division>
<Division number="687" book="pv">

<Paragraph number="737" type="verses" book="pv">
_Yathā gāmato nikkhamma,_  
_aññaṃ gāmaṃ pavīsati;_  
_Evamevaṃ ca so jīvo,_  
_aññaṃ bondiṃ pavīsati._  

</Paragraph>

</Division>
<Division number="688" book="pv">

<Paragraph number="738" type="verses" book="pv">
_Yathā gehato nikkhamma,_  
_aññaṃ gehaṃ pavīsati;_  
_Evamevaṃ ca so jīvo,_  
_aññaṃ bondiṃ pavīsati._  

</Paragraph>

</Division>
<Division number="689" book="pv">

<Paragraph number="739" type="verses" book="pv">
_Cullāsīti mahākappino,_  
_satasahassāni hi;_  
_Ye bālā ye ca paṇḍitā,_  
_saṃsāraṃ khepayitvāna;_  
_Dukkhassantaṃ karissare._  

</Paragraph>

</Division>
<Division number="690" book="pv">

<Paragraph number="740" type="verses" book="pv">
_Mitāni sukhadukkhāni,_  
_doṇehi piṭakehi ca;_  
_Jino sabbaṃ pajānāti,_  
_sammūḷhā itarā pajā._  

</Paragraph>

</Division>
<Division number="691" book="pv">

<Paragraph number="741" type="verses" book="pv">
_Evaṃdiṭṭhi pure āsiṃ,_  
_sammūḷho mohapāruto;_  
_Micchādiṭṭhi ca dussīlo,_  
_kadariyo paribhāsako._  

</Paragraph>

</Division>
<Division number="692" book="pv">

<Paragraph number="742" type="verses" book="pv">
_Oraṃ me chahi māsehi,_  
_kālakiriyā bhavissati;_  
_Ekantakaṭukaṃ ghoraṃ,_  
_nirayaṃ papatissahaṃ._  

</Paragraph>

</Division>
<Division number="693" book="pv">

<Paragraph number="743" type="verses" book="pv">
_Catukkaṇṇaṃ catudvāraṃ,_  
_vibhattaṃ bhāgaso mitaṃ;_  
_Ayopākārapariyantaṃ,_  
_ayasā paṭikujjitaṃ._  

</Paragraph>

</Division>
<Division number="694" book="pv">

<Paragraph number="744" type="verses" book="pv">
_Tassa ayomayā bhūmi,_  
_jalitā tejasā yutā;_  
_Samantā yojanasataṃ,_  
_pharitvā tiṭṭhati sabbadā._  

</Paragraph>

</Division>
<Division number="695" book="pv">

<Paragraph number="745" type="verses" book="pv">
_Vassāni satasahassāni,_  
_ghoso suyyati tāvade;_  
_Lakkho eso mahārāja,_  
_satabhāgavassakoṭiyo._  

</Paragraph>

</Division>
<Division number="696" book="pv">

<Paragraph number="746" type="verses" book="pv">
_Koṭisatasahassāni,_  
_niraye paccare janā;_  
_Micchādiṭṭhī ca dussīlā,_  
_ye ca ariyūpavādino._  

</Paragraph>

</Division>
<Division number="697" book="pv">

<Paragraph number="747" type="verses" book="pv">
_Tatthāhaṃ dīghamaddhānaṃ,_  
_dukkhaṃ vedissa vedanaṃ;_  
_Phalaṃ pāpassa kammassa,_  
_tasmā socāmahaṃ bhusaṃ._  

</Paragraph>

</Division>
<Division number="698" book="pv">

<Paragraph number="748" type="verses" book="pv">
_Taṃ suṇohi mahārāja,_  
_arindama raṭṭhavaḍḍhana;_  
_Dhītā mayhaṃ mahārāja,_  
_uttarā bhaddamatthu te._  

</Paragraph>

</Division>
<Division number="699" book="pv">

<Paragraph number="749" type="verses" book="pv">
_Karoti bhaddakaṃ kammaṃ,_  
_sīlesuposathe ratā;_  
_Saññatā saṃvibhāgī ca,_  
_vadaññū vītamaccharā._  

</Paragraph>

</Division>
<Division number="700" book="pv">

<Paragraph number="750" type="verses" book="pv">
_Akhaṇḍakārī sikkhāya,_  
_suṇhā parakulesu ca;_  
_Upāsikā sakyamunino,_  
_sambuddhassa sirīmato._  

</Paragraph>

</Division>
<Division number="701" book="pv">

<Paragraph number="751" type="verses" book="pv">
_Bhikkhu ca sīlasampanno,_  
_gāmaṃ piṇḍāya pāvisi;_  
_Okkhittacakkhu satimā,_  
_guttadvāro susaṃvuto._  

</Paragraph>

</Division>
<Division number="702" book="pv">

<Paragraph number="752" type="verses" book="pv">
_Sapadānaṃ caramāno,_  
_agamā taṃ nivesanaṃ;_  
_‘Tamaddasa mahārāja,_  
_uttarā bhaddamatthu te’._  

</Paragraph>

</Division>
<Division number="703" book="pv">

<Paragraph number="753" type="verses" book="pv">
_Pūraṃ pānīyasarakaṃ,_  
_pūve vitte ca sā adā;_  
_‘Pitā me kālakato bhante,_  
_tassetaṃ upakappatu’._  

</Paragraph>

</Division>
<Division number="704" book="pv">

<Paragraph number="754" type="verses" book="pv">
_Samanantarānuddiṭṭhe,_  
_vipāko udapajjatha;_  
_Bhuñjāmi kāmakāmīhaṃ,_  
_rājā vessavaṇo yathā._  

</Paragraph>

</Division>
<Division number="705" book="pv">

<Paragraph number="755" type="verses" book="pv">
_Taṃ suṇohi mahārāja,_  
_arindama raṭṭhavaḍḍhana;_  
_Sadevakassa lokassa,_  
_buddho aggo pavuccati;_  
_Taṃ buddhaṃ saraṇaṃ gaccha,_  
_saputtadāro arindama._  

</Paragraph>

</Division>
<Division number="706" book="pv">

<Paragraph number="756" type="verses" book="pv">
_Aṭṭhaṅgikena maggena,_  
_phusanti amataṃ padaṃ;_  
_Taṃ dhammaṃ saraṇaṃ gaccha,_  
_saputtadāro arindama._  

</Paragraph>

</Division>
<Division number="707" book="pv">

<Paragraph number="757" type="verses" book="pv">
_Cattāro ca paṭipannā,_  
_cattāro ca phale ṭhitā;_  
_Esa saṃgho ujubhūto,_  
_paññāsīlasamāhito;_  
_Taṃ saṃghaṃ saraṇaṃ gaccha,_  
_saputtadāro arindama._  

</Paragraph>

</Division>
<Division number="708" book="pv">

<Paragraph number="758" type="verses" book="pv">
_Pāṇātipātā viramassu khippaṃ,_  
_Loke adinnaṃ parivajjayassu;_  
_Amajjapo mā ca musā abhāṇi,_  
_Sakena dārena ca hohi tuṭṭho”ti._  

</Paragraph>

</Division>
<Division number="709" book="pv">

<Paragraph number="759" type="verses" book="pv">
_“Atthakāmosi me yakkha,_  
_hitakāmosi devate;_  
_Karomi tuyhaṃ vacanaṃ,_  
_tvaṃsi ācariyo mama._  

</Paragraph>

</Division>
<Division number="710" book="pv">

<Paragraph number="760" type="verses" book="pv">
_Upemi saraṇaṃ buddhaṃ,_  
_dhammañcāpi anuttaraṃ;_  
_Saṃghañca naradevassa,_  
_gacchāmi saraṇaṃ ahaṃ._  

</Paragraph>

</Division>
<Division number="711" book="pv">

<Paragraph number="761" type="verses" book="pv">
_Pāṇātipātā viramāmi khippaṃ,_  
_Loke adinnaṃ parivajjayāmi;_  
_Amajjapo no ca musā bhaṇāmi,_  
_Sakena dārena ca homi tuṭṭho._  

</Paragraph>

</Division>
<Division number="712" book="pv">

<Paragraph number="762" type="verses" book="pv">
_Ophuṇāmi mahāvāte,_  
_nadiyā sīghagāmiyā;_  
_Vamāmi pāpikaṃ diṭṭhiṃ,_  
_buddhānaṃ sāsane rato”._  

</Paragraph>

</Division>
<Division number="713" book="pv">

<Paragraph number="763" type="verses" book="pv">
_Idaṃ vatvāna soraṭṭho,_  
_viramitvā pāpadassanā;_  
_Namo bhagavato katvā,_  
_pāmokkho rathamāruhīti._  


</Paragraph>
</Division>

<Paragraph number="764" type="center" book="pv">
Nandakapetavatthu tatiyaṃ.
</Paragraph>
