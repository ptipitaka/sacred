<Paragraph number="249" type="verses" book="pv">
_“Naggo dubbaṇṇarūposi,_  
_kiso dhamanisanthato;_  
_Upphāsuliko kisiko,_  
_ko nu tvamasi mārisa”._  

</Paragraph>

</Division>
<Division number="228" book="pv" bookNo="19" page="151">

<Paragraph number="250" type="verses" book="pv">
_“Ahaṃ bhadante petomhi,_  
_duggato yamalokiko;_  
_Pāpakammaṃ karitvāna,_  
_petalokaṃ ito gato”._  

</Paragraph>

</Division>
<Division number="229" book="pv" bookNo="19" page="151">

<Paragraph number="251" type="verses" book="pv">
_“Kiṃ nu kāyena vācāya,_  
_manasā dukkaṭaṃ kataṃ;_  
_Kissa kammavipākena,_  
_petalokaṃ ito gato”._  

</Paragraph>

</Division>
<Division number="230" book="pv" bookNo="19" page="151">

<Paragraph number="252" type="verses" book="pv">
_“Nagaraṃ atthi paṇṇānaṃ,_  
_erakacchanti vissutaṃ;_  
_Tattha seṭṭhi pure āsiṃ,_  
_dhanapāloti maṃ vidū._  

</Paragraph>

</Division>
<Division number="231" book="pv" bookNo="19" page="151">

<Paragraph number="253" type="verses" book="pv">
_Asīti sakaṭavāhānaṃ,_  
_hiraññassa ahosi me;_  
_Pahūtaṃ me jātarūpaṃ,_  
_muttā veḷuriyā bahū._  

</Paragraph>

</Division>
<Division number="232" book="pv" bookNo="19" page="151">

<Paragraph number="254" type="verses" book="pv">
_Tāva mahaddhanassāpi,_  
_na me dātuṃ piyaṃ ahu;_  
_Pidahitvā dvāraṃ bhuñjiṃ,_  
_mā maṃ yācanakāddasuṃ._  

</Paragraph>

</Division>
<Division number="233" book="pv" bookNo="19" page="151">

<Paragraph number="255" type="verses" book="pv">
_Assaddho maccharī cāsiṃ,_  
_kadariyo paribhāsako;_  
_Dadantānaṃ karontānaṃ,_  
_vārayissaṃ bahū jane._  

</Paragraph>

</Division>
<Division number="234" book="pv" bookNo="19" page="151">

<Paragraph number="256" type="verses" book="pv">
_Vipāko natthi dānassa,_  
_saṃyamassa kuto phalaṃ;_  
_Pokkharaññodapānāni,_  
_ārāmāni ca ropite;_  
_Papāyo ca vināsesiṃ,_  
_dugge saṅkamanāni ca._  

</Paragraph>

</Division>
<Division number="235" book="pv" bookNo="19" page="151">

<Paragraph number="257" type="verses" book="pv">
_Svāhaṃ akatakalyāṇo,_  
_katapāpo tato cuto;_  
_Upapanno pettivisayaṃ,_  
_khuppipāsasamappito._  

</Paragraph>

</Division>
<Division number="236" book="pv" bookNo="19" page="151">

<Paragraph number="258" type="verses" book="pv">
_Pañcapaṇṇāsavassāni,_  
_yato kālaṅkato ahaṃ;_  
_Nābhijānāmi bhuttaṃ vā,_  
_pītaṃ vā pana pāniyaṃ._  

</Paragraph>

</Division>
<Division number="237" book="pv" bookNo="19" page="151">

<Paragraph number="259" type="verses" book="pv">
_Yo saṃyamo so vināso,_  
_yo vināso so saṃyamo;_  
_Petā hi kira jānanti,_  
_yo saṃyamo so vināso._  

</Paragraph>

</Division>
<Division number="238" book="pv" bookNo="19" page="152">

<Paragraph number="260" type="verses" book="pv">
_Ahaṃ pure saṃyamissaṃ,_  
_nādāsiṃ bahuke dhane;_  
_Santesu deyyadhammesu,_  
_dīpaṃ nākāsimattano;_  
_Svāhaṃ pacchānutappāmi,_  
_attakammaphalūpago._  

</Paragraph>

</Division>
<Division number="239" book="pv" bookNo="19" page="152">

<Paragraph number="261" type="verses" book="pv">
_Uddhaṃ catūhi māsehi,_  
_kālaṃkiriyā bhavissati;_  
_Ekantakaṭukaṃ ghoraṃ,_  
_nirayaṃ papatissahaṃ._  

</Paragraph>

</Division>
<Division number="240" book="pv" bookNo="19" page="152">

<Paragraph number="262" type="verses" book="pv">
_Catukkaṇṇaṃ catudvāraṃ,_  
_vibhattaṃ bhāgaso mitaṃ;_  
_Ayopākārapariyantaṃ,_  
_ayasā paṭikujjitaṃ._  

</Paragraph>

</Division>
<Division number="241" book="pv" bookNo="19" page="152">

<Paragraph number="263" type="verses" book="pv">
_Tassa ayomayā bhūmi,_  
_jalitā tejasā yutā;_  
_Samantā yojanasataṃ,_  
_pharitvā tiṭṭhati sabbadā._  

</Paragraph>

</Division>
<Division number="242" book="pv" bookNo="19" page="152">

<Paragraph number="264" type="verses" book="pv">
_Tatthāhaṃ dīghamaddhānaṃ,_  
_Dukkhaṃ vedissa vedanaṃ;_  
_Phalaṃ pāpassa kammassa,_  
_Tasmā socāmahaṃ bhusaṃ._  

</Paragraph>

</Division>
<Division number="243" book="pv" bookNo="19" page="152">

<Paragraph number="265" type="verses" book="pv">
_Taṃ vo vadāmi bhaddaṃ vo,_  
_yāvantettha samāgatā;_  
_Mākattha pāpakaṃ kammaṃ,_  
_āvi vā yadi vā raho._  

</Paragraph>

</Division>
<Division number="244" book="pv" bookNo="19" page="152">

<Paragraph number="266" type="verses" book="pv">
_Sace taṃ pāpakaṃ kammaṃ,_  
_karissatha karotha vā;_  
_Na vo dukkhā pamutyatthi,_  
_uppaccāpi palāyataṃ._  

</Paragraph>

</Division>
<Division number="245" book="pv" bookNo="19" page="152">

<Paragraph number="267" type="verses" book="pv">
_Matteyyā hotha petteyyā,_  
_kule jeṭṭhāpacāyikā;_  
_Sāmaññā hotha brahmaññā,_  
_evaṃ saggaṃ gamissathā”ti._  


</Paragraph>
</Division>

<Paragraph number="268" type="center" book="pv">
Dhanapālaseṭṭhipetavatthu sattamaṃ.
</Paragraph>
