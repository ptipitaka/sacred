<Division number="348" book="pv">

<Paragraph number="376" type="verses" book="pv">
_“Soṇṇasopānaphalakā,_  
_soṇṇavālukasanthatā;_  
_Tattha sogandhiyā vaggū,_  
_sucigandhā manoramā._  

</Paragraph>

</Division>
<Division number="349" book="pv">

<Paragraph number="377" type="verses" book="pv">
_Nānārukkhehi sañchannā,_  
_nānāgandhasameritā;_  
_Nānāpadumasañchannā,_  
_puṇḍarīkasamotatā._  

</Paragraph>

</Division>
<Division number="350" book="pv">

<Paragraph number="378" type="verses" book="pv">
_Surabhiṃ sampavāyanti,_  
_manuññā māluteritā;_  
_Haṃsakoñcābhirudā ca,_  
_cakkavakkābhikūjitā._  

</Paragraph>

</Division>
<Division number="351" book="pv">

<Paragraph number="379" type="verses" book="pv">
_Nānādijagaṇākiṇṇā,_  
_nānāsaragaṇāyutā;_  
_Nānāphaladharā rukkhā,_  
_nānāpupphadharā vanā._  

</Paragraph>

</Division>
<Division number="352" book="pv">

<Paragraph number="380" type="verses" book="pv">
_Na manussesu īdisaṃ,_  
_nagaraṃ yādisaṃ idaṃ;_  
_Pāsādā bahukā tuyhaṃ,_  
_sovaṇṇarūpiyāmayā;_  
_Daddallamānā ābhenti,_  
_samantā caturo disā._  

</Paragraph>

</Division>
<Division number="353" book="pv">

<Paragraph number="381" type="verses" book="pv">
_Pañca dāsisatā tuyhaṃ,_  
_yā temā paricārikā;_  
_Tā kambukāyūradharā,_  
_kañcanāveḷabhūsitā._  

</Paragraph>

</Division>
<Division number="354" book="pv">

<Paragraph number="382" type="verses" book="pv">
_Pallaṅkā bahukā tuyhaṃ,_  
_sovaṇṇarūpiyāmayā;_  
_Kadalimigasañchannā,_  
_sajjā gonakasanthatā._  

</Paragraph>

</Division>
<Division number="355" book="pv">

<Paragraph number="383" type="verses" book="pv">
_Yattha tvaṃ vāsūpagatā,_  
_sabbakāmasamiddhinī;_  
_Sampattāyaḍḍharattāya,_  
_tato uṭṭhāya gacchasi._  

</Paragraph>

</Division>
<Division number="356" book="pv">

<Paragraph number="384" type="verses" book="pv">
_Uyyānabhūmiṃ gantvāna,_  
_pokkharaññā samantato;_  
_Tassā tīre tuvaṃ ṭhāsi,_  
_harite saddale subhe._  

</Paragraph>

</Division>
<Division number="357" book="pv">

<Paragraph number="385" type="verses" book="pv">
_Tato te kaṇṇamuṇḍo sunakho,_  
_Aṅgamaṅgāni khādati;_  
_Yadā ca khāyitā āsi,_  
_Aṭṭhisaṅkhalikā katā;_  
_Ogāhasi pokkharaṇiṃ,_  
_Hoti kāyo yathā pure._  

</Paragraph>

</Division>
<Division number="358" book="pv">

<Paragraph number="386" type="verses" book="pv">
_Tato tvaṃ aṅgapaccaṅgī,_  
_sucāru piyadassanā;_  
_Vatthena pārupitvāna,_  
_āyāsi mama santikaṃ._  

</Paragraph>

</Division>
<Division number="359" book="pv">

<Paragraph number="387" type="verses" book="pv">
_Kiṃ nu kāyena vācāya,_  
_manasā dukkaṭaṃ kataṃ;_  
_Kissa kammavipākena,_  
_kaṇṇamuṇḍo sunakho tava;_  
_Aṅgamaṅgāni khādatī”ti._  

</Paragraph>

</Division>
<Division number="360" book="pv">

<Paragraph number="388" type="verses" book="pv">
_“Kimilāyaṃ gahapati,_  
_saddho āsi upāsako;_  
_Tassāhaṃ bhariyā āsiṃ,_  
_dussīlā aticārinī._  

</Paragraph>

</Division>
<Division number="361" book="pv">

<Paragraph number="389" type="verses" book="pv">
_So maṃ aticaramānāya,_  
_sāmiko etadabravi;_  
_‘Netaṃ channaṃ patirūpaṃ,_  
_yaṃ tvaṃ aticarāsi maṃ’._  

</Paragraph>

</Division>
<Division number="362" book="pv">

<Paragraph number="390" type="verses" book="pv">
_Sāhaṃ ghorañca sapathaṃ,_  
_musāvādañca bhāsisaṃ;_  
_‘Nāhaṃ taṃ aticarāmi,_  
_kāyena uda cetasā._  

</Paragraph>

</Division>
<Division number="363" book="pv">

<Paragraph number="391" type="verses" book="pv">
_Sacāhaṃ taṃ aticarāmi,_  
_kāyena uda cetasā;_  
_Kaṇṇamuṇḍoyaṃ sunakho,_  
_aṅgamaṅgāni khādatu’._  

</Paragraph>

</Division>
<Division number="364" book="pv">

<Paragraph number="392" type="verses" book="pv">
_Tassa kammassa vipākaṃ,_  
_musāvādassa cūbhayaṃ;_  
_Satteva vassasatāni,_  
_anubhūtaṃ yato hi me;_  
_Kaṇṇamuṇḍo ca sunakho,_  
_aṅgamaṅgāni khādati._  

</Paragraph>

</Division>
<Division number="365" book="pv">

<Paragraph number="393" type="verses" book="pv">
_Tvañca deva bahukāro,_  
_atthāya me idhāgato;_  
_Sumuttāhaṃ kaṇṇamuṇḍassa,_  
_asokā akutobhayā._  

</Paragraph>

</Division>
<Division number="366" book="pv">

<Paragraph number="394" type="verses" book="pv">
_Tāhaṃ deva namassāmi,_  
_yācāmi pañjalīkatā;_  
_Bhuñja amānuse kāme,_  
_rama deva mayā sahā”ti._  

</Paragraph>

</Division>
<Division number="367" book="pv">

<Paragraph number="395" type="verses" book="pv">
_“Bhuttā amānusā kāmā,_  
_ramitomhi tayā saha;_  
_Tāhaṃ subhage yācāmi,_  
_khippaṃ paṭinayāhi man”ti._  


</Paragraph>
</Division>

<Paragraph number="396" type="center" book="pv">
Kaṇṇamuṇḍapetivatthu dvādasamaṃ.
</Paragraph>
