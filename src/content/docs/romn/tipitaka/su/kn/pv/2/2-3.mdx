<Paragraph number="152" type="verses" book="pv">
_“Naggā dubbaṇṇarūpāsi,_  
_kisā dhamanisanthatā;_  
_Upphāsulike kisike,_  
_kā nu tvaṃ idha tiṭṭhasī”ti._  

</Paragraph>

</Division>
<Division number="135" book="pv">

<Paragraph number="153" type="verses" book="pv">
_“Ahaṃ mattā tuvaṃ tissā,_  
_sapattī te pure ahuṃ;_  
_Pāpakammaṃ karitvāna,_  
_petalokaṃ ito gatā”ti._  

</Paragraph>

</Division>
<Division number="136" book="pv">

<Paragraph number="154" type="verses" book="pv">
_“Kiṃ nu kāyena vācāya,_  
_manasā dukkaṭaṃ kataṃ;_  
_Kissa kammavipākena,_  
_petalokaṃ ito gatā”ti._  

</Paragraph>

</Division>
<Division number="137" book="pv">

<Paragraph number="155" type="verses" book="pv">
_“Caṇḍī ca pharusā cāsiṃ,_  
_issukī maccharī saṭhā;_  
_Tāhaṃ duruttaṃ vatvāna,_  
_petalokaṃ ito gatā”ti._  

</Paragraph>

</Division>
<Division number="138" book="pv">

<Paragraph number="156" type="verses" book="pv">
_“Sabbaṃ ahampi jānāmi,_  
_yathā tvaṃ caṇḍikā ahu;_  
_Aññañca kho taṃ pucchāmi,_  
_kenāsi paṃsukunthitā”ti._  

</Paragraph>

</Division>
<Division number="139" book="pv">

<Paragraph number="157" type="verses" book="pv">
_“Sīsaṃnhātā tuvaṃ āsi,_  
_sucivatthā alaṅkatā;_  
_Ahañca kho adhimattaṃ,_  
_samalaṅkatatarā tayā._  

</Paragraph>

</Division>
<Division number="140" book="pv">

<Paragraph number="158" type="verses" book="pv">
_Tassā me pekkhamānāya,_  
_sāmikena samantayi;_  
_Tato me issā vipulā,_  
_kodho me samajāyatha._  

</Paragraph>

</Division>
<Division number="141" book="pv">

<Paragraph number="159" type="verses" book="pv">
_Tato paṃsuṃ gahetvāna,_  
_paṃsunā taṃ hi okiriṃ;_  
_Tassa kammavipākena,_  
_tenamhi paṃsukunthitā”ti._  

</Paragraph>

</Division>
<Division number="142" book="pv">

<Paragraph number="160" type="verses" book="pv">
_“Sabbaṃ ahampi jānāmi,_  
_paṃsunā maṃ tvamokiri;_  
_Aññañca kho taṃ pucchāmi,_  
_kena khajjasi kacchuyā”ti._  

</Paragraph>

</Division>
<Division number="143" book="pv">

<Paragraph number="161" type="verses" book="pv">
_“Bhesajjahārī ubhayo,_  
_vanantaṃ agamimhase;_  
_Tvañca bhesajjamāhari,_  
_ahañca kapikacchuno._  

</Paragraph>

</Division>
<Division number="144" book="pv">

<Paragraph number="162" type="verses" book="pv">
_Tassā tyājānamānāya,_  
_seyyaṃ tyāhaṃ samokiriṃ;_  
_Tassa kammavipākena,_  
_tena khajjāmi kacchuyā”ti._  

</Paragraph>

</Division>
<Division number="145" book="pv">

<Paragraph number="163" type="verses" book="pv">
_“Sabbaṃ ahampi jānāmi,_  
_seyyaṃ me tvaṃ samokiri;_  
_Aññañca kho taṃ pucchāmi,_  
_kenāsi naggiyā tuvan”ti._  

</Paragraph>

</Division>
<Division number="146" book="pv">

<Paragraph number="164" type="verses" book="pv">
_“Sahāyānaṃ samayo āsi,_  
_ñātīnaṃ samitī ahu;_  
_Tvañca āmantitā āsi,_  
_sasāminī no ca kho ahaṃ._  

</Paragraph>

</Division>
<Division number="147" book="pv">

<Paragraph number="165" type="verses" book="pv">
_Tassā tyājānamānāya,_  
_dussaṃ tyāhaṃ apānudiṃ;_  
_Tassa kammavipākena,_  
_tenamhi naggiyā ahan”ti._  

</Paragraph>

</Division>
<Division number="148" book="pv">

<Paragraph number="166" type="verses" book="pv">
_“Sabbaṃ ahampi jānāmi,_  
_dussaṃ me tvaṃ apānudi;_  
_Aññañca kho taṃ pucchāmi,_  
_kenāsi gūthagandhinī”ti._  

</Paragraph>

</Division>
<Division number="149" book="pv">

<Paragraph number="167" type="verses" book="pv">
_“Tava gandhañca mālañca,_  
_paccagghañca vilepanaṃ;_  
_Gūthakūpe atāresiṃ,_  
_taṃ pāpaṃ pakataṃ mayā;_  
_Tassa kammavipākena,_  
_tenamhi gūthagandhinī”ti._  

</Paragraph>

</Division>
<Division number="150" book="pv">

<Paragraph number="168" type="verses" book="pv">
_“Sabbaṃ ahampi jānāmi,_  
_taṃ pāpaṃ pakataṃ tayā;_  
_Aññañca kho taṃ pucchāmi,_  
_kenāsi duggatā tuvan”ti._  

</Paragraph>

</Division>
<Division number="151" book="pv">

<Paragraph number="169" type="verses" book="pv">
_“Ubhinnaṃ samakaṃ āsi,_  
_yaṃ gehe vijjate dhanaṃ;_  
_Santesu deyyadhammesu,_  
_dīpaṃ nākāsimattano;_  
_Tassa kammavipākena,_  
_tenamhi duggatā ahaṃ._  

</Paragraph>

</Division>
<Division number="152" book="pv">

<Paragraph number="170" type="verses" book="pv">
_Tadeva maṃ tvaṃ avaca,_  
_‘pāpakammaṃ nisevasi;_  
_Na hi pāpehi kammehi,_  
_sulabhā hoti suggatī’”ti._  

</Paragraph>

</Division>
<Division number="153" book="pv">

<Paragraph number="171" type="verses" book="pv">
_“Vāmato maṃ tvaṃ paccesi,_  
_athopi maṃ usūyasi;_  
_Passa pāpānaṃ kammānaṃ,_  
_vipāko hoti yādiso._  

</Paragraph>

</Division>
<Division number="154" book="pv">

<Paragraph number="172" type="verses" book="pv">
_Te gharā tā ca dāsiyo,_  
_tānevābharaṇānime;_  
_Te aññe paricārenti,_  
_na bhogā honti sassatā._  

</Paragraph>

</Division>
<Division number="155" book="pv">

<Paragraph number="173" type="verses" book="pv">
_Idāni bhūtassa pitā,_  
_āpaṇā gehamehiti;_  
_Appeva te dade kiñci,_  
_mā su tāva ito agā”ti._  

</Paragraph>

</Division>
<Division number="156" book="pv">

<Paragraph number="174" type="verses" book="pv">
_“Naggā dubbaṇṇarūpāmhi,_  
_kisā dhamanisanthatā;_  
_Kopīnametaṃ itthīnaṃ,_  
_mā maṃ bhūtapitāddasā”ti._  

</Paragraph>

</Division>
<Division number="157" book="pv">

<Paragraph number="175" type="verses" book="pv">
_“Handa kiṃ vā tyāhaṃ dammi,_  
_Kiṃ vā tedha karomahaṃ;_  
_Yena tvaṃ sukhitā assa,_  
_Sabbakāmasamiddhinī”ti._  

</Paragraph>

</Division>
<Division number="158" book="pv">

<Paragraph number="176" type="verses" book="pv">
_“Cattāro bhikkhū saṃghato,_  
_cattāro pana puggale;_  
_Aṭṭha bhikkhū bhojayitvā,_  
_mama dakkhiṇamādisa;_  
_Tadāhaṃ sukhitā hessaṃ,_  
_sabbakāmasamiddhinī”ti._  

</Paragraph>

</Division>
<Division number="159" book="pv">

<Paragraph number="177" type="verses" book="pv">
_Sādhūti sā paṭissutvā,_  
_bhojayitvāṭṭha bhikkhavo;_  
_Vatthehacchādayitvāna,_  
_tassā dakkhiṇamādisī._  

</Paragraph>

</Division>
<Division number="160" book="pv">

<Paragraph number="178" type="verses" book="pv">
_Samanantarānuddiṭṭhe,_  
_vipāko udapajjatha;_  
_Bhojanacchādanapānīyaṃ,_  
_dakkhiṇāya idaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="161" book="pv">

<Paragraph number="179" type="verses" book="pv">
_Tato suddhā sucivasanā,_  
_kāsikuttamadhārinī;_  
_Vicittavatthābharaṇā,_  
_sapattiṃ upasaṅkami._  

</Paragraph>

</Division>
<Division number="162" book="pv">

<Paragraph number="180" type="verses" book="pv">
_“Abhikkantena vaṇṇena,_  
_yā tvaṃ tiṭṭhasi devate;_  
_Obhāsentī disā sabbā,_  
_osadhī viya tārakā._  

</Paragraph>

</Division>
<Division number="163" book="pv">

<Paragraph number="181" type="verses" book="pv">
_Kena tetādiso vaṇṇo,_  
_kena te idha mijjhati;_  
_Uppajjanti ca te bhogā,_  
_ye keci manaso piyā._  

</Paragraph>

</Division>
<Division number="164" book="pv">

<Paragraph number="182" type="verses" book="pv">
_Pucchāmi taṃ devi mahānubhāve,_  
_Manussabhūtā kimakāsi puññaṃ;_  
_Kenāsi evaṃ jalitānubhāvā,_  
_Vaṇṇo ca te sabbadisā pabhāsatī”ti._  

</Paragraph>

</Division>

</Paragraph>

<Division number="166" book="pv">

<Paragraph number="184" type="verses" book="pv">
_Tava dinnena dānena,_  
_modāmi akutobhayā;_  
_Ciraṃ jīvāhi bhagini,_  
_saha sabbehi ñātibhi;_  
_Asokaṃ virajaṃ ṭhānaṃ,_  
_āvāsaṃ vasavattinaṃ._  

</Paragraph>

</Division>
<Division number="167" book="pv">

<Paragraph number="185" type="verses" book="pv">
_Idha dhammaṃ caritvāna,_  
_Dānaṃ datvāna sobhane;_  
_Vineyya maccheramalaṃ samūlaṃ,_  
_Aninditā saggamupehi ṭhānan”ti._  


</Paragraph>
</Division>
