<Paragraph number="851" type="verses" book="pv">
_“Diṭṭhā tayā nirayā tiracchānayoni,_  
_Petā asurā athavāpi mānusā devā;_  
_Sayamaddasa kammavipākamattano,_  
_Nessāmi taṃ pāṭaliputtamakkhataṃ;_  
_Tattha gantvā kusalaṃ karohi kammaṃ”._  

</Paragraph>

</Division>
<Division number="794" book="pv" bookNo="19" page="215">

<Paragraph number="852" type="verses" book="pv">
_“Atthakāmosi me yakkha,_  
_hitakāmosi devate;_  
_Karomi tuyhaṃ vacanaṃ,_  
_tvaṃsi ācariyo mama._  

</Paragraph>

</Division>
<Division number="795" book="pv" bookNo="19" page="215">

<Paragraph number="853" type="verses" book="pv">
_Diṭṭhā mayā nirayā tiracchānayoni,_  
_Petā asurā athavāpi mānusā devā;_  
_Sayamaddasaṃ kammavipākamattano,_  
_Kāhāmi puññāni anappakānī”ti._  


</Paragraph>
</Division>

<Paragraph number="854" type="center" book="pv">
Pāṭaliputtapetavatthu ekādasamaṃ.
</Paragraph>
