<Division number="14" book="pv">

<Paragraph number="19" type="verses" book="pv">
_“Tirokuṭṭesu tiṭṭhanti,_  
_sandhisiṅghāṭakesu ca;_  
_Dvārabāhāsu tiṭṭhanti,_  
_āgantvāna sakaṃ gharaṃ._  

</Paragraph>

</Division>
<Division number="15" book="pv">

<Paragraph number="20" type="verses" book="pv">
_Pahūte annapānamhi,_  
_khajjabhojje upaṭṭhite;_  
_Na tesaṃ koci sarati,_  
_sattānaṃ kammapaccayā._  

</Paragraph>

</Division>
<Division number="16" book="pv">

<Paragraph number="21" type="verses" book="pv">
_Evaṃ dadanti ñātīnaṃ,_  
_ye honti anukampakā;_  
_Suciṃ paṇītaṃ kālena,_  
_kappiyaṃ pānabhojanaṃ;_  
_‘Idaṃ vo ñātīnaṃ hotu,_  
_sukhitā hontu ñātayo’._  

</Paragraph>

</Division>
<Division number="17" book="pv">

<Paragraph number="22" type="verses" book="pv">
_Te ca tattha samāgantvā,_  
_ñātipetā samāgatā;_  
_Pahūte annapānamhi,_  
_sakkaccaṃ anumodare._  

</Paragraph>

</Division>
<Division number="18" book="pv">

<Paragraph number="23" type="verses" book="pv">
_‘Ciraṃ jīvantu no ñātī,_  
_yesaṃ hetu labhāmase;_  
_Amhākañca katā pūjā,_  
_dāyakā ca anipphalā’._  

</Paragraph>

</Division>
<Division number="19" book="pv">

<Paragraph number="24" type="verses" book="pv">
_Na hi tattha kasī atthi,_  
_gorakkhettha na vijjati;_  
_Vaṇijjā tādisī natthi,_  
_hiraññena kayākayaṃ;_  
_Ito dinnena yāpenti,_  
_petā kālagatā tahiṃ._  

</Paragraph>

</Division>
<Division number="20" book="pv">

<Paragraph number="25" type="verses" book="pv">
_Unname udakaṃ vuṭṭhaṃ,_  
_yathā ninnaṃ pavattati;_  
_Evamevaṃ ito dinnaṃ,_  
_petānaṃ upakappati._  

</Paragraph>

</Division>
<Division number="21" book="pv">

<Paragraph number="26" type="verses" book="pv">
_Yathā vārivahā pūrā,_  
_paripūrenti sāgaraṃ;_  
_Evamevaṃ ito dinnaṃ,_  
_petānaṃ upakappati._  

</Paragraph>

</Division>
<Division number="22" book="pv">

<Paragraph number="27" type="verses" book="pv">
_‘Adāsi me akāsi me,_  
_ñāti mittā sakhā ca me’;_  
_Petānaṃ dakkhiṇaṃ dajjā,_  
_pubbe katamanussaraṃ._  

</Paragraph>

</Division>
<Division number="23" book="pv">

<Paragraph number="28" type="verses" book="pv">
_Na hi ruṇṇaṃ vā soko vā,_  
_yā caññā paridevanā;_  
_Na taṃ petānamatthāya,_  
_evaṃ tiṭṭhanti ñātayo._  

</Paragraph>

</Division>
<Division number="24" book="pv">

<Paragraph number="29" type="verses" book="pv">
_Ayañca kho dakkhiṇā dinnā,_  
_saṃghamhi suppatiṭṭhitā;_  
_Dīgharattaṃ hitāyassa,_  
_ṭhānaso upakappati._  

</Paragraph>

</Division>
<Division number="25" book="pv">

<Paragraph number="30" type="verses" book="pv">
_So ñātidhammo ca ayaṃ nidassito,_  
_Petāna pūjā ca katā uḷārā;_  
_Balañca bhikkhūnamanuppadinnaṃ,_  
_Tumhehi puññaṃ pasutaṃ anappakan”ti._  


</Paragraph>
</Division>

<Paragraph number="31" type="center" book="pv">
Tirokuṭṭapetavatthu pañcamaṃ.
</Paragraph>
