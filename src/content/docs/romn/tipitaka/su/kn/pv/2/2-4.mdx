<Paragraph number="187" type="verses" book="pv">
_“Kāḷī dubbaṇṇarūpāsi,_  
_pharusā bhīrudassanā;_  
_Piṅgalāsi kaḷārāsi,_  
_na taṃ maññāmi mānusin”ti._  

</Paragraph>

</Division>
<Division number="169" book="pv">

<Paragraph number="188" type="verses" book="pv">
_“Ahaṃ nandā nandisena,_  
_bhariyā te pure ahuṃ;_  
_Pāpakammaṃ karitvāna,_  
_petalokaṃ ito gatā”ti._  

</Paragraph>

</Division>
<Division number="170" book="pv">

<Paragraph number="189" type="verses" book="pv">
_“Kiṃ nu kāyena vācāya,_  
_manasā dukkaṭaṃ kataṃ;_  
_Kissa kammavipākena,_  
_petalokaṃ ito gatā”ti._  

</Paragraph>

</Division>
<Division number="171" book="pv">

<Paragraph number="190" type="verses" book="pv">
_“Caṇḍī ca pharusā cāsiṃ,_  
_tayi cāpi agāravā;_  
_Tāhaṃ duruttaṃ vatvāna,_  
_petalokaṃ ito gatā”ti._  

</Paragraph>

</Division>
<Division number="172" book="pv">

<Paragraph number="191" type="verses" book="pv">
_“Handuttarīyaṃ dadāmi te,_  
_imaṃ dussaṃ nivāsaya;_  
_Imaṃ dussaṃ nivāsetvā,_  
_ehi nessāmi taṃ gharaṃ._  

</Paragraph>

</Division>
<Division number="173" book="pv">

<Paragraph number="192" type="verses" book="pv">
_Vatthañca annapānañca,_  
_lacchasi tvaṃ gharaṃ gatā;_  
_Putte ca te passissasi,_  
_suṇisāyo ca dakkhasī”ti._  

</Paragraph>

</Division>
<Division number="174" book="pv">

<Paragraph number="193" type="verses" book="pv">
_“Hatthena hatthe te dinnaṃ,_  
_na mayhaṃ upakappati;_  
_Bhikkhū ca sīlasampanne,_  
_vītarāge bahussute._  

</Paragraph>

</Division>
<Division number="175" book="pv">

<Paragraph number="194" type="verses" book="pv">
_Tappehi annapānena,_  
_mama dakkhiṇamādisa;_  
_Tadāhaṃ sukhitā hessaṃ,_  
_sabbakāmasamiddhinī”ti._  

</Paragraph>

</Division>
<Division number="176" book="pv">

<Paragraph number="195" type="verses" book="pv">
_Sādhūti so paṭissutvā,_  
_dānaṃ vipulamākiri;_  
_Annaṃ pānaṃ khādanīyaṃ,_  
_vatthasenāsanāni ca;_  
_Chattaṃ gandhañca mālañca,_  
_vividhā ca upāhanā._  

</Paragraph>

</Division>
<Division number="177" book="pv">

<Paragraph number="196" type="verses" book="pv">
_Bhikkhū ca sīlasampanne,_  
_vītarāge bahussute;_  
_Tappetvā annapānena,_  
_tassā dakkhiṇamādisī._  

</Paragraph>

</Division>
<Division number="178" book="pv">

<Paragraph number="197" type="verses" book="pv">
_Samanantarānuddiṭṭhe,_  
_vipāko udapajjatha;_  
_Bhojanacchādanapānīyaṃ,_  
_dakkhiṇāya idaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="179" book="pv">

<Paragraph number="198" type="verses" book="pv">
_Tato suddhā sucivasanā,_  
_kāsikuttamadhārinī;_  
_Vicittavatthābharaṇā,_  
_sāmikaṃ upasaṅkami._  

</Paragraph>

</Division>
<Division number="180" book="pv">

<Paragraph number="199" type="verses" book="pv">
_“Abhikkantena vaṇṇena,_  
_yā tvaṃ tiṭṭhasi devate;_  
_Obhāsentī disā sabbā,_  
_osadhī viya tārakā._  

</Paragraph>

</Division>
<Division number="181" book="pv">

<Paragraph number="200" type="verses" book="pv">
_Kena tetādiso vaṇṇo,_  
_kena te idha mijjhati;_  
_Uppajjanti ca te bhogā,_  
_ye keci manaso piyā._  

</Paragraph>

</Division>
<Division number="182" book="pv">

<Paragraph number="201" type="verses" book="pv">
_Pucchāmi taṃ devi mahānubhāve,_  
_Manussabhūtā kimakāsi puññaṃ;_  
_Kenāsi evaṃ jalitānubhāvā,_  
_Vaṇṇo ca te sabbadisā pabhāsatī”ti._  

</Paragraph>

</Division>
<Division number="183" book="pv">

<Paragraph number="202" type="verses" book="pv">
_“Ahaṃ nandā nandisena,_  
_bhariyā te pure ahuṃ;_  
_Pāpakammaṃ karitvāna,_  
_petalokaṃ ito gatā._  

</Paragraph>

</Division>
<Division number="184" book="pv">

<Paragraph number="203" type="verses" book="pv">
_Tava dinnena dānena,_  
_modāmi akutobhayā;_  
_Ciraṃ jīva gahapati,_  
_saha sabbehi ñātibhi;_  
_Asokaṃ virajaṃ khemaṃ,_  
_āvāsaṃ vasavattinaṃ._  

</Paragraph>

</Division>
<Division number="185" book="pv">

<Paragraph number="204" type="verses" book="pv">
_Idha dhammaṃ caritvāna,_  
_Dānaṃ datvā gahapati;_  
_Vineyya maccheramalaṃ samūlaṃ,_  
_Anindito saggamupehi ṭhānan”ti._  


</Paragraph>
</Division>

<Paragraph number="205" type="center" book="pv">
Nandāpetivatthu catutthaṃ.
</Paragraph>
