<Division number="54" book="pv">

<Paragraph number="63" type="verses" book="pv">
_“Gūthañca muttaṃ ruhirañca pubbaṃ,_  
_Paribhuñjati kissa ayaṃ vipāko;_  
_Ayaṃ nu kiṃ kammamakāsi nārī,_  
_Yā sabbadā lohitapubbabhakkhā._  

</Paragraph>

</Division>
<Division number="55" book="pv">

<Paragraph number="64" type="verses" book="pv">
_Navāni vatthāni subhāni ceva,_  
_Mudūni suddhāni ca lomasāni;_  
_Dinnānimissā kitakā bhavanti,_  
_Ayaṃ nu kiṃ kammamakāsi nārī”ti._  

</Paragraph>

</Division>
<Division number="56" book="pv">

<Paragraph number="65" type="verses" book="pv">
_“Bhariyā mamesā ahū bhadante,_  
_Adāyikā maccharinī kadariyā;_  
_Sā maṃ dadantaṃ samaṇabrāhmaṇānaṃ,_  
_Akkosati ca paribhāsati ca._  

</Paragraph>

</Division>
<Division number="57" book="pv">

<Paragraph number="66" type="verses" book="pv">
_‘Gūthañca muttaṃ ruhirañca pubbaṃ,_  
_Paribhuñja tvaṃ asuciṃ sabbakālaṃ;_  
_Etaṃ te paralokasmiṃ hotu,_  
_Vatthā ca te kitakasamā bhavantu’;_  
_Etādisaṃ duccaritaṃ caritvā,_  
_Idhāgatā cirarattāya khādatī”ti._  


</Paragraph>
</Division>

<Paragraph number="67" type="center" book="pv">
Mahāpesakārapetivatthu navamaṃ.
</Paragraph>
