<Paragraph number="133" type="verses" book="pv">
_“Naggā dubbaṇṇarūpāsi,_  
_kisā dhamanisanthatā;_  
_Upphāsulike kisike,_  
_kā nu tvaṃ idha tiṭṭhasi”._  

</Paragraph>

</Division>
<Division number="117" book="pv">

<Paragraph number="134" type="verses" book="pv">
_“Ahaṃ te sakiyā mātā,_  
_pubbe aññāsu jātīsu;_  
_Upapannā pettivisayaṃ,_  
_khuppipāsasamappitā._  

</Paragraph>

</Division>
<Division number="118" book="pv">

<Paragraph number="135" type="verses" book="pv">
_Chaḍḍitaṃ khipitaṃ kheḷaṃ,_  
_siṅghāṇikaṃ silesumaṃ;_  
_Vasañca ḍayhamānānaṃ,_  
_vijātānañca lohitaṃ._  

</Paragraph>

</Division>
<Division number="119" book="pv">

<Paragraph number="136" type="verses" book="pv">
_Vaṇikānañca yaṃ ghāna-_  
_sīsacchinnāna lohitaṃ;_  
_Khudāparetā bhuñjāmi,_  
_itthipurisanissitaṃ._  

</Paragraph>

</Division>
<Division number="120" book="pv">

<Paragraph number="137" type="verses" book="pv">
_Pubbalohitaṃ bhakkhāmi,_  
_pasūnaṃ mānusāna ca;_  
_Aleṇā anagārā ca,_  
_nīlamañcaparāyaṇā._  

</Paragraph>

</Division>
<Division number="121" book="pv">

<Paragraph number="138" type="verses" book="pv">
_Dehi puttaka me dānaṃ,_  
_datvā anvādisāhi me;_  
_Appeva nāma mucceyyaṃ,_  
_pubbalohitabhojanā”ti._  

</Paragraph>

</Division>
<Division number="122" book="pv">

<Paragraph number="139" type="verses" book="pv">
_Mātuyā vacanaṃ sutvā,_  
_upatissonukampako;_  
_Āmantayi moggallānaṃ,_  
_anuruddhañca kappinaṃ._  

</Paragraph>

</Division>
<Division number="123" book="pv">

<Paragraph number="140" type="verses" book="pv">
_Catasso kuṭiyo katvā,_  
_saṃghe cātuddise adā;_  
_Kuṭiyo annapānañca,_  
_mātu dakkhiṇamādisī._  

</Paragraph>

</Division>
<Division number="124" book="pv">

<Paragraph number="141" type="verses" book="pv">
_Samanantarānuddiṭṭhe,_  
_vipāko udapajjatha;_  
_Bhojanaṃ pānīyaṃ vatthaṃ,_  
_dakkhiṇāya idaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="125" book="pv">

<Paragraph number="142" type="verses" book="pv">
_Tato suddhā sucivasanā,_  
_kāsikuttamadhārinī;_  
_Vicittavatthābharaṇā,_  
_kolitaṃ upasaṅkami._  

</Paragraph>

</Division>
<Division number="126" book="pv">

<Paragraph number="143" type="verses" book="pv">
_“Abhikkantena vaṇṇena,_  
_yā tvaṃ tiṭṭhasi devate;_  
_Obhāsentī disā sabbā,_  
_osadhī viya tārakā._  

</Paragraph>

</Division>
<Division number="127" book="pv">

<Paragraph number="144" type="verses" book="pv">
_Kena tetādiso vaṇṇo,_  
_kena te idha mijjhati;_  
_Uppajjanti ca te bhogā,_  
_ye keci manaso piyā._  

</Paragraph>

</Division>
<Division number="128" book="pv">

<Paragraph number="145" type="verses" book="pv">
_Pucchāmi taṃ devi mahānubhāve,_  
_Manussabhūtā kimakāsi puññaṃ;_  
_Kenāsi evaṃ jalitānubhāvā,_  
_Vaṇṇo ca te sabbadisā pabhāsatī”ti._  

</Paragraph>

</Division>

_pubbe aññāsu jātīsu;_  
_Upapannā pettivisayaṃ,_  
_khuppipāsasamappitā._  

</Paragraph>

</Division>

<Paragraph number="150" type="verses" book="pv">
_Sāriputtassa dānena,_  
_modāmi akutobhayā;_  
_Muniṃ kāruṇikaṃ loke,_  
_bhante vanditumāgatā”ti._  


</Paragraph>
