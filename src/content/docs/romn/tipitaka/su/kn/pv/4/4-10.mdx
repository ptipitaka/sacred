<Division number="782" book="pv">

<Paragraph number="839" type="verses" book="pv">
_“Naggā dubbaṇṇarūpāttha,_  
_kisā dhamanisanthatā;_  
_Upphāsulikā kisikā,_  
_ke nu tumhettha mārisā”ti._  

</Paragraph>

</Division>
<Division number="783" book="pv">

<Paragraph number="840" type="verses" book="pv">
_“Mayaṃ bhadante petāmhā,_  
_duggatā yamalokikā;_  
_Pāpakammaṃ karitvāna,_  
_petalokaṃ ito gatā”ti._  

</Paragraph>

</Division>
<Division number="784" book="pv">

<Paragraph number="841" type="verses" book="pv">
_“Kiṃ nu kāyena vācāya,_  
_manasā dukkaṭaṃ kataṃ;_  
_Kissa kammavipākena,_  
_petalokaṃ ito gatā”ti._  

</Paragraph>

</Division>
<Division number="785" book="pv">

<Paragraph number="842" type="verses" book="pv">
_“Anāvaṭesu titthesu,_  
_vicinimhaddhamāsakaṃ;_  
_Santesu deyyadhammesu,_  
_dīpaṃ nākamha attano._  

</Paragraph>

</Division>
<Division number="786" book="pv">

<Paragraph number="843" type="verses" book="pv">
_Nadiṃ upema tasitā,_  
_rittakā parivattati;_  
_Chāyaṃ upema uṇhesu,_  
_ātapo parivattati._  

</Paragraph>

</Division>
<Division number="787" book="pv">

<Paragraph number="844" type="verses" book="pv">
_Aggivaṇṇo ca no vāto,_  
_ḍahanto upavāyati;_  
_Etañca bhante arahāma,_  
_aññañca pāpakaṃ tato._  

</Paragraph>

</Division>
<Division number="788" book="pv">

<Paragraph number="845" type="verses" book="pv">
_Api yojanāni gacchāma,_  
_chātā āhāragedhino;_  
_Aladdhāva nivattāma,_  
_aho no appapuññatā._  

</Paragraph>

</Division>
<Division number="789" book="pv">

<Paragraph number="846" type="verses" book="pv">
_Chātā pamucchitā bhantā,_  
_bhūmiyaṃ paṭisumbhitā;_  
_Uttānā paṭikirāma,_  
_avakujjā patāmase._  

</Paragraph>

</Division>
<Division number="790" book="pv">

<Paragraph number="847" type="verses" book="pv">
_Te ca tattheva patitā,_  
_bhūmiyaṃ paṭisumbhitā;_  
_Uraṃ sīsañca ghaṭṭema,_  
_aho no appapuññatā._  

</Paragraph>

</Division>
<Division number="791" book="pv">

<Paragraph number="848" type="verses" book="pv">
_Etañca bhante arahāma,_  
_aññañca pāpakaṃ tato;_  
_Santesu deyyadhammesu,_  
_dīpaṃ nākamha attano._  

</Paragraph>

</Division>
<Division number="792" book="pv">

<Paragraph number="849" type="verses" book="pv">
_Te hi nūna ito gantvā,_  
_yoniṃ laddhāna mānusiṃ;_  
_Vadaññū sīlasampannā,_  
_kāhāma kusalaṃ bahun”ti._  


</Paragraph>
</Division>

<Paragraph number="850" type="center" book="pv">
Gaṇapetavatthu dasamaṃ.
</Paragraph>
