<Division number="465" book="pv">

<Paragraph number="503" type="verses" book="pv">
_“Ahaṃ bhadante petīmhi,_  
_duggatā yamalokikā;_  
_Pāpakammaṃ karitvāna,_  
_petalokaṃ ito gatā”ti._  

</Paragraph>

</Division>
<Division number="466" book="pv">

<Paragraph number="504" type="verses" book="pv">
_“Kiṃ nu kāyena vācāya,_  
_manasā dukkaṭaṃ kataṃ;_  
_Kissa kammavipākena,_  
_petalokaṃ ito gatā”ti._  

</Paragraph>

</Division>
<Division number="467" book="pv">

<Paragraph number="505" type="verses" book="pv">
_“Anāvaṭesu titthesu,_  
_viciniṃ aḍḍhamāsakaṃ;_  
_Santesu deyyadhammesu,_  
_dīpaṃ nākāsimattano._  

</Paragraph>

</Division>
<Division number="468" book="pv">

<Paragraph number="506" type="verses" book="pv">
_Nadiṃ upemi tasitā,_  
_rittakā parivattati;_  
_Chāyaṃ upemi uṇhesu,_  
_ātapo parivattati._  

</Paragraph>

</Division>
<Division number="469" book="pv">

<Paragraph number="507" type="verses" book="pv">
_Aggivaṇṇo ca me vāto,_  
_ḍahanto upavāyati;_  
_Etañca bhante arahāmi,_  
_aññañca pāpakaṃ tato._  

</Paragraph>

</Division>
<Division number="470" book="pv">

<Paragraph number="508" type="verses" book="pv">
_Gantvāna hatthiniṃ puraṃ,_  
_vajjesi mayha mātaraṃ;_  
_‘Dhītā ca te mayā diṭṭhā,_  
_duggatā yamalokikā;_  
_Pāpakammaṃ karitvāna,_  
_petalokaṃ ito gatā’._  

</Paragraph>

</Division>
<Division number="471" book="pv">

<Paragraph number="509" type="verses" book="pv">
_Atthi me ettha nikkhittaṃ,_  
_anakkhātañca taṃ mayā;_  
_Cattārisatasahassāni,_  
_pallaṅkassa ca heṭṭhato._  

</Paragraph>

</Division>
<Division number="472" book="pv">

<Paragraph number="510" type="verses" book="pv">
_Tato me dānaṃ dadatu,_  
_tassā ca hotu jīvikā;_  
_Dānaṃ datvā ca me mātā,_  
_dakkhiṇaṃ anudicchatu;_  
_Tadāhaṃ sukhitā hessaṃ,_  
_sabbakāmasamiddhinī”ti._  

</Paragraph>

</Division>
<Division number="473" book="pv">

</Paragraph>

</Paragraph>
