<Paragraph number="529" type="verses" book="pv">
_“Kūṭāgāre ca pāsāde,_  
_pallaṅke gonakatthate;_  
_Pañcaṅgikena turiyena,_  
_ramasi suppavādite._  

</Paragraph>

</Division>
<Division number="489" book="pv" bookNo="19" page="178">

<Paragraph number="530" type="verses" book="pv">
_Tato ratyā vivasāne,_  
_sūriyuggamanaṃ pati;_  
_Apaviddho susānasmiṃ,_  
_bahudukkhaṃ nigacchasi._  

</Paragraph>

</Division>
<Division number="490" book="pv" bookNo="19" page="178">

<Paragraph number="531" type="verses" book="pv">
_Kiṃ nu kāyena vācāya,_  
_manasā dukkaṭaṃ kataṃ;_  
_Kissa kammavipākena,_  
_idaṃ dukkhaṃ nigacchasi”._  

</Paragraph>

</Division>
<Division number="491" book="pv" bookNo="19" page="178">

<Paragraph number="532" type="verses" book="pv">
_“Ahaṃ rājagahe ramme,_  
_ramaṇīye giribbaje;_  
_Migaluddo pure āsiṃ,_  
_luddo cāsimasaññato._  

</Paragraph>

</Division>
<Division number="492" book="pv" bookNo="19" page="178">

<Paragraph number="533" type="verses" book="pv">
_Tassa me sahāyo suhadayo,_  
_Saddho āsi upāsako;_  
_Tassa kulupako bhikkhu,_  
_Āsi gotamasāvako;_  
_Sopi maṃ anukampanto,_  
_Nivāresi punappunaṃ._  

</Paragraph>

</Division>
<Division number="493" book="pv" bookNo="19" page="178">

<Paragraph number="534" type="verses" book="pv">
_‘Mākāsi pāpakaṃ kammaṃ,_  
_Mā tāta duggatiṃ agā;_  
_Sace icchasi pecca sukhaṃ,_  
_Virama pāṇavadhā asaṃyamā’._  

</Paragraph>

</Division>
<Division number="494" book="pv" bookNo="19" page="178">

<Paragraph number="535" type="verses" book="pv">
_Tassāhaṃ vacanaṃ sutvā,_  
_Sukhakāmassa hitānukampino;_  
_Nākāsiṃ sakalānusāsaniṃ,_  
_Cirapāpābhirato abuddhimā._  

</Paragraph>

</Division>
<Division number="495" book="pv" bookNo="19" page="178">

<Paragraph number="536" type="verses" book="pv">
_So maṃ puna bhūrisumedhaso,_  
_Anukampāya saṃyame nivesayi;_  
_‘Sace divā hanasi pāṇino,_  
_Atha te rattiṃ bhavatu saṃyamo’._  

</Paragraph>

</Division>
<Division number="496" book="pv" bookNo="19" page="178">

<Paragraph number="537" type="verses" book="pv">
_Svāhaṃ divā hanitvā pāṇino,_  
_Virato rattimahosi saññato;_  
_Rattāhaṃ paricāremi,_  
_Divā khajjāmi duggato._  

</Paragraph>

</Division>
<Division number="497" book="pv" bookNo="19" page="178">

<Paragraph number="538" type="verses" book="pv">
_Tassa kammassa kusalassa,_  
_Anubhomi rattiṃ amānusiṃ;_  
_Divā paṭihatāva kukkurā,_  
_Upadhāvanti samantā khādituṃ._  

</Paragraph>

</Division>
<Division number="498" book="pv" bookNo="19" page="178">

<Paragraph number="539" type="verses" book="pv">
_Ye ca te satatānuyogino,_  
_Dhuvaṃ payuttā sugatassa sāsane;_  
_Maññāmi te amatameva kevalaṃ,_  
_Adhigacchanti padaṃ asaṅkhatan”ti._  


</Paragraph>
</Division>

<Paragraph number="540" type="center" book="pv">
Dutiyamigaluddakapetavatthu aṭṭhamaṃ.
</Paragraph>
