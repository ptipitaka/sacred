<Division number="368" book="pv">

<Paragraph number="397" type="verses" book="pv">
_Ahu rājā brahmadatto,_  
_pañcālānaṃ rathesabho;_  
_Ahorattānamaccayā,_  
_rājā kālamakrubbatha._  

</Paragraph>

</Division>
<Division number="369" book="pv">

<Paragraph number="398" type="verses" book="pv">
_Tassa āḷāhanaṃ gantvā,_  
_bhariyā kandati ubbarī;_  
_Brahmadattaṃ apassantī,_  
_brahmadattāti kandati._  

</Paragraph>

</Division>
<Division number="370" book="pv">

<Paragraph number="399" type="verses" book="pv">
_Isi ca tattha āgacchi,_  
_sampannacaraṇo muni;_  
_So ca tattha apucchittha,_  
_ye tattha susamāgatā._  

</Paragraph>

</Division>
<Division number="371" book="pv">

<Paragraph number="400" type="verses" book="pv">
_“Kassa idaṃ āḷāhanaṃ,_  
_nānāgandhasameritaṃ;_  
_Kassāyaṃ kandati bhariyā,_  
_ito dūragataṃ patiṃ;_  
_Brahmadattaṃ apassantī,_  
_‘brahmadattā’ti kandati”._  

</Paragraph>

</Division>
<Division number="372" book="pv">

<Paragraph number="401" type="verses" book="pv">
_Te ca tattha viyākaṃsu,_  
_ye tattha susamāgatā;_  
_“Brahmadattassa bhadante,_  
_brahmadattassa mārisa._  

</Paragraph>

</Division>
<Division number="373" book="pv">

<Paragraph number="402" type="verses" book="pv">
_Tassa idaṃ āḷāhanaṃ,_  
_nānāgandhasameritaṃ;_  
_Tassāyaṃ kandati bhariyā,_  
_ito dūragataṃ patiṃ;_  
_Brahmadattaṃ apassantī,_  
_‘brahmadattā’ti kandati”._  

</Paragraph>

</Division>
<Division number="374" book="pv">

<Paragraph number="403" type="verses" book="pv">
_“Chaḷāsītisahassāni,_  
_brahmadattassanāmakā;_  
_Imasmiṃ āḷāhane daḍḍhā,_  
_tesaṃ kamanusocasī”ti._  

</Paragraph>

</Division>
<Division number="375" book="pv">

<Paragraph number="404" type="verses" book="pv">
_“Yo rājā cūḷanīputto,_  
_pañcālānaṃ rathesabho;_  
_Taṃ bhante anusocāmi,_  
_bhattāraṃ sabbakāmadan”ti._  

</Paragraph>

</Division>
<Division number="376" book="pv">

<Paragraph number="405" type="verses" book="pv">
_“Sabbe vāhesuṃ rājāno,_  
_brahmadattassanāmakā;_  
_Sabbeva cūḷanīputtā,_  
_pañcālānaṃ rathesabhā._  

</Paragraph>

</Division>
<Division number="377" book="pv">

<Paragraph number="406" type="verses" book="pv">
_Sabbesaṃ anupubbena,_  
_mahesittamakārayi;_  
_Kasmā purimake hitvā,_  
_pacchimaṃ anusocasī”ti._  

</Paragraph>

</Division>
<Division number="378" book="pv">

<Paragraph number="407" type="verses" book="pv">
_“Ātume itthibhūtāya,_  
_dīgharattāya mārisa;_  
_Yassā me itthibhūtāya,_  
_saṃsāre bahubhāsasī”ti._  

</Paragraph>

</Division>
<Division number="379" book="pv">

<Paragraph number="408" type="verses" book="pv">
_“Ahu itthī ahu puriso,_  
_pasuyonimpi āgamā;_  
_Evametaṃ atītānaṃ,_  
_pariyanto na dissatī”ti._  

</Paragraph>

</Division>
<Division number="380" book="pv">

<Paragraph number="409" type="verses" book="pv">
_“Ādittaṃ vata maṃ santaṃ,_  
_ghatasittaṃva pāvakaṃ;_  
_Vārinā viya osiñcaṃ,_  
_sabbaṃ nibbāpaye daraṃ._  

</Paragraph>

</Division>
<Division number="381" book="pv">

<Paragraph number="410" type="verses" book="pv">
_Abbahī vata me sallaṃ,_  
_sokaṃ hadayanissitaṃ;_  
_Yo me sokaparetāya,_  
_patisokaṃ apānudi._  

</Paragraph>

</Division>
<Division number="382" book="pv">

<Paragraph number="411" type="verses" book="pv">
_Sāhaṃ abbūḷhasallāsmi,_  
_sītibhūtāsmi nibbutā;_  
_Na socāmi na rodāmi,_  
_tava sutvā mahāmunī”ti._  

</Paragraph>

</Division>
<Division number="383" book="pv">

<Paragraph number="412" type="verses" book="pv">
_Tassa taṃ vacanaṃ sutvā,_  
_samaṇassa subhāsitaṃ;_  
_Pattacīvaramādāya,_  
_pabbaji anagāriyaṃ._  

</Paragraph>

</Division>
<Division number="384" book="pv">

<Paragraph number="413" type="verses" book="pv">
_Sā ca pabbajitā santā,_  
_agārasmā anagāriyaṃ;_  
_Mettacittaṃ abhāvesi,_  
_brahmalokūpapattiyā._  

</Paragraph>

</Division>
<Division number="385" book="pv">

<Paragraph number="414" type="verses" book="pv">
_Gāmā gāmaṃ vicarantī,_  
_nigame rājadhāniyo;_  
_Uruvelā nāma so gāmo,_  
_yattha kālamakrubbatha._  

</Paragraph>

</Division>
<Division number="386" book="pv">

<Paragraph number="415" type="verses" book="pv">
_Mettacittaṃ ābhāvetvā,_  
_brahmalokūpapattiyā;_  
_Itthicittaṃ virājetvā,_  
_brahmalokūpagā ahūti._  


</Paragraph>
</Division>

<Paragraph number="416" type="center" book="pv">
Ubbaripetavatthu terasamaṃ.



</Paragraph>

<Paragraph number="417" type="center" book="pv">
Ubbarivaggo dutiyo.



</Paragraph>

<Paragraph number="418" type="center" book="pv">
Tassuddānaṃ


</Paragraph>

_Mocakaṃ mātā mattā ca,_  
_Nandā kuṇḍalīnā ghaṭo;_  
_Dve seṭṭhī tunnavāyo ca,_  
_Uttara suttakaṇṇa ubbarīti._
</Paragraph>
