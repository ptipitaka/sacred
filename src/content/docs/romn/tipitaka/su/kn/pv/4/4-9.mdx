<Division number="774" book="pv">

<Paragraph number="830" type="verses" book="pv">
_“Gūthakūpato uggantvā,_  
_kā nu dīnā patiṭṭhasi;_  
_Nissaṃsayaṃ pāpakammantā,_  
_kiṃ nu saddahase tuvan”ti._  

</Paragraph>

</Division>
<Division number="775" book="pv">

<Paragraph number="831" type="verses" book="pv">
_“Ahaṃ bhadante petīmhi,_  
_duggatā yamalokikā;_  
_Pāpakammaṃ karitvāna,_  
_petalokaṃ ito gatā”ti._  

</Paragraph>

</Division>
<Division number="776" book="pv">

<Paragraph number="832" type="verses" book="pv">
_“Kiṃ nu kāyena vācāya,_  
_manasā dukkaṭaṃ kataṃ;_  
_Kissa kammavipākena,_  
_idaṃ dukkhaṃ nigacchasī”ti._  

</Paragraph>

</Division>
<Division number="777" book="pv">

<Paragraph number="833" type="verses" book="pv">
_“Ahu āvāsiko mayhaṃ,_  
_issukī kulamaccharī;_  
_Ajjhāsito mayhaṃ ghare,_  
_kadariyo paribhāsako._  

</Paragraph>

</Division>
<Division number="778" book="pv">

<Paragraph number="834" type="verses" book="pv">
_Tassāhaṃ vacanaṃ sutvā,_  
_bhikkhavo paribhāsisaṃ;_  
_Tassa kammavipākena,_  
_petalokaṃ ito gatā”ti._  

</Paragraph>

</Division>
<Division number="779" book="pv">

<Paragraph number="835" type="verses" book="pv">
_“Amitto mittavaṇṇena,_  
_yo te āsi kulūpako;_  
_Kāyassa bhedā duppañño,_  
_kiṃ nu pecca gatiṃ gato”ti._  

</Paragraph>

</Division>
<Division number="780" book="pv">

<Paragraph number="836" type="verses" book="pv">
_“Tassevāhaṃ pāpakammassa,_  
_sīse tiṭṭhāmi matthake;_  
_So ca paravisayaṃ patto,_  
_mameva paricārako._  

</Paragraph>

</Division>
<Division number="781" book="pv">

<Paragraph number="837" type="verses" book="pv">
_Yaṃ bhadante hadantaññe,_  
_etaṃ me hoti bhojanaṃ;_  
_Ahañca kho yaṃ hadāmi,_  
_etaṃ so upajīvatī”ti._  


</Paragraph>
</Division>

<Paragraph number="838" type="center" book="pv">
Gūthakhādakapetivatthu navamaṃ.
</Paragraph>
