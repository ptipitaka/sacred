<Paragraph number="42" type="verses" book="pv">
_“Naggā dubbaṇṇarūpāsi,_  
_duggandhā pūti vāyasi;_  
_Makkhikāhi parikiṇṇā,_  
_kā nu tvaṃ idha tiṭṭhasī”ti._  

</Paragraph>

</Division>
<Division number="36" book="pv" bookNo="19" page="131">

<Paragraph number="43" type="verses" book="pv">
_“Ahaṃ bhadante petīmhi,_  
_duggatā yamalokikā;_  
_Pāpakammaṃ karitvāna,_  
_petalokaṃ ito gatā._  

</Paragraph>

</Division>
<Division number="37" book="pv" bookNo="19" page="131">

<Paragraph number="44" type="verses" book="pv">
_Kālena satta puttāni,_  
_sāyaṃ satta punāpare;_  
_Vijāyitvāna khādāmi,_  
_tepi nā honti me alaṃ._  

</Paragraph>

</Division>
<Division number="38" book="pv" bookNo="19" page="131">

<Paragraph number="45" type="verses" book="pv">
_Pariḍayhati dhūmāyati,_  
_khudāya hadayaṃ mama;_  
_Nibbutiṃ nādhigacchāmi,_  
_aggidaḍḍhāva ātape”ti._  

</Paragraph>

</Division>
<Division number="39" book="pv" bookNo="19" page="131">

<Paragraph number="46" type="verses" book="pv">
_“Kiṃ nu kāyena vācāya,_  
_manasā dukkaṭaṃ kataṃ;_  
_Kissa kammavipākena,_  
_puttamaṃsāni khādasī”ti._  

</Paragraph>

</Division>
<Division number="40" book="pv" bookNo="19" page="131">

<Paragraph number="47" type="verses" book="pv">
_“Ahū mayhaṃ duve puttā,_  
_ubho sampattayobbanā;_  
_Sāhaṃ puttabalūpetā,_  
_sāmikaṃ atimaññisaṃ._  

</Paragraph>

</Division>
<Division number="41" book="pv" bookNo="19" page="131">

<Paragraph number="48" type="verses" book="pv">
_Tato me sāmiko kuddho,_  
_sapatiṃ mayhamānayi;_  
_Sā ca gabbhaṃ alabhittha,_  
_tassā pāpaṃ acetayiṃ._  

</Paragraph>

</Division>
<Division number="42" book="pv" bookNo="19" page="132">

<Paragraph number="49" type="verses" book="pv">
_Sāhaṃ paduṭṭhamanasā,_  
_akariṃ gabbhapātanaṃ;_  
_Tassā temāsiko gabbho,_  
_pūtilohitako pati._  

</Paragraph>

</Division>
<Division number="43" book="pv" bookNo="19" page="132">

<Paragraph number="50" type="verses" book="pv">
_Tadassā mātā kupitā,_  
_mayhaṃ ñātī samānayi;_  
_Sapathañca maṃ kāresi,_  
_paribhāsāpayī ca maṃ._  

</Paragraph>

</Division>
<Division number="44" book="pv" bookNo="19" page="132">

<Paragraph number="51" type="verses" book="pv">
_Sāhaṃ ghorañca sapathaṃ,_  
_musāvādaṃ abhāsisaṃ;_  
_‘Puttamaṃsāni khādāmi,_  
_sace taṃ pakataṃ mayā’._  

</Paragraph>

</Division>
<Division number="45" book="pv" bookNo="19" page="132">

<Paragraph number="52" type="verses" book="pv">
_Tassa kammassa vipākena,_  
_musāvādassa cūbhayaṃ;_  
_Puttamaṃsāni khādāmi,_  
_pubbalohitamakkhitā”ti._  


</Paragraph>
</Division>

<Paragraph number="53" type="center" book="pv">
Sattaputtakhādapetivatthu sattamaṃ.
</Paragraph>
