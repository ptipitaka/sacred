<Division number="478" book="pv">

<Paragraph number="518" type="verses" book="pv">
_“Naranāripurakkhato yuvā,_  
_Rajanīyehi kāmaguṇehi sobhasi;_  
_Divasaṃ anubhosi kāraṇaṃ,_  
_Kimakāsi purimāya jātiyā”ti._  

</Paragraph>

</Division>
<Division number="479" book="pv">

<Paragraph number="519" type="verses" book="pv">
_“Ahaṃ rājagahe ramme,_  
_ramaṇīye giribbaje;_  
_Migaluddo pure āsiṃ,_  
_lohitapāṇi dāruṇo._  

</Paragraph>

</Division>
<Division number="480" book="pv">

<Paragraph number="520" type="verses" book="pv">
_Avirodhakaresu pāṇisu,_  
_Puthusattesu paduṭṭhamānaso;_  
_Vicariṃ atidāruṇo sadā,_  
_Parahiṃsāya rato asaññato._  

</Paragraph>

</Division>
<Division number="481" book="pv">

<Paragraph number="521" type="verses" book="pv">
_Tassa me sahāyo suhadayo,_  
_Saddho āsi upāsako;_  
_Sopi maṃ anukampanto,_  
_Nivāresi punappunaṃ._  

</Paragraph>

</Division>
<Division number="482" book="pv">

<Paragraph number="522" type="verses" book="pv">
_‘Mākāsi pāpakaṃ kammaṃ,_  
_Mā tāta duggatiṃ agā;_  
_Sace icchasi pecca sukhaṃ,_  
_Virama pāṇavadhā asaṃyamā’._  

</Paragraph>

</Division>
<Division number="483" book="pv">

<Paragraph number="523" type="verses" book="pv">
_Tassāhaṃ vacanaṃ sutvā,_  
_Sukhakāmassa hitānukampino;_  
_Nākāsiṃ sakalānusāsaniṃ,_  
_Cirapāpābhirato abuddhimā._  

</Paragraph>

</Division>
<Division number="484" book="pv">

<Paragraph number="524" type="verses" book="pv">
_So maṃ puna bhūrisumedhaso,_  
_Anukampāya saṃyame nivesayi;_  
_‘Sace divā hanasi pāṇino,_  
_Atha te rattiṃ bhavatu saṃyamo’._  

</Paragraph>

</Division>
<Division number="485" book="pv">

<Paragraph number="525" type="verses" book="pv">
_Svāhaṃ divā hanitvā pāṇino,_  
_Virato rattimahosi saññato;_  
_Rattāhaṃ paricāremi,_  
_Divā khajjāmi duggato._  

</Paragraph>

</Division>
<Division number="486" book="pv">

<Paragraph number="526" type="verses" book="pv">
_Tassa kammassa kusalassa,_  
_Anubhomi rattiṃ amānusiṃ;_  
_Divā paṭihatāva kukkurā,_  
_Upadhāvanti samantā khādituṃ._  

</Paragraph>

</Division>
<Division number="487" book="pv">

<Paragraph number="527" type="verses" book="pv">
_Ye ca te satatānuyogino,_  
_Dhuvaṃ payuttā sugatassa sāsane;_  
_Maññāmi te amatameva kevalaṃ,_  
_Adhigacchanti padaṃ asaṅkhatan”ti._  


</Paragraph>
</Division>

<Paragraph number="528" type="center" book="pv">
Migaluddakapetavatthu sattamaṃ.
</Paragraph>
