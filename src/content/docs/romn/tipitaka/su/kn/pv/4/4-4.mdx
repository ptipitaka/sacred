<Paragraph number="765" type="verses" book="pv">
_“Uṭṭhehi revate supāpadhamme,_  
_Apārutadvāre adānasīle;_  
_Nessāma taṃ yattha thunanti duggatā,_  
_Samappitā nerayikā dukhenā”ti._  

</Paragraph>

</Division>
<Division number="715" book="pv" bookNo="19" page="206">

<Paragraph number="766" type="verses" book="pv">
_Icceva vatvāna yamassa dūtā,_  
_Te dve yakkhā lohitakkhā brahantā;_  
_Paccekabāhāsu gahetvā revataṃ,_  
_Pakkāmayuṃ devagaṇassa santike._  

</Paragraph>

</Division>
<Division number="716" book="pv" bookNo="19" page="207">

<Paragraph number="767" type="verses" book="pv">
_“Ādiccavaṇṇaṃ ruciraṃ pabhassaraṃ,_  
_Byamhaṃ subhaṃ kañcanajālachannaṃ;_  
_Kassetamākiṇṇajanaṃ vimānaṃ,_  
_Suriyassa raṃsīriva jotamānaṃ._  

</Paragraph>

</Division>
<Division number="717" book="pv" bookNo="19" page="207">

<Paragraph number="768" type="verses" book="pv">
_Nārīgaṇā candanasāralittā,_  
_Ubhato vimānaṃ upasobhayanti;_  
_Taṃ dissati suriyasamānavaṇṇaṃ,_  
_Ko modati saggapatto vimāne”ti._  

</Paragraph>

</Division>
<Division number="718" book="pv" bookNo="19" page="207">

<Paragraph number="769" type="verses" book="pv">
_“Bārāṇasiyaṃ nandiyo nāmāsi,_  
_Upāsako amaccharī dānapati vadaññū;_  
_Tassetamākiṇṇajanaṃ vimānaṃ,_  
_Suriyassa raṃsīriva jotamānaṃ._  

</Paragraph>

</Division>
<Division number="719" book="pv" bookNo="19" page="207">

_Nārīgaṇā candanasāralittā,_  
_Ubhato vimānaṃ upasobhayanti;_  
_Taṃ dissati suriyasamānavaṇṇaṃ,_  
_So modati saggapatto vimāne”ti._  

</Paragraph>

</Division>
<Division number="720" book="pv" bookNo="19" page="207">

<Paragraph number="771" type="verses" book="pv">
_“Nandiyassāhaṃ bhariyā,_  
_Agārinī sabbakulassa issarā;_  
_Bhattu vimāne ramissāmi dānahaṃ,_  
_Na patthaye nirayadassanāyā”ti._  

</Paragraph>

</Division>
<Division number="721" book="pv" bookNo="19" page="207">

<Paragraph number="772" type="verses" book="pv">
_“Eso te nirayo supāpadhamme,_  
_Puññaṃ tayā akataṃ jīvaloke;_  
_Na hi maccharī rosako pāpadhammo,_  
_Saggūpagānaṃ labhati sahabyatan”ti._  

</Paragraph>

</Division>
<Division number="722" book="pv" bookNo="19" page="207">

<Paragraph number="773" type="verses" book="pv">
_“Kiṃ nu gūthañca muttañca,_  
_asucī paṭidissati;_  
_Duggandhaṃ kimidaṃ mīḷhaṃ,_  
_kimetaṃ upavāyatī”ti._  

</Paragraph>

</Division>
<Division number="723" book="pv" bookNo="19" page="207">

<Paragraph number="774" type="verses" book="pv">
_“Esa saṃsavako nāma,_  
_gambhīro sataporiso;_  
_Yattha vassasahassāni,_  
_tuvaṃ paccasi revate”ti._  

</Paragraph>

</Division>
<Division number="724" book="pv" bookNo="19" page="208">

<Paragraph number="775" type="verses" book="pv">
_“Kiṃ nu kāyena vācāya,_  
_manasā dukkaṭaṃ kataṃ;_  
_Kena saṃsavako laddho,_  
_gambhīro sataporiso”ti._  

</Paragraph>

</Division>
<Division number="725" book="pv" bookNo="19" page="208">

<Paragraph number="776" type="verses" book="pv">
_“Samaṇe brāhmaṇe cāpi,_  
_aññe vāpi vanibbake;_  
_Musāvādena vañcesi,_  
_taṃ pāpaṃ pakataṃ tayā._  

</Paragraph>

</Division>
<Division number="726" book="pv" bookNo="19" page="208">

<Paragraph number="777" type="verses" book="pv">
_Tena saṃsavako laddho,_  
_gambhīro sataporiso;_  
_Tattha vassasahassāni,_  
_tuvaṃ paccasi revate._  

</Paragraph>

</Division>
<Division number="727" book="pv" bookNo="19" page="208">

<Paragraph number="778" type="verses" book="pv">
_Hatthepi chindanti athopi pāde,_  
_Kaṇṇepi chindanti athopi nāsaṃ;_  
_Athopi kākoḷagaṇā samecca,_  
_Saṅgamma khādanti viphandamānan”ti._  

</Paragraph>

</Division>
<Division number="728" book="pv" bookNo="19" page="208">

<Paragraph number="779" type="verses" book="pv">
_“Sādhu kho maṃ paṭinetha,_  
_kāhāmi kusalaṃ bahuṃ;_  
_Dānena samacariyāya,_  
_saṃyamena damena ca;_  
_Yaṃ katvā sukhitā honti,_  
_na ca pacchānutappare”ti._  

</Paragraph>

</Division>
<Division number="729" book="pv" bookNo="19" page="208">

<Paragraph number="780" type="verses" book="pv">
_“Pure tuvaṃ pamajjitvā,_  
_idāni paridevasi;_  
_Sayaṃ katānaṃ kammānaṃ,_  
_vipākaṃ anubhossasī”ti._  

</Paragraph>

</Division>
<Division number="730" book="pv" bookNo="19" page="208">

<Paragraph number="781" type="verses" book="pv">
_“Ko devalokato manussalokaṃ,_  
_Gantvāna puṭṭho me evaṃ vadeyya;_  
_‘Nikkhittadaṇḍesu dadātha dānaṃ,_  
_Acchādanaṃ seyya mathannapānaṃ;_  
_Na hi maccharī rosako pāpadhammo,_  
_Saggūpagānaṃ labhati sahabyataṃ’._  

</Paragraph>

</Division>
<Division number="731" book="pv" bookNo="19" page="208">

<Paragraph number="782" type="verses" book="pv">
_Sāhaṃ nūna ito gantvā,_  
_yoniṃ laddhāna mānusiṃ;_  
_Vadaññū sīlasampannā,_  
_kāhāmi kusalaṃ bahuṃ;_  
_Dānena samacariyāya,_  
_saṃyamena damena ca._  

</Paragraph>

</Division>
<Division number="732" book="pv" bookNo="19" page="208">

<Paragraph number="783" type="verses" book="pv">
_Ārāmāni ca ropissaṃ,_  
_dugge saṅkamanāni ca;_  
_Papañca udapānañca,_  
_vippasannena cetasā._  

</Paragraph>

</Division>
<Division number="733" book="pv" bookNo="19" page="209">

<Paragraph number="784" type="verses" book="pv">
_Cātuddasiṃ pañcadasiṃ,_  
_yā ca pakkhassa aṭṭhamī;_  
_Pāṭihāriyapakkhañca,_  
_aṭṭhaṅgasusamāgataṃ._  

</Paragraph>

</Division>
<Division number="734" book="pv" bookNo="19" page="209">

<Paragraph number="785" type="verses" book="pv">
_Uposathaṃ upavasissaṃ,_  
_sadā sīlesu saṃvutā;_  
_Na ca dāne pamajjissaṃ,_  
_sāmaṃ diṭṭhamidaṃ mayā”ti._  

</Paragraph>

</Division>
<Division number="735" book="pv" bookNo="19" page="209">

<Paragraph number="786" type="verses" book="pv">
_Iccevaṃ vippalapantiṃ,_  
_phandamānaṃ tato tato;_  
_Khipiṃsu niraye ghore,_  
_uddhampādaṃ avaṃsiraṃ._  

</Paragraph>

</Division>
<Division number="736" book="pv" bookNo="19" page="209">

<Paragraph number="787" type="verses" book="pv">
_“Ahaṃ pure maccharinī ahosiṃ,_  
_Paribhāsikā samaṇabrāhmaṇānaṃ;_  
_Vitathena ca sāmikaṃ vañcayitvā,_  
_Paccāmahaṃ niraye ghorarūpe”ti._  


</Paragraph>
