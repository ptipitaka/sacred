<Division number="95" book="pv">

<Paragraph number="111" type="verses" book="pv">
_“Naggā dubbaṇṇarūpāsi,_  
_kisā dhamanisanthatā;_  
_Upphāsulike kisike,_  
_kā nu tvaṃ idha tiṭṭhasī”ti._  

</Paragraph>

</Division>
<Division number="96" book="pv">

<Paragraph number="112" type="verses" book="pv">
_“Ahaṃ bhadante petīmhi,_  
_duggatā yamalokikā;_  
_Pāpakammaṃ karitvāna,_  
_petalokaṃ ito gatā”ti._  

</Paragraph>

</Division>
<Division number="97" book="pv">

<Paragraph number="113" type="verses" book="pv">
_“Kiṃ nu kāyena vācāya,_  
_manasā dukkaṭaṃ kataṃ;_  
_Kissa kammavipākena,_  
_petalokaṃ ito gatā”ti._  

</Paragraph>

</Division>
<Division number="98" book="pv">

<Paragraph number="114" type="verses" book="pv">
_“Anukampakā mayhaṃ nāhesuṃ bhante,_  
_Pitā ca mātā athavāpi ñātakā;_  
_Ye maṃ niyojeyyuṃ dadāhi dānaṃ,_  
_Pasannacittā samaṇabrāhmaṇānaṃ._  

</Paragraph>

</Division>
<Division number="99" book="pv">

<Paragraph number="115" type="verses" book="pv">
_Ito ahaṃ vassasatāni pañca,_  
_Yaṃ evarūpā vicarāmi naggā;_  
_Khudāya taṇhāya ca khajjamānā,_  
_Pāpassa kammassa phalaṃ mamedaṃ._  

</Paragraph>

</Division>
<Division number="100" book="pv">

<Paragraph number="116" type="verses" book="pv">
_Vandāmi taṃ ayya pasannacittā,_  
_Anukampa maṃ vīra mahānubhāva;_  
_Datvā ca me ādisa yaṃ hi kiñci,_  
_Mocehi maṃ duggatiyā bhadante”ti._  

</Paragraph>

</Division>
<Division number="101" book="pv">

<Paragraph number="117" type="verses" book="pv">
_Sādhūti so paṭissutvā,_  
_sāriputtonukampako;_  
_Bhikkhūnaṃ ālopaṃ datvā,_  
_pāṇimattañca coḷakaṃ;_  
_Thālakassa ca pānīyaṃ,_  
_tassā dakkhiṇamādisi._  

</Paragraph>

</Division>
<Division number="102" book="pv">

<Paragraph number="118" type="verses" book="pv">
_Samanantarānuddiṭṭhe,_  
_vipāko udapajjatha;_  
_Bhojanacchādanapānīyaṃ,_  
_dakkhiṇāya idaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="103" book="pv">

<Paragraph number="119" type="verses" book="pv">
_Tato suddhā sucivasanā,_  
_kāsikuttamadhārinī;_  
_Vicittavatthābharaṇā,_  
_sāriputtaṃ upasaṅkami._  

</Paragraph>

</Division>
<Division number="104" book="pv">

<Paragraph number="120" type="verses" book="pv">
_“Abhikkantena vaṇṇena,_  
_yā tvaṃ tiṭṭhasi devate;_  
_Obhāsentī disā sabbā,_  
_osadhī viya tārakā._  

</Paragraph>

</Division>
<Division number="105" book="pv">

<Paragraph number="121" type="verses" book="pv">
_Kena te tādiso vaṇṇo,_  
_kena te idha mijjhati;_  
_Uppajjanti ca te bhogā,_  
_ye keci manaso piyā._  

</Paragraph>

</Division>
<Division number="106" book="pv">

<Paragraph number="122" type="verses" book="pv">
_Pucchāmi taṃ devi mahānubhāve,_  
_Manussabhūtā kimakāsi puññaṃ;_  
_Kenāsi evaṃ jalitānubhāvā,_  
_Vaṇṇo ca te sabbadisā pabhāsatī”ti._  

</Paragraph>

</Division>
<Division number="107" book="pv">

<Paragraph number="123" type="verses" book="pv">
_“Uppaṇḍukiṃ kisaṃ chātaṃ,_  
_naggaṃ sampatitacchaviṃ;_  
_Muni kāruṇiko loke,_  
_taṃ maṃ addakkhi duggataṃ._  

</Paragraph>

</Division>
<Division number="108" book="pv">

<Paragraph number="124" type="verses" book="pv">
_Bhikkhūnaṃ ālopaṃ datvā,_  
_pāṇimattañca coḷakaṃ;_  
_Thālakassa ca pānīyaṃ,_  
_mama dakkhiṇamādisi._  

</Paragraph>

</Division>
<Division number="109" book="pv">

<Paragraph number="125" type="verses" book="pv">
_Ālopassa phalaṃ passa,_  
_bhattaṃ vassasataṃ dasa;_  
_Bhuñjāmi kāmakāminī,_  
_anekarasabyañjanaṃ._  

</Paragraph>

</Division>
<Division number="110" book="pv">

<Paragraph number="126" type="verses" book="pv">
_Pāṇimattassa coḷassa,_  
_vipākaṃ passa yādisaṃ;_  
_Yāvatā nandarājassa,_  
_vijitasmiṃ paṭicchadā._  

</Paragraph>

</Division>
<Division number="111" book="pv">

<Paragraph number="127" type="verses" book="pv">
_Tato bahutarā bhante,_  
_vatthānacchādanāni me;_  
_Koseyyakambalīyāni,_  
_khomakappāsikāni ca._  

</Paragraph>

</Division>
<Division number="112" book="pv">

<Paragraph number="128" type="verses" book="pv">
_Vipulā ca mahagghā ca,_  
_tepākāsevalambare;_  
_Sāhaṃ taṃ paridahāmi,_  
_yaṃ yaṃ hi manaso piyaṃ._  

</Paragraph>

</Division>
<Division number="113" book="pv">

<Paragraph number="129" type="verses" book="pv">
_Thālakassa ca pānīyaṃ,_  
_vipākaṃ passa yādisaṃ;_  
_Gambhīrā caturassā ca,_  
_pokkharañño sunimmitā._  

</Paragraph>

</Division>
<Division number="114" book="pv">

<Paragraph number="130" type="verses" book="pv">
_Setodakā suppatitthā,_  
_sītā appaṭigandhiyā;_  
_Padumuppalasañchannā,_  
_vārikiñjakkhapūritā._  

</Paragraph>

</Division>
<Division number="115" book="pv">

<Paragraph number="131" type="verses" book="pv">
_Sāhaṃ ramāmi kīḷāmi,_  
_modāmi akutobhayā;_  
_Muniṃ kāruṇikaṃ loke,_  
_bhante vanditumāgatā”ti._  


</Paragraph>
</Division>

<Paragraph number="132" type="center" book="pv">
Saṃsāramocakapetivatthu paṭhamaṃ.
</Paragraph>
