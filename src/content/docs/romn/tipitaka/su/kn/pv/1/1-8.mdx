<Paragraph number="54" type="verses" book="pv">
_“Kiṃ nu ummattarūpova,_  
_lāyitvā haritaṃ tiṇaṃ;_  
_Khāda khādāti lapasi,_  
_gatasattaṃ jaraggavaṃ._  

</Paragraph>

</Division>
<Division number="47" book="pv" bookNo="19" page="132">

<Paragraph number="55" type="verses" book="pv">
_Na hi annena pānena,_  
_mato goṇo samuṭṭhahe;_  
_Tvaṃsi bālo ca dummedho,_  
_yathā taññova dummatī”ti._  

</Paragraph>

</Division>
<Division number="48" book="pv" bookNo="19" page="132">

<Paragraph number="56" type="verses" book="pv">
_“Ime pādā idaṃ sīsaṃ,_  
_ayaṃ kāyo savāladhi;_  
_Nettā tatheva tiṭṭhanti,_  
_ayaṃ goṇo samuṭṭhahe._  

</Paragraph>

</Division>
<Division number="49" book="pv" bookNo="19" page="132">

<Paragraph number="57" type="verses" book="pv">
_Nāyyakassa hatthapādā,_  
_kāyo sīsañca dissati;_  
_Rudaṃ mattikathūpasmiṃ,_  
_nanu tvaññeva dummatī”ti._  

</Paragraph>

</Division>
<Division number="50" book="pv" bookNo="19" page="132">

<Paragraph number="58" type="verses" book="pv">
_“Ādittaṃ vata maṃ santaṃ,_  
_ghatasittaṃva pāvakaṃ;_  
_Vārinā viya osiñcaṃ,_  
_sabbaṃ nibbāpaye daraṃ._  

</Paragraph>

</Division>
<Division number="51" book="pv" bookNo="19" page="132">

<Paragraph number="59" type="verses" book="pv">
_Abbahī vata me sallaṃ,_  
_sokaṃ hadayanissitaṃ;_  
_Yo me sokaparetassa,_  
_pitusokaṃ apānudi._  

</Paragraph>

</Division>
<Division number="52" book="pv" bookNo="19" page="132">

<Paragraph number="60" type="verses" book="pv">
_Svāhaṃ abbūḷhasallosmi,_  
_sītibhūtosmi nibbuto;_  
_Na socāmi na rodāmi,_  
_tava sutvāna māṇava”._  

</Paragraph>

</Division>
<Division number="53" book="pv" bookNo="19" page="133">

<Paragraph number="61" type="verses" book="pv">
_Evaṃ karonti sappaññā,_  
_ye honti anukampakā;_  
_Vinivattayanti sokamhā,_  
_sujāto pitaraṃ yathāti._  


</Paragraph>
</Division>

Goṇapetavatthu aṭṭhamaṃ.
