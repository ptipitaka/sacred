<Division number="26" book="pv">

<Paragraph number="32" type="verses" book="pv">
_“Naggā dubbaṇṇarūpāsi,_  
_duggandhā pūti vāyasi;_  
_Makkhikāhi parikiṇṇā,_  
_kā nu tvaṃ idha tiṭṭhasī”ti._  

</Paragraph>

</Division>
<Division number="27" book="pv">

<Paragraph number="33" type="verses" book="pv">
_“Ahaṃ bhadante petīmhi,_  
_duggatā yamalokikā;_  
_Pāpakammaṃ karitvāna,_  
_petalokaṃ ito gatā._  

</Paragraph>

</Division>
<Division number="28" book="pv">

<Paragraph number="34" type="verses" book="pv">
_Kālena pañca puttāni,_  
_sāyaṃ pañca punāpare;_  
_Vijāyitvāna khādāmi,_  
_tepi nā honti me alaṃ._  

</Paragraph>

</Division>
<Division number="29" book="pv">

<Paragraph number="35" type="verses" book="pv">
_Pariḍayhati dhūmāyati,_  
_khudāya hadayaṃ mama;_  
_Pānīyaṃ na labhe pātuṃ,_  
_passa maṃ byasanaṃ gatan”ti._  

</Paragraph>

</Division>
<Division number="30" book="pv">

<Paragraph number="36" type="verses" book="pv">
_“Kiṃ nu kāyena vācāya,_  
_manasā dukkaṭaṃ kataṃ;_  
_Kissa kammavipākena,_  
_puttamaṃsāni khādasī”ti._  

</Paragraph>

</Division>
<Division number="31" book="pv">

<Paragraph number="37" type="verses" book="pv">
_“Sapatī me gabbhinī āsi,_  
_tassā pāpaṃ acetayiṃ;_  
_Sāhaṃ paduṭṭhamanasā,_  
_akariṃ gabbhapātanaṃ._  

</Paragraph>

</Division>
<Division number="32" book="pv">

<Paragraph number="38" type="verses" book="pv">
_Tassā dvemāsiko gabbho,_  
_lohitaññeva pagghari;_  
_Tadassā mātā kupitā,_  
_mayhaṃ ñātī samānayi;_  
_Sapathañca maṃ kāresi,_  
_paribhāsāpayī ca maṃ._  

</Paragraph>

</Division>
<Division number="33" book="pv">

<Paragraph number="39" type="verses" book="pv">
_Sāhaṃ ghorañca sapathaṃ,_  
_musāvādaṃ abhāsisaṃ;_  
_Puttamaṃsāni khādāmi,_  
_sace taṃ pakataṃ mayā._  

</Paragraph>

</Division>
<Division number="34" book="pv">

<Paragraph number="40" type="verses" book="pv">
_Tassa kammassa vipākena,_  
_musāvādassa cūbhayaṃ;_  
_Puttamaṃsāni khādāmi,_  
_pubbalohitamakkhitā”ti._  


</Paragraph>
</Division>

<Paragraph number="41" type="center" book="pv">
Pañcaputtakhādapetivatthu chaṭṭhaṃ.
</Paragraph>
