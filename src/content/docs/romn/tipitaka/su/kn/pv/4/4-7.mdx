<Division number="753" book="pv">

<Paragraph number="807" type="verses" book="pv">
_Pubbe katānaṃ kammānaṃ,_  
_vipāko mathaye manaṃ;_  
_Rūpe sadde rase gandhe,_  
_phoṭṭhabbe ca manorame._  

</Paragraph>

</Division>
<Division number="754" book="pv">

<Paragraph number="808" type="verses" book="pv">
_Naccaṃ gītaṃ ratiṃ khiḍḍaṃ,_  
_anubhutvā anappakaṃ;_  
_Uyyāne paricaritvā,_  
_pavisanto giribbajaṃ._  

</Paragraph>

</Division>
<Division number="755" book="pv">

<Paragraph number="809" type="verses" book="pv">
_Isiṃ sunetta maddakkhi,_  
_attadantaṃ samāhitaṃ;_  
_Appicchaṃ hirisampannaṃ,_  
_uñche pattagate rataṃ._  

</Paragraph>

</Division>
<Division number="756" book="pv">

<Paragraph number="810" type="verses" book="pv">
_Hatthikkhandhato oruyha,_  
_laddhā bhanteti cābravi;_  
_Tassa pattaṃ gahetvāna,_  
_uccaṃ paggayha khattiyo._  

</Paragraph>

</Division>
<Division number="757" book="pv">

<Paragraph number="811" type="verses" book="pv">
_Thaṇḍile pattaṃ bhinditvā,_  
_hasamāno apakkami;_  
_“Rañño kitavassāhaṃ putto,_  
_kiṃ maṃ bhikkhu karissasi”._  

</Paragraph>

</Division>
<Division number="758" book="pv">

<Paragraph number="812" type="verses" book="pv">
_Tassa kammassa pharusassa,_  
_vipāko kaṭuko ahu;_  
_Yaṃ rājaputto vedesi,_  
_nirayamhi samappito._  

</Paragraph>

</Division>
<Division number="759" book="pv">

<Paragraph number="813" type="verses" book="pv">
_Chaḷeva caturāsīti,_  
_vassāni navutāni ca;_  
_Bhusaṃ dukkhaṃ nigacchittho,_  
_niraye katakibbiso._  

</Paragraph>

</Division>
<Division number="760" book="pv">

<Paragraph number="814" type="verses" book="pv">
_Uttānopi ca paccittha,_  
_nikujjo vāmadakkhiṇo;_  
_Uddhampādo ṭhito ceva,_  
_ciraṃ bālo apaccatha._  

</Paragraph>

</Division>
<Division number="761" book="pv">

<Paragraph number="815" type="verses" book="pv">
_Bahūni vassasahassāni,_  
_pūgāni nahutāni ca;_  
_Bhusaṃ dukkhaṃ nigacchittho,_  
_niraye katakibbiso._  

</Paragraph>

</Division>
<Division number="762" book="pv">

<Paragraph number="816" type="verses" book="pv">
_Etādisaṃ kho kaṭukaṃ,_  
_appaduṭṭhappadosinaṃ;_  
_Paccanti pāpakammantā,_  
_isimāsajja subbataṃ._  

</Paragraph>

</Division>
<Division number="763" book="pv">

<Paragraph number="817" type="verses" book="pv">
_So tattha bahuvassāni,_  
_vedayitvā bahuṃ dukhaṃ;_  
_Khuppipāsahato nāma,_  
_peto āsi tato cuto._  

</Paragraph>

</Division>
<Division number="764" book="pv">

<Paragraph number="818" type="verses" book="pv">
_Etamādīnavaṃ ñatvā,_  
_issaramadasambhavaṃ;_  
_Pahāya issaramadaṃ,_  
_nivātamanuvattaye._  

</Paragraph>

</Division>
<Division number="765" book="pv">

<Paragraph number="819" type="verses" book="pv">
_Diṭṭheva dhamme pāsaṃso,_  
_yo buddhesu sagāravo;_  
_Kāyassa bhedā sappañño,_  
_saggaṃ so upapajjatīti._  


</Paragraph>
</Division>

<Paragraph number="820" type="center" book="pv">
Rājaputtapetavatthu sattamaṃ.
</Paragraph>
