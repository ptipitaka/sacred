<Division number="766" book="pv">

<Paragraph number="821" type="verses" book="pv">
_“Gūthakūpato uggantvā,_  
_ko nu dīno patiṭṭhasi;_  
_Nissaṃsayaṃ pāpakammanto,_  
_kiṃ nu saddahase tuvan”ti._  

</Paragraph>

</Division>
<Division number="767" book="pv">

<Paragraph number="822" type="verses" book="pv">
_“Ahaṃ bhadante petomhi,_  
_duggato yamalokiko;_  
_Pāpakammaṃ karitvāna,_  
_petalokaṃ ito gato”._  

</Paragraph>

</Division>
<Division number="768" book="pv">

<Paragraph number="823" type="verses" book="pv">
_“Kiṃ nu kāyena vācāya,_  
_manasā dukkaṭaṃ kataṃ;_  
_Kissa kammavipākena,_  
_idaṃ dukkhaṃ nigacchasī”ti._  

</Paragraph>

</Division>
<Division number="769" book="pv">

<Paragraph number="824" type="verses" book="pv">
_“Ahu āvāsiko mayhaṃ,_  
_issukī kulamaccharī;_  
_Ajjhāsito mayhaṃ ghare,_  
_kadariyo paribhāsako._  

</Paragraph>

</Division>
<Division number="770" book="pv">

<Paragraph number="825" type="verses" book="pv">
_Tassāhaṃ vacanaṃ sutvā,_  
_bhikkhavo paribhāsisaṃ;_  
_Tassa kammavipākena,_  
_petalokaṃ ito gato”ti._  

</Paragraph>

</Division>
<Division number="771" book="pv">

<Paragraph number="826" type="verses" book="pv">
_“Amitto mittavaṇṇena,_  
_yo te āsi kulūpako;_  
_Kāyassa bhedā duppañño,_  
_kiṃ nu pecca gatiṃ gato”ti._  

</Paragraph>

</Division>
<Division number="772" book="pv">

<Paragraph number="827" type="verses" book="pv">
_“Tassevāhaṃ pāpakammassa,_  
_sīse tiṭṭhāmi matthake;_  
_So ca paravisayaṃ patto,_  
_mameva paricārako._  

</Paragraph>

</Division>
<Division number="773" book="pv">

<Paragraph number="828" type="verses" book="pv">
_Yaṃ bhadante hadantaññe,_  
_etaṃ me hoti bhojanaṃ;_  
_Ahañca kho yaṃ hadāmi,_  
_etaṃ so upajīvatī”ti._  


</Paragraph>
</Division>

<Paragraph number="829" type="center" book="pv">
Gūthakhādakapetavatthu aṭṭhamaṃ.
</Paragraph>
