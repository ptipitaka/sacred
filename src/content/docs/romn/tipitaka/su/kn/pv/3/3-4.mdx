<Division number="447" book="pv">

<Paragraph number="483" type="verses" book="pv">
_“Bhusāni eko sāliṃ punāparo,_  
_Ayañca nārī sakamaṃsalohitaṃ;_  
_Tuvañca gūthaṃ asuciṃ akantaṃ,_  
_Paribhuñjasi kissa ayaṃ vipāko”ti._  

</Paragraph>

</Division>
<Division number="448" book="pv">

<Paragraph number="484" type="verses" book="pv">
_“Ayaṃ pure mātaraṃ hiṃsati,_  
_Ayaṃ pana kūṭavāṇijo;_  
_Ayaṃ maṃsāni khāditvā,_  
_Musāvādena vañceti._  

</Paragraph>

</Division>
<Division number="449" book="pv">

<Paragraph number="485" type="verses" book="pv">
_Ahaṃ manussesu manussabhūtā,_  
_Agārinī sabbakulassa issarā;_  
_Santesu pariguhāmi,_  
_Mā ca kiñci ito adaṃ._  

</Paragraph>

</Division>
<Division number="450" book="pv">

<Paragraph number="486" type="verses" book="pv">
_Musāvādena chādemi,_  
_‘natthi etaṃ mama gehe;_  
_Sace santaṃ niguhāmi,_  
_gūtho me hotu bhojanaṃ’._  

</Paragraph>

</Division>
<Division number="451" book="pv">

<Paragraph number="487" type="verses" book="pv">
_Tassa kammassa vipākena,_  
_musāvādassa cūbhayaṃ;_  
_Sugandhaṃ sālino bhattaṃ,_  
_gūthaṃ me parivattati._  

</Paragraph>

</Division>
<Division number="452" book="pv">

<Paragraph number="488" type="verses" book="pv">
_Avañjhāni ca kammāni,_  
_Na hi kammaṃ vinassati;_  
_Duggandhaṃ kiminaṃ mīḷaṃ,_  
_Bhuñjāmi ca pivāmi cā”ti._  


</Paragraph>
</Division>

<Paragraph number="489" type="center" book="pv">
Bhusapetavatthu catutthaṃ.
</Paragraph>
