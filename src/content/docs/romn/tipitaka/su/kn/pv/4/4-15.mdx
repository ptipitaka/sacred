<Division number="802" book="pv">

<Paragraph number="864" type="verses" book="pv">
_“Saṭṭhivassasahassāni,_  
_paripuṇṇāni sabbaso;_  
_Niraye paccamānānaṃ,_  
_kadā anto bhavissati._  

</Paragraph>

</Division>
<Division number="803" book="pv">

<Paragraph number="865" type="verses" book="pv">
_Natthi anto kuto anto,_  
_na anto paṭidissati;_  
_Tathā hi pakataṃ pāpaṃ,_  
_tuyhaṃ mayhañca mārisā._  

</Paragraph>

</Division>
<Division number="804" book="pv">

<Paragraph number="866" type="verses" book="pv">
_Dujjīvitamajīvamha,_  
_ye sante na dadamhase;_  
_Santesu deyyadhammesu,_  
_dīpaṃ nākamha attano._  

</Paragraph>

</Division>
<Division number="805" book="pv">

<Paragraph number="867" type="verses" book="pv">
_Sohaṃ nūna ito gantvā,_  
_yoniṃ laddhāna mānusiṃ;_  
_Vadaññū sīlasampanno,_  
_kāhāmi kusalaṃ bahun”ti._  


</Paragraph>
</Division>

<Paragraph number="868" type="center" book="pv">
Seṭṭhiputtapetavatthu pannarasamaṃ.
</Paragraph>
