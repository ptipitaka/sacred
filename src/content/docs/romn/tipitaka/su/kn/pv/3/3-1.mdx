<Division number="387" book="pv">

<Paragraph number="420" type="verses" book="pv">
_“Abhijjamāne vārimhi,_  
_gaṅgāya idha gacchasi;_  
_Naggo pubbaddhapetova,_  
_māladhārī alaṅkato;_  
_Kuhiṃ gamissasi peta,_  
_kattha vāso bhavissatī”ti._  

</Paragraph>

</Division>
<Division number="388" book="pv">

<Paragraph number="421" type="verses" book="pv">
_“Cundaṭṭhilaṃ gamissāmi,_  
_peto so iti bhāsati;_  
_Antare vāsabhagāmaṃ,_  
_bārāṇasiṃ ca santike”._  

</Paragraph>

</Division>
<Division number="389" book="pv">

<Paragraph number="422" type="verses" book="pv">
_Tañca disvā mahāmatto,_  
_koliyo iti vissuto;_  
_Sattuṃ bhattañca petassa,_  
_pītakañca yugaṃ adā._  

</Paragraph>

</Division>
<Division number="390" book="pv">

<Paragraph number="423" type="verses" book="pv">
_Nāvāya tiṭṭhamānāya,_  
_kappakassa adāpayi;_  
_Kappakassa padinnamhi,_  
_ṭhāne petassa dissatha._  

</Paragraph>

</Division>
<Division number="391" book="pv">

<Paragraph number="424" type="verses" book="pv">
_Tato suvatthavasano,_  
_māladhārī alaṅkato;_  
_Ṭhāne ṭhitassa petassa,_  
_dakkhiṇā upakappatha;_  
_Tasmā dajjetha petānaṃ,_  
_anukampāya punappunaṃ._  

</Paragraph>

</Division>
<Division number="392" book="pv">

<Paragraph number="425" type="verses" book="pv">
_Sātunnavasanā eke,_  
_aññe kesanivāsanā;_  
_Petā bhattāya gacchanti,_  
_pakkamanti disodisaṃ._  

</Paragraph>

</Division>
<Division number="393" book="pv">

<Paragraph number="426" type="verses" book="pv">
_Dūre eke padhāvitvā,_  
_aladdhāva nivattare;_  
_Chātā pamucchitā bhantā,_  
_bhūmiyaṃ paṭisumbhitā._  

</Paragraph>

</Division>
<Division number="394" book="pv">

<Paragraph number="427" type="verses" book="pv">
_Te ca tattha papatitā,_  
_bhūmiyaṃ paṭisumbhitā;_  
_Pubbe akatakalyāṇā,_  
_aggidaḍḍhāva ātape._  

</Paragraph>

</Division>
<Division number="395" book="pv">

<Paragraph number="428" type="verses" book="pv">
_“Mayaṃ pubbe pāpadhammā,_  
_gharaṇī kulamātaro;_  
_Santesu deyyadhammesu,_  
_dīpaṃ nākamha attano._  

</Paragraph>

</Division>
<Division number="396" book="pv">

<Paragraph number="429" type="verses" book="pv">
_Pahūtaṃ annapānampi,_  
_apissu avakirīyati;_  
_Sammaggate pabbajite,_  
_na ca kiñci adamhase._  

</Paragraph>

</Division>
<Division number="397" book="pv">

<Paragraph number="430" type="verses" book="pv">
_Akammakāmā alasā,_  
_Sādukāmā mahagghasā;_  
_Ālopapiṇḍadātāro,_  
_Paṭiggahe paribhāsimhase._  

</Paragraph>

</Division>
<Division number="398" book="pv">

<Paragraph number="431" type="verses" book="pv">
_Te gharā tā ca dāsiyo,_  
_tānevābharaṇāni no;_  
_Te aññe paricārenti,_  
_mayaṃ dukkhassa bhāgino._  

</Paragraph>

</Division>
<Division number="399" book="pv">

<Paragraph number="432" type="verses" book="pv">
_Veṇī vā avaññā honti,_  
_rathakārī ca dubbhikā;_  
_Caṇḍālī kapaṇā honti,_  
_kappakā ca punappunaṃ._  

</Paragraph>

</Division>
<Division number="400" book="pv">

<Paragraph number="433" type="verses" book="pv">
_Yāni yāni nihīnāni,_  
_kulāni kapaṇāni ca;_  
_Tesu tesveva jāyanti,_  
_esā maccharino gati._  

</Paragraph>

</Division>
<Division number="401" book="pv">

<Paragraph number="434" type="verses" book="pv">
_Pubbe ca katakalyāṇā,_  
_dāyakā vītamaccharā;_  
_Saggaṃ te paripūrenti,_  
_obhāsenti ca nandanaṃ._  

</Paragraph>

</Division>
<Division number="402" book="pv">

<Paragraph number="435" type="verses" book="pv">
_Vejayante ca pāsāde,_  
_ramitvā kāmakāmino;_  
_Uccākulesu jāyanti,_  
_sabhogesu tato cutā._  

</Paragraph>

</Division>
<Division number="403" book="pv">

<Paragraph number="436" type="verses" book="pv">
_Kūṭāgāre ca pāsāde,_  
_pallaṅke gonakatthate;_  
_Bījitaṅgā morahatthehi,_  
_kule jātā yasassino._  

</Paragraph>

</Division>
<Division number="404" book="pv">

<Paragraph number="437" type="verses" book="pv">
_Aṅkato aṅkaṃ gacchanti,_  
_māladhārī alaṅkatā;_  
_Dhātiyo upatiṭṭhanti,_  
_sāyaṃ pātaṃ sukhesino._  

</Paragraph>

</Division>
<Division number="405" book="pv">

<Paragraph number="438" type="verses" book="pv">
_Nayidaṃ akatapuññānaṃ,_  
_katapuññānamevidaṃ;_  
_Asokaṃ nandanaṃ rammaṃ,_  
_tidasānaṃ mahāvanaṃ._  

</Paragraph>

</Division>
<Division number="406" book="pv">

<Paragraph number="439" type="verses" book="pv">
_Sukhaṃ akatapuññānaṃ,_  
_idha natthi parattha ca;_  
_Sukhañca katapuññānaṃ,_  
_idha ceva parattha ca._  

</Paragraph>

</Division>
<Division number="407" book="pv">

<Paragraph number="440" type="verses" book="pv">
_Tesaṃ sahabyakāmānaṃ,_  
_kattabbaṃ kusalaṃ bahuṃ;_  
_Katapuññā hi modanti,_  
_sagge bhogasamaṅgino”ti._  


</Paragraph>
</Division>

<Paragraph number="441" type="center" book="pv">
Abhijjamānapetavatthu paṭhamaṃ.
</Paragraph>
