<Division number="499" book="pv">

<Paragraph number="541" type="verses" book="pv">
_“Mālī kiriṭī kāyūrī,_  
_gattā te candanussadā;_  
_Pasannamukhavaṇṇosi,_  
_sūriyavaṇṇova sobhasi._  

</Paragraph>

</Division>
<Division number="500" book="pv">

<Paragraph number="542" type="verses" book="pv">
_Amānusā pārisajjā,_  
_ye teme paricārakā;_  
_Dasa kaññāsahassāni,_  
_yā temā paricārikā;_  
_Tā kambukāyūradharā,_  
_kañcanāveḷabhūsitā._  

</Paragraph>

</Division>
<Division number="501" book="pv">

<Paragraph number="543" type="verses" book="pv">
_Mahānubhāvosi tuvaṃ,_  
_lomahaṃsanarūpavā;_  
_Piṭṭhimaṃsāni attano,_  
_sāmaṃ ukkacca khādasi._  

</Paragraph>

</Division>
<Division number="502" book="pv">

<Paragraph number="544" type="verses" book="pv">
_Kiṃ nu kāyena vācāya,_  
_manasā dukkuṭaṃ kataṃ;_  
_Kissa kammavipākena,_  
_piṭṭhimaṃsāni attano;_  
_Sāmaṃ ukkacca khādasī”ti._  

</Paragraph>

</Division>
<Division number="503" book="pv">

<Paragraph number="545" type="verses" book="pv">
_“Attanohaṃ anatthāya,_  
_jīvaloke acārisaṃ;_  
_Pesuññamusāvādena,_  
_nikativañcanāya ca._  

</Paragraph>

</Division>
<Division number="504" book="pv">

<Paragraph number="546" type="verses" book="pv">
_Tatthāhaṃ parisaṃ gantvā,_  
_Saccakāle upaṭṭhite;_  
_Atthaṃ dhammaṃ nirākatvā,_  
_Adhammamanuvattisaṃ._  

</Paragraph>

</Division>
<Division number="505" book="pv">

<Paragraph number="547" type="verses" book="pv">
_Evaṃ so khādatattānaṃ,_  
_yo hoti piṭṭhimaṃsiko;_  
_Yathāhaṃ ajja khādāmi,_  
_piṭṭhimaṃsāni attano._  

</Paragraph>

</Division>
<Division number="506" book="pv">

<Paragraph number="548" type="verses" book="pv">
_Tayidaṃ tayā nārada sāmaṃ diṭṭhaṃ,_  
_Anukampakā ye kusalā vadeyyuṃ;_  
_Mā pesuṇaṃ mā ca musā abhāṇi,_  
_Mā khosi piṭṭhimaṃsiko tuvan”ti._  


</Paragraph>
</Division>

<Paragraph number="549" type="center" book="pv">
Kūṭavinicchayikapetavatthu navamaṃ.
</Paragraph>
