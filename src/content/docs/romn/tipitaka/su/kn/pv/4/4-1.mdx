<Paragraph number="564" type="verses" book="pv">
_Vesālī nāma nagaratthi vajjīnaṃ,_  
_Tattha ahu licchavi ambasakkaro;_  
_Disvāna petaṃ nagarassa bāhiraṃ,_  
_Tattheva pucchittha taṃ kāraṇatthiko._  

</Paragraph>

</Division>
<Division number="518" book="pv" bookNo="19" page="181">

<Paragraph number="565" type="verses" book="pv">
_“Seyyā nisajjā nayimassa atthi,_  
_Abhikkamo natthi paṭikkamo ca;_  
_Asitapītakhāyitavatthabhogā,_  
_Paricārikā sāpi imassa natthi._  

</Paragraph>

</Division>
<Division number="519" book="pv" bookNo="19" page="181">

<Paragraph number="566" type="verses" book="pv">
_Ye ñātakā diṭṭhasutā suhajjā,_  
_Anukampakā yassa ahesuṃ pubbe;_  
_Daṭṭhumpi te dāni na taṃ labhanti,_  
_Virājitatto hi janena tena._  

</Paragraph>

</Division>
<Division number="520" book="pv" bookNo="19" page="181">

<Paragraph number="567" type="verses" book="pv">
_Na oggatattassa bhavanti mittā,_  
_Jahanti mittā vikalaṃ viditvā;_  
_Atthañca disvā parivārayanti,_  
_Bahū mittā uggatattassa honti._  

</Paragraph>

</Division>
<Division number="521" book="pv" bookNo="19" page="181">

<Paragraph number="568" type="verses" book="pv">
_Nihīnatto sabbabhogehi kiccho,_  
_Sammakkhito samparibhinnagatto;_  
_Ussāvabindūva palimpamāno,_  
_Ajja suve jīvitassūparodho._  

</Paragraph>

</Division>
<Division number="522" book="pv" bookNo="19" page="181">

<Paragraph number="569" type="verses" book="pv">
_Etādisaṃ uttamakicchappattaṃ,_  
_Uttāsitaṃ pucimandassa sūle;_  
_‘Atha tvaṃ kena vaṇṇena vadesi yakkha,_  
_Jīva bho jīvitameva seyyo’”ti._  

</Paragraph>

</Division>
<Division number="523" book="pv" bookNo="19" page="182">

<Paragraph number="570" type="verses" book="pv">
_“Sālohito esa ahosi mayhaṃ,_  
_Ahaṃ sarāmi purimāya jātiyā;_  
_Disvā ca me kāruññamahosi rāja,_  
_Mā pāpadhammo nirayaṃ patāyaṃ._  

</Paragraph>

</Division>
<Division number="524" book="pv" bookNo="19" page="182">

<Paragraph number="571" type="verses" book="pv">
_Ito cuto licchavi esa poso,_  
_Sattussadaṃ nirayaṃ ghorarūpaṃ;_  
_Upapajjati dukkaṭakammakārī,_  
_Mahābhitāpaṃ kaṭukaṃ bhayānakaṃ._  

</Paragraph>

</Division>
<Division number="525" book="pv" bookNo="19" page="182">

<Paragraph number="572" type="verses" book="pv">
_Anekabhāgena guṇena seyyo,_  
_Ayameva sūlo nirayena tena;_  
_Ekantadukkhaṃ kaṭukaṃ bhayānakaṃ,_  
_Ekantatibbaṃ nirayaṃ patāyaṃ._  

</Paragraph>

</Division>
<Division number="526" book="pv" bookNo="19" page="182">

<Paragraph number="573" type="verses" book="pv">
_Idañca sutvā vacanaṃ mameso,_  
_Dukkhūpanīto vijaheyya pāṇaṃ;_  
_Tasmā ahaṃ santike na bhaṇāmi,_  
_Mā me kato jīvitassūparodho”._  

</Paragraph>

</Division>
<Division number="527" book="pv" bookNo="19" page="182">

<Paragraph number="574" type="verses" book="pv">
_“Aññāto eso purisassa attho,_  
_Aññampi icchāmase pucchituṃ tuvaṃ;_  
_Okāsakammaṃ sace no karosi,_  
_Pucchāma taṃ no na ca kujjhitabban”ti._  

</Paragraph>

</Division>
<Division number="528" book="pv" bookNo="19" page="182">

<Paragraph number="575" type="verses" book="pv">
_“Addhā paṭiññā me tadā ahu,_  
_Nācikkhanā appasannassa hoti;_  
_Akāmā saddheyyavacoti katvā,_  
_Pucchassu maṃ kāmaṃ yathā visayhan”ti._  

</Paragraph>

</Division>
<Division number="529" book="pv" bookNo="19" page="182">

<Paragraph number="576" type="verses" book="pv">
_“Yaṃ kiñcahaṃ cakkhunā passissāmi,_  
_Sabbampi tāhaṃ abhisaddaheyyaṃ;_  
_Disvāva taṃ nopi ce saddaheyyaṃ,_  
_Kareyyāsi me yakkha niyassakamman”ti._  

</Paragraph>

</Division>
<Division number="530" book="pv" bookNo="19" page="183">

<Paragraph number="577" type="verses" book="pv">
_“Saccappaṭiññā tava mesā hotu,_  
_Sutvāna dhammaṃ labha suppasādaṃ;_  
_Aññatthiko no ca paduṭṭhacitto,_  
_Yaṃ te sutaṃ asutañcāpi dhammaṃ;_  
_Sabbampi akkhissaṃ yathā pajānanti”._  

</Paragraph>

</Division>
<Division number="531" book="pv" bookNo="19" page="183">

<Paragraph number="578" type="verses" book="pv">
_“Setena assena alaṅkatena,_  
_Upayāsi sūlāvutakassa santike;_  
_Yānaṃ idaṃ abbhutaṃ dassaneyyaṃ,_  
_Kissetaṃ kammassa ayaṃ vipāko”ti._  

</Paragraph>

</Division>
<Division number="532" book="pv" bookNo="19" page="183">

<Paragraph number="579" type="verses" book="pv">
_“Vesāliyā nagarassa majjhe,_  
_Cikkhallamagge narakaṃ ahosi;_  
_Gosīsamekāhaṃ pasannacitto,_  
_Setaṃ gahetvā narakasmiṃ nikkhipiṃ._  

</Paragraph>

</Division>
<Division number="533" book="pv" bookNo="19" page="183">

<Paragraph number="580" type="verses" book="pv">
_Etasmiṃ pādāni patiṭṭhapetvā,_  
_Mayañca aññe ca atikkamimhā;_  
_Yānaṃ idaṃ abbhutaṃ dassaneyyaṃ,_  
_Tasseva kammassa ayaṃ vipāko”ti._  

</Paragraph>

</Division>
<Division number="534" book="pv" bookNo="19" page="183">

<Paragraph number="581" type="verses" book="pv">
_“Vaṇṇo ca te sabbadisā pabhāsati,_  
_Gandho ca te sabbadisā pavāyati;_  
_Yakkhiddhipattosi mahānubhāvo,_  
_Naggo cāsi kissa ayaṃ vipāko”ti._  

</Paragraph>

</Division>
<Division number="535" book="pv" bookNo="19" page="183">

<Paragraph number="582" type="verses" book="pv">
_“Akkodhano niccapasannacitto,_  
_Saṇhāhi vācāhi janaṃ upemi;_  
_Tasseva kammassa ayaṃ vipāko,_  
_Dibbo me vaṇṇo satataṃ pabhāsati._  

</Paragraph>

</Division>
<Division number="536" book="pv" bookNo="19" page="183">

<Paragraph number="583" type="verses" book="pv">
_Yasañca kittiñca dhamme ṭhitānaṃ,_  
_Disvāna mantemi pasannacitto;_  
_Tasseva kammassa ayaṃ vipāko,_  
_Dibbo me gandho satataṃ pavāyati._  

</Paragraph>

</Division>
<Division number="537" book="pv" bookNo="19" page="184">

<Paragraph number="584" type="verses" book="pv">
_Sahāyānaṃ titthasmiṃ nhāyantānaṃ,_  
_Thale gahetvā nidahissa dussaṃ;_  
_Khiḍḍatthiko no ca paduṭṭhacitto,_  
_Tenamhi naggo kasirā ca vuttī”ti._  

</Paragraph>

</Division>
<Division number="538" book="pv" bookNo="19" page="184">

<Paragraph number="585" type="verses" book="pv">
_“Yo kīḷamāno pakaroti pāpaṃ,_  
_Tassedisaṃ kammavipākamāhu;_  
_Akīḷamāno pana yo karoti,_  
_Kiṃ tassa kammassa vipākamāhū”ti._  

</Paragraph>

</Division>
<Division number="539" book="pv" bookNo="19" page="184">

<Paragraph number="586" type="verses" book="pv">
_“Ye duṭṭhasaṅkappamanā manussā,_  
_Kāyena vācāya ca saṃkiliṭṭhā;_  
_Kāyassa bhedā abhisamparāyaṃ,_  
_Asaṃsayaṃ te nirayaṃ upenti._  

</Paragraph>

</Division>
<Division number="540" book="pv" bookNo="19" page="184">

<Paragraph number="587" type="verses" book="pv">
_Apare pana sugatimāsamānā,_  
_Dāne ratā saṅgahitattabhāvā;_  
_Kāyassa bhedā abhisamparāyaṃ,_  
_Asaṃsayaṃ te sugatiṃ upentī”ti._  

</Paragraph>

</Division>
<Division number="541" book="pv" bookNo="19" page="184">

<Paragraph number="588" type="verses" book="pv">
_“Taṃ kinti jāneyyamahaṃ avecca,_  
_Kalyāṇapāpassa ayaṃ vipāko;_  
_Kiṃ vāhaṃ disvā abhisaddaheyyaṃ,_  
_Ko vāpi maṃ saddahāpeyya etan”ti._  

</Paragraph>

</Division>
<Division number="542" book="pv" bookNo="19" page="184">

<Paragraph number="589" type="verses" book="pv">
_“Disvā ca sutvā abhisaddahassu,_  
_Kalyāṇapāpassa ayaṃ vipāko;_  
_Kalyāṇapāpe ubhaye asante,_  
_Siyā nu sattā sugatā duggatā vā._  

</Paragraph>

</Division>
<Division number="543" book="pv" bookNo="19" page="184">

<Paragraph number="590" type="verses" book="pv">
_No cettha kammāni kareyyuṃ maccā,_  
_Kalyāṇapāpāni manussaloke;_  
_Nāhesuṃ sattā sugatā duggatā vā,_  
_Hīnā paṇītā ca manussaloke._  

</Paragraph>

</Division>
<Division number="544" book="pv" bookNo="19" page="185">

<Paragraph number="591" type="verses" book="pv">
_Yasmā ca kammāni karonti maccā,_  
_Kalyāṇapāpāni manussaloke;_  
_Tasmā hi sattā sugatā duggatā vā,_  
_Hīnā paṇītā ca manussaloke._  

</Paragraph>

</Division>
<Division number="545" book="pv" bookNo="19" page="185">

<Paragraph number="592" type="verses" book="pv">
_Dvayajja kammānaṃ vipākamāhu,_  
_Sukhassa dukkhassa ca vedanīyaṃ;_  
_Tā devatāyo paricārayanti,_  
_Paccanti bālā dvayataṃ apassino._  

</Paragraph>

</Division>
<Division number="546" book="pv" bookNo="19" page="185">

<Paragraph number="593" type="verses" book="pv">
_Na matthi kammāni sayaṃkatāni,_  
_Datvāpi me natthi yo ādiseyya;_  
_Acchādanaṃ sayanamathannapānaṃ,_  
_Tenamhi naggo kasirā ca vuttī”ti._  

</Paragraph>

</Division>
<Division number="547" book="pv" bookNo="19" page="185">

<Paragraph number="594" type="verses" book="pv">
_“Siyā nu kho kāraṇaṃ kiñci yakkha,_  
_Acchādanaṃ yena tuvaṃ labhetha;_  
_Ācikkha me tvaṃ yadatthi hetu,_  
_Saddhāyikaṃ hetuvaco suṇomā”ti._  

</Paragraph>

</Division>
<Division number="548" book="pv" bookNo="19" page="185">

<Paragraph number="595" type="verses" book="pv">
_“Kappitako nāma idhatthi bhikkhu,_  
_Jhāyī susīlo arahā vimutto;_  
_Guttindriyo saṃvutapātimokkho,_  
_Sītibhūto uttamadiṭṭhipatto._  

</Paragraph>

</Division>
<Division number="549" book="pv" bookNo="19" page="185">

<Paragraph number="596" type="verses" book="pv">
_Sakhilo vadaññū suvaco sumukho,_  
_Svāgamo suppaṭimuttako ca;_  
_Puññassa khettaṃ araṇavihārī,_  
_Devamanussānañca dakkhiṇeyyo._  

</Paragraph>

</Division>
<Division number="550" book="pv" bookNo="19" page="185">

<Paragraph number="597" type="verses" book="pv">
_Santo vidhūmo anīgho nirāso,_  
_Mutto visallo amamo avaṅko;_  
_Nirūpadhī sabbapapañcakhīṇo,_  
_Tisso vijjā anuppatto jutimā._  

</Paragraph>

</Division>
<Division number="551" book="pv" bookNo="19" page="186">

<Paragraph number="598" type="verses" book="pv">
_Appaññāto disvāpi na ca sujāno,_  
_Munīti naṃ vajjisu voharanti;_  
_Jānanti taṃ yakkhabhūtā anejaṃ,_  
_Kalyāṇadhammaṃ vicarantaṃ loke._  

</Paragraph>

</Division>
<Division number="552" book="pv" bookNo="19" page="186">

<Paragraph number="599" type="verses" book="pv">
_Tassa tuvaṃ ekayugaṃ duve vā,_  
_Mamuddisitvāna sace dadetha;_  
_Paṭiggahītāni ca tāni assu,_  
_Mamañca passetha sannaddhadussan”ti._  

</Paragraph>

</Division>
<Division number="553" book="pv" bookNo="19" page="186">

<Paragraph number="600" type="verses" book="pv">
_“Kasmiṃ padese samaṇaṃ vasantaṃ,_  
_Gantvāna passemu mayaṃ idāni;_  
_Yo majja kaṅkhaṃ vicikicchitañca,_  
_Diṭṭhīvisūkāni vinodayeyyā”ti._  

</Paragraph>

</Division>
<Division number="554" book="pv" bookNo="19" page="186">

<Paragraph number="601" type="verses" book="pv">
_“Eso nisinno kapinaccanāyaṃ,_  
_Parivārito devatāhi bahūhi;_  
_Dhammiṃ kathaṃ bhāsati saccanāmo,_  
_Sakasmimācerake appamatto”ti._  

</Paragraph>

</Division>
<Division number="555" book="pv" bookNo="19" page="186">

<Paragraph number="602" type="verses" book="pv">
_“Tathāhaṃ kassāmi gantvā idāni,_  
_Acchādayissaṃ samaṇaṃ yugena;_  
_Paṭiggahitāni ca tāni assu,_  
_Tuvañca passemu sannaddhadussan”ti._  

</Paragraph>

</Division>
<Division number="556" book="pv" bookNo="19" page="186">

<Paragraph number="603" type="verses" book="pv">
_“Mā akkhaṇe pabbajitaṃ upāgami,_  
_Sādhu vo licchavi nesa dhammo;_  
_Tato ca kāle upasaṅkamitvā,_  
_Tattheva passāhi raho nisinnan”ti._  

</Paragraph>

</Division>
<Division number="557" book="pv" bookNo="19" page="186">

<Paragraph number="604" type="verses" book="pv">
_Tathāti vatvā agamāsi tattha,_  
_Parivārito dāsagaṇena licchavi;_  
_So taṃ nagaraṃ upasaṅkamitvā,_  
_Vāsūpagacchittha sake nivesane._  

</Paragraph>

</Division>
<Division number="558" book="pv" bookNo="19" page="187">

<Paragraph number="605" type="verses" book="pv">
_Tato ca kāle gihikiccāni katvā,_  
_Nhatvā pivitvā ca khaṇaṃ labhitvā;_  
_Viceyya peḷāto ca yugāni aṭṭha,_  
_Gāhāpayī dāsagaṇena licchavi._  

</Paragraph>

</Division>
<Division number="559" book="pv" bookNo="19" page="187">

<Paragraph number="606" type="verses" book="pv">
_So taṃ padesaṃ upasaṅkamitvā,_  
_Taṃ addasa samaṇaṃ santacittaṃ;_  
_Paṭikkantaṃ gocarato nivattaṃ,_  
_Sītibhūtaṃ rukkhamūle nisinnaṃ._  

</Paragraph>

</Division>
<Division number="560" book="pv" bookNo="19" page="187">

<Paragraph number="607" type="verses" book="pv">
_Tamenamavoca upasaṅkamitvā,_  
_Appābādhaṃ phāsuvihārañca pucchi;_  
_“Vesāliyaṃ licchavihaṃ bhadante,_  
_Jānanti maṃ licchavi ambasakkaro._  

</Paragraph>

</Division>
<Division number="561" book="pv" bookNo="19" page="187">

<Paragraph number="608" type="verses" book="pv">
_Imāni me aṭṭha yugā subhāni,_  
_Paṭigaṇha bhante padadāmi tuyhaṃ;_  
_Teneva atthena idhāgatosmi,_  
_Yathā ahaṃ attamano bhaveyyan”ti._  

</Paragraph>

</Division>
<Division number="562" book="pv" bookNo="19" page="187">

<Paragraph number="609" type="verses" book="pv">
_“Dūratova samaṇabrāhmaṇā ca,_  
_Nivesanaṃ te parivajjayanti;_  
_Pattāni bhijjanti ca te nivesane,_  
_Saṅghāṭiyo cāpi vidālayanti._  

</Paragraph>

</Division>
<Division number="563" book="pv" bookNo="19" page="187">

<Paragraph number="610" type="verses" book="pv">
_Athāpare pādakuṭhārikāhi,_  
_Avaṃsirā samaṇā pātayanti;_  
_Etādisaṃ pabbajitā vihesaṃ,_  
_Tayā kataṃ samaṇā pāpuṇanti._  

</Paragraph>

</Division>
<Division number="564" book="pv" bookNo="19" page="187">

<Paragraph number="611" type="verses" book="pv">
_Tiṇena telampi na tvaṃ adāsi,_  
_Muḷhassa maggampi na pāvadāsi;_  
_Andhassa daṇḍaṃ sayamādiyāsi,_  
_Etādiso kadariyo asaṃvuto tuvaṃ;_  
_Atha tvaṃ kena vaṇṇena kimeva disvā,_  
_Amhehi saha saṃvibhāgaṃ karosī”ti._  

</Paragraph>

</Division>
<Division number="565" book="pv" bookNo="19" page="188">

<Paragraph number="612" type="verses" book="pv">
_“Paccemi bhante yaṃ tvaṃ vadesi,_  
_Vihesayiṃ samaṇe brāhmaṇe ca;_  
_Khiḍḍatthiko no ca paduṭṭhacitto,_  
_Etampi me dukkaṭameva bhante._  

</Paragraph>

</Division>
<Division number="566" book="pv" bookNo="19" page="188">

<Paragraph number="613" type="verses" book="pv">
_Khiḍḍāya yakkho pasavitvā pāpaṃ,_  
_Vedeti dukkhaṃ asamattabhogī;_  
_Daharo yuvā nagganiyassa bhāgī,_  
_Kiṃ su tato dukkhatarassa hoti._  

</Paragraph>

</Division>
<Division number="567" book="pv" bookNo="19" page="188">

<Paragraph number="614" type="verses" book="pv">
_Taṃ disvā saṃvegamalatthaṃ bhante,_  
_Tappaccayā vāpi dadāmi dānaṃ;_  
_Paṭigaṇha bhante vatthayugāni aṭṭha,_  
_Yakkhassimā gacchantu dakkhiṇāyo”ti._  

</Paragraph>

</Division>
<Division number="568" book="pv" bookNo="19" page="188">

<Paragraph number="615" type="verses" book="pv">
_“Addhā hi dānaṃ bahudhā pasatthaṃ,_  
_Dadato ca te akkhayadhammamatthu;_  
_Paṭigaṇhāmi te vatthayugāni aṭṭha,_  
_Yakkhassimā gacchantu dakkhiṇāyo”ti._  

</Paragraph>

</Division>
<Division number="569" book="pv" bookNo="19" page="188">

<Paragraph number="616" type="verses" book="pv">
_Tato hi so ācamayitvā licchavi,_  
_Therassa datvāna yugāni aṭṭha;_  
_Paṭiggahitāni ca tāni assu,_  
_Yakkhañca passetha sannaddhadussaṃ._  

</Paragraph>

</Division>
<Division number="570" book="pv" bookNo="19" page="188">

<Paragraph number="617" type="verses" book="pv">
_Tamaddasā candanasāralittaṃ,_  
_Ājaññamārūḷhamuḷāravaṇṇaṃ;_  
_Alaṅkataṃ sādhunivatthadussaṃ,_  
_Parivāritaṃ yakkhamahiddhipattaṃ._  

</Paragraph>

</Division>
<Division number="571" book="pv" bookNo="19" page="188">

<Paragraph number="618" type="verses" book="pv">
_So taṃ disvā attamano udaggo,_  
_Pahaṭṭhacitto ca subhaggarūpo;_  
_Kammañca disvāna mahāvipākaṃ,_  
_Sandiṭṭhikaṃ cakkhunā sacchikatvā._  

</Paragraph>

</Division>
<Division number="572" book="pv" bookNo="19" page="189">

<Paragraph number="619" type="verses" book="pv">
_Tamenamavoca upasaṅkamitvā,_  
_“Dassāmi dānaṃ samaṇabrāhmaṇānaṃ;_  
_Na cāpi me kiñci adeyyamatthi,_  
_Tuvañca me yakkha bahūpakāro”ti._  

</Paragraph>

</Division>
<Division number="573" book="pv" bookNo="19" page="189">

<Paragraph number="620" type="verses" book="pv">
_“Tuvañca me licchavi ekadesaṃ,_  
_Adāsi dānāni amoghametaṃ;_  
_Svāhaṃ karissāmi tayāva sakkhiṃ,_  
_Amānuso mānusakena saddhin”ti._  

</Paragraph>

</Division>
<Division number="574" book="pv" bookNo="19" page="189">

<Paragraph number="621" type="verses" book="pv">
_“Gatī ca bandhū ca parāyaṇañca,_  
_Mitto mamāsi atha devatā me;_  
_Yācāmi taṃ pañjaliko bhavitvā,_  
_Icchāmi taṃ yakkha punāpi daṭṭhun”ti._  

</Paragraph>

</Division>
<Division number="575" book="pv" bookNo="19" page="189">

<Paragraph number="622" type="verses" book="pv">
