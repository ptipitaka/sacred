<Division number="58" book="pv">

<Paragraph number="68" type="verses" book="pv">
_“Kā nu antovimānasmiṃ,_  
_tiṭṭhantī nūpanikkhami;_  
_Upanikkhamassu bhadde,_  
_passāma taṃ bahiṭṭhitan”ti._  

</Paragraph>

</Division>
<Division number="59" book="pv">

<Paragraph number="69" type="verses" book="pv">
_“Aṭṭīyāmi harāyāmi,_  
_naggā nikkhamituṃ bahi;_  
_Kesehamhi paṭicchannā,_  
_puññaṃ me appakaṃ katan”ti._  

</Paragraph>

</Division>
<Division number="60" book="pv">

<Paragraph number="70" type="verses" book="pv">
_“Handuttarīyaṃ dadāmi te,_  
_idaṃ dussaṃ nivāsaya;_  
_Idaṃ dussaṃ nivāsetvā,_  
_ehi nikkhama sobhane;_  
_Upanikkhamassu bhadde,_  
_passāma taṃ bahiṭṭhitan”ti._  

</Paragraph>

</Division>
<Division number="61" book="pv">

<Paragraph number="71" type="verses" book="pv">
_“Hatthena hatthe te dinnaṃ,_  
_na mayhaṃ upakappati;_  
_Esetthupāsako saddho,_  
_sammāsambuddhasāvako._  

</Paragraph>

</Division>
<Division number="62" book="pv">

<Paragraph number="72" type="verses" book="pv">
_Etaṃ acchādayitvāna,_  
_mama dakkhiṇamādisa;_  
_Tathāhaṃ sukhitā hessaṃ,_  
_sabbakāmasamiddhinī”ti._  

</Paragraph>

</Division>
<Division number="63" book="pv">

<Paragraph number="73" type="verses" book="pv">
_Tañca te nhāpayitvāna,_  
_vilimpetvāna vāṇijā;_  
_Vatthehacchādayitvāna,_  
_tassā dakkhiṇamādisuṃ._  

</Paragraph>

</Division>
<Division number="64" book="pv">

<Paragraph number="74" type="verses" book="pv">
_Samanantarānuddiṭṭhe,_  
_vipāko udapajjatha;_  
_Bhojanacchādanapānīyaṃ,_  
_dakkhiṇāya idaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="65" book="pv">

<Paragraph number="75" type="verses" book="pv">
_Tato suddhā sucivasanā,_  
_kāsikuttamadhārinī;_  
_Hasantī vimānā nikkhami,_  
_dakkhiṇāya idaṃ phalanti._  

</Paragraph>

</Division>
<Division number="66" book="pv">

<Paragraph number="76" type="verses" book="pv">
_“Sucittarūpaṃ ruciraṃ,_  
_vimānaṃ te pabhāsati;_  
_Devate pucchitācikkha,_  
_kissa kammassidaṃ phalan”ti._  

</Paragraph>

</Division>
<Division number="67" book="pv">

<Paragraph number="77" type="verses" book="pv">
_“Bhikkhuno caramānassa,_  
_doṇinimmajjaniṃ ahaṃ;_  
_Adāsiṃ ujubhūtassa,_  
_vippasannena cetasā._  

</Paragraph>

</Division>
<Division number="68" book="pv">

<Paragraph number="78" type="verses" book="pv">
_Tassa kammassa kusalassa,_  
_vipākaṃ dīghamantaraṃ;_  
_Anubhomi vimānasmiṃ,_  
_tañca dāni parittakaṃ._  

</Paragraph>

</Division>
<Division number="69" book="pv">

<Paragraph number="79" type="verses" book="pv">
_Uddhaṃ catūhi māsehi,_  
_kālakiriyā bhavissati;_  
_Ekantakaṭukaṃ ghoraṃ,_  
_nirayaṃ papatissahaṃ._  

</Paragraph>

</Division>
<Division number="70" book="pv">

<Paragraph number="80" type="verses" book="pv">
_Catukkaṇṇaṃ catudvāraṃ,_  
_vibhattaṃ bhāgaso mitaṃ;_  
_Ayopākārapariyantaṃ,_  
_ayasā paṭikujjitaṃ._  

</Paragraph>

</Division>
<Division number="71" book="pv">

<Paragraph number="81" type="verses" book="pv">
_Tassa ayomayā bhūmi,_  
_jalitā tejasā yutā;_  
_Samantā yojanasataṃ,_  
_pharitvā tiṭṭhati sabbadā._  

</Paragraph>

</Division>
<Division number="72" book="pv">

<Paragraph number="82" type="verses" book="pv">
_Tatthāhaṃ dīghamaddhānaṃ,_  
_dukkhaṃ vedissa vedanaṃ;_  
_Phalañca pāpakammassa,_  
_tasmā socāmahaṃ bhusan”ti._  


</Paragraph>
</Division>

<Paragraph number="83" type="center" book="pv">
Khallāṭiyapetivatthu dasamaṃ.
</Paragraph>
