<Division number="737" book="pv">

<Paragraph number="789" type="verses" book="pv">
_“Idaṃ mama ucchuvanaṃ mahantaṃ,_  
_Nibbattati puññaphalaṃ anappakaṃ;_  
_Taṃ dāni me na paribhogameti,_  
_Ācikkha bhante kissa ayaṃ vipāko._  

</Paragraph>

</Division>
<Division number="738" book="pv">

<Paragraph number="790" type="verses" book="pv">
_Haññāmi khajjāmi ca vāyamāmi,_  
_Parisakkāmi paribhuñjituṃ kiñci;_  
_Svāhaṃ chinnathāmo kapaṇo lālapāmi,_  
_Kissa kammassa ayaṃ vipāko._  

</Paragraph>

</Division>
<Division number="739" book="pv">

<Paragraph number="791" type="verses" book="pv">
_Vighāto cāhaṃ paripatāmi chamāyaṃ,_  
_Parivattāmi vāricarova ghamme;_  
_Rudato ca me assukā niggalanti,_  
_Ācikkha bhante kissa ayaṃ vipāko._  

</Paragraph>

</Division>
<Division number="740" book="pv">

<Paragraph number="792" type="verses" book="pv">
_Chāto kilanto ca pipāsito ca,_  
_Santassito sātasukhaṃ na vinde;_  
_Pucchāmi taṃ etamatthaṃ bhadante,_  
_Kathaṃ nu ucchuparibhogaṃ labheyyan”ti._  

</Paragraph>

</Division>
<Division number="741" book="pv">

<Paragraph number="793" type="verses" book="pv">
_“Pure tuvaṃ kammamakāsi attanā,_  
_Manussabhūto purimāya jātiyā;_  
_Ahañca taṃ etamatthaṃ vadāmi,_  
_Sutvāna tvaṃ etamatthaṃ vijāna._  

</Paragraph>

</Division>
<Division number="742" book="pv">

<Paragraph number="794" type="verses" book="pv">
_Ucchuṃ tuvaṃ khādamāno payāto,_  
_Puriso ca te piṭṭhito anvagacchi;_  
_So ca taṃ paccāsanto kathesi,_  
_Tassa tuvaṃ na kiñci ālapittha._  

</Paragraph>

</Division>
<Division number="743" book="pv">

<Paragraph number="795" type="verses" book="pv">
_So ca taṃ abhaṇantaṃ ayāci,_  
_‘Dehayya ucchun’ti ca taṃ avoca;_  
_Tassa tuvaṃ piṭṭhito ucchuṃ adāsi,_  
_Tassetaṃ kammassa ayaṃ vipāko._  

</Paragraph>

</Division>
<Division number="744" book="pv">

<Paragraph number="796" type="verses" book="pv">
_Iṅgha tvaṃ gantvāna piṭṭhito gaṇheyyāsi,_  
_Gahetvāna taṃ khādassu yāvadatthaṃ;_  
_Teneva tvaṃ attamano bhavissasi,_  
_Haṭṭho cudaggo ca pamodito cā”ti._  

</Paragraph>

</Division>
<Division number="745" book="pv">

<Paragraph number="797" type="verses" book="pv">
_Gantvāna so piṭṭhito aggahesi,_  
_Gahetvāna taṃ khādi yāvadatthaṃ;_  
_Teneva so attamano ahosi,_  
_Haṭṭho cudaggo ca pamodito cāti._  


</Paragraph>
</Division>

<Paragraph number="798" type="center" book="pv">
Ucchupetavatthu pañcamaṃ.
</Paragraph>
