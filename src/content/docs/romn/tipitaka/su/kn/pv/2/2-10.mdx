<Division number="331" book="pv">

<Paragraph number="357" type="verses" book="pv">
_Divāvihāragataṃ bhikkhuṃ,_  
_gaṅgātīre nisinnakaṃ;_  
_Taṃ petī upasaṅkamma,_  
_dubbaṇṇā bhīrudassanā._  

</Paragraph>

</Division>
<Division number="332" book="pv">

<Paragraph number="358" type="verses" book="pv">
_Kesā cassā atidīghā,_  
_yāvabhūmāvalambare;_  
_Kesehi sā paṭicchannā,_  
_samaṇaṃ etadabravi._  

</Paragraph>

</Division>
<Division number="333" book="pv">

<Paragraph number="359" type="verses" book="pv">
_“Pañcapaṇṇāsavassāni,_  
_yato kālaṅkatā ahaṃ;_  
_Nābhijānāmi bhuttaṃ vā,_  
_pītaṃ vā pana pāniyaṃ;_  
_Dehi tvaṃ pāniyaṃ bhante,_  
_tasitā pāniyāya me”ti._  

</Paragraph>

</Division>
<Division number="334" book="pv">

<Paragraph number="360" type="verses" book="pv">
_“Ayaṃ sītodikā gaṅgā,_  
_himavantato sandati;_  
_Piva etto gahetvāna,_  
_kiṃ maṃ yācasi pāniyan”ti._  

</Paragraph>

</Division>
<Division number="335" book="pv">

<Paragraph number="361" type="verses" book="pv">
_“Sacāhaṃ bhante gaṅgāya,_  
_sayaṃ gaṇhāmi pāniyaṃ;_  
_Lohitaṃ me parivattati,_  
_tasmā yācāmi pāniyan”ti._  

</Paragraph>

</Division>
<Division number="336" book="pv">

<Paragraph number="362" type="verses" book="pv">
_“Kiṃ nu kāyena vācāya,_  
_manasā dukkaṭaṃ kataṃ;_  
_Kissa kammavipākena,_  
_gaṅgā te hoti lohitan”ti._  

</Paragraph>

</Division>
<Division number="337" book="pv">

<Paragraph number="363" type="verses" book="pv">
_“Putto me uttaro nāma,_  
_saddho āsi upāsako;_  
_So ca mayhaṃ akāmāya,_  
_samaṇānaṃ pavecchati._  

</Paragraph>

</Division>
<Division number="338" book="pv">

<Paragraph number="364" type="verses" book="pv">
_Cīvaraṃ piṇḍapātañca,_  
_paccayaṃ sayanāsanaṃ;_  
_Tamahaṃ paribhāsāmi,_  
_maccherena upaddutā._  

</Paragraph>

</Division>
<Division number="339" book="pv">

<Paragraph number="365" type="verses" book="pv">
_Yaṃ tvaṃ mayhaṃ akāmāya,_  
_samaṇānaṃ pavecchasi;_  
_Cīvaraṃ piṇḍapātañca,_  
_paccayaṃ sayanāsanaṃ._  

</Paragraph>

</Division>
<Division number="340" book="pv">

<Paragraph number="366" type="verses" book="pv">
_Etaṃ te paralokasmiṃ,_  
_lohitaṃ hotu uttara;_  
_Tassa kammassa vipākena,_  
_gaṅgā me hoti lohitan”ti._  


</Paragraph>
</Division>

<Paragraph number="367" type="center" book="pv">
Uttaramātupetivatthu dasamaṃ.
</Paragraph>
