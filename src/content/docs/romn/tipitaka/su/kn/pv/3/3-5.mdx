<Division number="453" book="pv">

<Paragraph number="490" type="verses" book="pv">
_Accherarūpaṃ sugatassa ñāṇaṃ,_  
_Satthā yathā puggalaṃ byākāsi;_  
_Ussannapuññāpi bhavanti heke,_  
_Parittapuññāpi bhavanti heke._  

</Paragraph>

</Division>
<Division number="454" book="pv">

<Paragraph number="491" type="verses" book="pv">
_Ayaṃ kumāro sīvathikāya chaḍḍito,_  
_Aṅguṭṭhasnehena yāpeti rattiṃ;_  
_Na yakkhabhūtā na sarīsapā vā,_  
_Viheṭhayeyyuṃ katapuññaṃ kumāraṃ._  

</Paragraph>

</Division>
<Division number="455" book="pv">

<Paragraph number="492" type="verses" book="pv">
_Sunakhāpimassa palihiṃsu pāde,_  
_Dhaṅkā siṅgālā parivattayanti;_  
_Gabbhāsayaṃ pakkhigaṇā haranti,_  
_Kākā pana akkhimalaṃ haranti._  

</Paragraph>

</Division>
<Division number="456" book="pv">

<Paragraph number="493" type="verses" book="pv">
_Nayimassa rakkhaṃ vidahiṃsu keci,_  
_Na osadhaṃ sāsapadhūpanaṃ vā;_  
_Nakkhattayogampi na aggahesuṃ,_  
_Na sabbadhaññānipi ākiriṃsu._  

</Paragraph>

</Division>
<Division number="457" book="pv">

<Paragraph number="494" type="verses" book="pv">
_Etādisaṃ uttamakicchapattaṃ,_  
_Rattābhataṃ sīvathikāya chaḍḍitaṃ;_  
_Nonītapiṇḍaṃva pavedhamānaṃ,_  
_Sasaṃsayaṃ jīvitasāvasesaṃ._  

</Paragraph>

</Division>
<Division number="458" book="pv">

<Paragraph number="495" type="verses" book="pv">
_Tamaddasā devamanussapūjito,_  
_Disvā ca taṃ byākari bhūripañño;_  
_“Ayaṃ kumāro nagarassimassa,_  
_Aggakuliko bhavissati bhogato ca”._  

</Paragraph>

</Division>
<Division number="459" book="pv">

<Paragraph number="496" type="verses" book="pv">
_“Kissa vataṃ kiṃ pana brahmacariyaṃ,_  
_Kissa suciṇṇassa ayaṃ vipāko;_  
_Etādisaṃ byasanaṃ pāpuṇitvā,_  
_Taṃ tādisaṃ paccanubhossatiddhin”ti._  

</Paragraph>

</Division>
<Division number="460" book="pv">

<Paragraph number="497" type="verses" book="pv">
_Buddhapamukhassa bhikkhusaṃghassa,_  
_Pūjaṃ akāsi janatā uḷāraṃ;_  
_Tatrassa cittassahu aññathattaṃ,_  
_Vācaṃ abhāsi pharusaṃ asabbhaṃ._  

</Paragraph>

</Division>
<Division number="461" book="pv">

<Paragraph number="498" type="verses" book="pv">
_So taṃ vitakkaṃ pavinodayitvā,_  
_Pītiṃ pasādaṃ paṭiladdhā pacchā;_  
_Tathāgataṃ jetavane vasantaṃ,_  
_Yāguyā upaṭṭhāsi sattarattaṃ._  

</Paragraph>

</Division>
<Division number="462" book="pv">

<Paragraph number="499" type="verses" book="pv">
_Tassa vataṃ taṃ pana brahmacariyaṃ,_  
_Tassa suciṇṇassa ayaṃ vipāko;_  
_Etādisaṃ byasanaṃ pāpuṇitvā,_  
_Taṃ tādisaṃ paccanubhossatiddhiṃ._  

</Paragraph>

</Division>
<Division number="463" book="pv">

<Paragraph number="500" type="verses" book="pv">
_Ṭhatvāna so vassasataṃ idheva,_  
_Sabbehi kāmehi samaṅgibhūto;_  
_Kāyassa bhedā abhisamparāyaṃ,_  
_Sahabyataṃ gacchati vāsavassāti._  


</Paragraph>
</Division>

<Paragraph number="501" type="center" book="pv">
Kumārapetavatthu pañcamaṃ.
</Paragraph>
