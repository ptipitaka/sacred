<Division number="83" book="cp">

<Paragraph number="360" type="verses" book="cp">
_“Punāparaṃ yadā homi,_  
_maccharājā mahāsare;_  
_Uṇhe sūriyasantāpe,_  
_sare udaka khīyatha._  

</Paragraph>

</Division>
<Division number="84" book="cp">

<Paragraph number="361" type="verses" book="cp">
_Tato kākā ca gijjhā ca,_  
_kaṅkā kulalasenakā;_  
_Bhakkhayanti divārattiṃ,_  
_macche upanisīdiya._  

</Paragraph>

</Division>
<Division number="85" book="cp">

<Paragraph number="362" type="verses" book="cp">
_Evaṃ cintesahaṃ tattha,_  
_saha ñātīhi pīḷito;_  
_‘Kena nu kho upāyena,_  
_ñātī dukkhā pamocaye’._  

</Paragraph>

</Division>
<Division number="86" book="cp">

<Paragraph number="363" type="verses" book="cp">
_Vicintayitvā dhammatthaṃ,_  
_saccaṃ addasa passayaṃ;_  
_Sacce ṭhatvā pamocesiṃ,_  
_ñātīnaṃ taṃ atikkhayaṃ._  

</Paragraph>

</Division>
<Division number="87" book="cp">

<Paragraph number="364" type="verses" book="cp">
_Anussaritvā sataṃ dhammaṃ,_  
_paramatthaṃ vicintayaṃ;_  
_Akāsiṃ saccakiriyaṃ,_  
_yaṃ loke dhuvasassataṃ._  

</Paragraph>

</Division>
<Division number="88" book="cp">

<Paragraph number="365" type="verses" book="cp">
_‘Yato sarāmi attānaṃ,_  
_yato pattosmi viññutaṃ;_  
_Nābhijānāmi sañcicca,_  
_ekapāṇampi hiṃsitaṃ._  

</Paragraph>

</Division>
<Division number="89" book="cp">

<Paragraph number="366" type="verses" book="cp">
_Etena saccavajjena,_  
_pajjunno abhivassatu;_  
_Abhitthanaya pajjunna,_  
_nidhiṃ kākassa nāsaya;_  
_Kākaṃ sokāya randhehi,_  
_macche sokā pamocaya’._  

</Paragraph>

</Division>
<Division number="90" book="cp">

<Paragraph number="367" type="verses" book="cp">
_Saha kate saccavare,_  
_pajjunno abhigajjiya;_  
_Thalaṃ ninnañca pūrento,_  
_khaṇena abhivassatha._  

</Paragraph>

</Division>
<Division number="91" book="cp">

<Paragraph number="368" type="verses" book="cp">
_Evarūpaṃ saccavaraṃ,_  
_katvā vīriyamuttamaṃ;_  
_Vassāpesiṃ mahāmeghaṃ,_  
_saccatejabalassito;_  
_Saccena me samo natthi,_  
_esā me saccapāramī”ti._  


</Paragraph>
</Division>

<Paragraph number="369" type="center" book="cp">
Maccharājacariyaṃ dasamaṃ.
</Paragraph>
