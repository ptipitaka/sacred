<Paragraph number="23" type="verses" book="cp">
_“Punāparaṃ yadā homi,_  
_indapatthe puruttame;_  
_Rājā dhanañcayo nāma,_  
_kusale dasahupāgato._  

</Paragraph>

</Division>
<Division number="21" book="cp">

<Paragraph number="24" type="verses" book="cp">
_Kaliṅgaraṭṭhavisayā,_  
_brāhmaṇā upagañchu maṃ;_  
_Āyācuṃ maṃ hatthināgaṃ,_  
_dhaññaṃ maṅgalasammataṃ._  

</Paragraph>

</Division>
<Division number="22" book="cp">

<Paragraph number="25" type="verses" book="cp">
_‘Avuṭṭhiko janapado,_  
_dubbhikkho chātako mahā;_  
_Dadāhi pavaraṃ nāgaṃ,_  
_nīlaṃ añjanasavhayaṃ’._  

</Paragraph>

</Division>
<Division number="23" book="cp">

<Paragraph number="26" type="verses" book="cp">
_‘Na me yācakamanuppatte,_  
_paṭikkhepo anucchavo;_  
_Mā me bhijji samādānaṃ,_  
_dassāmi vipulaṃ gajaṃ’._  

</Paragraph>

</Division>
<Division number="24" book="cp">

<Paragraph number="27" type="verses" book="cp">
_Nāgaṃ gahetvā soṇḍāya,_  
_bhiṅgāre ratanāmaye;_  
_Jalaṃ hatthe ākiritvā,_  
_brāhmaṇānaṃ adaṃ gajaṃ._  

</Paragraph>

</Division>
<Division number="25" book="cp">

<Paragraph number="28" type="verses" book="cp">
_Tassa nāge padinnamhi,_  
_amaccā etadabravuṃ;_  
_‘Kiṃ nu tuyhaṃ varaṃ nāgaṃ,_  
_yācakānaṃ padassasi._  

</Paragraph>

</Division>
<Division number="26" book="cp">

<Paragraph number="29" type="verses" book="cp">
_Dhaññaṃ maṅgalasampannaṃ,_  
_saṅgāmavijayuttamaṃ;_  
_Tasmiṃ nāge padinnamhi,_  
_kiṃ te rajjaṃ karissati’._  

</Paragraph>

</Division>
<Division number="27" book="cp">

<Paragraph number="30" type="verses" book="cp">
_‘Rajjampi me dade sabbaṃ,_  
_sarīraṃ dajjamattano;_  
_Sabbaññutaṃ piyaṃ mayhaṃ,_  
_tasmā nāgaṃ adāsahan’”ti._  


</Paragraph>
</Division>

<Paragraph number="31" type="center" book="cp">
Kururājacariyaṃ tatiyaṃ.
</Paragraph>
