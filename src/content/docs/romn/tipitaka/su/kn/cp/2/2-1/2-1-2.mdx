<Paragraph number="173" type="verses" book="cp">
_“Punāparaṃ yadā homi,_  
_bhūridatto mahiddhiko;_  
_Virūpakkhena mahāraññā,_  
_devalokamagañchahaṃ._  

</Paragraph>

</Division>
<Division number="12" book="cp" bookNo="21" page="400">

<Paragraph number="174" type="verses" book="cp">
_Tattha passitvāhaṃ deve,_  
_ekantaṃ sukhasamappite;_  
_Taṃ saggagamanatthāya,_  
_sīlabbataṃ samādiyiṃ._  

</Paragraph>

</Division>
<Division number="13" book="cp" bookNo="21" page="400">

<Paragraph number="175" type="verses" book="cp">
_Sarīrakiccaṃ katvāna,_  
_bhutvā yāpanamattakaṃ;_  
_Caturo aṅge adhiṭṭhāya,_  
_semi vammikamuddhani._  

</Paragraph>

</Division>
<Division number="14" book="cp" bookNo="21" page="400">

<Paragraph number="176" type="verses" book="cp">
_‘Chaviyā cammena maṃsena,_  
_Nahāruaṭṭhikehi vā;_  
_Yassa etena karaṇīyaṃ,_  
_Dinnaṃyeva harātu so’._  

</Paragraph>

</Division>
<Division number="15" book="cp" bookNo="21" page="400">

<Paragraph number="177" type="verses" book="cp">
_Saṃsito akataññunā,_  
_ālampāyano mamaggahi;_  
_Peḷāya pakkhipitvāna,_  
_kīḷeti maṃ tahiṃ tahiṃ._  

</Paragraph>

</Division>
<Division number="16" book="cp" bookNo="21" page="400">

<Paragraph number="178" type="verses" book="cp">
_Peḷāya pakkhipantepi,_  
_sammaddantepi pāṇinā;_  
_Ālampāyane na kuppāmi,_  
_sīlakhaṇḍabhayā mama._  

</Paragraph>

</Division>
<Division number="17" book="cp" bookNo="21" page="400">

<Paragraph number="179" type="verses" book="cp">
_Sakajīvitapariccāgo,_  
_tiṇato lahuko mama;_  
_Sīlavītikkamo mayhaṃ,_  
_pathavīuppatanaṃ viya._  

</Paragraph>

</Division>
<Division number="18" book="cp" bookNo="21" page="400">

<Paragraph number="180" type="verses" book="cp">
_Nirantaraṃ jātisataṃ,_  
_cajeyyaṃ mama jīvitaṃ;_  
_Neva sīlaṃ pabhindeyyaṃ,_  
_catuddīpāna hetupi._  

</Paragraph>

</Division>
<Division number="19" book="cp" bookNo="21" page="400">

<Paragraph number="181" type="verses" book="cp">
_Api cāhaṃ sīlarakkhāya,_  
_Sīlapāramipūriyā;_  
_Na karomi citte aññathattaṃ,_  
_Pakkhipantampi peḷake”ti._  


</Paragraph>
</Division>

<Paragraph number="182" type="center" book="cp">
Bhūridattacariyaṃ dutiyaṃ.
</Paragraph>
