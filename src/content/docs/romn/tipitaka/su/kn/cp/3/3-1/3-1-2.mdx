<Division number="7" book="cp">

<Paragraph number="276" type="verses" book="cp">
_“Punāparaṃ yadā homi,_  
_indapatthe puruttame;_  
_Kāmito dayito putto,_  
_somanassoti vissuto._  

</Paragraph>

</Division>
<Division number="8" book="cp">

<Paragraph number="277" type="verses" book="cp">
_Sīlavā guṇasampanno,_  
_kalyāṇapaṭibhānavā;_  
_Vuḍḍhāpacāyī hirīmā,_  
_saṅgahesu ca kovido._  

</Paragraph>

</Division>
<Division number="9" book="cp">

<Paragraph number="278" type="verses" book="cp">
_Tassa rañño patikaro,_  
_ahosi kuhakatāpaso;_  
_Ārāmaṃ mālāvacchañca,_  
_ropayitvāna jīvati._  

</Paragraph>

</Division>
<Division number="10" book="cp">

<Paragraph number="279" type="verses" book="cp">
_Tamahaṃ disvāna kuhakaṃ,_  
_thusarāsiṃva ataṇḍulaṃ;_  
_Dumaṃva anto susiraṃ,_  
_kadaliṃva asārakaṃ._  

</Paragraph>

</Division>
<Division number="11" book="cp">

<Paragraph number="280" type="verses" book="cp">
_Natthimassa sataṃ dhammo,_  
_sāmaññāpagato ayaṃ;_  
_Hirīsukkadhammajahito,_  
_jīvitavuttikāraṇā._  

</Paragraph>

</Division>
<Division number="12" book="cp">

<Paragraph number="281" type="verses" book="cp">
_Kupito ahu paccanto,_  
_aṭavīhi parantihi;_  
_Taṃ nisedhetuṃ gacchanto,_  
_anusāsi pitā mamaṃ._  

</Paragraph>

</Division>
<Division number="13" book="cp">

<Paragraph number="282" type="verses" book="cp">
_‘Mā pamajji tuvaṃ tāta,_  
_jaṭilaṃ uggatāpanaṃ;_  
_Yadicchakaṃ pavattehi,_  
_sabbakāmadado hi so’._  

</Paragraph>

</Division>
<Division number="14" book="cp">

<Paragraph number="283" type="verses" book="cp">
_Tamahaṃ gantvānupaṭṭhānaṃ,_  
_Idaṃ vacanamabraviṃ;_  
_‘Kacci te gahapati kusalaṃ,_  
_Kiṃ vā te āharīyatu’._  

</Paragraph>

</Division>
<Division number="15" book="cp">

<Paragraph number="284" type="verses" book="cp">
_Tena so kupito āsi,_  
_kuhako mānanissito;_  
_‘Ghātāpemi tuvaṃ ajja,_  
_raṭṭhā pabbājayāmi vā’._  

</Paragraph>

</Division>
<Division number="16" book="cp">

<Paragraph number="285" type="verses" book="cp">
_Nisedhayitvā paccantaṃ,_  
_rājā kuhakamabravi;_  
_‘Kacci te bhante khamanīyaṃ,_  
_sammāno te pavattito’._  

</Paragraph>

</Division>
<Division number="17" book="cp">

<Paragraph number="286" type="verses" book="cp">
_Tassa ācikkhatī pāpo,_  
_kumāro yathā nāsiyo;_  
_Tassa taṃ vacanaṃ sutvā,_  
_āṇāpesi mahīpati._  

</Paragraph>

</Division>
<Division number="18" book="cp">

<Paragraph number="287" type="verses" book="cp">
_‘Sīsaṃ tattheva chinditvā,_  
_katvāna catukhaṇḍikaṃ;_  
_Rathiyā rathiyaṃ dassetha,_  
_sā gati jaṭilahīḷitā’._  

</Paragraph>

</Division>
<Division number="19" book="cp">

<Paragraph number="288" type="verses" book="cp">
_Tattha kāraṇikā gantvā,_  
_caṇḍā luddā akāruṇā;_  
_Mātu aṅke nisinnassa,_  
_ākaḍḍhitvā nayanti maṃ._  

</Paragraph>

</Division>
<Division number="20" book="cp">

<Paragraph number="289" type="verses" book="cp">
_Tesāhaṃ evamavacaṃ,_  
_bandhataṃ gāḷhabandhanaṃ;_  
_‘Rañño dassetha maṃ khippaṃ,_  
_rājakiriyāni atthi me’._  

</Paragraph>

</Division>
<Division number="21" book="cp">

<Paragraph number="290" type="verses" book="cp">
_Te maṃ rañño dassayiṃsu,_  
_pāpassa pāpasevino;_  
_Disvāna taṃ saññāpesiṃ,_  
_mamañca vasamānayiṃ._  

</Paragraph>

</Division>
<Division number="22" book="cp">

<Paragraph number="291" type="verses" book="cp">
_So maṃ tattha khamāpesi,_  
_mahārajjamadāsi me;_  
_Sohaṃ tamaṃ dālayitvā,_  
_pabbajiṃ anagāriyaṃ._  

</Paragraph>

</Division>
<Division number="23" book="cp">

<Paragraph number="292" type="verses" book="cp">
_Na me dessaṃ mahārajjaṃ,_  
_kāmabhogo na dessiyo;_  
_Sabbaññutaṃ piyaṃ mayhaṃ,_  
_tasmā rajjaṃ pariccajin”ti._  


</Paragraph>
</Division>

<Paragraph number="293" type="center" book="cp">
Somanassacariyaṃ dutiyaṃ.
</Paragraph>
