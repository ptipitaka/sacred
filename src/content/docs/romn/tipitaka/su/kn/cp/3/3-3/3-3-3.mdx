<Division number="72" book="cp">

<Paragraph number="348" type="verses" book="cp">
_“Punāparaṃ yadā homi,_  
_magadhe vaṭṭapotako;_  
_Ajātapakkho taruṇo,_  
_maṃsapesi kulāvake._  

</Paragraph>

</Division>
<Division number="73" book="cp">

<Paragraph number="349" type="verses" book="cp">
_Mukhatuṇḍakenāharitvā,_  
_mātā posayatī mamaṃ;_  
_Tassā phassena jīvāmi,_  
_natthi me kāyikaṃ balaṃ._  

</Paragraph>

</Division>
<Division number="74" book="cp">

<Paragraph number="350" type="verses" book="cp">
_Saṃvacchare gimhasamaye,_  
_davaḍāho padippati;_  
_Upagacchati amhākaṃ,_  
_pāvako kaṇhavattanī._  

</Paragraph>

</Division>
<Division number="75" book="cp">

<Paragraph number="351" type="verses" book="cp">
_Dhamadhamāiti evaṃ,_  
_saddāyanto mahāsikhī;_  
_Anupubbena jhāpento,_  
_aggi mamamupāgami._  

</Paragraph>

</Division>
<Division number="76" book="cp">

<Paragraph number="352" type="verses" book="cp">
_Aggivegabhayātītā,_  
_tasitā mātāpitā mama;_  
_Kulāvake maṃ chaḍḍetvā,_  
_attānaṃ parimocayuṃ._  

</Paragraph>

</Division>
<Division number="77" book="cp">

<Paragraph number="353" type="verses" book="cp">
_Pāde pakkhe pajahāmi,_  
_natthi me kāyikaṃ balaṃ;_  
_Sohaṃ agatiko tattha,_  
_evaṃ cintesahaṃ tadā._  

</Paragraph>

</Division>
<Division number="78" book="cp">

<Paragraph number="354" type="verses" book="cp">
_‘Yesāhaṃ upadhāveyyaṃ,_  
_bhīto tasitavedhito;_  
_Te maṃ ohāya pakkantā,_  
_kathaṃ me ajja kātave._  

</Paragraph>

</Division>
<Division number="79" book="cp">

<Paragraph number="355" type="verses" book="cp">
_Atthi loke sīlaguṇo,_  
_saccaṃ soceyyanuddayā;_  
_Tena saccena kāhāmi,_  
_saccakiriyamuttamaṃ._  

</Paragraph>

</Division>
<Division number="80" book="cp">

<Paragraph number="356" type="verses" book="cp">
_Āvejjetvā dhammabalaṃ,_  
_saritvā pubbake jine;_  
_Saccabalamavassāya,_  
_saccakiriyamakāsahaṃ._  

</Paragraph>

</Division>
<Division number="81" book="cp">

<Paragraph number="357" type="verses" book="cp">
_Santi pakkhā apatanā,_  
_santi pādā avañcanā;_  
_Mātāpitā ca nikkhantā,_  
_jātaveda paṭikkama’._  

</Paragraph>

</Division>
<Division number="82" book="cp">

<Paragraph number="358" type="verses" book="cp">
_Sahasacce kate mayhaṃ,_  
_Mahāpajjalito sikhī;_  
_Vajjesi soḷasakarīsāni,_  
_Udakaṃ patvā yathā sikhī;_  
_Saccena me samo natthi,_  
_Esā me saccapāramī”ti._  


</Paragraph>
</Division>

<Paragraph number="359" type="center" book="cp">
Vaṭṭapotakacariyaṃ navamaṃ.
</Paragraph>
