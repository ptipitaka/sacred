<Paragraph number="13" type="verses" book="cp">
_“Punāparaṃ yadā homi,_  
_Brāhmaṇo saṅkhasavhayo;_  
_Mahāsamuddaṃ taritukāmo,_  
_Upagacchāmi paṭṭanaṃ._  

</Paragraph>

</Division>
<Division number="12" book="cp">

<Paragraph number="14" type="verses" book="cp">
_Tatthaddasaṃ paṭipathe,_  
_Sayambhuṃ aparājitaṃ;_  
_Kantāraddhānaṃ paṭipannaṃ,_  
_Tattāya kaṭhinabhūmiyā._  

</Paragraph>

</Division>
<Division number="13" book="cp">

<Paragraph number="15" type="verses" book="cp">
_Tamahaṃ paṭipathe disvā,_  
_imamatthaṃ vicintayiṃ;_  
_‘Idaṃ khettaṃ anuppattaṃ,_  
_puññakāmassa jantuno._  

</Paragraph>

</Division>
<Division number="14" book="cp">

<Paragraph number="16" type="verses" book="cp">
_Yathā kassako puriso,_  
_khettaṃ disvā mahāgamaṃ;_  
_Tattha bījaṃ na ropeti,_  
_na so dhaññena atthiko._  

</Paragraph>

</Division>
<Division number="15" book="cp">

<Paragraph number="17" type="verses" book="cp">
_Evamevāhaṃ puññakāmo,_  
_Disvā khettavaruttamaṃ;_  
_Yadi tattha kāraṃ na karomi,_  
_Nāhaṃ puññena atthiko._  

</Paragraph>

</Division>
<Division number="16" book="cp">

<Paragraph number="18" type="verses" book="cp">
_Yathā amacco muddikāmo,_  
_rañño antepure jane;_  
_Na deti tesaṃ dhanadhaññaṃ,_  
_muddito parihāyati._  

</Paragraph>

</Division>
<Division number="17" book="cp">

<Paragraph number="19" type="verses" book="cp">
_Evamevāhaṃ puññakāmo,_  
_Vipulaṃ disvāna dakkhiṇaṃ;_  
_Yadi tassa dānaṃ na dadāmi,_  
_Parihāyissāmi puññato’._  

</Paragraph>

</Division>
<Division number="18" book="cp">

<Paragraph number="20" type="verses" book="cp">
_Evāhaṃ cintayitvāna,_  
_orohitvā upāhanā;_  
_Tassa pādāni vanditvā,_  
_adāsiṃ chattupāhanaṃ._  

</Paragraph>

</Division>
<Division number="19" book="cp">

<Paragraph number="21" type="verses" book="cp">
_Tenevāhaṃ sataguṇato,_  
_Sukhumālo sukhedhito;_  
_Api ca dānaṃ paripūrento,_  
_Evaṃ tassa adāsahan”ti._  


</Paragraph>
</Division>

<Paragraph number="22" type="center" book="cp">
Saṅkhacariyaṃ dutiyaṃ.
</Paragraph>
