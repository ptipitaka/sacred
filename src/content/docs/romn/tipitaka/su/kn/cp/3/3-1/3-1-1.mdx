<Paragraph number="269" type="verses" book="cp">
_“Yadāhaṃ amitayaso,_  
_Rājaputto yudhañjayo;_  
_Ussāvabinduṃ sūriyātape,_  
_Patitaṃ disvāna saṃvijiṃ._  

</Paragraph>

</Division>
<Division number="2" book="cp" bookNo="21" page="408">

<Paragraph number="270" type="verses" book="cp">
_Taññevādhipatiṃ katvā,_  
_saṃvegamanubrūhayiṃ;_  
_Mātāpitū ca vanditvā,_  
_pabbajjamanuyācahaṃ._  

</Paragraph>

</Division>
<Division number="3" book="cp" bookNo="21" page="408">

<Paragraph number="271" type="verses" book="cp">
_Yācanti maṃ pañjalikā,_  
_sanegamā saraṭṭhakā;_  
_‘Ajjeva putta paṭipajja,_  
_iddhaṃ phītaṃ mahāmahiṃ’._  

</Paragraph>

</Division>
<Division number="4" book="cp" bookNo="21" page="408">

<Paragraph number="272" type="verses" book="cp">
_Sarājake sahorodhe,_  
_sanegame saraṭṭhake;_  
_Karuṇaṃ paridevante,_  
_anapekkhova pariccajiṃ._  

</Paragraph>

</Division>
<Division number="5" book="cp" bookNo="21" page="408">

<Paragraph number="273" type="verses" book="cp">
_Kevalaṃ pathaviṃ rajjaṃ,_  
_ñātiparijanaṃ yasaṃ;_  
_Cajamāno na cintesiṃ,_  
_bodhiyāyeva kāraṇā._  

</Paragraph>

</Division>
<Division number="6" book="cp" bookNo="21" page="408">

<Paragraph number="274" type="verses" book="cp">
_Mātāpitā na me dessā,_  
_napi me dessaṃ mahāyasaṃ;_  
_Sabbaññutaṃ piyaṃ mayhaṃ,_  
_tasmā rajjaṃ pariccajin”ti._  


</Paragraph>
</Division>

<Paragraph number="275" type="center" book="cp">
Yudhañjayacariyaṃ paṭhamaṃ.
</Paragraph>
