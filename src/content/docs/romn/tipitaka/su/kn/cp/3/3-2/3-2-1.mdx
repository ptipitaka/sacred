<Paragraph number="321" type="verses" book="cp">
_“Punāparaṃ yadā homi,_  
_kāsirājassa atrajo;_  
_Mūgapakkhoti nāmena,_  
_temiyoti vadanti maṃ._  

</Paragraph>

</Division>
<Division number="49" book="cp" bookNo="21" page="412">

<Paragraph number="322" type="verses" book="cp">
_Soḷasitthisahassānaṃ,_  
_na vijjati pumo tadā;_  
_Ahorattānaṃ accayena,_  
_nibbatto ahamekako._  

</Paragraph>

</Division>
<Division number="50" book="cp" bookNo="21" page="412">

<Paragraph number="323" type="verses" book="cp">
_Kicchā laddhaṃ piyaṃ puttaṃ,_  
_abhijātaṃ jutindharaṃ;_  
_Setacchattaṃ dhārayitvāna,_  
_sayane poseti maṃ pitā._  

</Paragraph>

</Division>
<Division number="51" book="cp" bookNo="21" page="412">

<Paragraph number="324" type="verses" book="cp">
_Niddāyamāno sayanavare,_  
_Pabujjhitvānahaṃ tadā;_  
_Addasaṃ paṇḍaraṃ chattaṃ,_  
_Yenāhaṃ nirayaṃ gato._  

</Paragraph>

</Division>
<Division number="52" book="cp" bookNo="21" page="412">

<Paragraph number="325" type="verses" book="cp">
_Saha diṭṭhassa me chattaṃ,_  
_tāso uppajji bheravo;_  
_Vinicchayaṃ samāpanno,_  
_‘kathāhaṃ imaṃ muñcissaṃ’._  

</Paragraph>

</Division>
<Division number="53" book="cp" bookNo="21" page="412">

<Paragraph number="326" type="verses" book="cp">
_Pubbasālohitā mayhaṃ,_  
_devatā atthakāminī;_  
_Sā maṃ disvāna dukkhitaṃ,_  
_tīsu ṭhānesu yojayi._  

</Paragraph>

</Division>
<Division number="54" book="cp" bookNo="21" page="412">

<Paragraph number="327" type="verses" book="cp">
_‘Mā paṇḍiccayaṃ vibhāvaya,_  
_Bālamato bhava sabbapāṇinaṃ;_  
_Sabbo taṃ jano ocināyatu,_  
_Evaṃ tava attho bhavissati’._  

</Paragraph>

</Division>
<Division number="55" book="cp" bookNo="21" page="412">

<Paragraph number="328" type="verses" book="cp">
_Evaṃ vuttāyahaṃ tassā,_  
_idaṃ vacanamabraviṃ;_  
_‘Karomi te taṃ vacanaṃ,_  
_yaṃ tvaṃ bhaṇasi devate;_  
_Atthakāmāsi me amma,_  
_hitakāmāsi devate’._  

</Paragraph>

</Division>
<Division number="56" book="cp" bookNo="21" page="412">

<Paragraph number="329" type="verses" book="cp">
_Tassāhaṃ vacanaṃ sutvā,_  
_sāgareva thalaṃ labhiṃ;_  
_Haṭṭho saṃviggamānaso,_  
_tayo aṅge adhiṭṭhahiṃ._  

</Paragraph>

</Division>
<Division number="57" book="cp" bookNo="21" page="413">

<Paragraph number="330" type="verses" book="cp">
_Mūgo ahosiṃ badhiro,_  
_pakkho gativivajjito;_  
_Ete aṅge adhiṭṭhāya,_  
_vassāni soḷasaṃ vasiṃ._  

</Paragraph>

</Division>
<Division number="58" book="cp" bookNo="21" page="413">

<Paragraph number="331" type="verses" book="cp">
_Tato me hatthapāde ca,_  
_jivhaṃ sotañca maddiya;_  
_Anūnataṃ me passitvā,_  
_‘kāḷakaṇṇī’ti nindisuṃ._  

</Paragraph>

</Division>
<Division number="59" book="cp" bookNo="21" page="413">

<Paragraph number="332" type="verses" book="cp">
_Tato jānapadā sabbe,_  
_senāpatipurohitā;_  
_Sabbe ekamanā hutvā,_  
_chaḍḍanaṃ anumodisuṃ._  

</Paragraph>

</Division>
<Division number="60" book="cp" bookNo="21" page="413">

<Paragraph number="333" type="verses" book="cp">
_Sohaṃ tesaṃ matiṃ sutvā,_  
_haṭṭho saṃviggamānaso;_  
_‘Yassatthāya tapociṇṇo,_  
_so me attho samijjhatha’._  

</Paragraph>

</Division>
<Division number="61" book="cp" bookNo="21" page="413">

<Paragraph number="334" type="verses" book="cp">
_Nhāpetvā anulimpitvā,_  
_veṭhetvā rājaveṭhanaṃ;_  
_Chattena abhisiñcitvā,_  
_kāresuṃ puraṃ padakkhiṇaṃ._  

</Paragraph>

</Division>
<Division number="62" book="cp" bookNo="21" page="413">

<Paragraph number="335" type="verses" book="cp">
_Sattāhaṃ dhārayitvāna,_  
_uggate ravimaṇḍale;_  
_Rathena maṃ nīharitvā,_  
_sārathī vanamupāgami._  

</Paragraph>

</Division>
<Division number="63" book="cp" bookNo="21" page="413">

<Paragraph number="336" type="verses" book="cp">
_Ekokāse rathaṃ katvā,_  
_sajjassaṃ hatthamuccito;_  
_Sārathī khaṇatī kāsuṃ,_  
_nikhātuṃ pathaviyā mamaṃ._  

</Paragraph>

</Division>
<Division number="64" book="cp" bookNo="21" page="413">

<Paragraph number="337" type="verses" book="cp">
_Adhiṭṭhitamadhiṭṭhānaṃ,_  
_tajjento vividhakāraṇā;_  
_Na bhindiṃ tamadhiṭṭhānaṃ,_  
_bodhiyāyeva kāraṇā._  

</Paragraph>

</Division>
<Division number="65" book="cp" bookNo="21" page="413">

<Paragraph number="338" type="verses" book="cp">
_Mātāpitā na me dessā,_  
_attā me na ca dessiyo;_  
_Sabbaññutaṃ piyaṃ mayhaṃ,_  
_tasmā vatamadhiṭṭhahiṃ._  

</Paragraph>

</Division>
<Division number="66" book="cp" bookNo="21" page="413">

<Paragraph number="339" type="verses" book="cp">
_Ete aṅge adhiṭṭhāya,_  
_Vassāni soḷasaṃ vasiṃ;_  
_Adhiṭṭhānena me samo natthi,_  
_Esā me adhiṭṭhānapāramī”ti._  


</Paragraph>
</Division>

<Paragraph number="340" type="center" book="cp">
Temiyacariyaṃ chaṭṭhaṃ.
</Paragraph>
