<Division number="48" book="cp">

<Paragraph number="214" type="verses" book="cp">
_“Punāparaṃ yadā homi,_  
_sutattakanakasannibho;_  
_Migarājā ruru nāma,_  
_paramasīlasamāhito._  

</Paragraph>

</Division>
<Division number="49" book="cp">

<Paragraph number="215" type="verses" book="cp">
_Ramme padese ramaṇīye,_  
_vivitte amanussake;_  
_Tattha vāsaṃ upagañchiṃ,_  
_gaṅgākūle manorame._  

</Paragraph>

</Division>
<Division number="50" book="cp">

<Paragraph number="216" type="verses" book="cp">
_Atha upari gaṅgāya,_  
_dhanikehi paripīḷito;_  
_Puriso gaṅgāya papati,_  
_‘jīvāmi vā marāmi vā’._  

</Paragraph>

</Division>
<Division number="51" book="cp">

<Paragraph number="217" type="verses" book="cp">
_Rattindivaṃ so gaṅgāya,_  
_vuyhamāno mahodake;_  
_Ravanto karuṇaṃ ravaṃ,_  
_majjhe gaṅgāya gacchati._  

</Paragraph>

</Division>
<Division number="52" book="cp">

<Paragraph number="218" type="verses" book="cp">
_Tassāhaṃ saddaṃ sutvāna,_  
_karuṇaṃ paridevato;_  
_Gaṅgāya tīre ṭhatvāna,_  
_apucchiṃ ‘kosi tvaṃ naro’._  

</Paragraph>

</Division>
<Division number="53" book="cp">

<Paragraph number="219" type="verses" book="cp">
_So me puṭṭho ca byākāsi,_  
_attano karaṇaṃ tadā;_  
_‘Dhanikehi bhīto tasito,_  
_pakkhandohaṃ mahānadiṃ’._  

</Paragraph>

</Division>
<Division number="54" book="cp">

<Paragraph number="220" type="verses" book="cp">
_Tassa katvāna kāruññaṃ,_  
_cajitvā mama jīvitaṃ;_  
_Pavisitvā nīhariṃ tassa,_  
_andhakāramhi rattiyā._  

</Paragraph>

</Division>
<Division number="55" book="cp">

<Paragraph number="221" type="verses" book="cp">
_Assatthakālamaññāya,_  
_tassāhaṃ idamabraviṃ;_  
_‘Ekaṃ taṃ varaṃ yācāmi,_  
_mā maṃ kassaci pāvada’._  

</Paragraph>

</Division>
<Division number="56" book="cp">

<Paragraph number="222" type="verses" book="cp">
_Nagaraṃ gantvāna ācikkhi,_  
_pucchito dhanahetuko;_  
_Rājānaṃ so gahetvāna,_  
_upagañchi mamantikaṃ._  

</Paragraph>

</Division>
<Division number="57" book="cp">

<Paragraph number="223" type="verses" book="cp">
_Yāvatā karaṇaṃ sabbaṃ,_  
_rañño ārocitaṃ mayā;_  
_Rājā sutvāna vacanaṃ,_  
_usuṃ tassa pakappayi;_  
_‘Idheva ghātayissāmi,_  
_mittadubbhiṃ anāriyaṃ’._  

</Paragraph>

</Division>
<Division number="58" book="cp">

<Paragraph number="224" type="verses" book="cp">
_Tamahaṃ anurakkhanto,_  
_nimminiṃ mama attanā;_  
_‘Tiṭṭhateso mahārāja,_  
_kāmakāro bhavāmi te’._  

</Paragraph>

</Division>
<Division number="59" book="cp">

<Paragraph number="225" type="verses" book="cp">
_Anurakkhiṃ mama sīlaṃ,_  
_nārakkhiṃ mama jīvitaṃ;_  
_Sīlavā hi tadā āsiṃ,_  
_bodhiyāyeva kāraṇā”ti._  


</Paragraph>
</Division>

<Paragraph number="226" type="center" book="cp">
Rurumigarājacariyaṃ chaṭṭhaṃ.
</Paragraph>
