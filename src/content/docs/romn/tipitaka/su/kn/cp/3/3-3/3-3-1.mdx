<Division number="67" book="cp">

<Paragraph number="341" type="verses" book="cp">
_“Yadā ahaṃ kapi āsiṃ,_  
_nadīkūle darīsaye;_  
_Pīḷito susumārena,_  
_gamanaṃ na labhāmahaṃ._  

</Paragraph>

</Division>
<Division number="68" book="cp">

<Paragraph number="342" type="verses" book="cp">
_Yamhokāse ahaṃ ṭhatvā,_  
_orā pāraṃ patāmahaṃ;_  
_Tatthacchi sattu vadhako,_  
_kumbhīlo luddadassano._  

</Paragraph>

</Division>
<Division number="69" book="cp">

<Paragraph number="343" type="verses" book="cp">
_So maṃ asaṃsi ‘ehī’ti,_  
_‘ahampemī’ti taṃ vatiṃ;_  
_Tassa matthakamakkamma,_  
_parakūle patiṭṭhahiṃ._  

</Paragraph>

</Division>
<Division number="70" book="cp">

<Paragraph number="344" type="verses" book="cp">
_Na tassa alikaṃ bhaṇitaṃ,_  
_yathā vācaṃ akāsahaṃ;_  
_Saccena me samo natthi,_  
_esā me saccapāramī”ti._  


</Paragraph>
</Division>

<Paragraph number="345" type="center" book="cp">
Kapirājacariyaṃ sattamaṃ.
</Paragraph>
