<Paragraph number="243" type="verses" book="cp">
_“Pañcālaraṭṭhe nagaravare,_  
_Kapilāyaṃ puruttame;_  
_Rājā jayaddiso nāma,_  
_Sīlaguṇamupāgato._  

</Paragraph>

</Division>
<Division number="75" book="cp">

<Paragraph number="244" type="verses" book="cp">
_Tassa rañño ahaṃ putto,_  
_Sutadhammo susīlavā;_  
_Alīnasatto guṇavā,_  
_Anurakkhaparijano sadā._  

</Paragraph>

</Division>
<Division number="76" book="cp">

<Paragraph number="245" type="verses" book="cp">
_Pitā me migavaṃ gantvā,_  
_porisādaṃ upāgami;_  
_So me pitumaggahesi,_  
_‘bhakkhosi mama mā cali’._  

</Paragraph>

</Division>
<Division number="77" book="cp">

<Paragraph number="246" type="verses" book="cp">
_Tassa taṃ vacanaṃ sutvā,_  
_bhīto tasitavedhito;_  
_Ūrukkhambho ahu tassa,_  
_disvāna porisādakaṃ._  

</Paragraph>

</Division>
<Division number="78" book="cp">

<Paragraph number="247" type="verses" book="cp">
_Migavaṃ gahetvā muñcassu,_  
_katvā āgamanaṃ puna;_  
_Brāhmaṇassa dhanaṃ datvā,_  
_pitā āmantayī mamaṃ._  

</Paragraph>

</Division>
<Division number="79" book="cp">

<Paragraph number="248" type="verses" book="cp">
_‘Rajjaṃ putta paṭipajja,_  
_mā pamajji puraṃ idaṃ;_  
_Kataṃ me porisādena,_  
_mama āgamanaṃ puna’._  

</Paragraph>

</Division>
<Division number="80" book="cp">

<Paragraph number="249" type="verses" book="cp">
_Mātāpitū ca vanditvā,_  
_nimminitvāna attanā;_  
_Nikkhipitvā dhanuṃ khaggaṃ,_  
_porisādaṃ upāgamiṃ._  

</Paragraph>

</Division>
<Division number="81" book="cp">

<Paragraph number="250" type="verses" book="cp">
_Sasatthahatthūpagataṃ,_  
_kadāci so tasissati;_  
_Tena bhijjissati sīlaṃ,_  
_parittāsaṃ kate mayi._  

</Paragraph>

</Division>
<Division number="82" book="cp">

<Paragraph number="251" type="verses" book="cp">
_Sīlakhaṇḍabhayā mayhaṃ,_  
_tassa dessaṃ na byāhariṃ;_  
_Mettacitto hitavādī,_  
_idaṃ vacanamabraviṃ._  

</Paragraph>

</Division>
<Division number="83" book="cp">

<Paragraph number="252" type="verses" book="cp">
_‘Ujjālehi mahāaggiṃ,_  
_papatissāmi rukkhato;_  
_Tvaṃ pakkakālamaññāya,_  
_bhakkhaya maṃ pitāmaha’._  

</Paragraph>

</Division>
<Division number="84" book="cp">

<Paragraph number="253" type="verses" book="cp">
_Iti sīlavataṃ hetu,_  
_nārakkhiṃ mama jīvitaṃ;_  
_Pabbājesiṃ cahaṃ tassa,_  
_sadā pāṇātipātikan”ti._  


</Paragraph>
</Division>

<Paragraph number="254" type="center" book="cp">
Alīnasattucariyaṃ navamaṃ.
</Paragraph>
