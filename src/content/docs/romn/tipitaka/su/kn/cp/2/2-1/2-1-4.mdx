<Division number="26" book="cp">

<Paragraph number="190" type="verses" book="cp">
_“Punāparaṃ yadā homi,_  
_cūḷabodhi susīlavā;_  
_Bhavaṃ disvāna bhayato,_  
_nekkhammaṃ abhinikkhamiṃ._  

</Paragraph>

</Division>
<Division number="27" book="cp">

<Paragraph number="191" type="verses" book="cp">
_Yā me dutiyikā āsi,_  
_brāhmaṇī kanakasannibhā;_  
_Sāpi vaṭṭe anapekkhā,_  
_nekkhammaṃ abhinikkhami._  

</Paragraph>

</Division>
<Division number="28" book="cp">

<Paragraph number="192" type="verses" book="cp">
_Nirālayā chinnabandhū,_  
_anapekkhā kule gaṇe;_  
_Carantā gāmanigamaṃ,_  
_bārāṇasimupāgamuṃ._  

</Paragraph>

</Division>
<Division number="29" book="cp">

<Paragraph number="193" type="verses" book="cp">
_Tattha vasāma nipakā,_  
_asaṃsaṭṭhā kule gaṇe;_  
_Nirākule appasadde,_  
_rājuyyāne vasāmubho._  

</Paragraph>

</Division>
<Division number="30" book="cp">

<Paragraph number="194" type="verses" book="cp">
_Uyyānadassanaṃ gantvā,_  
_rājā addasa brāhmaṇiṃ;_  
_Upagamma mamaṃ pucchi,_  
_‘tuyhesā kā kassa bhariyā’._  

</Paragraph>

</Division>
<Division number="31" book="cp">

<Paragraph number="195" type="verses" book="cp">
_Evaṃ vutte, ahaṃ tassa,_  
_idaṃ vacanamabraviṃ;_  
_‘Na mayhaṃ bhariyā esā,_  
_sahadhammā ekasāsanī’._  

</Paragraph>

</Division>
<Division number="32" book="cp">

<Paragraph number="196" type="verses" book="cp">
_Tissā sārattagadhito,_  
_gāhāpetvāna ceṭake;_  
_Nippīḷayanto balasā,_  
_antepuraṃ pavesayi._  

</Paragraph>

</Division>
<Division number="33" book="cp">

<Paragraph number="197" type="verses" book="cp">
_Odapattakiyā mayhaṃ,_  
_sahajā ekasāsanī;_  
_Ākaḍḍhitvā nayantiyā,_  
_kopo me upapajjatha._  

</Paragraph>

</Division>
<Division number="34" book="cp">

<Paragraph number="198" type="verses" book="cp">
_Saha kope samuppanne,_  
_sīlabbatamanussariṃ;_  
_Tattheva kopaṃ niggaṇhiṃ,_  
_nādāsiṃ vaḍḍhitūpari._  

</Paragraph>

</Division>
<Division number="35" book="cp">

<Paragraph number="199" type="verses" book="cp">
_Yadi naṃ brāhmaṇiṃ koci,_  
_koṭṭeyya tiṇhasattiyā;_  
_Neva sīlaṃ pabhindeyyaṃ,_  
_bodhiyāyeva kāraṇā._  

</Paragraph>

</Division>
<Division number="36" book="cp">

<Paragraph number="200" type="verses" book="cp">
_Na mesā brāhmaṇī dessā,_  
_napi me balaṃ na vijjati;_  
_Sabbaññutaṃ piyaṃ mayhaṃ,_  
_tasmā sīlānurakkhisan”ti._  


</Paragraph>
</Division>

<Paragraph number="201" type="center" book="cp">
Cūḷabodhicariyaṃ catutthaṃ.
</Paragraph>
