<Division number="24" book="cp">

<Paragraph number="294" type="verses" book="cp">
_“Punāparaṃ yadā homi,_  
_kāsirājassa atrajo;_  
_Ayogharamhi saṃvaḍḍho,_  
_nāmenāsi ayogharo._  

</Paragraph>

</Division>
<Division number="25" book="cp">

<Paragraph number="295" type="verses" book="cp">
_‘Dukkhena jīvito laddho,_  
_saṃpīḷe patiposito;_  
_Ajjeva putta paṭipajja,_  
_kevalaṃ vasudhaṃ imaṃ’._  

</Paragraph>

</Division>
<Division number="26" book="cp">

<Paragraph number="296" type="verses" book="cp">
_Saraṭṭhakaṃ sanigamaṃ,_  
_sajanaṃ vanditva khattiyaṃ;_  
_Añjaliṃ paggahetvāna,_  
_idaṃ vacanamabraviṃ._  

</Paragraph>

</Division>
<Division number="27" book="cp">

<Paragraph number="297" type="verses" book="cp">
_‘Ye keci mahiyā sattā,_  
_hīnamukkaṭṭhamajjhimā;_  
_Nirārakkhā sake gehe,_  
_vaḍḍhanti sakañātibhi._  

</Paragraph>

</Division>
<Division number="28" book="cp">

<Paragraph number="298" type="verses" book="cp">
_Idaṃ loke uttariyaṃ,_  
_saṃpīḷe mama posanaṃ;_  
_Ayogharamhi saṃvaḍḍho,_  
_appabhe candasūriye._  

</Paragraph>

</Division>
<Division number="29" book="cp">

<Paragraph number="299" type="verses" book="cp">
_Pūtikuṇapasampuṇṇā,_  
_muccitvā mātukucchito;_  
_Tato ghoratare dukkhe,_  
_puna pakkhittayoghare._  

</Paragraph>

</Division>
<Division number="30" book="cp">

<Paragraph number="300" type="verses" book="cp">
_Yadihaṃ tādisaṃ patvā,_  
_dukkhaṃ paramadāruṇaṃ;_  
_Rajjesu yadi rajjāmi,_  
_pāpānaṃ uttamo siyaṃ._  

</Paragraph>

</Division>
<Division number="31" book="cp">

<Paragraph number="301" type="verses" book="cp">
_Ukkaṇṭhitomhi kāyena,_  
_rajjenamhi anatthiko;_  
_Nibbutiṃ pariyesissaṃ,_  
_yattha maṃ maccu na maddiye’._  

</Paragraph>

</Division>
<Division number="32" book="cp">

<Paragraph number="302" type="verses" book="cp">
_Evāhaṃ cintayitvāna,_  
_viravante mahājane;_  
_Nāgova bandhanaṃ chetvā,_  
_pāvisiṃ kānanaṃ vanaṃ._  

</Paragraph>

</Division>
<Division number="33" book="cp">

<Paragraph number="303" type="verses" book="cp">
_Mātāpitā na me dessā,_  
_napi me dessaṃ mahāyasaṃ;_  
_Sabbaññutaṃ piyaṃ mayhaṃ,_  
_tasmā rajjaṃ pariccajin”ti._  


</Paragraph>
</Division>

<Paragraph number="304" type="center" book="cp">
Ayogharacariyaṃ tatiyaṃ.
</Paragraph>
