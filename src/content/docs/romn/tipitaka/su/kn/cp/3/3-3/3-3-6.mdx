<Division number="105" book="cp">

<Paragraph number="384" type="verses" book="cp">
_“Punāparaṃ yadā homi,_  
_sutasomo mahīpati;_  
_Gahito porisādena,_  
_brāhmaṇe saṅgaraṃ sariṃ._  

</Paragraph>

</Division>
<Division number="106" book="cp">

<Paragraph number="385" type="verses" book="cp">
_Khattiyānaṃ ekasataṃ,_  
_āvuṇitvā karattale;_  
_Etesaṃ pamilāpetvā,_  
_yaññatthe upanayī mamaṃ._  

</Paragraph>

</Division>
<Division number="107" book="cp">

<Paragraph number="386" type="verses" book="cp">
_Apucchi maṃ porisādo,_  
_‘kiṃ tvaṃ icchasi nissajaṃ;_  
_Yathāmati te kāhāmi,_  
_yadi me tvaṃ punehisi’._  

</Paragraph>

</Division>
<Division number="108" book="cp">

<Paragraph number="387" type="verses" book="cp">
_Tassa paṭissuṇitvāna,_  
_paṇhe āgamanaṃ mama;_  
_Upagantvā puraṃ rammaṃ,_  
_rajjaṃ niyyādayiṃ tadā._  

</Paragraph>

</Division>
<Division number="109" book="cp">

<Paragraph number="388" type="verses" book="cp">
_Anussaritvā sataṃ dhammaṃ,_  
_pubbakaṃ jinasevitaṃ;_  
_Brāhmaṇassa dhanaṃ datvā,_  
_porisādaṃ upāgamiṃ._  

</Paragraph>

</Division>
<Division number="110" book="cp">

<Paragraph number="389" type="verses" book="cp">
_Natthi me saṃsayo tattha,_  
_Ghātayissati vā na vā;_  
_Saccavācānurakkhanto,_  
_Jīvitaṃ cajitumupāgamiṃ;_  
_Saccena me samo natthi,_  
_Esā me saccapāramī”ti._  


</Paragraph>
</Division>

<Paragraph number="390" type="center" book="cp">
Sutasomacariyaṃ dvādasamaṃ.
</Paragraph>
