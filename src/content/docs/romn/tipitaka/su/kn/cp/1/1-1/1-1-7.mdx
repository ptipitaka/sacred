<Paragraph number="52" type="verses" book="cp">
_“Punāparaṃ yadā homi,_  
_ekarājassa atrajo;_  
_Nagare pupphavatiyā,_  
_kumāro candasavhayo._  

</Paragraph>

</Division>
<Division number="46" book="cp" bookNo="21" page="389">

<Paragraph number="53" type="verses" book="cp">
_Tadāhaṃ yajanā mutto,_  
_nikkhanto yaññavāṭato;_  
_Saṃvegaṃ janayitvāna,_  
_mahādānaṃ pavattayiṃ._  

</Paragraph>

</Division>
<Division number="47" book="cp" bookNo="21" page="389">

<Paragraph number="54" type="verses" book="cp">
_Nāhaṃ pivāmi khādāmi,_  
_napi bhuñjāmi bhojanaṃ;_  
_Dakkhiṇeyye adatvāna,_  
_api chappañcarattiyo._  

</Paragraph>

</Division>
<Division number="48" book="cp" bookNo="21" page="390">

<Paragraph number="55" type="verses" book="cp">
_Yathāpi vāṇijo nāma,_  
_Katvāna bhaṇḍasañcayaṃ;_  
_Yattha lābho mahā hoti,_  
_Tattha taṃ harati bhaṇḍakaṃ._  

</Paragraph>

</Division>
<Division number="49" book="cp" bookNo="21" page="390">

<Paragraph number="56" type="verses" book="cp">
_Tatheva sakabhuttāpi,_  
_pare dinnaṃ mahapphalaṃ;_  
_Tasmā parassa dātabbaṃ,_  
_satabhāgo bhavissati._  

</Paragraph>

</Division>
<Division number="50" book="cp" bookNo="21" page="390">

<Paragraph number="57" type="verses" book="cp">
_Etamatthavasaṃ ñatvā,_  
_demi dānaṃ bhavābhave;_  
_Na paṭikkamāmi dānato,_  
_sambodhimanupattiyā”ti._  


</Paragraph>
</Division>

<Paragraph number="58" type="center" book="cp">
Candakumāracariyaṃ sattamaṃ.
</Paragraph>
