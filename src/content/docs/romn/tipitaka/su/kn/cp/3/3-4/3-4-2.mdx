<Paragraph number="395" type="verses" book="cp">
_“Punāparaṃ yadā homi,_  
_ekarājāti vissuto;_  
_Paramaṃ sīlaṃ adhiṭṭhāya,_  
_pasāsāmi mahāmahiṃ._  

</Paragraph>

</Division>
<Division number="115" book="cp" bookNo="21" page="418">

<Paragraph number="396" type="verses" book="cp">
_Dasa kusalakammapathe,_  
_vattāmi anavasesato;_  
_Catūhi saṅgahavatthūhi,_  
_saṅgaṇhāmi mahājanaṃ._  

</Paragraph>

</Division>
<Division number="116" book="cp" bookNo="21" page="418">

<Paragraph number="397" type="verses" book="cp">
_Evaṃ me appamattassa,_  
_idha loke parattha ca;_  
_Dabbaseno upagantvā,_  
_acchindanto puraṃ mama._  

</Paragraph>

</Division>
<Division number="117" book="cp" bookNo="21" page="419">

<Paragraph number="398" type="verses" book="cp">
_Rājūpajīve nigame,_  
_sabalaṭṭhe saraṭṭhake;_  
_Sabbaṃ hatthagataṃ katvā,_  
_kāsuyā nikhaṇī mamaṃ._  

</Paragraph>

</Division>
<Division number="118" book="cp" bookNo="21" page="419">

<Paragraph number="399" type="verses" book="cp">
_Amaccamaṇḍalaṃ rajjaṃ,_  
_phītaṃ antepuraṃ mama;_  
_Acchinditvāna gahitaṃ,_  
_piyaṃ puttaṃva passahaṃ;_  
_Mettāya me samo natthi,_  
_esā me mettāpāramī”ti._  


</Paragraph>
</Division>

<Paragraph number="400" type="center" book="cp">
Ekarājacariyaṃ cuddasamaṃ.
</Paragraph>
