<Paragraph number="135" type="verses" book="cp">
_“Punāparaṃ yadā homi,_  
_Sasako pavanacārako;_  
_Tiṇapaṇṇasākaphalabhakkho,_  
_Paraheṭhanavivajjito._  

</Paragraph>

</Division>
<Division number="126" book="cp">

<Paragraph number="136" type="verses" book="cp">
_Makkaṭo ca siṅgālo ca,_  
_suttapoto cahaṃ tadā;_  
_Vasāma ekasāmantā,_  
_sāyaṃ pāto ca dissare._  

</Paragraph>

</Division>
<Division number="127" book="cp">

<Paragraph number="137" type="verses" book="cp">
_Ahaṃ te anusāsāmi,_  
_kiriye kalyāṇapāpake;_  
_‘Pāpāni parivajjetha,_  
_kalyāṇe abhinivissatha’._  

</Paragraph>

</Division>
<Division number="128" book="cp">

<Paragraph number="138" type="verses" book="cp">
_Uposathamhi divase,_  
_candaṃ disvāna pūritaṃ;_  
_Etesaṃ tattha ācikkhiṃ,_  
_‘divaso ajjuposatho._  

</Paragraph>

</Division>
<Division number="129" book="cp">

<Paragraph number="139" type="verses" book="cp">
_Dānāni paṭiyādetha,_  
_dakkhiṇeyyassa dātave;_  
_Datvā dānaṃ dakkhiṇeyye,_  
_upavassathuposathaṃ’._  

</Paragraph>

</Division>
<Division number="130" book="cp">

<Paragraph number="140" type="verses" book="cp">
_Te me sādhūti vatvāna,_  
_yathāsatti yathābalaṃ;_  
_Dānāni paṭiyādetvā,_  
_dakkhiṇeyyaṃ gavesisuṃ._  

</Paragraph>

</Division>
<Division number="131" book="cp">

<Paragraph number="141" type="verses" book="cp">
_Ahaṃ nisajja cintesiṃ,_  
_dānaṃ dakkhiṇanucchavaṃ;_  
_‘Yadihaṃ labhe dakkhiṇeyyaṃ,_  
_kiṃ me dānaṃ bhavissati._  

</Paragraph>

</Division>
<Division number="132" book="cp">

<Paragraph number="142" type="verses" book="cp">
_Na me atthi tilā muggā,_  
_māsā vā taṇḍulā ghataṃ;_  
_Ahaṃ tiṇena yāpemi,_  
_na sakkā tiṇa dātave._  

</Paragraph>

</Division>
<Division number="133" book="cp">

<Paragraph number="143" type="verses" book="cp">
_Yadi koci eti dakkhiṇeyyo,_  
_Bhikkhāya mama santike;_  
_Dajjāhaṃ sakamattānaṃ,_  
_Na so tuccho gamissati’._  

</Paragraph>

</Division>
<Division number="134" book="cp">

<Paragraph number="144" type="verses" book="cp">
_Mama saṅkappamaññāya,_  
_sakko brāhmaṇavaṇṇinā;_  
_Āsayaṃ me upāgacchi,_  
_dānavīmaṃsanāya me._  

</Paragraph>

</Division>
<Division number="135" book="cp">

<Paragraph number="145" type="verses" book="cp">
_Tamahaṃ disvāna santuṭṭho,_  
_idaṃ vacanamabraviṃ;_  
_‘Sādhu khosi anuppatto,_  
_ghāsahetu mamantike._  

</Paragraph>

</Division>
<Division number="136" book="cp">

<Paragraph number="146" type="verses" book="cp">
_Adinnapubbaṃ dānavaraṃ,_  
_ajja dassāmi te ahaṃ;_  
_Tuvaṃ sīlaguṇūpeto,_  
_ayuttaṃ te paraheṭhanaṃ._  

</Paragraph>

</Division>
<Division number="137" book="cp">

<Paragraph number="147" type="verses" book="cp">
_Ehi aggiṃ padīpehi,_  
_nānākaṭṭhe samānaya;_  
_Ahaṃ pacissamattānaṃ,_  
_pakkaṃ tvaṃ bhakkhayissasi’._  

</Paragraph>

</Division>
<Division number="138" book="cp">

<Paragraph number="148" type="verses" book="cp">
_‘Sādhū’ti so haṭṭhamano,_  
_nānākaṭṭhe samānayi;_  
_Mahantaṃ akāsi citakaṃ,_  
_katvā aṅgāragabbhakaṃ._  

</Paragraph>

</Division>
<Division number="139" book="cp">

<Paragraph number="149" type="verses" book="cp">
_Aggiṃ tattha padīpesi,_  
_yathā so khippaṃ mahā bhave;_  
_Phoṭetvā rajagate gatte,_  
_ekamantaṃ upāvisiṃ._  

</Paragraph>

</Division>
<Division number="140" book="cp">

<Paragraph number="150" type="verses" book="cp">
_Yadā mahākaṭṭhapuñjo,_  
_āditto dhamadhamāyati;_  
_Taduppatitvā papatiṃ,_  
_majjhe jālasikhantare._  

</Paragraph>

</Division>
<Division number="141" book="cp">

<Paragraph number="151" type="verses" book="cp">
_Yathā sītodakaṃ nāma,_  
_Paviṭṭhaṃ yassa kassaci;_  
_Sameti darathapariḷāhaṃ,_  
_Assādaṃ deti pīti ca._  

</Paragraph>

</Division>
<Division number="142" book="cp">

<Paragraph number="152" type="verses" book="cp">
_Tatheva jalitaṃ aggiṃ,_  
_paviṭṭhassa mamaṃ tadā;_  
_Sabbaṃ sameti darathaṃ,_  
_yathā sītodakaṃ viya._  

</Paragraph>

</Division>
<Division number="143" book="cp">

<Paragraph number="153" type="verses" book="cp">
_Chaviṃ cammaṃ maṃsaṃ nhāruṃ,_  
_Aṭṭhiṃ hadayabandhanaṃ;_  
_Kevalaṃ sakalaṃ kāyaṃ,_  
_Brāhmaṇassa adāsahan”ti. (143)_  


</Paragraph>
</Division>

Sasapaṇḍitacariyaṃ dasamaṃ.



</Paragraph>

<Paragraph number="155" type="center" book="cp">
Akittivaggo paṭhamo.



</Paragraph>

Tassuddānaṃ


</Paragraph>

<Paragraph number="157" type="verses" book="cp">
_Akittibrāhmaṇo saṅkho,_  
_kururājā dhanañcayo;_  
_Mahāsudassano rājā,_  
_mahāgovindabrāhmaṇo._  

</Paragraph>

<Paragraph number="158" type="verses" book="cp">
_Nimi candakumāro ca,_  
_sivi vessantaro saso;_  
_Ahameva tadā āsiṃ,_  
_yo te dānavare adā._  

</Paragraph>

<Paragraph number="159" type="verses" book="cp">
_Ete dānaparikkhārā,_  
_ete dānassa pāramī;_  
_Jīvitaṃ yācake datvā,_  
_imaṃ pārami pūrayiṃ._  

</Paragraph>

<Paragraph number="160" type="verses" book="cp">
_Bhikkhāya upagataṃ disvā,_  
_sakattānaṃ pariccajiṃ;_  
_Dānena me samo natthi,_  
_esā me dānapāramīti._  


</Paragraph>
