<Division number="42" book="cp">

<Paragraph number="314" type="verses" book="cp">
_“Punāparaṃ yadā homi,_  
_nagare brahmavaḍḍhane;_  
_Tattha kulavare seṭṭhe,_  
_mahāsāle ajāyahaṃ._  

</Paragraph>

</Division>
<Division number="43" book="cp">

<Paragraph number="315" type="verses" book="cp">
_Tadāpi lokaṃ disvāna,_  
_Andhībhūtaṃ tamotthaṭaṃ;_  
_Cittaṃ bhavato patikuṭati,_  
_Tuttavegahataṃ viya._  

</Paragraph>

</Division>
<Division number="44" book="cp">

<Paragraph number="316" type="verses" book="cp">
_Disvāna vividhaṃ pāpaṃ,_  
_evaṃ cintesahaṃ tadā;_  
_‘Kadāhaṃ gehā nikkhamma,_  
_pavisissāmi kānanaṃ’._  

</Paragraph>

</Division>
<Division number="45" book="cp">

<Paragraph number="317" type="verses" book="cp">
_Tadāpi maṃ nimantesuṃ,_  
_kāmabhogehi ñātayo;_  
_Tesampi chandamācikkhiṃ,_  
_‘mā nimantetha tehi maṃ’._  

</Paragraph>

</Division>
<Division number="46" book="cp">

<Paragraph number="318" type="verses" book="cp">
_Yo me kaniṭṭhako bhātā,_  
_nando nāmāsi paṇḍito;_  
_Sopi maṃ anusikkhanto,_  
_pabbajjaṃ samarocayi._  

</Paragraph>

</Division>
<Division number="47" book="cp">

<Paragraph number="319" type="verses" book="cp">
_Ahaṃ soṇo ca nando ca,_  
_ubho mātāpitā mama;_  
_Tadāpi bhoge chaḍḍetvā,_  
_pāvisimhā mahāvanan”ti._  


</Paragraph>
</Division>

<Paragraph number="320" type="center" book="cp">
Soṇapaṇḍitacariyaṃ pañcamaṃ.
</Paragraph>
