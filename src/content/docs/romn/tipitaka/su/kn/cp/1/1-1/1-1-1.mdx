<Paragraph number="2" type="verses" book="cp">
_“Kappe ca satasahasse,_  
_caturo ca asaṅkhiye;_  
_Etthantare yaṃ caritaṃ,_  
_sabbaṃ taṃ bodhipācanaṃ._  

</Paragraph>

</Division>
<Division number="2" book="cp" bookNo="21" page="385">

<Paragraph number="3" type="verses" book="cp">
_Atītakappe caritaṃ,_  
_ṭhapayitvā bhavābhave;_  
_Imamhi kappe caritaṃ,_  
_pavakkhissaṃ suṇohi me._  

</Paragraph>

</Division>
<Division number="3" book="cp" bookNo="21" page="385">

<Paragraph number="4" type="verses" book="cp">
_Yadā ahaṃ brahāraññe,_  
_suññe vipinakānane;_  
_Ajjhogāhetvā viharāmi,_  
_akitti nāma tāpaso._  

</Paragraph>

</Division>
<Division number="4" book="cp" bookNo="21" page="385">

<Paragraph number="5" type="verses" book="cp">
_Tadā maṃ tapatejena,_  
_santatto tidivābhibhū;_  
_Dhārento brāhmaṇavaṇṇaṃ,_  
_bhikkhāya maṃ upāgami._  

</Paragraph>

</Division>
<Division number="5" book="cp" bookNo="21" page="385">

<Paragraph number="6" type="verses" book="cp">
_Pavanā ābhataṃ paṇṇaṃ,_  
_atelañca aloṇikaṃ;_  
_Mama dvāre ṭhitaṃ disvā,_  
_sakaṭāhena ākiriṃ._  

</Paragraph>

</Division>
<Division number="6" book="cp" bookNo="21" page="385">

<Paragraph number="7" type="verses" book="cp">
_Tassa datvānahaṃ paṇṇaṃ,_  
_nikkujjitvāna bhājanaṃ;_  
_Punesanaṃ jahitvāna,_  
_pāvisiṃ paṇṇasālakaṃ._  

</Paragraph>

</Division>
<Division number="7" book="cp" bookNo="21" page="385">

<Paragraph number="8" type="verses" book="cp">
_Dutiyampi tatiyampi,_  
_upagañchi mamantikaṃ;_  
_Akampito anolaggo,_  
_evamevamadāsahaṃ._  

</Paragraph>

</Division>
<Division number="8" book="cp" bookNo="21" page="386">

<Paragraph number="9" type="verses" book="cp">
_Na me tappaccayā atthi,_  
_sarīrasmiṃ vivaṇṇiyaṃ;_  
_Pītisukhena ratiyā,_  
_vītināmemi taṃ divaṃ._  

</Paragraph>

</Division>
<Division number="9" book="cp" bookNo="21" page="386">

<Paragraph number="10" type="verses" book="cp">
_Yadi māsampi dvemāsaṃ,_  
_dakkhiṇeyyaṃ varaṃ labhe;_  
_Akampito anolīno,_  
_dadeyyaṃ dānamuttamaṃ._  

</Paragraph>

</Division>
<Division number="10" book="cp" bookNo="21" page="386">

<Paragraph number="11" type="verses" book="cp">
_Na tassa dānaṃ dadamāno,_  
_yasaṃ lābhañca patthayiṃ;_  
_Sabbaññutaṃ patthayāno,_  
_tāni kammāni ācarin”ti._  


</Paragraph>
</Division>

<Paragraph number="12" type="center" book="cp">
Akitticariyaṃ paṭhamaṃ.
</Paragraph>
