<Division number="92" book="cp">

<Paragraph number="370" type="verses" book="cp">
_“Punāparaṃ yadā homi,_  
_kaṇhadīpāyano isi;_  
_Paropaññāsavassāni,_  
_anabhiratocariṃ ahaṃ._  

</Paragraph>

</Division>
<Division number="93" book="cp">

<Paragraph number="371" type="verses" book="cp">
_Na koci etaṃ jānāti,_  
_Anabhiratimanaṃ mama;_  
_Ahañhi kassaci nācikkhiṃ,_  
_Arati me carati mānase._  

</Paragraph>

</Division>
<Division number="94" book="cp">

<Paragraph number="372" type="verses" book="cp">
_Sabrahmacārī maṇḍabyo,_  
_sahāyo me mahāisi;_  
_Pubbakammasamāyutto,_  
_sūlamāropanaṃ labhi._  

</Paragraph>

</Division>
<Division number="95" book="cp">

<Paragraph number="373" type="verses" book="cp">
_Tamahaṃ upaṭṭhahitvāna,_  
_ārogyamanupāpayiṃ;_  
_Āpucchitvāna āgañchiṃ,_  
_yaṃ mayhaṃ sakamassamaṃ._  

</Paragraph>

</Division>
<Division number="96" book="cp">

<Paragraph number="374" type="verses" book="cp">
_Sahāyo brāhmaṇo mayhaṃ,_  
_bhariyaṃ ādāya puttakaṃ;_  
_Tayo janā samāgantvā,_  
_āgañchuṃ pāhunāgataṃ._  

</Paragraph>

</Division>
<Division number="97" book="cp">

<Paragraph number="375" type="verses" book="cp">
_Sammodamāno tehi saha,_  
_nisinno sakamassame;_  
_Dārako vaṭṭamanukkhipaṃ,_  
_āsīvisamakopayi._  

</Paragraph>

</Division>
<Division number="98" book="cp">

<Paragraph number="376" type="verses" book="cp">
_Tato so vaṭṭagataṃ maggaṃ,_  
_anvesanto kumārako;_  
_Āsīvisassa hatthena,_  
_uttamaṅgaṃ parāmasi._  

</Paragraph>

</Division>
<Division number="99" book="cp">

<Paragraph number="377" type="verses" book="cp">
_Tassa āmasane kuddho,_  
_sappo visabalassito;_  
_Kupito paramakopena,_  
_aḍaṃsi dārakaṃ khaṇe._  

</Paragraph>

</Division>
<Division number="100" book="cp">

<Paragraph number="378" type="verses" book="cp">
_Sahadaṭṭho āsīvisena,_  
_dārako papati bhūmiyaṃ;_  
_Tenāhaṃ dukkhito āsiṃ,_  
_mama vāhasi taṃ dukkhaṃ._  

</Paragraph>

</Division>
<Division number="101" book="cp">

<Paragraph number="379" type="verses" book="cp">
_Tyāhaṃ assāsayitvāna,_  
_dukkhite sokasallite;_  
_Paṭhamaṃ akāsiṃ kiriyaṃ,_  
_aggaṃ saccaṃ varuttamaṃ._  

</Paragraph>

</Division>
<Division number="102" book="cp">

<Paragraph number="380" type="verses" book="cp">
_‘Sattāhamevāhaṃ pasannacitto,_  
_Puññatthiko acariṃ brahmacariyaṃ;_  
_Athāparaṃ yaṃ caritaṃ mamedaṃ,_  
_Vassāni paññāsasamādhikāni._  

</Paragraph>

</Division>
<Division number="103" book="cp">

<Paragraph number="381" type="verses" book="cp">
_Akāmako vāhi ahaṃ carāmi,_  
_Etena saccena suvatthi hotu;_  
_Hataṃ visaṃ jīvatu yaññadatto’._  

</Paragraph>

</Division>
<Division number="104" book="cp">

<Paragraph number="382" type="verses" book="cp">
_Saha sacce kate mayhaṃ,_  
_visavegena vedhito;_  
_Abujjhitvāna vuṭṭhāsi,_  
_arogo cāsi māṇavo;_  
_Saccena me samo natthi,_  
_esā me saccapāramī”ti._  


</Paragraph>
</Division>

<Paragraph number="383" type="center" book="cp">
Kaṇhadīpāyanacariyaṃ ekādasamaṃ.
</Paragraph>
