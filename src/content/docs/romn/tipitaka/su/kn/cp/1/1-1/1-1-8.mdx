<Paragraph number="59" type="verses" book="cp">
_“Ariṭṭhasavhaye nagare,_  
_sivināmāsi khattiyo;_  
_Nisajja pāsādavare,_  
_evaṃ cintesahaṃ tadā._  

</Paragraph>

</Division>
<Division number="52" book="cp" bookNo="21" page="390">

<Paragraph number="60" type="verses" book="cp">
_‘Yaṃ kiñci mānusaṃ dānaṃ,_  
_adinnaṃ me na vijjati;_  
_Yopi yāceyya maṃ cakkhuṃ,_  
_dadeyyaṃ avikampito’._  

</Paragraph>

</Division>
<Division number="53" book="cp" bookNo="21" page="390">

<Paragraph number="61" type="verses" book="cp">
_Mama saṅkappamaññāya,_  
_sakko devānamissaro;_  
_Nisinno devaparisāya,_  
_idaṃ vacanamabravi._  

</Paragraph>

</Division>
<Division number="54" book="cp" bookNo="21" page="390">

<Paragraph number="62" type="verses" book="cp">
_‘Nisajja pāsādavare,_  
_sivirājā mahiddhiko;_  
_Cintento vividhaṃ dānaṃ,_  
_adeyyaṃ so na passati._  

</Paragraph>

</Division>
<Division number="55" book="cp" bookNo="21" page="390">

<Paragraph number="63" type="verses" book="cp">
_Tathaṃ nu vitathaṃ netaṃ,_  
_handa vīmaṃsayāmi taṃ;_  
_Muhuttaṃ āgameyyātha,_  
_yāva jānāmi taṃ manaṃ’._  

</Paragraph>

</Division>
<Division number="56" book="cp" bookNo="21" page="390">

<Paragraph number="64" type="verses" book="cp">
_Pavedhamāno palitasiro,_  
_Valigatto jarāturo;_  
_Andhavaṇṇova hutvāna,_  
_Rājānaṃ upasaṅkami._  

</Paragraph>

</Division>
<Division number="57" book="cp" bookNo="21" page="390">

<Paragraph number="65" type="verses" book="cp">
_So tadā paggahetvāna,_  
_vāmaṃ dakkhiṇabāhu ca;_  
_Sirasmiṃ añjaliṃ katvā,_  
_idaṃ vacanamabravi._  

</Paragraph>

</Division>
<Division number="58" book="cp" bookNo="21" page="390">

<Paragraph number="66" type="verses" book="cp">
_‘Yācāmi taṃ mahārāja,_  
_dhammika raṭṭhavaḍḍhana;_  
_Tava dānaratā kitti,_  
_uggatā devamānuse._  

</Paragraph>

</Division>
<Division number="59" book="cp" bookNo="21" page="391">

<Paragraph number="67" type="verses" book="cp">
_Ubhopi nettā nayanā,_  
_andhā upahatā mama;_  
_Ekaṃ me nayanaṃ dehi,_  
_tvampi ekena yāpaya’._  

</Paragraph>

</Division>
<Division number="60" book="cp" bookNo="21" page="391">

<Paragraph number="68" type="verses" book="cp">
_Tassāhaṃ vacanaṃ sutvā,_  
_haṭṭho saṃviggamānaso;_  
_Katañjalī vedajāto,_  
_idaṃ vacanamabraviṃ._  

</Paragraph>

</Division>
<Division number="61" book="cp" bookNo="21" page="391">

<Paragraph number="69" type="verses" book="cp">
_‘Idānāhaṃ cintayitvāna,_  
_pāsādato idhāgato;_  
_Tvaṃ mama cittamaññāya,_  
_nettaṃ yācitumāgato._  

</Paragraph>

</Division>
<Division number="62" book="cp" bookNo="21" page="391">

<Paragraph number="70" type="verses" book="cp">
_Aho me mānasaṃ siddhaṃ,_  
_saṅkappo paripūrito;_  
_Adinnapubbaṃ dānavaraṃ,_  
_ajja dassāmi yācake._  

</Paragraph>

</Division>
<Division number="63" book="cp" bookNo="21" page="391">

<Paragraph number="71" type="verses" book="cp">
_Ehi sivaka uṭṭhehi,_  
_mā dandhayi mā pavedhayi;_  
_Ubhopi nayanaṃ dehi,_  
_uppāṭetvā vaṇibbake’._  

</Paragraph>

</Division>
<Division number="64" book="cp" bookNo="21" page="391">

<Paragraph number="72" type="verses" book="cp">
_Tato so codito mayhaṃ,_  
_sivako vacanaṃ karo;_  
_Uddharitvāna pādāsi,_  
_tālamiñjaṃva yācake._  

</Paragraph>

</Division>
<Division number="65" book="cp" bookNo="21" page="391">

<Paragraph number="73" type="verses" book="cp">
_Dadamānassa dentassa,_  
_dinnadānassa me sato;_  
_Cittassa aññathā natthi,_  
_bodhiyāyeva kāraṇā._  

</Paragraph>

</Division>
<Division number="66" book="cp" bookNo="21" page="391">

<Paragraph number="74" type="verses" book="cp">
_Na me dessā ubho cakkhū,_  
_attā na me na dessiyo;_  
_Sabbaññutaṃ piyaṃ mayhaṃ,_  
_tasmā cakkhuṃ adāsahan”ti._  


</Paragraph>
</Division>

<Paragraph number="75" type="center" book="cp">
Sivirājacariyaṃ aṭṭhamaṃ.
</Paragraph>
