<Division number="20" book="cp">

<Paragraph number="183" type="verses" book="cp">
_“Punāparaṃ yadā homi,_  
_campeyyako mahiddhiko;_  
_Tadāpi dhammiko āsiṃ,_  
_sīlabbatasamappito._  

</Paragraph>

</Division>
<Division number="21" book="cp">

<Paragraph number="184" type="verses" book="cp">
_Tadāpi maṃ dhammacāriṃ,_  
_upavutthaṃ uposathaṃ;_  
_Ahituṇḍiko gahetvāna,_  
_rājadvāramhi kīḷati._  

</Paragraph>

</Division>
<Division number="22" book="cp">

<Paragraph number="185" type="verses" book="cp">
_Yaṃ yaṃ so vaṇṇaṃ cintayi,_  
_nīlaṃva pītalohitaṃ;_  
_Tassa cittānuvattanto,_  
_homi cintitasannibho._  

</Paragraph>

</Division>
<Division number="23" book="cp">

<Paragraph number="186" type="verses" book="cp">
_Thalaṃ kareyyamudakaṃ,_  
_udakampi thalaṃ kare;_  
_Yadihaṃ tassa pakuppeyyaṃ,_  
_khaṇena chārikaṃ kare._  

</Paragraph>

</Division>
<Division number="24" book="cp">

<Paragraph number="187" type="verses" book="cp">
_Yadi cittavasī hessaṃ,_  
_parihāyissāmi sīlato;_  
_Sīlena parihīnassa,_  
_uttamattho na sijjhati._  

</Paragraph>

</Division>
<Division number="25" book="cp">

<Paragraph number="188" type="verses" book="cp">
_Kāmaṃ bhijjatuyaṃ kāyo,_  
_idheva vikirīyatu;_  
_Neva sīlaṃ pabhindeyyaṃ,_  
_vikirante bhusaṃ viyā”ti._  


</Paragraph>
</Division>

<Paragraph number="189" type="center" book="cp">
Campeyyanāgacariyaṃ tatiyaṃ.
</Paragraph>
