<Paragraph number="202" type="verses" book="cp">
_“Punāparaṃ yadā homi,_  
_mahiṃso pavanacārako;_  
_Pavaḍḍhakāyo balavā,_  
_mahanto bhīmadassano._  

</Paragraph>

</Division>
<Division number="38" book="cp" bookNo="21" page="402">

<Paragraph number="203" type="verses" book="cp">
_Pabbhāre giridugge ca,_  
_rukkhamūle dakāsaye;_  
_Hotettha ṭhānaṃ mahiṃsānaṃ,_  
_koci koci tahiṃ tahiṃ._  

</Paragraph>

</Division>
<Division number="39" book="cp" bookNo="21" page="402">

<Paragraph number="204" type="verses" book="cp">
_Vicaranto brahāraññe,_  
_ṭhānaṃ addasa bhaddakaṃ;_  
_Taṃ ṭhānaṃ upagantvāna,_  
_tiṭṭhāmi ca sayāmi ca._  

</Paragraph>

</Division>
<Division number="40" book="cp" bookNo="21" page="402">

<Paragraph number="205" type="verses" book="cp">
_Athettha kapimāgantvā,_  
_pāpo anariyo lahu;_  
_Khandhe nalāṭe bhamuke,_  
_mutteti ohadeti taṃ._  

</Paragraph>

</Division>
<Division number="41" book="cp" bookNo="21" page="402">

<Paragraph number="206" type="verses" book="cp">
_Sakimpi divasaṃ dutiyaṃ,_  
_tatiyaṃ catutthampi ca;_  
_Dūseti maṃ sabbakālaṃ,_  
_tena homi upadduto._  

</Paragraph>

</Division>
<Division number="42" book="cp" bookNo="21" page="402">

<Paragraph number="207" type="verses" book="cp">
_Mamaṃ upaddutaṃ disvā,_  
_yakkho maṃ idamabravi;_  
_‘Nāsehetaṃ chavaṃ pāpaṃ,_  
_siṅgehi ca khurehi ca’._  

</Paragraph>

</Division>
<Division number="43" book="cp" bookNo="21" page="402">

<Paragraph number="208" type="verses" book="cp">
_Evaṃ vutte, tadā yakkhe,_  
_ahaṃ taṃ idamabraviṃ;_  
_‘Kiṃ tvaṃ makkhesi kuṇapena,_  
_pāpena anariyena maṃ._  

</Paragraph>

</Division>
<Division number="44" book="cp" bookNo="21" page="403">

<Paragraph number="209" type="verses" book="cp">
_Yadihaṃ tassa pakuppeyyaṃ,_  
_tato hīnataro bhave;_  
_Sīlañca me pabhijjeyya,_  
_viññū ca garaheyyu maṃ._  

</Paragraph>

</Division>
<Division number="45" book="cp" bookNo="21" page="403">

<Paragraph number="210" type="verses" book="cp">
_Hīḷitā jīvitā vāpi,_  
_parisuddhena mataṃ varaṃ;_  
_Kyāhaṃ jīvitahetūpi,_  
_kāhāmi paraheṭhanaṃ’._  

</Paragraph>

</Division>
<Division number="46" book="cp" bookNo="21" page="403">

<Paragraph number="211" type="verses" book="cp">
_Mamevāyaṃ maññamāno,_  
_aññepevaṃ karissati;_  
_Teva tassa vadhissanti,_  
_sā me mutti bhavissati._  

</Paragraph>

</Division>
<Division number="47" book="cp" bookNo="21" page="403">

<Paragraph number="212" type="verses" book="cp">
_Hīnamajjhimaukkaṭṭhe,_  
_sahanto avamānitaṃ;_  
_Evaṃ labhati sappañño,_  
_manasā yathāpatthitan”ti._  


</Paragraph>
</Division>
