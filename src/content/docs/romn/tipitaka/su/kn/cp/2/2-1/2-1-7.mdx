<Paragraph number="227" type="verses" book="cp">
_“Punāparaṃ yadā homi,_  
_jaṭilo uggatāpano;_  
_Mātaṅgo nāma nāmena,_  
_sīlavā susamāhito._  

</Paragraph>

</Division>
<Division number="61" book="cp">

<Paragraph number="228" type="verses" book="cp">
_Ahañca brāhmaṇo eko,_  
_gaṅgākūle vasāmubho;_  
_Ahaṃ vasāmi upari,_  
_heṭṭhā vasati brāhmaṇo._  

</Paragraph>

</Division>
<Division number="62" book="cp">

<Paragraph number="229" type="verses" book="cp">
_Vicaranto anukūlamhi,_  
_uddhaṃ me assamaddasa;_  
_Tattha maṃ paribhāsetvā,_  
_abhisapi muddhaphālanaṃ._  

</Paragraph>

</Division>
<Division number="63" book="cp">

<Paragraph number="230" type="verses" book="cp">
_Yadihaṃ tassa pakuppeyyaṃ,_  
_yadi sīlaṃ na gopaye;_  
_Oloketvānahaṃ tassa,_  
_kareyyaṃ chārikaṃ viya._  

</Paragraph>

</Division>
<Division number="64" book="cp">

<Paragraph number="231" type="verses" book="cp">
_Yaṃ so tadā maṃ abhisapi,_  
_kupito duṭṭhamānaso;_  
_Tasseva matthake nipati,_  
_yogena taṃ pamocayiṃ._  

</Paragraph>

</Division>
<Division number="65" book="cp">

<Paragraph number="232" type="verses" book="cp">
_Anurakkhiṃ mama sīlaṃ,_  
_nārakkhiṃ mama jīvitaṃ;_  
_Sīlavā hi tadā āsiṃ,_  
_bodhiyāyeva kāraṇā”ti._  


</Paragraph>
</Division>

<Paragraph number="233" type="center" book="cp">
Mātaṅgacariyaṃ sattamaṃ.
</Paragraph>
