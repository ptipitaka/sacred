<Paragraph number="305" type="verses" book="cp">
_“Punāparaṃ yadā homi,_  
_kāsīnaṃ puravaruttame;_  
_Bhaginī ca bhātaro satta,_  
_nibbattā sotthiye kule._  

</Paragraph>

</Division>
<Division number="35" book="cp" bookNo="21" page="411">

<Paragraph number="306" type="verses" book="cp">
_Etesaṃ pubbajo āsiṃ,_  
_hirīsukkamupāgato;_  
_Bhavaṃ disvāna bhayato,_  
_nekkhammābhirato ahaṃ._  

</Paragraph>

</Division>
<Division number="36" book="cp" bookNo="21" page="411">

<Paragraph number="307" type="verses" book="cp">
_Mātāpitūhi pahitā,_  
_sahāyā ekamānasā;_  
_Kāmehi maṃ nimantenti,_  
_‘kulavaṃsaṃ dharehi’ti._  

</Paragraph>

</Division>
<Division number="37" book="cp" bookNo="21" page="411">

<Paragraph number="308" type="verses" book="cp">
_Yaṃ tesaṃ vacanaṃ vuttaṃ,_  
_gihīdhamme sukhāvahaṃ;_  
_Taṃ me ahosi kaṭhinaṃ,_  
_tattaphālasamaṃ viya._  

</Paragraph>

</Division>
<Division number="38" book="cp" bookNo="21" page="411">

<Paragraph number="309" type="verses" book="cp">
_Te maṃ tadā ukkhipantaṃ,_  
_pucchiṃsu patthitaṃ mama;_  
_‘Kiṃ tvaṃ patthayase samma,_  
_yadi kāme na bhuñjasi’._  

</Paragraph>

</Division>
<Division number="39" book="cp" bookNo="21" page="411">

<Paragraph number="310" type="verses" book="cp">
_Tesāhaṃ evamavacaṃ,_  
_atthakāmo hitesinaṃ;_  
_‘Nāhaṃ patthemi gihībhāvaṃ,_  
_nekkhammābhirato ahaṃ’._  

</Paragraph>

</Division>
<Division number="40" book="cp" bookNo="21" page="411">

<Paragraph number="311" type="verses" book="cp">
_Te mayhaṃ vacanaṃ sutvā,_  
_pitu mātu ca sāvayuṃ;_  
_Mātāpitā evamāhu,_  
_‘sabbeva pabbajāma bho’._  

</Paragraph>

</Division>
<Division number="41" book="cp" bookNo="21" page="411">

<Paragraph number="312" type="verses" book="cp">
_Ubho mātāpitā mayhaṃ,_  
_bhaginī ca satta bhātaro;_  
_Amitadhanaṃ chaḍḍayitvā,_  
_pāvisimhā mahāvanan”ti._  


</Paragraph>
</Division>

<Paragraph number="313" type="center" book="cp">
Bhisacariyaṃ catutthaṃ.
</Paragraph>
