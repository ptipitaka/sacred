<Paragraph number="162" type="verses" book="cp">
_“Yadā ahosiṃ pavane,_  
_kuñjaro mātuposako;_  
_Na tadā atthi mahiyā,_  
_guṇena mama sādiso._  

</Paragraph>

</Division>
<Division number="2" book="cp" bookNo="21" page="399">

<Paragraph number="163" type="verses" book="cp">
_Pavane disvā vanacaro,_  
_rañño maṃ paṭivedayi;_  
_‘Tavānucchavo mahārāja,_  
_gajo vasati kānane._  

</Paragraph>

</Division>
<Division number="3" book="cp" bookNo="21" page="399">

<Paragraph number="164" type="verses" book="cp">
_Na tassa parikkhāyattho,_  
_napi āḷakakāsuyā;_  
_Saha gahite soṇḍāya,_  
_sayameva idhehi’ti._  

</Paragraph>

</Division>
<Division number="4" book="cp" bookNo="21" page="399">

<Paragraph number="165" type="verses" book="cp">
_Tassa taṃ vacanaṃ sutvā,_  
_rājāpi tuṭṭhamānaso;_  
_Pesesi hatthidamakaṃ,_  
_chekācariyaṃ susikkhitaṃ._  

</Paragraph>

</Division>
<Division number="5" book="cp" bookNo="21" page="399">

<Paragraph number="166" type="verses" book="cp">
_Gantvā so hatthidamako,_  
_addasa padumassare;_  
_Bhisamuḷālaṃ uddharantaṃ,_  
_yāpanatthāya mātuyā._  

</Paragraph>

</Division>
<Division number="6" book="cp" bookNo="21" page="399">

<Paragraph number="167" type="verses" book="cp">
_Viññāya me sīlaguṇaṃ,_  
_lakkhaṇaṃ upadhārayi;_  
_‘Ehi puttā’ti vatvāna,_  
_mama soṇḍāya aggahi._  

</Paragraph>

</Division>
<Division number="7" book="cp" bookNo="21" page="399">

<Paragraph number="168" type="verses" book="cp">
_Yaṃ me tadā pākatikaṃ,_  
_sarīrānugataṃ balaṃ;_  
_Ajja nāgasahassānaṃ,_  
_balena samasādisaṃ._  

</Paragraph>

</Division>
<Division number="8" book="cp" bookNo="21" page="399">

<Paragraph number="169" type="verses" book="cp">
_Yadihaṃ tesaṃ pakuppeyyaṃ,_  
_upetānaṃ gahaṇāya maṃ;_  
_Paṭibalo bhave tesaṃ,_  
_yāva rajjampi mānusaṃ._  

</Paragraph>

</Division>
<Division number="9" book="cp" bookNo="21" page="399">

<Paragraph number="170" type="verses" book="cp">
_Api cāhaṃ sīlarakkhāya,_  
_Sīlapāramipūriyā;_  
_Na karomi citte aññathattaṃ,_  
_Pakkhipantaṃ mamāḷake._  

</Paragraph>

</Division>
<Division number="10" book="cp" bookNo="21" page="399">

<Paragraph number="171" type="verses" book="cp">
_Yadi te maṃ tattha koṭṭeyyuṃ,_  
_pharasūhi tomarehi ca;_  
_Neva tesaṃ pakuppeyyaṃ,_  
_sīlakhaṇḍabhayā mamā”ti._  


</Paragraph>
</Division>

<Paragraph number="172" type="center" book="cp">
Mātuposakacariyaṃ paṭhamaṃ.
</Paragraph>
