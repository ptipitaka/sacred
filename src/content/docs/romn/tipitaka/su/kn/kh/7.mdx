<Paragraph number="64" type="verses" book="kh">
_Tirokuṭṭesu tiṭṭhanti,_  
_sandhisiṅghāṭakesu ca;_  
_Dvārabāhāsu tiṭṭhanti,_  
_āgantvāna sakaṃ gharaṃ._  

</Paragraph>

</Division>
<Division number="2" book="kh" bookNo="18" page="8">

<Paragraph number="65" type="verses" book="kh">
_Pahūte annapānamhi,_  
_khajjabhojje upaṭṭhite;_  
_Na tesaṃ koci sarati,_  
_sattānaṃ kammapaccayā._  

</Paragraph>

</Division>
<Division number="3" book="kh" bookNo="18" page="8">

<Paragraph number="66" type="verses" book="kh">
_Evaṃ dadanti ñātīnaṃ,_  
_ye honti anukampakā;_  
_Suciṃ paṇītaṃ kālena,_  
_kappiyaṃ pānabhojanaṃ._  

</Paragraph>

</Division>
<Division number="4" book="kh" bookNo="18" page="8">

<Paragraph number="67" type="verses" book="kh">
_Idaṃ vo ñātīnaṃ hotu,_  
_sukhitā hontu ñātayo;_  
_Te ca tattha samāgantvā,_  
_ñātipetā samāgatā._  

</Paragraph>

</Division>
<Division number="5" book="kh" bookNo="18" page="8">

<Paragraph number="68" type="verses" book="kh">
_Pahūte annapānamhi,_  
_sakkaccaṃ anumodare;_  
_Ciraṃ jīvantu no ñātī,_  
_yesaṃ hetu labhāmase._  

</Paragraph>

</Division>
<Division number="6" book="kh" bookNo="18" page="8">

<Paragraph number="69" type="verses" book="kh">
_Amhākañca katā pūjā,_  
_dāyakā ca anipphalā;_  
_Na hi tattha kasī atthi,_  
_gorakkhettha na vijjati._  

</Paragraph>

</Division>
<Division number="7" book="kh" bookNo="18" page="8">

<Paragraph number="70" type="verses" book="kh">
_Vaṇijjā tādisī natthi,_  
_hiraññena kayākayaṃ;_  
_Ito dinnena yāpenti,_  
_petā kālagatā tahiṃ._  

</Paragraph>

</Division>
<Division number="8" book="kh" bookNo="18" page="8">

<Paragraph number="71" type="verses" book="kh">
_Unname udakaṃ vuṭṭhaṃ,_  
_yathā ninnaṃ pavattati;_  
_Evamevaṃ ito dinnaṃ,_  
_petānaṃ upakappati._  

</Paragraph>

</Division>
<Division number="9" book="kh" bookNo="18" page="9">

<Paragraph number="72" type="verses" book="kh">
_Yathā vārivahā pūrā,_  
_paripūrenti sāgaraṃ;_  
_Evamevaṃ ito dinnaṃ,_  
_petānaṃ upakappati._  

</Paragraph>

</Division>
<Division number="10" book="kh" bookNo="18" page="9">

<Paragraph number="73" type="verses" book="kh">
_Adāsi me akāsi me,_  
_ñāti mittā sakhā ca me;_  
_Petānaṃ dakkhiṇaṃ dajjā,_  
_pubbe katamanussaraṃ._  

</Paragraph>

</Division>
<Division number="11" book="kh" bookNo="18" page="9">

<Paragraph number="74" type="verses" book="kh">
_Na hi ruṇṇaṃ vā soko vā,_  
_yā caññā paridevanā;_  
_Na taṃ petānamatthāya,_  
_evaṃ tiṭṭhanti ñātayo._  

</Paragraph>

</Division>
<Division number="12" book="kh" bookNo="18" page="9">

<Paragraph number="75" type="verses" book="kh">
_Ayañca kho dakkhiṇā dinnā,_  
_saṃghamhi suppatiṭṭhitā;_  
_Dīgharattaṃ hitāyassa,_  
_ṭhānaso upakappati._  

</Paragraph>

</Division>
<Division number="13" book="kh" bookNo="18" page="10">

<Paragraph number="76" type="verses" book="kh">
_So ñātidhammo ca ayaṃ nidassito,_  
_Petāna pūjā ca katā uḷārā;_  
_Balañca bhikkhūnamanuppadinnaṃ,_  
_Tumhehi puññaṃ pasutaṃ anappakanti._  


</Paragraph>
</Division>

<Paragraph number="77" type="center" book="kh">
Tirokuṭṭasuttaṃ.
</Paragraph>
