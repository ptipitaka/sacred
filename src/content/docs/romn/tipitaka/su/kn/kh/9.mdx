<Paragraph number="95" type="verses" book="kh">
_Karaṇīyamatthakusalena,_  
_Yanta santaṃ padaṃ abhisamecca;_  
_Sakko ujū ca suhujū ca,_  
_Sūvaco cassa mudu anatimānī._  

</Paragraph>

</Division>
<Division number="2" book="kh" bookNo="18" page="11">

<Paragraph number="96" type="verses" book="kh">
_Santussako ca subharo ca,_  
_Appakicco ca sallahukavutti;_  
_Santindriyo ca nipako ca,_  
_Appagabbho kulesvananugiddho._  

</Paragraph>

</Division>
<Division number="3" book="kh" bookNo="18" page="11">

<Paragraph number="97" type="verses" book="kh">
_Na ca khuddamācare kiñci,_  
_Yena viññū pare upavadeyyuṃ;_  
_Sukhino va khemino hontu,_  
_Sabbasattā bhavantu sukhitattā._  

</Paragraph>

</Division>
<Division number="4" book="kh" bookNo="18" page="11">

<Paragraph number="98" type="verses" book="kh">
_Ye keci pāṇabhūtatthi,_  
_Tasā vā thāvarā vanavasesā;_  
_Dīghā vā ye va mahantā,_  
_Majjhimā rassakā aṇukathūlā._  

</Paragraph>

</Division>
<Division number="5" book="kh" bookNo="18" page="11">

<Paragraph number="99" type="verses" book="kh">
_Diṭṭhā vā ye va adiṭṭhā,_  
_Ye va dūre vasanti avidūre;_  
_Bhūtā va sambhavesī va,_  
_Sabbasattā bhavantu sukhitattā._  

</Paragraph>

</Division>
<Division number="6" book="kh" bookNo="18" page="11">

<Paragraph number="100" type="verses" book="kh">
_Na paro paraṃ nikubbetha,_  
_Nātimaññetha katthaci na kañci;_  
_Byārosanā paṭighasañña,_  
_Nāññamaññassa dukkhamiccheyya._  

</Paragraph>

</Division>
<Division number="7" book="kh" bookNo="18" page="11">

<Paragraph number="101" type="verses" book="kh">
_Mātā yathā niyaṃputta,_  
_Māyusā ekaputtamanurakkhe;_  
_Evampi sabbabhūtesu,_  
_Mānasaṃ bhāvaye aparimāṇaṃ._  

</Paragraph>

</Division>
<Division number="8" book="kh" bookNo="18" page="11">

<Paragraph number="102" type="verses" book="kh">
_Mettañca sabbalokasmi,_  
_Mānasaṃ bhāvaye aparimāṇaṃ;_  
_Uddhaṃ adho ca tiriyañca,_  
_Asambādhaṃ averamasapattaṃ._  

</Paragraph>

</Division>
<Division number="9" book="kh" bookNo="18" page="11">

<Paragraph number="103" type="verses" book="kh">
_Tiṭṭhaṃ caraṃ nisinno va,_  
_Sayāno yāvatāssa vitamiddho;_  
_Etaṃ satiṃ adhiṭṭheyya,_  
_Brahmametaṃ vihāramidhamāhu._  

</Paragraph>

</Division>
<Division number="10" book="kh" bookNo="18" page="11">

<Paragraph number="104" type="verses" book="kh">
_Diṭṭhiñca anupaggamma,_  
_Sīlavā dassanena sampanno;_  
_Kāmesu vinaya gedhaṃ,_  
_Na hi jātuggabbhaseyya punaretīti._  


</Paragraph>
</Division>

<Paragraph number="105" type="center" book="kh">
Mettasuttaṃ.
