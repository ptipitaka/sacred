<Division number="1" book="kh">

<Paragraph number="78" type="verses" book="kh">
_Nidhiṃ nidheti puriso,_  
_gambhīre odakantike;_  
_Atthe kicce samuppanne,_  
_atthāya me bhavissati._  

</Paragraph>

</Division>
<Division number="2" book="kh">

<Paragraph number="79" type="verses" book="kh">
_Rājato vā duruttassa,_  
_corato pīḷitassa vā;_  
_Iṇassa vā pamokkhāya,_  
_dubbhikkhe āpadāsu vā;_  
_Etadatthāya lokasmiṃ,_  
_nidhi nāma nidhīyati._  

</Paragraph>

</Division>
<Division number="3" book="kh">

<Paragraph number="80" type="verses" book="kh">
_Tāvassunihito santo,_  
_gambhīre odakantike;_  
_Na sabbo sabbadā eva,_  
_tassa taṃ upakappati._  

</Paragraph>

</Division>
<Division number="4" book="kh">

<Paragraph number="81" type="verses" book="kh">
_Nidhi vā ṭhānā cavati,_  
_saññā vāssa vimuyhati;_  
_Nāgā vā apanāmenti,_  
_yakkhā vāpi haranti naṃ._  

</Paragraph>

</Division>
<Division number="5" book="kh">

<Paragraph number="82" type="verses" book="kh">
_Appiyā vāpi dāyādā,_  
_uddharanti apassato;_  
_Yadā puññakkhayo hoti,_  
_sabbametaṃ vinassati._  

</Paragraph>

</Division>
<Division number="6" book="kh">

<Paragraph number="83" type="verses" book="kh">
_Yassa dānena sīlena,_  
_saṃyamena damena ca;_  
_Nidhī sunihito hoti,_  
_itthiyā purisassa vā._  

</Paragraph>

</Division>
<Division number="7" book="kh">

<Paragraph number="84" type="verses" book="kh">
_Cetiyamhi ca saṃghe vā,_  
_puggale atithīsu vā;_  
_Mātari pitari cāpi,_  
_atho jeṭṭhamhi bhātari._  

</Paragraph>

</Division>
<Division number="8" book="kh">

<Paragraph number="85" type="verses" book="kh">
_Eso nidhi sunihito,_  
_ajeyyo anugāmiko;_  
_Pahāya gamanīyesu,_  
_etaṃ ādāya gacchati._  

</Paragraph>

</Division>
<Division number="9" book="kh">

<Paragraph number="86" type="verses" book="kh">
_Asādhāraṇamaññesaṃ,_  
_acorāharaṇo nidhi;_  
_Kayirātha dhīro puññāni,_  
_yo nidhi anugāmiko._  

</Paragraph>

</Division>
<Division number="10" book="kh">

<Paragraph number="87" type="verses" book="kh">
_Esa devamanussānaṃ,_  
_sabbakāmadado nidhi;_  
_Yaṃ yadevābhipatthenti,_  
_sabbametena labbhati._  

</Paragraph>

</Division>
<Division number="11" book="kh">

<Paragraph number="88" type="verses" book="kh">
_Suvaṇṇatā susaratā,_  
_susaṇṭhānā surūpatā;_  
_Ādhipaccaparivāro,_  
_sabbametena labbhati._  

</Paragraph>

</Division>
<Division number="12" book="kh">

<Paragraph number="89" type="verses" book="kh">
_Padesarajjaṃ issariyaṃ,_  
_cakkavattisukhaṃ piyaṃ;_  
_Devarajjampi dibbesu,_  
_sabbametena labbhati._  

</Paragraph>

</Division>
<Division number="13" book="kh">

<Paragraph number="90" type="verses" book="kh">
_Mānussikā ca sampatti,_  
_devaloke ca yā rati;_  
_Yā ca nibbānasampatti,_  
_sabbametena labbhati._  

</Paragraph>

</Division>
<Division number="14" book="kh">

<Paragraph number="91" type="verses" book="kh">
_Mittasampadamāgamma,_  
_yonisova payuñjato;_  
_Vijjāvimuttivasībhāvo,_  
_sabbametena labbhati._  

</Paragraph>

</Division>
<Division number="15" book="kh">

<Paragraph number="92" type="verses" book="kh">
_Paṭisambhidā vimokkhā ca,_  
_yā ca sāvakapāramī;_  
_Paccekabodhi buddhabhūmi,_  
_sabbametena labbhati._  

</Paragraph>

</Division>
<Division number="16" book="kh">

<Paragraph number="93" type="verses" book="kh">
_Evaṃ mahatthikā esā,_  
_yadidaṃ puññasampadā;_  
_Tasmā dhīrā pasaṃsanti,_  
_paṇḍitā katapuññatanti._  


</Paragraph>
</Division>

<Paragraph number="94" type="center" book="kh">
Nidhikaṇḍasuttaṃ.
</Paragraph>
