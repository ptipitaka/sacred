<Paragraph number="151" type="verses" book="th2">
_“Dasa putte vijāyitvā,_  
_asmiṃ rūpasamussaye;_  
_Tatohaṃ dubbalā jiṇṇā,_  
_bhikkhuniṃ upasaṅkamiṃ._  

</Paragraph>

</Division>
<Division number="103" book="th2">

<Paragraph number="152" type="verses" book="th2">
_Sā me dhammamadesesi,_  
_khandhāyatanadhātuyo;_  
_Tassā dhammaṃ suṇitvāna,_  
_kese chetvāna pabbajiṃ._  

</Paragraph>

</Division>
<Division number="104" book="th2">

<Paragraph number="153" type="verses" book="th2">
_Tassā me sikkhamānāya,_  
_dibbacakkhu visodhitaṃ;_  
_Pubbenivāsaṃ jānāmi,_  
_yattha me vusitaṃ pure._  

</Paragraph>

</Division>
<Division number="105" book="th2">

<Paragraph number="154" type="verses" book="th2">
_Animittañca bhāvemi,_  
_ekaggā susamāhitā;_  
_Anantarāvimokkhāsiṃ,_  
_anupādāya nibbutā._  

</Paragraph>

</Division>
<Division number="106" book="th2">

<Paragraph number="155" type="verses" book="th2">
_Pañcakkhandhā pariññātā,_  
_tiṭṭhanti chinnamūlakā;_  
_Dhi tavatthu jare jamme,_  
_natthi dāni punabbhavo”ti._  


</Paragraph>
</Division>
