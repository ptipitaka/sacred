<Paragraph number="175" type="verses" book="th2">
_“Duggatāhaṃ pure āsiṃ,_  
_vidhavā ca aputtikā;_  
_Vinā mittehi ñātīhi,_  
_bhattacoḷassa nādhigaṃ._  

</Paragraph>

</Division>
<Division number="123" book="th2" bookNo="19" page="393">

<Paragraph number="176" type="verses" book="th2">
_Pattaṃ daṇḍañca gaṇhitvā,_  
_bhikkhamānā kulā kulaṃ;_  
_Sītuṇhena ca ḍayhantī,_  
_satta vassāni cārihaṃ._  

</Paragraph>

</Division>
<Division number="124" book="th2" bookNo="19" page="393">

<Paragraph number="177" type="verses" book="th2">
_Bhikkhuniṃ puna disvāna,_  
_annapānassa lābhiniṃ;_  
_Upasaṅkamma avocaṃ,_  
_‘pabbajjaṃ anagāriyaṃ’._  

</Paragraph>

</Division>
<Division number="125" book="th2" bookNo="19" page="393">

<Paragraph number="178" type="verses" book="th2">
_Sā ca maṃ anukampāya,_  
_pabbājesi paṭācārā;_  
_Tato maṃ ovaditvāna,_  
_paramatthe niyojayi._  

</Paragraph>

</Division>
<Division number="126" book="th2" bookNo="19" page="393">

<Paragraph number="179" type="verses" book="th2">
_Tassāhaṃ vacanaṃ sutvā,_  
_akāsiṃ anusāsaniṃ;_  
_Amogho ayyāyovādo,_  
_tevijjāmhi anāsavā”ti._  


</Paragraph>
</Division>

…  Candā therī… .
