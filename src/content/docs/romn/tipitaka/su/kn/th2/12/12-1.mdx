<Division number="236" book="th2">

<Paragraph number="312" type="verses" book="th2">
_“Udahārī ahaṃ sīte,_  
_sadā udakamotariṃ;_  
_Ayyānaṃ daṇḍabhayabhītā,_  
_vācādosabhayaṭṭitā._  

</Paragraph>

</Division>
<Division number="237" book="th2">

<Paragraph number="313" type="verses" book="th2">
_Kassa brāhmaṇa tvaṃ bhīto,_  
_sadā udakamotari;_  
_Vedhamānehi gattehi,_  
_sītaṃ vedayase bhusaṃ”._  

</Paragraph>

</Division>
<Division number="238" book="th2">

<Paragraph number="314" type="verses" book="th2">
_“Jānantī vata maṃ bhoti,_  
_puṇṇike paripucchasi;_  
_Karontaṃ kusalaṃ kammaṃ,_  
_rundhantaṃ katapāpakaṃ._  

</Paragraph>

</Division>
<Division number="239" book="th2">

<Paragraph number="315" type="verses" book="th2">
_Yo ca vuḍḍho daharo vā,_  
_pāpakammaṃ pakubbati;_  
_Dakābhisecanā sopi,_  
_pāpakammā pamuccati”._  

</Paragraph>

</Division>
<Division number="240" book="th2">

<Paragraph number="316" type="verses" book="th2">
_“Ko nu te idamakkhāsi,_  
_ajānantassa ajānako;_  
_Dakābhisecanā nāma,_  
_pāpakammā pamuccati._  

</Paragraph>

</Division>
<Division number="241" book="th2">

<Paragraph number="317" type="verses" book="th2">
_Saggaṃ nūna gamissanti,_  
_sabbe maṇḍūkakacchapā;_  
_Nāgā ca susumārā ca,_  
_ye caññe udake carā._  

</Paragraph>

</Division>
<Division number="242" book="th2">

<Paragraph number="318" type="verses" book="th2">
_Orabbhikā sūkarikā,_  
_macchikā migabandhakā;_  
_Corā ca vajjhaghātā ca,_  
_ye caññe pāpakammino;_  
_Dakābhisecanā tepi,_  
_pāpakammā pamuccare._  

</Paragraph>

</Division>
<Division number="243" book="th2">

<Paragraph number="319" type="verses" book="th2">
_Sace imā nadiyo te,_  
_pāpaṃ pubbe kataṃ vahuṃ;_  
_Puññampimā vaheyyuṃ te,_  
_tena tvaṃ paribāhiro._  

</Paragraph>

</Division>
<Division number="244" book="th2">

<Paragraph number="320" type="verses" book="th2">
_Yassa brāhmaṇa tvaṃ bhīto,_  
_sadā udakamotari;_  
_Tameva brahme mā kāsi,_  
_mā te sītaṃ chaviṃ hane”._  

</Paragraph>

</Division>
<Division number="245" book="th2">

<Paragraph number="321" type="verses" book="th2">
_“Kummaggapaṭipannaṃ maṃ,_  
_ariyamaggaṃ samānayi;_  
_Dakābhisecanā bhoti,_  
_imaṃ sāṭaṃ dadāmi te”._  

</Paragraph>

</Division>
<Division number="246" book="th2">

<Paragraph number="322" type="verses" book="th2">
_“Tuyheva sāṭako hotu,_  
_nāhamicchāmi sāṭakaṃ;_  
_Sace bhāyasi dukkhassa,_  
_sace te dukkhamappiyaṃ._  

</Paragraph>

</Division>
<Division number="247" book="th2">

<Paragraph number="323" type="verses" book="th2">
_Mākāsi pāpakaṃ kammaṃ,_  
_āvi vā yadi vā raho;_  
_Sace ca pāpakaṃ kammaṃ,_  
_karissasi karosi vā._  

</Paragraph>

</Division>
<Division number="248" book="th2">

<Paragraph number="324" type="verses" book="th2">
_Na te dukkhā pamutyatthi,_  
_upeccāpi palāyato;_  
_Sace bhāyasi dukkhassa,_  
_sace te dukkhamappiyaṃ._  

</Paragraph>

</Division>
<Division number="249" book="th2">

<Paragraph number="325" type="verses" book="th2">
_Upehi saraṇaṃ buddhaṃ,_  
_dhammaṃ saṃghañca tādinaṃ;_  
_Samādiyāhi sīlāni,_  
_taṃ te atthāya hehiti”._  

</Paragraph>

</Division>
<Division number="250" book="th2">

<Paragraph number="326" type="verses" book="th2">
_“Upemi saraṇaṃ buddhaṃ,_  
_dhammaṃ saṃghañca tādinaṃ;_  
_Samādiyāmi sīlāni,_  
_taṃ me atthāya hehiti._  

</Paragraph>

</Division>
<Division number="251" book="th2">

<Paragraph number="327" type="verses" book="th2">
_Brahmabandhu pure āsiṃ,_  
_ajjamhi saccabrāhmaṇo;_  
_Tevijjo vedasampanno,_  
_sottiyo camhi nhātako”ti._  


</Paragraph>
</Division>

…  Puṇṇā therī… .



</Paragraph>
