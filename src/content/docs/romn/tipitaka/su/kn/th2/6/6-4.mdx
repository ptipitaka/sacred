<Paragraph number="203" type="verses" book="th2">
_“Alaṅkatā suvasanā,_  
_mālinī candanokkhitā;_  
_Sabbābharaṇasañchannā,_  
_dāsīgaṇapurakkhatā._  

</Paragraph>

</Division>
<Division number="146" book="th2" bookNo="19" page="395">

<Paragraph number="204" type="verses" book="th2">
_Annaṃ pānañca ādāya,_  
_khajjaṃ bhojjaṃ anappakaṃ;_  
_Gehato nikkhamitvāna,_  
_uyyānamabhihārayiṃ._  

</Paragraph>

</Division>
<Division number="147" book="th2" bookNo="19" page="395">

<Paragraph number="205" type="verses" book="th2">
_Tattha ramitvā kīḷitvā,_  
_āgacchantī sakaṃ gharaṃ;_  
_Vihāraṃ daṭṭhuṃ pāvisiṃ,_  
_sākete añjanaṃ vanaṃ._  

</Paragraph>

</Division>
<Division number="148" book="th2" bookNo="19" page="396">

<Paragraph number="206" type="verses" book="th2">
_Disvāna lokapajjotaṃ,_  
_vanditvāna upāvisiṃ;_  
_So me dhammamadesesi,_  
_anukampāya cakkhumā._  

</Paragraph>

</Division>
<Division number="149" book="th2" bookNo="19" page="396">

<Paragraph number="207" type="verses" book="th2">
_Sutvā ca kho mahesissa,_  
_saccaṃ sampaṭivijjhahaṃ;_  
_Tattheva virajaṃ dhammaṃ,_  
_phusayiṃ amataṃ padaṃ._  

</Paragraph>

</Division>
<Division number="150" book="th2" bookNo="19" page="396">

<Paragraph number="208" type="verses" book="th2">
_Tato viññātasaddhammā,_  
_pabbajiṃ anagāriyaṃ;_  
_Tisso vijjā anuppattā,_  
_amoghaṃ buddhasāsanan”ti._  


</Paragraph>
</Division>
