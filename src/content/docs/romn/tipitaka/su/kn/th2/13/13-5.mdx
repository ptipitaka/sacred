<Division number="339" book="th2">

<Paragraph number="421" type="verses" book="th2">
_“Daharāhaṃ suddhavasanā,_  
_yaṃ pure dhammamassuṇiṃ;_  
_Tassā me appamattāya,_  
_saccābhisamayo ahu._  

</Paragraph>

</Division>
<Division number="340" book="th2">

<Paragraph number="422" type="verses" book="th2">
_Tatohaṃ sabbakāmesu,_  
_bhusaṃ aratimajjhagaṃ;_  
_Sakkāyasmiṃ bhayaṃ disvā,_  
_nekkhammameva pīhaye._  

</Paragraph>

</Division>
<Division number="341" book="th2">

<Paragraph number="423" type="verses" book="th2">
_Hitvānahaṃ ñātigaṇaṃ,_  
_dāsakammakarāni ca;_  
_Gāmakhettāni phītāni,_  
_ramaṇīye pamodite._  

</Paragraph>

</Division>
<Division number="342" book="th2">

<Paragraph number="424" type="verses" book="th2">
_Pahāyahaṃ pabbajitā,_  
_sāpateyyamanappakaṃ;_  
_Evaṃ saddhāya nikkhamma,_  
_saddhamme suppavedite._  

</Paragraph>

</Division>
<Division number="343" book="th2">

<Paragraph number="425" type="verses" book="th2">
_Netaṃ assa patirūpaṃ,_  
_Ākiñcaññañhi patthaye;_  
_Yo jātarūpaṃ rajataṃ,_  
_Chaḍḍetvā punarāgame._  

</Paragraph>

</Division>
<Division number="344" book="th2">

<Paragraph number="426" type="verses" book="th2">
_Rajataṃ jātarūpaṃ vā,_  
_na bodhāya na santiyā;_  
_Netaṃ samaṇasāruppaṃ,_  
_na etaṃ ariyaddhanaṃ._  

</Paragraph>

</Division>
<Division number="345" book="th2">

<Paragraph number="427" type="verses" book="th2">
_Lobhanaṃ madanañcetaṃ,_  
_mohanaṃ rajavaḍḍhanaṃ;_  
_Sāsaṅkaṃ bahuāyāsaṃ,_  
_natthi cettha dhuvaṃ ṭhiti._  

</Paragraph>

</Division>
<Division number="346" book="th2">

<Paragraph number="428" type="verses" book="th2">
_Ettha rattā pamattā ca,_  
_saṃkiliṭṭhamanā narā;_  
_Aññamaññena byāruddhā,_  
_puthu kubbanti medhagaṃ._  

</Paragraph>

</Division>
<Division number="347" book="th2">

<Paragraph number="429" type="verses" book="th2">
_Vadho bandho parikleso,_  
_jāni sokapariddavo;_  
_Kāmesu adhipannānaṃ,_  
_dissate byasanaṃ bahuṃ._  

</Paragraph>

</Division>
<Division number="348" book="th2">

<Paragraph number="430" type="verses" book="th2">
_Taṃ maṃ ñātī amittāva,_  
_kiṃ vo kāmesu yuñjatha;_  
_Jānātha maṃ pabbajitaṃ,_  
_kāmesu bhayadassiniṃ._  

</Paragraph>

</Division>
<Division number="349" book="th2">

<Paragraph number="431" type="verses" book="th2">
_Na hiraññasuvaṇṇena,_  
_parikkhīyanti āsavā;_  
_Amittā vadhakā kāmā,_  
_sapattā sallabandhanā._  

</Paragraph>

</Division>
<Division number="350" book="th2">

<Paragraph number="432" type="verses" book="th2">
_Taṃ maṃ ñātī amittāva,_  
_kiṃ vo kāmesu yuñjatha;_  
_Jānātha maṃ pabbajitaṃ,_  
_muṇḍaṃ saṅghāṭipārutaṃ._  

</Paragraph>

</Division>
<Division number="351" book="th2">

<Paragraph number="433" type="verses" book="th2">
_Uttiṭṭhapiṇḍo uñcho ca,_  
_paṃsukūlañca cīvaraṃ;_  
_Etaṃ kho mama sāruppaṃ,_  
_anagārūpanissayo._  

</Paragraph>

</Division>
<Division number="352" book="th2">

<Paragraph number="434" type="verses" book="th2">
_Vantā mahesīhi kāmā,_  
_ye dibbā ye ca mānusā;_  
_Khemaṭṭhāne vimuttā te,_  
_pattā te acalaṃ sukhaṃ._  

</Paragraph>

</Division>
<Division number="353" book="th2">

<Paragraph number="435" type="verses" book="th2">
_Māhaṃ kāmehi saṅgacchiṃ,_  
_yesu tāṇaṃ na vijjati;_  
_Amittā vadhakā kāmā,_  
_aggikkhandhūpamā dukhā._  

</Paragraph>

</Division>
<Division number="354" book="th2">

<Paragraph number="436" type="verses" book="th2">
_Paripantho esa bhayo,_  
_savighāto sakaṇṭako;_  
_Gedho suvisamo ceso,_  
_mahanto mohanāmukho._  

</Paragraph>

</Division>
<Division number="355" book="th2">

<Paragraph number="437" type="verses" book="th2">
_Upasaggo bhīmarūpo,_  
_kāmā sappasirūpamā;_  
_Ye bālā abhinandanti,_  
_andhabhūtā puthujjanā._  

</Paragraph>

</Division>
<Division number="356" book="th2">

<Paragraph number="438" type="verses" book="th2">
_Kāmapaṅkena sattā hi,_  
_bahū loke aviddasū;_  
_Pariyantaṃ na jānanti,_  
_jātiyā maraṇassa ca._  

</Paragraph>

</Division>
<Division number="357" book="th2">

<Paragraph number="439" type="verses" book="th2">
_Duggatigamanaṃ maggaṃ,_  
_manussā kāmahetukaṃ;_  
_Bahuṃ ve paṭipajjanti,_  
_attano rogamāvahaṃ._  

</Paragraph>

</Division>
<Division number="358" book="th2">

<Paragraph number="440" type="verses" book="th2">
_Evaṃ amittajananā,_  
_tāpanā saṃkilesikā;_  
_Lokāmisā bandhanīyā,_  
_kāmā maraṇabandhanā._  

</Paragraph>

</Division>
<Division number="359" book="th2">

<Paragraph number="441" type="verses" book="th2">
_Ummādanā ullapanā,_  
_kāmā cittappamaddino;_  
_Sattānaṃ saṃkilesāya,_  
_khippaṃ mārena oḍḍitaṃ._  

</Paragraph>

</Division>
<Division number="360" book="th2">

<Paragraph number="442" type="verses" book="th2">
_Anantādīnavā kāmā,_  
_bahudukkhā mahāvisā;_  
_Appassādā raṇakarā,_  
_sukkapakkhavisosanā._  

</Paragraph>

</Division>
<Division number="361" book="th2">

<Paragraph number="443" type="verses" book="th2">
_Sāhaṃ etādisaṃ katvā,_  
_byasanaṃ kāmahetukaṃ;_  
_Na taṃ paccāgamissāmi,_  
_nibbānābhiratā sadā._  

</Paragraph>

</Division>
<Division number="362" book="th2">

<Paragraph number="444" type="verses" book="th2">
_Raṇaṃ karitvā kāmānaṃ,_  
_sītibhāvābhikaṅkhinī;_  
_Appamattā vihassāmi,_  
_sabbasaṃyojanakkhaye._  

</Paragraph>

</Division>
<Division number="363" book="th2">

<Paragraph number="445" type="verses" book="th2">
_Asokaṃ virajaṃ khemaṃ,_  
_ariyaṭṭhaṅgikaṃ ujuṃ;_  
_Taṃ maggaṃ anugacchāmi,_  
_yena tiṇṇā mahesino”._  

</Paragraph>

</Division>
<Division number="364" book="th2">

<Paragraph number="446" type="verses" book="th2">
_“Imaṃ passatha dhammaṭṭhaṃ,_  
_subhaṃ kammāradhītaraṃ;_  
_Anejaṃ upasampajja,_  
_rukkhamūlamhi jhāyati._  

</Paragraph>

</Division>
<Division number="365" book="th2">

<Paragraph number="447" type="verses" book="th2">
_Ajjaṭṭhamī pabbajitā,_  
_saddhā saddhammasobhanā;_  
_Vinītuppalavaṇṇāya,_  
_tevijjā maccuhāyinī._  

</Paragraph>

</Division>
<Division number="366" book="th2">

<Paragraph number="448" type="verses" book="th2">
_Sāyaṃ bhujissā anaṇā,_  
_bhikkhunī bhāvitindriyā;_  
_Sabbayogavisaṃyuttā,_  
_katakiccā anāsavā”._  

</Paragraph>

</Division>
<Division number="367" book="th2">

<Paragraph number="449" type="verses" book="th2">
_Taṃ sakko devasaṅghena,_  
_upasaṅkamma iddhiyā;_  
_Namassati bhūtapati,_  
_subhaṃ kammāradhītaranti._  


</Paragraph>
</Division>

<Paragraph number="450" type="center" book="th2">
…  Subhā kammāradhītā therī… .



</Paragraph>

<Paragraph number="451" type="center" book="th2">
Vīsatinipāto niṭṭhito.
</Paragraph>
