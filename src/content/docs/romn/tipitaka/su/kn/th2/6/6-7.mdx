<Paragraph number="224" type="verses" book="th2">
_“Gutte yadatthaṃ pabbajjā,_  
_hitvā puttaṃ vasuṃ piyaṃ;_  
_Tameva anubrūhehi,_  
_mā cittassa vasaṃ gami._  

</Paragraph>

</Division>
<Division number="164" book="th2" bookNo="19" page="397">

<Paragraph number="225" type="verses" book="th2">
_Cittena vañcitā sattā,_  
_mārassa visaye ratā;_  
_Anekajātisaṃsāraṃ,_  
_sandhāvanti aviddasū._  

</Paragraph>

</Division>
<Division number="165" book="th2" bookNo="19" page="397">

<Paragraph number="226" type="verses" book="th2">
_Kāmacchandañca byāpādaṃ,_  
_sakkāyadiṭṭhimeva ca;_  
_Sīlabbataparāmāsaṃ,_  
_vicikicchañca pañcamaṃ._  

</Paragraph>

</Division>
<Division number="166" book="th2" bookNo="19" page="397">

<Paragraph number="227" type="verses" book="th2">
_Saṃyojanāni etāni,_  
_pajahitvāna bhikkhunī;_  
_Orambhāgamanīyāni,_  
_nayidaṃ punarehisi._  

</Paragraph>

</Division>
<Division number="167" book="th2" bookNo="19" page="397">

<Paragraph number="228" type="verses" book="th2">
_Rāgaṃ mānaṃ avijjañca,_  
_uddhaccañca vivajjiya;_  
_Saṃyojanāni chetvāna,_  
_dukkhassantaṃ karissasi._  

</Paragraph>

</Division>
<Division number="168" book="th2" bookNo="19" page="397">

<Paragraph number="229" type="verses" book="th2">
_Khepetvā jātisaṃsāraṃ,_  
_pariññāya punabbhavaṃ;_  
_Diṭṭheva dhamme nicchātā,_  
_upasantā carissatī”ti._  


</Paragraph>
</Division>
