<Paragraph number="115" type="verses" book="th2">
_“Mattā vaṇṇena rūpena,_  
_sobhaggena yasena ca;_  
_Yobbanena cupatthaddhā,_  
_aññāsamatimaññihaṃ._  

</Paragraph>

</Division>
<Division number="73" book="th2">

<Paragraph number="116" type="verses" book="th2">
_Vibhūsetvā imaṃ kāyaṃ,_  
_sucittaṃ bālalāpanaṃ;_  
_Aṭṭhāsiṃ vesidvāramhi,_  
_luddo pāsamivoḍḍiya._  

</Paragraph>

</Division>
<Division number="74" book="th2">

<Paragraph number="117" type="verses" book="th2">
_Piḷandhanaṃ vidaṃsentī,_  
_guyhaṃ pakāsikaṃ bahuṃ;_  
_Akāsiṃ vividhaṃ māyaṃ,_  
_ujjagghantī bahuṃ janaṃ._  

</Paragraph>

</Division>
<Division number="75" book="th2">

<Paragraph number="118" type="verses" book="th2">
_Sājja piṇḍaṃ caritvāna,_  
_muṇḍā saṅghāṭipārutā;_  
_Nisinnā rukkhamūlamhi,_  
_avitakkassa lābhinī._  

</Paragraph>

</Division>
<Division number="76" book="th2">

<Paragraph number="119" type="verses" book="th2">
_Sabbe yogā samucchinnā,_  
_ye dibbā ye ca mānusā;_  
_Khepetvā āsave sabbe,_  
_sītibhūtāmhi nibbutā”ti._  


</Paragraph>
</Division>
