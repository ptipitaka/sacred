<Division number="204" book="th2">

<Paragraph number="274" type="verses" book="th2">
_“Mā su te vaḍḍha lokamhi,_  
_vanatho āhu kudācanaṃ;_  
_Mā puttaka punappunaṃ,_  
_ahu dukkhassa bhāgimā._  

</Paragraph>

</Division>
<Division number="205" book="th2">

<Paragraph number="275" type="verses" book="th2">
_Sukhañhi vaḍḍha munayo,_  
_anejā chinnasaṃsayā;_  
_Sītibhūtā damappattā,_  
_viharanti anāsavā._  

</Paragraph>

</Division>
<Division number="206" book="th2">

<Paragraph number="276" type="verses" book="th2">
_Tehānuciṇṇaṃ isībhi,_  
_maggaṃ dassanapattiyā;_  
_Dukkhassantakiriyāya,_  
_tvaṃ vaḍḍha anubrūhaya”._  

</Paragraph>

</Division>
<Division number="207" book="th2">

<Paragraph number="277" type="verses" book="th2">
_“Visāradāva bhaṇasi,_  
_etamatthaṃ janetti me;_  
_Maññāmi nūna māmike,_  
_vanatho te na vijjati”._  

</Paragraph>

</Division>
<Division number="208" book="th2">

<Paragraph number="278" type="verses" book="th2">
_“Ye keci vaḍḍha saṅkhārā,_  
_hīnā ukkaṭṭhamajjhimā;_  
_Aṇūpi aṇumattopi,_  
_vanatho me na vijjati._  

</Paragraph>

</Division>
<Division number="209" book="th2">

<Paragraph number="279" type="verses" book="th2">
_Sabbe me āsavā khīṇā,_  
_appamattassa jhāyato;_  
_Tisso vijjā anuppattā,_  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

</Division>
<Division number="210" book="th2">

<Paragraph number="280" type="verses" book="th2">
_“Uḷāraṃ vata me mātā,_  
_patodaṃ samavassari;_  
_Paramatthasañhitā gāthā,_  
_yathāpi anukampikā._  

</Paragraph>

</Division>
<Division number="211" book="th2">

<Paragraph number="281" type="verses" book="th2">
_Tassāhaṃ vacanaṃ sutvā,_  
_anusiṭṭhiṃ janettiyā;_  
_Dhammasaṃvegamāpādiṃ,_  
_yogakkhemassa pattiyā._  

</Paragraph>

</Division>
<Division number="212" book="th2">

<Paragraph number="282" type="verses" book="th2">
_Sohaṃ padhānapahitatto,_  
_rattindivamatandito;_  
_Mātarā codito santo,_  
_aphusiṃ santimuttamaṃ”._  


</Paragraph>
</Division>

<Paragraph number="283" type="center" book="th2">
…  Vaḍḍhamātā therī… .



</Paragraph>

<Paragraph number="284" type="center" book="th2">
Navakanipāto niṭṭhito.
</Paragraph>
