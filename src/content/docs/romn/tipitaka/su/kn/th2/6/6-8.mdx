<Paragraph number="231" type="verses" book="th2">
_“Catukkhattuṃ pañcakkhattuṃ,_  
_vihārā upanikkhamiṃ;_  
_Aladdhā cetaso santiṃ,_  
_citte avasavattinī._  

</Paragraph>

</Division>
<Division number="170" book="th2" bookNo="19" page="398">

<Paragraph number="232" type="verses" book="th2">
_Bhikkhuniṃ upasaṅkamma,_  
_sakkaccaṃ paripucchahaṃ;_  
_Sā me dhammamadesesi,_  
_dhātuāyatanāni ca._  

</Paragraph>

</Division>
<Division number="171" book="th2" bookNo="19" page="398">

<Paragraph number="233" type="verses" book="th2">
_Cattāri ariyasaccāni,_  
_indriyāni balāni ca;_  
_Bojjhaṅgaṭṭhaṅgikaṃ maggaṃ,_  
_uttamatthassa pattiyā._  

</Paragraph>

</Division>
<Division number="172" book="th2" bookNo="19" page="398">

<Paragraph number="234" type="verses" book="th2">
_Tassāhaṃ vacanaṃ sutvā,_  
_karontī anusāsaniṃ;_  
_Rattiyā purime yāme,_  
_pubbajātimanussariṃ._  

</Paragraph>

</Division>
<Division number="173" book="th2" bookNo="19" page="398">

<Paragraph number="235" type="verses" book="th2">
_Rattiyā majjhime yāme,_  
_dibbacakkhuṃ visodhayiṃ;_  
_Rattiyā pacchime yāme,_  
_tamokhandhaṃ padālayiṃ._  

</Paragraph>

</Division>
<Division number="174" book="th2" bookNo="19" page="398">

<Paragraph number="236" type="verses" book="th2">
_Pītisukhena ca kāyaṃ,_  
_Pharitvā vihariṃ tadā;_  
_Sattamiyā pāde pasāresiṃ,_  
_Tamokhandhaṃ padāliyā”ti._  


</Paragraph>
</Division>

<Paragraph number="237" type="center" book="th2">
…  Vijayā therī… .



</Paragraph>
