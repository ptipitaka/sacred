<Division number="68" book="th2">

<Paragraph number="110" type="verses" book="th2">

_kāmarāgenavassutā;_  
_Bāhā paggayha kandantī,_  
_vihāraṃ pāvisiṃ ahaṃ._  

</Paragraph>

</Division>
<Division number="69" book="th2">

<Paragraph number="111" type="verses" book="th2">
_Sā bhikkhuniṃ upāgacchiṃ,_  
_yā me saddhāyikā ahu;_  
_Sā me dhammamadesesi,_  
_khandhāyatanadhātuyo._  

</Paragraph>

</Division>
<Division number="70" book="th2">

<Paragraph number="112" type="verses" book="th2">
_Tassā dhammaṃ suṇitvāna,_  
_ekamante upāvisiṃ;_  
_Pubbenivāsaṃ jānāmi,_  
_dibbacakkhu visodhitaṃ._  

</Paragraph>

</Division>
<Division number="71" book="th2">

<Paragraph number="113" type="verses" book="th2">

_sotadhātu visodhitā;_  
_Iddhīpi me sacchikatā,_  
_patto me āsavakkhayo;_  
_Chaḷabhiññā sacchikatā,_  
_kataṃ buddhassa sāsanan”ti._  


</Paragraph>
