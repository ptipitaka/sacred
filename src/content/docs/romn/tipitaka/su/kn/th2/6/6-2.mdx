<Paragraph number="189" type="verses" book="th2">
_“Puttasokenahaṃ aṭṭā,_  
_khittacittā visaññinī;_  
_Naggā pakiṇṇakesī ca,_  
_tena tena vicārihaṃ._  

</Paragraph>

</Division>
<Division number="134" book="th2" bookNo="19" page="394">

<Paragraph number="190" type="verses" book="th2">
_Vīthi saṅkārakūṭesu,_  
_susāne rathiyāsu ca;_  
_Acariṃ tīṇi vassāni,_  
_khuppipāsāsamappitā._  

</Paragraph>

</Division>
<Division number="135" book="th2" bookNo="19" page="394">

<Paragraph number="191" type="verses" book="th2">
_Athaddasāsiṃ sugataṃ,_  
_nagaraṃ mithilaṃ pati;_  
_Adantānaṃ dametāraṃ,_  
_sambuddhamakutobhayaṃ._  

</Paragraph>

</Division>
<Division number="136" book="th2" bookNo="19" page="394">

<Paragraph number="192" type="verses" book="th2">
_Sacittaṃ paṭiladdhāna,_  
_vanditvāna upāvisiṃ;_  
_So me dhammamadesesi,_  
_anukampāya gotamo._  

</Paragraph>

</Division>
<Division number="137" book="th2" bookNo="19" page="395">

<Paragraph number="193" type="verses" book="th2">
_Tassa dhammaṃ suṇitvāna,_  
_pabbajiṃ anagāriyaṃ;_  
_Yuñjantī satthuvacane,_  
_sacchākāsiṃ padaṃ sivaṃ._  

</Paragraph>

</Division>
<Division number="138" book="th2" bookNo="19" page="395">

<Paragraph number="194" type="verses" book="th2">
_Sabbe sokā samucchinnā,_  
_pahīnā etadantikā;_  
_Pariññātā hi me vatthū,_  
_yato sokāna sambhavo”ti._  


</Paragraph>
</Division>
