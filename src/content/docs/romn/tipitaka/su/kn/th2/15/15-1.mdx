<Paragraph number="488" type="verses" book="th2">
_Nagaramhi kusumanāme,_  
_Pāṭaliputtamhi pathaviyā maṇḍe;_  
_Sakyakulakulīnāyo,_  
_Dve bhikkhuniyo hi guṇavatiyo._  

</Paragraph>

</Division>
<Division number="403" book="th2" bookNo="19" page="425">

<Paragraph number="489" type="verses" book="th2">
_Isidāsī tattha ekā,_  
_Dutiyā bodhīti sīlasampannā ca;_  
_Jhānajjhāyanaratāyo,_  
_Bahussutāyo dhutakilesāyo._  

</Paragraph>

</Division>
<Division number="404" book="th2" bookNo="19" page="425">

<Paragraph number="490" type="verses" book="th2">
_Tā piṇḍāya caritvā,_  
_Bhattatthaṃ kariya dhotapattāyo;_  
_Rahitamhi sukhanisinnā,_  
_Imā girā abbhudīresuṃ._  

</Paragraph>

</Division>
<Division number="405" book="th2" bookNo="19" page="425">

<Paragraph number="491" type="verses" book="th2">
_“Pāsādikāsi ayye,_  
_Isidāsi vayopi te aparihīno;_  
_Kiṃ disvāna byālikaṃ,_  
_Athāsi nekkhammamanuyuttā”._  

</Paragraph>

</Division>
<Division number="406" book="th2" bookNo="19" page="425">

<Paragraph number="492" type="verses" book="th2">
_Evamanuyuñjiyamānā sā,_  
_Rahite dhammadesanākusalā;_  
_Isidāsī vacanamabravi,_  
_“Suṇa bodhi yathāmhi pabbajitā”._  

</Paragraph>

</Division>
<Division number="407" book="th2" bookNo="19" page="425">

<Paragraph number="493" type="verses" book="th2">
_“Ujjeniyā puravare,_  
_Mayhaṃ pitā sīlasaṃvuto seṭṭhi;_  
_Tassamhi ekadhītā,_  
_Piyā manāpā ca dayitā ca._  

</Paragraph>

</Division>
<Division number="408" book="th2" bookNo="19" page="425">

<Paragraph number="494" type="verses" book="th2">
_Atha me sāketato varakā,_  
_Āgacchumuttamakulīnā;_  
_Seṭṭhī pahūtaratano,_  
_Tassa mamaṃ suṇhamadāsi tāto._  

</Paragraph>

</Division>
<Division number="409" book="th2" bookNo="19" page="425">

<Paragraph number="495" type="verses" book="th2">
_Sassuyā sasurassa ca,_  
_Sāyaṃ pātaṃ paṇāmamupagamma;_  
_Sirasā karomi pāde,_  
_Vandāmi yathāmhi anusiṭṭhā._  

</Paragraph>

</Division>
<Division number="410" book="th2" bookNo="19" page="425">

<Paragraph number="496" type="verses" book="th2">
_Yā mayhaṃ sāmikassa,_  
_Bhaginiyo bhātuno parijano vā;_  
_Tamekavarakampi disvā,_  
_Ubbiggā āsanaṃ demi._  

</Paragraph>

</Division>
<Division number="411" book="th2" bookNo="19" page="425">

<Paragraph number="497" type="verses" book="th2">
_Annena ca pānena ca,_  
_Khajjena ca yañca tattha sannihitaṃ;_  
_Chādemi upanayāmi ca,_  
_Demi ca yaṃ yassa patirūpaṃ._  

</Paragraph>

</Division>
<Division number="412" book="th2" bookNo="19" page="425">

<Paragraph number="498" type="verses" book="th2">
_Kālena upaṭṭhahitvā,_  
_Gharaṃ samupagamāmi ummāre;_  
_Dhovantī hatthapāde,_  
_Pañjalikā sāmikamupemi._  

</Paragraph>

</Division>
<Division number="413" book="th2" bookNo="19" page="425">

<Paragraph number="499" type="verses" book="th2">
_Kocchaṃ pasādaṃ añjaniñca,_  
_Ādāsakañca gaṇhitvā;_  
_Parikammakārikā viya,_  
_Sayameva patiṃ vibhūsemi._  

</Paragraph>

</Division>
<Division number="414" book="th2" bookNo="19" page="426">

<Paragraph number="500" type="verses" book="th2">
_Sayameva odanaṃ sādhayāmi,_  
_Sayameva bhājanaṃ dhovantī;_  
_Mātāva ekaputtakaṃ,_  
_Tathā bhattāraṃ paricarāmi._  

</Paragraph>

</Division>
<Division number="415" book="th2" bookNo="19" page="426">

<Paragraph number="501" type="verses" book="th2">
_Evaṃ maṃ bhattikataṃ,_  
_Anurattaṃ kārikaṃ nihatamānaṃ;_  
_Uṭṭhāyikaṃ analasaṃ,_  
_Sīlavatiṃ dussate bhattā._  

</Paragraph>

</Division>
<Division number="416" book="th2" bookNo="19" page="426">

<Paragraph number="502" type="verses" book="th2">
_So mātarañca pitarañca,_  
_Bhaṇati ‘āpucchahaṃ gamissāmi;_  
_Isidāsiyā na saha vacchaṃ,_  
_Ekāgārehaṃ saha vatthuṃ’._  

</Paragraph>

</Division>
<Division number="417" book="th2" bookNo="19" page="426">

<Paragraph number="503" type="verses" book="th2">
_‘Mā evaṃ putta avaca,_  
_Isidāsī paṇḍitā paribyattā;_  
_Uṭṭhāyikā analasā,_  
_Kiṃ tuyhaṃ na rocate putta’._  

</Paragraph>

</Division>
<Division number="418" book="th2" bookNo="19" page="426">

<Paragraph number="504" type="verses" book="th2">
_‘Na ca me hiṃsati kiñci,_  
_Na cahaṃ isidāsiyā saha vacchaṃ;_  
_Dessāva me alaṃ me,_  
_Apucchāhaṃ gamissāmi’._  

</Paragraph>

</Division>
<Division number="419" book="th2" bookNo="19" page="426">

<Paragraph number="505" type="verses" book="th2">
_Tassa vacanaṃ suṇitvā,_  
_Sassu sasuro ca maṃ apucchiṃsu;_  
_‘Kissa tayā aparaddhaṃ,_  
_Bhaṇa vissaṭṭhā yathābhūtaṃ’._  

</Paragraph>

</Division>
<Division number="420" book="th2" bookNo="19" page="426">

<Paragraph number="506" type="verses" book="th2">
_‘Napihaṃ aparajjhaṃ kiñci,_  
_Napi hiṃsemi na bhaṇāmi dubbacanaṃ;_  
_Kiṃ sakkā kātuyye,_  
_Yaṃ maṃ viddessate bhattā’._  

</Paragraph>

</Division>
<Division number="421" book="th2" bookNo="19" page="426">

<Paragraph number="507" type="verses" book="th2">
_Te maṃ pitugharaṃ paṭinayiṃsu,_  
_Vimanā dukhena adhibhūtā;_  
_Puttamanurakkhamānā,_  
_‘Jitāmhase rūpiniṃ lakkhiṃ’._  

</Paragraph>

</Division>
<Division number="422" book="th2" bookNo="19" page="426">

<Paragraph number="508" type="verses" book="th2">
_Atha maṃ adāsi tāto,_  
_Aḍḍhassa gharamhi dutiyakulikassa;_  
_Tato upaḍḍhasuṅkena,_  
_Yena maṃ vindatha seṭṭhi._  

</Paragraph>

</Division>
<Division number="423" book="th2" bookNo="19" page="426">

<Paragraph number="509" type="verses" book="th2">
_Tassapi gharamhi māsaṃ,_  
_Avasiṃ atha sopi maṃ paṭiccharayi;_  
_Dāsīva upaṭṭhahantiṃ,_  
_Adūsikaṃ sīlasampannaṃ._  

</Paragraph>

</Division>
<Division number="424" book="th2" bookNo="19" page="426">

<Paragraph number="510" type="verses" book="th2">
_Bhikkhāya ca vicarantaṃ,_  
_Damakaṃ dantaṃ me pitā bhaṇati;_  
_‘Hohisi me jāmātā,_  
_Nikkhipa poṭṭhiñca ghaṭikañca’._  

</Paragraph>

</Division>
<Division number="425" book="th2" bookNo="19" page="426">

<Paragraph number="511" type="verses" book="th2">
_Sopi vasitvā pakkhaṃ,_  
_Atha tātaṃ bhaṇati ‘dehi me poṭṭhiṃ;_  
_Ghaṭikañca mallakañca,_  
_Punapi bhikkhaṃ carissāmi’._  

</Paragraph>

</Division>
<Division number="426" book="th2" bookNo="19" page="427">

<Paragraph number="512" type="verses" book="th2">
_Atha naṃ bhaṇatī tāto,_  
_Ammā sabbo ca me ñātigaṇavaggo;_  
_‘Kiṃ te na kīrati idha,_  
_Bhaṇa khippaṃ taṃ te karihi’ti._  

</Paragraph>

</Division>
<Division number="427" book="th2" bookNo="19" page="427">

<Paragraph number="513" type="verses" book="th2">
_Evaṃ bhaṇito bhaṇati,_  
_‘Yadi me attā sakkoti alaṃ mayhaṃ;_  
_Isidāsiyā na saha vacchaṃ,_  
_Ekagharehaṃ saha vatthuṃ’._  

</Paragraph>

</Division>
<Division number="428" book="th2" bookNo="19" page="427">

<Paragraph number="514" type="verses" book="th2">
_Vissajjito gato so,_  
_Ahampi ekākinī vicintemi;_  
_‘Āpucchitūna gacchaṃ,_  
_Marituye vā pabbajissaṃ vā’._  

</Paragraph>

</Division>
<Division number="429" book="th2" bookNo="19" page="427">

<Paragraph number="515" type="verses" book="th2">
_Atha ayyā jinadattā,_  
_Āgacchī gocarāya caramānā;_  
_Tātakulaṃ vinayadharī,_  
_Bahussutā sīlasampannā._  

</Paragraph>

</Division>
<Division number="430" book="th2" bookNo="19" page="427">

<Paragraph number="516" type="verses" book="th2">
_Taṃ disvāna amhākaṃ,_  
_Uṭṭhāyāsanaṃ tassā paññāpayiṃ;_  
_Nisinnāya ca pāde,_  
_Vanditvā bhojanamadāsiṃ._  

</Paragraph>

</Division>
<Division number="431" book="th2" bookNo="19" page="427">

<Paragraph number="517" type="verses" book="th2">
_Annena ca pānena ca,_  
_Khajjena ca yañca tattha sannihitaṃ;_  
_Santappayitvā avacaṃ,_  
_‘Ayye icchāmi pabbajituṃ’._  

</Paragraph>

</Division>
<Division number="432" book="th2" bookNo="19" page="427">

<Paragraph number="518" type="verses" book="th2">
_Atha maṃ bhaṇatī tāto,_  
_‘Idheva puttaka carāhi tvaṃ dhammaṃ;_  
_Annena ca pānena ca,_  
_Tappaya samaṇe dvijātī ca’._  

</Paragraph>

</Division>
<Division number="433" book="th2" bookNo="19" page="427">

<Paragraph number="519" type="verses" book="th2">
_Athahaṃ bhaṇāmi tātaṃ,_  
_Rodantī añjaliṃ paṇāmetvā;_  
_‘Pāpañhi mayā pakataṃ,_  
_Kammaṃ taṃ nijjaressāmi’._  

</Paragraph>

</Division>
<Division number="434" book="th2" bookNo="19" page="427">

<Paragraph number="520" type="verses" book="th2">
_Atha maṃ bhaṇatī tāto,_  
_‘Pāpuṇa bodhiñca aggadhammañca;_  
_Nibbānañca labhassu,_  
_Yaṃ sacchikarī dvipadaseṭṭho’._  

</Paragraph>

</Division>
<Division number="435" book="th2" bookNo="19" page="427">

<Paragraph number="521" type="verses" book="th2">
_Mātāpitū abhivādayitvā,_  
_Sabbañca ñātigaṇavaggaṃ;_  
_Sattāhaṃ pabbajitā,_  
_Tisso vijjā aphassayiṃ._  

</Paragraph>

</Division>
<Division number="436" book="th2" bookNo="19" page="427">

<Paragraph number="522" type="verses" book="th2">
_Jānāmi attano satta,_  
_Jātiyo yassayaṃ phalavipāko;_  
_Taṃ tava ācikkhissaṃ,_  
_Taṃ ekamanā nisāmehi._  

</Paragraph>

</Division>
<Division number="437" book="th2" bookNo="19" page="427">

<Paragraph number="523" type="verses" book="th2">
_Nagaramhi erakacche,_  
_Suvaṇṇakāro ahaṃ pahūtadhano;_  
_Yobbanamadena matto,_  
_So paradāraṃ asevihaṃ._  

</Paragraph>

</Division>
<Division number="438" book="th2" bookNo="19" page="427">

<Paragraph number="524" type="verses" book="th2">
_Sohaṃ tato cavitvā,_  
_Nirayamhi apaccisaṃ ciraṃ;_  
_Pakko tato ca uṭṭhahitvā,_  
_Makkaṭiyā kucchimokkamiṃ._  

</Paragraph>

</Division>
<Division number="439" book="th2" bookNo="19" page="428">

<Paragraph number="525" type="verses" book="th2">
_Sattāhajātakaṃ maṃ,_  
_Mahākapi yūthapo nillacchesi;_  
_Tassetaṃ kammaphalaṃ,_  
_Yathāpi gantvāna paradāraṃ._  

</Paragraph>

</Division>
<Division number="440" book="th2" bookNo="19" page="428">

<Paragraph number="526" type="verses" book="th2">
_Sohaṃ tato cavitvā,_  
_Kālaṃ karitvā sindhavāraññe;_  
_Kāṇāya ca khañjāya ca,_  
_Eḷakiyā kucchimokkamiṃ._  

</Paragraph>

</Division>
<Division number="441" book="th2" bookNo="19" page="428">

<Paragraph number="527" type="verses" book="th2">
_Dvādasa vassāni ahaṃ,_  
_Nillacchito dārake parivahitvā;_  
_Kimināvaṭṭo akallo,_  
_Yathāpi gantvāna paradāraṃ._  

</Paragraph>

</Division>
<Division number="442" book="th2" bookNo="19" page="428">

<Paragraph number="528" type="verses" book="th2">
_Sohaṃ tato cavitvā,_  
_Govāṇijakassa gāviyā jāto;_  
_Vaccho lākhātambo,_  
_Nillacchito dvādase māse._  

</Paragraph>

</Division>
<Division number="443" book="th2" bookNo="19" page="428">

<Paragraph number="529" type="verses" book="th2">
_Voḍhūna naṅgalamahaṃ,_  
_Sakaṭañca dhārayāmi;_  
_Andhovaṭṭo akallo,_  
_Yathāpi gantvāna paradāraṃ._  

</Paragraph>

</Division>
<Division number="444" book="th2" bookNo="19" page="428">

<Paragraph number="530" type="verses" book="th2">
_Sohaṃ tato cavitvā,_  
_Vīthiyā dāsiyā ghare jāto;_  
_Neva mahilā na puriso,_  
_Yathāpi gantvāna paradāraṃ._  

</Paragraph>

</Division>
<Division number="445" book="th2" bookNo="19" page="428">

<Paragraph number="531" type="verses" book="th2">
_Tiṃsativassamhi mato,_  
_Sākaṭikakulamhi dārikā jātā;_  
_Kapaṇamhi appabhoge,_  
_Dhanika purisapātabahulamhi._  

</Paragraph>

</Division>
<Division number="446" book="th2" bookNo="19" page="428">

<Paragraph number="532" type="verses" book="th2">
_Taṃ maṃ tato satthavāho,_  
_Ussannāya vipulāya vaḍḍhiyā;_  
_Okaḍḍhati vilapantiṃ,_  
_Acchinditvā kulagharasmā._  

</Paragraph>

</Division>
<Division number="447" book="th2" bookNo="19" page="428">

<Paragraph number="533" type="verses" book="th2">
_Atha soḷasame vasse,_  
_Disvā maṃ pattayobbanaṃ kaññaṃ;_  
_Orundhatassa putto,_  
_Giridāso nāma nāmena._  

</Paragraph>

</Division>
<Division number="448" book="th2" bookNo="19" page="428">

<Paragraph number="534" type="verses" book="th2">
_Tassapi aññā bhariyā,_  
_Sīlavatī guṇavatī yasavatī ca;_  
_Anurattā bhattāraṃ,_  
_Tassāhaṃ viddesanamakāsiṃ._  

</Paragraph>

</Division>
<Division number="449" book="th2" bookNo="19" page="428">

<Paragraph number="535" type="verses" book="th2">
_Tassetaṃ kammaphalaṃ,_  
_Yaṃ maṃ apakīritūna gacchanti;_  
_Dāsīva upaṭṭhahantiṃ,_  
_Tassapi anto kato mayā”ti._  


</Paragraph>
</Division>
