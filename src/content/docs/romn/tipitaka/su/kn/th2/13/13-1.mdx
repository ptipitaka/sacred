<Division number="252" book="th2">

<Paragraph number="330" type="verses" book="th2">
_“Kāḷakā bhamaravaṇṇasādisā,_  
_Vellitaggā mama muddhajā ahuṃ;_  
_Te jarāya sāṇavākasādisā,_  
_Saccavādivacanaṃ anaññathā._  

</Paragraph>

</Division>
<Division number="253" book="th2">

<Paragraph number="331" type="verses" book="th2">
_Vāsitova surabhī karaṇḍako,_  
_Pupphapūra mama uttamaṅgajo;_  
_Taṃ jarāyatha salomagandhikaṃ,_  
_Saccavādivacanaṃ anaññathā._  

</Paragraph>

</Division>
<Division number="254" book="th2">

<Paragraph number="332" type="verses" book="th2">
_Kānanaṃva sahitaṃ suropitaṃ,_  
_Kocchasūcivicitaggasobhitaṃ;_  
_Taṃ jarāya viralaṃ tahiṃ tahiṃ,_  
_Saccavādivacanaṃ anaññathā._  

</Paragraph>

</Division>
<Division number="255" book="th2">

<Paragraph number="333" type="verses" book="th2">
_Kaṇhakhandhakasuvaṇṇamaṇḍitaṃ,_  
_Sobhate suveṇīhilaṅkataṃ;_  
_Taṃ jarāya khalitaṃ siraṃ kataṃ,_  
_Saccavādivacanaṃ anaññathā._  

</Paragraph>

</Division>
<Division number="256" book="th2">

<Paragraph number="334" type="verses" book="th2">
_Cittakārasukatāva lekhikā,_  
_Sobhare su bhamukā pure mama;_  
_Tā jarāya valibhippalambitā,_  
_Saccavādivacanaṃ anaññathā._  

</Paragraph>

</Division>
<Division number="257" book="th2">

<Paragraph number="335" type="verses" book="th2">
_Bhassarā surucirā yathā maṇī,_  
_Nettahesumabhinīlamāyatā;_  
_Te jarāyabhihatā na sobhare,_  
_Saccavādivacanaṃ anaññathā._  

</Paragraph>

</Division>
<Division number="258" book="th2">

<Paragraph number="336" type="verses" book="th2">
_Saṇhatuṅgasadisī ca nāsikā,_  
_Sobhate su abhiyobbanaṃ pati;_  
_Sā jarāya upakūlitā viya,_  
_Saccavādivacanaṃ anaññathā._  

</Paragraph>

</Division>
<Division number="259" book="th2">

<Paragraph number="337" type="verses" book="th2">
_Kaṅkaṇaṃva sukataṃ suniṭṭhitaṃ,_  
_Sobhare su mama kaṇṇapāḷiyo;_  
_Tā jarāya valibhippalambitā,_  
_Saccavādivacanaṃ anaññathā._  

</Paragraph>

</Division>
<Division number="260" book="th2">

<Paragraph number="338" type="verses" book="th2">
_Pattalīmakulavaṇṇasādisā,_  
_Sobhare su dantā pure mama;_  
_Te jarāya khaṇḍitā cāsitā,_  
_Saccavādivacanaṃ anaññathā._  

</Paragraph>

</Division>
<Division number="261" book="th2">

<Paragraph number="339" type="verses" book="th2">
_Kānanamhi vanasaṇḍacārinī,_  
_Kokilāva madhuraṃ nikūjihaṃ;_  
_Taṃ jarāya khalitaṃ tahiṃ tahiṃ,_  
_Saccavādivacanaṃ anaññathā._  

</Paragraph>

</Division>
<Division number="262" book="th2">

<Paragraph number="340" type="verses" book="th2">
_Saṇhakamburiva suppamajjitā,_  
_Sobhate su gīvā pure mama;_  
_Sā jarāya bhaggā vināmitā,_  
_Saccavādivacanaṃ anaññathā._  

</Paragraph>

</Division>
<Division number="263" book="th2">

<Paragraph number="341" type="verses" book="th2">
_Vaṭṭapalighasadisopamā ubho,_  
_Sobhare su bāhā pure mama;_  
_Tā jarāya yatha pāṭalibbalitā,_  
_Saccavādivacanaṃ anaññathā._  

</Paragraph>

</Division>
<Division number="264" book="th2">

<Paragraph number="342" type="verses" book="th2">
_Saṇhamuddikasuvaṇṇamaṇḍitā,_  
_Sobhare su hatthā pure mama;_  
_Te jarāya yathā mūlamūlikā,_  
_Saccavādivacanaṃ anaññathā._  

</Paragraph>

</Division>
<Division number="265" book="th2">

<Paragraph number="343" type="verses" book="th2">
_Pīnavaṭṭasahituggatā ubho,_  
_Sobhare su thanakā pure mama;_  
_Thevikīva lambanti nodakā,_  
_Saccavādivacanaṃ anaññathā._  

</Paragraph>

</Division>
<Division number="266" book="th2">

<Paragraph number="344" type="verses" book="th2">
_Kañcanassa phalakaṃva sammaṭṭhaṃ,_  
_Sobhate su kāyo pure mama;_  
_So valīhi sukhumāhi otato,_  
_Saccavādivacanaṃ anaññathā._  

</Paragraph>

</Division>
<Division number="267" book="th2">

<Paragraph number="345" type="verses" book="th2">
_Nāgabhogasadisopamā ubho,_  
_Sobhare su ūrū pure mama;_  
_Te jarāya yathā veḷunāḷiyo,_  
_Saccavādivacanaṃ anaññathā._  

</Paragraph>

</Division>
<Division number="268" book="th2">

<Paragraph number="346" type="verses" book="th2">
_Saṇhanūpurasuvaṇṇamaṇḍitā,_  
_Sobhare su jaṅghā pure mama;_  
_Tā jarāya tiladaṇḍakāriva,_  
_Saccavādivacanaṃ anaññathā._  

</Paragraph>

</Division>
<Division number="269" book="th2">

<Paragraph number="347" type="verses" book="th2">
_Tūlapuṇṇasadisopamā ubho,_  
_Sobhare su pādā pure mama;_  
_Te jarāya phuṭitā valīmatā,_  
_Saccavādivacanaṃ anaññathā._  

</Paragraph>

</Division>
<Division number="270" book="th2">

<Paragraph number="348" type="verses" book="th2">
_Ediso ahu ayaṃ samussayo,_  
_Jajjaro bahudukkhānamālayo;_  
_Sopalepapatito jarāgharo,_  
_Saccavādivacanaṃ anaññathā”._  


</Paragraph>
</Division>
