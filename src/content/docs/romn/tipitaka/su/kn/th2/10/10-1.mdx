<Paragraph number="285" type="verses" book="th2">
_“Kalyāṇamittatā muninā,_  
_lokaṃ ādissa vaṇṇitā;_  
_Kalyāṇamitte bhajamāno,_  
_api bālo paṇḍito assa._  

</Paragraph>

</Division>
<Division number="214" book="th2" bookNo="19" page="403">

<Paragraph number="286" type="verses" book="th2">
_Bhajitabbā sappurisā,_  
_Paññā tathā vaḍḍhati bhajantānaṃ;_  
_Bhajamāno sappurise,_  
_Sabbehipi dukkhehi pamucceyya._  

</Paragraph>

</Division>
<Division number="215" book="th2" bookNo="19" page="403">

<Paragraph number="287" type="verses" book="th2">
_Dukkhañca vijāneyya,_  
_Dukkhassa ca samudayaṃ nirodhaṃ;_  
_Aṭṭhaṅgikañca maggaṃ,_  
_Cattāripi ariyasaccāni”._  

</Paragraph>

</Division>
<Division number="216" book="th2" bookNo="19" page="403">

<Paragraph number="288" type="verses" book="th2">
_“Dukkho itthibhāvo,_  
_Akkhāto purisadammasārathinā;_  
_Sapattikampi hi dukkhaṃ,_  
_Appekaccā sakiṃ vijātāyo._  

</Paragraph>

</Division>
<Division number="217" book="th2" bookNo="19" page="403">

<Paragraph number="289" type="verses" book="th2">
_Galake api kantanti,_  
_Sukhumāliniyo visāni khādanti;_  
_Janamārakamajjhagatā,_  
_Ubhopi byasanāni anubhonti”._  

</Paragraph>

</Division>
<Division number="218" book="th2" bookNo="19" page="403">

<Paragraph number="290" type="verses" book="th2">
_“Upavijaññā gacchantī,_  
_addasāhaṃ patiṃ mataṃ;_  
_Panthamhi vijāyitvāna,_  
_appattāva sakaṃ gharaṃ._  

</Paragraph>

</Division>
<Division number="219" book="th2" bookNo="19" page="403">

<Paragraph number="291" type="verses" book="th2">
_Dve puttā kālakatā,_  
_Patī ca panthe mato kapaṇikāya;_  
_Mātā pitā ca bhātā,_  
_Ḍayhanti ca ekacitakāyaṃ”._  

</Paragraph>

</Division>
<Division number="220" book="th2" bookNo="19" page="403">

<Paragraph number="292" type="verses" book="th2">
_“Khīṇakulīne kapaṇe,_  
_Anubhūtaṃ te dukhaṃ aparimāṇaṃ;_  
_Assū ca te pavattaṃ,_  
_Bahūni ca jātisahassāni._  

</Paragraph>

</Division>
<Division number="221" book="th2" bookNo="19" page="403">

<Paragraph number="293" type="verses" book="th2">
_Vasitā susānamajjhe,_  
_Athopi khāditāni puttamaṃsāni;_  
_Hatakulikā sabbagarahitā,_  
_Matapatikā amatamadhigacchiṃ._  

</Paragraph>

</Division>
<Division number="222" book="th2" bookNo="19" page="403">

<Paragraph number="294" type="verses" book="th2">
_Bhāvito me maggo,_  
_Ariyo aṭṭhaṅgiko amatagāmī;_  
_Nibbānaṃ sacchikataṃ,_  
_Dhammādāsaṃ avekkhiṃhaṃ._  

</Paragraph>

</Division>
<Division number="223" book="th2" bookNo="19" page="403">

<Paragraph number="295" type="verses" book="th2">
_Ahamamhi kantasallā,_  
_Ohitabhārā katañhi karaṇīyaṃ;_  
_Kisā gotamī therī,_  
_Vimuttacittā imaṃ bhaṇī”ti._  


</Paragraph>
</Division>

<Paragraph number="296" type="center" book="th2">
…  Kisā gotamī therī… .



</Paragraph>

<Paragraph number="297" type="center" book="th2">
Ekādasakanipāto niṭṭhito.
</Paragraph>
