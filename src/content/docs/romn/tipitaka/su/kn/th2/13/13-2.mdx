<Division number="271" book="th2">

<Paragraph number="350" type="verses" book="th2">
_“‘Samaṇā’ti bhoti supi,_  
_‘samaṇā’ti pabujjhasi;_  
_Samaṇāneva kittesi,_  
_samaṇī nūna bhavissasi._  

</Paragraph>

</Division>
<Division number="272" book="th2">

<Paragraph number="351" type="verses" book="th2">
_Vipulaṃ annañca pānañca,_  
_samaṇānaṃ paveccasi;_  
_Rohinī dāni pucchāmi,_  
_kena te samaṇā piyā._  

</Paragraph>

</Division>
<Division number="273" book="th2">

<Paragraph number="352" type="verses" book="th2">
_Akammakāmā alasā,_  
_paradattūpajīvino;_  
_Āsaṃsukā sādukāmā,_  
_kena te samaṇā piyā”._  

</Paragraph>

</Division>
<Division number="274" book="th2">

<Paragraph number="353" type="verses" book="th2">
_“Cirassaṃ vata maṃ tāta,_  
_samaṇānaṃ paripucchasi;_  
_Tesaṃ te kittayissāmi,_  
_paññāsīlaparakkamaṃ._  

</Paragraph>

</Division>
<Division number="275" book="th2">

<Paragraph number="354" type="verses" book="th2">
_Kammakāmā analasā,_  
_kammaseṭṭhassa kārakā;_  
_Rāgaṃ dosaṃ pajahanti,_  
_tena me samaṇā piyā._  

</Paragraph>

</Division>
<Division number="276" book="th2">

<Paragraph number="355" type="verses" book="th2">
_Tīṇi pāpassa mūlāni,_  
_dhunanti sucikārino;_  
_Sabbaṃ pāpaṃ pahīnesaṃ,_  
_tena me samaṇā piyā._  

</Paragraph>

</Division>
<Division number="277" book="th2">

<Paragraph number="356" type="verses" book="th2">
_Kāyakammaṃ suci nesaṃ,_  
_vacīkammañca tādisaṃ;_  
_Manokammaṃ suci nesaṃ,_  
_tena me samaṇā piyā._  

</Paragraph>

</Division>
<Division number="278" book="th2">

<Paragraph number="357" type="verses" book="th2">
_Vimalā saṅkhamuttāva,_  
_suddhā santarabāhirā;_  
_Puṇṇā sukkāna dhammānaṃ,_  
_tena me samaṇā piyā._  

</Paragraph>

</Division>
<Division number="279" book="th2">

<Paragraph number="358" type="verses" book="th2">
_Bahussutā dhammadharā,_  
_ariyā dhammajīvino;_  
_Atthaṃ dhammañca desenti,_  
_tena me samaṇā piyā._  

</Paragraph>

</Division>
<Division number="280" book="th2">

<Paragraph number="359" type="verses" book="th2">
_Bahussutā dhammadharā,_  
_ariyā dhammajīvino;_  
_Ekaggacittā satimanto,_  
_tena me samaṇā piyā._  

</Paragraph>

</Division>
<Division number="281" book="th2">

<Paragraph number="360" type="verses" book="th2">
_Dūraṅgamā satimanto,_  
_mantabhāṇī anuddhatā;_  
_Dukkhassantaṃ pajānanti,_  
_tena me samaṇā piyā._  

</Paragraph>

</Division>
<Division number="282" book="th2">

<Paragraph number="361" type="verses" book="th2">
_Yasmā gāmā pakkamanti,_  
_na vilokenti kiñcanaṃ;_  
_Anapekkhāva gacchanti,_  
_tena me samaṇā piyā._  

</Paragraph>

</Division>
<Division number="283" book="th2">

<Paragraph number="362" type="verses" book="th2">
_Na te saṃ koṭṭhe openti,_  
_na kumbhiṃ na khaḷopiyaṃ;_  
_Pariniṭṭhitamesānā,_  
_tena me samaṇā piyā._  

</Paragraph>

</Division>
<Division number="284" book="th2">

<Paragraph number="363" type="verses" book="th2">
_Na te hiraññaṃ gaṇhanti,_  
_na suvaṇṇaṃ na rūpiyaṃ;_  
_Paccuppannena yāpenti,_  
_tena me samaṇā piyā._  

</Paragraph>

</Division>
<Division number="285" book="th2">

<Paragraph number="364" type="verses" book="th2">
_Nānākulā pabbajitā,_  
_nānājanapadehi ca;_  
_Aññamaññaṃ piyāyanti,_  
_tena me samaṇā piyā”._  

</Paragraph>

</Division>
<Division number="286" book="th2">

<Paragraph number="365" type="verses" book="th2">
_“Atthāya vata no bhoti,_  
_kule jātāsi rohinī;_  
_Saddhā buddhe ca dhamme ca,_  
_saṃghe ca tibbagāravā._  

</Paragraph>

</Division>
<Division number="287" book="th2">

<Paragraph number="366" type="verses" book="th2">
_Tuvaṃ hetaṃ pajānāsi,_  
_puññakkhettaṃ anuttaraṃ;_  
_Amhampi ete samaṇā,_  
_paṭigaṇhanti dakkhiṇaṃ._  

</Paragraph>

</Division>
<Division number="288" book="th2">

<Paragraph number="367" type="verses" book="th2">
_Patiṭṭhito hettha yañño,_  
_vipulo no bhavissati”;_  
_“Sace bhāyasi dukkhassa,_  
_sace te dukkhamappiyaṃ._  

</Paragraph>

</Division>
<Division number="289" book="th2">

<Paragraph number="368" type="verses" book="th2">
_Upehi saraṇaṃ buddhaṃ,_  
_dhammaṃ saṃghañca tādinaṃ;_  
_Samādiyāhi sīlāni,_  
_taṃ te atthāya hehiti”._  

</Paragraph>

</Division>
<Division number="290" book="th2">

<Paragraph number="369" type="verses" book="th2">
_“Upemi saraṇaṃ buddhaṃ,_  
_dhammaṃ saṃghañca tādinaṃ;_  
_Samādiyāmi sīlāni,_  
_taṃ me atthāya hehiti._  

</Paragraph>

</Division>
<Division number="291" book="th2">

<Paragraph number="370" type="verses" book="th2">
_Brahmabandhu pure āsiṃ,_  
_so idānimhi brāhmaṇo;_  
_Tevijjo sottiyo camhi,_  
_vedagū camhi nhātako”._  


</Paragraph>
</Division>
