<Paragraph number="394" type="verses" book="th2">
_“Petāni bhoti puttāni,_  
_khādamānā tuvaṃ pure;_  
_Tuvaṃ divā ca ratto ca,_  
_atīva paritappasi._  

</Paragraph>

</Division>
<Division number="314" book="th2" bookNo="19" page="414">

<Paragraph number="395" type="verses" book="th2">
_Sājja sabbāni khāditvā,_  
_sataputtāni brāhmaṇī;_  
_Vāseṭṭhi kena vaṇṇena,_  
_na bāḷhaṃ paritappasi”._  

</Paragraph>

</Division>
<Division number="315" book="th2" bookNo="19" page="414">

<Paragraph number="396" type="verses" book="th2">
_“Bahūni puttasatāni,_  
_ñātisaṅghasatāni ca;_  
_Khāditāni atītaṃse,_  
_mama tuyhañca brāhmaṇa._  

</Paragraph>

</Division>
<Division number="316" book="th2" bookNo="19" page="414">

<Paragraph number="397" type="verses" book="th2">
_Sāhaṃ nissaraṇaṃ ñatvā,_  
_jātiyā maraṇassa ca;_  
_Na socāmi na rodāmi,_  
_na cāpi paritappayiṃ”._  

</Paragraph>

</Division>
<Division number="317" book="th2" bookNo="19" page="414">

<Paragraph number="398" type="verses" book="th2">
_“Abbhutaṃ vata vāseṭṭhi,_  
_vācaṃ bhāsasi edisiṃ;_  
_Kassa tvaṃ dhammamaññāya,_  
_giraṃ bhāsasi edisiṃ”._  

</Paragraph>

</Division>
<Division number="318" book="th2" bookNo="19" page="414">

<Paragraph number="399" type="verses" book="th2">
_“Esa brāhmaṇa sambuddho,_  
_nagaraṃ mithilaṃ pati;_  
_Sabbadukkhappahānāya,_  
_dhammaṃ desesi pāṇinaṃ._  

</Paragraph>

</Division>
<Division number="319" book="th2" bookNo="19" page="414">

<Paragraph number="400" type="verses" book="th2">
_Tassa brahme arahato,_  
_dhammaṃ sutvā nirūpadhiṃ;_  
_Tattha viññātasaddhammā,_  
_puttasokaṃ byapānudiṃ”._  

</Paragraph>

</Division>
<Division number="320" book="th2" bookNo="19" page="414">

<Paragraph number="401" type="verses" book="th2">
_“So ahampi gamissāmi,_  
_nagaraṃ mithilaṃ pati;_  
_Appeva maṃ so bhagavā,_  
_sabbadukkhā pamocaye”._  

</Paragraph>

</Division>
<Division number="321" book="th2" bookNo="19" page="415">

<Paragraph number="402" type="verses" book="th2">
_Addasa brāhmaṇo buddhaṃ,_  
_vippamuttaṃ nirūpadhiṃ;_  
_Svassa dhammamadesesi,_  
_muni dukkhassa pāragū._  

</Paragraph>

</Division>
<Division number="322" book="th2" bookNo="19" page="415">

<Paragraph number="403" type="verses" book="th2">
_Dukkhaṃ dukkhasamuppādaṃ,_  
_dukkhassa ca atikkamaṃ;_  
_Ariyaṃ caṭṭhaṅgikaṃ maggaṃ,_  
_dukkhūpasamagāminaṃ._  

</Paragraph>

</Division>
<Division number="323" book="th2" bookNo="19" page="415">

<Paragraph number="404" type="verses" book="th2">
_Tattha viññātasaddhammo,_  
_pabbajjaṃ samarocayi;_  
_Sujāto tīhi rattīhi,_  
_tisso vijjā aphassayi._  

</Paragraph>

</Division>
<Division number="324" book="th2" bookNo="19" page="415">

<Paragraph number="405" type="verses" book="th2">
_“Ehi sārathi gacchāhi,_  
_rathaṃ niyyādayāhimaṃ;_  
_Ārogyaṃ brāhmaṇiṃ vajja,_  
_‘pabbaji dāni brāhmaṇo;_  
_Sujāto tīhi rattīhi,_  
_tisso vijjā aphassayi’”._  

</Paragraph>

</Division>
<Division number="325" book="th2" bookNo="19" page="415">

<Paragraph number="406" type="verses" book="th2">
_Tato ca rathamādāya,_  
_sahassañcāpi sārathi;_  
_Ārogyaṃ brāhmaṇiṃvoca,_  
_“pabbaji dāni brāhmaṇo;_  
_Sujāto tīhi rattīhi,_  
_tisso vijjā aphassayi”._  

</Paragraph>

</Division>
<Division number="326" book="th2" bookNo="19" page="415">

<Paragraph number="407" type="verses" book="th2">
_“Etañcāhaṃ assarathaṃ,_  
_sahassañcāpi sārathi;_  
_Tevijjaṃ brāhmaṇaṃ sutvā,_  
_puṇṇapattaṃ dadāmi te”._  

</Paragraph>

</Division>
<Division number="327" book="th2" bookNo="19" page="415">

<Paragraph number="408" type="verses" book="th2">
_“Tuyheva hotvassaratho,_  
_sahassañcāpi brāhmaṇi;_  
_Ahampi pabbajissāmi,_  
_varapaññassa santike”._  

</Paragraph>

</Division>
<Division number="328" book="th2" bookNo="19" page="415">

<Paragraph number="409" type="verses" book="th2">
_“Hatthī gavassaṃ maṇikuṇḍalañca,_  
_Phītañcimaṃ gahavibhavaṃ pahāya;_  
_Pitā pabbajito tuyhaṃ,_  
_Bhuñja bhogāni sundarī;_  
_Tuvaṃ dāyādikā kule”._  

</Paragraph>

</Division>
<Division number="329" book="th2" bookNo="19" page="415">

<Paragraph number="410" type="verses" book="th2">
_“Hatthī gavassaṃ maṇikuṇḍalañca,_  
_Rammaṃ cimaṃ gahavibhavaṃ pahāya;_  
_Pitā pabbajito mayhaṃ,_  
_Puttasokena aṭṭito;_  
_Ahampi pabbajissāmi,_  
_Bhātusokena aṭṭitā”._  

</Paragraph>

</Division>
<Division number="330" book="th2" bookNo="19" page="415">

<Paragraph number="411" type="verses" book="th2">
_“So te ijjhatu saṅkappo,_  
_yaṃ tvaṃ patthesi sundarī;_  
_Uttiṭṭhapiṇḍo uñcho ca,_  
_paṃsukūlañca cīvaraṃ;_  
_Etāni abhisambhontī,_  
_paraloke anāsavā”._  

</Paragraph>

</Division>
<Division number="331" book="th2" bookNo="19" page="416">

<Paragraph number="412" type="verses" book="th2">
_“Sikkhamānāya me ayye,_  
_dibbacakkhu visodhitaṃ;_  
_Pubbenivāsaṃ jānāmi,_  
_yattha me vusitaṃ pure._  

</Paragraph>

</Division>
<Division number="332" book="th2" bookNo="19" page="416">

<Paragraph number="413" type="verses" book="th2">
_Tuvaṃ nissāya kalyāṇi,_  
_theri saṃghassa sobhane;_  
_Tisso vijjā anuppattā,_  
_kataṃ buddhassa sāsanaṃ._  

</Paragraph>

</Division>
<Division number="333" book="th2" bookNo="19" page="416">

<Paragraph number="414" type="verses" book="th2">
_Anujānāhi me ayye,_  
_icche sāvatthi gantave;_  
_Sīhanādaṃ nadissāmi,_  
_buddhaseṭṭhassa santike”._  

</Paragraph>

</Division>
<Division number="334" book="th2" bookNo="19" page="416">

<Paragraph number="415" type="verses" book="th2">
_“Passa sundari satthāraṃ,_  
_hemavaṇṇaṃ harittacaṃ;_  
_Adantānaṃ dametāraṃ,_  
_sambuddhamakutobhayaṃ”._  

</Paragraph>

</Division>
<Division number="335" book="th2" bookNo="19" page="416">

<Paragraph number="416" type="verses" book="th2">
_“Passa sundarimāyantiṃ,_  
_vippamuttaṃ nirūpadhiṃ;_  
_Vītarāgaṃ visaṃyuttaṃ,_  
_katakiccamanāsavaṃ._  

</Paragraph>

</Division>
<Division number="336" book="th2" bookNo="19" page="416">

<Paragraph number="417" type="verses" book="th2">
_Bārāṇasito nikkhamma,_  
_tava santikamāgatā;_  
_Sāvikā te mahāvīra,_  
_pāde vandati sundarī._  

</Paragraph>

</Division>
<Division number="337" book="th2" bookNo="19" page="416">

<Paragraph number="418" type="verses" book="th2">
_Tuvaṃ buddho tuvaṃ satthā,_  
_tuyhaṃ dhītāmhi brāhmaṇa;_  
_Orasā mukhato jātā,_  
_katakiccā anāsavā”._  

</Paragraph>

</Division>
<Division number="338" book="th2" bookNo="19" page="416">

<Paragraph number="419" type="verses" book="th2">
_“Tassā te svāgataṃ bhadde,_  
_tato te adurāgataṃ;_  
_Evañhi dantā āyanti,_  
_satthu pādāni vandikā;_  
_Vītarāgā visaṃyuttā,_  
_katakiccā anāsavā”._  


</Paragraph>
</Division>
