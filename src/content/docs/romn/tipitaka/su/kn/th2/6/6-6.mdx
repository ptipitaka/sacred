<Paragraph number="217" type="verses" book="th2">
_“Buddha vīra namo tyatthu,_  
_sabbasattānamuttama;_  
_Yo maṃ dukkhā pamocesi,_  
_aññañca bahukaṃ janaṃ._  

</Paragraph>

</Division>
<Division number="158" book="th2" bookNo="19" page="396">

<Paragraph number="218" type="verses" book="th2">
_Sabbadukkhaṃ pariññātaṃ,_  
_Hetutaṇhā visositā;_  
_Bhāvito aṭṭhaṅgiko maggo,_  
_Nirodho phusito mayā._  

</Paragraph>

</Division>
<Division number="159" book="th2" bookNo="19" page="397">

<Paragraph number="219" type="verses" book="th2">
_Mātā putto pitā bhātā,_  
_ayyakā ca pure ahuṃ;_  
_Yathābhuccamajānantī,_  
_saṃsariṃhaṃ anibbisaṃ._  

</Paragraph>

</Division>
<Division number="160" book="th2" bookNo="19" page="397">

<Paragraph number="220" type="verses" book="th2">
_Diṭṭho hi me so bhagavā,_  
_antimoyaṃ samussayo;_  
_Vikkhīṇo jātisaṃsāro,_  
_natthi dāni punabbhavo._  

</Paragraph>

</Division>
<Division number="161" book="th2" bookNo="19" page="397">

<Paragraph number="221" type="verses" book="th2">
_Āraddhavīriye pahitatte,_  
_Niccaṃ daḷhaparakkame;_  
_Samagge sāvake passe,_  
_Esā buddhāna vandanā._  

</Paragraph>

</Division>
<Division number="162" book="th2" bookNo="19" page="397">

<Paragraph number="222" type="verses" book="th2">
_Bahūnaṃ vata atthāya,_  
_Māyā janayi gotamaṃ;_  
_Byādhimaraṇatunnānaṃ,_  
_Dukkhakkhandhaṃ byapānudī”ti._  


</Paragraph>
</Division>
