<Paragraph number="255" type="verses" book="th2">
_“Satimatī cakkhumatī,_  
_bhikkhunī bhāvitindriyā;_  
_Paṭivijjhi padaṃ santaṃ,_  
_akāpurisasevitaṃ”._  

</Paragraph>

</Division>
<Division number="190" book="th2">

<Paragraph number="256" type="verses" book="th2">
_“Kiṃ nu jātiṃ na rocesi,_  
_jāto kāmāni bhuñjati;_  
_Bhuñjāhi kāmaratiyo,_  
_māhu pacchānutāpinī”._  

</Paragraph>

</Division>
<Division number="191" book="th2">

<Paragraph number="257" type="verses" book="th2">
_“Jātassa maraṇaṃ hoti,_  
_hatthapādāna chedanaṃ;_  
_Vadhabandhapariklesaṃ,_  
_jāto dukkhaṃ nigacchati._  

</Paragraph>

</Division>
<Division number="192" book="th2">

<Paragraph number="258" type="verses" book="th2">
_Atthi sakyakule jāto,_  
_sambuddho aparājito;_  
_So me dhammamadesesi,_  
_jātiyā samatikkamaṃ._  

</Paragraph>

</Division>
<Division number="193" book="th2">

<Paragraph number="259" type="verses" book="th2">
_Dukkhaṃ dukkhasamuppādaṃ,_  
_dukkhassa ca atikkamaṃ;_  
_Ariyaṃ caṭṭhaṅgikaṃ maggaṃ,_  
_dukkhūpasamagāminaṃ._  

</Paragraph>

</Division>
<Division number="194" book="th2">

<Paragraph number="260" type="verses" book="th2">
_Tassāhaṃ vacanaṃ sutvā,_  
_vihariṃ sāsane ratā;_  
_Tisso vijjā anuppattā,_  
_kataṃ buddhassa sāsanaṃ._  

</Paragraph>

</Division>
<Division number="195" book="th2">

<Paragraph number="261" type="verses" book="th2">
_Sabbattha vihatā nandī,_  
_tamokhandho padālito;_  
_Evaṃ jānāhi pāpima,_  
_nihato tvamasi antaka”._  


</Paragraph>
</Division>
