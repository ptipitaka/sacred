<Division number="175" book="th2">

<Paragraph number="239" type="verses" book="th2">
_“Musalāni gahetvāna,_  
_dhaññaṃ koṭṭenti māṇavā;_  
_Puttadārāni posentā,_  
_dhanaṃ vindanti māṇavā._  

</Paragraph>

</Division>
<Division number="176" book="th2">

<Paragraph number="240" type="verses" book="th2">
_Ghaṭetha buddhasāsane,_  
_yaṃ katvā nānutappati;_  
_Khippaṃ pādāni dhovitvā,_  
_ekamantaṃ nisīdatha._  

</Paragraph>

</Division>
<Division number="177" book="th2">

<Paragraph number="241" type="verses" book="th2">
_Cittaṃ upaṭṭhapetvāna,_  
_ekaggaṃ susamāhitaṃ;_  
_Paccavekkhatha saṅkhāre,_  
_parato no ca attato”._  

</Paragraph>

</Division>
<Division number="178" book="th2">

<Paragraph number="242" type="verses" book="th2">
_“Tassāhaṃ vacanaṃ sutvā,_  
_paṭācārānusāsaniṃ;_  
_Pāde pakkhālayitvāna,_  
_ekamante upāvisiṃ._  

</Paragraph>

</Division>
<Division number="179" book="th2">

<Paragraph number="243" type="verses" book="th2">
_Rattiyā purime yāme,_  
_pubbajātimanussariṃ;_  
_Rattiyā majjhime yāme,_  
_dibbacakkhuṃ visodhayiṃ._  

</Paragraph>

</Division>
<Division number="180" book="th2">

<Paragraph number="244" type="verses" book="th2">
_Rattiyā pacchime yāme,_  
_tamokhandhaṃ padālayiṃ;_  
_Tevijjā atha vuṭṭhāsiṃ,_  
_katā te anusāsanī._  

</Paragraph>

</Division>
<Division number="181" book="th2">

<Paragraph number="245" type="verses" book="th2">
_Sakkaṃva devā tidasā,_  
_saṅgāme aparājitaṃ;_  
_Purakkhatvā vihassāmi,_  
_tevijjāmhi anāsavā”._  


</Paragraph>
</Division>
