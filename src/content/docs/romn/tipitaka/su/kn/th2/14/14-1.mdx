<Paragraph number="452" type="verses" book="th2">
_Jīvakambavanaṃ rammaṃ,_  
_Gacchantiṃ bhikkhuniṃ subhaṃ;_  
_Dhuttako sannivāresi,_  
_Tamenaṃ abravī subhā._  

</Paragraph>

</Division>
<Division number="369" book="th2" bookNo="19" page="420">

<Paragraph number="453" type="verses" book="th2">
_“Kiṃ te aparādhitaṃ mayā,_  
_Yaṃ maṃ ovariyāna tiṭṭhasi;_  
_Na hi pabbajitāya āvuso,_  
_Puriso samphusanāya kappati._  

</Paragraph>

</Division>
<Division number="370" book="th2" bookNo="19" page="420">

<Paragraph number="454" type="verses" book="th2">
_Garuke mama satthusāsane,_  
_Yā sikkhā sugatena desitā;_  
_Parisuddhapadaṃ anaṅgaṇaṃ,_  
_Kiṃ maṃ ovariyāna tiṭṭhasi._  

</Paragraph>

</Division>
<Division number="371" book="th2" bookNo="19" page="420">

<Paragraph number="455" type="verses" book="th2">
_Āvilacitto anāvilaṃ,_  
_Sarajo vītarajaṃ anaṅgaṇaṃ;_  
_Sabbattha vimuttamānasaṃ,_  
_Kiṃ maṃ ovariyāna tiṭṭhasi”._  

</Paragraph>

</Division>
<Division number="372" book="th2" bookNo="19" page="420">

<Paragraph number="456" type="verses" book="th2">
_“Daharā ca apāpikā casi,_  
_Kiṃ te pabbajjā karissati;_  
_Nikkhipa kāsāyacīvaraṃ,_  
_Ehi ramāma supupphite vane._  

</Paragraph>

</Division>
<Division number="373" book="th2" bookNo="19" page="420">

<Paragraph number="457" type="verses" book="th2">
_Madhurañca pavanti sabbaso,_  
_Kusumarajena samuṭṭhitā dumā;_  
_Paṭhamavasanto sukho utu,_  
_Ehi ramāma supupphite vane._  

</Paragraph>

</Division>
<Division number="374" book="th2" bookNo="19" page="420">

<Paragraph number="458" type="verses" book="th2">
_Kusumitasikharā ca pādapā,_  
_Abhigajjantiva māluteritā;_  
_Kā tuyhaṃ rati bhavissati,_  
_Yadi ekā vanamogahissasi._  

</Paragraph>

</Division>
<Division number="375" book="th2" bookNo="19" page="421">

<Paragraph number="459" type="verses" book="th2">
_Vāḷamigasaṅghasevitaṃ,_  
_Kuñjaramattakareṇuloḷitaṃ;_  
_Asahāyikā gantumicchasi,_  
_Rahitaṃ bhiṃsanakaṃ mahāvanaṃ._  

</Paragraph>

</Division>
<Division number="376" book="th2" bookNo="19" page="421">

<Paragraph number="460" type="verses" book="th2">
_Tapanīyakatāva dhītikā,_  
_Vicarasi cittalateva accharā;_  
_Kāsikasukhumehi vaggubhi,_  
_Sobhasī suvasanehi nūpame._  

</Paragraph>

</Division>
<Division number="377" book="th2" bookNo="19" page="421">

<Paragraph number="461" type="verses" book="th2">
_Ahaṃ tava vasānugo siyaṃ,_  
_Yadi viharemase kānanantare;_  
_Na hi matthi tayā piyattaro,_  
_Pāṇo kinnarimandalocane._  

</Paragraph>

</Division>
<Division number="378" book="th2" bookNo="19" page="421">

<Paragraph number="462" type="verses" book="th2">
_Yadi me vacanaṃ karissasi,_  
_Sukhitā ehi agāramāvasa;_  
_Pāsādanivātavāsinī,_  
_Parikammaṃ te karontu nāriyo._  

</Paragraph>

</Division>
<Division number="379" book="th2" bookNo="19" page="421">

<Paragraph number="463" type="verses" book="th2">
_Kāsikasukhumāni dhāraya,_  
_Abhiropehi ca mālavaṇṇakaṃ;_  
_Kañcanamaṇimuttakaṃ bahuṃ,_  
_Vividhaṃ ābharaṇaṃ karomi te._  

</Paragraph>

</Division>
<Division number="380" book="th2" bookNo="19" page="421">

<Paragraph number="464" type="verses" book="th2">
_Sudhotarajapacchadaṃ subhaṃ,_  
_Gonakatūlikasanthataṃ navaṃ;_  
_Abhiruha sayanaṃ mahārahaṃ,_  
_Candanamaṇḍitasāragandhikaṃ._  

</Paragraph>

</Division>
<Division number="381" book="th2" bookNo="19" page="421">

<Paragraph number="465" type="verses" book="th2">
_Uppalaṃ cudakā samuggataṃ,_  
_Yathā taṃ amanussasevitaṃ;_  
_Evaṃ tvaṃ brahmacārinī,_  
_Sakesaṅgesu jaraṃ gamissasi”._  

</Paragraph>

</Division>
<Division number="382" book="th2" bookNo="19" page="422">

<Paragraph number="466" type="verses" book="th2">
_“Kiṃ te idha sārasammataṃ,_  
_Kuṇapapūramhi susānavaḍḍhane;_  
_Bhedanadhamme kaḷevare,_  
_Yaṃ disvā vimano udikkhasi”._  

</Paragraph>

</Division>
<Division number="383" book="th2" bookNo="19" page="422">

<Paragraph number="467" type="verses" book="th2">
_“Akkhīni ca tūriyāriva,_  
_Kinnariyāriva pabbatantare;_  
_Tava me nayanāni dakkhiya,_  
_Bhiyyo kāmaratī pavaḍḍhati._  

</Paragraph>

</Division>
<Division number="384" book="th2" bookNo="19" page="422">

<Paragraph number="468" type="verses" book="th2">
_Uppalasikharopamāni te,_  
_Vimale hāṭakasannibhe mukhe;_  
_Tava me nayanāni dakkhiya,_  
_Bhiyyo kāmaguṇo pavaḍḍhati._  

</Paragraph>

</Division>
<Division number="385" book="th2" bookNo="19" page="422">

<Paragraph number="469" type="verses" book="th2">
_Api dūragatā saramhase,_  
_Āyatapamhe visuddhadassane;_  
_Na hi matthi tayā piyattaro,_  
_Nayanā kinnarimandalocane”._  

</Paragraph>

</Division>
<Division number="386" book="th2" bookNo="19" page="422">

<Paragraph number="470" type="verses" book="th2">
_“Apathena payātumicchasi,_  
_Candaṃ kīḷanakaṃ gavesasi;_  
_Meruṃ laṅghetumicchasi,_  
_Yo tvaṃ buddhasutaṃ maggayasi._  

</Paragraph>

</Division>
<Division number="387" book="th2" bookNo="19" page="422">

<Paragraph number="471" type="verses" book="th2">
_Natthi hi loke sadevake,_  
_Rāgo yatthapi dāni me siyā;_  
_Napi naṃ jānāmi kīriso,_  
_Atha maggena hato samūlako._  

</Paragraph>

</Division>
<Division number="388" book="th2" bookNo="19" page="422">

<Paragraph number="472" type="verses" book="th2">
_Iṅgālakuyāva ujjhito,_  
_Visapattoriva aggito kato;_  
_Napi naṃ passāmi kīriso,_  
_Atha maggena hato samūlako._  

</Paragraph>

</Division>
<Division number="389" book="th2" bookNo="19" page="423">

<Paragraph number="473" type="verses" book="th2">
_Yassā siyā apaccavekkhitaṃ,_  
_Satthā vā anupāsito siyā;_  
_Tvaṃ tādisikaṃ palobhaya,_  
_Jānantiṃ so imaṃ vihaññasi._  

</Paragraph>

</Division>
<Division number="390" book="th2" bookNo="19" page="423">

<Paragraph number="474" type="verses" book="th2">
_Mayhañhi akkuṭṭhavandite,_  
_Sukhadukkhe ca satī upaṭṭhitā;_  
_Saṅkhatamasubhanti jāniya,_  
_Sabbattheva mano na limpati._  

</Paragraph>

</Division>
<Division number="391" book="th2" bookNo="19" page="423">

<Paragraph number="475" type="verses" book="th2">
_Sāhaṃ sugatassa sāvikā,_  
_Maggaṭṭhaṅgikayānayāyinī;_  
_Uddhaṭasallā anāsavā,_  
_Suññāgāragatā ramāmahaṃ._  

</Paragraph>

</Division>
<Division number="392" book="th2" bookNo="19" page="423">

<Paragraph number="476" type="verses" book="th2">
_Diṭṭhā hi mayā sucittitā,_  
_Sombhā dārukapillakāni vā;_  
_Tantīhi ca khīlakehi ca,_  
_Vinibaddhā vividhaṃ panaccakā._  

</Paragraph>

</Division>
<Division number="393" book="th2" bookNo="19" page="423">

<Paragraph number="477" type="verses" book="th2">
_Tamhuddhaṭe tantikhīlake,_  
_Vissaṭṭhe vikale parikrite;_  
_Na vindeyya khaṇḍaso kate,_  
_Kimhi tattha manaṃ nivesaye._  

</Paragraph>

</Division>
<Division number="394" book="th2" bookNo="19" page="423">

<Paragraph number="478" type="verses" book="th2">
_Tathūpamā dehakāni maṃ,_  
_Tehi dhammehi vinā na vattanti;_  
_Dhammehi vinā na vattati,_  
_Kimhi tattha manaṃ nivesaye._  

</Paragraph>

</Division>
<Division number="395" book="th2" bookNo="19" page="423">

<Paragraph number="479" type="verses" book="th2">
_Yathā haritālena makkhitaṃ,_  
_Addasa cittikaṃ bhittiyā kataṃ;_  
_Tamhi te viparītadassanaṃ,_  
_Saññā mānusikā niratthikā._  

</Paragraph>

</Division>
<Division number="396" book="th2" bookNo="19" page="424">

<Paragraph number="480" type="verses" book="th2">
_Māyaṃ viya aggato kataṃ,_  
_Supinanteva suvaṇṇapādapaṃ;_  
_Upagacchasi andha rittakaṃ,_  
_Janamajjheriva rupparūpakaṃ._  

</Paragraph>

</Division>
<Division number="397" book="th2" bookNo="19" page="424">

<Paragraph number="481" type="verses" book="th2">
_Vaṭṭaniriva koṭarohitā,_  
_Majjhe pubbuḷakā saassukā;_  
_Pīḷakoḷikā cettha jāyati,_  
_Vividhā cakkhuvidhā ca piṇḍitā”._  

</Paragraph>

</Division>
<Division number="398" book="th2" bookNo="19" page="424">

<Paragraph number="482" type="verses" book="th2">
_Uppāṭiya cārudassanā,_  
_Na ca pajjittha asaṅgamānasā;_  
_“Handa te cakkhuṃ harassu taṃ”,_  
_Tassa narassa adāsi tāvade._  

</Paragraph>

</Division>
<Division number="399" book="th2" bookNo="19" page="424">

<Paragraph number="483" type="verses" book="th2">
_Tassa ca viramāsi tāvade,_  
_Rāgo tattha khamāpayī ca naṃ;_  
_“Sotthi siyā brahmacārinī,_  
_Na puno edisakaṃ bhavissati”._  

</Paragraph>

</Division>
<Division number="400" book="th2" bookNo="19" page="424">

<Paragraph number="484" type="verses" book="th2">
_“Āsādiya edisaṃ janaṃ,_  
_Aggiṃ pajjalitaṃva liṅgiya;_  
_Gaṇhiya āsīvisaṃ viya,_  
_Api nu sotthi siyā khamehi no”._  

</Paragraph>

</Division>
<Division number="401" book="th2" bookNo="19" page="424">

<Paragraph number="485" type="verses" book="th2">
_Muttā ca tato sā bhikkhunī,_  
_Agamī buddhavarassa santikaṃ;_  
_Passiya varapuññalakkhaṇaṃ,_  
_Cakkhu āsi yathā purāṇakanti._  


</Paragraph>
</Division>

<Paragraph number="486" type="center" book="th2">
…  Subhā jīvakambavanikā therī… .



</Paragraph>
