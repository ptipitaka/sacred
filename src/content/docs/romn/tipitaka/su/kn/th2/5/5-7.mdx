<Paragraph number="145" type="verses" book="th2">
_“Agārasmiṃ vasantīhaṃ,_  
_dhammaṃ sutvāna bhikkhuno;_  
_Addasaṃ virajaṃ dhammaṃ,_  
_nibbānaṃ padamaccutaṃ._  

</Paragraph>

</Division>
<Division number="98" book="th2">

<Paragraph number="146" type="verses" book="th2">
_Sāhaṃ puttaṃ dhītarañca,_  
_dhanadhaññañca chaḍḍiya;_  
_Kese chedāpayitvāna,_  
_pabbajiṃ anagāriyaṃ._  

</Paragraph>

</Division>
<Division number="99" book="th2">

<Paragraph number="147" type="verses" book="th2">
_Sikkhamānā ahaṃ santī,_  
_bhāventī maggamañjasaṃ;_  
_Pahāsiṃ rāgadosañca,_  
_tadekaṭṭhe ca āsave._  

</Paragraph>

</Division>
<Division number="100" book="th2">

<Paragraph number="148" type="verses" book="th2">
_Bhikkhunī upasampajja,_  
_pubbajātimanussariṃ;_  
_Dibbacakkhu visodhitaṃ,_  
_vimalaṃ sādhubhāvitaṃ._  

</Paragraph>

</Division>
<Division number="101" book="th2">

<Paragraph number="149" type="verses" book="th2">
_Saṅkhāre parato disvā,_  
_hetujāte palokite;_  
_Pahāsiṃ āsave sabbe,_  
_sītibhūtāmhi nibbutā”ti._  


</Paragraph>
</Division>
