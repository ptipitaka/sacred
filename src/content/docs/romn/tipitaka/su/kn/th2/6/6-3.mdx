<Paragraph number="196" type="verses" book="th2">
_“Daharā tvaṃ rūpavatī,_  
_ahampi daharo yuvā;_  
_Pañcaṅgikena turiyena,_  
_ehi kheme ramāmase”._  

</Paragraph>

</Division>
<Division number="140" book="th2" bookNo="19" page="395">

<Paragraph number="197" type="verses" book="th2">
_“Iminā pūtikāyena,_  
_āturena pabhaṅgunā;_  
_Aṭṭiyāmi harāyāmi,_  
_kāmataṇhā samūhatā._  

</Paragraph>

</Division>
<Division number="141" book="th2" bookNo="19" page="395">

<Paragraph number="198" type="verses" book="th2">
_Sattisūlūpamā kāmā,_  
_khandhāsaṃ adhikuṭṭanā;_  
_Yaṃ ‘tvaṃ kāmaratiṃ’ brūsi,_  
_‘aratī’ dāni sā mama._  

</Paragraph>

</Division>
<Division number="142" book="th2" bookNo="19" page="395">

<Paragraph number="199" type="verses" book="th2">
_Sabbattha vihatā nandī,_  
_tamokhandho padālito;_  
_Evaṃ jānāhi pāpima,_  
_nihato tvamasi antaka._  

</Paragraph>

</Division>
<Division number="143" book="th2" bookNo="19" page="395">

<Paragraph number="200" type="verses" book="th2">
_Nakkhattāni namassantā,_  
_aggiṃ paricaraṃ vane;_  
_Yathābhuccamajānantā,_  
_bālā suddhimamaññatha._  

</Paragraph>

</Division>
<Division number="144" book="th2" bookNo="19" page="395">

<Paragraph number="201" type="verses" book="th2">
_Ahañca kho namassantī,_  
_sambuddhaṃ purisuttamaṃ;_  
_Pamuttā sabbadukkhehi,_  
_satthusāsanakārikā”ti._  


</Paragraph>
</Division>
