<Paragraph number="163" type="verses" book="th2">
_“Naṅgalehi kasaṃ khettaṃ,_  
_bījāni pavapaṃ chamā;_  
_Puttadārāni posentā,_  
_dhanaṃ vindanti māṇavā._  

</Paragraph>

</Division>
<Division number="113" book="th2" bookNo="19" page="392">

<Paragraph number="164" type="verses" book="th2">
_Kimahaṃ sīlasampannā,_  
_satthusāsanakārikā;_  
_Nibbānaṃ nādhigacchāmi,_  
_akusītā anuddhatā._  

</Paragraph>

</Division>
<Division number="114" book="th2" bookNo="19" page="392">

<Paragraph number="165" type="verses" book="th2">
_Pāde pakkhālayitvāna,_  
_udakesu karomahaṃ;_  
_Pādodakañca disvāna,_  
_thalato ninnamāgataṃ._  

</Paragraph>

</Division>
<Division number="115" book="th2" bookNo="19" page="392">

<Paragraph number="166" type="verses" book="th2">
_Tato cittaṃ samādhesiṃ,_  
_assaṃ bhadraṃvajāniyaṃ;_  
_Tato dīpaṃ gahetvāna,_  
_vihāraṃ pāvisiṃ ahaṃ;_  
_Seyyaṃ olokayitvāna,_  
_mañcakamhi upāvisiṃ._  

</Paragraph>

</Division>

_vaṭṭiṃ okassayāmahaṃ;_  
_Padīpasseva nibbānaṃ,_  
_vimokkho ahu cetaso”ti._  


</Paragraph>
