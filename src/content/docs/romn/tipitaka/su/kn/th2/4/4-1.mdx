<Division number="63" book="th2">

<Paragraph number="103" type="verses" book="th2">
_“Putto buddhassa dāyādo,_  
_kassapo susamāhito;_  
_Pubbenivāsaṃ yovedi,_  
_saggāpāyañca passati._  

</Paragraph>

</Division>
<Division number="64" book="th2">

<Paragraph number="104" type="verses" book="th2">
_Atho jātikkhayaṃ patto,_  
_abhiññāvosito muni;_  
_Etāhi tīhi vijjāhi,_  
_tevijjo hoti brāhmaṇo._  

</Paragraph>

</Division>
<Division number="65" book="th2">

<Paragraph number="105" type="verses" book="th2">
_Tatheva bhaddā kāpilānī,_  
_tevijjā maccuhāyinī;_  
_Dhāreti antimaṃ dehaṃ,_  
_jetvā māraṃ savāhiniṃ._  

</Paragraph>

</Division>
<Division number="66" book="th2">

<Paragraph number="106" type="verses" book="th2">
_Disvā ādīnavaṃ loke,_  
_ubho pabbajitā mayaṃ;_  
_Tyamha khīṇāsavā dantā,_  
_sītibhūtamha nibbutā”ti._  


</Paragraph>
</Division>

<Paragraph number="107" type="center" book="th2">
…  Bhaddā kāpilānī therī… .



</Paragraph>
