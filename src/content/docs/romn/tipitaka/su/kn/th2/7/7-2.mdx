<Paragraph number="247" type="verses" book="th2">
_“Satiṃ upaṭṭhapetvāna,_  
_bhikkhunī bhāvitindriyā;_  
_Paṭivijjhi padaṃ santaṃ,_  
_saṅkhārūpasamaṃ sukhaṃ”._  

</Paragraph>

</Division>
<Division number="183" book="th2">

<Paragraph number="248" type="verses" book="th2">
_“Kaṃ nu uddissa muṇḍāsi,_  
_samaṇī viya dissasi;_  
_Na ca rocesi pāsaṇḍe,_  
_kimidaṃ carasi momuhā”._  

</Paragraph>

</Division>
<Division number="184" book="th2">

<Paragraph number="249" type="verses" book="th2">
_“Ito bahiddhā pāsaṇḍā,_  
_diṭṭhiyo upanissitā;_  
_Na te dhammaṃ vijānanti,_  
_na te dhammassa kovidā._  

</Paragraph>

</Division>
<Division number="185" book="th2">

<Paragraph number="250" type="verses" book="th2">
_Atthi sakyakule jāto,_  
_buddho appaṭipuggalo;_  
_So me dhammamadesesi,_  
_diṭṭhīnaṃ samatikkamaṃ._  

</Paragraph>

</Division>
<Division number="186" book="th2">

<Paragraph number="251" type="verses" book="th2">
_Dukkhaṃ dukkhasamuppādaṃ,_  
_Dukkhassa ca atikkamaṃ;_  
_Ariyaṃ caṭṭhaṅgikaṃ maggaṃ,_  
_Dukkhūpasamagāminaṃ._  

</Paragraph>

</Division>
<Division number="187" book="th2">

<Paragraph number="252" type="verses" book="th2">
_Tassāhaṃ vacanaṃ sutvā,_  
_vihariṃ sāsane ratā;_  
_Tisso vijjā anuppattā,_  
_kataṃ buddhassa sāsanaṃ._  

</Paragraph>

</Division>
<Division number="188" book="th2">

<Paragraph number="253" type="verses" book="th2">
_Sabbattha vihatā nandī,_  
_tamokhandho padālito;_  
_Evaṃ jānāhi pāpima,_  
_nihato tvamasi antaka”._  


</Paragraph>
</Division>
