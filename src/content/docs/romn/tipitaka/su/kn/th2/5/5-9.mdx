<Paragraph number="157" type="verses" book="th2">
_“Lūnakesī paṅkadharī,_  
_ekasāṭī pure cariṃ;_  
_Avajje vajjamatinī,_  
_vajje cāvajjadassinī._  

</Paragraph>

</Division>
<Division number="108" book="th2" bookNo="19" page="391">

<Paragraph number="158" type="verses" book="th2">
_Divāvihārā nikkhamma,_  
_gijjhakūṭamhi pabbate;_  
_Addasaṃ virajaṃ buddhaṃ,_  
_bhikkhusaṃghapurakkhataṃ._  

</Paragraph>

</Division>
<Division number="109" book="th2" bookNo="19" page="391">

<Paragraph number="159" type="verses" book="th2">
_Nihacca jāṇuṃ vanditvā,_  
_sammukhā añjaliṃ akaṃ;_  
_‘Ehi bhadde’ti maṃ avaca,_  
_sā me āsūpasampadā._  

</Paragraph>

</Division>
<Division number="110" book="th2" bookNo="19" page="392">

<Paragraph number="160" type="verses" book="th2">
_Ciṇṇā aṅgā ca magadhā,_  
_vajjī kāsī ca kosalā;_  
_Anaṇā paṇṇāsa vassāni,_  
_raṭṭhapiṇḍaṃ abhuñjahaṃ._  

</Paragraph>

</Division>
<Division number="111" book="th2" bookNo="19" page="392">

<Paragraph number="161" type="verses" book="th2">
_Puññaṃ vata pasavi bahuṃ,_  
_Sappañño vatāyaṃ upāsako;_  
_Yo bhaddāya cīvaraṃ adāsi,_  
_Vippamuttāya sabbaganthehī”ti._  


</Paragraph>
</Division>
