<Paragraph number="372" type="verses" book="th2">
_“Laṭṭhihattho pure āsi,_  
_so dāni migaluddako;_  
_Āsāya palipā ghorā,_  
_nāsakkhi pārametave._  

</Paragraph>

</Division>
<Division number="293" book="th2" bookNo="19" page="412">

<Paragraph number="373" type="verses" book="th2">
_Sumattaṃ maṃ maññamānā,_  
_cāpā puttamatosayi;_  
_Cāpāya bandhanaṃ chetvā,_  
_pabbajissaṃ punopahaṃ”._  

</Paragraph>

</Division>
<Division number="294" book="th2" bookNo="19" page="412">

<Paragraph number="374" type="verses" book="th2">
_“Mā me kujjhi mahāvīra,_  
_mā me kujjhi mahāmuni;_  
_Na hi kodhaparetassa,_  
_suddhi atthi kuto tapo”._  

</Paragraph>

</Division>
<Division number="295" book="th2" bookNo="19" page="412">

<Paragraph number="375" type="verses" book="th2">
_“Pakkamissañca nāḷāto,_  
_kodha nāḷāya vacchati;_  
_Bandhantī itthirūpena,_  
_samaṇe dhammajīvino”._  

</Paragraph>

</Division>
<Division number="296" book="th2" bookNo="19" page="412">

<Paragraph number="376" type="verses" book="th2">
_“Ehi kāḷa nivattassu,_  
_bhuñja kāme yathā pure;_  
_Ahañca te vasīkatā,_  
_ye ca me santi ñātakā”._  

</Paragraph>

</Division>
<Division number="297" book="th2" bookNo="19" page="413">

<Paragraph number="377" type="verses" book="th2">
_“Etto cāpe catubbhāgaṃ,_  
_yathā bhāsasi tvañca me;_  
_Tayi rattassa posassa,_  
_uḷāraṃ vata taṃ siyā”._  

</Paragraph>

</Division>
<Division number="298" book="th2" bookNo="19" page="413">

<Paragraph number="378" type="verses" book="th2">
_“Kāḷaṅginiṃva takkāriṃ,_  
_pupphitaṃ girimuddhani;_  
_Phullaṃ dālimalaṭṭhiṃva,_  
_antodīpeva pāṭaliṃ._  

</Paragraph>

</Division>
<Division number="299" book="th2" bookNo="19" page="413">

<Paragraph number="379" type="verses" book="th2">
_Haricandanalittaṅgiṃ,_  
_kāsikuttamadhāriniṃ;_  
_Taṃ maṃ rūpavatiṃ santiṃ,_  
_kassa ohāya gacchasi”._  

</Paragraph>

</Division>
<Division number="300" book="th2" bookNo="19" page="413">

<Paragraph number="380" type="verses" book="th2">
_“Sākuntikova sakuṇiṃ,_  
_yathā bandhitumicchati;_  
_Āharimena rūpena,_  
_na maṃ tvaṃ bādhayissasi”._  

</Paragraph>

</Division>
<Division number="301" book="th2" bookNo="19" page="413">

<Paragraph number="381" type="verses" book="th2">
_“Imañca me puttaphalaṃ,_  
_kāḷa uppāditaṃ tayā;_  
_Taṃ maṃ puttavatiṃ santiṃ,_  
_kassa ohāya gacchasi”._  

</Paragraph>

</Division>
<Division number="302" book="th2" bookNo="19" page="413">

<Paragraph number="382" type="verses" book="th2">
_“Jahanti putte sappaññā,_  
_tato ñātī tato dhanaṃ;_  
_Pabbajanti mahāvīrā,_  
_nāgo chetvāva bandhanaṃ”._  

</Paragraph>

</Division>
<Division number="303" book="th2" bookNo="19" page="413">

<Paragraph number="383" type="verses" book="th2">
_“Idāni te imaṃ puttaṃ,_  
_Daṇḍena churikāya vā;_  
_Bhūmiyaṃ vā nisumbhissaṃ,_  
_Puttasokā na gacchasi”._  

</Paragraph>

</Division>
<Division number="304" book="th2" bookNo="19" page="413">

<Paragraph number="384" type="verses" book="th2">
_“Sace puttaṃ siṅgālānaṃ,_  
_kukkurānaṃ padāhisi;_  
_Na maṃ puttakatte jammi,_  
_punarāvattayissasi”._  

</Paragraph>

</Division>
<Division number="305" book="th2" bookNo="19" page="413">

<Paragraph number="385" type="verses" book="th2">
_“Handa kho dāni bhaddante,_  
_kuhiṃ kāḷa gamissasi;_  
_Katamaṃ gāmanigamaṃ,_  
_nagaraṃ rājadhāniyo”._  

</Paragraph>

</Division>
<Division number="306" book="th2" bookNo="19" page="413">

<Paragraph number="386" type="verses" book="th2">
_“Ahumha pubbe gaṇino,_  
_Assamaṇā samaṇamānino;_  
_Gāmena gāmaṃ vicarimha,_  
_Nagare rājadhāniyo._  

</Paragraph>

</Division>
<Division number="307" book="th2" bookNo="19" page="413">

<Paragraph number="387" type="verses" book="th2">
_Eso hi bhagavā buddho,_  
_nadiṃ nerañjaraṃ pati;_  
_Sabbadukkhappahānāya,_  
_dhammaṃ deseti pāṇinaṃ;_  
_Tassāhaṃ santikaṃ gacchaṃ,_  
_so me satthā bhavissati”._  

</Paragraph>

</Division>
<Division number="308" book="th2" bookNo="19" page="413">

<Paragraph number="388" type="verses" book="th2">
_“Vandanaṃ dāni vajjāsi,_  
_lokanāthaṃ anuttaraṃ;_  
_Padakkhiṇañca katvāna,_  
_ādiseyyāsi dakkhiṇaṃ”._  

</Paragraph>

</Division>
<Division number="309" book="th2" bookNo="19" page="413">

<Paragraph number="389" type="verses" book="th2">
_“Etaṃ kho labbhamamhehi,_  
_yathā bhāsasi tvañca me;_  
_Vandanaṃ dāni te vajjaṃ,_  
_lokanāthaṃ anuttaraṃ;_  
_Padakkhiṇañca katvāna,_  
_ādisissāmi dakkhiṇaṃ”._  

</Paragraph>

</Division>
<Division number="310" book="th2" bookNo="19" page="414">

<Paragraph number="390" type="verses" book="th2">
_Tato ca kāḷo pakkāmi,_  
_nadiṃ nerañjaraṃ pati;_  
_So addasāsi sambuddhaṃ,_  
_desentaṃ amataṃ padaṃ._  

</Paragraph>

</Division>
<Division number="311" book="th2" bookNo="19" page="414">

<Paragraph number="391" type="verses" book="th2">
_Dukkhaṃ dukkhasamuppādaṃ,_  
_dukkhassa ca atikkamaṃ;_  
_Ariyaṃ caṭṭhaṅgikaṃ maggaṃ,_  
_dukkhūpasamagāminaṃ._  

</Paragraph>

</Division>
<Division number="312" book="th2" bookNo="19" page="414">

<Paragraph number="392" type="verses" book="th2">
_Tassa pādāni vanditvā,_  
_Katvāna naṃ padakkhiṇaṃ;_  
_Cāpāya ādisitvāna,_  
_Pabbajiṃ anagāriyaṃ;_  
_Tisso vijjā anuppattā,_  
_Kataṃ buddhassa sāsanaṃ._  


</Paragraph>
</Division>
