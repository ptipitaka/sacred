<Division number="224" book="th2">

<Paragraph number="298" type="verses" book="th2">
_“Ubho mātā ca dhītā ca,_  
_mayaṃ āsuṃ sapattiyo;_  
_Tassā me ahu saṃvego,_  
_abbhuto lomahaṃsano._  

</Paragraph>

</Division>
<Division number="225" book="th2">

<Paragraph number="299" type="verses" book="th2">
_Dhiratthu kāmā asucī,_  
_duggandhā bahukaṇṭakā;_  
_Yattha mātā ca dhītā ca,_  
_sabhariyā mayaṃ ahuṃ._  

</Paragraph>

</Division>
<Division number="226" book="th2">

<Paragraph number="300" type="verses" book="th2">
_Kāmesvādīnavaṃ disvā,_  
_nekkhammaṃ daṭṭhu khemato;_  
_Sā pabbajjiṃ rājagahe,_  
_agārasmānagāriyaṃ._  

</Paragraph>

</Division>
<Division number="227" book="th2">

<Paragraph number="301" type="verses" book="th2">
_Pubbenivāsaṃ jānāmi,_  
_dibbacakkhu visodhitaṃ;_  
_Cetopariccañāṇañca,_  
_sotadhātu visodhitā._  

</Paragraph>

</Division>
<Division number="228" book="th2">

<Paragraph number="302" type="verses" book="th2">
_Iddhīpi me sacchikatā,_  
_patto me āsavakkhayo;_  
_Chaḷabhiññā sacchikatā,_  
_kataṃ buddhassa sāsanaṃ._  

</Paragraph>

</Division>
<Division number="229" book="th2">

<Paragraph number="303" type="verses" book="th2">
_Iddhiyā abhinimmitvā,_  
_caturassaṃ rathaṃ ahaṃ;_  
_Buddhassa pāde vanditvā,_  
_lokanāthassa tādino”._  

</Paragraph>

</Division>
<Division number="230" book="th2">

<Paragraph number="304" type="verses" book="th2">
_“Supupphitaggaṃ upagamma pādapaṃ,_  
_Ekā tuvaṃ tiṭṭhasi sālamūle;_  
_Na cāpi te dutiyo atthi koci,_  
_Bāle na tvaṃ bhāyasi dhuttakānaṃ”._  

</Paragraph>

</Division>
<Division number="231" book="th2">

<Paragraph number="305" type="verses" book="th2">
_“Sataṃ sahassānipi dhuttakānaṃ,_  
_Samāgatā edisakā bhaveyyuṃ;_  
_Lomaṃ na iñje napi sampavedhe,_  
_Kiṃ me tuvaṃ māra karissaseko._  

</Paragraph>

</Division>
<Division number="232" book="th2">

<Paragraph number="306" type="verses" book="th2">
_Esā antaradhāyāmi,_  
_kucchiṃ vā pavisāmi te;_  
_Bhamukantare tiṭṭhāmi,_  
_tiṭṭhantiṃ maṃ na dakkhasi._  

</Paragraph>

</Division>
<Division number="233" book="th2">

<Paragraph number="307" type="verses" book="th2">
_Cittamhi vasībhūtāhaṃ,_  
_iddhipādā subhāvitā;_  
_Chaḷabhiññā sacchikatā,_  
_kataṃ buddhassa sāsanaṃ._  

</Paragraph>

</Division>
<Division number="234" book="th2">

<Paragraph number="308" type="verses" book="th2">
_Sattisūlūpamā kāmā,_  
_khandhāsaṃ adhikuṭṭanā;_  
_Yaṃ tvaṃ ‘kāmaratiṃ’ brūsi,_  
_‘āratī’ dāni sā mama._  

</Paragraph>

</Division>
<Division number="235" book="th2">

<Paragraph number="309" type="verses" book="th2">
_Sabbattha vihatā nandī,_  
_Tamokhandho padālito;_  
_Evaṃ jānāhi pāpima,_  
_Nihato tvamasi antakā”ti._  


</Paragraph>
</Division>

<Paragraph number="310" type="center" book="th2">
…  Uppalavaṇṇā therī… .



</Paragraph>

<Paragraph number="311" type="center" book="th2">
Dvādasakanipāto niṭṭhito.
</Paragraph>
