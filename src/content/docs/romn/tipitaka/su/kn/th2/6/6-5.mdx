<Division number="151" book="th2">

<Paragraph number="210" type="verses" book="th2">
_“Ucce kule ahaṃ jātā,_  
_bahuvitte mahaddhane;_  
_Vaṇṇarūpena sampannā,_  
_dhītā majjhassa atrajā._  

</Paragraph>

</Division>
<Division number="152" book="th2">

<Paragraph number="211" type="verses" book="th2">
_Patthitā rājaputtehi,_  
_seṭṭhiputtehi gijjhitā;_  
_Pitu me pesayī dūtaṃ,_  
_detha mayhaṃ anopamaṃ._  

</Paragraph>

</Division>
<Division number="153" book="th2">

<Paragraph number="212" type="verses" book="th2">
_Yattakaṃ tulitā esā,_  
_tuyhaṃ dhītā anopamā;_  
_Tato aṭṭhaguṇaṃ dassaṃ,_  
_hiraññaṃ ratanāni ca._  

</Paragraph>

</Division>
<Division number="154" book="th2">

<Paragraph number="213" type="verses" book="th2">
_Sāhaṃ disvāna sambuddhaṃ,_  
_lokajeṭṭhaṃ anuttaraṃ;_  
_Tassa pādāni vanditvā,_  
_ekamantaṃ upāvisiṃ._  

</Paragraph>

</Division>
<Division number="155" book="th2">

<Paragraph number="214" type="verses" book="th2">
_So me dhammamadesesi,_  
_anukampāya gotamo;_  
_Nisinnā āsane tasmiṃ,_  
_phusayiṃ tatiyaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="156" book="th2">

<Paragraph number="215" type="verses" book="th2">
_Tato kesāni chetvāna,_  
_pabbajiṃ anagāriyaṃ;_  
_Ajja me sattamī ratti,_  
_yato taṇhā visositā”ti._  


</Paragraph>
</Division>
