<Paragraph number="127" type="verses" book="th2">
_“Āturaṃ asuciṃ pūtiṃ,_  
_passa nande samussayaṃ;_  
_Asubhāya cittaṃ bhāvehi,_  
_ekaggaṃ susamāhitaṃ._  

</Paragraph>

</Division>
<Division number="83" book="th2" bookNo="19" page="389">

<Paragraph number="128" type="verses" book="th2">
_Yathā idaṃ tathā etaṃ,_  
_yathā etaṃ tathā idaṃ;_  
_Duggandhaṃ pūtikaṃ vāti,_  
_bālānaṃ abhinanditaṃ._  

</Paragraph>

</Division>
<Division number="84" book="th2" bookNo="19" page="389">

<Paragraph number="129" type="verses" book="th2">
_Evametaṃ avekkhantī,_  
_rattindivamatanditā;_  
_Tato sakāya paññāya,_  
_abhinibbijjha dakkhisaṃ”._  

</Paragraph>

</Division>
<Division number="85" book="th2" bookNo="19" page="389">

<Paragraph number="130" type="verses" book="th2">
_“Tassā me appamattāya,_  
_vicinantiyā yoniso;_  
_Yathābhūtaṃ ayaṃ kāyo,_  
_diṭṭho santarabāhiro._  

</Paragraph>

</Division>
<Division number="86" book="th2" bookNo="19" page="389">

<Paragraph number="131" type="verses" book="th2">
_Atha nibbindahaṃ kāye,_  
_ajjhattañca virajjahaṃ;_  
_Appamattā visaṃyuttā,_  
_upasantāmhi nibbutā”ti._  


</Paragraph>
</Division>
