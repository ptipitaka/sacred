<Paragraph number="1424" type="verses" book="ap2">
_“Yo raṃsiphusitāveḷo,_  
_phusso nāma mahāmuni;_  
_Tassāhaṃ bhaginī āsiṃ,_  
_ajāyiṃ khattiye kule._  

</Paragraph>

</Division>
<Division number="205" book="ap2" bookNo="20" page="159">

<Paragraph number="1425" type="verses" book="ap2">
_Tassa dhammaṃ suṇitvāhaṃ,_  
_vippasannena cetasā;_  
_Mahādānaṃ daditvāna,_  
_patthayiṃ rūpasampadaṃ._  

</Paragraph>

</Division>
<Division number="206" book="ap2" bookNo="20" page="159">

<Paragraph number="1426" type="verses" book="ap2">
_Ekattiṃse ito kappe,_  
_sikhī lokagganāyako;_  
_Uppanno lokapajjoto,_  
_tilokasaraṇo jino._  

</Paragraph>

</Division>
<Division number="207" book="ap2" bookNo="20" page="159">

<Paragraph number="1427" type="verses" book="ap2">
_Tadāruṇapure ramme,_  
_brāhmaññakulasambhavā;_  
_Vimuttacittaṃ kupitā,_  
_bhikkhuniṃ abhisāpayiṃ._  

</Paragraph>

</Division>
<Division number="208" book="ap2" bookNo="20" page="159">

<Paragraph number="1428" type="verses" book="ap2">
_‘Vesikāva anācārā’,_  
_jinasāsanadūsikā;_  
_Evaṃ akkosayitvāna,_  
_tena pāpena kammunā._  

</Paragraph>

</Division>
<Division number="209" book="ap2" bookNo="20" page="160">

<Paragraph number="1429" type="verses" book="ap2">
_Dāruṇaṃ nirayaṃ gantvā,_  
_mahādukkhasamappitā;_  
_Tato cutā manussesu,_  
_upapannā tapassinī._  

</Paragraph>

</Division>
<Division number="210" book="ap2" bookNo="20" page="160">

<Paragraph number="1430" type="verses" book="ap2">
_Dasajātisahassāni,_  
_gaṇikattamakārayiṃ;_  
_Tamhā pāpā na muccissaṃ,_  
_bhutvā duṭṭhavisaṃ yathā._  

</Paragraph>

</Division>
<Division number="211" book="ap2" bookNo="20" page="160">

<Paragraph number="1431" type="verses" book="ap2">
_Brahmacariyamasevissaṃ,_  
_kassape jinasāsane;_  
_Tena kammavipākena,_  
_ajāyiṃ tidase pure._  

</Paragraph>

</Division>
<Division number="212" book="ap2" bookNo="20" page="160">

<Paragraph number="1432" type="verses" book="ap2">
_Pacchime bhave sampatte,_  
_ahosiṃ opapātikā;_  
_Ambasākhantare jātā,_  
_ambapālīti tenahaṃ._  

</Paragraph>

</Division>
<Division number="213" book="ap2" bookNo="20" page="160">

<Paragraph number="1433" type="verses" book="ap2">
_Parivutā pāṇakoṭīhi,_  
_pabbajiṃ jinasāsane;_  
_Pattāhaṃ acalaṃ ṭhānaṃ,_  
_dhītā buddhassa orasā._  

</Paragraph>

</Division>
<Division number="214" book="ap2" bookNo="20" page="160">

<Paragraph number="1434" type="verses" book="ap2">
_Iddhīsu ca vasī homi,_  
_sotadhātuvisuddhiyā;_  
_Cetopariyañāṇassa,_  
_vasī homi mahāmuni._  

</Paragraph>

</Division>
<Division number="215" book="ap2" bookNo="20" page="160">

<Paragraph number="1435" type="verses" book="ap2">
_Pubbenivāsaṃ jānāmi,_  
_dibbacakkhu visodhitaṃ;_  
_Sabbāsavā parikkhīṇā,_  
_natthi dāni punabbhavo._  

</Paragraph>

</Division>
<Division number="216" book="ap2" bookNo="20" page="160">

<Paragraph number="1436" type="verses" book="ap2">
_Atthadhammaniruttīsu,_  
_paṭibhāne tatheva ca;_  
_Ñāṇaṃ me vimalaṃ suddhaṃ,_  
_buddhaseṭṭhassa vāhasā._  

</Paragraph>

</Division>

_ …pe… _  
_viharāmi anāsavā._  

</Paragraph>

_ …pe… _

</Paragraph>

_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._
