<Paragraph number="58" type="verses" book="ap2">
_“Nagare bandhumatiyā,_  
_bandhumā nāma khattiyo;_  
_Tassa rañño ahuṃ bhariyā,_  
_ekajjhaṃ cārayāmahaṃ._  

</Paragraph>

</Division>
<Division number="47" book="ap2" bookNo="20" page="5">

<Paragraph number="59" type="verses" book="ap2">
_Rahogatā nisīditvā,_  
_evaṃ cintesahaṃ tadā;_  
_‘Ādāya gamanīyaṃ hi,_  
_kusalaṃ natthi me kataṃ._  

</Paragraph>

</Division>
<Division number="48" book="ap2" bookNo="20" page="5">

<Paragraph number="60" type="verses" book="ap2">
_Mahābhitāpaṃ kaṭukaṃ,_  
_ghorarūpaṃ sudāruṇaṃ;_  
_Nirayaṃ nūna gacchāmi,_  
_ettha me natthi saṃsayo’._  

</Paragraph>

</Division>
<Division number="49" book="ap2" bookNo="20" page="5">

<Paragraph number="61" type="verses" book="ap2">
_Rājānaṃ upasaṅkamma,_  
_idaṃ vacanamabraviṃ;_  
_‘Ekaṃ me samaṇaṃ dehi,_  
_bhojayissāmi khattiya’._  

</Paragraph>

</Division>
<Division number="50" book="ap2" bookNo="20" page="5">

<Paragraph number="62" type="verses" book="ap2">
_Adāsi me mahārājā,_  
_samaṇaṃ bhāvitindriyaṃ;_  
_Tassa pattaṃ gahetvāna,_  
_paramannena pūrayiṃ._  

</Paragraph>

</Division>
<Division number="51" book="ap2" bookNo="20" page="5">

<Paragraph number="63" type="verses" book="ap2">
_Pūrayitvā paramannaṃ,_  
_gandhālepaṃ akāsahaṃ;_  
_Jālena pidahitvāna,_  
_vatthayugena chādayiṃ._  

</Paragraph>

</Division>
<Division number="52" book="ap2" bookNo="20" page="5">

<Paragraph number="64" type="verses" book="ap2">
_Ārammaṇaṃ mamaṃ etaṃ,_  
_sarāmi yāvajīvihaṃ;_  
_Tattha cittaṃ pasādetvā,_  
_tāvatiṃsamagacchahaṃ._  

</Paragraph>

</Division>
<Division number="53" book="ap2" bookNo="20" page="5">

<Paragraph number="65" type="verses" book="ap2">
_Tiṃsānaṃ devarājūnaṃ,_  
_mahesittamakārayiṃ;_  
_Manasā patthitaṃ mayhaṃ,_  
_nibbattati yathicchitaṃ._  

</Paragraph>

</Division>
<Division number="54" book="ap2" bookNo="20" page="5">

<Paragraph number="66" type="verses" book="ap2">
_Vīsānaṃ cakkavattīnaṃ,_  
_mahesittamakārayiṃ;_  
_Ocitattāva hutvāna,_  
_saṃsarāmi bhavesvahaṃ._  

</Paragraph>

</Division>
<Division number="55" book="ap2" bookNo="20" page="6">

<Paragraph number="67" type="verses" book="ap2">
_Sabbabandhanamuttāhaṃ,_  
_apetā me upādikā;_  
_Sabbāsavā parikkhīṇā,_  
_natthi dāni punabbhavo._  

</Paragraph>

</Division>
<Division number="56" book="ap2" bookNo="20" page="6">

<Paragraph number="68" type="verses" book="ap2">
_Ekanavutito kappe,_  
_yaṃ dānamadadiṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_piṇḍapātassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="57" book="ap2" bookNo="20" page="6">

<Paragraph number="69" type="verses" book="ap2">
_Kilesā jhāpitā mayhaṃ,_  
_ …pe… _  
_viharāmi anāsavā._  

</Paragraph>

</Division>
<Division number="58" book="ap2" bookNo="20" page="6">

<Paragraph number="70" type="verses" book="ap2">
_Svāgataṃ vata me āsi,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ._  

</Paragraph>

</Division>
<Division number="59" book="ap2" bookNo="20" page="6">

<Paragraph number="71" type="verses" book="ap2">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="72" book="ap2">
Itthaṃ sudaṃ ekapiṇḍapātadāyikā bhikkhunī imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>
