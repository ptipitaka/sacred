_“Bhikkhunī uppalavaṇṇā,_  
_iddhiyā pāramiṃ gatā;_  
_Vanditvā satthuno pāde,_  
_idaṃ vacanamabravi._  

</Paragraph>

</Division>
<Division number="385" book="ap2">

<Paragraph number="559" type="verses" book="ap2">
_‘Nitthiṇṇā jātisaṃsāraṃ,_  
_pattāhaṃ acalaṃ padaṃ;_  
_Sabbadukkhaṃ mayā khīṇaṃ,_  
_ārocemi mahāmuni._  

</Paragraph>

</Division>
<Division number="386" book="ap2">

<Paragraph number="560" type="verses" book="ap2">
_Yāvatā parisā atthi,_  
_pasannā jinasāsane;_  
_Yassā ca meparādhotthi,_  
_khamantu jinasammukhā._  

</Paragraph>

</Division>
<Division number="387" book="ap2">

<Paragraph number="561" type="verses" book="ap2">
_Saṃsāre saṃsarantiyā,_  
_khalitaṃ me sace bhave;_  
_Ārocemi mahāvīra,_  
_aparādhaṃ khamassu taṃ’._  

</Paragraph>

</Division>
<Division number="388" book="ap2">

<Paragraph number="562" type="verses" book="ap2">
_‘Iddhiñcāpi nidassehi,_  
_mama sāsanakārike;_  
_Catasso parisā ajja,_  
_kaṅkhaṃ chindāhi yāvatā’._  

</Paragraph>

</Division>
<Division number="389" book="ap2">

<Paragraph number="563" type="verses" book="ap2">
_‘Dhītā tuyhaṃ mahāvīra,_  
_paññavanta jutindhara;_  
_Bahuñca dukkaraṃ kammaṃ,_  
_kataṃ me atidukkaraṃ._  

</Paragraph>

</Division>
<Division number="390" book="ap2">

<Paragraph number="564" type="verses" book="ap2">
_Uppalasseva me vaṇṇo,_  
_nāmenuppalanāmikā;_  
_Sāvikā te mahāvīra,_  
_pāde vandāmi cakkhuma._  

</Paragraph>

</Division>
<Division number="391" book="ap2">

<Paragraph number="565" type="verses" book="ap2">

_nekajātisate bahū;_  
_Ekasmiṃ sambhave jātā,_  
_samānachandamānasā._  

</Paragraph>

</Division>
<Division number="392" book="ap2">

_jātiyāpi ca ekato;_  
_Pacchime bhave sampatte,_  
_ubhopi nānāsambhavā._  

</Paragraph>

</Division>
<Division number="393" book="ap2">

_dhītā uppalasavhayā;_  
_Passa vīra mamaṃ iddhiṃ,_  
_balaṃ dassemi satthuno’._  

</Paragraph>

</Division>

_khiḍḍo komārako yathā._  

</Paragraph>

</Division>
<Division number="395" book="ap2">

_pakkhipi hatthapātiyaṃ;_

_luñci komārako yuvā._  

</Paragraph>

</Division>

_chādayitvāna matthake;_

_nānāvaṇṇaṃ punappunaṃ._  

</Paragraph>

</Division>

</Paragraph>

</Division>

</Paragraph>

</Paragraph>

</Division>

</Paragraph>

_natthi dāni punabbhavo._  

</Paragraph>

</Paragraph>

</Paragraph>

</Paragraph>

_tavatthāya mahāmuni’._  

</Paragraph>

</Division>

_atuliddhiparakkamā’._  

</Paragraph>

</Division>

</Paragraph>

</Paragraph>

</Paragraph>

</Paragraph>
