<Paragraph number="157" type="verses" book="ap2">
_“Nagare bandhumatiyā,_  
_bandhumā nāma khattiyo;_  
_Divase puṇṇamāya so,_  
_upavasi uposathaṃ._  

</Paragraph>

</Division>
<Division number="2" book="ap2">

<Paragraph number="158" type="verses" book="ap2">
_Ahaṃ tena samayena,_  
_kumbhadāsī ahaṃ tahiṃ;_  
_Disvā sarājakaṃ senaṃ,_  
_evāhaṃ cintayiṃ tadā._  

</Paragraph>

</Division>
<Division number="3" book="ap2">

<Paragraph number="159" type="verses" book="ap2">
_‘Rājāpi rajjaṃ chaḍḍetvā,_  
_upavasi uposathaṃ;_  
_Saphalaṃ nūna taṃ kammaṃ,_  
_janakāyo pamodito’._  

</Paragraph>

</Division>
<Division number="4" book="ap2">

<Paragraph number="160" type="verses" book="ap2">
_Yoniso paccavekkhitvā,_  
_duggaccañca daliddataṃ;_  
_Mānasaṃ sampahaṃsitvā,_  
_upavasiṃ uposathaṃ._  

</Paragraph>

</Division>
<Division number="5" book="ap2">

<Paragraph number="161" type="verses" book="ap2">
_Ahaṃ uposathaṃ katvā,_  
_sammāsambuddhasāsane;_  
_Tena kammena sukatena,_  
_tāvatiṃsamagacchahaṃ._  

</Paragraph>

</Division>
<Division number="6" book="ap2">

<Paragraph number="162" type="verses" book="ap2">
_Tattha me sukataṃ byamhaṃ,_  
_ubbhayojanamuggataṃ;_  
_Kūṭāgāravarūpetaṃ,_  
_mahāsanasubhūsitaṃ._  

</Paragraph>

</Division>
<Division number="7" book="ap2">

<Paragraph number="163" type="verses" book="ap2">
_Accharā satasahassā,_  
_upatiṭṭhanti maṃ sadā;_  
_Aññe deve atikkamma,_  
_atirocāmi sabbadā._  

</Paragraph>

</Division>
<Division number="8" book="ap2">

<Paragraph number="164" type="verses" book="ap2">
_Catusaṭṭhi devarājūnaṃ,_  
_mahesittamakārayiṃ;_  
_Tesaṭṭhi cakkavattīnaṃ,_  
_mahesittamakārayiṃ._  

</Paragraph>

</Division>
<Division number="9" book="ap2">

<Paragraph number="165" type="verses" book="ap2">
_Suvaṇṇavaṇṇā hutvāna,_  
_bhavesu saṃsarāmahaṃ;_  
_Sabbattha pavarā homi,_  
_uposathassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="10" book="ap2">

<Paragraph number="166" type="verses" book="ap2">
_Hatthiyānaṃ assayānaṃ,_  
_rathayānañca sīvikaṃ;_  
_Labhāmi sabbamevetaṃ,_  
_uposathassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="11" book="ap2">

<Paragraph number="167" type="verses" book="ap2">
_Soṇṇamayaṃ rūpimayaṃ,_  
_athopi phalikāmayaṃ;_  
_Lohitaṅgamayañceva,_  
_sabbaṃ paṭilabhāmahaṃ._  

</Paragraph>

</Division>
<Division number="12" book="ap2">

<Paragraph number="168" type="verses" book="ap2">
_Koseyyakambaliyāni,_  
_khomakappāsikāni ca;_  
_Mahagghāni ca vatthāni,_  
_sabbaṃ paṭilabhāmahaṃ._  

</Paragraph>

</Division>
<Division number="13" book="ap2">

<Paragraph number="169" type="verses" book="ap2">
_Annaṃ pānaṃ khādanīyaṃ,_  
_vatthasenāsanāni ca;_  
_Sabbametaṃ paṭilabhe,_  
_uposathassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="14" book="ap2">

<Paragraph number="170" type="verses" book="ap2">
_Varagandhañca mālañca,_  
_cuṇṇakañca vilepanaṃ;_  
_Sabbametaṃ paṭilabhe,_  
_uposathassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="15" book="ap2">

<Paragraph number="171" type="verses" book="ap2">
_Kūṭāgārañca pāsādaṃ,_  
_maṇḍapaṃ hammiyaṃ guhaṃ;_  
_Sabbametaṃ paṭilabhe,_  
_uposathassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="16" book="ap2">

<Paragraph number="172" type="verses" book="ap2">
_Jātiyā sattavassāhaṃ,_  
_pabbajiṃ anagāriyaṃ;_  
_Aḍḍhamāse asampatte,_  
_arahattamapāpuṇiṃ._  

</Paragraph>

</Division>
<Division number="17" book="ap2">

<Paragraph number="173" type="verses" book="ap2">
_Kilesā jhāpitā mayhaṃ,_  
_bhavā sabbe samūhatā;_  
_Sabbāsavā parikkhīṇā,_  
_natthi dāni punabbhavo._  

</Paragraph>

</Division>
<Division number="18" book="ap2">

<Paragraph number="174" type="verses" book="ap2">

_yaṃ kammamakariṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_uposathassidaṃ phalaṃ._  

</Paragraph>

</Division>
