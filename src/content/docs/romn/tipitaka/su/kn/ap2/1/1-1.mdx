<Paragraph number="2" book="ap2">
Atha therikāpadānāni suṇātha—
</Paragraph>

</Division>
<Division number="1" book="ap2" bookNo="20" page="1">

<Paragraph number="3" type="verses" book="ap2">
_“Bhagavati koṇāgamane,_  
_Saṃghārāmamhi navanivesanamhi;_  
_Sakhiyo tisso janiyo,_  
_Vihāradānaṃ adāsimha._  

</Paragraph>

</Division>
<Division number="2" book="ap2" bookNo="20" page="1">

<Paragraph number="4" type="verses" book="ap2">
_Dasakkhattuṃ satakkhattuṃ,_  
_Dasasatakkhattuṃ satānañca satakkhattuṃ;_  
_Devesu upapajjimha,_  
_Ko vādo mānuse bhave._  

</Paragraph>

</Division>
<Division number="3" book="ap2" bookNo="20" page="1">

<Paragraph number="5" type="verses" book="ap2">
_Deve mahiddhikā ahumha,_  
_mānusakamhi ko vādo;_  
_Sattaratanamahesī,_  
_itthiratanaṃ ahaṃ bhaviṃ._  

</Paragraph>

</Division>
<Division number="4" book="ap2" bookNo="20" page="1">

<Paragraph number="6" type="verses" book="ap2">
_Idha sañcitakusalā,_  
_susamiddhakulappajā;_  
_Dhanañjānī ca khemā ca,_  
_ahampi ca tayo janā._  

</Paragraph>

</Division>
<Division number="5" book="ap2" bookNo="20" page="1">

<Paragraph number="7" type="verses" book="ap2">
_Ārāmaṃ sukataṃ katvā,_  
_sabbāvayavamaṇḍitaṃ;_  
_Buddhappamukhasaṃghassa,_  
_niyyādetvā samoditā._  

</Paragraph>

</Division>
<Division number="6" book="ap2" bookNo="20" page="1">

<Paragraph number="8" type="verses" book="ap2">
_Yattha yatthūpapajjāmi,_  
_tassa kammassa vāhasā;_  
_Devesu aggataṃ pattā,_  
_manussesu tatheva ca._  

</Paragraph>

</Division>
<Division number="7" book="ap2" bookNo="20" page="1">

<Paragraph number="9" type="verses" book="ap2">
_Imasmiṃyeva kappamhi,_  
_brahmabandhu mahāyaso;_  
_Kassapo nāma gottena,_  
_uppajji vadataṃ varo._  

</Paragraph>

</Division>
<Division number="8" book="ap2" bookNo="20" page="2">

<Paragraph number="10" type="verses" book="ap2">
_Upaṭṭhāko mahesissa,_  
_tadā āsi narissaro;_  
_Kāsirājā kikī nāma,_  
_bārāṇasipuruttame._  

</Paragraph>

</Division>
<Division number="9" book="ap2" bookNo="20" page="2">

<Paragraph number="11" type="verses" book="ap2">
_Tassāsuṃ satta dhītaro,_  
_rājakaññā sukhedhitā;_  
_Buddhopaṭṭhānaniratā,_  
_brahmacariyaṃ cariṃsu tā._  

</Paragraph>

</Division>
<Division number="10" book="ap2" bookNo="20" page="2">

<Paragraph number="12" type="verses" book="ap2">
_Tāsaṃ sahāyikā hutvā,_  
_sīlesu susamāhitā;_  
_Datvā dānāni sakkaccaṃ,_  
_agāreva vataṃ cariṃ._  

</Paragraph>

</Division>
<Division number="11" book="ap2" bookNo="20" page="2">

<Paragraph number="13" type="verses" book="ap2">
_Tena kammena sukatena,_  
_cetanāpaṇidhīhi ca;_  
_Jahitvā mānusaṃ dehaṃ,_  
_tāvatiṃsūpagā ahaṃ._  

</Paragraph>

</Division>
<Division number="12" book="ap2" bookNo="20" page="2">

<Paragraph number="14" type="verses" book="ap2">
_Tato cutā yāmamagaṃ,_  
_tatohaṃ tusitaṃ gatā;_  
_Tato ca nimmānaratiṃ,_  
_vasavattipuraṃ tato._  

</Paragraph>

</Division>
<Division number="13" book="ap2" bookNo="20" page="2">

<Paragraph number="15" type="verses" book="ap2">
_Yattha yatthūpapajjāmi,_  
_puññakammasamohitā;_  
_Tattha tattheva rājūnaṃ,_  
_mahesittamakārayiṃ._  

</Paragraph>

</Division>
<Division number="14" book="ap2" bookNo="20" page="2">

<Paragraph number="16" type="verses" book="ap2">
_Tato cutā manussatte,_  
_rājūnaṃ cakkavattinaṃ;_  
_Maṇḍalīnañca rājūnaṃ,_  
_mahesittamakārayiṃ._  

</Paragraph>

</Division>
<Division number="15" book="ap2" bookNo="20" page="2">

<Paragraph number="17" type="verses" book="ap2">
_Sampattimanubhotvāna,_  
_devesu mānusesu ca;_  
_Sabbattha sukhitā hutvā,_  
_nekajātīsu saṃsariṃ._  

</Paragraph>

</Division>
<Division number="16" book="ap2" bookNo="20" page="2">

<Paragraph number="18" type="verses" book="ap2">
_So hetu ca so pabhavo,_  
_Tammūlaṃ sāsane khamaṃ;_  
_Paṭhamaṃ taṃ samodhānaṃ,_  
_Taṃ dhammaratāya nibbānaṃ._  

</Paragraph>

</Division>
<Division number="17" book="ap2" bookNo="20" page="2">

<Paragraph number="19" type="verses" book="ap2">
_Kilesā jhāpitā mayhaṃ,_  
_bhavā sabbe samūhatā;_  
_Nāgīva bandhanaṃ chetvā,_  
_viharāmi anāsavā._  

</Paragraph>

</Division>
<Division number="18" book="ap2" bookNo="20" page="3">

<Paragraph number="20" type="verses" book="ap2">
_Svāgataṃ vata me āsi,_  
_buddhaseṭṭhassa santike;_  
_Tisso vijjā anuppattā,_  
_kataṃ buddhassa sāsanaṃ._  

</Paragraph>

</Division>
<Division number="19" book="ap2" bookNo="20" page="3">

<Paragraph number="21" type="verses" book="ap2">
_Paṭisambhidā catasso,_  
_vimokkhāpi ca aṭṭhime;_  
_Chaḷabhiññā sacchikatā,_  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="22" book="ap2">
Itthaṃ sudaṃ sumedhā bhikkhunī imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="23" type="center" book="ap2">
Sumedhātheriyāpadānaṃ paṭhamaṃ.
</Paragraph>
