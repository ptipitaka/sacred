<Division number="91" book="ap2">

<Paragraph number="109" type="verses" book="ap2">
_“Nagare haṃsavatiyā,_  
_cārikī āsahaṃ tadā;_  
_Ārāmena ca ārāmaṃ,_  
_carāmi kusalatthikā._  

</Paragraph>

</Division>
<Division number="92" book="ap2">

<Paragraph number="110" type="verses" book="ap2">
_Kāḷapakkhamhi divase,_  
_addasaṃ bodhimuttamaṃ;_  
_Tattha cittaṃ pasādetvā,_  
_bodhimūle nisīdahaṃ._  

</Paragraph>

</Division>
<Division number="93" book="ap2">

<Paragraph number="111" type="verses" book="ap2">
_Garucittaṃ upaṭṭhetvā,_  
_sire katvāna añjaliṃ;_  
_Somanassaṃ pavedetvā,_  
_evaṃ cintesi tāvade._  

</Paragraph>

</Division>
<Division number="94" book="ap2">

<Paragraph number="112" type="verses" book="ap2">
_‘Yadi buddho amitaguṇo,_  
_asamappaṭipuggalo;_  
_Dassetu pāṭihīraṃ me,_  
_bodhi obhāsatu ayaṃ’._  

</Paragraph>

</Division>
<Division number="95" book="ap2">

<Paragraph number="113" type="verses" book="ap2">
_Saha āvajjite mayhaṃ,_  
_bodhi pajjali tāvade;_  
_Sabbasoṇṇamayā āsi,_  
_disā sabbā virocati._  

</Paragraph>

</Division>
<Division number="96" book="ap2">

<Paragraph number="114" type="verses" book="ap2">
_Sattarattindivaṃ tattha,_  
_bodhimūle nisīdahaṃ;_  
_Sattame divase patte,_  
_dīpapūjaṃ akāsahaṃ._  

</Paragraph>

</Division>
<Division number="97" book="ap2">

<Paragraph number="115" type="verses" book="ap2">
_Āsanaṃ parivāretvā,_  
_pañca dīpāni pajjaluṃ;_  
_Yāva udeti sūriyo,_  
_dīpā me pajjaluṃ tadā._  

</Paragraph>

</Division>
<Division number="98" book="ap2">

<Paragraph number="116" type="verses" book="ap2">
_Tena kammena sukatena,_  
_cetanāpaṇidhīhi ca;_  
_Jahitvā mānusaṃ dehaṃ,_  
_tāvatiṃsamagacchahaṃ._  

</Paragraph>

</Division>
<Division number="99" book="ap2">

<Paragraph number="117" type="verses" book="ap2">
_Tattha me sukataṃ byamhaṃ,_  
_pañcadīpāti vuccati;_  
_Saṭṭhiyojanamubbedhaṃ,_  
_tiṃsayojanavitthataṃ._  

</Paragraph>

</Division>
<Division number="100" book="ap2">

<Paragraph number="118" type="verses" book="ap2">
_Asaṅkhiyāni dīpāni,_  
_parivāre jalanti me;_  
_Yāvatā devabhavanaṃ,_  
_dīpālokena jotati._  

</Paragraph>

</Division>
<Division number="101" book="ap2">

<Paragraph number="119" type="verses" book="ap2">
_Parammukhā nisīditvā,_  
_yadi icchāmi passituṃ;_  
_Uddhaṃ adho ca tiriyaṃ,_  
_sabbaṃ passāmi cakkhunā._  

</Paragraph>

</Division>
<Division number="102" book="ap2">

<Paragraph number="120" type="verses" book="ap2">
_Yāvatā abhikaṅkhāmi,_  
_daṭṭhuṃ sugataduggate;_  
_Tattha āvaraṇaṃ natthi,_  
_rukkhesu pabbatesu vā._  

</Paragraph>

</Division>
<Division number="103" book="ap2">

<Paragraph number="121" type="verses" book="ap2">
_Asīti devarājūnaṃ,_  
_mahesittamakārayiṃ;_  
_Satānaṃ cakkavattīnaṃ,_  
_mahesittamakārayiṃ._  

</Paragraph>

</Division>
<Division number="104" book="ap2">

<Paragraph number="122" type="verses" book="ap2">
_Yaṃ yaṃ yonupapajjāmi,_  
_devattaṃ atha mānusaṃ;_  
_Dīpasatasahassāni,_  
_parivāre jalanti me._  

</Paragraph>

</Division>
<Division number="105" book="ap2">

<Paragraph number="123" type="verses" book="ap2">
_Devalokā cavitvāna,_  
_uppajjiṃ mātukucchiyaṃ;_  
_Mātukucchigatā santī,_  
_akkhi me na nimīlati._  

</Paragraph>

</Division>
<Division number="106" book="ap2">

<Paragraph number="124" type="verses" book="ap2">
_Dīpasatasahassāni,_  
_puññakammasamaṅgitā;_  
_Jalanti sūtikāgehe,_  
_pañcadīpānidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="107" book="ap2">

<Paragraph number="125" type="verses" book="ap2">
_Pacchime bhave sampatte,_  
_mānasaṃ vinivattayiṃ;_  
_Ajarāmataṃ sītibhāvaṃ,_  
_nibbānaṃ phassayiṃ ahaṃ._  

</Paragraph>

</Division>
<Division number="108" book="ap2">

<Paragraph number="126" type="verses" book="ap2">
_Jātiyā sattavassāhaṃ,_  
_arahattamapāpuṇiṃ;_  
_Upasampādayī buddho,_  
_guṇamaññāya gotamo._  

</Paragraph>

</Division>
<Division number="109" book="ap2">

<Paragraph number="127" type="verses" book="ap2">
_Maṇḍape rukkhamūle vā,_  
_pāsādesu guhāsu vā;_  
_Suññāgāre vasantiyā,_  
_pañcadīpā jalanti me._  

</Paragraph>

</Division>
<Division number="110" book="ap2">

<Paragraph number="128" type="verses" book="ap2">
_Dibbacakkhuvisuddhaṃ me,_  
_samādhikusalā ahaṃ;_  
_Abhiññāpāramippattā,_  
_pañcadīpānidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="111" book="ap2">

<Paragraph number="129" type="verses" book="ap2">
_Sabbavositavosānā,_  
_katakiccā anāsavā;_  
_Pañcadīpā mahāvīra,_  
_pāde vandāmi cakkhuma._  

</Paragraph>

</Division>
<Division number="112" book="ap2">

<Paragraph number="130" type="verses" book="ap2">
_Satasahassito kappe,_  
_yaṃ dīpamadadiṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_pañcadīpānidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="113" book="ap2">

<Paragraph number="131" type="verses" book="ap2">
_Kilesā jhāpitā mayhaṃ,_  
_ …pe… _  
_viharāmi anāsavā._  

</Paragraph>

</Division>
<Division number="114" book="ap2">

<Paragraph number="132" type="verses" book="ap2">
_Svāgataṃ vata me āsi,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ._  

</Paragraph>

</Division>
<Division number="115" book="ap2">

<Paragraph number="133" type="verses" book="ap2">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="134" book="ap2">
Itthaṃ sudaṃ pañcadīpikā bhikkhunī imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

Pañcadīpikātheriyāpadānaṃ navamaṃ.
