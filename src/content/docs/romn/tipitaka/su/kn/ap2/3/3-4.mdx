_sukhāya vadataṃ varo;_  
_Atthāya purisājañño,_  
_paṭipanno sadevake._  

</Paragraph>

</Division>
<Division number="133" book="ap2" bookNo="20" page="66">

_kittivaṇṇagato jino;_  
_Pūjito sabbalokassa,_  
_disāsabbāsu vissuto._  

</Paragraph>

</Division>
<Division number="134" book="ap2" bookNo="20" page="66">

_vītivattakathaṃkatho;_  
_Sampuṇṇamanasaṅkappo,_  
_patto sambodhimuttamaṃ._  

</Paragraph>

</Division>
<Division number="135" book="ap2" bookNo="20" page="66">

_uppādetā naruttamo;_  
_Anakkhātañca akkhāsi,_  
_asañjātañca sañjanī._  

</Paragraph>

</Division>
<Division number="136" book="ap2" bookNo="20" page="66">

_maggakkhāyī narāsabho;_  
_Maggassa kusalo satthā,_  
_sārathīnaṃ varuttamo._  

</Paragraph>

</Division>
<Division number="137" book="ap2" bookNo="20" page="66">

<Paragraph number="837" type="verses" book="ap2">

_dhammaṃ desesi nāyako;_  
_Nimugge kāmapaṅkamhi,_  
_samuddharati pāṇine._  

</Paragraph>

</Division>
<Division number="138" book="ap2" bookNo="20" page="66">

_jātā khattiyanandanā;_  
_Surūpā sadhanā cāpi,_  
_dayitā ca sirīmatī._  

</Paragraph>

</Division>
<Division number="139" book="ap2" bookNo="20" page="66">

<Paragraph number="839" type="verses" book="ap2">
_Ānandassa mahārañño,_  
_dhītā paramasobhaṇā;_  
_Vemātā bhaginī cāpi,_  
_padumuttaranāmino._  

</Paragraph>

</Division>
<Division number="140" book="ap2" bookNo="20" page="66">

<Paragraph number="840" type="verses" book="ap2">
_Rājakaññāhi sahitā,_  
_sabbābharaṇabhūsitā;_  
_Upāgamma mahāvīraṃ,_  
_assosiṃ dhammadesanaṃ._  

</Paragraph>

</Division>
<Division number="141" book="ap2" bookNo="20" page="66">

<Paragraph number="841" type="verses" book="ap2">
_Tadā hi so lokagaru,_  
_bhikkhuniṃ dibbacakkhukaṃ;_  
_Kittayaṃ parisāmajjhe,_  
_aggaṭṭhāne ṭhapesi taṃ._  

</Paragraph>

</Division>
<Division number="142" book="ap2" bookNo="20" page="66">

<Paragraph number="842" type="verses" book="ap2">

_dānaṃ datvāna satthuno;_  
_Pūjitvāna ca sambuddhaṃ,_  
_dibbacakkhuṃ apatthayiṃ._  

</Paragraph>

</Division>
<Division number="143" book="ap2" bookNo="20" page="66">

<Paragraph number="843" type="verses" book="ap2">
_Tato avoca maṃ satthā,_  
_‘nande lacchasi patthitaṃ;_  
_Padīpadhammadānānaṃ,_  
_phalametaṃ sunicchitaṃ._  

</Paragraph>

</Division>
<Division number="144" book="ap2" bookNo="20" page="66">

_okkākakulasambhavo;_  
_Gotamo nāma gottena,_  
_satthā loke bhavissati._  

</Paragraph>

</Division>
<Division number="145" book="ap2" bookNo="20" page="67">

_orasā dhammanimmitā;_  
_Sakulā nāma nāmena,_  
_hessati satthu sāvikā’._  

</Paragraph>

</Division>

</Paragraph>

</Paragraph>

_alabhiṃ telamattakaṃ._  

</Paragraph>

</Division>

_vippasannena cetasā._  

</Paragraph>

_dīpadānassidaṃ phalaṃ._  

</Paragraph>

_mudite rājapūjite._  

</Paragraph>

_sabbābharaṇabhūsitā;_  
_Purappavese sugataṃ,_  
_vātapāne ṭhitā ahaṃ._  

</Paragraph>

_Anubyañjanasampannaṃ,_  
_lakkhaṇehi vibhūsitaṃ._  

</Paragraph>

_arahattamapāpuṇiṃ._  

</Paragraph>

_satthusāsanakārikā._  

</Paragraph>

_visuddhāsiṃ sunimmalā._  

</Paragraph>

_Ohito garuko bhāro,_  
_bhavanetti samūhatā._  

</Paragraph>

_sabbasaṃyojanakkhayo._  

</Paragraph>
