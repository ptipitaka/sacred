</Paragraph>

_anāgatasamāgame;_  
_Sabbāva tuyhaṃ bhariyā,_  
_manāpā piyavādikā._  

</Paragraph>

</Division>
<Division number="415" book="ap2">

<Paragraph number="1126" type="verses" book="ap2">

_bhāvanā ca subhāvitā;_  
_Dīgharattañca no sabbaṃ,_  
_pariccattaṃ mahāmune._  

</Paragraph>

</Division>
<Division number="416" book="ap2">

<Paragraph number="1127" type="verses" book="ap2">

_dīpañca ratanāmayaṃ;_  
_Yaṃ kiñci patthitaṃ sabbaṃ,_  
_pariccattaṃ mahāmuni._  

</Paragraph>

</Division>
<Division number="417" book="ap2">

<Paragraph number="1128" type="verses" book="ap2">

_paribhogañca mānusaṃ;_  
_Dīgharattañhi no sabbaṃ,_  
_pariccattaṃ mahāmuni._  

</Paragraph>

</Division>
<Division number="418" book="ap2">

<Paragraph number="1129" type="verses" book="ap2">

_Issaramanubhotvāna,_  
_saṃsaritvā bhavābhave._  

</Paragraph>

</Division>

</Paragraph>

</Division>

</Paragraph>

</Division>

</Paragraph>
