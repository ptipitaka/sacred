<Paragraph number="225" type="verses" book="ap2">
_“Nagare haṃsavatiyā,_  
_cārikī āsahaṃ tadā;_  
_Ārāmena ca ārāmaṃ,_  
_carāmi kusalatthikā._  

</Paragraph>

</Division>
<Division number="62" book="ap2" bookNo="20" page="21">

<Paragraph number="226" type="verses" book="ap2">
_Kāḷapakkhamhi divase,_  
_addasaṃ bodhimuttamaṃ;_  
_Tattha cittaṃ pasādetvā,_  
_bodhimūle nisīdahaṃ._  

</Paragraph>

</Division>
<Division number="63" book="ap2" bookNo="20" page="21">

<Paragraph number="227" type="verses" book="ap2">
_Garucittaṃ upaṭṭhetvā,_  
_sire katvāna añjaliṃ;_  
_Somanassaṃ pavedetvā,_  
_evaṃ cintesi tāvade._  

</Paragraph>

</Division>
<Division number="64" book="ap2" bookNo="20" page="21">

<Paragraph number="228" type="verses" book="ap2">
_‘Yadi buddho amitaguṇo,_  
_asamappaṭipuggalo;_  
_Dassetu pāṭihīraṃ me,_  
_bodhi obhāsatu ayaṃ’._  

</Paragraph>

</Division>
<Division number="65" book="ap2" bookNo="20" page="21">

<Paragraph number="229" type="verses" book="ap2">
_Saha āvajjite mayhaṃ,_  
_bodhi pajjali tāvade;_  
_Sabbasoṇṇamayā āsi,_  
_disā sabbā virocati._  

</Paragraph>

</Division>
<Division number="66" book="ap2" bookNo="20" page="21">

<Paragraph number="230" type="verses" book="ap2">
_Sattarattindivaṃ tattha,_  
_bodhimūle nisīdahaṃ;_  
_Sattame divase patte,_  
_dīpapūjaṃ akāsahaṃ._  

</Paragraph>

</Division>
<Division number="67" book="ap2" bookNo="20" page="21">

<Paragraph number="231" type="verses" book="ap2">
_Āsanaṃ parivāretvā,_  
_pañca dīpāni pajjaluṃ;_  
_Yāva udeti sūriyo,_  
_dīpā me pajjaluṃ tadā._  

</Paragraph>

</Division>
<Division number="68" book="ap2" bookNo="20" page="21">

<Paragraph number="232" type="verses" book="ap2">
_Tena kammena sukatena,_  
_cetanāpaṇidhīhi ca;_  
_Jahitvā mānusaṃ dehaṃ,_  
_tāvatiṃsamagacchahaṃ._  

</Paragraph>

</Division>
<Division number="69" book="ap2" bookNo="20" page="22">

<Paragraph number="233" type="verses" book="ap2">
_Tattha me sukataṃ byamhaṃ,_  
_pañcadīpāti vuccati;_  
_Saṭṭhiyojanamubbedhaṃ,_  
_tiṃsayojanavitthataṃ._  

</Paragraph>

</Division>
<Division number="70" book="ap2" bookNo="20" page="22">

<Paragraph number="234" type="verses" book="ap2">
_Asaṅkhiyāni dīpāni,_  
_parivāre jaliṃsu me;_  
_Yāvatā devabhavanaṃ,_  
_dīpālokena jotati._  

</Paragraph>

</Division>
<Division number="71" book="ap2" bookNo="20" page="22">

<Paragraph number="235" type="verses" book="ap2">
_Parammukhā nisīditvā,_  
_yadi icchāmi passituṃ;_  
_Uddhaṃ adho ca tiriyaṃ,_  
_sabbaṃ passāmi cakkhunā._  

</Paragraph>

</Division>
<Division number="72" book="ap2" bookNo="20" page="22">

<Paragraph number="236" type="verses" book="ap2">
_Yāvatā abhikaṅkhāmi,_  
_daṭṭhuṃ sugataduggate;_  
_Tattha āvaraṇaṃ natthi,_  
_rukkhesu pabbatesu vā._  

</Paragraph>

</Division>
<Division number="73" book="ap2" bookNo="20" page="22">

<Paragraph number="237" type="verses" book="ap2">
_Asīti devarājūnaṃ,_  
_mahesittamakārayiṃ;_  
_Satānaṃ cakkavattīnaṃ,_  
_mahesittamakārayiṃ._  

</Paragraph>

</Division>
<Division number="74" book="ap2" bookNo="20" page="22">

<Paragraph number="238" type="verses" book="ap2">
_Yaṃ yaṃ yonupapajjāmi,_  
_devattaṃ atha mānusaṃ;_  
_Dīpasatasahassāni,_  
_parivāre jalanti me._  

</Paragraph>

</Division>
<Division number="75" book="ap2" bookNo="20" page="22">

<Paragraph number="239" type="verses" book="ap2">
_Devalokā cavitvāna,_  
_uppajjiṃ mātukucchiyaṃ;_  
_Mātukucchigatā santī,_  
_akkhi me na nimīlati._  

</Paragraph>

</Division>
<Division number="76" book="ap2" bookNo="20" page="22">

<Paragraph number="240" type="verses" book="ap2">
_Dīpasatasahassāni,_  
_puññakammasamaṅgitā;_  
_Jalanti sūtikāgehe,_  
_pañcadīpānidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="77" book="ap2" bookNo="20" page="22">

<Paragraph number="241" type="verses" book="ap2">
_Pacchime bhave sampatte,_  
_mānasaṃ vinivattayiṃ;_  
_Ajarāmataṃ sītibhāvaṃ,_  
_nibbānaṃ phassayiṃ ahaṃ._  

</Paragraph>

</Division>
<Division number="78" book="ap2" bookNo="20" page="22">

<Paragraph number="242" type="verses" book="ap2">
_Jātiyā sattavassāhaṃ,_  
_arahattamapāpuṇiṃ;_  
_Upasampādayī buddho,_  
_guṇamaññāya gotamo._  

</Paragraph>

</Division>
<Division number="79" book="ap2" bookNo="20" page="22">

<Paragraph number="243" type="verses" book="ap2">
_Maṇḍape rukkhamūle vā,_  
_suññāgāre vasantiyā;_  
_Sadā pajjalate dīpaṃ,_  
_pañcadīpānidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="80" book="ap2" bookNo="20" page="22">

<Paragraph number="244" type="verses" book="ap2">
_Dibbacakkhuvisuddhaṃ me,_  
_samādhikusalā ahaṃ;_  
_Abhiññāpāramippattā,_  
_pañcadīpānidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="81" book="ap2" bookNo="20" page="23">

<Paragraph number="245" type="verses" book="ap2">
_Sabbavositavosānā,_  
_katakiccā anāsavā;_  
_Pañcadīpā mahāvīra,_  
_pāde vandāmi cakkhuma._  

</Paragraph>

</Division>
<Division number="82" book="ap2" bookNo="20" page="23">

<Paragraph number="246" type="verses" book="ap2">
_Satasahassito kappe,_  
_yaṃ dīpamadadiṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_pañcadīpānidaṃ phalaṃ._  

</Paragraph>

</Division>
