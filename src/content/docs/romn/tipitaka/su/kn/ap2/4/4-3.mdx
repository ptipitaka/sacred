<Paragraph number="1265" type="verses" book="ap2">
_“Nagare aruṇavatiyā,_  
_aruṇo nāma khattiyo;_  
_Tassa rañño ahuṃ bhariyā,_  
_ekajjhaṃ cārayāmahaṃ._  

</Paragraph>

</Division>
<Division number="58" book="ap2" bookNo="20" page="37">

<Paragraph number="1266" type="verses" book="ap2">
_Rahogatā nisīditvā,_  
_evaṃ cintesahaṃ tadā;_  
_‘Kusalaṃ me kataṃ natthi,_  
_ādāya gamiyaṃ mama._  

</Paragraph>

</Division>
<Division number="59" book="ap2" bookNo="20" page="37">

<Paragraph number="1267" type="verses" book="ap2">
_Mahābhitāpaṃ kaṭukaṃ,_  
_ghorarūpaṃ sudāruṇaṃ;_  
_Nirayaṃ nūna gacchāmi,_  
_ettha me natthi saṃsayo’._  

</Paragraph>

</Division>
<Division number="60" book="ap2" bookNo="20" page="37">

<Paragraph number="1268" type="verses" book="ap2">
_Evāhaṃ cintayitvāna,_  
_pahaṃsetvāna mānasaṃ;_  
_Rājānaṃ upagantvāna,_  
_idaṃ vacanamabraviṃ._  

</Paragraph>

</Division>
<Division number="61" book="ap2" bookNo="20" page="37">

<Paragraph number="1269" type="verses" book="ap2">
_‘Itthī nāma mayaṃ deva,_  
_purisānittarā ahu;_  
_Ekaṃ me samaṇaṃ dehi,_  
_bhojayissāmi khattiya’._  

</Paragraph>

</Division>
<Division number="62" book="ap2" bookNo="20" page="37">

<Paragraph number="1270" type="verses" book="ap2">
_Adāsi me tadā rājā,_  
_samaṇaṃ bhāvitindriyaṃ;_  
_Tassa pattaṃ gahetvāna,_  
_paramannena pūrayiṃ._  

</Paragraph>

</Division>
<Division number="63" book="ap2" bookNo="20" page="37">

<Paragraph number="1271" type="verses" book="ap2">
_Pūretvā paramaṃ annaṃ,_  
_saha sugandhalepanaṃ;_  
_Mahācelena chāditvā,_  
_adāsiṃ tuṭṭhamānasā._  

</Paragraph>

</Division>
<Division number="64" book="ap2" bookNo="20" page="37">

<Paragraph number="1272" type="verses" book="ap2">
_Tena kammena sukatena,_  
_cetanāpaṇidhīhi ca;_  
_Jahitvā mānusaṃ dehaṃ,_  
_tāvatiṃsamagacchahaṃ._  

</Paragraph>

</Division>
<Division number="65" book="ap2" bookNo="20" page="37">

<Paragraph number="1273" type="verses" book="ap2">
_Sahassadevarājūnaṃ,_  
_mahesittamakārayiṃ;_  
_Sahassacakkavattīnaṃ,_  
_mahesittamakārayiṃ._  

</Paragraph>

</Division>
<Division number="66" book="ap2" bookNo="20" page="37">

<Paragraph number="1274" type="verses" book="ap2">
_Padesarajjaṃ vipulaṃ,_  
_gaṇanāto asaṅkhiyaṃ;_  
_Nānāvidhaṃ bahuṃ aññaṃ,_  
_tassa kammaphalaṃ tato._  

</Paragraph>

</Division>
<Division number="67" book="ap2" bookNo="20" page="38">

<Paragraph number="1275" type="verses" book="ap2">
_Uppalasseva me vaṇṇo,_  
_abhirūpā sudassanā;_  
_Itthisabbaṅgasampannā,_  
_abhijātā jutindharā._  

</Paragraph>

</Division>
<Division number="68" book="ap2" bookNo="20" page="45">

<Paragraph number="1276" type="verses" book="ap2">
_Pacchime bhave sampatte,_  
_ajāyiṃ sākiye kule;_  
_Nārīsahassapāmokkhā,_  
_suddhodanasutassahaṃ._  

</Paragraph>

</Division>
<Division number="69" book="ap2" bookNo="20" page="45">

<Paragraph number="1277" type="verses" book="ap2">
_Nibbinditvā agārehaṃ,_  
_pabbajiṃ anagāriyaṃ;_  
_Sattamīrattisampattā,_  
_catusaccamapāpuṇiṃ._  

</Paragraph>

</Division>
<Division number="70" book="ap2" bookNo="20" page="45">

<Paragraph number="1278" type="verses" book="ap2">
_Cīvaraṃ piṇḍapātañca,_  
_paccayaṃ sayanāsanaṃ;_  
_Parimetuṃ na sakkomi,_  
_piṇḍapātassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="71" book="ap2" bookNo="20" page="45">

<Paragraph number="1279" type="verses" book="ap2">
_Yaṃ mayhaṃ pūritaṃ kammaṃ,_  
_Kusalaṃ sarase muni;_  
_Tuyhatthāya mahāvīra,_  
_Paricattaṃ bahuṃ mayā._  

</Paragraph>

</Division>
<Division number="72" book="ap2" bookNo="20" page="45">

<Paragraph number="1280" type="verses" book="ap2">
_Ekattiṃse ito kappe,_  
_yaṃ dānamadadiṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_piṇḍapātassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="73" book="ap2" bookNo="20" page="52">

<Paragraph number="1281" type="verses" book="ap2">
_Dve gatiyo pajānāmi,_  
_devattaṃ atha mānusaṃ;_  
_Aññaṃ gatiṃ na jānāmi,_  
_piṇḍapātassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="74" book="ap2" bookNo="20" page="52">

<Paragraph number="1282" type="verses" book="ap2">
_Ucce kule pajānāmi,_  
_mahāsāle mahaddhane;_  
_Aññe kule na jānāmi,_  
_piṇḍapātassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="75" book="ap2" bookNo="20" page="52">

<Paragraph number="1283" type="verses" book="ap2">
_Bhavābhave saṃsaritvā,_  
_sukkamūlena coditā;_  
_Amanāpaṃ na passāmi,_  
_somanassakataṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="76" book="ap2" bookNo="20" page="52">

<Paragraph number="1284" type="verses" book="ap2">
_Iddhīsu ca vasī homi,_  
_dibbāya sotadhātuyā;_  
_Cetopariyañāṇassa,_  
_vasī homi mahāmune._  

</Paragraph>

</Division>
<Division number="77" book="ap2" bookNo="20" page="52">

<Paragraph number="1285" type="verses" book="ap2">
_Pubbenivāsaṃ jānāmi,_  
_dibbacakkhu visodhitaṃ;_  
_Sabbāsavā parikkhīṇā,_  
_natthi dāni punabbhavo._  

</Paragraph>

</Division>
<Division number="78" book="ap2" bookNo="20" page="52">

<Paragraph number="1286" type="verses" book="ap2">
_Atthadhammaniruttīsu,_  
_paṭibhāne tatheva ca;_  
_Ñāṇaṃ mama mahāvīra,_  
_uppannaṃ tava santike._  

</Paragraph>

</Division>
<Division number="79" book="ap2" bookNo="20" page="52">

<Paragraph number="1287" type="verses" book="ap2">
_Kilesā jhāpitā mayhaṃ,_  
_ …pe… _  
_viharāmi anāsavā._  

</Paragraph>

</Division>
<Division number="80" book="ap2" bookNo="20" page="52">

<Paragraph number="1288" type="verses" book="ap2">
_Svāgataṃ vata me āsi,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ._  

</Paragraph>

</Division>
<Division number="81" book="ap2" bookNo="20" page="61">

<Paragraph number="1289" type="verses" book="ap2">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="1290" book="ap2">
Itthaṃ sudaṃ uppaladāyikā bhikkhunī bhagavato sammukhā imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="1291" type="center" book="ap2">
Uppaladāyikātheriyāpadānaṃ tatiyaṃ.
</Paragraph>
