<Paragraph number="1442" type="verses" book="ap2">
_“Imamhi bhaddake kappe,_  
_brahmabandhu mahāyaso;_  
_Kassapo nāma gottena,_  
_uppajji vadataṃ varo._  

</Paragraph>

</Division>
<Division number="221" book="ap2" bookNo="20" page="216">

<Paragraph number="1443" type="verses" book="ap2">
_Sāvatthiyaṃ pure vare,_  
_upāsakakule ahaṃ;_  
_Pasūtā taṃ jinavaraṃ,_  
_disvā sutvā ca desanaṃ._  

</Paragraph>

</Division>
<Division number="222" book="ap2" bookNo="20" page="216">

<Paragraph number="1444" type="verses" book="ap2">
_Taṃ vīraṃ saraṇaṃ gantvā,_  
_sīlāni ca samādiyiṃ;_  
_Kadāci so mahāvīro,_  
_mahājanasamāgame._  

</Paragraph>

</Division>
<Division number="223" book="ap2" bookNo="20" page="216">

<Paragraph number="1445" type="verses" book="ap2">
_Attano abhisambodhiṃ,_  
_pakāsesi narāsabho;_  
_Ananussutadhammesu,_  
_pubbe dukkhādikesu ca._  

</Paragraph>

</Division>
<Division number="224" book="ap2" bookNo="20" page="216">

<Paragraph number="1446" type="verses" book="ap2">
_Cakkhu ñāṇañca paññā ca,_  
_vijjāloko ca āsi me;_  
_Taṃ sutvā uggahetvāna,_  
_paripucchiñca bhikkhavo._  

</Paragraph>

</Division>
<Division number="225" book="ap2" bookNo="20" page="216">

<Paragraph number="1447" type="verses" book="ap2">
_Tena kammena sukatena,_  
_cetanāpaṇidhīhi ca;_  
_Jahitvā mānusaṃ dehaṃ,_  
_tāvatiṃsamagacchahaṃ._  

</Paragraph>

</Division>
<Division number="226" book="ap2" bookNo="20" page="216">

<Paragraph number="1448" type="verses" book="ap2">
_Pacchime ca bhave dāni,_  
_Jātā seṭṭhimahākule;_  
_Upecca buddhaṃ saddhammaṃ,_  
_Sutvā saccūpasaṃhitaṃ._  

</Paragraph>

</Division>
<Division number="227" book="ap2" bookNo="20" page="216">

<Paragraph number="1449" type="verses" book="ap2">
_Pabbajitvācireneva,_  
_saccatthāni vicintayaṃ;_  
_Khepetvā āsave sabbe,_  
_arahattamapāpuṇiṃ._  

</Paragraph>

</Division>
<Division number="228" book="ap2" bookNo="20" page="216">

<Paragraph number="1450" type="verses" book="ap2">
_Iddhīsu ca vasī homi,_  
_dibbāya sotadhātuyā;_  
_Cetopariyañāṇassa,_  
_vasī homi mahāmune._  

</Paragraph>

</Division>
<Division number="229" book="ap2" bookNo="20" page="217">

<Paragraph number="1451" type="verses" book="ap2">
_Pubbenivāsaṃ jānāmi,_  
_dibbacakkhu visodhitaṃ;_  
_Sabbāsavā parikkhīṇā,_  
_natthi dāni punabbhavo._  

</Paragraph>

</Division>
<Division number="230" book="ap2" bookNo="20" page="217">

<Paragraph number="1452" type="verses" book="ap2">
_Atthadhammaniruttīsu,_  
_paṭibhāne tatheva ca;_  
_Ñāṇaṃ me vimalaṃ suddhaṃ,_  
_buddhaseṭṭhassa vāhasā._  

</Paragraph>

</Division>
<Division number="231" book="ap2" bookNo="20" page="217">

<Paragraph number="1453" type="verses" book="ap2">
_Kilesā jhāpitā mayhaṃ,_  
_bhavā sabbe samūhatā;_  
_Nāgīva bandhanaṃ chetvā,_  
_viharāmi anāsavā._  

</Paragraph>

</Division>
<Division number="232" book="ap2" bookNo="20" page="217">

<Paragraph number="1454" type="verses" book="ap2">
_Svāgataṃ vata me āsi,_  
_mama buddhassa santike;_  
_Tisso vijjā anuppattā,_  
_kataṃ buddhassa sāsanaṃ._  

</Paragraph>

</Division>
<Division number="233" book="ap2" bookNo="20" page="217">

<Paragraph number="1455" type="verses" book="ap2">
_Paṭisambhidā catasso,_  
_Vimokkhāpi ca aṭṭhime;_  
_Chaḷabhiññā sacchikatā,_  
_Kataṃ buddhassa sāsanaṃ”. (1359)_  

</Paragraph>

<Paragraph number="1456" book="ap2">
Itthaṃ sudaṃ pesalā bhikkhunī imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="1457" type="center" book="ap2">
Pesalātheriyāpadānaṃ dasamaṃ.

  
<Emphasis>Khattiyāvaggo catuttho.</Emphasis>



</Paragraph>

<Paragraph number="1458" type="center" book="ap2">
Tassuddānaṃ


</Paragraph>

<Paragraph number="1459" type="verses" book="ap2">
_Khattiyā brāhmaṇī ceva,_  
_tathā uppaladāyikā;_  
_Siṅgālamātā sukkā ca,_  
_abhirūpā aḍḍhakāsikā._  

</Paragraph>

<Paragraph number="1460" type="verses" book="ap2">
_Puṇṇā ca ambapālī ca,_  
_pesalāti ca tā dasa;_  
_Gāthāyo dvisatānettha,_  
_dvicattālīsa cuttari._  


</Paragraph>

<Paragraph number="1461" type="center" book="ap2">
Atha vagguddānaṃ


</Paragraph>

<Paragraph number="1462" type="verses" book="ap2">
_Sumedhā ekūposathā,_  
_kuṇḍalakesī khattiyā;_  
_Sahassaṃ tisatā gāthā,_  
_sattatālīsa piṇḍitā._  

</Paragraph>

<Paragraph number="1463" type="verses" book="ap2">
_Saha uddānagāthāhi,_  
_gaṇitāyo vibhāvibhi;_  
_Sahassaṃ tisataṃ gāthā,_  
_sattapaññāsameva cāti._  


</Paragraph>

<Paragraph number="1464" type="center" book="ap2">
Therikāpadānaṃ samattaṃ.



</Paragraph>

Apadānapāḷi samattā.
