<Division number="55" book="ap2">

<Paragraph number="750" type="verses" book="ap2">
_“Padumuttaro nāma jino,_  
_sabbadhammāna pāragū;_  
_Ito satasahassamhi,_  
_kappe uppajji nāyako._  

</Paragraph>

</Division>
<Division number="56" book="ap2">

<Paragraph number="751" type="verses" book="ap2">
_Tadāhaṃ haṃsavatiyaṃ,_  
_jātā aññatare kule;_  
_Upetvā taṃ naravaraṃ,_  
_saraṇaṃ samupāgamiṃ._  

</Paragraph>

</Division>
<Division number="57" book="ap2">

<Paragraph number="752" type="verses" book="ap2">
_Dhammañca tassa assosiṃ,_  
_catusaccūpasañhitaṃ;_  
_Madhuraṃ paramassādaṃ,_  
_vaṭṭasantisukhāvahaṃ._  

</Paragraph>

</Division>
<Division number="58" book="ap2">

<Paragraph number="753" type="verses" book="ap2">
_Tadā ca bhikkhuniṃ vīro,_  
_lūkhacīvaradhāriniṃ;_  
_Ṭhapento etadaggamhi,_  
_vaṇṇayī purisuttamo._  

</Paragraph>

</Division>
<Division number="59" book="ap2">

<Paragraph number="754" type="verses" book="ap2">
_Janetvānappakaṃ pītiṃ,_  
_sutvā bhikkhuniyā guṇe;_  
_Kāraṃ katvāna buddhassa,_  
_yathāsatti yathābalaṃ._  

</Paragraph>

</Division>
<Division number="60" book="ap2">

<Paragraph number="755" type="verses" book="ap2">
_Nipacca munivaraṃ taṃ,_  
_taṃ ṭhānamabhipatthayiṃ;_  
_Tadānumodi sambuddho,_  
_ṭhānalābhāya nāyako._  

</Paragraph>

</Division>
<Division number="61" book="ap2">

<Paragraph number="756" type="verses" book="ap2">
_‘Satasahassito kappe,_  
_okkākakulasambhavo;_  
_Gotamo nāma gottena,_  
_satthā loke bhavissati._  

</Paragraph>

</Division>
<Division number="62" book="ap2">

<Paragraph number="757" type="verses" book="ap2">
_Tassa dhammesu dāyādā,_  
_orasā dhammanimmitā;_  
_Kisāgotamī nāmena,_  
_hessati satthu sāvikā’._  

</Paragraph>

</Division>
<Division number="63" book="ap2">

<Paragraph number="758" type="verses" book="ap2">
_Taṃ sutvā muditā hutvā,_  
_yāvajīvaṃ tadā jinaṃ;_  
_Mettacittā paricariṃ,_  
_paccayehi vināyakaṃ._  

</Paragraph>

</Division>
<Division number="64" book="ap2">

<Paragraph number="759" type="verses" book="ap2">
_Tena kammena sukatena,_  
_cetanāpaṇidhīhi ca;_  
_Jahitvā mānusaṃ dehaṃ,_  
_tāvatiṃsamagacchahaṃ._  

</Paragraph>

</Division>
<Division number="65" book="ap2">

<Paragraph number="760" type="verses" book="ap2">
_Imamhi bhaddake kappe,_  
_brahmabandhu mahāyaso;_  
_Kassapo nāma gottena,_  
_uppajji vadataṃ varo._  

</Paragraph>

</Division>
<Division number="66" book="ap2">

<Paragraph number="761" type="verses" book="ap2">
_Upaṭṭhāko mahesissa,_  
_tadā āsi narissaro;_  
_Kāsirājā kikī nāma,_  
_bārāṇasipuruttame._  

</Paragraph>

</Division>
<Division number="67" book="ap2">

<Paragraph number="762" type="verses" book="ap2">
_Pañcamī tassa dhītāsiṃ,_  
_dhammā nāmena vissutā;_  
_Dhammaṃ sutvā jinaggassa,_  
_pabbajjaṃ samarocayiṃ._  

</Paragraph>

</Division>
<Division number="68" book="ap2">

<Paragraph number="763" type="verses" book="ap2">
_Anujāni na no tāto,_  
_agāreva tadā mayaṃ;_  
_Vīsavassasahassāni,_  
_vicarimha atanditā._  

</Paragraph>

</Division>
<Division number="69" book="ap2">

<Paragraph number="764" type="verses" book="ap2">
_Komāribrahmacariyaṃ,_  
_rājakaññā sukhedhitā;_  
_Buddhopaṭṭhānaniratā,_  
_muditā satta dhītaro._  

</Paragraph>

</Division>
<Division number="70" book="ap2">

<Paragraph number="765" type="verses" book="ap2">
_Samaṇī samaṇaguttā ca,_  
_bhikkhunī bhikkhudāyikā;_  
_Dhammā ceva sudhammā ca,_  
_sattamī saṃghadāyikā._  

</Paragraph>

</Division>
<Division number="71" book="ap2">

<Paragraph number="766" type="verses" book="ap2">
_Khemā uppalavaṇṇā ca,_  
_paṭācārā ca kuṇḍalā;_  
_Ahañca dhammadinnā ca,_  
_visākhā hoti sattamī._  

</Paragraph>

</Division>
<Division number="72" book="ap2">

<Paragraph number="767" type="verses" book="ap2">
_Tehi kammehi sukatehi,_  
_cetanāpaṇidhīhi ca;_  
_Jahitvā mānusaṃ dehaṃ,_  
_tāvatiṃsamagacchahaṃ._  

</Paragraph>

</Division>
<Division number="73" book="ap2">

<Paragraph number="768" type="verses" book="ap2">
_Pacchime ca bhave dāni,_  
_jātā seṭṭhikule ahaṃ;_  
_Duggate adhane naṭṭhe,_  
_gatā ca sadhanaṃ kulaṃ._  

</Paragraph>

</Division>
<Division number="74" book="ap2">

<Paragraph number="769" type="verses" book="ap2">
_Patiṃ ṭhapetvā sesā me,_  
_dessanti adhanā iti;_  
_Yadā ca pasūtā āsiṃ,_  
_sabbesaṃ dayitā tadā._  

</Paragraph>

</Division>
<Division number="75" book="ap2">

<Paragraph number="770" type="verses" book="ap2">
_Yadā so taruṇo bhaddo,_  
_komalako sukhedhito;_  
_Sapāṇamiva kanto me,_  
_tadā yamavasaṃ gato._  

</Paragraph>

</Division>
<Division number="76" book="ap2">

<Paragraph number="771" type="verses" book="ap2">
_Sokaṭṭādīnavadanā,_  
_assunettā rudammukhā;_  
_Mataṃ kuṇapamādāya,_  
_vilapantī gamāmahaṃ._  

</Paragraph>

</Division>
<Division number="77" book="ap2">

<Paragraph number="772" type="verses" book="ap2">
_Tadā ekena sandiṭṭhā,_  
_upetvābhisakkuttamaṃ;_  
_Avocaṃ ‘dehi bhesajjaṃ,_  
_puttasañjīvananti bho’._  

</Paragraph>

</Division>
<Division number="78" book="ap2">

<Paragraph number="773" type="verses" book="ap2">
_‘Na vijjante matā yasmiṃ,_  
_gehe siddhatthakaṃ tato;_  
_Āharā’ti jino āha,_  
_vinayopāyakovido._  

</Paragraph>

</Division>
<Division number="79" book="ap2">

<Paragraph number="774" type="verses" book="ap2">
_Tadā gamitvā sāvatthiṃ,_  
_na labhiṃ tādisaṃ gharaṃ;_  
_Kuto siddhatthakaṃ tasmā,_  
_tato laddhā satiṃ ahaṃ._  

</Paragraph>

</Division>
<Division number="80" book="ap2">

<Paragraph number="775" type="verses" book="ap2">
_Kuṇapaṃ chaḍḍayitvāna,_  
_upesiṃ lokanāyakaṃ;_  
_Dūratova mamaṃ disvā,_  
_avoca madhurassaro._  

</Paragraph>

</Division>
<Division number="81" book="ap2">

<Paragraph number="776" type="verses" book="ap2">
_‘Yo ca vassasataṃ jīve,_  
_apassaṃ udayabbayaṃ;_  
_Ekāhaṃ jīvitaṃ seyyo,_  
_passato udayabbayaṃ._  

</Paragraph>

</Division>
<Division number="82" book="ap2">

<Paragraph number="777" type="verses" book="ap2">
_Na gāmadhammo nigamassa dhammo,_  
_Na cāpiyaṃ ekakulassa dhammo;_  
_Sabbassa lokassa sadevakassa,_  
_Eseva dhammo yadidaṃ aniccatā’._  

</Paragraph>

</Division>
<Division number="83" book="ap2">

<Paragraph number="778" type="verses" book="ap2">
_Sāhaṃ sutvānimā gāthā,_  
_dhammacakkhuṃ visodhayiṃ;_  
_Tato viññātasaddhammā,_  
_pabbajiṃ anagāriyaṃ._  

</Paragraph>

</Division>
<Division number="84" book="ap2">

<Paragraph number="779" type="verses" book="ap2">
_Tathā pabbajitā santī,_  
_yuñjantī jinasāsane;_  
_Na cireneva kālena,_  
_arahattamapāpuṇiṃ._  

</Paragraph>

</Division>
<Division number="85" book="ap2">

<Paragraph number="780" type="verses" book="ap2">
_Iddhīsu ca vasī homi,_  
_dibbāya sotadhātuyā;_  
_Paracittāni jānāmi,_  
_satthusāsanakārikā._  

</Paragraph>

</Division>
<Division number="86" book="ap2">

<Paragraph number="781" type="verses" book="ap2">
_Pubbenivāsaṃ jānāmi,_  
_dibbacakkhu visodhitaṃ;_  
_Khepetvā āsave sabbe,_  
_visuddhāsiṃ sunimmalā._  

</Paragraph>

</Division>
<Division number="87" book="ap2">

<Paragraph number="782" type="verses" book="ap2">
_Pariciṇṇo mayā satthā,_  
_kataṃ buddhassa sāsanaṃ;_  
_Ohito garuko bhāro,_  
_bhavanetti samūhatā._  

</Paragraph>

</Division>
<Division number="88" book="ap2">

<Paragraph number="783" type="verses" book="ap2">
_Yassatthāya pabbajitā,_  
_agārasmānagāriyaṃ;_  
_So me attho anuppatto,_  
_sabbasaṃyojanakkhayo._  

</Paragraph>

</Division>
<Division number="89" book="ap2">

<Paragraph number="784" type="verses" book="ap2">
_Atthadhammaniruttīsu,_  
_paṭibhāne tatheva ca;_  
_Ñāṇaṃ me vimalaṃ suddhaṃ,_  
_buddhaseṭṭhassa vāhasā._  

</Paragraph>

</Division>
<Division number="90" book="ap2">

<Paragraph number="785" type="verses" book="ap2">
_Saṅkārakūṭā āhitvā,_  
_susānā rathiyāpi ca;_  
_Tato saṅghāṭikaṃ katvā,_  
_lūkhaṃ dhāremi cīvaraṃ._  

</Paragraph>

</Division>
<Division number="91" book="ap2">

<Paragraph number="786" type="verses" book="ap2">
_Jino tasmiṃ guṇe tuṭṭho,_  
_lūkhacīvaradhāraṇe;_  
_Ṭhapesi etadaggamhi,_  
_parisāsu vināyako._  

</Paragraph>

</Division>
<Division number="92" book="ap2">

<Paragraph number="787" type="verses" book="ap2">
_Kilesā jhāpitā mayhaṃ,_  
_ …pe… _  
_viharāmi anāsavā._  

</Paragraph>

</Division>
<Division number="93" book="ap2">

<Paragraph number="788" type="verses" book="ap2">
_Svāgataṃ vata me āsi,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ._  

</Paragraph>

</Division>
<Division number="94" book="ap2">

<Paragraph number="789" type="verses" book="ap2">

_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="790" book="ap2">
Itthaṃ sudaṃ kisāgotamī bhikkhunī imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="791" type="center" book="ap2">
Kisāgotamītheriyāpadānaṃ dutiyaṃ.
