<Division number="82" book="ap2">

<Paragraph number="1292" type="verses" book="ap2">
_“Padumuttaro nāma jino,_  
_sabbadhammāna pāragū;_  
_Ito satasahassamhi,_  
_kappe uppajji nāyako._  

</Paragraph>

</Division>
<Division number="83" book="ap2">

<Paragraph number="1293" type="verses" book="ap2">
_Tadāhaṃ haṃsavatiyaṃ,_  
_jātāmaccakule ahuṃ;_  
_Nānāratanapajjote,_  
_iddhe phīte mahaddhane._  

</Paragraph>

</Division>
<Division number="84" book="ap2">

<Paragraph number="1294" type="verses" book="ap2">
_Pitunā saha gantvāna,_  
_mahājanapurakkhatā;_  
_Dhammaṃ buddhassa sutvāna,_  
_pabbajiṃ anagāriyaṃ._  

</Paragraph>

</Division>
<Division number="85" book="ap2">

<Paragraph number="1295" type="verses" book="ap2">
_Pabbajitvāna kāyena,_  
_pāpakammaṃ vivajjayiṃ;_  
_Vacīduccaritaṃ hitvā,_  
_ājīvaṃ parisodhayiṃ._  

</Paragraph>

</Division>
<Division number="86" book="ap2">

<Paragraph number="1296" type="verses" book="ap2">
_Buddhe pasannā dhamme ca,_  
_saṃghe ca tibbagāravā;_  
_Saddhammassavane yuttā,_  
_buddhadassanalālasā._  

</Paragraph>

</Division>
<Division number="87" book="ap2">

<Paragraph number="1297" type="verses" book="ap2">
_Aggaṃ saddhādhimuttānaṃ,_  
_assosiṃ bhikkhuniṃ tadā;_  
_Taṃ ṭhānaṃ patthayitvāna,_  
_tisso sikkhā apūrayiṃ._  

</Paragraph>

</Division>
<Division number="88" book="ap2">

<Paragraph number="1298" type="verses" book="ap2">
_Tato maṃ sugato āha,_  
_karuṇānugatāsayo;_  
_‘Yassa saddhā tathāgate,_  
_acalā suppatiṭṭhitā;_  
_Sīlañca yassa kalyāṇaṃ,_  
_ariyakantaṃ pasaṃsitaṃ._  

</Paragraph>

</Division>
<Division number="89" book="ap2">

<Paragraph number="1299" type="verses" book="ap2">
_Saṃghe pasādo yassatthi,_  
_ujubhūtañca dassanaṃ;_  
_Adaliddoti taṃ āhu,_  
_amoghaṃ tassa jīvitaṃ._  

</Paragraph>

</Division>
<Division number="90" book="ap2">

<Paragraph number="1300" type="verses" book="ap2">
_Tasmā saddhañca sīlañca,_  
_pasādaṃ dhammadassanaṃ;_  
_Anuyuñjetha medhāvī,_  
_saraṃ buddhāna sāsanaṃ’._  

</Paragraph>

</Division>
<Division number="91" book="ap2">

<Paragraph number="1301" type="verses" book="ap2">
_Taṃ sutvāhaṃ pamuditā,_  
_apucchiṃ paṇidhiṃ mama;_  
_Tadā anomo amito,_  
_byākarittha vināyako;_  
_‘Buddhe pasannā kalyāṇī,_  
_lacchase taṃ supatthitaṃ._  

</Paragraph>

</Division>
<Division number="92" book="ap2">

<Paragraph number="1302" type="verses" book="ap2">
_Satasahassito kappe,_  
_okkākakulasambhavo;_  
_Gotamo nāma gottena,_  
_satthā loke bhavissati._  

</Paragraph>

</Division>
<Division number="93" book="ap2">

<Paragraph number="1303" type="verses" book="ap2">
_Tassa dhammesu dāyādā,_  
_orasā dhammanimmitā;_  
_Siṅgālakassa mātāti,_  
_hessati satthu sāvikā’._  

</Paragraph>

</Division>
<Division number="94" book="ap2">

<Paragraph number="1304" type="verses" book="ap2">
_Taṃ sutvā muditā hutvā,_  
_yāvajīvaṃ tadā jinaṃ;_  
_Mettacittā paricariṃ,_  
_paṭipattīhi nāyakaṃ._  

</Paragraph>

</Division>
<Division number="95" book="ap2">

<Paragraph number="1305" type="verses" book="ap2">
_Tena kammena sukatena,_  
_cetanāpaṇidhīhi ca;_  
_Jahitvā mānusaṃ dehaṃ,_  
_tāvatiṃsamagacchahaṃ._  

</Paragraph>

</Division>
<Division number="96" book="ap2">

<Paragraph number="1306" type="verses" book="ap2">
_Pacchime ca bhave dāni,_  
_giribbajapuruttame;_  
_Jātā seṭṭhikule phīte,_  
_mahāratanasañcaye._  

</Paragraph>

</Division>
<Division number="97" book="ap2">

<Paragraph number="1307" type="verses" book="ap2">
_Putto siṅgālako nāma,_  
_mamāsi vipathe rato;_  
_Diṭṭhigahanapakkhando,_  
_disāpūjanatapparo._  

</Paragraph>

</Division>
<Division number="98" book="ap2">

<Paragraph number="1308" type="verses" book="ap2">
_Nānādisā namassantaṃ,_  
_piṇḍāya nagaraṃ vajaṃ;_  
_Taṃ disvā ovadī buddho,_  
_magge ṭhatvā vināyako._  

</Paragraph>

</Division>
<Division number="99" book="ap2">

<Paragraph number="1309" type="verses" book="ap2">
_Tassa desayato dhammaṃ,_  
_panādo vimhayo ahu;_  
_Dvekoṭinaranārīnaṃ,_  
_dhammābhisamayo ahu._  

</Paragraph>

</Division>
<Division number="100" book="ap2">

<Paragraph number="1310" type="verses" book="ap2">
_Tadāhaṃ parisaṃ gantvā,_  
_sutvā sugatabhāsitaṃ;_  
_Sotāpattiphalaṃ pattā,_  
_pabbajiṃ anagāriyaṃ._  

</Paragraph>

</Division>
<Division number="101" book="ap2">

<Paragraph number="1311" type="verses" book="ap2">
_Na cireneva kālena,_  
_buddhadassanalālasā;_  
_Anussatiṃ taṃ bhāvetvā,_  
_arahattamapāpuṇiṃ._  

</Paragraph>

</Division>
<Division number="102" book="ap2">

<Paragraph number="1312" type="verses" book="ap2">
_Dassanatthāya buddhassa,_  
_sabbadā ca vajāmahaṃ;_  
_Atittāyeva passāmi,_  
_rūpaṃ nayananandanaṃ._  

</Paragraph>

</Division>
<Division number="103" book="ap2">

<Paragraph number="1313" type="verses" book="ap2">
_Sabbapāramisambhūtaṃ,_  
_lakkhīnilayanaṃ varaṃ;_  
_Rūpaṃ sabbasubhākiṇṇaṃ,_  
_atittā viharāmahaṃ._  

</Paragraph>

</Division>
<Division number="104" book="ap2">

<Paragraph number="1314" type="verses" book="ap2">
_Jino tasmiṃ guṇe tuṭṭho,_  
_etadagge ṭhapesi maṃ;_  
_‘Siṅgālakassa yā mātā,_  
_aggā saddhādhimuttikā’._  

</Paragraph>

</Division>
<Division number="105" book="ap2">

<Paragraph number="1315" type="verses" book="ap2">
_Iddhīsu ca vasī homi,_  
_dibbāya sotadhātuyā;_  
_Cetopariyañāṇassa,_  
_vasī homi mahāmuni._  

</Paragraph>

</Division>
<Division number="106" book="ap2">

<Paragraph number="1316" type="verses" book="ap2">
_Pubbenivāsaṃ jānāmi,_  
_dibbacakkhu visodhitaṃ;_  
_Sabbāsavā parikkhīṇā,_  
_natthi dāni punabbhavo._  

</Paragraph>

</Division>
<Division number="107" book="ap2">

<Paragraph number="1317" type="verses" book="ap2">
_Atthadhammaniruttīsu,_  
_paṭibhāne tatheva ca;_  
_Ñāṇaṃ mama mahāvīra,_  
_uppannaṃ tava santike._  

</Paragraph>

</Division>
<Division number="108" book="ap2">

<Paragraph number="1318" type="verses" book="ap2">

_ …pe… _  
_viharāmi anāsavā._  

</Paragraph>

</Division>

_ …pe… _  
_kataṃ buddhassa sāsanaṃ._  

</Paragraph>

</Division>
<Division number="110" book="ap2">

_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>
