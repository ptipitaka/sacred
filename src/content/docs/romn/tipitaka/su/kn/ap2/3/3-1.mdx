<Paragraph number="694" type="verses" book="ap2">
_“Padumuttaro nāma jino,_  
_sabbadhammāna pāragū;_  
_Ito satasahassamhi,_  
_kappe uppajji nāyako._  

</Paragraph>

</Division>
<Division number="2" book="ap2" bookNo="20" page="24">

<Paragraph number="695" type="verses" book="ap2">
_Tadāhaṃ haṃsavatiyaṃ,_  
_jātā seṭṭhikule ahuṃ;_  
_Nānāratanapajjote,_  
_mahāsukhasamappitā._  

</Paragraph>

</Division>
<Division number="3" book="ap2" bookNo="20" page="24">

<Paragraph number="696" type="verses" book="ap2">
_Upetvā taṃ mahāvīraṃ,_  
_assosiṃ dhammadesanaṃ;_  
_Tato jātappasādāhaṃ,_  
_upesiṃ saraṇaṃ jinaṃ._  

</Paragraph>

</Division>
<Division number="4" book="ap2" bookNo="20" page="24">

<Paragraph number="697" type="verses" book="ap2">
_Tadā mahākāruṇiko,_  
_padumuttaranāmako;_  
_Khippābhiññānamagganti,_  
_ṭhapesi bhikkhuniṃ subhaṃ._  

</Paragraph>

</Division>
<Division number="5" book="ap2" bookNo="20" page="24">

<Paragraph number="698" type="verses" book="ap2">
_Taṃ sutvā muditā hutvā,_  
_dānaṃ datvā mahesino;_  
_Nipacca sirasā pāde,_  
_taṃ ṭhānamabhipatthayiṃ._  

</Paragraph>

</Division>
<Division number="6" book="ap2" bookNo="20" page="24">

<Paragraph number="699" type="verses" book="ap2">
_Anumodi mahāvīro,_  
_‘bhadde yaṃ tebhipatthitaṃ;_  
_Samijjhissati taṃ sabbaṃ,_  
_sukhinī hohi nibbutā._  

</Paragraph>

</Division>
<Division number="7" book="ap2" bookNo="20" page="24">

<Paragraph number="700" type="verses" book="ap2">
_Satasahassito kappe,_  
_okkākakulasambhavo;_  
_Gotamo nāma gottena,_  
_satthā loke bhavissati._  

</Paragraph>

</Division>
<Division number="8" book="ap2" bookNo="20" page="24">

<Paragraph number="701" type="verses" book="ap2">
_Tassa dhammesu dāyādā,_  
_orasā dhammanimmitā;_  
_Bhaddākuṇḍalakesāti,_  
_hessati satthu sāvikā’._  

</Paragraph>

</Division>
<Division number="9" book="ap2" bookNo="20" page="25">

<Paragraph number="702" type="verses" book="ap2">
_Tena kammena sukatena,_  
_cetanāpaṇidhīhi ca;_  
_Jahitvā mānusaṃ dehaṃ,_  
_tāvatiṃsamagacchahaṃ._  

</Paragraph>

</Division>
<Division number="10" book="ap2" bookNo="20" page="25">

<Paragraph number="703" type="verses" book="ap2">
_Tato cutā yāmamagaṃ,_  
_tatohaṃ tusitaṃ gatā;_  
_Tato ca nimmānaratiṃ,_  
_vasavattipuraṃ tato._  

</Paragraph>

</Division>
<Division number="11" book="ap2" bookNo="20" page="25">

<Paragraph number="704" type="verses" book="ap2">
_Yattha yatthūpapajjāmi,_  
_tassa kammassa vāhasā;_  
_Tattha tattheva rājūnaṃ,_  
_mahesittamakārayiṃ._  

</Paragraph>

</Division>
<Division number="12" book="ap2" bookNo="20" page="25">

<Paragraph number="705" type="verses" book="ap2">
_Tato cutā manussesu,_  
_rājūnaṃ cakkavattinaṃ;_  
_Maṇḍalīnañca rājūnaṃ,_  
_mahesittamakārayiṃ._  

</Paragraph>

</Division>
<Division number="13" book="ap2" bookNo="20" page="25">

<Paragraph number="706" type="verses" book="ap2">
_Sampattiṃ anubhotvāna,_  
_devesu mānusesu ca;_  
_Sabbattha sukhitā hutvā,_  
_nekakappesu saṃsariṃ._  

</Paragraph>

</Division>
<Division number="14" book="ap2" bookNo="20" page="25">

<Paragraph number="707" type="verses" book="ap2">
_Imamhi bhaddake kappe,_  
_brahmabandhu mahāyaso;_  
_Kassapo nāma gottena,_  
_uppajji vadataṃ varo._  

</Paragraph>

</Division>
<Division number="15" book="ap2" bookNo="20" page="25">

<Paragraph number="708" type="verses" book="ap2">
_Upaṭṭhāko mahesissa,_  
_tadā āsi narissaro;_  
_Kāsirājā kikī nāma,_  
_bārāṇasipuruttame._  

</Paragraph>

</Division>
<Division number="16" book="ap2" bookNo="20" page="25">

<Paragraph number="709" type="verses" book="ap2">
_Tassa dhītā catutthāsiṃ,_  
_bhikkhudāyīti vissutā;_  
_Dhammaṃ sutvā jinaggassa,_  
_pabbajjaṃ samarocayiṃ._  

</Paragraph>

</Division>
<Division number="17" book="ap2" bookNo="20" page="25">

<Paragraph number="710" type="verses" book="ap2">
_Anujāni na no tāto,_  
_agāreva tadā mayaṃ;_  
_Vīsavassasahassāni,_  
_vicarimha atanditā._  

</Paragraph>

</Division>
<Division number="18" book="ap2" bookNo="20" page="25">

<Paragraph number="711" type="verses" book="ap2">
_Komāribrahmacariyaṃ,_  
_rājakaññā sukhedhitā;_  
_Buddhopaṭṭhānaniratā,_  
_muditā satta dhītaro._  

</Paragraph>

</Division>
<Division number="19" book="ap2" bookNo="20" page="25">

<Paragraph number="712" type="verses" book="ap2">
_Samaṇī samaṇaguttā ca,_  
_bhikkhunī bhikkhudāyikā;_  
_Dhammā ceva sudhammā ca,_  
_sattamī saṃghadāyikā._  

</Paragraph>

</Division>
<Division number="20" book="ap2" bookNo="20" page="25">

<Paragraph number="713" type="verses" book="ap2">
_Khemā uppalavaṇṇā ca,_  
_paṭācārā ahaṃ tadā;_  
_Kisāgotamī dhammadinnā,_  
_visākhā hoti sattamī._  

</Paragraph>

</Division>
<Division number="21" book="ap2" bookNo="20" page="26">

<Paragraph number="714" type="verses" book="ap2">
_Tehi kammehi sukatehi,_  
_cetanāpaṇidhīhi ca;_  
_Jahitvā mānusaṃ dehaṃ,_  
_tāvatiṃsamagacchahaṃ._  

</Paragraph>

</Division>
<Division number="22" book="ap2" bookNo="20" page="26">

<Paragraph number="715" type="verses" book="ap2">
_Pacchime ca bhave dāni,_  
_giribbajapuruttame;_  
_Jātā seṭṭhikule phīte,_  
_yadāhaṃ yobbane ṭhitā._  

</Paragraph>

</Division>
<Division number="23" book="ap2" bookNo="20" page="26">

<Paragraph number="716" type="verses" book="ap2">
_Coraṃ vadhatthaṃ nīyantaṃ,_  
_disvā rattā tahiṃ ahaṃ;_  
_Pitā me taṃ sahassena,_  
_mocayitvā vadhā tato._  

</Paragraph>

</Division>
<Division number="24" book="ap2" bookNo="20" page="26">

<Paragraph number="717" type="verses" book="ap2">
_Adāsi tassa maṃ tāto,_  
_viditvāna manaṃ mama;_  
_Tassāhamāsiṃ visaṭṭhā,_  
_atīva dayitā hitā._  

</Paragraph>

</Division>
<Division number="25" book="ap2" bookNo="20" page="26">

<Paragraph number="718" type="verses" book="ap2">
_So me bhūsanalobhena,_  
_balimajjhāsayo diso;_  
_Corappapātaṃ netvāna,_  
_pabbataṃ cetayī vadhaṃ._  

</Paragraph>

</Division>
<Division number="26" book="ap2" bookNo="20" page="26">

<Paragraph number="719" type="verses" book="ap2">
_Tadāhaṃ paṇamitvāna,_  
_sattukaṃ sukatañjalī;_  
_Rakkhantī attano pāṇaṃ,_  
_idaṃ vacanamabraviṃ._  

</Paragraph>

</Division>
<Division number="27" book="ap2" bookNo="20" page="26">

<Paragraph number="720" type="verses" book="ap2">
_‘Idaṃ suvaṇṇakeyūraṃ,_  
_muttā veḷuriyā bahū;_  
_Sabbaṃ harassu bhaddante,_  
_mañca dāsīti sāvaya’._  

</Paragraph>

</Division>
<Division number="28" book="ap2" bookNo="20" page="26">

<Paragraph number="721" type="verses" book="ap2">
_‘Oropayassu kalyāṇī,_  
_mā bāḷhaṃ paridevasi;_  
_Na cāhaṃ abhijānāmi,_  
_ahantvā dhanamābhataṃ’._  

</Paragraph>

</Division>
<Division number="29" book="ap2" bookNo="20" page="26">

<Paragraph number="722" type="verses" book="ap2">
_‘Yato sarāmi attānaṃ,_  
_yato pattosmi viññutaṃ;_  
_Na cāhaṃ abhijānāmi,_  
_aññaṃ piyataraṃ tayā._  

</Paragraph>

</Division>
<Division number="30" book="ap2" bookNo="20" page="26">

<Paragraph number="723" type="verses" book="ap2">
_Ehi taṃ upagūhissaṃ,_  
_katvāna taṃ padakkhiṇaṃ;_  
_Na ca dāni puno atthi,_  
_mama tuyhañca saṅgamo’._  

</Paragraph>

</Division>
<Division number="31" book="ap2" bookNo="20" page="27">

<Paragraph number="724" type="verses" book="ap2">
_‘Na hi sabbesu ṭhānesu,_  
_puriso hoti paṇḍito;_  
_Itthīpi paṇḍitā hoti,_  
_tattha tattha vicakkhaṇā._  

</Paragraph>

</Division>
<Division number="32" book="ap2" bookNo="20" page="27">

<Paragraph number="725" type="verses" book="ap2">
_Na hi sabbesu ṭhānesu,_  
_puriso hoti paṇḍito;_  
_Itthīpi paṇḍitā hoti,_  
_lahuṃ atthavicintikā’._  

</Paragraph>

</Division>
<Division number="33" book="ap2" bookNo="20" page="27">

<Paragraph number="726" type="verses" book="ap2">
_Lahuñca vata khippañca,_  
_nikaṭṭhe samacetayiṃ;_  
_Migaṃ uṇṇā yathā evaṃ,_  
_tadāhaṃ sattukaṃ vadhiṃ._  

</Paragraph>

</Division>
<Division number="34" book="ap2" bookNo="20" page="27">

<Paragraph number="727" type="verses" book="ap2">
_Yo ca uppatitaṃ atthaṃ,_  
_na khippamanubujjhati;_  
_So haññate mandamati,_  
_corova girigabbhare._  

</Paragraph>

</Division>
<Division number="35" book="ap2" bookNo="20" page="27">

<Paragraph number="728" type="verses" book="ap2">
_Yo ca uppatitaṃ atthaṃ,_  
_khippameva nibodhati;_  
_Muccate sattusambādhā,_  
_tadāhaṃ sattukā yathā._  

</Paragraph>

</Division>
<Division number="36" book="ap2" bookNo="20" page="27">

<Paragraph number="729" type="verses" book="ap2">
_Tadāhaṃ pātayitvāna,_  
_giriduggamhi sattukaṃ;_  
_Santikaṃ setavatthānaṃ,_  
_upetvā pabbajiṃ ahaṃ._  

</Paragraph>

</Division>
<Division number="37" book="ap2" bookNo="20" page="27">

<Paragraph number="730" type="verses" book="ap2">
_Saṇḍāsena ca kese me,_  
_luñcitvā sabbaso tadā;_  
_Pabbajitvāna samayaṃ,_  
_ācikkhiṃsu nirantaraṃ._  

</Paragraph>

</Division>
<Division number="38" book="ap2" bookNo="20" page="27">

<Paragraph number="731" type="verses" book="ap2">
_Tato taṃ uggahetvāhaṃ,_  
_nisīditvāna ekikā;_  
_Samayaṃ taṃ vicintesiṃ,_  
_suvāno mānusaṃ karaṃ._  

</Paragraph>

</Division>
<Division number="39" book="ap2" bookNo="20" page="27">

<Paragraph number="732" type="verses" book="ap2">
_Chinnaṃ gayha samīpe me,_  
_pātayitvā apakkami;_  
_Disvā nimittamalabhiṃ,_  
_hatthaṃ taṃ puḷavākulaṃ._  

</Paragraph>

</Division>
<Division number="40" book="ap2" bookNo="20" page="27">

<Paragraph number="733" type="verses" book="ap2">
_Tato uṭṭhāya saṃviggā,_  
_apucchiṃ sahadhammike;_  
_Te avocuṃ ‘vijānanti,_  
_taṃ atthaṃ sakyabhikkhavo’._  

</Paragraph>

</Division>
<Division number="41" book="ap2" bookNo="20" page="27">

<Paragraph number="734" type="verses" book="ap2">
_Sāhaṃ tamatthaṃ pucchissaṃ,_  
_upetvā buddhasāvake;_  
_Te mamādāya gacchiṃsu,_  
_buddhaseṭṭhassa santikaṃ._  

</Paragraph>

</Division>
<Division number="42" book="ap2" bookNo="20" page="28">

<Paragraph number="735" type="verses" book="ap2">
_So me dhammamadesesi,_  
_khandhāyatanadhātuyo;_  
_Asubhāniccadukkhāti,_  
_anattāti ca nāyako._  

</Paragraph>

</Division>
<Division number="43" book="ap2" bookNo="20" page="28">

<Paragraph number="736" type="verses" book="ap2">
_Tassa dhammaṃ suṇitvāhaṃ,_  
_dhammacakkhuṃ visodhayiṃ;_  
_Tato viññātasaddhammā,_  
_pabbajjaṃ upasampadaṃ._  

</Paragraph>

</Division>
<Division number="44" book="ap2" bookNo="20" page="28">

<Paragraph number="737" type="verses" book="ap2">
_Āyācito tadā āha,_  
_‘ehi bhadde’ti nāyako;_  
_Tadāhaṃ upasampannā,_  
_parittaṃ toyamaddasaṃ._  

</Paragraph>

</Division>
<Division number="45" book="ap2" bookNo="20" page="28">

<Paragraph number="738" type="verses" book="ap2">
_Pādapakkhālanenāhaṃ,_  
_ñatvā saudayabbayaṃ;_  
_Tathā sabbepi saṅkhāre,_  
_īdisaṃ cintayiṃ tadā._  

</Paragraph>

</Division>
<Division number="46" book="ap2" bookNo="20" page="28">

<Paragraph number="739" type="verses" book="ap2">
_Tato cittaṃ vimucci me,_  
_anupādāya sabbaso;_  
_Khippābhiññānamaggaṃ me,_  
_tadā paññāpayī jino._  

</Paragraph>

</Division>
<Division number="47" book="ap2" bookNo="20" page="28">

<Paragraph number="740" type="verses" book="ap2">
_Iddhīsu ca vasī homi,_  
_dibbāya sotadhātuyā;_  
_Paracittāni jānāmi,_  
_satthusāsanakārikā._  

</Paragraph>

</Division>
<Division number="48" book="ap2" bookNo="20" page="28">

<Paragraph number="741" type="verses" book="ap2">
_Pubbenivāsaṃ jānāmi,_  
_dibbacakkhu visodhitaṃ;_  
_Khepetvā āsave sabbe,_  
_visuddhāsiṃ sunimmalā._  

</Paragraph>

</Division>
<Division number="49" book="ap2" bookNo="20" page="28">

<Paragraph number="742" type="verses" book="ap2">
_Pariciṇṇo mayā satthā,_  
_kataṃ buddhassa sāsanaṃ;_  
_Ohito garuko bhāro,_  
_bhavanetti samūhatā._  

</Paragraph>

</Division>
<Division number="50" book="ap2" bookNo="20" page="28">

<Paragraph number="743" type="verses" book="ap2">
_Yassatthāya pabbajitā,_  
_agārasmānagāriyaṃ;_  
_So me attho anuppatto,_  
_sabbasaṃyojanakkhayo._  

</Paragraph>

</Division>
<Division number="51" book="ap2" bookNo="20" page="28">

<Paragraph number="744" type="verses" book="ap2">
_Atthadhammaniruttīsu,_  
_paṭibhāne tatheva ca;_  
_Ñāṇaṃ me vimalaṃ suddhaṃ,_  
_buddhaseṭṭhassa sāsane._  

</Paragraph>

</Division>
<Division number="52" book="ap2" bookNo="20" page="28">

<Paragraph number="745" type="verses" book="ap2">
_Kilesā jhāpitā mayhaṃ,_  
_bhavā sabbe samūhatā;_  
_Nāgīva bandhanaṃ chetvā,_  
_viharāmi anāsavā._  

</Paragraph>

</Division>
<Division number="53" book="ap2" bookNo="20" page="29">

<Paragraph number="746" type="verses" book="ap2">
_Svāgataṃ vata me āsi,_  
_mama buddhassa santike;_  
_Tisso vijjā anuppattā,_  
_kataṃ buddhassa sāsanaṃ._  

</Paragraph>

</Division>
<Division number="54" book="ap2" bookNo="20" page="29">

<Paragraph number="747" type="verses" book="ap2">

_vimokkhāpi ca aṭṭhime;_  
_Chaḷabhiññā sacchikatā,_  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="748" book="ap2">
Itthaṃ sudaṃ bhaddākuṇḍalakesā bhikkhunī imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>
