<Division number="37" book="ap2">

<Paragraph number="199" type="verses" book="ap2">
_“Nagare haṃsavatiyā,_  
_ahosiṃ bālikā tadā;_  
_Mātā ca me pitā ceva,_  
_kammantaṃ agamaṃsu te._  

</Paragraph>

</Division>
<Division number="38" book="ap2">

<Paragraph number="200" type="verses" book="ap2">
_Majjhanhikamhi sūriye,_  
_addasaṃ samaṇaṃ ahaṃ;_  
_Vīthiyā anugacchantaṃ,_  
_āsanaṃ paññapesahaṃ._  

</Paragraph>

</Division>
<Division number="39" book="ap2">

<Paragraph number="201" type="verses" book="ap2">
_Gonakāvikatikāhi,_  
_paññapetvā mamāsanaṃ;_  
_Pasannacittā sumanā,_  
_idaṃ vacanamabraviṃ._  

</Paragraph>

</Division>
<Division number="40" book="ap2">

<Paragraph number="202" type="verses" book="ap2">
_‘Santattā kuthitā bhūmi,_  
_sūro majjhanhike ṭhito;_  
_Mālutā ca na vāyanti,_  
_kālo cevettha mehiti._  

</Paragraph>

</Division>
<Division number="41" book="ap2">

<Paragraph number="203" type="verses" book="ap2">
_Paññattamāsanamidaṃ,_  
_tavatthāya mahāmuni;_  
_Anukampaṃ upādāya,_  
_nisīda mama āsane’._  

</Paragraph>

</Division>
<Division number="42" book="ap2">

<Paragraph number="204" type="verses" book="ap2">
_Nisīdi tattha samaṇo,_  
_sudanto suddhamānaso;_  
_Tassa pattaṃ gahetvāna,_  
_yathārandhaṃ adāsahaṃ._  

</Paragraph>

</Division>
<Division number="43" book="ap2">

<Paragraph number="205" type="verses" book="ap2">
_Tena kammena sukatena,_  
_cetanāpaṇidhīhi ca;_  
_Jahitvā mānusaṃ dehaṃ,_  
_tāvatiṃsamagacchahaṃ._  

</Paragraph>

</Division>
<Division number="44" book="ap2">

<Paragraph number="206" type="verses" book="ap2">
_Tattha me sukataṃ byamhaṃ,_  
_āsanena sunimmitaṃ;_  
_Saṭṭhiyojanamubbedhaṃ,_  
_tiṃsayojanavitthataṃ._  

</Paragraph>

</Division>
<Division number="45" book="ap2">

<Paragraph number="207" type="verses" book="ap2">
_Soṇṇamayā maṇimayā,_  
_athopi phalikāmayā;_  
_Lohitaṅgamayā ceva,_  
_pallaṅkā vividhā mama._  

</Paragraph>

</Division>
<Division number="46" book="ap2">

<Paragraph number="208" type="verses" book="ap2">
_Tūlikāvikatikāhi,_  
_kaṭṭissacittakāhi ca;_  
_Uddaekantalomī ca,_  
_pallaṅkā me susaṇṭhitā._  

</Paragraph>

</Division>
<Division number="47" book="ap2">

<Paragraph number="209" type="verses" book="ap2">
_Yadā icchāmi gamanaṃ,_  
_hāsakhiḍḍasamappitā;_  
_Saha pallaṅkaseṭṭhena,_  
_gacchāmi mama patthitaṃ._  

</Paragraph>

</Division>
<Division number="48" book="ap2">

<Paragraph number="210" type="verses" book="ap2">
_Asīti devarājūnaṃ,_  
_mahesittamakārayiṃ;_  
_Sattati cakkavattīnaṃ,_  
_mahesittamakārayiṃ._  

</Paragraph>

</Division>
<Division number="49" book="ap2">

<Paragraph number="211" type="verses" book="ap2">
_Bhavābhave saṃsarantī,_  
_mahābhogaṃ labhāmahaṃ;_  
_Bhoge me ūnatā natthi,_  
_ekāsanassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="50" book="ap2">

<Paragraph number="212" type="verses" book="ap2">
_Duve bhave saṃsarāmi,_  
_devatte atha mānuse;_  
_Aññe bhave na jānāmi,_  
_ekāsanassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="51" book="ap2">

<Paragraph number="213" type="verses" book="ap2">
_Duve kule pajāyāmi,_  
_khattiye cāpi brāhmaṇe;_  
_Uccākulīnā sabbattha,_  
_ekāsanassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="52" book="ap2">

<Paragraph number="214" type="verses" book="ap2">
_Domanassaṃ na jānāmi,_  
_cittasantāpanaṃ mama;_  
_Vevaṇṇiyaṃ na jānāmi,_  
_ekāsanassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="53" book="ap2">

<Paragraph number="215" type="verses" book="ap2">
_Dhātiyo maṃ upaṭṭhanti,_  
_khujjā celāpikā bahū;_  
_Aṅkena aṅkaṃ gacchāmi,_  
_ekāsanassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="54" book="ap2">

<Paragraph number="216" type="verses" book="ap2">
_Aññā nhāpenti bhojenti,_  
_aññā ramenti maṃ sadā;_  
_Aññā gandhaṃ vilimpanti,_  
_ekāsanassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="55" book="ap2">

<Paragraph number="217" type="verses" book="ap2">
_Maṇḍape rukkhamūle vā,_  
_suññāgāre vasantiyā;_  
_Mama saṅkappamaññāya,_  
_pallaṅko upatiṭṭhati._  

</Paragraph>

</Division>
<Division number="56" book="ap2">

<Paragraph number="218" type="verses" book="ap2">
_Ayaṃ pacchimako mayhaṃ,_  
_carimo vattate bhavo;_  
_Ajjāpi rajjaṃ chaḍḍetvā,_  
_pabbajiṃ anagāriyaṃ._  

</Paragraph>

</Division>
<Division number="57" book="ap2">

<Paragraph number="219" type="verses" book="ap2">
_Satasahassito kappe,_  
_yaṃ dānamadadiṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_ekāsanassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="58" book="ap2">

<Paragraph number="220" type="verses" book="ap2">
_Kilesā jhāpitā mayhaṃ,_  
_ …pe… _  
_viharāmi anāsavā._  

</Paragraph>

</Division>
<Division number="59" book="ap2">

<Paragraph number="221" type="verses" book="ap2">
_Svāgataṃ vata me āsi,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ._  

</Paragraph>

</Division>
<Division number="60" book="ap2">

<Paragraph number="222" type="verses" book="ap2">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="223" book="ap2">
Itthaṃ sudaṃ ekāsanadāyikā bhikkhunī imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="224" type="center" book="ap2">
Ekāsanadāyikātheriyāpadānaṃ catutthaṃ.
