<Paragraph number="792" type="verses" book="ap2">
_“Padumuttaro nāma jino,_  
_sabbadhammāna pāragū;_  
_Ito satasahassamhi,_  
_kappe uppajji nāyako._  

</Paragraph>

</Division>
<Division number="96" book="ap2" bookNo="20" page="82">

<Paragraph number="793" type="verses" book="ap2">
_Tadāhaṃ haṃsavatiyaṃ,_  
_kule aññatare ahuṃ;_  
_Parakammakārī āsiṃ,_  
_nipakā sīlasaṃvutā._  

</Paragraph>

</Division>
<Division number="97" book="ap2" bookNo="20" page="82">

<Paragraph number="794" type="verses" book="ap2">
_Padumuttarabuddhassa,_  
_sujāto aggasāvako;_  
_Vihārā abhinikkhamma,_  
_piṇḍapātāya gacchati._  

</Paragraph>

</Division>
<Division number="98" book="ap2" bookNo="20" page="82">

<Paragraph number="795" type="verses" book="ap2">
_Ghaṭaṃ gahetvā gacchantī,_  
_tadā udakahārikā;_  
_Taṃ disvā adadaṃ pūpaṃ,_  
_pasannā sehi pāṇibhi._  

</Paragraph>

</Division>
<Division number="99" book="ap2" bookNo="20" page="82">

<Paragraph number="796" type="verses" book="ap2">
_Paṭiggahetvā tattheva,_  
_nisinno paribhuñji so;_  
_Tato netvāna taṃ gehaṃ,_  
_adāsiṃ tassa bhojanaṃ._  

</Paragraph>

</Division>
<Division number="100" book="ap2" bookNo="20" page="82">

<Paragraph number="797" type="verses" book="ap2">
_Tato me ayyako tuṭṭho,_  
_akarī suṇisaṃ sakaṃ;_  
_Sassuyā saha gantvāna,_  
_sambuddhaṃ abhivādayiṃ._  

</Paragraph>

</Division>
<Division number="101" book="ap2" bookNo="20" page="82">

<Paragraph number="798" type="verses" book="ap2">
_Tadā so dhammakathikaṃ,_  
_bhikkhuniṃ parikittayaṃ;_  
_Ṭhapesi etadaggamhi,_  
_taṃ sutvā muditā ahaṃ._  

</Paragraph>

</Division>
<Division number="102" book="ap2" bookNo="20" page="82">

<Paragraph number="799" type="verses" book="ap2">
_Nimantayitvā sugataṃ,_  
_sasaṃghaṃ lokanāyakaṃ;_  
_Mahādānaṃ daditvāna,_  
_taṃ ṭhānamabhipatthayiṃ._  

</Paragraph>

</Division>
<Division number="103" book="ap2" bookNo="20" page="82">

<Paragraph number="800" type="verses" book="ap2">
_Tato maṃ sugato āha,_  
_ghananinnādasussaro;_  
_‘Mamupaṭṭhānanirate,_  
_sasaṃghaparivesike._  

</Paragraph>

</Division>
<Division number="104" book="ap2" bookNo="20" page="82">

<Paragraph number="801" type="verses" book="ap2">
_Saddhammassavane yutte,_  
_guṇavaddhitamānase;_  
_Bhadde bhavassu muditā,_  
_lacchase paṇidhīphalaṃ._  

</Paragraph>

</Division>
<Division number="105" book="ap2" bookNo="20" page="82">

<Paragraph number="802" type="verses" book="ap2">
_Satasahassito kappe,_  
_okkākakulasambhavo;_  
_Gotamo nāma gottena,_  
_satthā loke bhavissati._  

</Paragraph>

</Division>
<Division number="106" book="ap2" bookNo="20" page="83">

<Paragraph number="803" type="verses" book="ap2">
_Tassa dhammesu dāyādā,_  
_orasā dhammanimmitā;_  
_Dhammadinnāti nāmena,_  
_hessati satthu sāvikā’._  

</Paragraph>

</Division>
<Division number="107" book="ap2" bookNo="20" page="83">

<Paragraph number="804" type="verses" book="ap2">
_Taṃ sutvā muditā hutvā,_  
_yāvajīvaṃ mahāmuniṃ;_  
_Mettacittā paricariṃ,_  
_paccayehi vināyakaṃ._  

</Paragraph>

</Division>
<Division number="108" book="ap2" bookNo="20" page="83">

<Paragraph number="805" type="verses" book="ap2">
_Tena kammena sukatena,_  
_cetanāpaṇidhīhi ca;_  
_Jahitvā mānusaṃ dehaṃ,_  
_tāvatiṃsamagacchahaṃ._  

</Paragraph>

</Division>
<Division number="109" book="ap2" bookNo="20" page="83">

<Paragraph number="806" type="verses" book="ap2">
_Imamhi bhaddake kappe,_  
_brahmabandhu mahāyaso;_  
_Kassapo nāma gottena,_  
_uppajji vadataṃ varo._  

</Paragraph>

</Division>
<Division number="110" book="ap2" bookNo="20" page="110">

<Paragraph number="807" type="verses" book="ap2">
_Upaṭṭhāko mahesissa,_  
_tadā āsi narissaro;_  
_Kāsirājā kikī nāma,_  
_bārāṇasipuruttame._  

</Paragraph>

</Division>
<Division number="111" book="ap2" bookNo="20" page="110">

<Paragraph number="808" type="verses" book="ap2">
_Chaṭṭhā tassāsahaṃ dhītā,_  
_sudhammā iti vissutā;_  
_Dhammaṃ sutvā jinaggassa,_  
_pabbajjaṃ samarocayiṃ._  

</Paragraph>

</Division>
<Division number="112" book="ap2" bookNo="20" page="110">

<Paragraph number="809" type="verses" book="ap2">
_Anujāni na no tāto,_  
_agāreva tadā mayaṃ;_  
_Vīsavassasahassāni,_  
_vicarimha atanditā._  


</Paragraph>
</Division>

<Paragraph number="810" type="center" book="ap2">
Tatiyaṃ bhāṇavāraṃ.


</Paragraph>

<Division number="113" book="ap2" bookNo="20" page="111">

<Paragraph number="811" type="verses" book="ap2">
_Komāribrahmacariyaṃ,_  
_rājakaññā sukhedhitā;_  
_Buddhopaṭṭhānaniratā,_  
_muditā satta dhītaro._  

</Paragraph>

</Division>
<Division number="114" book="ap2" bookNo="20" page="111">

<Paragraph number="812" type="verses" book="ap2">
_Samaṇī samaṇaguttā ca,_  
_bhikkhunī bhikkhudāyikā;_  
_Dhammā ceva sudhammā ca,_  
_sattamī saṃghadāyikā._  

</Paragraph>

</Division>
<Division number="115" book="ap2" bookNo="20" page="111">

<Paragraph number="813" type="verses" book="ap2">
_Khemā uppalavaṇṇā ca,_  
_paṭācārā ca kuṇḍalā;_  
_Gotamī ca ahañceva,_  
_visākhā hoti sattamī._  

</Paragraph>

</Division>
<Division number="116" book="ap2" bookNo="20" page="111">

<Paragraph number="814" type="verses" book="ap2">
_Tehi kammehi sukatehi,_  
_cetanāpaṇidhīhi ca;_  
_Jahitvā mānusaṃ dehaṃ,_  
_tāvatiṃsamagacchahaṃ._  

</Paragraph>

</Division>
<Division number="117" book="ap2" bookNo="20" page="111">

<Paragraph number="815" type="verses" book="ap2">
_Pacchime ca bhave dāni,_  
_giribbajapuruttame;_  
_Jātā seṭṭhikule phīte,_  
_sabbakāmasamiddhine._  

</Paragraph>

</Division>
<Division number="118" book="ap2" bookNo="20" page="111">

<Paragraph number="816" type="verses" book="ap2">
_Yadā rūpaguṇūpetā,_  
_paṭhame yobbane ṭhitā;_  
_Tadā parakulaṃ gantvā,_  
_vasiṃ sukhasamappitā._  

</Paragraph>

</Division>
<Division number="119" book="ap2" bookNo="20" page="111">

<Paragraph number="817" type="verses" book="ap2">
_Upetvā lokasaraṇaṃ,_  
_suṇitvā dhammadesanaṃ;_  
_Anāgāmiphalaṃ patto,_  
_sāmiko me subuddhimā._  

</Paragraph>

</Division>
<Division number="120" book="ap2" bookNo="20" page="111">

<Paragraph number="818" type="verses" book="ap2">
_Tadāhaṃ anujānetvā,_  
_pabbajiṃ anagāriyaṃ;_  
_Na cireneva kālena,_  
_arahattamapāpuṇiṃ._  

</Paragraph>

</Division>
<Division number="121" book="ap2" bookNo="20" page="111">

<Paragraph number="819" type="verses" book="ap2">
_Tadā upāsako so maṃ,_  
_upagantvā apucchatha;_  
_Gambhīre nipuṇe pañhe,_  
_te sabbe byākariṃ ahaṃ._  

</Paragraph>

</Division>
<Division number="122" book="ap2" bookNo="20" page="111">

<Paragraph number="820" type="verses" book="ap2">
_Jino tasmiṃ guṇe tuṭṭho,_  
_etadagge ṭhapesi maṃ;_  
_‘Bhikkhuniṃ dhammakathikaṃ,_  
_nāññaṃ passāmi edisiṃ._  

</Paragraph>

</Division>
<Division number="123" book="ap2" bookNo="20" page="111">

<Paragraph number="821" type="verses" book="ap2">
_Dhammadinnā yathā dhīrā,_  
_evaṃ dhāretha bhikkhavo’;_  
_Evāhaṃ paṇḍitā homi,_  
_nāyakenānukampitā._  

</Paragraph>

</Division>
<Division number="124" book="ap2" bookNo="20" page="111">

<Paragraph number="822" type="verses" book="ap2">
_Pariciṇṇo mayā satthā,_  
_kataṃ buddhassa sāsanaṃ;_  
_Ohito garuko bhāro,_  
_bhavanetti samūhatā._  

</Paragraph>

</Division>
<Division number="125" book="ap2" bookNo="20" page="111">

<Paragraph number="823" type="verses" book="ap2">
_Yassatthāya pabbajitā,_  
_agārasmānagāriyaṃ;_  
_So me attho anuppatto,_  
_sabbasaṃyojanakkhayo._  

</Paragraph>

</Division>
<Division number="126" book="ap2" bookNo="20" page="112">

<Paragraph number="824" type="verses" book="ap2">
_Iddhīsu ca vasī homi,_  
_dibbāya sotadhātuyā;_  
_Paracittāni jānāmi,_  
_satthusāsanakārikā._  

</Paragraph>

</Division>
<Division number="127" book="ap2" bookNo="20" page="123">

<Paragraph number="825" type="verses" book="ap2">
_Pubbenivāsaṃ jānāmi,_  
_dibbacakkhu visodhitaṃ;_  
_Khepetvā āsave sabbe,_  
_visuddhāsiṃ sunimmalā._  

</Paragraph>

</Division>
