<Paragraph number="74" type="verses" book="ap2">
_“Piṇḍacāraṃ carantassa,_  
_tissanāmassa satthuno;_  
_Kaṭacchubhikkhaṃ paggayha,_  
_buddhaseṭṭhassa dāsahaṃ._  

</Paragraph>

</Division>
<Division number="61" book="ap2" bookNo="20" page="6">

<Paragraph number="75" type="verses" book="ap2">
_Paṭiggahetvā sambuddho,_  
_tisso lokagganāyako;_  
_Vīthiyā saṇṭhito satthā,_  
_akā me anumodanaṃ._  

</Paragraph>

</Division>
<Division number="62" book="ap2" bookNo="20" page="6">

<Paragraph number="76" type="verses" book="ap2">
_‘Kaṭacchubhikkhaṃ datvāna,_  
_tāvatiṃsaṃ gamissasi;_  
_Chattiṃsadevarājūnaṃ,_  
_mahesittaṃ karissasi._  

</Paragraph>

</Division>
<Division number="63" book="ap2" bookNo="20" page="6">

<Paragraph number="77" type="verses" book="ap2">
_Paññāsaṃ cakkavattīnaṃ,_  
_mahesittaṃ karissasi;_  
_Manasā patthitaṃ sabbaṃ,_  
_paṭilacchasi sabbadā._  

</Paragraph>

</Division>
<Division number="64" book="ap2" bookNo="20" page="6">

<Paragraph number="78" type="verses" book="ap2">
_Sampattiṃ anubhotvāna,_  
_pabbajissasikiñcanā;_  
_Sabbāsave pariññāya,_  
_nibbāyissasināsavā’._  

</Paragraph>

</Division>
<Division number="65" book="ap2" bookNo="20" page="6">

<Paragraph number="79" type="verses" book="ap2">
_Idaṃ vatvāna sambuddho,_  
_tisso lokagganāyako;_  
_Nabhaṃ abbhuggamī vīro,_  
_haṃsarājāva ambare._  

</Paragraph>

</Division>
<Division number="66" book="ap2" bookNo="20" page="6">

<Paragraph number="80" type="verses" book="ap2">
_Sudinnaṃ me dānavaraṃ,_  
_suyiṭṭhā yāgasampadā;_  
_Kaṭacchubhikkhaṃ datvāna,_  
_pattāhaṃ acalaṃ padaṃ._  

</Paragraph>

</Division>
<Division number="67" book="ap2" bookNo="20" page="6">

<Paragraph number="81" type="verses" book="ap2">
_Dvenavute ito kappe,_  
_yaṃ dānamadadiṃ tadā;_  
_Duggatiṃ nābhijānāmi,_  
_bhikkhādānassidaṃ phalaṃ._  

</Paragraph>

</Division>
<Division number="68" book="ap2" bookNo="20" page="7">

<Paragraph number="82" type="verses" book="ap2">
_Kilesā jhāpitā mayhaṃ,_  
_ …pe… _  
_viharāmi anāsavā._  

</Paragraph>

</Division>
<Division number="69" book="ap2" bookNo="20" page="7">

<Paragraph number="83" type="verses" book="ap2">
_Svāgataṃ vata me āsi,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ._  

</Paragraph>

</Division>
<Division number="70" book="ap2" bookNo="20" page="7">

<Paragraph number="84" type="verses" book="ap2">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="85" book="ap2">
Itthaṃ sudaṃ kaṭacchubhikkhādāyikā bhikkhunī imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="86" type="center" book="ap2">

</Paragraph>
