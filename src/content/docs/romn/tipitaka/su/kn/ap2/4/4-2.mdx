<Division number="13" book="ap2">

<Paragraph number="1219" type="verses" book="ap2">
_“Cullāsītisahassāni,_  
_brāhmaññakulasambhavā;_  
_Sukhumālahatthapādā,_  
_pure tuyhaṃ mahāmune._  

</Paragraph>

</Division>
<Division number="14" book="ap2">

<Paragraph number="1220" type="verses" book="ap2">
_Vessasuddakule jātā,_  
_devā nāgā ca kinnarā;_  
_Cātuddīpā bahū kaññā,_  
_pure tuyhaṃ mahāmune._  

</Paragraph>

</Division>
<Division number="15" book="ap2">

<Paragraph number="1221" type="verses" book="ap2">
_Kāci pabbajitā atthi,_  
_sabbadassāvino bahū;_  
_Devā ca kinnarā nāgā,_  
_phusissanti anāgate._  

</Paragraph>

</Division>
<Division number="16" book="ap2">

<Paragraph number="1222" type="verses" book="ap2">
_Anubhotvā yasaṃ sabbaṃ,_  
_Patvāna sabbasampadā;_  
_Tumhaṃ pasādaṃ paṭiladdhā,_  
_Bujjhissanti anāgate._  

</Paragraph>

</Division>
<Division number="17" book="ap2">

<Paragraph number="1223" type="verses" book="ap2">
_Amhe brāhmaṇadhītā tu,_  
_brāhmaññakulasambhavā;_  
_Pekkhato no mahāvīra,_  
_pāde vandāma cakkhuma._  

</Paragraph>

</Division>
<Division number="18" book="ap2">

<Paragraph number="1224" type="verses" book="ap2">
_Upāhatā bhavā sabbe,_  
_mūlataṇhā samūhatā;_  
_Samucchinnā anusayā,_  
_puññasaṅkhāradālitā._  

</Paragraph>

</Division>
<Division number="19" book="ap2">

<Paragraph number="1225" type="verses" book="ap2">
_Samādhigocarā sabbā,_  
_samāpattivasī katā;_  
_Jhānena dhammaratiyā,_  
_viharissāma no sadā._  

</Paragraph>

</Division>
<Division number="20" book="ap2">

<Paragraph number="1226" type="verses" book="ap2">
_Bhavanetti avijjā ca,_  
_saṅkhārāpi ca khepitā;_  
_Sududdasaṃ padaṃ gantvā,_  
_anujānātha nāyaka”._  

</Paragraph>

</Division>
<Division number="21" book="ap2">

<Paragraph number="1227" type="verses" book="ap2">
_“Upakārā mamaṃ tumhe,_  
_dīgharattaṃ katāvino;_  
_Catunnaṃ saṃsayaṃ chetvā,_  
_sabbā gacchantu nibbutiṃ”._  

</Paragraph>

</Division>
<Division number="22" book="ap2">

<Paragraph number="1228" type="verses" book="ap2">
_Vanditvā munino pāde,_  
_katvā iddhivikubbanaṃ;_  
_Kāci dassenti ālokaṃ,_  
_andhakāramathāparā._  

</Paragraph>

</Division>
<Division number="23" book="ap2">

<Paragraph number="1229" type="verses" book="ap2">
_Dassenti candasūriye,_  
_sāgarañca samacchakaṃ;_  
_Sineruṃ paribhaṇḍañca,_  
_dassenti pārichattakaṃ._  

</Paragraph>

</Division>
<Division number="24" book="ap2">

<Paragraph number="1230" type="verses" book="ap2">
_Tāvatiṃsañca bhavanaṃ,_  
_yāmaṃ dassenti iddhiyā;_  
_Tusitaṃ nimmite deve,_  
_vasavattī mahissare._  

</Paragraph>

</Division>
<Division number="25" book="ap2">

<Paragraph number="1231" type="verses" book="ap2">
_Brahmāno kāci dassenti,_  
_caṅkamañca mahārahaṃ;_  
_Brahmavaṇṇañca māpetvā,_  
_dhammaṃ desenti suññataṃ._  

</Paragraph>

</Division>
<Division number="26" book="ap2">

<Paragraph number="1232" type="verses" book="ap2">
_Nānāvikubbanaṃ katvā,_  
_iddhiṃ dassiya satthuno;_  
_Dassayiṃsu balaṃ sabbā,_  
_pāde vandiṃsu satthuno._  

</Paragraph>

</Division>
<Division number="27" book="ap2">

<Paragraph number="1233" type="verses" book="ap2">
_“Iddhīsu ca vasī homa,_  
_dibbāya sotadhātuyā;_  
_Cetopariyañāṇassa,_  
_vasī homa mahāmune._  

</Paragraph>

</Division>
<Division number="28" book="ap2">

<Paragraph number="1234" type="verses" book="ap2">
_Pubbenivāsaṃ jānāma,_  
_dibbacakkhu visodhitaṃ;_  
_Sabbāsavā parikkhīṇā,_  
_natthi dāni punabbhavo._  

</Paragraph>

</Division>
<Division number="29" book="ap2">

<Paragraph number="1235" type="verses" book="ap2">
_Atthadhammaniruttīsu,_  
_paṭibhāne tatheva ca;_  
_Ñāṇaṃ amhaṃ mahāvīra,_  
_uppannaṃ tava santike._  

</Paragraph>

</Division>
<Division number="30" book="ap2">

<Paragraph number="1236" type="verses" book="ap2">
_Pubbānaṃ lokanāthānaṃ,_  
_saṅgamaṃ no nidassitaṃ;_  
_Adhikāraṃ bahuṃ amhaṃ,_  
_tuyhatthāya mahāmune._  

</Paragraph>

</Division>
<Division number="31" book="ap2">

<Paragraph number="1237" type="verses" book="ap2">
_Yaṃ amhehi kataṃ kammaṃ,_  
_kusalaṃ sara taṃ mune;_  
_Tuyhatthāya mahāvīra,_  
_puññānupacitāni no._  

</Paragraph>

</Division>
<Division number="32" book="ap2">

<Paragraph number="1238" type="verses" book="ap2">
_Satasahassito kappe,_  
_padumuttaro mahāmuni;_  
_Puraṃ haṃsavatī nāma,_  
_sambuddhassa kulāsayaṃ._  

</Paragraph>

</Division>
<Division number="33" book="ap2">

<Paragraph number="1239" type="verses" book="ap2">
_Dvārena haṃsavatiyā,_  
_gaṅgā sandati sabbadā;_  
_Ubbāḷhā nadiyā bhikkhū,_  
_gamanaṃ na labhanti te._  

</Paragraph>

</Division>
<Division number="34" book="ap2">

<Paragraph number="1240" type="verses" book="ap2">
_Divasaṃ dve tayo ceva,_  
_sattāhaṃ māsikaṃ tato;_  
_Catumāsampi sampuṇṇaṃ,_  
_gamanaṃ na labhanti te._  

</Paragraph>

</Division>
<Division number="35" book="ap2">

<Paragraph number="1241" type="verses" book="ap2">
_Tadā ahu sattasāro,_  
_jaṭilo nāma raṭṭhiko;_  
_Oruddhe bhikkhavo disvā,_  
_setuṃ gaṅgāya kārayi._  

</Paragraph>

</Division>
<Division number="36" book="ap2">

<Paragraph number="1242" type="verses" book="ap2">
_Tadā satasahassehi,_  
_setuṃ gaṅgāya kārayi;_  
_Saṃghassa orime tīre,_  
_vihārañca akārayi._  

</Paragraph>

</Division>
<Division number="37" book="ap2">

<Paragraph number="1243" type="verses" book="ap2">
_Itthiyo purisā ceva,_  
_uccanīcakulāni ca;_  
_Tassa setuṃ vihārañca,_  
_samabhāgaṃ akaṃsu te._  

</Paragraph>

</Division>
<Division number="38" book="ap2">

<Paragraph number="1244" type="verses" book="ap2">
_Amhe aññe ca mānujā,_  
_vippasannena cetasā;_  
_Tassa dhammesu dāyādā,_  
_nagare janapadesu ca._  

</Paragraph>

</Division>
<Division number="39" book="ap2">

<Paragraph number="1245" type="verses" book="ap2">
_Itthī pumā kumārā ca,_  
_bahū ceva kumārikā;_  
_Setuno ca vihārassa,_  
_vālukā ākiriṃsu te._  

</Paragraph>

</Division>
<Division number="40" book="ap2">

<Paragraph number="1246" type="verses" book="ap2">
_Vīthiṃ sammajjanaṃ katvā,_  
_kadalīpuṇṇaghaṭe dhaje;_  
_Dhūpaṃ cuṇṇañca mālañca,_  
_kāraṃ katvāna satthuno._  

</Paragraph>

</Division>
<Division number="41" book="ap2">

<Paragraph number="1247" type="verses" book="ap2">
_Setuvihāre kāretvā,_  
_nimantetvā vināyakaṃ;_  
_Mahādānaṃ daditvāna,_  
_sambodhiṃ abhipatthayiṃ._  

</Paragraph>

</Division>
<Division number="42" book="ap2">

<Paragraph number="1248" type="verses" book="ap2">
_Padumuttaro mahāvīro,_  
_tārako sabbapāṇinaṃ;_  
_Anumodanīyaṃkāsi,_  
_jaṭilassa mahāmuni._  

</Paragraph>

</Division>
<Division number="43" book="ap2">

<Paragraph number="1249" type="verses" book="ap2">
_‘Satasahassātikkante,_  
_kappo hessati bhaddako;_  
_Bhavābhavenubhotvāna,_  
_pāpuṇissati bodhiyaṃ._  

</Paragraph>

</Division>
<Division number="44" book="ap2">

<Paragraph number="1250" type="verses" book="ap2">
_Kāci hatthaparikammaṃ,_  
_katāvī naranāriyo;_  
_Anāgatamhi addhāne,_  
_sabbā hessanti sammukhā’._  

</Paragraph>

</Division>
<Division number="45" book="ap2">

<Paragraph number="1251" type="verses" book="ap2">
_Tena kammavipākena,_  
_cetanāpaṇidhīhi ca;_  
_Uppannā devabhavanaṃ,_  
_tuyhaṃ tā paricārikā._  

</Paragraph>

</Division>
<Division number="46" book="ap2">

<Paragraph number="1252" type="verses" book="ap2">
_Dibbasukhaṃ asaṅkhiyaṃ,_  
_mānusañca asaṅkhiyaṃ;_  
_Tuyhaṃ te paricārema,_  
_saṃsarimha bhavābhave._  

</Paragraph>

</Division>
<Division number="47" book="ap2">

<Paragraph number="1253" type="verses" book="ap2">
_Satasahassito kappe,_  
_sukataṃ kammasampadaṃ;_  
_Sukhumālī manussānaṃ,_  
_atho devapure vare._  

</Paragraph>

</Division>
<Division number="48" book="ap2">

<Paragraph number="1254" type="verses" book="ap2">
_Rūpabhogayase ceva,_  
_atho kittiñca sakkataṃ;_  
_Labhāma satataṃ sabbaṃ,_  
_sukataṃ kammasampadaṃ._  

</Paragraph>

</Division>
<Division number="49" book="ap2">

<Paragraph number="1255" type="verses" book="ap2">
_Pacchime bhave sampatte,_  
_jātāmha brāhmaṇe kule;_  
_Sukhumālahatthapādā,_  
_sakyaputtanivesane._  

</Paragraph>

</Division>
<Division number="50" book="ap2">

<Paragraph number="1256" type="verses" book="ap2">
_Sabbakālampi pathaviṃ,_  
_na passāma na laṅkataṃ;_  
_Cikkhallabhūmimasuciṃ,_  
_na passāma mahāmune._  

</Paragraph>

</Division>
<Division number="51" book="ap2">

<Paragraph number="1257" type="verses" book="ap2">
_Agāraṃ vasante amhe,_  
_sakkāraṃ sabbakālikaṃ;_  
_Upanenti sadā sabbaṃ,_  
_pubbakammaphalena no._  

</Paragraph>

</Division>
<Division number="52" book="ap2">

<Paragraph number="1258" type="verses" book="ap2">
_Agāraṃ pajahitvāna,_  
_pabbajitvānagāriyaṃ;_  
_Saṃsārapathanitthiṇṇā,_  
_vītarāgā bhavāmase._  

</Paragraph>

</Division>
<Division number="53" book="ap2">

<Paragraph number="1259" type="verses" book="ap2">
_Cīvaraṃ piṇḍapātañca,_  
_paccayaṃ sayanāsanaṃ;_  
_Upanenti sadā amhe,_  
_sahassāni tato tato._  

</Paragraph>

</Division>
<Division number="54" book="ap2">

<Paragraph number="1260" type="verses" book="ap2">
_Kilesā jhāpitā amhaṃ,_  
_ …pe… _  
_viharāma anāsavā._  

</Paragraph>

</Division>
<Division number="55" book="ap2">

<Paragraph number="1261" type="verses" book="ap2">
_Svāgataṃ vata no āsi,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ._  

</Paragraph>

</Division>
<Division number="56" book="ap2">

<Paragraph number="1262" type="verses" book="ap2">
_Paṭisambhidā catasso,_  
_ …pe… _  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="1263" book="ap2">
Itthaṃ sudaṃ caturāsītibrāhmaṇakaññābhikkhunīsahassāni bhagavato sammukhā imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="1264" type="center" book="ap2">
Caturāsītibhikkhunīsahassāpadānaṃ dutiyaṃ.
</Paragraph>
