<Division number="87" book="ap2">

<Paragraph number="253" type="verses" book="ap2">

_vedajātā katañjalī;_  
_Naḷamālaṃ gahetvāna,_  
_sayambhuṃ abhipūjayiṃ._  

</Paragraph>

</Division>
<Division number="88" book="ap2">

<Paragraph number="254" type="verses" book="ap2">

_cetanāpaṇidhīhi ca;_  
_Jahitvā kinnarīdehaṃ,_  
_tāvatiṃsamagacchahaṃ._  

</Paragraph>

</Division>
<Division number="89" book="ap2">

<Paragraph number="255" type="verses" book="ap2">

_mahesittamakārayiṃ;_  
_Manasā patthitaṃ mayhaṃ,_  
_nibbattati yathicchitaṃ._  

</Paragraph>

</Division>
<Division number="90" book="ap2">

<Paragraph number="256" type="verses" book="ap2">

_mahesittamakārayiṃ;_  
_Ocitattāva hutvāna,_  
_saṃsarāmi bhavesvahaṃ._  

</Paragraph>

</Division>
<Division number="91" book="ap2">

<Paragraph number="257" type="verses" book="ap2">

_pabbajiṃ anagāriyaṃ;_  
_Pūjārahā ahaṃ ajja,_  
_sakyaputtassa sāsane._  

</Paragraph>

</Division>

<Paragraph number="258" type="verses" book="ap2">

_apetamanapāpikā;_  
_Sabbāsavā parikkhīṇā,_  
_natthi dāni punabbhavo._  

</Paragraph>

</Division>

</Paragraph>
