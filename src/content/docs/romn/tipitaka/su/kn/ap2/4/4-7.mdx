<Division number="169" book="ap2">

<Paragraph number="1385" type="verses" book="ap2">

_tassa buddhassa sāsane;_  
_Saṃvutā pātimokkhamhi,_  
_indriyesu ca pañcasu._  

</Paragraph>

</Division>
<Division number="170" book="ap2">

<Paragraph number="1386" type="verses" book="ap2">
_Mattaññunī ca asane,_  
_yuttā jāgariyepi ca;_  
_Vasantī yuttayogāhaṃ,_  
_bhikkhuniṃ vigatāsavaṃ._  

</Paragraph>

</Division>
<Division number="171" book="ap2">

<Paragraph number="1387" type="verses" book="ap2">
_Akkosiṃ duṭṭhacittāhaṃ,_  
_gaṇiketi bhaṇiṃ tadā;_  
_Tena pāpena kammena,_  
_nirayamhi apaccisaṃ._  

</Paragraph>

</Division>
<Division number="172" book="ap2">

<Paragraph number="1388" type="verses" book="ap2">
_Tena kammāvasesena,_  
_ajāyiṃ gaṇikākule;_  
_Bahusova parādhīnā,_  
_pacchimāya ca jātiyaṃ._  

</Paragraph>

</Division>
<Division number="173" book="ap2">

<Paragraph number="1389" type="verses" book="ap2">
_Kāsīsu seṭṭhikulajā,_  
_brahmacārībalenahaṃ;_  
_Accharā viya devesu,_  
_ahosiṃ rūpasampadā._  

</Paragraph>

</Division>
<Division number="174" book="ap2">

<Paragraph number="1390" type="verses" book="ap2">
_Disvāna dassanīyaṃ maṃ,_  
_giribbajapuruttame;_  
_Gaṇikatte nivesesuṃ,_  
_akkosanabalena me._  

</Paragraph>

</Division>
<Division number="175" book="ap2">

<Paragraph number="1391" type="verses" book="ap2">
_Sāhaṃ sutvāna saddhammaṃ,_  
_buddhaseṭṭhena desitaṃ;_  
_Pubbavāsanasampannā,_  
_pabbajiṃ anagāriyaṃ._  

</Paragraph>

</Division>
<Division number="176" book="ap2">

<Paragraph number="1392" type="verses" book="ap2">
_Tadūpasampadatthāya,_  
_gacchantī jinasantikaṃ;_  
_Magge dhutte ṭhite sutvā,_  
_labhiṃ dūtopasampadaṃ._  

</Paragraph>

</Division>
<Division number="177" book="ap2">

<Paragraph number="1393" type="verses" book="ap2">
_Sabbakammaṃ parikkhīṇaṃ,_  
_puññaṃ pāpaṃ tatheva ca;_  
_Sabbasaṃsāramuttiṇṇā,_  
_gaṇikattañca khepitaṃ._  

</Paragraph>

</Division>
<Division number="178" book="ap2">

<Paragraph number="1394" type="verses" book="ap2">
_Iddhīsu ca vasī homi,_  
_dibbāya sotadhātuyā;_  
_Cetopariyañāṇassa,_  
_vasī homi mahāmune._  

</Paragraph>

</Division>
<Division number="179" book="ap2">

<Paragraph number="1395" type="verses" book="ap2">
_Pubbenivāsaṃ jānāmi,_  
_dibbacakkhu visodhitaṃ;_  
_Sabbāsavā parikkhīṇā,_  
_natthi dāni punabbhavo._  

</Paragraph>

</Division>
<Division number="180" book="ap2">

<Paragraph number="1396" type="verses" book="ap2">

_paṭibhāne tatheva ca;_  
_Ñāṇaṃ mama mahāvīra,_  
_uppannaṃ tava santike._  

</Paragraph>

</Division>
