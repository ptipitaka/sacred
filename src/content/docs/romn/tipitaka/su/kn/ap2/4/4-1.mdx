<Division number="1" book="ap2">

<Paragraph number="1205" type="verses" book="ap2">
_“Bhavā sabbe parikkhīṇā,_  
_bhavā santi vimocitā;_  
_Sabbāsavā ca no natthi,_  
_ārocema mahāmune._  

</Paragraph>

</Division>
<Division number="2" book="ap2">

<Paragraph number="1206" type="verses" book="ap2">
_Purimaṃ kusalaṃ kammaṃ,_  
_yaṃ kiñci sādhupatthitaṃ;_  
_Paribhogamayaṃ dinnaṃ,_  
_tuyhatthāya mahāmune._  

</Paragraph>

</Division>
<Division number="3" book="ap2">

<Paragraph number="1207" type="verses" book="ap2">
_Buddhapaccekabuddhānaṃ,_  
_sāvakānañca patthitaṃ;_  
_Paribhogamayaṃ dinnaṃ,_  
_tuyhatthāya mahāmune._  

</Paragraph>

</Division>
<Division number="4" book="ap2">

<Paragraph number="1208" type="verses" book="ap2">
_Uccanīcamayaṃ kammaṃ,_  
_bhikkhūnaṃ sādhupatthitaṃ;_  
_Uccākulaparikammaṃ,_  
_katametaṃ mahāmune._  

</Paragraph>

</Division>
<Division number="5" book="ap2">

<Paragraph number="1209" type="verses" book="ap2">
_Teneva sukkamūlena,_  
_coditā kammasampadā;_  
_Mānusikamatikkantā,_  
_jāyiṃsu khattiye kule._  

</Paragraph>

</Division>
<Division number="6" book="ap2">

<Paragraph number="1210" type="verses" book="ap2">
_Uppatte ca kate kamme,_  
_jātiyā vāpi ekato;_  
_Pacchime ekato jātā,_  
_khattiyā kulasambhavā._  

</Paragraph>

</Division>
<Division number="7" book="ap2">

<Paragraph number="1211" type="verses" book="ap2">
_Rūpavatī bhogavatī,_  
_lābhasakkārapūjitā;_  
_Antepure mahāvīra,_  
_devānaṃ viya nandane._  

</Paragraph>

</Division>
<Division number="8" book="ap2">

<Paragraph number="1212" type="verses" book="ap2">
_Nibbinditvā agāramhā,_  
_pabbajimhanagāriyaṃ;_  
_Katipāhaṃ upādāya,_  
_sabbā pattāmha nibbutiṃ._  

</Paragraph>

</Division>
<Division number="9" book="ap2">

<Paragraph number="1213" type="verses" book="ap2">
_Cīvaraṃ piṇḍapātañca,_  
_paccayaṃ sayanāsanaṃ;_  
_Upanenti bahū amhe,_  
_sadā sakkatapūjitā._  

</Paragraph>

</Division>
<Division number="10" book="ap2">

<Paragraph number="1214" type="verses" book="ap2">
_Kilesā jhāpitā amhaṃ,_  
_bhavā sabbe samūhatā;_  
_Nāgīva bandhanaṃ chetvā,_  
_viharāma anāsavā._  

</Paragraph>

</Division>
<Division number="11" book="ap2">

<Paragraph number="1215" type="verses" book="ap2">
_Svāgataṃ vata no āsi,_  
_mama buddhassa santike;_  
_Tisso vijjā anuppattā,_  
_kataṃ buddhassa sāsanaṃ._  

</Paragraph>

</Division>
<Division number="12" book="ap2">

<Paragraph number="1216" type="verses" book="ap2">
_Paṭisambhidā catasso,_  
_vimokkhāpi ca aṭṭhime;_  
_Chaḷabhiññā sacchikatā,_  
_kataṃ buddhassa sāsanaṃ”._  

</Paragraph>

<Paragraph number="1217" book="ap2">
Itthaṃ sudaṃ yasavatīpamukhāni khattiyakaññābhikkhuniyo aṭṭhārasasahassāni bhagavato sammukhā imā gāthāyo abhāsitthāti.

</Paragraph>
</Division>

<Paragraph number="1218" type="center" book="ap2">
Yasavatīpamukhaaṭṭhārasabhikkhunīsahassāpadānaṃ paṭhamaṃ.
</Paragraph>
