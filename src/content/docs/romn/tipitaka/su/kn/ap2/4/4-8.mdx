<Paragraph number="1403" type="verses" book="ap2">

_pabbajitvāna sāsane;_  
_Bhikkhunī sīlasampannā,_  
_nipakā saṃvutindriyā._  

</Paragraph>

</Division>
<Division number="186" book="ap2" bookNo="20" page="158">

<Paragraph number="1404" type="verses" book="ap2">
_Bahussutā dhammadharā,_  
_dhammatthapaṭipucchikā;_  
_Uggahetā ca dhammānaṃ,_  
_sotā payirupāsitā._  

</Paragraph>

</Division>
<Division number="187" book="ap2" bookNo="20" page="158">

<Paragraph number="1405" type="verses" book="ap2">
_Desentī janamajjhehaṃ,_  
_ahosiṃ jinasāsane;_  
_Bāhusaccena tenāhaṃ,_  
_pesalā atimaññisaṃ._  

</Paragraph>

</Division>
<Division number="188" book="ap2" bookNo="20" page="158">

<Paragraph number="1406" type="verses" book="ap2">
_Pacchime ca bhave dāni,_  
_sāvatthiyaṃ puruttame;_  
_Anāthapiṇḍino gehe,_  
_jātāhaṃ kumbhadāsiyā._  

</Paragraph>

</Division>
<Division number="189" book="ap2" bookNo="20" page="158">

<Paragraph number="1407" type="verses" book="ap2">
_Gatā udakahāriyaṃ,_  
_sotthiyaṃ dijamaddasaṃ;_  
_Sītaṭṭaṃ toyamajjhamhi,_  
_taṃ disvā idamabraviṃ._  

</Paragraph>

</Division>
<Division number="190" book="ap2" bookNo="20" page="158">

<Paragraph number="1408" type="verses" book="ap2">
_‘Udahārī ahaṃ sīte,_  
_sadā udakamotariṃ;_  
_Ayyānaṃ daṇḍabhayabhītā,_  
_vācādosabhayaṭṭitā._  

</Paragraph>

</Division>
<Division number="191" book="ap2" bookNo="20" page="158">

<Paragraph number="1409" type="verses" book="ap2">
_Kassa brāhmaṇa tvaṃ bhīto,_  
_sadā udakamotari;_  
_Vedhamānehi gattehi,_  
_sītaṃ vedayase bhusaṃ’._  

</Paragraph>

</Division>
<Division number="192" book="ap2" bookNo="20" page="158">

<Paragraph number="1410" type="verses" book="ap2">
_‘Jānantī vata maṃ bhoti,_  
_puṇṇike paripucchasi;_  
_Karontaṃ kusalaṃ kammaṃ,_  
_rundhantaṃ katapāpakaṃ._  

</Paragraph>

</Division>
<Division number="193" book="ap2" bookNo="20" page="158">

<Paragraph number="1411" type="verses" book="ap2">
_Yo ce vuḍḍho daharo vā,_  
_pāpakammaṃ pakubbati;_  
_Dakābhisiñcanā sopi,_  
_pāpakammā pamuccati’._  

</Paragraph>

</Division>
<Division number="194" book="ap2" bookNo="20" page="158">

<Paragraph number="1412" type="verses" book="ap2">
_Uttarantassa akkhāsiṃ,_  
_dhammatthasaṃhitaṃ padaṃ;_  
_Tañca sutvā sa saṃviggo,_  
_pabbajitvārahā ahu._  

</Paragraph>

</Division>
<Division number="195" book="ap2" bookNo="20" page="185">

<Paragraph number="1413" type="verses" book="ap2">
_Pūrentī ūnakasataṃ,_  
_jātā dāsikule yato;_  
_Tato puṇṇāti nāmaṃ me,_  
_bhujissaṃ maṃ akaṃsu te._  

</Paragraph>

</Division>
<Division number="196" book="ap2" bookNo="20" page="159">

<Paragraph number="1414" type="verses" book="ap2">
_Seṭṭhiṃ tatonujānetvā,_  
_pabbajiṃ anagāriyaṃ;_  
_Na cireneva kālena,_  
_arahattamapāpuṇiṃ._  

</Paragraph>

</Division>
<Division number="197" book="ap2" bookNo="20" page="159">

<Paragraph number="1415" type="verses" book="ap2">
_Iddhīsu ca vasī homi,_  
_dibbāya sotadhātuyā;_  
_Cetopariyañāṇassa,_  
_vasī homi mahāmune._  

</Paragraph>

</Division>
<Division number="198" book="ap2" bookNo="20" page="159">

<Paragraph number="1416" type="verses" book="ap2">
_Pubbenivāsaṃ jānāmi,_  
_dibbacakkhu visodhitaṃ;_  
_Sabbāsavā parikkhīṇā,_  
_natthi dāni punabbhavo._  

</Paragraph>

</Division>
<Division number="199" book="ap2" bookNo="20" page="159">

<Paragraph number="1417" type="verses" book="ap2">
_Atthadhammaniruttīsu,_  
_paṭibhāne tatheva ca;_  
_Ñāṇaṃ me vimalaṃ suddhaṃ,_  
_buddhaseṭṭhassa vāhasā._  

</Paragraph>

</Division>
<Division number="200" book="ap2" bookNo="20" page="159">

_suteneva sutāvinī;_  
_Mānena nīcakulajā,_  
_na hi kammaṃ vinassati._  

</Paragraph>

</Division>
