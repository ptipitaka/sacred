<Paragraph number="924" type="verses" book="ap2">
_“Padumuttaro nāma jino,_  
_sabbadhammāna pāragū;_  
_Ito satasahassamhi,_  
_kappe uppajji nāyako._  

</Paragraph>

</Division>
<Division number="221" book="ap2" bookNo="20" page="160">

<Paragraph number="925" type="verses" book="ap2">
_Tadā seṭṭhikule jātā,_  
_sukhitā pūjitā piyā;_  
_Upetvā taṃ munivaraṃ,_  
_assosiṃ madhuraṃ vacaṃ._  

</Paragraph>

</Division>
<Division number="222" book="ap2" bookNo="20" page="161">

<Paragraph number="926" type="verses" book="ap2">
_Āraddhavīriyānaggaṃ,_  
_vaṇṇesi bhikkhuniṃ jino;_  
_Taṃ sutvā muditā hutvā,_  
_kāraṃ katvāna satthuno._  

</Paragraph>

</Division>
<Division number="223" book="ap2" bookNo="20" page="161">

<Paragraph number="927" type="verses" book="ap2">
_Abhivādiya sambuddhaṃ,_  
_ṭhānaṃ taṃ patthayiṃ tadā;_  
_Anumodi mahāvīro,_  
_‘sijjhataṃ paṇidhī tava._  

</Paragraph>

</Division>
<Division number="224" book="ap2" bookNo="20" page="161">

<Paragraph number="928" type="verses" book="ap2">
_Satasahassito kappe,_  
_okkākakulasambhavo;_  
_Gotamo nāma gottena,_  
_satthā loke bhavissati._  

</Paragraph>

</Division>
<Division number="225" book="ap2" bookNo="20" page="161">

<Paragraph number="929" type="verses" book="ap2">
_Tassa dhammesu dāyādā,_  
_orasā dhammanimmitā;_  
_Soṇāti nāma nāmena,_  
_hessati satthu sāvikā’._  

</Paragraph>

</Division>
<Division number="226" book="ap2" bookNo="20" page="161">

<Paragraph number="930" type="verses" book="ap2">
_Taṃ sutvā muditā hutvā,_  
_yāvajīvaṃ tadā jinaṃ;_  
_Mettacittā paricariṃ,_  
_paccayehi vināyakaṃ._  

</Paragraph>

</Division>
<Division number="227" book="ap2" bookNo="20" page="161">

<Paragraph number="931" type="verses" book="ap2">
_Tena kammena sukatena,_  
_cetanāpaṇidhīhi ca;_  
_Jahitvā mānusaṃ dehaṃ,_  
_tāvatiṃsamagacchahaṃ._  

</Paragraph>

</Division>
<Division number="228" book="ap2" bookNo="20" page="161">

<Paragraph number="932" type="verses" book="ap2">
_Pacchime ca bhave dāni,_  
_jātā seṭṭhikule ahaṃ;_  
_Sāvatthiyaṃ puravare,_  
_iddhe phīte mahaddhane._  

</Paragraph>

</Division>
<Division number="229" book="ap2" bookNo="20" page="161">

<Paragraph number="933" type="verses" book="ap2">
_Yadā ca yobbanappattā,_  
_gantvā patikulaṃ ahaṃ;_  
_Dasa puttāni ajaniṃ,_  
_surūpāni visesato._  

</Paragraph>

</Division>
<Division number="230" book="ap2" bookNo="20" page="161">

<Paragraph number="934" type="verses" book="ap2">
_Sukhedhitā ca te sabbe,_  
_jananettamanoharā;_  
_Amittānampi rucitā,_  
_mama pageva te piyā._  

</Paragraph>

</Division>
<Division number="231" book="ap2" bookNo="20" page="161">

<Paragraph number="935" type="verses" book="ap2">
_Tato mayhaṃ akāmāya,_  
_dasaputtapurakkhato;_  
_Pabbajittha sa me bhattā,_  
_devadevassa sāsane._  

</Paragraph>

</Division>
<Division number="232" book="ap2" bookNo="20" page="161">

<Paragraph number="936" type="verses" book="ap2">
_Tadekikā vicintesiṃ,_  
_‘jīvitenālamatthu me;_  
_Cattāya patiputtehi,_  
_vuḍḍhāya ca varākiyā._  

</Paragraph>

</Division>
<Division number="233" book="ap2" bookNo="20" page="161">

<Paragraph number="937" type="verses" book="ap2">
_Ahampi tattha gacchissaṃ,_  
_sampatto yattha me pati’;_  
_Evāhaṃ cintayitvāna,_  
_pabbajiṃ anagāriyaṃ._  

</Paragraph>

</Division>
<Division number="234" book="ap2" bookNo="20" page="162">

<Paragraph number="938" type="verses" book="ap2">
_Tato ca maṃ bhikkhuniyo,_  
_ekaṃ bhikkhunupassaye;_  
_Vihāya gacchumovādaṃ,_  
_‘tāpehi udakaṃ’ iti._  

</Paragraph>

</Division>
<Division number="235" book="ap2" bookNo="20" page="162">

<Paragraph number="939" type="verses" book="ap2">
_Tadā udakamāhitvā,_  
_okiritvāna kumbhiyā;_  
_Culle ṭhapetvā āsīnā,_  
_tato cittaṃ samādahiṃ._  

</Paragraph>

</Division>
<Division number="236" book="ap2" bookNo="20" page="162">

<Paragraph number="940" type="verses" book="ap2">
_Khandhe aniccato disvā,_  
_dukkhato ca anattato;_  
_Khepetvā āsave sabbe,_  
_arahattamapāpuṇiṃ._  

</Paragraph>

</Division>
<Division number="237" book="ap2" bookNo="20" page="162">

<Paragraph number="941" type="verses" book="ap2">
_Tadāgantvā bhikkhuniyo,_  
_uṇhodakamapucchisuṃ;_  
_Tejodhātumadhiṭṭhāya,_  
_khippaṃ santāpayiṃ jalaṃ._  

</Paragraph>

</Division>
<Division number="238" book="ap2" bookNo="20" page="162">

<Paragraph number="942" type="verses" book="ap2">
_Vimhitā tā jinavaraṃ,_  
_etamatthamasāvayuṃ;_  
_Taṃ sutvā mudito nātho,_  
_imaṃ gāthaṃ abhāsatha._  

</Paragraph>

</Division>
<Division number="239" book="ap2" bookNo="20" page="162">

<Paragraph number="943" type="verses" book="ap2">
_‘Yo ca vassasataṃ jīve,_  
_kusīto hīnavīriyo;_  
_Ekāhaṃ jīvitaṃ seyyo,_  
_vīriyamārabhato daḷhaṃ’._  

</Paragraph>

</Division>
<Division number="240" book="ap2" bookNo="20" page="162">

<Paragraph number="944" type="verses" book="ap2">
_Ārādhito mahāvīro,_  
_mayā suppaṭipattiyā;_  
_Āraddhavīriyānaggaṃ,_  
_mamāha sa mahāmuni._  

</Paragraph>

</Division>

_ …pe… _
