<Division number="257" book="pari">

<Paragraph number="1076" type="verses" book="pari">
_अनिच्चा सब्बे सङ्खारा,_  
_दुक्खानत्ता च सङ्खता;_  
_निब्बानञ्चेव पञ्ञत्ति,_  
_अनत्ता इति निच्छया._  

</Paragraph>

<Paragraph number="1077" type="verses" book="pari">
_बुद्धचन्दे अनुप्पन्ने,_  
_बुद्धादिच्चे अनुग्गते;_  
_तेसं सभागधम्मानं,_  
_नाममत्तं न नायति._  

</Paragraph>

<Paragraph number="1078" type="verses" book="pari">
_दुक्करं विविधं कत्वा,_  
_पूरयित्वान पारमी;_  
उप्पज्जन्ति महावीरा,_  
_चक्खुभूता सब्रह्मके._  

</Paragraph>

<Paragraph number="1079" type="verses" book="pari">
_ते देसयन्ति सद्धम्मं,_  
_दुक्खहानिं सुखावहं;_  
_अङ्गीरसो सक्यमुनि,_  
_सब्बभूतानुकम्पको._  

</Paragraph>

<Paragraph number="1080" type="verses" book="pari">
_सब्बसत्तुत्तमो सीहो,_  
_पिटके तीणि देसयि;_  
_सुत्तन्तमभिधम्मञ्च,_  
_विनयञ्च महागुणं._  

</Paragraph>

<Paragraph number="1081" type="verses" book="pari">
_एवं नीयति सद्धम्मो,_  
_विनयो यदि तिट्ठति;_  
उभतो च विभङ्गानि,_  
_खन्धका या च मातिका._  

</Paragraph>

<Paragraph number="1082" type="verses" book="pari">
_माला सुत्तगुणेनेव,_  
_परिवारेन गन्थिता;_  
_तस्सेव परिवारस्स,_  
_समुट्ठानं नियतो कतं._  

</Paragraph>

<Paragraph number="1083" type="verses" book="pari">
_सम्भेदं निदानञ्चञ्ञं,_  
_सुत्ते दिस्सन्ति उपरि;_  
_तस्मा सिक्खे परिवारं,_  
_धम्मकामो सुपेसलोति._
</Paragraph>
</Division>
